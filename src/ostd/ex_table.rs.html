3use crate::prelude::Vaddr;
4
5#[repr(C)]
6struct ExTableItem {
7    inst_addr: Vaddr,
8    recovery_inst_addr: Vaddr,
9}
10
11unsafe extern "C" {
12    fn __ex_table();
13    fn __ex_table_end();
14}
15
16/// A structure representing the usage of exception table (ExTable).
17/// This table is used for recovering from specific exception handling faults
18/// occurring at known points in the code.
19///
20/// To add a recovery instruction for a target assembly instruction, one should add
21/// the following statements:
22///
23/// ```
24/// .pushsection .ex_table, "a"
25/// .align 8
26/// .quad [.target_label],
27/// .quad [.recovery_label],
28/// .popsection
29/// ```
30///
31/// where the `target_label` and `recovery_label` are the labels of the target instruction
32/// and the label of recovery instruction respectively.
33///
34/// For example, we have the following assembly code snippets in an input file:
35/// ```
36/// .label1:    
37///     rep movsb
38///     mov rax, rcx
39/// .label2:
40///     ret
41/// ```
42///
43/// We can add the following statements in the same file (`label1` and `label2` are local
44/// labels):
45///
46/// ```
47/// .pushsection .ex_table, "a"
48/// .align 8
49/// .quad [.label1],
50/// .quad [.label2],
51/// .popsection
52/// ```
53///
54/// After that, we can use the API of `ExTable` to resume execution when handling
55/// exceptions caused by `rep movsb` (which `label1` point to) failing.
56pub(crate) struct ExTable;
57
58impl ExTable {
59    /// Finds the recovery instruction address for a given instruction address.
60    ///
61    /// This function is generally used when an exception (such as a page fault) occurs.
62    /// if the exception handling fails and there is a predefined recovery action,
63    /// then the found recovery action will be taken.
64    #[cfg_attr(target_arch = "loongarch64", expect(unused))]
65    pub fn find_recovery_inst_addr(inst_addr: Vaddr) -> Option<Vaddr> {
66        let table_size = (__ex_table_end as *const () as usize - __ex_table as *const () as usize)
67            / size_of::<ExTableItem>();
68        // SAFETY: `__ex_table` is a static section consisting of `ExTableItem`.
69        let ex_table =
70            unsafe { core::slice::from_raw_parts(__ex_table as *const ExTableItem, table_size) };
71        for item in ex_table {
72            if item.inst_addr == inst_addr {
73                return Some(item.recovery_inst_addr);
74            }
75        }
76        None
77    }
78}
