9/// Initializes the allocatable MMIO area based on the x86-64 memory distribution map.
10///
11/// In x86-64, the available physical memory area is divided into two regions below 32 bits (Low memory)
12/// and above (High memory). The area from the top of low memory to 0xffff_ffff and the area after the
13/// top of high memory are available MMIO areas.
14///
15/// This memory layout is documented in Intel's datasheets. The MMIO area is defined by specific
16/// registers that are configured by the BIOS and locked down, preventing the OS from reconfiguring
17/// them. For the details, one can read the "Processor Configuration Register Definitions and
18/// Address Ranges" section in the "10th Generation Intel(R) Processor Families" datasheet
19/// (<https://www.intel.com/content/dam/www/public/us/en/documents/datasheets/10th-gen-core-families-datasheet-vol-2-datasheet.pdf>).
20/// However, note that these specifics may differ between CPU generations and those manufactured by
21/// other vendors.
22///
23/// # Safety
24///
25/// 1. This function must be called only once in the boot context of the
26///    bootstrapping processor.
27/// 2. This function must be called after the kernel page table is activated on
28///    the bootstrapping processor.
29pub(super) unsafe fn construct_io_mem_allocator_builder() -> IoMemAllocatorBuilder {
30    // TODO: Add MMIO regions below 1MB (e.g., VGA framebuffer).
31    let regions = &crate::boot::EARLY_INFO.get().unwrap().memory_regions;
32    let mut ranges = Vec::with_capacity(2);
33
34    let reserved_filter = regions.iter().filter(|r| {
35        r.typ() != MemoryRegionType::Unknown
36            && r.typ() != MemoryRegionType::Reserved
37            && r.typ() != MemoryRegionType::Framebuffer
38    });
39
40    // Find the TOLM (Top of Low Memory) and initialize Low MMIO region (TOLM ~ LOW_MMIO_TOP).
41    // Align start address to LOW_MMIO_ALIGN (according to Intel's datasheets, the lower 20 bits
42    // are zeroed).
43    const LOW_MMIO_TOP: usize = 0x1_0000_0000; // 4 GiB, 32 bits
44    const LOW_MMIO_ALIGN: usize = 0x10_0000; // 1 MiB, 20 bits
45    let (lower_half_base, lower_half_len) = reserved_filter
46        .clone()
47        .filter(|r| r.base() < LOW_MMIO_TOP)
48        .max_by(|a, b| a.base().cmp(&b.base()))
49        .map(|reg| (reg.base(), reg.len()))
50        .unwrap();
51
52    let mmio_start_addr = (lower_half_base + lower_half_len).align_up(LOW_MMIO_ALIGN);
53    assert!(mmio_start_addr < LOW_MMIO_TOP);
54    ranges.push(mmio_start_addr..LOW_MMIO_TOP);
55
56    // Find the TOHM (Top of High Memory) and initialize High MMIO region (TOHM ~ HIGH_MMIO_TOP).
57    // Align start address to HIGH_MMIO_ALIGN (according to Intel's datasheets, the lower 20 bits
58    // are zeroed).
59    //
60    // TODO: Use the CPUID instruction to determine the maximum number of bits in physical
61    // addresses. We use 52 bits here, which is the architectural limit for physical addresses.
62    const HIGH_MMIO_TOP: usize = 0x10_0000_0000_0000; // 4 PiB, 52 bits
63    const HIGH_MMIO_ALIGN: usize = 0x10_0000; // 1 MiB, 20 bits
64    let (upper_half_base, upper_half_len) = reserved_filter
65        .filter(|r| r.base() >= LOW_MMIO_TOP)
66        .max_by(|a, b| a.base().cmp(&b.base()))
67        .map(|reg| (reg.base(), reg.len()))
68        .unwrap_or((LOW_MMIO_TOP, 0));
69
70    let mmio_start_addr = (upper_half_base + upper_half_len).align_up(HIGH_MMIO_ALIGN);
71    assert!(mmio_start_addr < HIGH_MMIO_TOP);
72    ranges.push(mmio_start_addr..HIGH_MMIO_TOP);
73
74    // SAFETY:
75    // 1. This is the only place that creates an `IoMemAllocatorBuilder`. The
76    //    caller ensures that the function is only called once.
77    // 2. The caller ensures that the kernel page table is already activated.
78    // 3. The range is guaranteed not to access physical memory.
79    unsafe { IoMemAllocatorBuilder::new(ranges) }
80}
