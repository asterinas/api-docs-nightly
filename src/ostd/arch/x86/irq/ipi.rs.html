3//! Inter-processor interrupts.
4
5use spin::Once;
6
7use crate::{cpu::PinCurrentCpu, irq::IrqLine, smp::do_inter_processor_call};
8
9/// Hardware-specific, architecture-dependent CPU ID.
10///
11/// This is the Local APIC ID in the x86_64 architecture.
12#[derive(Debug, Clone, Copy, PartialEq, Eq)]
13pub(crate) struct HwCpuId(u32);
14
15impl HwCpuId {
16    pub(crate) fn read_current(guard: &dyn PinCurrentCpu) -> Self {
17        use crate::arch::kernel::apic;
18
19        let apic = apic::get_or_init(guard);
20        Self(apic.id())
21    }
22}
23
24static IPI_IRQ: Once<IrqLine> = Once::new();
25
26/// Initializes global IPI state.
27pub(in crate::arch) fn init() {
28    let mut irq = IrqLine::alloc().unwrap();
29    // SAFETY: This will be called upon an inter-processor interrupt.
30    irq.on_active(|f| unsafe { do_inter_processor_call(f) });
31    IPI_IRQ.call_once(|| irq);
32}
33
34/// Sends a general inter-processor interrupt (IPI) to the specified CPU.
35pub(crate) fn send_ipi(hw_cpu_id: HwCpuId, guard: &dyn PinCurrentCpu) {
36    use crate::arch::kernel::apic::{self, Icr};
37
38    let irq_num = IPI_IRQ.get().unwrap().num();
39
40    let icr = Icr::new(
41        apic::ApicId::from(hw_cpu_id.0),
42        apic::DestinationShorthand::NoShorthand,
43        apic::TriggerMode::Edge,
44        apic::Level::Assert,
45        apic::DeliveryStatus::Idle,
46        apic::DestinationMode::Physical,
47        apic::DeliveryMode::Fixed,
48        irq_num,
49    );
50
51    let apic = apic::get_or_init(guard);
52    // SAFETY: The ICR is valid to generate the request IPI. Generating the
53    // request IPI is safe.
54    unsafe { apic.send_ipi(icr) };
55}
