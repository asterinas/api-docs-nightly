3use spin::once::Once;
4
5use crate::arch::iommu::{IrtEntryHandle, alloc_irt_entry, has_interrupt_remapping};
6
7pub(crate) struct IrqRemapping {
8    entry: Once<IrtEntryHandle>,
9}
10
11impl IrqRemapping {
12    pub(crate) const fn new() -> Self {
13        Self { entry: Once::new() }
14    }
15
16    /// Initializes the remapping entry for the specific IRQ number.
17    ///
18    /// This will do nothing if the entry is already initialized or interrupt
19    /// remapping is disabled or not supported by the architecture.
20    pub(crate) fn init(&self, irq_num: u8) {
21        if !has_interrupt_remapping() {
22            return;
23        }
24
25        self.entry.call_once(|| {
26            // Allocate and enable the IRT entry.
27            let handle = alloc_irt_entry().unwrap();
28            handle.enable(irq_num as u32);
29            handle
30        });
31    }
32
33    /// Gets the remapping index of the IRQ line.
34    ///
35    /// This method will return `None` if interrupt remapping is disabled or
36    /// not supported by the architecture.
37    pub(crate) fn remapping_index(&self) -> Option<u16> {
38        Some(self.entry.get()?.index())
39    }
40}
