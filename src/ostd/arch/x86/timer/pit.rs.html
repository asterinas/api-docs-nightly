5//! The Programmable Interval Timer (PIT) chip (Intel 8253/8254) basically consists of an oscillator,
6//! a prescaler and 3 independent frequency dividers. Each frequency divider has an output, which is
7//! used to allow the timer to control external circuitry (for example, IRQ 0).
8//!
9//! Reference: <https://wiki.osdev.org/Programmable_Interval_Timer>
10//!
11

22/// PIT Operating Mode.
23///
24/// Usually, only the rate generator, which is used to determine the base frequency of other timers
25/// (e.g. APIC Timer), and the Square wave generator, which is used to generate interrupts directly, are used.
26///
27/// Note that if IOAPIC is used to manage interrupts and square wave mode is enabled, the frequency at which
28/// clock interrupts are generated is `Frequency/2`.
29#[repr(u8)]
30pub enum OperatingMode {
31    /// Triggers an interrupt (only on channel 0) when the counter is terminated (1 -> 0).
32    /// The data port needs to be reset before the next interrupt.
33    /// ```text,ignore
34    ///            software reload counter
35    ///                      ⬇
36    ///               +------+             +----
37    ///               |      |             |
38    /// --------------+      +-------------+
39    /// ⬆             ⬆                    ⬆
40    /// init()   counter 1 -> 0         counter 1 -> 0
41    /// ```
42    InterruptOnTerminalCount = 0b000,
43    /// This mode is similar to `InterruptOnTerminalCount` mode, however counting doesn't start until
44    /// a rising edge of the gate input is detected. For this reason it is not usable for PIT channels
45    /// 0 or 1(where the gate input can't be changed).
46    OneShotHardwareRetriggerable = 0b001,
47    /// Rate generator, which produces a pulse at a fixed frequency.
48    /// ```text,ignore
49    /// init()   counter 2 -> 1    counter 2 -> 1
50    /// ⬇             ⬇                ⬇
51    /// --------------+  +-------------+
52    ///               |  |             |
53    ///               +--+             +--
54    ///                  ⬆
55    ///     counter 1 -> 0, auto reload counter
56    /// ```
57    RateGenerator = 0b010,
58    /// In this mode, the current count is **decremented twice** on each falling edge of the input signal.
59    /// The output will change state and then set to reload value.
60    /// ```text,ignore
61    /// init()  auto reload counter
62    /// ⬇             ⬇
63    /// --------------+              +--------------
64    ///               |              |
65    ///               +--------------+
66    ///                              ⬆
67    ///                       auto reload counter
68    /// ```
69    SquareWaveGenerator = 0b011,
70    /// Similar to a Rate generator, but requires a software reset to start counting.
71    /// ```text,ignore
72    /// init()   counter: 1  software reload counter
73    /// ⬇             ⬇              ⬇
74    /// --------------+ +---------------------------+ +--
75    ///               | |                           | |
76    ///               +-+                           +-+
77    ///                 ⬆
78    ///              counter: 0
79    /// ```
80    SoftwareTriggeredStrobe = 0b100,
81    /// This mode is similar to `SoftwareTriggeredStrobe` mode, except that it waits for the rising
82    /// edge of the gate input to trigger (or re-trigger) the delay period (like `OneShotHardwareRetriggerable`
83    /// mode).
84    HardwareTriggeredStrobe = 0b101,
85    // 0b110 -> Rate Generator
86    // 0b111 -> Square Wave Generator
87}

89/// This bits tell the PIT what access mode is used for the selected channel.
90#[repr(u8)]
91enum AccessMode {
92    /// When this command is sent, the current count is copied into latch register which can be read
93    /// through the data port corresponding to the selected channel (I/O ports 0x40 to 0x42).
94    LatchCountValueCommand = 0b00,
95    /// Only the lowest 8 bits of the count value are used in this mode.
96    LowByteOnly = 0b01,
97    /// Only the highest 8 bits of the count value are used in this mode.
98    HighByteOnly = 0b10,
99    /// 16 bits are used in this mode. User should sent the lowest 8 bits followed by the highest 8 bits
100    /// to the same data port.
101    LowAndHighByte = 0b11,
102}
103
104/// Used to select the configured channel in the `MODE_COMMAND_PORT` of the PIT.
105#[repr(u8)]
106enum Channel {
107    /// Channel 0. For more details, check `CHANNEL0_PORT` static variable
108    Channel0 = 0b00,
109    /// Channel 1. For more details, check `CHANNEL1_PORT` static variable
110    Channel1 = 0b01,
111    /// Channel 2. For more details, check `CHANNEL2_PORT` static variable
112    Channel2 = 0b10,
113    /// The read back command is a special command sent to the mode/command register.
114    /// The register uses the following format if set to read back command:
115    /// ```text
116    /// Bits         Usage
117    /// 7 and 6      Must be set for the read back command
118    /// 5            Latch count flag (0 = latch count, 1 = don't latch count)
119    /// 4            Latch status flag (0 = latch status, 1 = don't latch status)
120    /// 3            Read back timer channel 2 (1 = yes, 0 = no)
121    /// 2            Read back timer channel 1 (1 = yes, 0 = no)
122    /// 1            Read back timer channel 0 (1 = yes, 0 = no)
123    /// 0            Reserved
124    /// ```
125    /// Bits 1 to 3 of the read back command select which PIT channels are affected,
126    /// and allow multiple channels to be selected at the same time.
127    ///
128    /// If bit 5 is clear, then any/all PIT channels selected with bits 1 to 3 will
129    /// have their current count copied into their latch register.
130    ///
131    /// If bit 4 is clear, then for any/all PIT channels selected with bits 1 to 3,
132    /// the next read of the corresponding data port will return a status byte.
133    ///
134    /// Ref: <https://wiki.osdev.org/Programmable_Interval_Timer#Read_Back_Command>.
135    ReadBackCommand = 0b11,
136}

138sensitive_io_port! {
139    unsafe {
140        /// The output from PIT channel 0 is connected to the PIC chip and generate "IRQ 0".
141        /// If connected to PIC, the IRQ0 will generate by the **rising edge** of the output voltage.
142        static CHANNEL0_PORT: IoPort<u8, WriteOnlyAccess> = IoPort::new(0x40);
143
144        /// The output from PIT channel 1 was once used for refreshing the DRAM or RAM so that
145        /// the capacitors don't forget their state.
146        ///
147        /// On later machines, the DRAM refresh is done with dedicated hardware and this channel
148        /// is no longer used.
149        static CHANNEL1_PORT: IoPort<u8, WriteOnlyAccess> = IoPort::new(0x41);
150
151        /// The output from PIT channel 2 is connected to the PC speaker, so the frequency of the
152        /// output determines the frequency of the sound produced by the speaker. For more information,
153        /// check <https://wiki.osdev.org/PC_Speaker>.
154        static CHANNEL2_PORT: IoPort<u8, WriteOnlyAccess> = IoPort::new(0x42);
155
156        /// PIT command port.
157        /// ```text
158        /// Bits         Usage
159        /// 6 and 7      channel
160        /// 4 and 5      Access mode
161        /// 1 to 3       Operating mode
162        /// 0            BCD/Binary mode: 0 = 16-bit binary, 1 = four-digit BCD
163        /// ```
164        static MODE_COMMAND_PORT: IoPort<u8, WriteOnlyAccess> = IoPort::new(0x43);
165    }
166}

168const TIMER_RATE: u32 = 1193182;
169const TIMER_INTERRUPT: u8 = 0; // ISA interrupt.
170
171pub(crate) fn init(operating_mode: OperatingMode) {
172    // Set PIT mode
173    // Bit 0 is BCD/binary mode, which is always set to binary mode(value: 0)
174    MODE_COMMAND_PORT.write(
175        ((operating_mode as u8) << 1)
176            | ((AccessMode::LowAndHighByte as u8) << 4)
177            | ((Channel::Channel0 as u8) << 6),
178    );
179
180    // Set timer frequency
181    const CYCLE: u32 = TIMER_RATE / TIMER_FREQ as u32;
182    CHANNEL0_PORT.write((CYCLE & 0xFF) as _);
183    CHANNEL0_PORT.write((CYCLE >> 8) as _);
184}

186/// Enables the interrupt line that is connected to the PIT.
187pub(crate) fn enable_interrupt(irq_line: IrqLine) -> MappedIrqLine {
188    IRQ_CHIP
189        .get()
190        .unwrap()
191        .map_isa_pin_to(irq_line, TIMER_INTERRUPT)
192        .unwrap()
193}
