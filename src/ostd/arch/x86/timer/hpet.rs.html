22static HPET_INSTANCE: Once<Hpet> = Once::new();

24const OFFSET_ID_REGISTER: usize = 0x000;
25const OFFSET_CONFIGURATION_REGISTER: usize = 0x010;
26const OFFSET_INTERRUPT_STATUS_REGISTER: usize = 0x020;
27#[expect(dead_code)]
28const OFFSET_MAIN_COUNTER_VALUE_REGISTER: usize = 0x0F0;
29
30#[expect(dead_code)]
31const HPET_FREQ: usize = 1_000_000_000_000_000;
32
33#[derive(Debug)]
34#[repr(C)]
35struct HpetTimerRegister {
36    configuration_and_capabilities_register: u32,
37    timer_comparator_value_register: u32,
38    fsb_interrupt_route_register: u32,
39}
40
41struct Hpet {
42    information_register: VolatileRef<'static, u32, ReadOnly>,
43    _general_configuration_register: VolatileRef<'static, u32, ReadWrite>,
44    _general_interrupt_status_register: VolatileRef<'static, u32, ReadWrite>,
45
46    _timer_registers: Vec<VolatileRef<'static, HpetTimerRegister, ReadWrite>>,
47    _irq: MappedIrqLine,
48}
49
50impl Hpet {
51    /// # Safety
52    ///
53    /// The caller must ensure that the address is valid and points to the HPET MMIO region.
54    unsafe fn new(base_address: NonNull<u8>) -> Hpet {
55        // SAFETY: The safety is upheld by the caller.
56        let (
57            information_register,
58            general_configuration_register,
59            general_interrupt_status_register,
60        ) = unsafe {
61            (
62                VolatileRef::new_read_only(base_address.add(OFFSET_ID_REGISTER).cast::<u32>()),
63                VolatileRef::new(
64                    base_address
65                        .add(OFFSET_CONFIGURATION_REGISTER)
66                        .cast::<u32>(),
67                ),
68                VolatileRef::new(
69                    base_address
70                        .add(OFFSET_INTERRUPT_STATUS_REGISTER)
71                        .cast::<u32>(),
72                ),
73            )
74        };
75
76        let num_comparator = ((information_register.as_ptr().read() & 0x1F00) >> 8) as u8 + 1;
77        let num_comparator = num_comparator as usize;
78
79        // FIXME: We now trust the hardware. We should instead find a way to check that
80        // `num_comparator` are reasonable values before proceeding.
81
82        let mut comparators = Vec::with_capacity(num_comparator);
83        for i in 0..num_comparator {
84            // SAFETY: The safety is upheld by the caller and the correctness of the information
85            // value.
86            let comp = unsafe {
87                VolatileRef::new(
88                    base_address
89                        .add(0x100)
90                        .add(i * 0x20)
91                        .cast::<HpetTimerRegister>(),
92                )
93            };
94            comparators.push(comp);
95        }
96
97        let irq = IrqLine::alloc().unwrap();
98        // FIXME: The index of HPET interrupt needs to be tested.
99        let irq = IRQ_CHIP.get().unwrap().map_isa_pin_to(irq, 0).unwrap();
100
101        Hpet {
102            information_register,
103            _general_configuration_register: general_configuration_register,
104            _general_interrupt_status_register: general_interrupt_status_register,
105            _timer_registers: comparators,
106            _irq: irq,
107        }
108    }
109
110    #[expect(dead_code)]
111    pub fn hardware_rev(&self) -> u8 {
112        (self.information_register.as_ptr().read() & 0xFF) as u8
113    }
114
115    #[expect(dead_code)]
116    pub fn num_comparators(&self) -> u8 {
117        ((self.information_register.as_ptr().read() & 0x1F00) >> 8) as u8 + 1
118    }
119
120    #[expect(dead_code)]
121    pub fn main_counter_is_64bits(&self) -> bool {
122        (self.information_register.as_ptr().read() & 0x2000) != 0
123    }
124
125    #[expect(dead_code)]
126    pub fn legacy_irq_capable(&self) -> bool {
127        (self.information_register.as_ptr().read() & 0x8000) != 0
128    }
129
130    #[expect(dead_code)]
131    pub fn pci_vendor_id(&self) -> u16 {
132        ((self.information_register.as_ptr().read() & 0xFFFF_0000) >> 16) as u16
133    }
134}

136/// HPET init, need to init IOAPIC before init this function
137#[expect(dead_code)]
138pub fn init() -> Result<(), AcpiError> {
139    let tables = get_acpi_tables().unwrap();
140
141    let hpet_info = HpetInfo::new(&tables)?;
142    assert_ne!(hpet_info.base_address, 0, "HPET address should not be zero");
143
144    let base = NonNull::new(paddr_to_vaddr(hpet_info.base_address) as *mut u8).unwrap();
145    // SAFETY: The base address is from the ACPI table and points to the HPET MMIO region.
146    let hpet = unsafe { Hpet::new(base) };
147    HPET_INSTANCE.call_once(|| hpet);
148
149    Ok(())
150}
