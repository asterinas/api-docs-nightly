3use bitflags::bitflags;
4
5bitflags! {
6    /// Global Status of the IOMMU. All fields is read-only. Some description of the fields
7    /// is related to the fields in `GlobalCommand`.
8    pub struct GlobalStatus: u32{
9        /// Compatibility Format Interrupt Status. The value reported in this field is
10        /// applicable only when interrupt-remapping is enabled and extended interrupt
11        /// model (x2APIC) mode is not enables.
12        ///
13        /// - 0: Compatibility format interrupts are blocked.
14        /// - 1: Compatibility format interrupts are processed as pass-through (bypassing
15        /// interrupt remapping).
16        const CFIS =        1 << 23;
17        /// Interrupt Remapping Table Pointer Status.
18        ///
19        /// This field is cleared by hardware when software sets the SIRTP field in the Global
20        /// Command register. This field is Set by hardware when hardware completes the
21        /// `Set Interrupt Remap Table Pointer` operation using the value provided in the
22        /// Interrupt Remapping Table Address register.
23        const IRTPS =       1 << 24;
24        /// Interrupt Remapping Enable Status.
25        ///
26        /// - 0: Interrupt-remapping hardware is not enabled.
27        /// - 1: Interrupt-remapping hardware is enabled.
28        const IRES =        1 << 25;
29        /// Queued Invalidation Enable Status.
30        ///
31        /// - 0: queued invalidation is not enabled.
32        /// - 1: queued invalidation is enabled.
33        const QIES =        1 << 26;
34        /// Write Buffer Flush Status. This field is valid only for implementations requiring
35        /// write buffer flushing. This field indicates the status of the write buffer flush
36        /// command.
37        ///
38        /// - Set by hardware when software sets the WBF field in the Global Command register.
39        /// - Cleared by hardware when hardware completes the write buffer flushing operation.
40        const WBFS =        1 << 27;
41        /// Root Table Pointer Status.
42        ///
43        /// This field is cleared by hardware when software sets the SRTP field in the Global
44        /// Command register. This field is set by hardware when hardware completes the
45        /// `Set Root Table Pointer`` operation using the value provided in the Root Table
46        /// Address register.
47        const RTPS =        1 << 30;
48        /// Translation Enable Status.
49        ///
50        /// - 0: DMA remapping is not enabled.
51        /// - 1: DMA remapping is enabled.
52        const TES =         1 << 31;
53    }
54}
