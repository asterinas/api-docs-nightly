3use bitflags::bitflags;
4
5bitflags! {
6    /// Global Command to enable functions in IOMMU. All field is write-only.
7    pub struct GlobalCommand: u32{
8        /// Compatibility Format Interrupt, only valid if interrupt-remapping is supported.
9        ///
10        /// Interrupt remapping will block Compatibility format interrupts if set to 0.
11        /// Otherwise these interrupts will bypass interrupt remapping.
12        const CFI =         1 << 23;
13        /// Set Interrupt Remap Table Pointer, only valid if interrupt-remapping is supported.
14        ///
15        /// Software sets this filed to set/update the interrupt remapping table pointer used
16        /// by hardware. The interrupt remapping table pointer is specified through the Interrupt
17        /// Remapping Table Address (IRTA_REG) register.
18        const SIRTP =       1 << 24;
19        /// Interrupt Remapping Enable, only valid if hardware support interrupt remapping.
20        /// Set to 1 if enable interrupt-remapping hardware.
21        ///
22        /// Hardware reports the status of the interrupt remapping enable operation through the
23        /// IRES field in the Global Status register.
24        const IRE =         1 << 25;
25        /// Queued Invalidation Enable, only valid if hardware support queued invalidations.
26        /// Set to 1 to enable use of queued validations.
27        ///
28        /// Hardware reports the status of queued invalidation enable operation through QIES
29        /// field in Global Status register.
30        const QIE =         1 << 26;
31        /// Write Buffer Flush, only valid for implementations requiring write buffer flushing.
32        ///
33        /// Software sets this field to request that hardware flush the Root-Complex
34        /// internal write buffers. This is done to ensure any updates to the memory resident
35        /// remapping structures are not held in any internal write posting buffers.
36        ///
37        /// Hardware reports the status of the write buffer flushing operation through WBFS
38        /// field in Global Status register.
39        const WBF =         1 << 27;
40        /// Set Root Table Pointer.
41        ///
42        /// Software sets this field to set/update the root-table pointer (and translation
43        /// table mode) used by hardware. The root-table pointer (and translation table
44        /// mode) is specified through the Root Table Address (RTADDR_REG) register.
45        const SRTP =        1 << 30;
46        /// Translation Enable.
47        ///
48        /// Software writes to this field to request hardware to enable/disable DMA remapping.
49        ///
50        /// 0: Disable DMA remapping; 1: Enable DMA remapping.
51        ///
52        /// Hardware reports the status of the translation enable operation through TES field
53        /// in the Global Status register.
54        const TE =          1 << 31;
55    }
56}
