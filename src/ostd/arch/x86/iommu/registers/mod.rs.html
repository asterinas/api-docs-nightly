67/// Important registers used by IOMMU.
68#[derive(Debug)]
69pub struct IommuRegisters {
70    version: VolatileRef<'static, u32, ReadOnly>,
71    capability: VolatileRef<'static, u64, ReadOnly>,
72    extended_capability: VolatileRef<'static, u64, ReadOnly>,
73    global_command: VolatileRef<'static, u32, WriteOnly>,
74    global_status: VolatileRef<'static, u32, ReadOnly>,
75    root_table_address: VolatileRef<'static, u64, ReadWrite>,
76    context_command: VolatileRef<'static, u64, ReadWrite>,
77
78    interrupt_remapping_table_addr: VolatileRef<'static, u64, ReadWrite>,
79
80    invalidate: InvalidationRegisters,
81}
82
83impl IommuRegisters {
84    /// Reads the version of IOMMU
85    #[expect(dead_code)]
86    pub fn read_version(&self) -> IommuVersion {
87        let version = self.version.as_ptr().read();
88        IommuVersion {
89            major: version.get_bits(4..8) as u8,
90            minor: version.get_bits(0..4) as u8,
91        }
92    }
93
94    /// Reads the capability of IOMMU
95    pub fn read_capability(&self) -> Capability {
96        Capability::new(self.capability.as_ptr().read())
97    }
98
99    /// Reads the extended Capability of IOMMU
100    pub fn read_extended_capability(&self) -> ExtendedCapability {
101        ExtendedCapability::new(self.extended_capability.as_ptr().read())
102    }
103
104    /// Reads the global Status of IOMMU
105    pub fn read_global_status(&self) -> GlobalStatus {
106        GlobalStatus::from_bits_truncate(self.global_status.as_ptr().read())
107    }
108
109    /// Enables DMA remapping with static RootTable
110    pub(super) fn enable_dma_remapping(
111        &mut self,
112        root_table: &'static SpinLock<RootTable, LocalIrqDisabled>,
113    ) {
114        // Set root table address
115        self.root_table_address
116            .as_mut_ptr()
117            .write(root_table.lock().root_paddr() as u64);
118        self.write_global_command(GlobalCommand::SRTP, true);
119        while !self.read_global_status().contains(GlobalStatus::RTPS) {}
120
121        // Enable DMA remapping
122        self.write_global_command(GlobalCommand::TE, true);
123        while !self.read_global_status().contains(GlobalStatus::TES) {}
124    }
125
126    /// Enables Interrupt Remapping with IntRemappingTable
127    pub(super) fn enable_interrupt_remapping(&mut self, table: &'static IntRemappingTable) {
128        assert!(
129            self.read_extended_capability()
130                .flags()
131                .contains(ExtendedCapabilityFlags::IR)
132        );
133        // Set interrupt remapping table address
134        self.interrupt_remapping_table_addr
135            .as_mut_ptr()
136            .write(table.encode());
137        self.write_global_command(GlobalCommand::SIRTP, true);
138        while !self.read_global_status().contains(GlobalStatus::IRTPS) {}
139
140        // Enable Interrupt Remapping
141        self.write_global_command(GlobalCommand::IRE, true);
142        while !self.read_global_status().contains(GlobalStatus::IRES) {}
143
144        // Disable Compatibility format interrupts
145        if self.read_global_status().contains(GlobalStatus::CFIS) {
146            self.write_global_command(GlobalCommand::CFI, false);
147            while self.read_global_status().contains(GlobalStatus::CFIS) {}
148        }
149    }
150
151    pub(super) fn enable_queued_invalidation(&mut self, queue: &Queue) {
152        assert!(
153            self.read_extended_capability()
154                .flags()
155                .contains(ExtendedCapabilityFlags::QI)
156        );
157        self.invalidate.queue_tail.as_mut_ptr().write(0);
158
159        let mut write_value = queue.base_paddr() as u64;
160        // By default, we set descriptor width to 128-bit(0)
161        let descriptor_width = 0b0;
162        write_value |= descriptor_width << 11;
163
164        let write_queue_size = {
165            let mut queue_size = queue.size();
166            assert!(queue_size.is_power_of_two());
167            let mut write_queue_size = 0;
168
169            if descriptor_width == 0 {
170                // 2^(write_queue_size + 8) = number of entries = queue_size
171                assert!(queue_size >= (1 << 8));
172                queue_size >>= 8;
173            } else {
174                // 2^(write_queue_size + 7) = number of entries = queue_size
175                assert!(queue_size >= (1 << 7));
176                queue_size >>= 7;
177            };
178
179            while queue_size & 0b1 == 0 {
180                queue_size >>= 1;
181                write_queue_size += 1;
182            }
183            write_queue_size
184        };
185
186        write_value |= write_queue_size;
187
188        self.invalidate.queue_addr.as_mut_ptr().write(write_value);
189
190        // Enable queued invalidation
191        self.write_global_command(GlobalCommand::QIE, true);
192        while !self.read_global_status().contains(GlobalStatus::QIES) {}
193
194        // Clear the Invalidation Completion Status Register
195        self.invalidate.completion_status.as_mut_ptr().write(1);
196    }
197
198    // Invalidates Interrupt-remapping cache.
199    pub(super) fn invalidate_interrupt_cache(&mut self) {
200        if !self.read_global_status().contains(GlobalStatus::QIES) {
201            self.global_invalidation();
202            return;
203        }
204
205        let mut queue = QUEUE.get().unwrap().lock();
206
207        // Currently, we don't support asynchronous processing in the queue. Therefore, when we
208        // lock the queue, we know that it is empty and that the Invalidation Completion Status
209        // Register has been cleared.
210
211        // Construct an Interrupt Entry Cache Invalidate Descriptor.
212        queue.append_descriptor(InterruptEntryCache::global_invalidation().0);
213        // Construct an Invalidation Wait Descriptor. We need to set the interrupt flag so that the
214        // Invalidation Completion Status Register can report the completion status.
215        queue.append_descriptor(InvalidationWait::with_interrupt_flag().0);
216
217        // Update the queue tail.
218        let tail = queue.tail();
219        self.invalidate
220            .queue_tail
221            .as_mut_ptr()
222            .write((tail << 4) as u64);
223
224        // Wait for completion.
225        while self.invalidate.completion_status.as_ptr().read() == 0 {}
226        // Clear the Invalidation Completion Status Register.
227        self.invalidate.completion_status.as_mut_ptr().write(1);
228    }
229
230    fn global_invalidation(&mut self) {
231        // Set ICC(63) to 1 to requests invalidation and CIRG(62:61) to 01 to indicate global invalidation request.
232        self.context_command
233            .as_mut_ptr()
234            .write(0xA000_0000_0000_0000);
235
236        // Wait for invalidation complete (ICC set to 0).
237        let mut value = 0x8000_0000_0000_0000;
238        while (value & 0x8000_0000_0000_0000) != 0 {
239            value = self.context_command.as_ptr().read();
240        }
241
242        // Set IVT(63) to 1 to requests IOTLB invalidation and IIRG(61:60) to 01 to indicate global invalidation request.
243        self.invalidate
244            .iotlb_invalidate
245            .as_mut_ptr()
246            .write(0x9000_0000_0000_0000);
247    }
248
249    /// Writes value to the global command register. This function will not wait until the command
250    /// is serviced. User need to check the global status register.
251    fn write_global_command(&mut self, command: GlobalCommand, enable: bool) {
252        const ONE_SHOT_STATUS_MASK: u32 = 0x96FF_FFFF;
253        let status = self.global_status.as_ptr().read() & ONE_SHOT_STATUS_MASK;
254        if enable {
255            self.global_command
256                .as_mut_ptr()
257                .write(status | command.bits());
258        } else {
259            self.global_command
260                .as_mut_ptr()
261                .write(status & !command.bits());
262        }
263    }

265    /// Creates an instance from base address
266    fn new(io_mem_builder: &IoMemAllocatorBuilder) -> Option<Self> {
267        let dmar = Dmar::new()?;
268        debug!("DMAR: {:#x?}", dmar);
269
270        let base_address = dmar
271            .remapping_iter()
272            // TODO: Add support for multiple DMA remapping hardware unit definitions (DRHDs). Note
273            // that we use `rev()` here to select the last one, since DRHDs that control specific
274            // devices tend to be reported first.
275            //
276            // For example, Intel(R) Virtualization Technology for Directed I/O (Revision 5.0), 8.4
277            // DMA Remapping Hardware Unit Definition Structure says "If a DRHD structure with
278            // INCLUDE_PCI_ALL flag Set is reported for a Segment, it must be enumerated by BIOS
279            // after all other DRHD structures for the same Segment".
280            .rev()
281            .find_map(|remapping| match remapping {
282                Remapping::Drhd(drhd) => Some(drhd.register_base_addr()),
283                _ => None,
284            })
285            .expect("no DRHD structure found in the DMAR table");
286        assert_ne!(base_address, 0, "IOMMU address should not be zero");
287        debug!("IOMMU base address: {:#x?}", base_address);
288
289        io_mem_builder.remove(base_address as usize..(base_address as usize + PAGE_SIZE));
290        let base = NonNull::new(paddr_to_vaddr(base_address as usize) as *mut u8).unwrap();
291
292        // SAFETY:
293        // - We trust the ACPI tables (as well as the DRHD in them), from which the base address is
294        //   obtained, so it is a valid IOMMU base address.
295        // - `io_mem_builder.remove()` guarantees that we have exclusive ownership of all the IOMMU
296        //   registers.
297        let iommu_regs = unsafe {
298            fault::init(base);
299
300            Self {
301                version: VolatileRef::new_read_only(base.cast::<u32>()),
302                capability: VolatileRef::new_read_only(base.add(0x08).cast::<u64>()),
303                extended_capability: VolatileRef::new_read_only(base.add(0x10).cast::<u64>()),
304                global_command: VolatileRef::new_restricted(
305                    WriteOnly,
306                    base.add(0x18).cast::<u32>(),
307                ),
308                global_status: VolatileRef::new_read_only(base.add(0x1C).cast::<u32>()),
309                root_table_address: VolatileRef::new(base.add(0x20).cast::<u64>()),
310                context_command: VolatileRef::new(base.add(0x28).cast::<u64>()),
311
312                interrupt_remapping_table_addr: VolatileRef::new(base.add(0xb8).cast::<u64>()),
313
314                invalidate: InvalidationRegisters::new(base),
315            }
316        };
317
318        debug!("IOMMU registers:{:#x?}", iommu_regs);
319        debug!("IOMMU capability:{:#x?}", iommu_regs.read_capability());
320        debug!(
321            "IOMMU extend capability:{:#x?}",
322            iommu_regs.read_extended_capability()
323        );
324
325        Some(iommu_regs)
326    }
327}

329pub(super) static IOMMU_REGS: Once<SpinLock<IommuRegisters, LocalIrqDisabled>> = Once::new();
330
331pub(super) fn init(io_mem_builder: &IoMemAllocatorBuilder) -> Result<(), IommuError> {
332    let iommu_regs = IommuRegisters::new(io_mem_builder).ok_or(IommuError::NoIommu)?;
333    IOMMU_REGS.call_once(|| SpinLock::new(iommu_regs));
334    Ok(())
