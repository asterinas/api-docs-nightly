3use core::fmt::Debug;
4
5use bitflags::bitflags;

7/// Capability in IOMMU.
8pub struct Capability(u64);
9
10impl Capability {
11    /// Create Capability from `value`
12    pub const fn new(value: u64) -> Self {
13        Self(value)
14    }
15
16    /// Capability flags
17    pub const fn flags(&self) -> CapabilityFlags {
18        CapabilityFlags::from_bits_truncate(self.0)
19    }
20
21    /// Number of Fault-recording. The maximum number of fault recording registers per
22    /// remapping hardware unit is 256.
23    ///
24    /// Number of fault recording registers is computed as N+1, where N is the value
25    /// reported in this field.
26    pub const fn fault_recording_number(&self) -> u64 {
27        const NFR_MASK: u64 = 0xFF << 40;
28        (self.0 & NFR_MASK) >> 40
29    }
30
31    /// Maximum Address Mask Value, indicates the maximum supported value for them Address
32    /// Mask (AM) field in the Invalidation Address register (IVA_REG), and IOTLB Invalidation
33    /// Descriptor (iotlb_inv_dsc) used for invalidations of second-stage translation.
34    pub const fn maximum_address_mask_value(&self) -> u64 {
35        const MAMV_MASK: u64 = 0x3F << 48;
36        (self.0 & MAMV_MASK) >> 48
37    }
38
39    /// Number of domain support.
40    ///
41    /// ```text
42    /// 0 => 4-bit domain-ids with support for up to 16 domains.
43    /// 1 => 6-bit domain-ids with support for up to 64 domains.
44    /// 2 => 8-bit domain-ids with support for up to 256 domains.
45    /// 3 => 10-bit domain-ids with support for up to 1024 domains.
46    /// 4 => 12-bit domain-ids with support for up to 4K domains.
47    /// 5 => 14-bit domain-ids with support for up to 16K domains.
48    /// 6 => 16-bit domain-ids with support for up to 64K domains.
49    /// 7 => Reserved.
50    /// ```
51    pub const fn domain_support_number(&self) -> u64 {
52        const ND_MASK: u64 = 0x7;
53        self.0 & ND_MASK
54    }
55
56    /// Supported Adjusted Guest Address Widths.
57    pub const fn supported_adjusted_guest_address_widths(&self) -> CapabilitySagaw {
58        CapabilitySagaw::from_bits_truncate(self.0 >> 8)
59    }
60
61    /// Fault-recording Register offset, specifies the offset of the first fault recording
62    /// register relative to the register base address of this remapping hardware unit.
63    ///
64    /// If the register base address is X, and the value reported in this field
65    /// is Y, the address for the first fault recording register is calculated as X+(16*Y).
66    pub const fn fault_recording_register_offset(&self) -> u64 {
67        const FRO_MASK: u64 = 0x3FF << 24;
68        (self.0 & FRO_MASK) >> 24
69    }
70
71    /// Second Stage Large Page Support.
72    pub const fn second_stage_large_page_support(&self) -> CapabilitySslps {
73        CapabilitySslps::from_bits_truncate(self.0 >> 34)
74    }
75
76    /// Maximum Guest Address Width. The maximum guest physical address width supported
77    /// by second-stage translation in remapping hardware.
78    /// MGAW is computed as (N+1), where N is the valued reported in this field.
79    pub const fn maximum_guest_address_width(&self) -> u64 {
80        const MGAW_MASK: u64 = 0x3F << 16;
81        (self.0 & MGAW_MASK) >> 16
82    }
83}
84
85impl Debug for Capability {
86    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
87        f.debug_struct("Capability")
88            .field("flags", &self.flags())
89            .field(
90                "maximum_guest_address_width",
91                &self.maximum_guest_address_width(),
92            )
93            .field(
94                "second_stage_large_page_support",
95                &self.second_stage_large_page_support(),
96            )
97            .field(
98                "fault_recording_register_offset",
99                &self.fault_recording_register_offset(),
100            )
101            .field(
102                "supported_adjusted_guest_address_widths",
103                &self.supported_adjusted_guest_address_widths(),
104            )
105            .field("domain_support_number", &self.domain_support_number())
106            .field(
107                "maximum_address_mask_value",
108                &self.maximum_address_mask_value(),
109            )
110            .field("fault_recording_number", &self.fault_recording_number())
111            .field("raw", &self.0)
112            .finish()
113    }
114}

116bitflags! {
117    /// Capability flags in IOMMU.
118    pub struct CapabilityFlags: u64 {
119        /// Required Write-Buffer Flushing.
120        const RWBF =        1 << 4;
121        /// Protected Low-Memory Region
122        const PLMR =        1 << 5;
123        /// Protected High-Memory Region
124        const PHMR =        1 << 6;
125        /// Caching Mode
126        const CM =          1 << 7;
127        /// Zero Length Read. Whether the remapping hardware unit supports zero length DMA
128        /// read requests to write-only pages.
129        const ZLR =         1 << 22;
130        /// Page Selective Invalidation. Whether hardware supports page-selective invalidation
131        /// for IOTLB.
132        const PSI =         1 << 39;
133        /// Write Draining.
134        const DWD =         1 << 54;
135        /// Read Draining.
136        const DRD =         1 << 55;
137        /// First Stage 1-GByte Page Support.
138        const FS1GP =       1 << 56;
139        /// Posted Interrupts Support.
140        const PI =          1 << 59;
141        /// First Stage 5-level Paging Support.
142        const FS5LP =       1 << 60;
143        /// Enhanced Command Support.
144        const ECMDS =       1 << 61;
145        /// Enhanced Set Interrupt Remap Table Pointer Support.
146        const ESIRTPS =     1 << 62;
147        /// Enhanced Set Root Table Pointer Support.
148        const ESRTPS =      1 << 63;
149    }
150}

152bitflags! {
153    /// Supported Adjusted Guest Address Widths (SAGAW) in IOMMU.
154    pub struct CapabilitySagaw: u64 {
155        /// 39-bit AGAW (3-level page-table).
156        const AGAW_39BIT_3LP = 1 << 1;
157        /// 48-bit AGAW (4-level page-table).
158        const AGAW_48BIT_4LP = 1 << 2;
159        /// 57-bit AGAW (5-level page-table).
160        const AGAW_57BIT_5LP = 1 << 3;
161        // 0th and 4th bits are reserved.
162    }
163}
164
165bitflags! {
166    /// Second Stage Large Page Support (SSLPS) in IOMMU.
167    pub struct CapabilitySslps: u64 {
168        /// 21-bit offset to page frame (2MB).
169        const PAGE_21BIT_2MB = 1 << 0;
170        /// 30-bit offset to page frame (1GB).
171        const PAGE_30BIT_1GB = 1 << 1;
172        // 2nd and 3rd bits are reserved.
173    }
