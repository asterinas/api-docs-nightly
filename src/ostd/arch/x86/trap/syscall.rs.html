39/// # Safety
40///
41/// The caller needs to ensure that `gdt::init_on_cpu` has been called before,
42/// so the segment selectors used in the `syscall` and `sysret` instructions
43/// have been properly initialized.
44pub(super) unsafe fn init_on_cpu() {
45    // We now assume that all x86-64 CPUs should support the `syscall` and `sysret` instructions.
46    // Otherwise, we should check `has_extensions(IsaExtensions::SYSCALL)` here.
47
48    // Flags to clear on syscall.
49    //
50    // Linux 5.0 uses TF|DF|IF|IOPL|AC|NT. Reference:
51    // <https://github.com/torvalds/linux/blob/v5.0/arch/x86/kernel/cpu/common.c#L1559-L1562>
52    const RFLAGS_MASK: u64 = 0x47700;
53
54    // SAFETY: The segment selectors are correctly initialized (as upheld by the caller), and the
55    // entry point and flags to clear are also correctly set, so enabling the `syscall` and
56    // `sysret` instructions is safe.
57    unsafe {
58        LStar::write(VirtAddr::new(syscall_entry as *const () as usize as u64));
59        SFMask::write(RFlags::from_bits(RFLAGS_MASK).unwrap());
60
61        // Enable the `syscall` and `sysret` instructions.
62        Efer::update(|efer| {
63            efer.insert(EferFlags::SYSTEM_CALL_EXTENSIONS);
64        });
65    }
66}
67
68unsafe extern "C" {
69    unsafe fn syscall_entry();
70    unsafe fn syscall_return(regs: &mut RawUserContext);
71}
72
73impl RawUserContext {
74    /// Goes to user space with the context, and comes back when a trap occurs.
75    ///
76    /// On return, the context will be reset to the status before the trap.
77    /// Trap reason and error code will be placed at `trap_num` and `error_code`.
78    ///
79    /// If the trap was triggered by `syscall` instruction, the `trap_num` will be set to `0x100`.
80    ///
81    /// If `trap_num` is `0x100`, it will go user by `sysret` (`rcx` and `r11` are dropped),
82    /// otherwise it will use `iret`.
83    pub(in crate::arch) fn run(&mut self) {
84        // Return to userspace with interrupts disabled. Otherwise, interrupts
85        // after executing `swapgs` will mess up the CPU state.
86        crate::arch::irq::disable_local();
87        unsafe {
88            syscall_return(self);
89        }
90    }
91}
