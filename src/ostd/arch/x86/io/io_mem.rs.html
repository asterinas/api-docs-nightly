13/// Reads from a pointer with a non-tearing memory load.
14///
15/// This function is semantically equivalent to `core::ptr::read_volatile`
16/// but is manually implemented with a single memory load instruction.
17///
18/// # Safety
19///
20/// Same as the safety requirement of `core::ptr::read_volatile`.
21///
22/// # Guarantee
23///
24/// The single-memory-load-instruction guarantee is particularly useful for
25/// Confidential VMs (CVMs) such as Intel TDX and AMD SEV,
26/// where memory loads may cause CPU exceptions (#VE and #VC, respectively)
27/// and the kernel has to handle such exceptions
28/// by decoding the faulting CPU instruction.
29/// As such, the kernel must be compiled to emit simple load/store CPU instructions.
30pub(crate) unsafe fn read_once<T: PodOnce>(ptr: *const T) -> T {
31    debug_assert!(ptr.is_aligned());
32    let mut val: u64 = 0;
33
34    // SAFETY: The caller guarantees `ptr` is valid for reads.
35    unsafe {
36        // EFFICIENCY: The match should be optimized out for release builds.
37        //
38        // This match is resolved at compile-time via monomorphization.
39        // Since `size_of::<T>()` is a constant for each concrete instance of this
40        // function, the compiler eliminates the branch and only emits the
41        // `MOV` instruction for the matching size.
42        match size_of::<T>() {
43            1 => {
44                asm!("mov {0:l}, [{1}]", out(reg) val, in(reg) ptr, options(nostack, readonly, preserves_flags))
45            }
46            2 => {
47                asm!("mov {0:x}, [{1}]", out(reg) val, in(reg) ptr, options(nostack, readonly, preserves_flags))
48            }
49            4 => {
50                asm!("mov {0:e}, [{1}]", out(reg) val, in(reg) ptr, options(nostack, readonly, preserves_flags))
51            }
52            8 => {
53                asm!("mov {0}, [{1}]", out(reg) val, in(reg) ptr, options(nostack, readonly, preserves_flags))
54            }
55            _ => core::hint::unreachable_unchecked(),
56        }
57        // EFFICIENCY: This should compile to a no-op for release builds.
58        // This is because both the source and destination locations fit in a register.
59        // So it only _re-interprets_ bits and no copying is needed.
60        core::ptr::read((&val as *const u64).cast::<T>())
61    }
62}

64/// Writes to a pointer with a non-tearing memory store.
65///
66/// # Safety
67///
68/// Same as the safety requirement of `core::ptr::write_volatile`.
69///
70/// # Guarantee
71///
72/// Refer to the "Guarantee" section of [`read_once`].
73pub(crate) unsafe fn write_once<T: PodOnce>(ptr: *mut T, val: T) {
74    debug_assert!(ptr.is_aligned());
75    let mut tmp: u64 = 0;
76
77    // SAFETY: The caller guarantees `ptr` is valid for writes.
78    unsafe {
79        // EFFICIENCY: This should be a no-op for release build.
80        // This is because both the source and destination locations fit in a register.
81        // So it only _re-interprets_ bits and no copying is needed.
82        core::ptr::write((&mut tmp as *mut u64).cast::<T>(), val);
83
84        // EFFICIENCY: The match here has no overhead for release build.
85        //
86        // This match is resolved at compile-time via monomorphization.
87        // Since `size_of::<T>()` is a constant for each concrete instance of this
88        // function, the compiler eliminates the branch and only emits the
89        // `MOV` instruction for the matching size.
90        match size_of::<T>() {
91            1 => {
92                asm!("mov [{0}], {1:l}", in(reg) ptr, in(reg) tmp, options(nostack, preserves_flags))
93            }
94            2 => {
95                asm!("mov [{0}], {1:x}", in(reg) ptr, in(reg) tmp, options(nostack, preserves_flags))
96            }
97            4 => {
98                asm!("mov [{0}], {1:e}", in(reg) ptr, in(reg) tmp, options(nostack, preserves_flags))
99            }
100            8 => {
101                asm!("mov [{0}], {1}", in(reg) ptr, in(reg) tmp, options(nostack, preserves_flags))
102            }
103            _ => core::hint::unreachable_unchecked(),
104        }
105    }
106}

108macro_rules! do_copy_mmio_impl {
109    (dst: $dst:ident, src: $src:ident, count: $count:ident) => {{
110        if $count == 0 { return; }
111
112        // SAFETY: The caller guarantees source and destination are valid for `count` bytes.
113        unsafe {
114            asm!(
115                "rep movsb",
116                inout("rdi") $dst => _,
117                inout("rsi") $src => _,
118                inout("rcx") $count => _,
119                options(nostack, preserves_flags)
120            );
121        }
122    }};
123}
124
125macro_rules! do_memset_mmio_impl {
126    (dst: $dst:ident, value: $value:ident, count: $count:ident) => {{
127        if $count == 0 { return; }
128
129        // SAFETY: The caller guarantees destination pointer is valid for `count` bytes.
130        unsafe {
131            asm!(
132                "rep stosb",
133                inout("rdi") $dst => _,
134                inout("rcx") $count => _,
135                in("al") $value,
136                options(nostack, preserves_flags)
137            );
138        }
139    }};
140}

142/// Copies from MMIO to regular memory.
143///
144/// # Safety
145///
146/// - `dst_ptr` must be valid for writes of `count` bytes.
147/// - `src_io_ptr` must be valid for MMIO reads of `count` bytes.
148pub(crate) unsafe fn copy_from_mmio(dst_ptr: *mut u8, src_io_ptr: *const u8, count: usize) {
149    if_tdx_enabled!({
150        // SAFETY: The caller guarantees both pointers are valid for `count` bytes.
151        return unsafe { cvm::copy_from_mmio(dst_ptr, src_io_ptr, count) };
152    });
153
154    do_copy_mmio_impl!(dst: dst_ptr, src: src_io_ptr, count: count);
155}

157/// Copies from regular memory to MMIO.
158///
159/// # Safety
160///
161/// - `src_ptr` must be valid for reads of `count` bytes.
162/// - `dst_io_ptr` must be valid for MMIO writes of `count` bytes.
163pub(crate) unsafe fn copy_to_mmio(src_ptr: *const u8, dst_io_ptr: *mut u8, count: usize) {
164    if_tdx_enabled!({
165        // SAFETY: The caller guarantees both pointers are valid for `count` bytes.
166        return unsafe { cvm::copy_to_mmio(src_ptr, dst_io_ptr, count) };
167    });
168
169    do_copy_mmio_impl!(dst: dst_io_ptr, src: src_ptr, count: count);
170}

172/// Copies from MMIO to fallible memory and returns bytes copied.
173///
174/// # Safety
175///
176/// - `dst_ptr` must be valid or in user space for writes of `count` bytes.
177/// - `src_io_ptr` must be valid for MMIO reads of `count` bytes.
178pub(crate) unsafe fn copy_from_mmio_fallible(
179    dst_ptr: *mut u8,
180    src_io_ptr: *const u8,
181    count: usize,
182) -> usize {
183    if_tdx_enabled!({
184        // SAFETY: The caller guarantees both pointers are valid for `count` bytes.
185        return unsafe { cvm::copy_from_mmio_fallible(dst_ptr, src_io_ptr, count) };
186    });
187
188    // SAFETY: The caller guarantees the source and destination are valid.
189    let failed_bytes = unsafe { __memcpy_fallible(dst_ptr, src_io_ptr, count) };
190    count - failed_bytes
191}

193/// Copies from fallible memory to MMIO and returns bytes copied.
194///
195/// # Safety
196///
197/// - `src_ptr` must be valid or in user space for reads of `count` bytes.
198/// - `dst_io_ptr` must be valid for MMIO writes of `count` bytes.
199pub(crate) unsafe fn copy_to_mmio_fallible(
200    src_ptr: *const u8,
201    dst_io_ptr: *mut u8,
202    count: usize,
203) -> usize {
204    if_tdx_enabled!({
205        // SAFETY: The caller guarantees both pointers are valid for `count` bytes.
206        return unsafe { cvm::copy_to_mmio_fallible(src_ptr, dst_io_ptr, count) };
207    });
208
209    // SAFETY: The caller guarantees the source and destination are valid.
210    let failed_bytes = unsafe { __memcpy_fallible(dst_io_ptr, src_ptr, count) };
211    count - failed_bytes
212}

214/// Fills MMIO with `value` for `count` bytes.
215///
216/// # Safety
217///
218/// - `dst_io_ptr` must be valid for MMIO writes of `count` bytes.
219pub(crate) unsafe fn memset_mmio(dst_io_ptr: *mut u8, value: u8, count: usize) {
220    if_tdx_enabled!({
221        // SAFETY: The caller guarantees the destination pointer is valid for `count` bytes.
222        return unsafe { cvm::memset_mmio(dst_io_ptr, value, count) };
223    });
224
225    do_memset_mmio_impl!(dst: dst_io_ptr, value: value, count: count);
