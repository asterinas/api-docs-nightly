17fn parse_bootloader_name(mb1_info: &MultibootLegacyInfo) -> Option<&str> {
18    // SAFETY: The bootloader name is a C-style NUL-terminated string because of the contract with
19    // the Multiboot loader.
20    unsafe { parse_as_cstr(mb1_info.boot_loader_name) }
21}
22
23fn parse_kernel_commandline(mb1_info: &MultibootLegacyInfo) -> Option<&str> {
24    // SAFETY: The bootloader name is a C-style NUL-terminated string because of the contract with
25    // the Multiboot loader.
26    unsafe { parse_as_cstr(mb1_info.cmdline) }
27}
28
29unsafe fn parse_as_cstr<'a>(ptr: u32) -> Option<&'a str> {
30    if ptr == 0 {
31        return None;
32    }
33
34    let name_ptr = paddr_to_vaddr(ptr as usize) as *const core::ffi::c_char;
35    // SAFETY: The safety is upheld by the caller.
36    let name_cstr = unsafe { core::ffi::CStr::from_ptr(name_ptr) };
37
38    name_cstr.to_str().ok()
39}

41fn parse_initramfs(mb1_info: &MultibootLegacyInfo) -> Option<&[u8]> {
42    // FIXME: We think all modules are initramfs, can this cause problems?
43    if mb1_info.mods_count == 0 {
44        return None;
45    }
46
47    let mods_addr = paddr_to_vaddr(mb1_info.mods_addr as usize) as *const u32;
48    // SAFETY: We have checked `mods_count` above. By the contract with the Multiboot loader, the
49    // module addresses are available.
50    let (start, end) = unsafe { (*mods_addr, *mods_addr.add(1)) };
51
52    let ptr = paddr_to_vaddr(start as usize) as *const u8;
53    let len = (end - start) as usize;
54
55    // SAFETY: The initramfs is safe to read because of the contract with the loader.
56    Some(unsafe { core::slice::from_raw_parts(ptr, len) })
57}

59fn parse_acpi_arg(_mb1_info: &MultibootLegacyInfo) -> BootloaderAcpiArg {
60    // The multiboot protocol does not contain RSDP address.
61    // TODO: What about UEFI?
62    BootloaderAcpiArg::NotProvided
63}

65fn parse_framebuffer_info(mb1_info: &MultibootLegacyInfo) -> Option<BootloaderFramebufferArg> {
66    if mb1_info.framebuffer_table.addr == 0 {
67        return None;
68    }
69
70    Some(BootloaderFramebufferArg {
71        address: mb1_info.framebuffer_table.addr as usize,
72        width: mb1_info.framebuffer_table.width as usize,
73        height: mb1_info.framebuffer_table.height as usize,
74        bpp: mb1_info.framebuffer_table.bpp as usize,
75    })
76}

78fn parse_memory_regions(mb1_info: &MultibootLegacyInfo) -> MemoryRegionArray {
79    let mut regions = MemoryRegionArray::new();
80
81    // Add the regions in the multiboot protocol.
82    for entry in mb1_info.get_memory_map() {
83        let region = MemoryRegion::new(
84            entry.base_addr().try_into().unwrap(),
85            entry.length().try_into().unwrap(),
86            entry.memory_type(),
87        );
88        regions.push(region).unwrap();
89    }
90
91    // Add the framebuffer region.
92    if let Some(fb) = parse_framebuffer_info(mb1_info) {
93        regions.push(MemoryRegion::framebuffer(&fb)).unwrap();
94    }
95
96    // Add the kernel region.
97    regions.push(MemoryRegion::kernel()).unwrap();
98
99    // Add the initramfs region.
100    if let Some(initramfs) = parse_initramfs(mb1_info) {
101        regions.push(MemoryRegion::module(initramfs)).unwrap();
102    }
103
104    // Add the AP boot code region that will be copied into by the BSP.
105    regions
106        .push(super::smp::reclaimable_memory_region())
107        .unwrap();
108
109    // Add the kernel cmdline and boot loader name region since Grub does not specify it.
110    if let Some(kcmdline) = parse_kernel_commandline(mb1_info) {
111        regions
112            .push(MemoryRegion::module(kcmdline.as_bytes()))
113            .unwrap();
114    }
115    if let Some(bootloader_name) = parse_bootloader_name(mb1_info) {
116        regions
117            .push(MemoryRegion::module(bootloader_name.as_bytes()))
118            .unwrap();
119    }
120
121    regions.into_non_overlapping()
122}

124/// Representation of Multiboot Information according to specification.
125///
126/// Ref: <https://www.gnu.org/software/grub/manual/multiboot/multiboot.html#Boot-information-format>.
127///
128///```text
129///         +-------------------+
130/// 0       | flags             |    (required)
131///         +-------------------+
132/// 4       | mem_lower         |    (present if flags[0] is set)
133/// 8       | mem_upper         |    (present if flags[0] is set)
134///         +-------------------+
135/// 12      | boot_device       |    (present if flags[1] is set)
136///         +-------------------+
137/// 16      | cmdline           |    (present if flags[2] is set)
138///         +-------------------+
139/// 20      | mods_count        |    (present if flags[3] is set)
140/// 24      | mods_addr         |    (present if flags[3] is set)
141///         +-------------------+
142/// 28 - 40 | syms              |    (present if flags[4] or
143///         |                   |                flags[5] is set)
144///         +-------------------+
145/// 44      | mmap_length       |    (present if flags[6] is set)
146/// 48      | mmap_addr         |    (present if flags[6] is set)
147///         +-------------------+
148/// 52      | drives_length     |    (present if flags[7] is set)
149/// 56      | drives_addr       |    (present if flags[7] is set)
150///         +-------------------+
151/// 60      | config_table      |    (present if flags[8] is set)
152///         +-------------------+
153/// 64      | boot_loader_name  |    (present if flags[9] is set)
154///         +-------------------+
155/// 68      | apm_table         |    (present if flags[10] is set)
156///         +-------------------+
157/// 72      | vbe_control_info  |    (present if flags[11] is set)
158/// 76      | vbe_mode_info     |
159/// 80      | vbe_mode          |
160/// 82      | vbe_interface_seg |
161/// 84      | vbe_interface_off |
162/// 86      | vbe_interface_len |
163///         +-------------------+
164/// 88      | framebuffer_addr  |    (present if flags[12] is set)
165/// 96      | framebuffer_pitch |
166/// 100     | framebuffer_width |
167/// 104     | framebuffer_height|
168/// 108     | framebuffer_bpp   |
169/// 109     | framebuffer_type  |
170/// 110-115 | color_info        |
171///         +-------------------+
172///```
173///
174#[derive(Debug, Copy, Clone)]
175#[repr(C, packed)]
176struct MultibootLegacyInfo {
177    /// Indicate whether the below field exists.
178    flags: u32,
179
180    /// Physical memory low.
181    mem_lower: u32,
182    /// Physical memory high.
183    mem_upper: u32,
184
185    /// Indicates which BIOS disk device the boot loader loaded the OS image from.
186    boot_device: u32,
187
188    /// Command line passed to kernel.
189    cmdline: u32,
190
191    /// Modules count.
192    mods_count: u32,
193    /// The start address of modules list, each module structure format:
194    /// ```text
195    ///         +-------------------+
196    /// 0       | mod_start         |
197    /// 4       | mod_end           |
198    ///         +-------------------+
199    /// 8       | string            |
200    ///         +-------------------+
201    /// 12      | reserved (0)      |
202    ///         +-------------------+
203    /// ```
204    mods_addr: u32,
205
206    /// If flags\[4\] = 1, then the field starting at byte 28 are valid:
207    /// ```text
208    ///         +-------------------+
209    /// 28      | tabsize           |
210    /// 32      | strsize           |
211    /// 36      | addr              |
212    /// 40      | reserved (0)      |
213    ///         +-------------------+
214    /// ```
215    /// These indicate where the symbol table from kernel image can be found.
216    ///
217    /// If flags\[5\] = 1, then the field starting at byte 28 are valid:
218    /// ```text
219    ///         +-------------------+
220    /// 28      | num               |
221    /// 32      | size              |
222    /// 36      | addr              |
223    /// 40      | shndx             |
224    ///         +-------------------+
225    /// ```
226    /// These indicate where the section header table from an ELF kernel is,
227    /// the size of each entry, number of entries, and the string table used as the index of names.
228    symbols: [u8; 16],
229
230    memory_map_len: u32,
231    memory_map_addr: u32,
232
233    drives_length: u32,
234    drives_addr: u32,
235
236    config_table: u32,
237
238    boot_loader_name: u32,
239
240    apm_table: u32,
241
242    vbe_table: VbeTable,
243
244    framebuffer_table: FramebufferTable,
245}

247impl MultibootLegacyInfo {
248    fn get_memory_map(&self) -> MemoryEntryIter {
249        let cur_ptr = paddr_to_vaddr(self.memory_map_addr as Paddr) as *const u8;
250        // SAFETY: The range is in bounds by the contract with the Multiboot loader.
251        let region_end = unsafe { cur_ptr.add(self.memory_map_len as usize) };
252
253        MemoryEntryIter {
254            cur_ptr,
255            region_end,
256        }
257    }
258}

283/// A memory entry in the memory map header info region.
284///
285/// The memory layout of the entry structure doesn't fit in any scheme
286/// provided by Rust:
287///
288/// ```text
289///         +-------------------+   <- start of the struct pointer
290/// -4      | size              |
291///         +-------------------+
292/// 0       | base_addr         |
293/// 8       | length            |
294/// 16      | type              |
295///         +-------------------+
296/// ```
297///
298/// The start of a entry is not 64-bit aligned. Although the boot
299/// protocol may provide the `mmap_addr` 64-bit aligned when added with
300/// 4, it is not guaranteed. So we need to use pointer arithmetic to
301/// access the fields.
302struct MemoryEntry {
303    ptr: *const u8,
304}
305
306impl MemoryEntry {
307    fn size(&self) -> u32 {
308        // SAFETY: The entry can only be constructed from a valid address. The offset is in bounds.
309        unsafe { self.ptr.cast::<u32>().read_unaligned() }
310    }
311
312    fn base_addr(&self) -> u64 {
313        // SAFETY: The entry can only be constructed from a valid address. The offset is in bounds.
314        unsafe { self.ptr.byte_add(4).cast::<u64>().read_unaligned() }
315    }
316
317    fn length(&self) -> u64 {
318        // SAFETY: The entry can only be constructed from a valid address. The offset is in bounds.
319        unsafe { self.ptr.byte_add(12).cast::<u64>().read_unaligned() }
320    }
321
322    fn memory_type(&self) -> MemoryRegionType {
323        // The multiboot (v1) manual doesn't specify the length of the type field.
324        // Experimental result shows that "u8" works. So be it.
325        // SAFETY: The entry can only be constructed from a valid address. The offset is in bounds.
326        let typ_val = unsafe { self.ptr.add(20).cast::<u8>().read_unaligned() };
327
328        // The meaning of the values are however documented clearly by the manual.
329        match typ_val {
330            1 => MemoryRegionType::Usable,
331            2 => MemoryRegionType::Reserved,
332            3 => MemoryRegionType::Reclaimable,
333            4 => MemoryRegionType::NonVolatileSleep,
334            5 => MemoryRegionType::BadMemory,
335            _ => MemoryRegionType::Reserved,
336        }
337    }
338}

340/// A memory entry iterator in the memory map header info region.
341#[derive(Debug, Copy, Clone)]
342struct MemoryEntryIter {
343    cur_ptr: *const u8,
344    region_end: *const u8,
345}
346
347impl Iterator for MemoryEntryIter {
348    type Item = MemoryEntry;
349
350    fn next(&mut self) -> Option<Self::Item> {
351        if self.cur_ptr >= self.region_end {
352            return None;
353        }
354
355        let entry = MemoryEntry { ptr: self.cur_ptr };
356
357        let entry_size = entry.size() as usize + 4;
358        // SAFETY: The offset is in bounds according to the Multiboot specification.
359        unsafe { self.cur_ptr = self.cur_ptr.add(entry_size) };
360
361        Some(entry)
362    }
363}

365/// The entry point of the Rust code portion of Asterinas (with multiboot parameters).
366///
367/// # Safety
368///
369/// - This function must be called only once at a proper timing in the BSP's boot assembly code.
370/// - The caller must follow C calling conventions and put the right arguments in registers.
371/// - If this function is called, entry points of other boot protocols must never be called.
372// SAFETY: The name does not collide with other symbols.
373#[unsafe(no_mangle)]
374unsafe extern "sysv64" fn __multiboot_entry(boot_magic: u32, boot_params: u64) -> ! {
375    assert_eq!(boot_magic, MULTIBOOT_ENTRY_MAGIC);
376    let mb1_info =
377        unsafe { &*(paddr_to_vaddr(boot_params as usize) as *const MultibootLegacyInfo) };
378
379    use crate::boot::{EARLY_INFO, EarlyBootInfo, start_kernel};
380
381    EARLY_INFO.call_once(|| EarlyBootInfo {
382        bootloader_name: parse_bootloader_name(mb1_info).unwrap_or("Unknown Multiboot Loader"),
383        kernel_cmdline: parse_kernel_commandline(mb1_info).unwrap_or(""),
384        initramfs: parse_initramfs(mb1_info),
385        acpi_arg: parse_acpi_arg(mb1_info),
386        framebuffer_arg: parse_framebuffer_info(mb1_info),
387        memory_regions: parse_memory_regions(mb1_info),
388    });
389
390    // SAFETY: The safety is guaranteed by the safety preconditions and the fact that we call it
391    // once after setting up necessary resources.
392    unsafe { start_kernel() };
