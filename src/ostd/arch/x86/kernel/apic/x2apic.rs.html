3use x86::msr::{
4    IA32_APIC_BASE, IA32_X2APIC_APICID, IA32_X2APIC_CUR_COUNT, IA32_X2APIC_DIV_CONF,
5    IA32_X2APIC_EOI, IA32_X2APIC_ESR, IA32_X2APIC_ICR, IA32_X2APIC_INIT_COUNT,
6    IA32_X2APIC_LVT_TIMER, IA32_X2APIC_SIVR, IA32_X2APIC_VERSION, rdmsr, wrmsr,
7};
8
9use super::ApicTimer;
10
11#[derive(Debug)]
12pub(super) struct X2Apic {
13    _private: (),
14}
15
16// The APIC instance can be shared among threads running on the same CPU, but not among those
17// running on different CPUs. Therefore, it is not `Send`/`Sync`.
18impl !Send for X2Apic {}
19impl !Sync for X2Apic {}
20
21impl X2Apic {
22    pub(super) fn new() -> Option<Self> {
23        if !Self::has_x2apic() {
24            return None;
25        }
26
27        Some(Self { _private: () })
28    }
29
30    pub(super) fn has_x2apic() -> bool {
31        use crate::arch::cpu::extension::{IsaExtensions, has_extensions};
32
33        has_extensions(IsaExtensions::X2APIC)
34    }
35
36    pub(super) fn enable(&mut self) {
37        const X2APIC_ENABLE_BITS: u64 = {
38            // IA32_APIC_BASE MSR's EN bit: xAPIC global enable/disable
39            const EN_BIT_IDX: u8 = 11;
40            // IA32_APIC_BASE MSR's EXTD bit: Enable x2APIC mode
41            const EXTD_BIT_IDX: u8 = 10;
42
43            (1 << EN_BIT_IDX) | (1 << EXTD_BIT_IDX)
44        };
45
46        // SAFETY: These operations enable x2APIC, which is safe because `X2Apic` will only be
47        // constructed if x2APIC is known to be present.
48        unsafe {
49            // Enable x2APIC and xAPIC if they are not enabled by default.
50            let mut base = rdmsr(IA32_APIC_BASE);
51            if base & X2APIC_ENABLE_BITS != X2APIC_ENABLE_BITS {
52                base |= X2APIC_ENABLE_BITS;
53                wrmsr(IA32_APIC_BASE, base);
54            }
55
56            // Set SVR. Enable APIC and set Spurious Vector to 15 (reserved IRQ number).
57            let svr: u64 = (1 << 8) | 15;
58            wrmsr(IA32_X2APIC_SIVR, svr);
59        }
60    }
61}
62
63impl super::Apic for X2Apic {
64    fn id(&self) -> u32 {
65        unsafe { rdmsr(IA32_X2APIC_APICID) as u32 }
66    }
67
68    fn version(&self) -> u32 {
69        unsafe { rdmsr(IA32_X2APIC_VERSION) as u32 }
70    }
71
72    fn eoi(&self) {
73        unsafe { wrmsr(IA32_X2APIC_EOI, 0) };
74    }
75
76    unsafe fn send_ipi(&self, icr: super::Icr) {
77        let _guard = crate::irq::disable_local();
78
79        // SAFETY: These operations write the interrupt command to APIC and wait for results. The
80        // caller guarantees it's safe to execute this interrupt command.
81        unsafe {
82            wrmsr(IA32_X2APIC_ESR, 0);
83            wrmsr(IA32_X2APIC_ICR, icr.0);
84            loop {
85                let icr = rdmsr(IA32_X2APIC_ICR);
86                if ((icr >> 12) & 0x1) == 0 {
87                    break;
88                }
89                if rdmsr(IA32_X2APIC_ESR) > 0 {
90                    break;
91                }
92            }
93        }
94    }
95}
