14#[derive(Debug)]
15pub(super) struct XApic {
16    io_mem: IoMem<Sensitive>,
17}
18
19// The APIC instance can be shared among threads running on the same CPU, but not among those
20// running on different CPUs. Therefore, it is not `Send`/`Sync`.
21impl !Send for XApic {}
22impl !Sync for XApic {}
23
24impl XApic {
25    pub(super) fn new(io_mem: &IoMem<Sensitive>) -> Option<Self> {
26        if !Self::has_xapic() {
27            return None;
28        }
29
30        // SAFETY: xAPIC is present.
31        assert_eq!(io_mem.paddr(), unsafe { read_xapic_base_address() });
32        assert_eq!(io_mem.size(), XAPIC_MMIO_SIZE);
33        Some(Self {
34            io_mem: io_mem.clone(),
35        })
36    }
37
38    pub(super) fn has_xapic() -> bool {
39        use crate::arch::cpu::extension::{IsaExtensions, has_extensions};
40
41        has_extensions(IsaExtensions::XAPIC)
42    }
43
44    pub(super) fn enable(&mut self) {
45        const XAPIC_ENABLE_BITS: u64 = {
46            // IA32_APIC_BASE MSR's EN bit: xAPIC global enable/disable
47            const EN_BIT_IDX: u8 = 11;
48
49            1 << EN_BIT_IDX
50        };
51
52        // SAFETY: These operations enable xAPIC, which is safe because `XApic` will only be
53        // constructed if xAPIC is known to be present.
54        unsafe {
55            // Enable xAPIC if it is not enabled by default.
56            wrmsr(
57                IA32_APIC_BASE,
58                self.io_mem.paddr() as u64 | XAPIC_ENABLE_BITS,
59            );
60
61            // Set SVR. Enable APIC and set Spurious Vector to 15 (reserved IRQ number).
62            let svr: u32 = (1 << 8) | 15;
63            self.io_mem.write_once(xapic::XAPIC_SVR as usize, &svr);
64        }
65    }
66}
67
68impl super::Apic for XApic {
69    fn id(&self) -> u32 {
70        unsafe { self.io_mem.read_once(xapic::XAPIC_ID as usize) }
71    }
72
73    fn version(&self) -> u32 {
74        unsafe { self.io_mem.read_once(xapic::XAPIC_VERSION as usize) }
75    }
76
77    fn eoi(&self) {
78        unsafe { self.io_mem.write_once(xapic::XAPIC_EOI as usize, &0u32) };
79    }
80
81    unsafe fn send_ipi(&self, icr: super::Icr) {
82        let _guard = crate::irq::disable_local();
83
84        // SAFETY: These operations write the interrupt command to APIC and wait for results. The
85        // caller guarantees it's safe to execute this interrupt command.
86        unsafe {
87            self.io_mem.write_once(xapic::XAPIC_ESR as usize, &0u32);
88            // The upper 32 bits of ICR must be written into XAPIC_ICR1 first,
89            // because writing into XAPIC_ICR0 will trigger the action of
90            // interrupt sending.
91            self.io_mem
92                .write_once(xapic::XAPIC_ICR1 as usize, &icr.upper());
93            self.io_mem
94                .write_once(xapic::XAPIC_ICR0 as usize, &icr.lower());
95            loop {
96                let icr = self.io_mem.read_once::<u32>(xapic::XAPIC_ICR0 as usize);
97                if ((icr >> 12) & 0x1) == 0 {
98                    break;
99                }
100                if self.io_mem.read_once::<u32>(xapic::XAPIC_ESR as usize) > 0 {
101                    break;
102                }
103            }
104        }
105    }
106}

138/// Reads xAPIC base address from the MSR.
139///
140/// # Safety
141///
142/// The caller must ensure that xAPIC is present.
143pub(super) unsafe fn read_xapic_base_address() -> usize {
144    (unsafe { rdmsr(IA32_APIC_BASE) & 0xffff_f000 }) as usize
145}
146
147/// The size of the xAPIC MMIO region.
148///
149/// Intel(R) 64 and IA-32 Architectures Software Developer's Manual, CHAPTER 11 ADVANCED
150/// PROGRAMMABLE INTERRUPT CONTROLLER (APIC) says "The local-APIC register-address space comprises
151/// the 4 KBytes at the physical address specified in the IA32_APIC_BASE MSR."
152pub(super) const XAPIC_MMIO_SIZE: usize = crate::mm::PAGE_SIZE;
