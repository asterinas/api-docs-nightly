42bitflags::bitflags! {
43    #[derive(Pod)]
44    #[repr(C)]
45    /// Possible flags for a page table entry.
46    pub(crate) struct PteFlags: usize {
47        /// Specifies whether the mapped frame or page table is loaded in memory.
48        const PRESENT =         1 << 0;
49        /// Controls whether writes to the mapped frames are allowed.
50        const WRITABLE =        1 << 1;
51        /// Controls whether accesses from userspace (i.e. ring 3) are permitted.
52        const USER =            1 << 2;
53        /// If this bit is set, a “write-through” policy is used for the cache, else a “write-back”
54        /// policy is used.
55        const WRITE_THROUGH =   1 << 3;
56        /// Disables caching for the pointed entry is cacheable.
57        const NO_CACHE =        1 << 4;
58        /// Whether this entry has been used for linear-address translation.
59        const ACCESSED =        1 << 5;
60        /// Whether the memory area represented by this entry is modified.
61        const DIRTY =           1 << 6;
62        /// In level 2 or 3 it indicates that it map to a huge page.
63        /// In level 1, it is the PAT (page attribute table) bit.
64        /// We use this bit in level 1, 2 and 3 to indicate that this entry is
65        /// "valid". For levels above 3, `PRESENT` is used for "valid".
66        const HUGE =            1 << 7;
67        /// Indicates that the mapping is present in all address spaces, so it isn't flushed from
68        /// the TLB on an address space switch.
69        const GLOBAL =          1 << 8;
70        /// TDX shared bit.
71        #[cfg(feature = "cvm_guest")]
72        const SHARED =          1 << 51;
73
74        /// Ignored by the hardware. Free to use.
75        const HIGH_IGN1 =       1 << 52;
76        /// Ignored by the hardware. Free to use.
77        const HIGH_IGN2 =       1 << 53;
78
79        /// Forbid execute codes on the page. The NXE bits in EFER msr must be set.
80        const NO_EXECUTE =      1 << 63;
81    }
82}

84/// Flush any TLB entry that contains the map of the given virtual address.
85///
86/// This flush performs regardless of the global-page bit. So it can flush both global
87/// and non-global entries.
88pub(crate) fn tlb_flush_addr(vaddr: Vaddr) {
89    tlb::flush(VirtAddr::new(vaddr as u64));
90}
91
92/// Flush any TLB entry that intersects with the given address range.
93pub(crate) fn tlb_flush_addr_range(range: &Range<Vaddr>) {
94    for vaddr in range.clone().step_by(PAGE_SIZE) {
95        tlb_flush_addr(vaddr);
96    }
97}
98
99/// Flush all TLB entries except for the global-page entries.
100pub(crate) fn tlb_flush_all_excluding_global() {
101    tlb::flush_all();
102}
103
104/// Flush all TLB entries, including global-page entries.
105pub(crate) fn tlb_flush_all_including_global() {
106    // SAFETY: updates to CR4 here only change the global-page bit, the side effect
107    // is only to invalidate the TLB, which doesn't affect the memory safety.
108    unsafe {
109        // To invalidate all entries, including global-page
110        // entries, disable global-page extensions (CR4.PGE=0).
111        x86_64::registers::control::Cr4::update(|cr4| {
112            *cr4 -= x86_64::registers::control::Cr4Flags::PAGE_GLOBAL;
113        });
114        x86_64::registers::control::Cr4::update(|cr4| {
115            *cr4 |= x86_64::registers::control::Cr4Flags::PAGE_GLOBAL;
116        });
117    }
118}

136/// Activates the given root-level page table.
137///
138/// The cache policy of the root page table node is controlled by `root_pt_cache`.
139///
140/// # Safety
141///
142/// Changing the root-level page table is unsafe, because it's possible to violate memory safety by
143/// changing the page mapping.
144pub(crate) unsafe fn activate_page_table(root_paddr: Paddr, root_pt_cache: CachePolicy) {
145    let addr = PhysFrame::from_start_address(x86_64::PhysAddr::new(root_paddr as u64)).unwrap();
146    let flags = match root_pt_cache {
147        CachePolicy::Writeback => x86_64::registers::control::Cr3Flags::empty(),
148        CachePolicy::Writethrough => x86_64::registers::control::Cr3Flags::PAGE_LEVEL_WRITETHROUGH,
149        CachePolicy::Uncacheable => x86_64::registers::control::Cr3Flags::PAGE_LEVEL_CACHE_DISABLE,
150        // Write-combining and write-protected are not supported for root page table (CR3)
151        // as CR3 only supports WB, WT, and UC via PCD/PWT bits
152        _ => {
153            panic!(
154                "unsupported cache policy for the root page table (only WB, WT, and UC are allowed)"
155            )
156        }
157    };
158
159    // SAFETY: The safety is upheld by the caller.
160    unsafe { x86_64::registers::control::Cr3::write(addr, flags) };
161}

170#[derive(Debug, Clone, Copy, Pod)]
171#[repr(C)]
172pub(crate) struct PageTableEntry(usize);
173
174/// Parses a bit-flag bits `val` in the representation of `from` to `to` in bits.
175macro_rules! parse_flags {
176    ($val:expr, $from:expr, $to:expr) => {
177        (($val as usize & $from.bits() as usize) >> $from.bits().ilog2() << $to.bits().ilog2())
178    };
179}
180
181impl PageTableEntry {
182    cfg_if! {
183        if #[cfg(feature = "cvm_guest")] {
184            const PHYS_ADDR_MASK_LVL1: usize = 0x7_ffff_ffff_f000;
185            const PHYS_ADDR_MASK_LVL2: usize = 0x7_ffff_ffe0_0000;
186            const PHYS_ADDR_MASK_LVL3: usize = 0x7_ffff_c000_0000;
187        } else {
188            const PHYS_ADDR_MASK_LVL1: usize = 0xf_ffff_ffff_f000;
189            const PHYS_ADDR_MASK_LVL2: usize = 0xf_ffff_ffe0_0000;
190            const PHYS_ADDR_MASK_LVL3: usize = 0xf_ffff_c000_0000;
191        }
192    }
193
194    const CHILD_PT_ADDR_MASK: usize = Self::PHYS_ADDR_MASK_LVL1;
195
196    fn pa_mask_at_level(level: PagingLevel) -> usize {
197        match level {
198            1 => Self::PHYS_ADDR_MASK_LVL1,
199            2 => Self::PHYS_ADDR_MASK_LVL2,
200            3 => Self::PHYS_ADDR_MASK_LVL3,
201            _ => panic!("invalid level {} for page entry", level),
202        }
203    }
204
205    fn is_present(&self) -> bool {
206        // For PT child, `PRESENT` should be set; for huge page, `HUGE` should
207        // be set; for the leaf child page, `PAT`, which is the same bit as
208        // the `HUGE` bit in upper levels, should be set.
209        self.0 & PteFlags::PRESENT.bits() != 0 || self.0 & PteFlags::HUGE.bits() != 0
210    }
211
212    fn prop(&self) -> PageProperty {
213        let flags = parse_flags!(self.0, PteFlags::PRESENT, PageFlags::R)
214            | parse_flags!(self.0, PteFlags::WRITABLE, PageFlags::W)
215            | parse_flags!(!self.0, PteFlags::NO_EXECUTE, PageFlags::X)
216            | parse_flags!(self.0, PteFlags::ACCESSED, PageFlags::ACCESSED)
217            | parse_flags!(self.0, PteFlags::DIRTY, PageFlags::DIRTY)
218            | parse_flags!(self.0, PteFlags::HIGH_IGN2, PageFlags::AVAIL2);
219
220        let priv_flags = parse_flags!(self.0, PteFlags::USER, PrivFlags::USER)
221            | parse_flags!(self.0, PteFlags::GLOBAL, PrivFlags::GLOBAL)
222            | parse_flags!(self.0, PteFlags::HIGH_IGN1, PrivFlags::AVAIL1);
223
224        #[cfg(feature = "cvm_guest")]
225        let priv_flags = priv_flags | parse_flags!(self.0, PteFlags::SHARED, PrivFlags::SHARED);
226
227        // Determine cache policy from PCD, PWT bits.
228        let cache = flags_to_cache_policy(PteFlags::from_bits_truncate(self.0));
229
230        PageProperty {
231            flags: PageFlags::from_bits(flags as u8).unwrap(),
232            cache,
233            priv_flags: PrivFlags::from_bits(priv_flags as u8).unwrap(),
234        }
235    }
236
237    fn pt_flags(&self) -> PageTableFlags {
238        let bits = PageTableFlags::empty().bits() as usize
239            | parse_flags!(self.0, PteFlags::HIGH_IGN1, PageTableFlags::AVAIL1)
240            | parse_flags!(self.0, PteFlags::HIGH_IGN2, PageTableFlags::AVAIL2);
241        PageTableFlags::from_bits(bits as u8).unwrap()
242    }
243
244    fn new_page(paddr: Paddr, level: PagingLevel, prop: PageProperty) -> Self {
245        let mut flags = PteFlags::HUGE.bits();
246
247        flags |= parse_flags!(prop.flags.bits(), PageFlags::R, PteFlags::PRESENT)
248            | parse_flags!(prop.flags.bits(), PageFlags::W, PteFlags::WRITABLE)
249            | parse_flags!(!prop.flags.bits(), PageFlags::X, PteFlags::NO_EXECUTE)
250            | parse_flags!(prop.flags.bits(), PageFlags::ACCESSED, PteFlags::ACCESSED)
251            | parse_flags!(prop.flags.bits(), PageFlags::DIRTY, PteFlags::DIRTY)
252            | parse_flags!(
253                prop.priv_flags.bits(),
254                PrivFlags::AVAIL1,
255                PteFlags::HIGH_IGN1
256            )
257            | parse_flags!(prop.flags.bits(), PageFlags::AVAIL2, PteFlags::HIGH_IGN2)
258            | parse_flags!(prop.priv_flags.bits(), PrivFlags::USER, PteFlags::USER)
259            | parse_flags!(prop.priv_flags.bits(), PrivFlags::GLOBAL, PteFlags::GLOBAL);
260        #[cfg(feature = "cvm_guest")]
261        {
262            flags |= parse_flags!(prop.priv_flags.bits(), PrivFlags::SHARED, PteFlags::SHARED);
263        }
264
265        flags |= cache_policy_to_flags(prop.cache).bits();
266
267        assert_eq!(
268            paddr & !Self::pa_mask_at_level(level),
269            0,
270            "page physical address contains invalid bits"
271        );
272        Self(paddr | flags)
273    }
274
275    fn new_pt(paddr: Paddr, flags: PageTableFlags) -> Self {
276        // In x86 if it's an intermediate PTE, it's better to have the same permissions
277        // as the most permissive child (to reduce hardware page walk accesses). But we
278        // don't have a mechanism to keep it generic across architectures, thus just
279        // setting it to be the most permissive.
280        let flags = PteFlags::PRESENT.bits()
281            | PteFlags::WRITABLE.bits()
282            | PteFlags::USER.bits()
283            | parse_flags!(flags.bits(), PageTableFlags::AVAIL1, PteFlags::HIGH_IGN1)
284            | parse_flags!(flags.bits(), PageTableFlags::AVAIL2, PteFlags::HIGH_IGN2);
285
286        assert_eq!(
287            paddr & !Self::CHILD_PT_ADDR_MASK,
288            0,
289            "page table physical address contains invalid bits"
290        );
291        Self(paddr | flags)
292    }
293}

295impl PodOnce for PageTableEntry {}

297/// SAFETY: The implementation is safe because:
298///  - `from_usize` and `into_usize` are not overridden;
299///  - `from_repr` and `repr` are correctly implemented;
300///  - a zeroed PTE represents an absent entry.
301unsafe impl PteTrait for PageTableEntry {
302    fn from_repr(repr: &PteScalar, level: PagingLevel) -> Self {
303        match repr {
304            PteScalar::Absent => PageTableEntry(0),
305            PteScalar::PageTable(paddr, flags) => Self::new_pt(*paddr, *flags),
306            PteScalar::Mapped(paddr, prop) => Self::new_page(*paddr, level, *prop),
307        }
308    }
309
310    fn to_repr(&self, level: PagingLevel) -> PteScalar {
311        if !self.is_present() {
312            return PteScalar::Absent;
313        }
314
315        if self.0 & PteFlags::HUGE.bits() != 0 {
316            let paddr = self.0 & Self::pa_mask_at_level(level);
317            PteScalar::Mapped(paddr, self.prop())
318        } else {
319            let paddr = self.0 & Self::CHILD_PT_ADDR_MASK;
320            PteScalar::PageTable(paddr, self.pt_flags())
321        }
322    }
323}

325/// Enables memory-management essential features for the x86 MMU.
326pub(super) fn enable_essential_features() {
327    // Page Attribute Table (PAT) has been available since Pentium III (1999)
328    // and is ubiquitous in modern 64-bit CPUs. Therefore, we assume that all
329    // x86-64 CPUs should have PAT support. Otherwise, we should check
330    // `cpu::extension::has_extensions(IsaExtensions::PAT)` before programming it.
331    configure_pat();
332
333    unsafe {
334        // Enable non-executable page protection.
335        Efer::update(|efer| {
336            *efer |= EferFlags::NO_EXECUTE_ENABLE;
337        });
338    }
