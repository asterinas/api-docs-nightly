3//! Power management.
4
5mod qemu_isa_debug {
6    //! The isa-debug-exit device in QEMU.
7    //!
8    //! Reference: <https://elixir.bootlin.com/qemu/v10.1.2/source/hw/misc/debugexit.c>
9
10    use spin::Once;
11
12    use crate::{
13        arch::device::io_port::WriteOnlyAccess,
14        io::IoPort,
15        power::{ExitCode, inject_poweroff_handler},
16    };
17
18    // For `qemu-system-x86_64`, the exit code will be `(code << 1) | 1`. So it is not possible to
19    // let QEMU invoke `exit(0)`. We also need to check if the exit code is returned by the kernel,
20    // so we cannot use `0` as `EXIT_SUCCESS` because it may conflict with QEMU's return value `1`,
21    // which indicates that QEMU itself fails.
22    const EXIT_SUCCESS: u32 = 0x10;
23    const EXIT_FAILURE: u32 = 0x20;
24
25    static DEBUG_EXIT_PORT: Once<IoPort<u32, WriteOnlyAccess>> = Once::new();
26
27    fn try_exit_qemu(code: ExitCode) {
28        let value = match code {
29            ExitCode::Success => EXIT_SUCCESS,
30            ExitCode::Failure => EXIT_FAILURE,
31        };
32
33        // If possible, keep this method panic-free because it may be called by the panic handler.
34        if let Some(port) = DEBUG_EXIT_PORT.get() {
35            port.write(value);
36        }
37    }
38
39    pub(super) fn init() {
40        const DEBUG_EXIT_PORT_NUM: u16 = 0xF4;
41
42        let debug_exit_port = IoPort::acquire(DEBUG_EXIT_PORT_NUM).unwrap();
43
44        DEBUG_EXIT_PORT.call_once(|| debug_exit_port);
45        inject_poweroff_handler(try_exit_qemu);
46    }
47}
48
49pub(super) fn init() {
50    use super::cpu::cpuid;
51
52    if !cpuid::query_is_running_in_qemu() {
53        return;
54    }
55
56    // FIXME: We assume that the kernel is running in QEMU with the following QEMU command line
57    // arguments that specify the isa-debug-exit device:
58    // `-device isa-debug-exit,iobase=0xf4,iosize=0x04`.
59    log::info!("QEMU hypervisor detected, assuming that the isa-debug-exit device exists");
60
61    qemu_isa_debug::init();
62}
