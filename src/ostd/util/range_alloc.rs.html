3use alloc::collections::btree_map::BTreeMap;
4use core::ops::Range;
5
6use crate::sync::{PreemptDisabled, SpinLock, SpinLockGuard};
7
8pub struct RangeAllocator {
9    fullrange: Range<usize>,
10    freelist: SpinLock<Option<BTreeMap<usize, FreeRange>>>,
11}
12
13/// An error returned when allocating from a [`RangeAllocator`].
14#[derive(Debug)]
15pub struct RangeAllocError;
16
17impl RangeAllocator {
18    pub const fn new(fullrange: Range<usize>) -> Self {
19        Self {
20            fullrange,
21            freelist: SpinLock::new(None),
22        }
23    }
24
25    pub const fn fullrange(&self) -> &Range<usize> {
26        &self.fullrange
27    }
28
29    /// Allocates a specific kernel virtual area.
30    pub fn alloc_specific(&self, allocate_range: &Range<usize>) -> Result<(), RangeAllocError> {
31        if allocate_range.is_empty() {
32            return Err(RangeAllocError);
33        }
34
35        let mut lock_guard = self.get_freelist_guard();
36        let freelist = lock_guard.as_mut().unwrap();
37        let mut target_node = None;
38        let mut left_length = 0;
39        let mut right_length = 0;
40
41        for (key, value) in freelist.iter() {
42            if value.block.end >= allocate_range.end && value.block.start <= allocate_range.start {
43                target_node = Some(*key);
44                left_length = allocate_range.start - value.block.start;
45                right_length = value.block.end - allocate_range.end;
46                break;
47            }
48        }
49
50        if let Some(key) = target_node {
51            if left_length == 0 {
52                freelist.remove(&key);
53            } else if let Some(freenode) = freelist.get_mut(&key) {
54                freenode.block.end = allocate_range.start;
55            }
56
57            if right_length != 0 {
58                freelist.insert(
59                    allocate_range.end,
60                    FreeRange::new(allocate_range.end..(allocate_range.end + right_length)),
61                );
62            }
63        }
64
65        if target_node.is_some() {
66            Ok(())
67        } else {
68            Err(RangeAllocError)
69        }
70    }
71
72    /// Allocates a range specific by the `size`.
73    ///
74    /// This is currently implemented with a simple FIRST-FIT algorithm.
75    pub fn alloc(&self, size: usize) -> Result<Range<usize>, RangeAllocError> {
76        let mut lock_guard = self.get_freelist_guard();
77        let freelist = lock_guard.as_mut().unwrap();
78        let mut allocate_range = None;
79        let mut to_remove = None;
80
81        for (key, value) in freelist.iter() {
82            if value.block.end - value.block.start >= size {
83                allocate_range = Some((value.block.end - size)..value.block.end);
84                to_remove = Some(*key);
85                break;
86            }
87        }
88
89        if let Some(key) = to_remove
90            && let Some(freenode) = freelist.get_mut(&key)
91        {
92            if freenode.block.end - size == freenode.block.start {
93                freelist.remove(&key);
94            } else {
95                freenode.block.end -= size;
96            }
97        }
98
99        if let Some(range) = allocate_range {
100            Ok(range)
101        } else {
102            Err(RangeAllocError)
103        }
104    }
105
106    /// Frees a `range`.
107    pub fn free(&self, range: Range<usize>) {
108        let mut lock_guard = self.freelist.lock();
109        let freelist = lock_guard.as_mut().unwrap_or_else(|| {
110            panic!("Free a 'KVirtArea' when 'VirtAddrAllocator' has not been initialized.")
111        });
112        // 1. get the previous free block, check if we can merge this block with the free one
113        //     - if contiguous, merge this area with the free block.
114        //     - if not contiguous, create a new free block, insert it into the list.
115        let mut free_range = range.clone();
116
117        if let Some((prev_va, prev_node)) = freelist
118            .upper_bound_mut(core::ops::Bound::Excluded(&free_range.start))
119            .peek_prev()
120            && prev_node.block.end == free_range.start
121        {
122            let prev_va = *prev_va;
123            free_range.start = prev_node.block.start;
124            freelist.remove(&prev_va);
125        }
126
127        freelist.insert(free_range.start, FreeRange::new(free_range.clone()));
128
129        // 2. check if we can merge the current block with the next block, if we can, do so.
130        if let Some((next_va, next_node)) = freelist
131            .lower_bound_mut(core::ops::Bound::Excluded(&free_range.start))
132            .peek_next()
133            && free_range.end == next_node.block.start
134        {
135            let next_va = *next_va;
136            free_range.end = next_node.block.end;
137            freelist.remove(&next_va);
138            freelist.get_mut(&free_range.start).unwrap().block.end = free_range.end;
139        }
140    }
141
142    fn get_freelist_guard(
143        &self,
144    ) -> SpinLockGuard<'_, Option<BTreeMap<usize, FreeRange>>, PreemptDisabled> {
145        let mut lock_guard = self.freelist.lock();
146        if lock_guard.is_none() {
147            let mut freelist: BTreeMap<usize, FreeRange> = BTreeMap::new();
148            freelist.insert(self.fullrange.start, FreeRange::new(self.fullrange.clone()));
149            *lock_guard = Some(freelist);
150        }
151        lock_guard
152    }
153}
