34/// The accessor to the boot page table singleton [`BootPageTable`].
35///
36/// The user should provide a closure to access the boot page table. The
37/// function will acquire the lock and call the closure with a mutable
38/// reference to the boot page table as the argument.
39///
40/// The boot page table will be dropped when all CPUs have dismissed it.
41/// This function will return an [`Err`] if the boot page table is dropped.
42pub(crate) fn with_borrow<F, R>(f: F) -> Result<R, ()>
43where
44    F: FnOnce(&mut BootPageTable) -> R,
45{
46    let mut boot_pt = BOOT_PAGE_TABLE.lock();
47
48    if IS_DISMISSED.load() {
49        return Err(());
50    }
51
52    // Lazy initialization.
53    if boot_pt.is_none() {
54        // SAFETY: This function is called only once.
55        *boot_pt = Some(unsafe { BootPageTable::from_current_pt() });
56    }
57
58    let r = f(boot_pt.as_mut().unwrap());
59
60    Ok(r)
61}

63/// Dismiss the boot page table.
64///
65/// By calling it on a CPU, the caller claims that the boot page table is no
66/// longer needed on this CPU.
67///
68/// # Safety
69///
70/// The caller should ensure that:
71///  - another legitimate page table is activated on this CPU;
72///  - this function should be called only once per CPU;
73///  - no [`with_borrow`] calls are performed on this CPU after this dismissal;
74pub(crate) unsafe fn dismiss() {
75    IS_DISMISSED.store(true);
76    if DISMISS_COUNT.fetch_add(1, Ordering::SeqCst) as usize == num_cpus() - 1 {
77        let boot_pt = BOOT_PAGE_TABLE.lock().take().unwrap();
78
79        dfs_walk_on_leave::<PageTableEntry, PagingConsts>(
80            boot_pt.root_pt,
81            PagingConsts::NR_LEVELS,
82            &mut |pte, pa, _, flags| {
83                if !flags.contains(PTE_POINTS_TO_FIRMWARE_PT) {
84                    // SAFETY: The pointed frame is allocated and forgotten with `into_raw`.
85                    drop(unsafe { Frame::<EarlyAllocatedFrameMeta>::from_raw(pa) })
86                }
87                // Firmware provided page tables may be a DAG instead of a tree.
88                // Clear it to avoid double-free when we meet it the second time.
89                *pte = PageTableEntry::new_zeroed();
90            },
91        );
92    }
93}

95/// The boot page table singleton instance.
96static BOOT_PAGE_TABLE: SpinLock<Option<BootPageTable>> = SpinLock::new(None);
97/// If it reaches the number of CPUs, the boot page table will be dropped.
98static DISMISS_COUNT: AtomicU32 = AtomicU32::new(0);
99cpu_local_cell! {
100    /// If the boot page table is dismissed on this CPU.
101    static IS_DISMISSED: bool = false;
102}

104/// A simple boot page table singleton for boot stage mapping management.
105///
106/// If applicable, the boot page table could track the lifetime of page table
107/// frames that are set up by the firmware, loader or the setup code.
108///
109/// All the newly allocated page table frames have the first unused bit in
110/// parent PTEs. This allows us to deallocate them when the boot page table
111/// is dropped.
112pub(crate) struct BootPageTable<E: PteTrait = PageTableEntry, C: PagingConstsTrait = PagingConsts> {
113    root_pt: Paddr,
114    _phantom: core::marker::PhantomData<(E, C)>,
115}
116
117// We use extra two available bits in the boot PT for memory management.
118//
119// The first available bit is used to differentiate firmware page tables from
120// the page tables allocated here. The second is for identifying double-visits
121// when walking the page tables since the PT can be a DAG.
122const PTE_POINTS_TO_FIRMWARE_PT: PageTableFlags = PageTableFlags::AVAIL1;
123
124impl<E: PteTrait, C: PagingConstsTrait> BootPageTable<E, C> {
125    /// Creates a new boot page table from the current page table root
126    /// physical address.
127    ///
128    /// # Safety
129    ///
130    /// This function should be called only once in the initialization phase.
131    /// Otherwise, It would lead to double-drop of the page table frames set up
132    /// by the firmware, loader or the setup code.
133    unsafe fn from_current_pt() -> Self {
134        let root_pt = crate::arch::mm::current_page_table_paddr();
135        // Make sure the 2 available bits are not set for firmware page tables.
136        dfs_walk_on_leave::<E, C>(
137            root_pt,
138            C::NR_LEVELS,
139            &mut |pte: &mut E, pa, level, mut flags| {
140                flags |= PTE_POINTS_TO_FIRMWARE_PT;
141                *pte = E::from_repr(&PteScalar::PageTable(pa, flags), level);
142            },
143        );
144        Self {
145            root_pt,
146            _phantom: core::marker::PhantomData,
147        }
148    }
149
150    /// Returns the root physical address of the boot page table.
151    pub(crate) fn root_address(&self) -> Paddr {
152        self.root_pt
153    }
154
155    /// Maps a base page to a frame.
156    ///
157    /// # Panics
158    ///
159    /// This function will panic if the page is already mapped.
160    ///
161    /// # Safety
162    ///
163    /// This function is unsafe because it can cause undefined behavior if the caller
164    /// maps a page in the kernel address space.
165    pub unsafe fn map_base_page(&mut self, from: Vaddr, to: Paddr, prop: PageProperty) {
166        let mut pt = self.root_pt;
167        let mut level = C::NR_LEVELS;
168        // Walk to the last level of the page table.
169        while level > 1 {
170            let index = pte_index::<C>(from, level);
171            // SAFETY: The result pointer is within the PT frame.
172            let pte_ptr = unsafe { (paddr_to_vaddr(pt) as *mut E).add(index) };
173            // SAFETY: The pointer to the entry is valid to read.
174            let pte = unsafe { pte_ptr.read() };
175            match pte.to_repr(level) {
176                PteScalar::Absent => {
177                    let (pte, child_pt) = self.alloc_child(level);
178                    // SAFETY: The pointer to the entry is valid to write.
179                    unsafe { pte_ptr.write(pte) };
180                    pt = child_pt;
181                }
182                PteScalar::Mapped(_, _) => {
183                    panic!("mapping an already mapped huge page in the boot page table");
184                }
185                PteScalar::PageTable(child_pt, _) => {
186                    pt = child_pt;
187                }
188            };
189            level -= 1;
190        }
191        // Map the page in the last level page table.
192        let index = pte_index::<C>(from, 1);
193        // SAFETY: The result pointer is within the PT frame.
194        let pte_ptr = unsafe { (paddr_to_vaddr(pt) as *mut E).add(index) };
195        // SAFETY: The pointer to the entry is valid to read.
196        let pte = unsafe { pte_ptr.read() };
197        if matches!(pte.to_repr(1), PteScalar::Mapped(_, _)) {
198            panic!("mapping an already mapped page in the boot page table");
199        }
200        // SAFETY: The pointer to the entry is valid to write.
201        unsafe { pte_ptr.write(E::from_repr(&PteScalar::Mapped(to, prop), 1)) };
202    }
203
204    /// Set protections of a base page mapping.
205    ///
206    /// This function may split a huge page into base pages, causing page allocations
207    /// if the original mapping is a huge page.
208    ///
209    /// # Panics
210    ///
211    /// This function will panic if the page is already mapped.
212    ///
213    /// # Safety
214    ///
215    /// This function is unsafe because it can cause undefined behavior if the caller
216    /// maps a page in the kernel address space.
217    #[cfg_attr(not(ktest), expect(dead_code))]
218    pub unsafe fn protect_base_page(
219        &mut self,
220        virt_addr: Vaddr,
221        mut op: impl FnMut(&mut PageProperty),
222    ) {
223        let mut pt = self.root_pt;
224        let mut level = C::NR_LEVELS;
225        // Walk to the last level of the page table.
226        while level > 1 {
227            let index = pte_index::<C>(virt_addr, level);
228            // SAFETY: The result pointer is within the PT frame.
229            let pte_ptr = unsafe { (paddr_to_vaddr(pt) as *mut E).add(index) };
230            // SAFETY: The pointer to the entry is valid to read.
231            let pte = unsafe { pte_ptr.read() };
232            match pte.to_repr(level) {
233                PteScalar::Absent => {
234                    panic!("protecting an unmapped page in the boot page table");
235                }
236                PteScalar::PageTable(child_pt, _) => {
237                    pt = child_pt;
238                }
239                PteScalar::Mapped(huge_pa, prop) => {
240                    // Split the huge page.
241                    let (child_pte, child_frame_pa) = self.alloc_child(level);
242                    for i in 0..nr_subpage_per_huge::<C>() {
243                        // SAFETY: The result pointer is within the new PT frame.
244                        let nxt_ptr = unsafe { (paddr_to_vaddr(child_frame_pa) as *mut E).add(i) };
245                        // SAFETY: The pointer to the entry is valid to write.
246                        unsafe {
247                            nxt_ptr.write(E::from_repr(
248                                &PteScalar::Mapped(huge_pa + i * C::BASE_PAGE_SIZE, prop),
249                                level - 1,
250                            ))
251                        };
252                    }
253                    // SAFETY: The pointer to the entry is valid to write.
254                    unsafe { pte_ptr.write(child_pte) };
255                    pt = child_frame_pa;
256                }
257            };
258            level -= 1;
259        }
260        // Do protection in the last level page table.
261        let index = pte_index::<C>(virt_addr, 1);
262        // SAFETY: The result pointer is within the PT frame.
263        let pte_ptr = unsafe { (paddr_to_vaddr(pt) as *mut E).add(index) };
264        // SAFETY: The pointer to the entry is valid to read.
265        let pte = unsafe { pte_ptr.read() };
266        let PteScalar::Mapped(pa, mut prop) = pte.to_repr(1) else {
267            panic!("protecting an unmapped page in the boot page table");
268        };
269        op(&mut prop);
270        // SAFETY: The pointer to the entry is valid to write.
271        unsafe { pte_ptr.write(E::from_repr(&PteScalar::Mapped(pa, prop), 1)) };
272    }
273
274    fn alloc_child(&mut self, level: PagingLevel) -> (E, Paddr) {
275        let frame_paddr = if frame::meta::is_initialized() {
276            let frame = FrameAllocOptions::new()
277                .zeroed(false)
278                .alloc_frame_with(EarlyAllocatedFrameMeta)
279                .unwrap();
280            frame.into_raw()
281        } else {
282            allocator::early_alloc(
283                Layout::from_size_align(C::BASE_PAGE_SIZE, C::BASE_PAGE_SIZE).unwrap(),
284            )
285            .unwrap()
286        };
287
288        // Zero it out.
289        let vaddr = paddr_to_vaddr(frame_paddr) as *mut u8;
290        // SAFETY: The allocated frame is valid to write.
291        unsafe { core::ptr::write_bytes(vaddr, 0, PAGE_SIZE) };
292
293        (
294            E::from_repr(
295                &PteScalar::PageTable(frame_paddr, PageTableFlags::empty()),
296                level,
297            ),
298            frame_paddr,
299        )
300    }
301
302    #[cfg(ktest)]
303    pub(super) fn new(root_pt: Paddr) -> Self {
304        Self {
305            root_pt,
306            _phantom: core::marker::PhantomData,
307        }
308    }
309}

311/// A helper function to walk on the page table frames.
312///
313/// Once leaving a page table frame, the closure will be called with the PTE to
314/// the frame.
315fn dfs_walk_on_leave<E: PteTrait, C: PagingConstsTrait>(
316    pt: Paddr,
317    level: PagingLevel,
318    op: &mut impl FnMut(&mut E, Paddr, PagingLevel, PageTableFlags),
319) {
320    if level >= 2 {
321        let pt_vaddr = paddr_to_vaddr(pt) as *mut E;
322        for offset in 0..nr_subpage_per_huge::<C>() {
323            // SAFETY: The result pointer is within the PT frame.
324            let pte_ptr = unsafe { pt_vaddr.add(offset) };
325            // SAFETY: The pointer to the entry is valid to read.
326            let mut pte = unsafe { pte_ptr.read() };
327
328            if let PteScalar::PageTable(child_pt, flags) = pte.to_repr(level) {
329                dfs_walk_on_leave::<E, C>(child_pt, level - 1, op);
330                op(&mut pte, child_pt, level, flags);
331                // SAFETY: The pointer to the entry is valid to write.
332                unsafe { pte_ptr.write(pte) };
333            }
334        }
335    }
