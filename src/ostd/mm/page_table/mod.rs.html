48/// The configurations of a page table.
49///
50/// It abstracts away both the usage and the architecture specifics from the
51/// general page table implementation. For examples:
52///  - the managed virtual address range;
53///  - the trackedness of physical mappings;
54///  - the PTE layout;
55///  - the number of page table levels, etc.
56///
57/// # Safety
58///
59/// The implementor must ensure that the `item_into_raw` and `item_from_raw`
60/// are implemented correctly so that:
61///  - `item_into_raw` consumes the ownership of the item;
62///  - if the provided raw form matches the item that was consumed by
63///    `item_into_raw`, `item_from_raw` restores the exact item that was
64///    consumed by `item_into_raw`.
65pub(crate) unsafe trait PageTableConfig:
66    Clone + Debug + Send + Sync + 'static
67{
68    /// The index range at the top level (`C::NR_LEVELS`) page table.
69    ///
70    /// When configured with this value, the [`PageTable`] instance will only
71    /// be allowed to manage the virtual address range that is covered by
72    /// this range. The range can be smaller than the actual allowed range
73    /// specified by the hardware MMU (limited by `C::ADDRESS_WIDTH`).
74    const TOP_LEVEL_INDEX_RANGE: Range<usize>;
75
76    /// If we can remove the top-level page table entries.
77    ///
78    /// This is for the kernel page table, whose second-top-level page
79    /// tables need `'static` lifetime to be shared with user page tables.
80    /// Other page tables do not need to set this to `false`.
81    const TOP_LEVEL_CAN_UNMAP: bool = true;
82
83    /// The type of the page table entry.
84    type E: PteTrait;
85
86    /// The paging constants.
87    type C: PagingConstsTrait;
88
89    /// The item that can be mapped into the virtual memory space using the
90    /// page table.
91    ///
92    /// Usually, this item is a [`crate::mm::Frame`], which we call a "tracked"
93    /// frame. The page table can also do "untracked" mappings that only maps
94    /// to certain physical addresses without tracking the ownership of the
95    /// mapped physical frame. The user of the page table APIs can choose by
96    /// defining this type and the corresponding methods [`item_into_raw`] and
97    /// [`item_from_raw`].
98    ///
99    /// [`item_from_raw`]: PageTableConfig::item_from_raw
100    /// [`item_into_raw`]: PageTableConfig::item_into_raw
101    type Item: Clone;
102
103    /// Consumes the item and returns the physical address, the paging level,
104    /// and the page property.
105    ///
106    /// The ownership of the item will be consumed, i.e., the item will be
107    /// forgotten after this function is called.
108    fn item_into_raw(item: Self::Item) -> (Paddr, PagingLevel, PageProperty);
109
110    /// Restores the item from the physical address and the paging level.
111    ///
112    /// There could be transformations after [`PageTableConfig::item_into_raw`]
113    /// and before [`PageTableConfig::item_from_raw`], which include:
114    ///  - splitting and coalescing the items, for example, splitting one item
115    ///    into 512 `level - 1` items with and contiguous physical addresses;
116    ///  - protecting the items, for example, changing the page property.
117    ///
118    /// Splitting and coalescing maintains ownership rules, i.e., if one
119    /// physical address is within the range of one item, after splitting/
120    /// coalescing, there should be exactly one item that contains the address.
121    ///
122    /// # Safety
123    ///
124    /// The caller must ensure that:
125    ///  - the physical address and the paging level represent a page table
126    ///    item or part of it (as described above);
127    ///  - either the ownership of the item is properly transferred to the
128    ///    return value, or the return value is wrapped in a
129    ///    [`core::mem::ManuallyDrop`] that won't outlive the original item;
130    ///  - the [`super::PageFlags::AVAIL1`] flag is preserved, i.e., it is
131    ///    the same as that returned from [`PageTableConfig::item_into_raw`].
132    unsafe fn item_from_raw(paddr: Paddr, level: PagingLevel, prop: PageProperty) -> Self::Item;
133}

136// like `page_size::<C>` without typing `C::C` everywhere.
137impl<C: PageTableConfig> PagingConstsTrait for C {
138    const BASE_PAGE_SIZE: usize = C::C::BASE_PAGE_SIZE;
139    const NR_LEVELS: PagingLevel = C::C::NR_LEVELS;
140    const HIGHEST_TRANSLATION_LEVEL: PagingLevel = C::C::HIGHEST_TRANSLATION_LEVEL;
141    const PTE_SIZE: usize = C::C::PTE_SIZE;
142    const ADDRESS_WIDTH: usize = C::C::ADDRESS_WIDTH;
143    const VA_SIGN_EXT: bool = C::C::VA_SIGN_EXT;
144}

146/// Splits the address range into largest page table items.
147///
148/// Each of the returned items is a tuple of the physical address and the
149/// paging level. It is helpful when you want to map a physical address range
150/// into the provided virtual address.
151///
152/// For example, on x86-64, `C: PageTableConfig` may specify level 1 page as
153/// 4KiB, level 2 page as 2MiB, and level 3 page as 1GiB. Suppose that the
154/// supplied physical address range is from `0x3fdff000` to `0x80002000`,
155/// and the virtual address is also `0x3fdff000`, the following 5 items will
156/// be returned:
157///
158/// ```text
159/// 0x3fdff000                                                 0x80002000
160/// start                                                             end
161///   |----|----------------|--------------------------------|----|----|
162///    4KiB      2MiB                       1GiB              4KiB 4KiB
163/// ```
164///
165/// # Panics
166///
167/// Panics if:
168///  - any of `va`, `pa`, or `len` is not aligned to the base page size;
169///  - the range `va..(va + len)` is not valid for the page table.
170pub(crate) fn largest_pages<C: PageTableConfig>(
171    mut va: Vaddr,
172    mut pa: Paddr,
173    mut len: usize,
174) -> impl Iterator<Item = (Paddr, PagingLevel)> {
175    assert_eq!(va % C::BASE_PAGE_SIZE, 0);
176    assert_eq!(pa % C::BASE_PAGE_SIZE, 0);
177    assert_eq!(len % C::BASE_PAGE_SIZE, 0);
178    assert!(is_valid_range::<C>(&(va..(va + len))));
179
180    core::iter::from_fn(move || {
181        if len == 0 {
182            return None;
183        }
184
185        let mut level = C::HIGHEST_TRANSLATION_LEVEL;
186        while page_size::<C>(level) > len
187            || !va.is_multiple_of(page_size::<C>(level))
188            || !pa.is_multiple_of(page_size::<C>(level))
189        {
190            level -= 1;
191        }
192
193        let item_start = pa;
194        va += page_size::<C>(level);
195        pa += page_size::<C>(level);
196        len -= page_size::<C>(level);
197
198        Some((item_start, level))
199    })
200}

202/// Gets the managed virtual addresses range for the page table.
203///
204/// It returns a [`RangeInclusive`] because the end address, if being
205/// [`Vaddr::MAX`], overflows [`Range<Vaddr>`].
206pub(super) const fn vaddr_range<C: PageTableConfig>() -> RangeInclusive<Vaddr> {
207    const fn top_level_index_width<C: PageTableConfig>() -> usize {
208        C::ADDRESS_WIDTH - pte_index_bit_offset::<C>(C::NR_LEVELS)
209    }
210
211    const {
212        assert!(C::TOP_LEVEL_INDEX_RANGE.start < C::TOP_LEVEL_INDEX_RANGE.end);
213        assert!(top_level_index_width::<C>() <= nr_pte_index_bits::<C>(),);
214        assert!(C::TOP_LEVEL_INDEX_RANGE.start < 1 << top_level_index_width::<C>());
215        assert!(C::TOP_LEVEL_INDEX_RANGE.end <= 1 << top_level_index_width::<C>());
216    };
217
218    const fn pt_va_range_start<C: PageTableConfig>() -> Vaddr {
219        C::TOP_LEVEL_INDEX_RANGE.start << pte_index_bit_offset::<C>(C::NR_LEVELS)
220    }
221
222    const fn pt_va_range_end<C: PageTableConfig>() -> Vaddr {
223        C::TOP_LEVEL_INDEX_RANGE
224            .end
225            .unbounded_shl(pte_index_bit_offset::<C>(C::NR_LEVELS) as u32)
226            .wrapping_sub(1) // Inclusive end.
227    }
228
229    const fn sign_bit_of_va<C: PageTableConfig>(va: Vaddr) -> bool {
230        (va >> (C::ADDRESS_WIDTH - 1)) & 1 != 0
231    }
232
233    let mut start = pt_va_range_start::<C>();
234    let mut end = pt_va_range_end::<C>();
235
236    const {
237        assert!(
238            !C::VA_SIGN_EXT
239                || sign_bit_of_va::<C>(pt_va_range_start::<C>())
240                    == sign_bit_of_va::<C>(pt_va_range_end::<C>()),
241            "The sign bit of both range endpoints must be the same if sign extension is enabled"
242        )
243    }
244
245    if C::VA_SIGN_EXT && sign_bit_of_va::<C>(pt_va_range_start::<C>()) {
246        start |= !0 ^ ((1 << C::ADDRESS_WIDTH) - 1);
247        end |= !0 ^ ((1 << C::ADDRESS_WIDTH) - 1);
248    }
249
250    start..=end
251}

253/// Checks if the given range is covered by the valid range of the page table.
254const fn is_valid_range<C: PageTableConfig>(r: &Range<Vaddr>) -> bool {
255    let va_range = vaddr_range::<C>();
256    (r.start == 0 && r.end == 0) || (*va_range.start() <= r.start && r.end - 1 <= *va_range.end())
257}

259// Here are some const values that are determined by the paging constants.
260
261/// The number of virtual address bits used to index a PTE in a page.
262const fn nr_pte_index_bits<C: PagingConstsTrait>() -> usize {
263    nr_subpage_per_huge::<C>().ilog2() as usize
264}
265
266/// The index of a VA's PTE in a page table node at the given level.
267const fn pte_index<C: PagingConstsTrait>(va: Vaddr, level: PagingLevel) -> usize {
268    (va >> pte_index_bit_offset::<C>(level)) & (nr_subpage_per_huge::<C>() - 1)
269}
270
271/// The bit offset of the entry offset part in a virtual address.
272///
273/// This function returns the bit offset of the least significant bit. Take
274/// x86-64 as an example, the `pte_index_bit_offset(2)` should return 21, which
275/// is 12 (the 4KiB in-page offset) plus 9 (index width in the level-1 table).
276const fn pte_index_bit_offset<C: PagingConstsTrait>(level: PagingLevel) -> usize {
277    C::BASE_PAGE_SIZE.ilog2() as usize + nr_pte_index_bits::<C>() * (level as usize - 1)
278}

297impl PageTable<KernelPtConfig> {
298    /// Create a new kernel page table.
299    pub(crate) fn new_kernel_page_table() -> Self {
300        let kpt = Self::empty();
301
302        // Make shared the page tables mapped by the root table in the kernel space.
303        {
304            let preempt_guard = disable_preempt();
305            let mut root_node = kpt.root.borrow().lock(&preempt_guard);
306
307            for i in KernelPtConfig::TOP_LEVEL_INDEX_RANGE {
308                let mut root_entry = root_node.entry(i);
309                let _ = root_entry.alloc_if_none(&preempt_guard).unwrap();
310            }
311        }
312
313        kpt
314    }
315
316    /// Create a new user page table.
317    ///
318    /// This should be the only way to create the user page table, that is to
319    /// duplicate the kernel page table with all the kernel mappings shared.
320    pub(in crate::mm) fn create_user_page_table(&'static self) -> PageTable<UserPtConfig> {
321        let new_root = PageTableNode::alloc(PagingConsts::NR_LEVELS);
322
323        let preempt_guard = disable_preempt();
324        let mut root_node = self.root.borrow().lock(&preempt_guard);
325        let mut new_node = new_root.borrow().lock(&preempt_guard);
326
327        const {
328            assert!(!KernelPtConfig::TOP_LEVEL_CAN_UNMAP);
329            assert!(
330                UserPtConfig::TOP_LEVEL_INDEX_RANGE.end
331                    <= KernelPtConfig::TOP_LEVEL_INDEX_RANGE.start
332            );
333        }
334
335        for i in KernelPtConfig::TOP_LEVEL_INDEX_RANGE {
336            let root_entry = root_node.entry(i);
337            let child = root_entry.to_ref();
338            let ChildRef::PageTable(pt) = child else {
339                panic!("The kernel page table doesn't contain shared nodes");
340            };
341
342            // We do not add additional reference count specifically for the
343            // shared kernel page tables. It requires user page tables to
344            // outlive the kernel page table, which is trivially true.
345            // See also `<PageTablePageMeta as AnyFrameMeta>::on_drop`.
346            let pt_addr = pt.paddr();
347            let pte = PageTableEntry::from_repr(
348                &PteScalar::PageTable(pt_addr, PageTableFlags::empty()),
349                UserPtConfig::NR_LEVELS,
350            );
351            // SAFETY: The index is within the bounds and the PTE is at the
352            // correct paging level. However, neither it's a `UserPtConfig`
353            // child nor the node has the ownership of the child. It is
354            // still safe because `UserPtConfig::TOP_LEVEL_INDEX_RANGE`
355            // guarantees that the cursor won't access it.
356            unsafe { new_node.write_pte(i, pte) };
357        }
358        drop(new_node);
359
360        PageTable::<UserPtConfig> { root: new_root }
361    }
362}

434/// A software emulation of the MMU address translation process.
435///
436/// This method returns the physical address of the given virtual address and
437/// the page property if a valid mapping exists for the given virtual address.
438///
439/// # Safety
440///
441/// The caller must ensure that the `root_paddr` is a pointer to a valid root
442/// page table node.
443///
444/// # Notes on the page table use-after-free problem
445///
446/// Neither the hardware MMU nor the software page walk method acquires the page
447/// table locks while reading. They can enter a to-be-recycled page table node
448/// and read the page table entries after the node is recycled and reused.
449///
450/// For the hardware MMU page walk, we mitigate this problem by dropping the page
451/// table nodes only after the TLBs have been flushed on all the CPUs that
452/// activate the page table.
453///
454/// For the software page walk, we only need to disable preemption at the beginning
455/// since the page table nodes won't be recycled in the RCU critical section.
456#[cfg(ktest)]
457pub(super) unsafe fn page_walk<C: PageTableConfig>(
458    root_paddr: Paddr,
459    vaddr: Vaddr,
460) -> Option<(Paddr, PageProperty)> {
461    use super::paddr_to_vaddr;
462
463    let _rcu_guard = disable_preempt();
464
465    let mut pt_addr = paddr_to_vaddr(root_paddr);
466    for cur_level in (1..=C::NR_LEVELS).rev() {
467        let offset = pte_index::<C>(vaddr, cur_level);
468        // SAFETY:
469        //  - The page table node is alive because (1) the root node is alive and
470        //    (2) all child nodes cannot be recycled because we're in the RCU critical section.
471        //  - The index is inside the bound, so the page table entry is valid.
472        //  - All page table entries are aligned and accessed with atomic operations only.
473        let cur_pte = unsafe { load_pte((pt_addr as *mut C::E).add(offset), Ordering::Acquire) };
474
475        match cur_pte.to_repr(cur_level) {
476            PteScalar::Absent => return None,
477            PteScalar::PageTable(next_pt_addr, _) => {
478                pt_addr = paddr_to_vaddr(next_pt_addr);
479                continue;
480            }
481            PteScalar::Mapped(frame_paddr, prop) => {
482                debug_assert!(cur_level <= C::HIGHEST_TRANSLATION_LEVEL);
483                return Some((
484                    frame_paddr + (vaddr & (page_size::<C>(cur_level) - 1)),
485                    prop,
486                ));
487            }
488        }
489    }
490
491    unreachable!("All present PTEs at the level 1 must be last-level PTEs");
492}

494/// The scalar representation of a page table entry (PTE).
495///
496/// This is an architecture-agnostic representation that can be converted
497/// to/from architecture-specific PTEs via the [`PteTrait`]. This is a scalar
498/// value that can be cloned or compared, and does not own the underlying page
499/// table node or mapped item if present.
500#[derive(Debug, Clone, PartialEq, Eq)]
501pub(crate) enum PteScalar {
502    /// A PTE that is considered absent by the MMU.
503    Absent,
504    /// A PTE that points to the next-level page table.
505    PageTable(Paddr, PageTableFlags),
506    /// A PTE that establishes the mapping to a physical frame.
507    Mapped(Paddr, PageProperty),
508}
509
510/// A trait that abstracts architecture-specific page table entries (PTEs).
511///
512/// A PTE refers to an entry in any level of a page table. This trait requires
513/// that any architecture-specific PTEs are scalar types that essentially
514/// encodes [`PteScalar`].
515///
516/// # Safety
517///
518/// An implementor must ensure that:
519///  - the methods `as_usize` and `from_usize` are not overridden;
520///  - the return value of `from_repr`, when called with `repr`, should return
521///    the same [`PteScalar`] that is passed to `from_repr`, if the level
522///    passed to `repr` is the same as that passed to `from_repr`;
523///  - a zeroed PTE represents an absent entry.
524pub(crate) unsafe trait PteTrait:
525    Clone + Copy + Debug + Pod + PodOnce + Sized + Send + Sync + 'static
526{
527    /// Returns architecture-specific representation of the PTE.
528    fn from_repr(repr: &PteScalar, level: PagingLevel) -> Self;
529
530    /// Returns the representation of the PTE.
531    ///
532    /// The caller must ensure that the level is the correct level of the PTE,
533    /// otherwise the implementation can return arbitrary value.
534    fn to_repr(&self, level: PagingLevel) -> PteScalar;
535
536    /// Converts the PTE into a raw `usize` value.
537    fn as_usize(self) -> usize {
538        const { assert!(size_of::<Self>() == size_of::<usize>()) };
539
540        // SAFETY: `Self` is `Pod` and has the same memory representation as `usize`.
541        unsafe { transmute_unchecked(self) }
542    }
543
544    /// Converts the raw `usize` value into a PTE.
545    fn from_usize(pte_raw: usize) -> Self {
546        const { assert!(size_of::<Self>() == size_of::<usize>()) };
547
548        // SAFETY: `Self` is `Pod` and has the same memory representation as `usize`.
549        unsafe { transmute_unchecked(pte_raw) }
550    }
551}

553/// Loads a page table entry with an atomic instruction.
554///
555/// # Safety
556///
557/// The safety preconditions are same as those of [`AtomicUsize::from_ptr`].
558pub unsafe fn load_pte<E: PteTrait>(ptr: *mut E, ordering: Ordering) -> E {
559    // SAFETY: The safety is upheld by the caller.
560    let atomic = unsafe { AtomicUsize::from_ptr(ptr.cast()) };
561    let pte_raw = atomic.load(ordering);
562    E::from_usize(pte_raw)
563}

565/// Stores a page table entry with an atomic instruction.
566///
567/// # Safety
568///
569/// The safety preconditions are same as those of [`AtomicUsize::from_ptr`].
570pub unsafe fn store_pte<E: PteTrait>(ptr: *mut E, new_val: E, ordering: Ordering) {
571    let new_raw = new_val.as_usize();
572    // SAFETY: The safety is upheld by the caller.
573    let atomic = unsafe { AtomicUsize::from_ptr(ptr.cast()) };
574    atomic.store(new_raw, ordering)
575}
