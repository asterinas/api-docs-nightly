46/// The configurations of a page table.
47///
48/// It abstracts away both the usage and the architecture specifics from the
49/// general page table implementation. For examples:
50///  - the managed virtual address range;
51///  - the trackedness of physical mappings;
52///  - the PTE layout;
53///  - the number of page table levels, etc.
54///
55/// # Safety
56///
57/// The implementor must ensure that the `item_into_raw` and `item_from_raw`
58/// are implemented correctly so that:
59///  - `item_into_raw` consumes the ownership of the item;
60///  - if the provided raw form matches the item that was consumed by
61///    `item_into_raw`, `item_from_raw` restores the exact item that was
62///    consumed by `item_into_raw`.
63pub(crate) unsafe trait PageTableConfig:
64    Clone + Debug + Send + Sync + 'static
65{
66    /// The index range at the top level (`C::NR_LEVELS`) page table.
67    ///
68    /// When configured with this value, the [`PageTable`] instance will only
69    /// be allowed to manage the virtual address range that is covered by
70    /// this range. The range can be smaller than the actual allowed range
71    /// specified by the hardware MMU (limited by `C::ADDRESS_WIDTH`).
72    const TOP_LEVEL_INDEX_RANGE: Range<usize>;
73
74    /// If we can remove the top-level page table entries.
75    ///
76    /// This is for the kernel page table, whose second-top-level page
77    /// tables need `'static` lifetime to be shared with user page tables.
78    /// Other page tables do not need to set this to `false`.
79    const TOP_LEVEL_CAN_UNMAP: bool = true;
80
81    /// The type of the page table entry.
82    type E: PageTableEntryTrait;
83
84    /// The paging constants.
85    type C: PagingConstsTrait;
86
87    /// The item that can be mapped into the virtual memory space using the
88    /// page table.
89    ///
90    /// Usually, this item is a [`crate::mm::Frame`], which we call a "tracked"
91    /// frame. The page table can also do "untracked" mappings that only maps
92    /// to certain physical addresses without tracking the ownership of the
93    /// mapped physical frame. The user of the page table APIs can choose by
94    /// defining this type and the corresponding methods [`item_into_raw`] and
95    /// [`item_from_raw`].
96    ///
97    /// [`item_from_raw`]: PageTableConfig::item_from_raw
98    /// [`item_into_raw`]: PageTableConfig::item_into_raw
99    type Item: Clone;
100
101    /// Consumes the item and returns the physical address, the paging level,
102    /// and the page property.
103    ///
104    /// The ownership of the item will be consumed, i.e., the item will be
105    /// forgotten after this function is called.
106    fn item_into_raw(item: Self::Item) -> (Paddr, PagingLevel, PageProperty);
107
108    /// Restores the item from the physical address and the paging level.
109    ///
110    /// There could be transformations after [`PageTableConfig::item_into_raw`]
111    /// and before [`PageTableConfig::item_from_raw`], which include:
112    ///  - splitting and coalescing the items, for example, splitting one item
113    ///    into 512 `level - 1` items with and contiguous physical addresses;
114    ///  - protecting the items, for example, changing the page property.
115    ///
116    /// Splitting and coalescing maintains ownership rules, i.e., if one
117    /// physical address is within the range of one item, after splitting/
118    /// coalescing, there should be exactly one item that contains the address.
119    ///
120    /// # Safety
121    ///
122    /// The caller must ensure that:
123    ///  - the physical address and the paging level represent a page table
124    ///    item or part of it (as described above);
125    ///  - either the ownership of the item is properly transferred to the
126    ///    return value, or the return value is wrapped in a
127    ///    [`core::mem::ManuallyDrop`] that won't outlive the original item;
128    ///  - the [`super::PageFlags::AVAIL1`] flag is preserved, i.e., it is
129    ///    the same as that returned from [`PageTableConfig::item_into_raw`].
130    unsafe fn item_from_raw(paddr: Paddr, level: PagingLevel, prop: PageProperty) -> Self::Item;
131}

133// Implement it so that we can comfortably use low level functions
134// like `page_size::<C>` without typing `C::C` everywhere.
135impl<C: PageTableConfig> PagingConstsTrait for C {
136    const BASE_PAGE_SIZE: usize = C::C::BASE_PAGE_SIZE;
137    const NR_LEVELS: PagingLevel = C::C::NR_LEVELS;
138    const HIGHEST_TRANSLATION_LEVEL: PagingLevel = C::C::HIGHEST_TRANSLATION_LEVEL;
139    const PTE_SIZE: usize = C::C::PTE_SIZE;
140    const ADDRESS_WIDTH: usize = C::C::ADDRESS_WIDTH;
141    const VA_SIGN_EXT: bool = C::C::VA_SIGN_EXT;
142}

144/// Splits the address range into largest page table items.
145///
146/// Each of the returned items is a tuple of the physical address and the
147/// paging level. It is helpful when you want to map a physical address range
148/// into the provided virtual address.
149///
150/// For example, on x86-64, `C: PageTableConfig` may specify level 1 page as
151/// 4KiB, level 2 page as 2MiB, and level 3 page as 1GiB. Suppose that the
152/// supplied physical address range is from `0x3fdff000` to `0x80002000`,
153/// and the virtual address is also `0x3fdff000`, the following 5 items will
154/// be returned:
155///
156/// ```text
157/// 0x3fdff000                                                 0x80002000
158/// start                                                             end
159///   |----|----------------|--------------------------------|----|----|
160///    4KiB      2MiB                       1GiB              4KiB 4KiB
161/// ```
162///
163/// # Panics
164///
165/// Panics if:
166///  - any of `va`, `pa`, or `len` is not aligned to the base page size;
167///  - the range `va..(va + len)` is not valid for the page table.
168pub(crate) fn largest_pages<C: PageTableConfig>(
169    mut va: Vaddr,
170    mut pa: Paddr,
171    mut len: usize,
172) -> impl Iterator<Item = (Paddr, PagingLevel)> {
173    assert_eq!(va % C::BASE_PAGE_SIZE, 0);
174    assert_eq!(pa % C::BASE_PAGE_SIZE, 0);
175    assert_eq!(len % C::BASE_PAGE_SIZE, 0);
176    assert!(is_valid_range::<C>(&(va..(va + len))));
177
178    core::iter::from_fn(move || {
179        if len == 0 {
180            return None;
181        }
182
183        let mut level = C::HIGHEST_TRANSLATION_LEVEL;
184        while page_size::<C>(level) > len
185            || !va.is_multiple_of(page_size::<C>(level))
186            || !pa.is_multiple_of(page_size::<C>(level))
187        {
188            level -= 1;
189        }
190
191        let item_start = pa;
192        va += page_size::<C>(level);
193        pa += page_size::<C>(level);
194        len -= page_size::<C>(level);
195
196        Some((item_start, level))
197    })
198}

200/// Gets the managed virtual addresses range for the page table.
201///
202/// It returns a [`RangeInclusive`] because the end address, if being
203/// [`Vaddr::MAX`], overflows [`Range<Vaddr>`].
204const fn vaddr_range<C: PageTableConfig>() -> RangeInclusive<Vaddr> {
205    const fn top_level_index_width<C: PageTableConfig>() -> usize {
206        C::ADDRESS_WIDTH - pte_index_bit_offset::<C>(C::NR_LEVELS)
207    }
208
209    const {
210        assert!(C::TOP_LEVEL_INDEX_RANGE.start < C::TOP_LEVEL_INDEX_RANGE.end);
211        assert!(top_level_index_width::<C>() <= nr_pte_index_bits::<C>(),);
212        assert!(C::TOP_LEVEL_INDEX_RANGE.start < 1 << top_level_index_width::<C>());
213        assert!(C::TOP_LEVEL_INDEX_RANGE.end <= 1 << top_level_index_width::<C>());
214    };
215
216    const fn pt_va_range_start<C: PageTableConfig>() -> Vaddr {
217        C::TOP_LEVEL_INDEX_RANGE.start << pte_index_bit_offset::<C>(C::NR_LEVELS)
218    }
219
220    const fn pt_va_range_end<C: PageTableConfig>() -> Vaddr {
221        C::TOP_LEVEL_INDEX_RANGE
222            .end
223            .unbounded_shl(pte_index_bit_offset::<C>(C::NR_LEVELS) as u32)
224            .wrapping_sub(1) // Inclusive end.
225    }
226
227    const fn sign_bit_of_va<C: PageTableConfig>(va: Vaddr) -> bool {
228        (va >> (C::ADDRESS_WIDTH - 1)) & 1 != 0
229    }
230
231    let mut start = pt_va_range_start::<C>();
232    let mut end = pt_va_range_end::<C>();
233
234    const {
235        assert!(
236            !C::VA_SIGN_EXT
237                || sign_bit_of_va::<C>(pt_va_range_start::<C>())
238                    == sign_bit_of_va::<C>(pt_va_range_end::<C>()),
239            "The sign bit of both range endpoints must be the same if sign extension is enabled"
240        )
241    }
242
243    if C::VA_SIGN_EXT && sign_bit_of_va::<C>(pt_va_range_start::<C>()) {
244        start |= !0 ^ ((1 << C::ADDRESS_WIDTH) - 1);
245        end |= !0 ^ ((1 << C::ADDRESS_WIDTH) - 1);
246    }
247
248    start..=end
249}

251/// Checks if the given range is covered by the valid range of the page table.
252const fn is_valid_range<C: PageTableConfig>(r: &Range<Vaddr>) -> bool {
253    let va_range = vaddr_range::<C>();
254    (r.start == 0 && r.end == 0) || (*va_range.start() <= r.start && r.end - 1 <= *va_range.end())
255}

257// Here are some const values that are determined by the paging constants.
258
259/// The number of virtual address bits used to index a PTE in a page.
260const fn nr_pte_index_bits<C: PagingConstsTrait>() -> usize {
261    nr_subpage_per_huge::<C>().ilog2() as usize
262}
263
264/// The index of a VA's PTE in a page table node at the given level.
265const fn pte_index<C: PagingConstsTrait>(va: Vaddr, level: PagingLevel) -> usize {
266    (va >> pte_index_bit_offset::<C>(level)) & (nr_subpage_per_huge::<C>() - 1)
267}
268
269/// The bit offset of the entry offset part in a virtual address.
270///
271/// This function returns the bit offset of the least significant bit. Take
272/// x86-64 as an example, the `pte_index_bit_offset(2)` should return 21, which
273/// is 12 (the 4KiB in-page offset) plus 9 (index width in the level-1 table).
274const fn pte_index_bit_offset<C: PagingConstsTrait>(level: PagingLevel) -> usize {
275    C::BASE_PAGE_SIZE.ilog2() as usize + nr_pte_index_bits::<C>() * (level as usize - 1)
276}

295impl PageTable<KernelPtConfig> {
296    /// Create a new kernel page table.
297    pub(crate) fn new_kernel_page_table() -> Self {
298        let kpt = Self::empty();
299
300        // Make shared the page tables mapped by the root table in the kernel space.
301        {
302            let preempt_guard = disable_preempt();
303            let mut root_node = kpt.root.borrow().lock(&preempt_guard);
304
305            for i in KernelPtConfig::TOP_LEVEL_INDEX_RANGE {
306                let mut root_entry = root_node.entry(i);
307                let _ = root_entry.alloc_if_none(&preempt_guard).unwrap();
308            }
309        }
310
311        kpt
312    }
313
314    /// Create a new user page table.
315    ///
316    /// This should be the only way to create the user page table, that is to
317    /// duplicate the kernel page table with all the kernel mappings shared.
318    pub(in crate::mm) fn create_user_page_table(&'static self) -> PageTable<UserPtConfig> {
319        let new_root = PageTableNode::alloc(PagingConsts::NR_LEVELS);
320
321        let preempt_guard = disable_preempt();
322        let mut root_node = self.root.borrow().lock(&preempt_guard);
323        let mut new_node = new_root.borrow().lock(&preempt_guard);
324
325        const {
326            assert!(!KernelPtConfig::TOP_LEVEL_CAN_UNMAP);
327            assert!(
328                UserPtConfig::TOP_LEVEL_INDEX_RANGE.end
329                    <= KernelPtConfig::TOP_LEVEL_INDEX_RANGE.start
330            );
331        }
332
333        for i in KernelPtConfig::TOP_LEVEL_INDEX_RANGE {
334            let root_entry = root_node.entry(i);
335            let child = root_entry.to_ref();
336            let ChildRef::PageTable(pt) = child else {
337                panic!("The kernel page table doesn't contain shared nodes");
338            };
339
340            // We do not add additional reference count specifically for the
341            // shared kernel page tables. It requires user page tables to
342            // outlive the kernel page table, which is trivially true.
343            // See also `<PageTablePageMeta as AnyFrameMeta>::on_drop`.
344            let pt_addr = pt.paddr();
345            let pte = PageTableEntry::new_pt(pt_addr);
346            // SAFETY: The index is within the bounds and the PTE is at the
347            // correct paging level. However, neither it's a `UserPtConfig`
348            // child nor the node has the ownership of the child. It is
349            // still safe because `UserPtConfig::TOP_LEVEL_INDEX_RANGE`
350            // guarantees that the cursor won't access it.
351            unsafe { new_node.write_pte(i, pte) };
352        }
353        drop(new_node);
354
355        PageTable::<UserPtConfig> { root: new_root }
356    }
357
358    /// Protect the given virtual address range in the kernel page table.
359    ///
360    /// This method flushes the TLB entries when doing protection.
361    ///
362    /// # Safety
363    ///
364    /// The caller must ensure that the protection operation does not affect
365    /// the memory safety of the kernel.
366    pub unsafe fn protect_flush_tlb(
367        &self,
368        vaddr: &Range<Vaddr>,
369        mut op: impl FnMut(&mut PageProperty),
370    ) -> Result<(), PageTableError> {
371        let preempt_guard = disable_preempt();
372        let mut cursor = CursorMut::new(self, &preempt_guard, vaddr)?;
373        // SAFETY: The safety is upheld by the caller.
374        while let Some(range) =
375            unsafe { cursor.protect_next(vaddr.end - cursor.virt_addr(), &mut op) }
376        {
377            crate::arch::mm::tlb_flush_addr(range.start);
378        }
379        Ok(())
380    }
381}

453/// A software emulation of the MMU address translation process.
454///
455/// This method returns the physical address of the given virtual address and
456/// the page property if a valid mapping exists for the given virtual address.
457///
458/// # Safety
459///
460/// The caller must ensure that the `root_paddr` is a pointer to a valid root
461/// page table node.
462///
463/// # Notes on the page table use-after-free problem
464///
465/// Neither the hardware MMU nor the software page walk method acquires the page
466/// table locks while reading. They can enter a to-be-recycled page table node
467/// and read the page table entries after the node is recycled and reused.
468///
469/// For the hardware MMU page walk, we mitigate this problem by dropping the page
470/// table nodes only after the TLBs have been flushed on all the CPUs that
471/// activate the page table.
472///
473/// For the software page walk, we only need to disable preemption at the beginning
474/// since the page table nodes won't be recycled in the RCU critical section.
475#[cfg(ktest)]
476pub(super) unsafe fn page_walk<C: PageTableConfig>(
477    root_paddr: Paddr,
478    vaddr: Vaddr,
479) -> Option<(Paddr, PageProperty)> {
480    use super::paddr_to_vaddr;
481
482    let _rcu_guard = disable_preempt();
483
484    let mut pt_addr = paddr_to_vaddr(root_paddr);
485    for cur_level in (1..=C::NR_LEVELS).rev() {
486        let offset = pte_index::<C>(vaddr, cur_level);
487        // SAFETY:
488        //  - The page table node is alive because (1) the root node is alive and
489        //    (2) all child nodes cannot be recycled because we're in the RCU critical section.
490        //  - The index is inside the bound, so the page table entry is valid.
491        //  - All page table entries are aligned and accessed with atomic operations only.
492        let cur_pte = unsafe { load_pte((pt_addr as *mut C::E).add(offset), Ordering::Acquire) };
493
494        if !cur_pte.is_present() {
495            return None;
496        }
497
498        if cur_pte.is_last(cur_level) {
499            debug_assert!(cur_level <= C::HIGHEST_TRANSLATION_LEVEL);
500            return Some((
501                cur_pte.paddr() + (vaddr & (page_size::<C>(cur_level) - 1)),
502                cur_pte.prop(),
503            ));
504        }
505
506        pt_addr = paddr_to_vaddr(cur_pte.paddr());
507    }
508
509    unreachable!("All present PTEs at the level 1 must be last-level PTEs");
510}

512/// A trait that abstracts architecture-specific page table entries (PTEs).
513///
514/// Note that a default PTE should be a PTE that points to nothing.
515pub trait PageTableEntryTrait:
516    Clone + Copy + Debug + Default + Pod + PodOnce + Sized + Send + Sync + 'static
517{
518    /// Creates a PTE that points to nothing.
519    ///
520    /// Note that currently the implementation requires a zeroed PTE to be an absent PTE.
521    fn new_absent() -> Self {
522        Self::default()
523    }
524
525    /// Returns if the PTE points to something.
526    ///
527    /// For PTEs created by [`Self::new_absent`], this method should return
528    /// false. For PTEs created by [`Self::new_page`] or [`Self::new_pt`]
529    /// and modified with [`Self::set_prop`], this method should return true.
530    fn is_present(&self) -> bool;
531
532    /// Creates a new PTE that maps to a page.
533    fn new_page(paddr: Paddr, level: PagingLevel, prop: PageProperty) -> Self;
534
535    /// Creates a new PTE that maps to a child page table.
536    fn new_pt(paddr: Paddr) -> Self;
537
538    /// Returns the physical address from the PTE.
539    ///
540    /// The physical address recorded in the PTE is either:
541    /// - the physical address of the next-level page table, or
542    /// - the physical address of the page that the PTE maps to.
543    fn paddr(&self) -> Paddr;
544
545    /// Returns the page property of the PTE.
546    fn prop(&self) -> PageProperty;
547
548    /// Sets the page property of the PTE.
549    ///
550    /// This methold has an impact only if the PTE is present. If not, this
551    /// method will do nothing.
552    fn set_prop(&mut self, prop: PageProperty);
553
554    /// Returns if the PTE maps a page rather than a child page table.
555    ///
556    /// The method needs to know the level of the page table where the PTE resides,
557    /// since architectures like x86-64 have a huge bit only in intermediate levels.
558    fn is_last(&self, level: PagingLevel) -> bool;
559
560    /// Converts the PTE into a raw `usize` value.
561    fn as_usize(self) -> usize {
562        const { assert!(size_of::<Self>() == size_of::<usize>()) };
563
564        // SAFETY: `Self` is `Pod` and has the same memory representation as `usize`.
565        unsafe { transmute_unchecked(self) }
566    }
567
568    /// Converts the raw `usize` value into a PTE.
569    fn from_usize(pte_raw: usize) -> Self {
570        const { assert!(size_of::<Self>() == size_of::<usize>()) };
571
572        // SAFETY: `Self` is `Pod` and has the same memory representation as `usize`.
573        unsafe { transmute_unchecked(pte_raw) }
574    }
575}

577/// Loads a page table entry with an atomic instruction.
578///
579/// # Safety
580///
581/// The safety preconditions are same as those of [`AtomicUsize::from_ptr`].
582pub unsafe fn load_pte<E: PageTableEntryTrait>(ptr: *mut E, ordering: Ordering) -> E {
583    // SAFETY: The safety is upheld by the caller.
584    let atomic = unsafe { AtomicUsize::from_ptr(ptr.cast()) };
585    let pte_raw = atomic.load(ordering);
586    E::from_usize(pte_raw)
587}

589/// Stores a page table entry with an atomic instruction.
590///
591/// # Safety
592///
593/// The safety preconditions are same as those of [`AtomicUsize::from_ptr`].
594pub unsafe fn store_pte<E: PageTableEntryTrait>(ptr: *mut E, new_val: E, ordering: Ordering) {
595    let new_raw = new_val.as_usize();
596    // SAFETY: The safety is upheld by the caller.
597    let atomic = unsafe { AtomicUsize::from_ptr(ptr.cast()) };
598    atomic.store(new_raw, ordering)
599}
