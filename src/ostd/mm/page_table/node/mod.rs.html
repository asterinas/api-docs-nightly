3//! This module defines page table node abstractions and the handle.
4//!
5//! The page table node is also frequently referred to as a page table in many architectural
6//! documentations. It is essentially a page that contains page table entries (PTEs) that map
7//! to child page tables nodes or mapped pages.
8//!
9//! This module leverages the page metadata to manage the page table pages, which makes it
10//! easier to provide the following guarantees:
11//!
12//! The page table node is not freed when it is still in use by:
13//!    - a parent page table node,
14//!    - or a handle to a page table node,
15//!    - or a processor.
16//!
17//! This is implemented by using a reference counter in the page metadata. If the above
18//! conditions are not met, the page table node is ensured to be freed upon dropping the last
19//! reference.
20//!
21//! One can acquire exclusive access to a page table node using merely the physical address of
22//! the page table node. This is implemented by a lock in the page metadata. Here the
23//! exclusiveness is only ensured for kernel code, and the processor's MMU is able to access the
24//! page table node while a lock is held. So the modification to the PTEs should be done after
25//! the initialization of the entity that the PTE points to. This is taken care in this module.
26//!
27

53/// A smart pointer to a page table node.
54///
55/// This smart pointer is an owner of a page table node. Thus creating and
56/// dropping it will affect the reference count of the page table node. If
57/// dropped it as the last reference, the page table node and subsequent
58/// children will be freed.
59///
60/// [`PageTableNode`] is read-only. To modify the page table node, lock and use
61/// [`PageTableGuard`].
62pub(super) type PageTableNode<C> = Frame<PageTablePageMeta<C>>;
63
64impl<C: PageTableConfig> PageTableNode<C> {
65    pub(super) fn level(&self) -> PagingLevel {
66        self.meta().level
67    }
68
69    /// Allocates a new empty page table node.
70    pub(super) fn alloc(level: PagingLevel) -> Self {
71        let meta = PageTablePageMeta::new(level);
72        FrameAllocOptions::new()
73            .zeroed(true)
74            .alloc_frame_with(meta)
75            .expect("Failed to allocate a page table node")
76    }
77
78    /// Activates the page table assuming it is a root page table.
79    ///
80    /// Here we ensure not dropping an active page table by making a
81    /// processor a page table owner. When activating a page table, the
82    /// reference count of the last activated page table is decremented.
83    /// And that of the current page table is incremented.
84    ///
85    /// # Safety
86    ///
87    /// The caller must ensure that the page table to be activated has
88    /// proper mappings for the kernel and has the correct const parameters
89    /// matching the current CPU.
90    ///
91    /// # Panics
92    ///
93    /// Only top-level page tables can be activated using this function.
94    pub(crate) unsafe fn activate(&self) {
95        use crate::{
96            arch::mm::{activate_page_table, current_page_table_paddr},
97            mm::CachePolicy,
98        };
99
100        assert_eq!(self.level(), C::NR_LEVELS);
101
102        let last_activated_paddr = current_page_table_paddr();
103        if last_activated_paddr == self.paddr() {
104            return;
105        }
106
107        // SAFETY: The safety is upheld by the caller.
108        unsafe { activate_page_table(self.clone().into_raw(), CachePolicy::Writeback) };
109
110        // Restore and drop the last activated page table.
111        // SAFETY: The physical address is valid and points to a forgotten page table node.
112        drop(unsafe { Self::from_raw(last_activated_paddr) });
113    }
114
115    /// Activates the (root) page table assuming it is the first activation.
116    ///
117    /// It will not try dropping the last activate page table. It is the same
118    /// with [`Self::activate()`] in other senses.
119    pub(super) unsafe fn first_activate(&self) {
120        use crate::{arch::mm::activate_page_table, mm::CachePolicy};
121
122        // SAFETY: The safety is upheld by the caller.
123        unsafe { activate_page_table(self.clone().into_raw(), CachePolicy::Writeback) };
124    }
125}

127/// A reference to a page table node.
128pub(super) type PageTableNodeRef<'a, C> = FrameRef<'a, PageTablePageMeta<C>>;
129
130impl<'a, C: PageTableConfig> PageTableNodeRef<'a, C> {
131    /// Locks the page table node.
132    ///
133    /// An atomic mode guard is required to
134    ///  1. prevent deadlocks;
135    ///  2. provide a lifetime (`'rcu`) that the nodes are guaranteed to outlive.
136    pub(super) fn lock<'rcu>(self, _guard: &'rcu dyn InAtomicMode) -> PageTableGuard<'rcu, C>
137    where
138        'a: 'rcu,
139    {
140        while self
141            .meta()
142            .lock
143            .compare_exchange(0, 1, Ordering::Acquire, Ordering::Relaxed)
144            .is_err()
145        {
146            core::hint::spin_loop();
147        }
148
149        PageTableGuard::<'rcu, C> { inner: self }
150    }
151
152    /// Creates a new [`PageTableGuard`] without checking if the page table lock is held.
153    ///
154    /// # Safety
155    ///
156    /// This function must be called if this task logically holds the lock.
157    ///
158    /// Calling this function when a guard is already created is undefined behavior
159    /// unless that guard was already forgotten.
160    pub(super) unsafe fn make_guard_unchecked<'rcu>(
161        self,
162        _guard: &'rcu dyn InAtomicMode,
163    ) -> PageTableGuard<'rcu, C>
164    where
165        'a: 'rcu,
166    {
167        PageTableGuard { inner: self }
168    }
169}

171/// A guard that holds the lock of a page table node.
172#[derive(Debug)]
173pub(super) struct PageTableGuard<'rcu, C: PageTableConfig> {
174    inner: PageTableNodeRef<'rcu, C>,
175}
176
177impl<'rcu, C: PageTableConfig> PageTableGuard<'rcu, C> {
178    /// Borrows an entry in the node at a given index.
179    ///
180    /// # Panics
181    ///
182    /// Panics if the index is not within the bound of
183    /// [`nr_subpage_per_huge<C>`].
184    pub(super) fn entry(&mut self, idx: usize) -> Entry<'_, 'rcu, C> {
185        assert!(idx < nr_subpage_per_huge::<C>());
186        // SAFETY: The index is within the bound.
187        unsafe { Entry::new_at(self, idx) }
188    }
189
190    /// Gets the number of valid PTEs in the node.
191    pub(super) fn nr_children(&self) -> u16 {
192        // SAFETY: The lock is held so we have an exclusive access.
193        unsafe { *self.meta().nr_children.get() }
194    }
195
196    /// Returns if the page table node is detached from its parent.
197    pub(super) fn stray_mut(&mut self) -> &mut bool {
198        // SAFETY: The lock is held so we have an exclusive access.
199        unsafe { &mut *self.meta().stray.get() }
200    }
201
202    /// Reads a non-owning PTE at the given index.
203    ///
204    /// A non-owning PTE means that it does not account for a reference count
205    /// of the a page if the PTE points to a page. The original PTE still owns
206    /// the child page.
207    ///
208    /// # Safety
209    ///
210    /// The caller must ensure that the index is within the bound.
211    pub(super) unsafe fn read_pte(&self, idx: usize) -> C::E {
212        debug_assert!(idx < nr_subpage_per_huge::<C>());
213        let ptr = paddr_to_vaddr(self.paddr()) as *mut C::E;
214        // SAFETY:
215        // - The page table node is alive. The index is inside the bound, so the page table entry is valid.
216        // - All page table entries are aligned and accessed with atomic operations only.
217        unsafe { load_pte(ptr.add(idx), Ordering::Relaxed) }
218    }
219
220    /// Writes a page table entry at a given index.
221    ///
222    /// This operation will leak the old child if the old PTE is present.
223    ///
224    /// # Safety
225    ///
226    /// The caller must ensure that:
227    ///  1. The index must be within the bound;
228    ///  2. The PTE must represent a [`Child`] in the same [`PageTableConfig`]
229    ///     and at the right paging level (`self.level() - 1`).
230    ///  3. The page table node will have the ownership of the [`Child`]
231    ///     after this method.
232    pub(super) unsafe fn write_pte(&mut self, idx: usize, pte: C::E) {
233        debug_assert!(idx < nr_subpage_per_huge::<C>());
234        let ptr = paddr_to_vaddr(self.paddr()) as *mut C::E;
235        // SAFETY:
236        // - The page table node is alive. The index is inside the bound, so the page table entry is valid.
237        // - All page table entries are aligned and accessed with atomic operations only.
238        unsafe { store_pte(ptr.add(idx), pte, Ordering::Release) }
239    }
240
241    /// Gets the mutable reference to the number of valid PTEs in the node.
242    fn nr_children_mut(&mut self) -> &mut u16 {
243        // SAFETY: The lock is held so we have an exclusive access.
244        unsafe { &mut *self.meta().nr_children.get() }
245    }
246}

248impl<'rcu, C: PageTableConfig> Deref for PageTableGuard<'rcu, C> {
249    type Target = PageTableNodeRef<'rcu, C>;
250
251    fn deref(&self) -> &Self::Target {
252        &self.inner
253    }
254}
255
256impl<C: PageTableConfig> Drop for PageTableGuard<'_, C> {
257    fn drop(&mut self) {
258        self.inner.meta().lock.store(0, Ordering::Release);
259    }
260}

262/// The metadata of any kinds of page table pages.
263/// Make sure the the generic parameters don't effect the memory layout.
264#[derive(Debug)]
265pub(in crate::mm) struct PageTablePageMeta<C: PageTableConfig> {
266    /// The number of valid PTEs. It is mutable if the lock is held.
267    pub nr_children: SyncUnsafeCell<u16>,
268    /// If the page table is detached from its parent.
269    ///
270    /// A page table can be detached from its parent while still being accessed,
271    /// since we use a RCU scheme to recycle page tables. If this flag is set,
272    /// it means that the parent is recycling the page table.
273    pub stray: SyncUnsafeCell<bool>,
274    /// The level of the page table page. A page table page cannot be
275    /// referenced by page tables of different levels.
276    pub level: PagingLevel,
277    /// The lock for the page table page.
278    pub lock: AtomicU8,
279    _phantom: core::marker::PhantomData<C>,
280}
281
282impl<C: PageTableConfig> PageTablePageMeta<C> {
283    pub fn new(level: PagingLevel) -> Self {
284        Self {
285            nr_children: SyncUnsafeCell::new(0),
286            stray: SyncUnsafeCell::new(false),
287            level,
288            lock: AtomicU8::new(0),
289            _phantom: PhantomData,
290        }
291    }
292}

294// FIXME: The safe APIs in the `page_table/node` module allow `Child::Frame`s with
295// arbitrary addresses to be stored in the page table nodes. Therefore, they may not
296// be valid `C::Item`s. The soundness of the following `on_drop` implementation must
297// be reasoned in conjunction with the `page_table/cursor` implementation.
298unsafe impl<C: PageTableConfig> AnyFrameMeta for PageTablePageMeta<C> {
299    fn on_drop(&mut self, reader: &mut VmReader<Infallible>) {
300        let nr_children = self.nr_children.get_mut();
301        if *nr_children == 0 {
302            return;
303        }
304
305        let level = self.level;
306        let range = if level == C::NR_LEVELS {
307            C::TOP_LEVEL_INDEX_RANGE.clone()
308        } else {
309            0..nr_subpage_per_huge::<C>()
310        };
311
312        // Drop the children.
313        reader.skip(range.start * size_of::<C::E>());
314        for _ in range {
315            // Non-atomic read is OK because we have mutable access.
316            let pte = reader.read_once::<C::E>().unwrap();
317            match pte.to_repr(level) {
318                PteScalar::PageTable(child_pt_addr, _) => {
319                    // SAFETY: The PTE points to a page table node. The ownership
320                    // of the child is transferred to the child then dropped.
321                    drop(unsafe { PageTableNode::<C>::from_raw(child_pt_addr) });
322                }
323                PteScalar::Mapped(pa, prop) => {
324                    // SAFETY: The PTE points to a mapped item. The ownership
325                    // of the item is transferred here then dropped.
326                    drop(unsafe { C::item_from_raw(pa, level, prop) });
327                }
328                PteScalar::Absent => {}
329            }
330        }
331    }
