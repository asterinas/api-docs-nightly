16/// I/O port allocator that allocates port I/O access to device drivers.
17pub(super) struct IoPortAllocator {
18    /// Each ID indicates whether a Port I/O (1B) is allocated.
19    ///
20    /// Instead of using `RangeAllocator` like `IoMemAllocator` does, it is more reasonable to use `IdAlloc`,
21    /// as PIO space includes only a small region; for example, x86 module in OSTD allows just 65536 I/O ports.
22    allocator: SpinLock<IdAlloc, LocalIrqDisabled>,
23}
24
25impl IoPortAllocator {
26    /// Acquires an `IoPort`. Returns `None` if the PIO range is unavailable.
27    ///
28    /// `is_overlapping` indicates whether another `IoPort` can have a PIO range that overlaps with
29    /// this one. If it is true, only the first port in the PIO range will be marked as occupied;
30    /// otherwise, all ports in the PIO range will be marked as occupied.
31    pub(super) fn acquire<T, A>(&self, port: u16, is_overlapping: bool) -> Option<IoPort<T, A>> {
32        let range = if !is_overlapping {
33            port..port.checked_add(size_of::<T>().try_into().ok()?)?
34        } else {
35            port..port.checked_add(1)?
36        };
37        debug!("Try to acquire PIO range: {:#x?}", range);
38
39        let mut allocator = self.allocator.lock();
40        if range.clone().any(|i| allocator.is_allocated(i as usize)) {
41            return None;
42        }
43
44        for i in range.clone() {
45            allocator.alloc_specific(i as usize);
46        }
47
48        // SAFETY: The created `IoPort` is guaranteed not to access system device I/O.
49        unsafe { Some(IoPort::new_overlapping(port, is_overlapping)) }
50    }
51
52    /// Recycles an PIO range.
53    ///
54    /// # Safety
55    ///
56    /// The caller must have ownership of the PIO region through the `IoPortAllocator::acquire` interface.
57    pub(super) unsafe fn recycle(&self, range: Range<u16>) {
58        debug!("Recycling PIO range: {:#x?}", range);
59
60        self.allocator
61            .lock()
62            .free_consecutive(range.start as usize..range.end as usize);
63    }
64}
65
66pub(super) static IO_PORT_ALLOCATOR: Once<IoPortAllocator> = Once::new();
67
68/// Initializes the static `IO_PORT_ALLOCATOR` and removes the system device I/O port regions.
69///
70/// # Safety
71///
72/// User must ensure that:
73///
74/// 1. All the port I/O regions belonging to the system device are defined using the macros
75///    `sensitive_io_port` and `reserve_io_port_range`.
76///
77/// 2. `MAX_IO_PORT` defined in `crate::arch::io` is guaranteed not to exceed the maximum
78///    value specified by architecture.
79pub(in crate::io) unsafe fn init() {
80    // SAFETY: `MAX_IO_PORT` is guaranteed not to exceed the maximum value specified by architecture.
81    let mut allocator = IdAlloc::with_capacity(crate::arch::io::MAX_IO_PORT as usize);
82
83    unsafe extern "C" {
84        fn __sensitive_io_ports_start();
85        fn __sensitive_io_ports_end();
86    }
87    let start = __sensitive_io_ports_start as *const () as usize;
88    let end = __sensitive_io_ports_end as *const () as usize;
89    assert!((end - start).is_multiple_of(size_of::<RawIoPortRange>()));
90
91    // Iterate through the sensitive I/O port ranges and remove them from the allocator.
92    let io_port_range_count = (end - start) / size_of::<RawIoPortRange>();
93    for i in 0..io_port_range_count {
94        let range_base_addr =
95            __sensitive_io_ports_start as *const () as usize + i * size_of::<RawIoPortRange>();
96        // SAFETY: The range is guaranteed to be valid as it is defined in the `.sensitive_io_ports` section.
97        let port_range = unsafe { *(range_base_addr as *const RawIoPortRange) };
98
99        assert!(port_range.begin < port_range.end);
100        debug!("Removing sensitive I/O port range: {:#x?}", port_range);
101
102        for i in port_range.begin..port_range.end {
103            allocator.alloc_specific(i as usize);
104        }
105    }
106
107    IO_PORT_ALLOCATOR.call_once(|| IoPortAllocator {
108        allocator: SpinLock::new(allocator),
109    });
110}

112#[cfg(ktest)]
113mod test {
114    use crate::{arch::device::io_port::ReadWriteAccess, prelude::*};
115
116    type IoPort = crate::io::IoPort<u32, ReadWriteAccess>;
117    type ByteIoPort = crate::io::IoPort<u8, ReadWriteAccess>;
118
119    #[ktest]
120    fn illegal_region() {
121        let io_port_a = IoPort::acquire(0xffff);
122        assert!(io_port_a.is_err());
123
124        type IllegalIoPort = crate::io::IoPort<[u8; 0x10008], ReadWriteAccess>;
125        let io_port_b = IllegalIoPort::acquire(0);
126        assert!(io_port_b.is_err());
127    }
128
129    #[ktest]
130    fn conflict_region() {
131        let io_port_a = IoPort::acquire(0x60);
132        assert!(io_port_a.is_ok());
133
134        // This allocation will fail because its range conflicts with `io_port_a`.
135        let io_port_b = IoPort::acquire(0x62);
136        assert!(io_port_b.is_err());
137
138        drop(io_port_a);
139
140        // After dropping `io_port_a`, conflicts no longer exist so the allocation will succeed.
141        let io_port_b = IoPort::acquire(0x62);
142        assert!(io_port_b.is_ok());
143    }
144
145    #[ktest]
146    fn overlapping_region() {
147        // Reference: <https://wiki.osdev.org/PCI#Configuration_Space_Access_Mechanism_#1>
148        let pci_data = IoPort::acquire_overlapping(0xcf8);
149        // Reference: <https://www.intel.com/Assets/PDF/datasheet/290562.pdf>
150        let rst_ctrl = ByteIoPort::acquire(0xcf9);
151        assert!(pci_data.is_ok());
152        assert!(rst_ctrl.is_ok());
153
154        let pci_data2 = IoPort::acquire_overlapping(0xcf8);
155        let rst_ctrl2 = ByteIoPort::acquire(0xcf9);
156        assert!(pci_data2.is_err());
157        assert!(rst_ctrl2.is_err());
158
159        drop(pci_data);
160        drop(rst_ctrl);
161
162        let rst_ctrl3 = ByteIoPort::acquire(0xcf9);
163        assert!(rst_ctrl3.is_ok());
164
165        let pci_data3 = IoPort::acquire_overlapping(0xcf8);
166        assert!(pci_data3.is_ok());
167    }
