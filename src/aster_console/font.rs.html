9/// A bitmap font.
10///
11/// Currently it's mainly used to draw texts on the framebuffer console.
12#[derive(Debug)]
13pub struct BitmapFont {
14    width: usize,
15    height: usize,
16    char_size: usize,
17    bitmap: Box<[u8]>,
18}
19
20impl BitmapFont {
21    /// Creates a new bitmap font.
22    ///
23    /// In a bitmap, a one in a given position indicates that the foreground pixel should be drawn
24    /// at that row and column.
25    ///
26    /// The font should contain a set of characters, starting from the NUL character. Each
27    /// character contains `height` rows and each row contains `width` columns. So a row contains
28    /// `width` bits and occupies `width.div_ceil(u8::BITS as _)` bytes. Note that the least
29    /// significant bit of a byte represents the smallest column number.
30    ///
31    /// # Panics
32    ///
33    /// This method will panic if the bitmap contains no characters or the last character in the
34    /// bitmap is incomplete.
35    pub fn new(width: usize, height: usize, bitmap: Box<[u8]>) -> Self {
36        let row_size = width.div_ceil(u8::BITS as usize);
37        let char_size = row_size.checked_mul(height).unwrap();
38
39        assert_ne!(bitmap.len(), 0);
40        assert_eq!(bitmap.len() % char_size, 0);
41
42        Self {
43            width,
44            height,
45            char_size,
46            bitmap,
47        }
48    }
49
50    /// Creates a new bitmap font that represents the basic 8x8 font.
51    ///
52    /// This is the default font for the framebuffer console.
53    pub fn new_basic8x8() -> Self {
54        const CHAR_COUNT: u32 = 0x7F;
55
56        const FONT_WIDTH: usize = 8;
57        const FONT_HEIGHT: usize = 8;
58
59        let bitmap = (0..CHAR_COUNT)
60            .flat_map(|ch| {
61                font8x8::BASIC_FONTS
62                    .get(char::from_u32(ch).unwrap())
63                    .unwrap()
64                    .into_iter()
65            })
66            .collect();
67
68        Self::new(FONT_WIDTH, FONT_HEIGHT, bitmap)
69    }
70
71    /// Creates a new bitmap font with the specified `vpitch` value.
72    ///
73    /// This method is similar to [`BitmapFont::new`], except that each character in the bitmap
74    /// contains `vpitch` rows. Note that the character height is still `height`, so the remaining
75    /// rows in the bitmap will be ignored.
76    ///
77    /// # Panics
78    ///
79    /// This method will panic if `height` is smaller than `vpitch`.
80    ///
81    /// Besides, this method will panic for the same reasons specified in [`BitmapFont::new`].
82    pub fn new_with_vpitch(
83        width: usize,
84        height: usize,
85        vpitch: usize,
86        mut bitmap: Vec<u8>,
87    ) -> Self {
88        if height == vpitch {
89            return Self::new(width, height, bitmap.into_boxed_slice());
90        }
91        assert!(height < vpitch);
92
93        let row_size = width.div_ceil(u8::BITS as usize);
94        let char_size_old = row_size.checked_mul(vpitch).unwrap();
95        let char_size_new = row_size.checked_mul(height).unwrap();
96
97        assert_ne!(bitmap.len(), 0);
98        assert_eq!(bitmap.len() % char_size_old, 0);
99
100        let mut old_pos = char_size_old;
101        let mut new_pos = char_size_new;
102        while old_pos < bitmap.len() {
103            bitmap.copy_within(old_pos..old_pos + char_size_new, new_pos);
104            old_pos += char_size_old;
105            new_pos += char_size_new;
106        }
107
108        bitmap.truncate(new_pos);
109
110        Self::new(width, height, bitmap.into_boxed_slice())
111    }
112
113    /// Returns the width of the font.
114    pub fn width(&self) -> usize {
115        self.width
116    }
117
118    /// Returns the height of the font.
119    pub fn height(&self) -> usize {
120        self.height
121    }
122
123    /// Returns the bitmap of the specified character in the font.
124    ///
125    /// This method will return [`None`] if the font does not contain a bitmap for the specified
126    /// character.
127    pub fn char(&self, ch: u8) -> Option<BitmapChar<'_>> {
128        let pos = (ch as usize) * self.char_size;
129        let data = self.bitmap.get(pos..pos + self.char_size)?;
130
131        Some(BitmapChar {
132            font: self,
133            char_data: data,
134        })
135    }
136}

138/// A bitmap of a character in a [`BitmapFont`].
139#[derive(Debug)]
140pub struct BitmapChar<'a> {
141    font: &'a BitmapFont,
142    char_data: &'a [u8],
143}
144
145impl<'a> BitmapChar<'a> {
146    /// Iterates over the rows of the character bitmap.
147    pub fn rows(&self) -> impl Iterator<Item = BitmapCharRow<'a>> {
148        let row_size = self.font.width.div_ceil(u8::BITS as usize);
149        self.char_data
150            .chunks_exact(row_size)
151            .map(|chunk| BitmapCharRow {
152                font: self.font,
153                row_data: chunk,
154            })
155    }
156}
157
158/// A bitmap of a row in a [`BitmapChar`].
159#[derive(Debug)]
160pub struct BitmapCharRow<'a> {
161    font: &'a BitmapFont,
162    row_data: &'a [u8],
163}
164
165impl BitmapCharRow<'_> {
166    /// Iterates over the bits (i.e., the columns) of the row bitmap.
167    pub fn bits(&self) -> impl Iterator<Item = bool> + '_ {
168        (0..self.font.width).map(|i| {
169            let nbyte = i / (u8::BITS as usize);
170            let nbit = i % (u8::BITS as usize);
171            self.row_data[nbyte] & (1 << nbit) != 0
172        })
173    }
174}
