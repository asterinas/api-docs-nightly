3//! A module represents Linux kernel device IDs.
4//!
5//! In Linux, each device is identified by a **major** number and a **minor** number.
6//! These two numbers together form a unique identifier for the device, which is used
7//! in device files (e.g., `/dev/sda`, `/dev/null`) and system calls like `mknod`.
8//!
9//! For more information about device ID allocation and usage in Linux, see:
10//! <https://www.kernel.org/doc/Documentation/admin-guide/devices.txt>
11
12#![no_std]
13#![deny(unsafe_code)]
14
15use aster_util::ranged_integer::{RangedU16, RangedU32};
16
17/// A device ID, embedding the major ID and minor ID.
18#[derive(Clone, Copy, Debug, PartialEq, Eq)]
19pub struct DeviceId(u32);
20
21impl DeviceId {
22    /// Creates a device ID from the major device number and the minor device number.
23    pub fn new(major: MajorId, minor: MinorId) -> Self {
24        Self(((major.get() as u32) << 20) | minor.get())
25    }
26
27    /// Returns the encoded `u32` value.
28    pub fn to_raw(&self) -> u32 {
29        self.0
30    }
31
32    /// Returns the major device number.
33    pub fn major(&self) -> MajorId {
34        MajorId::new((self.0 >> 20) as u16)
35    }
36
37    /// Returns the minor device number.
38    pub fn minor(&self) -> MinorId {
39        MinorId::new(self.0 & 0xf_ffff)
40    }
41}
42
43impl DeviceId {
44    /// Creates a device ID from the encoded `u64` value.
45    ///
46    /// Returns `None` if the major or minor device number is not falling in the valid range.
47    pub fn from_encoded_u64(raw: u64) -> Option<Self> {
48        let (major, minor) = decode_device_numbers(raw);
49        let major = MajorId::try_from(major as u16).ok()?;
50        let minor = MinorId::try_from(minor).ok()?;
51        Some(Self::new(major, minor))
52    }
53
54    /// Encodes the device ID as a `u64` value.
55    pub fn as_encoded_u64(&self) -> u64 {
56        encode_device_numbers(self.major().get() as u32, self.minor().get())
57    }
58}
59
60/// Decodes the major and minor numbers from the encoded `u64` value.
61///
62/// See [`DeviceId::as_encoded_u64`] for details about how to encode a device ID to a `u64` value.
63pub fn decode_device_numbers(raw: u64) -> (u32, u32) {
64    let major = ((raw >> 32) & 0xffff_f000 | (raw >> 8) & 0x0000_0fff) as u32;
65    let minor = ((raw >> 12) & 0xffff_ff00 | raw & 0x0000_00ff) as u32;
66    (major, minor)
67}
68
69/// Encodes the major and minor numbers as a `u64` value.
70///
71/// The lower 32 bits use the same encoding strategy as Linux. See the Linux implementation at:
72/// <https://github.com/torvalds/linux/blob/0ff41df1cb268fc69e703a08a57ee14ae967d0ca/include/linux/kdev_t.h#L39-L44>.
73///
74/// If the major or minor device number is too large, the additional bits will be recorded
75/// using the higher 32 bits. Note that as of 2025, the Linux kernel still has no support for
76/// 64-bit device IDs:
77/// <https://github.com/torvalds/linux/blob/0ff41df1cb268fc69e703a08a57ee14ae967d0ca/include/linux/types.h#L18>.
78/// So this encoding follows the implementation in glibc:
79/// <https://github.com/bminor/glibc/blob/632d895f3e5d98162f77b9c3c1da4ec19968b671/bits/sysmacros.h#L26-L34>.
80pub fn encode_device_numbers(major: u32, minor: u32) -> u64 {
81    let major = major as u64;
82    let minor = minor as u64;
83    ((major & 0xffff_f000) << 32)
84        | ((major & 0x0000_0fff) << 8)
85        | ((minor & 0xffff_ff00) << 12)
86        | (minor & 0x0000_00ff)
87}
88
89const MAX_MAJOR_ID: u16 = 0x0fff;
90
91const MAX_MINOR_ID: u32 = 0x000f_ffff;
92
93/// The major component of a device ID.
94///
95/// A major ID is a non-zero, 12-bit integer, thus falling in the range of `1..(1u16 << 12)`.
96///
97/// Reference: <https://elixir.bootlin.com/linux/v6.13/source/include/linux/kdev_t.h#L10>.
98pub type MajorId = RangedU16<1, MAX_MAJOR_ID>;
99
100/// The minor component of a device ID.
101///
102/// A minor ID is a 20-bit integer, thus falling in the range of `0..(1u32 << 20)`.
103///
104/// Reference: <https://elixir.bootlin.com/linux/v6.13/source/include/linux/kdev_t.h#L11>.
105pub type MinorId = RangedU32<0, MAX_MINOR_ID>;
