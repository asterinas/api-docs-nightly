21/// A text console rendered onto the framebuffer.
22pub struct FramebufferConsole {
23    callbacks: SpinLock<ConsoleCallbacks, LocalIrqDisabled>,
24    inner: SpinLock<(ConsoleState, EscapeFsm), LocalIrqDisabled>,
25}
26
27pub const CONSOLE_NAME: &str = "Framebuffer-Console";
28
29pub static FRAMEBUFFER_CONSOLE: Once<Arc<FramebufferConsole>> = Once::new();
30
31pub(crate) fn init() {
32    let Some(fb) = FRAMEBUFFER.get() else {
33        ostd::warn!("Framebuffer not initialized");
34        return;
35    };
36
37    FRAMEBUFFER_CONSOLE.call_once(|| Arc::new(FramebufferConsole::new(fb.clone())));
38}

40impl AnyConsoleDevice for FramebufferConsole {
41    fn send(&self, buf: &[u8]) {
42        let mut inner = self.inner.lock();
43        let (state, esc_fsm) = &mut *inner;
44
45        for byte in buf {
46            if esc_fsm.eat(*byte, state) {
47                // The character is part of an ANSI escape sequence.
48                continue;
49            }
50
51            if *byte == 0 {
52                // The character is a NUL character.
53                continue;
54            }
55
56            state.send_char(*byte);
57        }
58    }
59
60    fn register_callback(&self, callback: &'static ConsoleCallback) {
61        self.callbacks.lock().callbacks.push(callback);
62    }
63
64    fn set_font(&self, font: BitmapFont) -> Result<(), ConsoleSetFontError> {
65        self.inner.lock().0.set_font(font)
66    }
67
68    fn set_mode(&self, mode: ConsoleMode) -> bool {
69        self.inner.lock().0.set_mode(mode);
70        true
71    }
72
73    fn mode(&self) -> Option<ConsoleMode> {
74        Some(self.inner.lock().0.mode())
75    }
76
77    fn set_keyboard_mode(&self, mode: KeyboardMode) -> bool {
78        let mut callbacks = self.callbacks.lock();
79        match mode {
80            // TODO: Add support for Raw mode.
81            KeyboardMode::Raw => false,
82            KeyboardMode::Xlate
83            | KeyboardMode::MediumRaw
84            | KeyboardMode::Unicode
85            | KeyboardMode::Off => {
86                callbacks.keyboard_mode = mode;
87                true
88            }
89        }
90    }
91
92    fn keyboard_mode(&self) -> Option<KeyboardMode> {
93        Some(self.callbacks.lock().keyboard_mode())
94    }
95}

97impl FramebufferConsole {
98    /// Creates a new framebuffer console.
99    pub(self) fn new(framebuffer: Arc<FrameBuffer>) -> Self {
100        let callbacks = ConsoleCallbacks {
101            callbacks: Vec::new(),
102            keyboard_mode: KeyboardMode::Unicode,
103            // Linux default: REPEAT | META
104            // Reference: <https://elixir.bootlin.com/linux/v6.17.4/source/drivers/tty/vt/keyboard.c#L56>
105            keyboard_mode_flags: KeyboardModeFlags::REPEAT | KeyboardModeFlags::META,
106        };
107
108        let state = ConsoleState {
109            x_pos: 0,
110            y_pos: 0,
111            fg_color: Pixel::WHITE,
112            bg_color: Pixel::BLACK,
113            font: BitmapFont::new_basic8x8(),
114            is_output_enabled: true,
115
116            bytes: alloc::vec![0u8; framebuffer.io_mem().size()],
117            backend: framebuffer,
118        };
119
120        let esc_fsm = EscapeFsm::new();
121
122        Self {
123            callbacks: SpinLock::new(callbacks),
124            inner: SpinLock::new((state, esc_fsm)),
125        }
126    }
127
128    /// Locks the console callbacks.
129    pub fn lock_callbacks(&self) -> SpinLockGuard<'_, ConsoleCallbacks, LocalIrqDisabled> {
130        self.callbacks.lock()
131    }
132}

166#[derive(Debug)]
167struct ConsoleState {
168    x_pos: usize,
169    y_pos: usize,
170    fg_color: Pixel,
171    bg_color: Pixel,
172    font: BitmapFont,
173    /// Whether the output characters will be drawn in the framebuffer.
174    is_output_enabled: bool,
175
176    bytes: Vec<u8>,
177    backend: Arc<FrameBuffer>,
178}
179
180impl ConsoleState {
181    /// Sends a single character to be drawn on the framebuffer.
182    pub(self) fn send_char(&mut self, ch: u8) {
183        if ch == b'\n' {
184            self.newline();
185            return;
186        } else if ch == b'\r' {
187            self.carriage_return();
188            return;
189        } else if ch == b'\x08' {
190            self.backspace();
191            return;
192        }
193
194        if self.x_pos > self.backend.width() - self.font.width() {
195            self.newline();
196        }
197
198        self.draw_char(ch);
199
200        self.x_pos += self.font.width();
201    }
202
203    fn newline(&mut self) {
204        self.y_pos += self.font.height();
205        self.x_pos = 0;
206
207        if self.y_pos > self.backend.height() - self.font.height() {
208            self.shift_lines_up();
209        }
210    }
211
212    fn shift_lines_up(&mut self) {
213        let offset = self.backend.calc_offset(0, self.font.height()).as_usize();
214        self.bytes.copy_within(offset.., 0);
215        self.bytes[self.backend.io_mem().size() - offset..].fill(0);
216
217        if self.is_output_enabled {
218            self.backend.write_bytes_at(0, &self.bytes).unwrap();
219        }
220
221        self.y_pos -= self.font.height();
222    }
223
224    fn carriage_return(&mut self) {
225        self.x_pos = 0;
226    }
227
228    fn backspace(&mut self) {
229        if self.x_pos < self.font.width() {
230            // TODO: What should we do if we're at the beginning of the line?
231            return;
232        }
233
234        self.x_pos -= self.font.width();
235        self.draw_char(b' ');
236    }
237
238    fn draw_char(&mut self, ch: u8) {
239        let Some(font_ch) = self.font.char(ch) else {
240            return;
241        };
242
243        let fg_pixel = self.backend.render_pixel(self.fg_color);
244        let bg_pixel = self.backend.render_pixel(self.bg_color);
245
246        let pixel_size = fg_pixel.nbytes();
247
248        let mut offset = self.backend.calc_offset(self.x_pos, self.y_pos);
249
250        for row in font_ch.rows() {
251            let off_st = offset.as_usize();
252            let off_ed = off_st + pixel_size * self.font.width();
253            let render_buf = &mut self.bytes[off_st..off_ed];
254
255            // Write pixels to the console buffer.
256            let chunks = render_buf.chunks_exact_mut(pixel_size);
257            for (chunk, is_fg) in chunks.zip(row.bits()) {
258                let pixel = if is_fg { fg_pixel } else { bg_pixel };
259                chunk.copy_from_slice(pixel.as_slice());
260            }
261
262            // Write pixels to the framebuffer.
263            if self.is_output_enabled {
264                self.backend.write_bytes_at(off_st, render_buf).unwrap();
265            }
266
267            offset.y_add(1);
268        }
269    }
270
271    /// Sets the font for the framebuffer console.
272    pub(self) fn set_font(&mut self, font: BitmapFont) -> Result<(), ConsoleSetFontError> {
273        // Note that the font height cannot exceed the half the height of the framebuffer.
274        // Otherwise, `shift_lines_up` will underflow `x_pos`.
275        if font.width() > self.backend.width() || font.height() > self.backend.height() / 2 {
276            return Err(ConsoleSetFontError::InvalidFont);
277        }
278
279        self.font = font;
280
281        if self.y_pos > self.backend.height() - self.font.height() {
282            self.shift_lines_up();
283        }
284
285        Ok(())
286    }
287
288    /// Sets the console mode (text or graphics).
289    pub(self) fn set_mode(&mut self, mode: ConsoleMode) {
290        if mode == ConsoleMode::Graphics {
291            self.is_output_enabled = false;
292            return;
293        }
294
295        if self.is_output_enabled {
296            return;
297        }
298
299        // We're switching from the graphics mode back to the text mode. The characters need to be
300        // redrawn in the framebuffer.
301        self.is_output_enabled = true;
302        self.backend.write_bytes_at(0, &self.bytes).unwrap();
303    }
304
305    /// Gets the current console mode.
306    pub(self) fn mode(&self) -> ConsoleMode {
307        if self.is_output_enabled {
308            ConsoleMode::Text
309        } else {
310            ConsoleMode::Graphics
311        }
312    }
313
314    /// Fills a rectangular pixel region `[x0, x1) × [y0, y1)` with the given color.
315    ///
316    /// The caller must pass arguments that form a valid rectangle within the framebuffer, i.e.,
317    /// `x0 ≤ x1 ≤ w` and `y0 ≤ y1 ≤ h`, where `w` and `h` are the framebuffer width and height.
318    fn fill_rect_pixels(&mut self, x0: usize, y0: usize, x1: usize, y1: usize, color: Pixel) {
319        debug_assert!(x1 <= self.backend.width());
320        debug_assert!(y1 <= self.backend.height());
321
322        let rendered_pixel = self.backend.render_pixel(color);
323        let rendered_pixel_size = rendered_pixel.nbytes();
324        let row_bytes = (x1 - x0) * rendered_pixel_size;
325
326        for y in y0..y1 {
327            let off = self.backend.calc_offset(x0, y).as_usize();
328            let buf = &mut self.bytes[off..off + row_bytes];
329
330            // Write pixels to the console buffer.
331            for chunk in buf.chunks_exact_mut(rendered_pixel_size) {
332                chunk.copy_from_slice(rendered_pixel.as_slice());
333            }
334
335            // Write pixels to the framebuffer.
336            if self.is_output_enabled {
337                self.backend.write_bytes_at(off, buf).unwrap();
338            }
339        }
340    }
341
342    /// Calculates the pixel coordinates for the cursor cell.
343    fn cursor_cell_rect(&self) -> (usize, usize, usize, usize) {
344        let cx0 = self.x_pos;
345        let cx1 = cx0 + self.font.width();
346        // `min` is necessary if we are at the end of the line.
347        let cx1 = cx1.min(self.backend.width());
348
349        let cy0 = self.y_pos;
350        let cy1 = cy0 + self.font.height();
351        // This will always hold because we will scroll if it is no longer true after a new line or
352        // font change.
353        debug_assert!(cy1 <= self.backend.height());
354
355        (cx0, cy0, cx1, cy1)
356    }
357}

359impl EscapeOp for ConsoleState {
360    fn set_cursor(&mut self, x: usize, y: usize) {
361        let max_x = self.backend.width() / self.font.width() - 1;
362        let max_y = self.backend.height() / self.font.height() - 1;
363
364        // Note that if the Y (or X) position is too large, the cursor will move to the last line
365        // (or the line end).
366        self.x_pos = x.min(max_x) * self.font.width();
367        self.y_pos = y.min(max_y) * self.font.height();
368    }
369
370    fn set_fg_color(&mut self, val: Pixel) {
371        self.fg_color = val;
372    }
373
374    fn set_bg_color(&mut self, val: Pixel) {
375        self.bg_color = val;
376    }
377
378    fn erase_in_display(&mut self, mode: EraseInDisplay) {
379        let bg = self.bg_color;
380        let w = self.backend.width();
381        let h = self.backend.height();
382
383        let (cx0, cy0, cx1, cy1) = self.cursor_cell_rect();
384
385        match mode {
386            EraseInDisplay::CursorToEnd => {
387                // Clear from the cursor to the end of the line, within the cursor row.
388                self.fill_rect_pixels(cx0, cy0, w, cy1, bg);
389
390                // Clear all rows below the cursor row.
391                if cy1 < h {
392                    self.fill_rect_pixels(0, cy1, w, h, bg);
393                }
394            }
395            EraseInDisplay::CursorToBeginning => {
396                // Clear all rows above the cursor row.
397                if cy0 > 0 {
398                    self.fill_rect_pixels(0, 0, w, cy0, bg);
399                }
400
401                // Clear from the start of the line to the cursor, within the cursor row.
402                self.fill_rect_pixels(0, cy0, cx1, cy1, bg);
403            }
404            EraseInDisplay::EntireScreen | EraseInDisplay::EntireScreenAndScrollback => {
405                // Clear the entire screen.
406                self.fill_rect_pixels(0, 0, w, h, bg);
407            }
408        }
409    }
