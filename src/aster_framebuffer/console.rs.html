21/// A text console rendered onto the framebuffer.
22pub struct FramebufferConsole {
23    callbacks: SpinLock<ConsoleCallbacks, LocalIrqDisabled>,
24    inner: SpinLock<(ConsoleState, EscapeFsm), LocalIrqDisabled>,
25}
26
27pub const CONSOLE_NAME: &str = "Framebuffer-Console";
28
29pub static FRAMEBUFFER_CONSOLE: Once<Arc<FramebufferConsole>> = Once::new();
30
31pub(crate) fn init() {
32    let Some(fb) = FRAMEBUFFER.get() else {
33        log::warn!("Framebuffer not initialized");
34        return;
35    };
36
37    FRAMEBUFFER_CONSOLE.call_once(|| Arc::new(FramebufferConsole::new(fb.clone())));
38}

40impl AnyConsoleDevice for FramebufferConsole {
41    fn send(&self, buf: &[u8]) {
42        let mut inner = self.inner.lock();
43        let (state, esc_fsm) = &mut *inner;
44
45        for byte in buf {
46            if esc_fsm.eat(*byte, state) {
47                // The character is part of an ANSI escape sequence.
48                continue;
49            }
50
51            if *byte == 0 {
52                // The character is a NUL character.
53                continue;
54            }
55
56            state.send_char(*byte);
57        }
58    }
59
60    fn register_callback(&self, callback: &'static ConsoleCallback) {
61        self.callbacks.lock().callbacks.push(callback);
62    }
63
64    fn set_font(&self, font: BitmapFont) -> Result<(), ConsoleSetFontError> {
65        self.inner.lock().0.set_font(font)
66    }
67
68    fn set_mode(&self, mode: ConsoleMode) -> bool {
69        self.inner.lock().0.set_mode(mode);
70        true
71    }
72
73    fn mode(&self) -> Option<ConsoleMode> {
74        Some(self.inner.lock().0.mode())
75    }
76
77    fn set_keyboard_mode(&self, mode: KeyboardMode) -> bool {
78        match mode {
79            KeyboardMode::Xlate => self.callbacks.lock().is_input_enabled = true,
80            KeyboardMode::Off => self.callbacks.lock().is_input_enabled = false,
81            _ => return false,
82        }
83        true
84    }
85
86    fn keyboard_mode(&self) -> Option<KeyboardMode> {
87        if self.callbacks.lock().is_input_enabled {
88            Some(KeyboardMode::Xlate)
89        } else {
90            Some(KeyboardMode::Off)
91        }
92    }
93}

95impl FramebufferConsole {
96    /// Creates a new framebuffer console.
97    pub(self) fn new(framebuffer: Arc<FrameBuffer>) -> Self {
98        let callbacks = ConsoleCallbacks {
99            callbacks: Vec::new(),
100            is_input_enabled: true,
101        };
102
103        let state = ConsoleState {
104            x_pos: 0,
105            y_pos: 0,
106            fg_color: Pixel::WHITE,
107            bg_color: Pixel::BLACK,
108            font: BitmapFont::new_basic8x8(),
109            is_output_enabled: true,
110
111            bytes: alloc::vec![0u8; framebuffer.io_mem().size()],
112            backend: framebuffer,
113        };
114
115        let esc_fsm = EscapeFsm::new();
116
117        Self {
118            callbacks: SpinLock::new(callbacks),
119            inner: SpinLock::new((state, esc_fsm)),
120        }
121    }
122
123    /// Triggers the registered input callbacks with the given data.
124    pub(crate) fn trigger_input_callbacks(&self, bytes: &[u8]) {
125        let callbacks = self.callbacks.lock();
126        if !callbacks.is_input_enabled {
127            return;
128        }
129
130        let reader = VmReader::from(bytes);
131        for callback in callbacks.callbacks.iter() {
132            callback(reader.clone());
133        }
134    }
135}

149#[derive(Debug)]
150struct ConsoleState {
151    x_pos: usize,
152    y_pos: usize,
153    fg_color: Pixel,
154    bg_color: Pixel,
155    font: BitmapFont,
156    /// Whether the output characters will be drawn in the framebuffer.
157    is_output_enabled: bool,
158
159    bytes: Vec<u8>,
160    backend: Arc<FrameBuffer>,
161}
162
163impl ConsoleState {
164    /// Sends a single character to be drawn on the framebuffer.
165    pub(self) fn send_char(&mut self, ch: u8) {
166        if ch == b'\n' {
167            self.newline();
168            return;
169        } else if ch == b'\r' {
170            self.carriage_return();
171            return;
172        } else if ch == b'\x08' {
173            self.backspace();
174            return;
175        }
176
177        if self.x_pos > self.backend.width() - self.font.width() {
178            self.newline();
179        }
180
181        self.draw_char(ch);
182
183        self.x_pos += self.font.width();
184    }
185
186    fn newline(&mut self) {
187        self.y_pos += self.font.height();
188        self.x_pos = 0;
189
190        if self.y_pos > self.backend.height() - self.font.height() {
191            self.shift_lines_up();
192        }
193    }
194
195    fn shift_lines_up(&mut self) {
196        let offset = self.backend.calc_offset(0, self.font.height()).as_usize();
197        self.bytes.copy_within(offset.., 0);
198        self.bytes[self.backend.io_mem().size() - offset..].fill(0);
199
200        if self.is_output_enabled {
201            self.backend.write_bytes_at(0, &self.bytes).unwrap();
202        }
203
204        self.y_pos -= self.font.height();
205    }
206
207    fn carriage_return(&mut self) {
208        self.x_pos = 0;
209    }
210
211    fn backspace(&mut self) {
212        if self.x_pos < self.font.width() {
213            // TODO: What should we do if we're at the beginning of the line?
214            return;
215        }
216
217        self.x_pos -= self.font.width();
218        self.draw_char(b' ');
219    }
220
221    fn draw_char(&mut self, ch: u8) {
222        let Some(font_ch) = self.font.char(ch) else {
223            return;
224        };
225
226        let fg_pixel = self.backend.render_pixel(self.fg_color);
227        let bg_pixel = self.backend.render_pixel(self.bg_color);
228
229        let pixel_size = fg_pixel.nbytes();
230
231        let mut offset = self.backend.calc_offset(self.x_pos, self.y_pos);
232
233        for row in font_ch.rows() {
234            let off_st = offset.as_usize();
235            let off_ed = off_st + pixel_size * self.font.width();
236            let render_buf = &mut self.bytes[off_st..off_ed];
237
238            // Write pixels to the console buffer.
239            let chunks = render_buf.chunks_exact_mut(pixel_size);
240            for (chunk, is_fg) in chunks.zip(row.bits()) {
241                let pixel = if is_fg { fg_pixel } else { bg_pixel };
242                chunk.copy_from_slice(pixel.as_slice());
243            }
244
245            // Write pixels to the framebuffer.
246            if self.is_output_enabled {
247                self.backend.write_bytes_at(off_st, render_buf).unwrap();
248            }
249
250            offset.y_add(1);
251        }
252    }
253
254    /// Sets the font for the framebuffer console.
255    pub(self) fn set_font(&mut self, font: BitmapFont) -> Result<(), ConsoleSetFontError> {
256        // Note that the font height cannot exceed the half the height of the framebuffer.
257        // Otherwise, `shift_lines_up` will underflow `x_pos`.
258        if font.width() > self.backend.width() || font.height() > self.backend.height() / 2 {
259            return Err(ConsoleSetFontError::InvalidFont);
260        }
261
262        self.font = font;
263
264        if self.y_pos > self.backend.height() - self.font.height() {
265            self.shift_lines_up();
266        }
267
268        Ok(())
269    }
270
271    /// Sets the console mode (text or graphics).
272    pub(self) fn set_mode(&mut self, mode: ConsoleMode) {
273        if mode == ConsoleMode::Graphics {
274            self.is_output_enabled = false;
275            return;
276        }
277
278        if self.is_output_enabled {
279            return;
280        }
281
282        // We're switching from the graphics mode back to the text mode. The characters need to be
283        // redrawn in the framebuffer.
284        self.is_output_enabled = true;
285        self.backend.write_bytes_at(0, &self.bytes).unwrap();
286    }
287
288    /// Gets the current console mode.
289    pub(self) fn mode(&self) -> ConsoleMode {
290        if self.is_output_enabled {
291            ConsoleMode::Text
292        } else {
293            ConsoleMode::Graphics
294        }
295    }
296}

298impl EscapeOp for ConsoleState {
299    fn set_cursor(&mut self, x: usize, y: usize) {
300        let max_x = self.backend.width() / self.font.width() - 1;
301        let max_y = self.backend.height() / self.font.height() - 1;
302
303        // Note that if the Y (or X) position is too large, the cursor will move to the last line
304        // (or the line end).
305        self.x_pos = x.min(max_x) * self.font.width();
306        self.y_pos = y.min(max_y) * self.font.height();
307    }
308
309    fn set_fg_color(&mut self, val: Pixel) {
310        self.fg_color = val;
311    }
312
313    fn set_bg_color(&mut self, val: Pixel) {
314        self.bg_color = val;
315    }
