30/// IRQ line for i8042 mouse.
31static IRQ_LINE: Once<MappedIrqLine> = Once::new();

33/// Registered device instance for event submission.
34static REGISTERED_DEVICE: Once<RegisteredInputDevice> = Once::new();

36/// ISA interrupt number for i8042 mouse.
37const ISA_INTR_NUM: u8 = 12;
38
39/// Mouse packet state machine.
40static PACKET_STATE: SpinLock<PacketState, LocalIrqDisabled> = SpinLock::new(PacketState::new());

42pub(super) fn init(controller: &mut I8042Controller) -> Result<(), I8042ControllerError> {
43    let mut init_ctx = InitCtx(controller);
44
45    let device_id = init_ctx
46        .reset()?
47        .ok_or(I8042ControllerError::DeviceUnknown)?;
48    log::info!("PS/2 mouse device ID: 0x{:02X}", device_id);
49
50    // Determine the mouse's type.
51    let mut mouse_type =
52        MouseType::from_device_id(device_id).ok_or(I8042ControllerError::DeviceUnknown)?;
53    if mouse_type == MouseType::Standard && init_ctx.enable_intellimouse().is_ok() {
54        // Query the device ID again.
55        let new_device_id = init_ctx.get_device_id()?;
56        log::info!("PS/2 mouse upgraded device ID: 0x{:02X}", new_device_id);
57        mouse_type =
58            MouseType::from_device_id(new_device_id).ok_or(I8042ControllerError::DeviceUnknown)?;
59    }
60
61    // Enable data reporting.
62    init_ctx.enable_data_reporting()?;
63
64    let mut irq_line = IrqLine::alloc()
65        .and_then(|irq_line| {
66            IRQ_CHIP
67                .get()
68                .unwrap()
69                .map_isa_pin_to(irq_line, ISA_INTR_NUM)
70        })
71        .map_err(|_| I8042ControllerError::DeviceAllocIrqFailed)?;
72    irq_line.on_active(handle_mouse_input);
73    IRQ_LINE.call_once(|| irq_line);
74
75    // Create and register the i8042 mouse device.
76    let mouse_device = Arc::new(I8042Mouse::new());
77    let registered_device = aster_input::register_device(mouse_device);
78    REGISTERED_DEVICE.call_once(|| registered_device);
79
80    // Configure the mouse's type based on its device ID.
81    let mut state = PACKET_STATE.lock();
82    state.set_mouse_type(mouse_type);
83
84    Ok(())
85}

87struct InitCtx<'a>(&'a mut I8042Controller);
88
89impl InitCtx<'_> {
90    fn get_device_id(&mut self) -> Result<u8, I8042ControllerError> {
91        let mut buf = [0u8; cmd::GetDeviceId::RES_LEN];
92        self.command::<cmd::GetDeviceId>(&[], &mut buf)?;
93        Ok(buf[0])
94    }
95
96    fn enable_intellimouse(&mut self) -> Result<(), I8042ControllerError> {
97        const SAMPLE_RATE_200: u8 = 200;
98        const SAMPLE_RATE_100: u8 = 100;
99        const SAMPLE_RATE_80: u8 = 80;
100
101        // Set the sample rate to 200, then to 100, and finally to 80.
102        // Reference: <https://wiki.osdev.org/Mouse_Input#Init/Detection_Command_Sequences>
103        self.set_sample_rate(SAMPLE_RATE_200)?;
104        self.set_sample_rate(SAMPLE_RATE_100)?;
105        self.set_sample_rate(SAMPLE_RATE_80)?;
106        Ok(())
107    }
108
109    fn set_sample_rate(&mut self, rate: u8) -> Result<(), I8042ControllerError> {
110        let mut buf = [0u8; cmd::SetSampleRate::RES_LEN];
111        self.command::<cmd::SetSampleRate>(&[rate], &mut buf)?;
112        Ok(())
113    }
114
115    fn enable_data_reporting(&mut self) -> Result<(), I8042ControllerError> {
116        let mut buf = [0u8; cmd::EnableDataReporting::RES_LEN];
117        self.command::<cmd::EnableDataReporting>(&[], &mut buf)?;
118        Ok(())
119    }
120}
121
122impl CommandCtx for InitCtx<'_> {
123    fn controller(&mut self) -> &mut I8042Controller {
124        self.0
125    }
126
127    fn write_to_port(&mut self, data: u8) -> Result<(), I8042ControllerError> {
128        self.0.write_to_second_port(data)
129    }

142#[derive(Debug)]
143struct I8042Mouse {
144    name: String,
145    phys: String,
146    uniq: String,
147    id: InputId,
148    capability: InputCapability,
149}
150
151impl I8042Mouse {
152    fn new() -> Self {
153        let mut capability = InputCapability::new();
154
155        // Mouse supports key events and relative movement events.
156        capability.set_supported_event_type(aster_input::event_type_codes::EventTypes::KEY);
157        capability.set_supported_event_type(aster_input::event_type_codes::EventTypes::REL);
158        capability.set_supported_event_type(aster_input::event_type_codes::EventTypes::SYN);
159
160        // Add mouse buttons.
161        capability.set_supported_key(KeyCode::BtnLeft);
162        capability.set_supported_key(KeyCode::BtnRight);
163        capability.set_supported_key(KeyCode::BtnMiddle);
164
165        // Add relative axes for movement.
166        capability.set_supported_relative_axis(RelCode::X);
167        capability.set_supported_relative_axis(RelCode::Y);
168        capability.set_supported_relative_axis(RelCode::Wheel);
169
170        Self {
171            // Standard name for i8042 PS/2 mouse devices.
172            name: "i8042_mouse".to_string(),
173
174            // Physical path describing the device's connection topology
175            // isa0060: ISA bus port 0x60 (i8042 data port)
176            // serio1: Serial I/O device 1 (second PS/2 port)
177            // input0: Input device 0 (first mouse device on this controller)
178            phys: "isa0060/serio1/input0".to_string(),
179
180            // Unique identifier - empty string because traditional i8042 mice
181            // don't have unique hardware identifiers
182            uniq: "".to_string(),
183
184            // Device ID with standard values for i8042 mice
185            // BUS_I8042 (0x11): PS/2 interface bus type
186            // vendor (0x0002): Generic vendor ID for standard mice
187            // product (0x0001): Generic product ID for standard mice
188            // version (0x0001): Version 1.0 - standard PS/2 mouse protocol
189            id: InputId::new(InputId::BUS_I8042, 0x0002, 0x0001, 0x0001),
190
191            capability,
192        }
193    }
194}

218fn handle_mouse_input(_trap_frame: &TrapFrame) {
219    let Some(controller) = I8042_CONTROLLER.get() else {
220        return;
221    };
222    let Some(data) = controller.lock().receive_data() else {
223        log::warn!("PS/2 mouse has no input data");
224        return;
225    };
226
227    let mut packet_state = PACKET_STATE.lock();
228    if let Some(events) = packet_state.process_byte(data)
229        && let Some(registered_device) = REGISTERED_DEVICE.get()
230    {
231        registered_device.submit_events(&events);
232    }
233}

235#[derive(Debug, Clone, Copy, PartialEq)]
236enum MouseType {
237    Standard,
238    IntelliMouse,
239}
240
241impl MouseType {
242    const PACKET_LEN_STANDARD: usize = 3;
243    const PACKET_LEN_INTELLIMOUSE: usize = 4;
244    const PACKET_LEN_MAX: usize = 4;
245
246    fn from_device_id(device_id: u8) -> Option<Self> {
247        // Reference: <https://wiki.osdev.org/Mouse_Input#MouseID_Byte>
248        const DEVICE_ID_STANDARD_MOUSE: u8 = 0x00;
249        const DEVICE_ID_INTELLIMOUSE: u8 = 0x03;
250        const DEVICE_ID_INTELLIMOUSE_EXPLORER: u8 = 0x04;
251
252        match device_id {
253            DEVICE_ID_STANDARD_MOUSE => Some(MouseType::Standard),
254            DEVICE_ID_INTELLIMOUSE | DEVICE_ID_INTELLIMOUSE_EXPLORER => {
255                Some(MouseType::IntelliMouse)
256            }
257            _ => None,
258        }
259    }
260
261    fn packet_len(&self) -> usize {
262        match self {
263            MouseType::Standard => Self::PACKET_LEN_STANDARD,
264            MouseType::IntelliMouse => Self::PACKET_LEN_INTELLIMOUSE,
265        }
266    }
267}

269/// Mouse packet state machine for handling 3-byte or 4-byte PS/2 mouse packets.
270struct PacketState {
271    packet: [u8; MouseType::PACKET_LEN_MAX],
272    byte_index: usize,
273    mouse_type: MouseType,
274    prev_left: bool,
275    prev_right: bool,
276    prev_middle: bool,
277}
278
279impl PacketState {
280    const fn new() -> Self {
281        Self {
282            packet: [0; MouseType::PACKET_LEN_MAX],
283            byte_index: 0,
284            mouse_type: MouseType::Standard,
285            prev_left: false,
286            prev_right: false,
287            prev_middle: false,
288        }
289    }
290
291    fn set_mouse_type(&mut self, mouse_type: MouseType) {
292        self.mouse_type = mouse_type;
293        self.reset();
294    }
295
296    /// Processes a mouse data byte and returns input events if a complete packet is ready.
297    fn process_byte(&mut self, data: u8) -> Option<Vec<InputEvent>> {
298        const PACKET_SYNC_MASK: u8 = 0x08;
299
300        // The first byte must have bit 3 set (always 1 in PS/2 mouse packets).
301        if self.byte_index == 0 && (data & PACKET_SYNC_MASK) == 0 {
302            return None;
303        }
304
305        // Collect the byte into the packet.
306        self.packet[self.byte_index] = data;
307        self.byte_index += 1;
308        if self.byte_index < self.mouse_type.packet_len() {
309            return None;
310        }
311
312        // A complete packet has been received.
313        let events = self.parse_packet();
314        self.reset();
315        Some(events)
316    }
317
318    /// Parses a complete PS/2 mouse packet and returns input events.
319    ///
320    /// Reference: <https://wiki.osdev.org/Mouse_Input#Format_of_First_3_Packet_Bytes>.
321    fn parse_packet(&mut self) -> Vec<InputEvent> {
322        const BUTTON_LEFT_MASK: u8 = 0x01;
323        const BUTTON_RIGHT_MASK: u8 = 0x02;
324        const BUTTON_MIDDLE_MASK: u8 = 0x04;
325        const X_SIGN_MASK: u8 = 0x10;
326        const Y_SIGN_MASK: u8 = 0x20;
327        const OVERFLOWED_MASK: u8 = 0xC0;
328        const INDEX_STATUS: usize = 0;
329        const INDEX_X: usize = 1;
330        const INDEX_Y: usize = 2;
331        const INDEX_WHEEL: usize = 3;
332
333        // Currently, this method can generate at most 7 events. Don't forget to update this when
334        // modifying the logic below!
335        let mut events = Vec::with_capacity(7);
336
337        let status = self.packet[INDEX_STATUS];
338        if (status & OVERFLOWED_MASK) != 0 {
339            log::warn!("PS/2 mouse packet overflowed");
340            return events;
341        }
342
343        let x_delta = add_sign_bit(self.packet[INDEX_X], (status & X_SIGN_MASK) != 0);
344        let y_delta = add_sign_bit(self.packet[INDEX_Y], (status & Y_SIGN_MASK) != 0);
345
346        let left_button = (status & BUTTON_LEFT_MASK) != 0;
347        let right_button = (status & BUTTON_RIGHT_MASK) != 0;
348        let middle_button = (status & BUTTON_MIDDLE_MASK) != 0;
349
350        let prev_left = self.prev_left;
351        let prev_right = self.prev_right;
352        let prev_middle = self.prev_middle;
353        self.prev_left = left_button;
354        self.prev_right = right_button;
355        self.prev_middle = middle_button;
356
357        if left_button != prev_left {
358            events.push(InputEvent::from_key_and_status(
359                KeyCode::BtnLeft,
360                if left_button {
361                    KeyStatus::Pressed
362                } else {
363                    KeyStatus::Released
364                },
365            ));
366        }
367
368        if right_button != prev_right {
369            events.push(InputEvent::from_key_and_status(
370                KeyCode::BtnRight,
371                if right_button {
372                    KeyStatus::Pressed
373                } else {
374                    KeyStatus::Released
375                },
376            ));
377        }
378
379        if middle_button != prev_middle {
380            events.push(InputEvent::from_key_and_status(
381                KeyCode::BtnMiddle,
382                if middle_button {
383                    KeyStatus::Pressed
384                } else {
385                    KeyStatus::Released
386                },
387            ));
388        }
389
390        if x_delta != 0 {
391            events.push(InputEvent::from_relative_move(RelCode::X, x_delta));
392        }
393
394        if y_delta != 0 {
395            events.push(InputEvent::from_relative_move(RelCode::Y, -y_delta));
396        }
397
398        if self.mouse_type == MouseType::IntelliMouse {
399            let z_delta = self.packet[INDEX_WHEEL] as i8;
400            if z_delta != 0 {
401                events.push(InputEvent::from_relative_move(
402                    RelCode::Wheel,
403                    -(z_delta as i32),
404                ));
405            }
406        }
407
408        // Add a sync event to indicate end of this input report.
409        if !events.is_empty() {
410            events.push(InputEvent::from_sync_event(SynEvent::Report));
411        }
412
413        events
414    }
415
416    /// Resets the packet state machine to wait for the next packet.
417    fn reset(&mut self) {
418        self.byte_index = 0;
419    }
420}

422/// Extends a byte to a signed 32-bit integer using the sign bit from the status byte.
423fn add_sign_bit(byte: u8, sign_bit: bool) -> i32 {
424    let mut byte = byte as u32;
425    if sign_bit {
426        byte |= 0xFFFFFF00;
427    }
428    byte as i32
