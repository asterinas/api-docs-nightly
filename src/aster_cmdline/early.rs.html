3//! A set of the kernel command line parameters to be handled early on by OSTD.
4//!
5//! Most parameters are recognized and handled by the Asterinas kernel,
6//! but a few must be done early on by OSTD.
7//! For example,
8//! `earlycon` specifies why an early console (most commonly based on UART) should be initialized.
9//! The early console is the only viable sink for log messages
10//! generated by OSTD in the early stage,
11//! before it handles the control of CPU to the Asterinas kernel.
12//!
13//! OSTD allows injecting a parser for these early command line parameters
14//! via [`ostd::early_cmdline_parser`].
15//! The injected parser is invoked so early
16//! even the heap allocator is not ready.
17//! As such, all parser logic in this module has to be written as const expressions.
18
19use ostd::{boot::EarlyCmdline, log::LevelFilter};
20
21/// Kernel command-line keys consumed by [`early_cmdline_parser`].
22///
23/// These are stripped during dispatch so they are not forwarded to init.
24const EARLY_PARAMS: &[&str] = &["earlycon", "loglevel"];
25
26/// Returns whether `key` is handled only by the early parser.
27pub(super) fn is_early_param(key: &str) -> bool {
28    EARLY_PARAMS.contains(&key)
29}
30
31/// Registers the kernel's early command-line parser with OSTD.
32///
33/// Scans space-separated tokens in `cmdline` and returns [`EarlyCmdline`] for
34/// OSTD to configure the early serial console and log filter.
35///
36/// # Recognized tokens
37///
38/// - `earlycon` — enables the early UART console (exact token match).
39/// - `loglevel=N` — sets the OSTD log filter (`N` in `0..=8`, see [`LevelFilter`]).
40/// - `loglevel=LEVEL` — also accepts textual levels such as `error` or `debug`.
41/// - `--` — stops parsing; all following tokens are for the init process.
42///
43/// Double quotes (`"`) protect spaces inside one token (Linux-compatible).
44/// Other tokens are ignored. When `loglevel` is absent, the default is `8`
45/// ([`LevelFilter::Debug`]). When `earlycon` is absent, the early console stays disabled.
46#[ostd::early_cmdline_parser]
47const fn early_cmdline_parser(cmdline: &str) -> EarlyCmdline {
48    let mut has_early_console = false;
49    let mut log_level = LevelFilter::Debug;
50
51    let bytes: &[u8] = cmdline.as_bytes();
52    let mut index = 0;
53    while index < bytes.len() {
54        // Skip leading separators before scanning the next token.
55        while index < bytes.len() && bytes[index].is_ascii_whitespace() {
56            index += 1;
57        }
58        if index >= bytes.len() {
59            break;
60        }
61
62        let start: usize = index;
63        let mut inside_quotes = false;
64        // Linux-style tokenization: whitespace splits only outside double quotes.
65        while index < bytes.len() {
66            let byte = bytes[index];
67            if byte == b'"' {
68                inside_quotes = !inside_quotes;
69            } else if byte.is_ascii_whitespace() && !inside_quotes {
70                break;
71            }
72            index += 1;
73        }
74
75        let token = strip_linux_double_quotes(bytes, start, index);
76
77        // Only early-only keys are consumed here; other tokens are ignored.
78        if matches!(token, b"--") {
79            // `--` terminates kernel-side parsing; remaining tokens belong to init.
80            break;
81        } else if matches!(token, b"earlycon") {
82            has_early_console = true;
83        } else if let Some((key, value)) = token.split_at_checked(9)
84            && matches!(key, b"loglevel=")
85            && let Some(parsed_log_level) = parse_loglevel_at(value)
86        {
87            log_level = parsed_log_level;
88        }
89    }
90
91    EarlyCmdline {
92        log_level,
93        has_early_console,
94    }
95}
96
97/// Strips Linux-style surrounding quotes from a token or value.
98const fn strip_linux_double_quotes(bytes: &[u8], start: usize, end: usize) -> &[u8] {
99    let (new_start, new_end) = if start < end && bytes[start] == b'"' {
100        let new_start = start + 1;
101        if new_start < end && bytes[end - 1] == b'"' {
102            (new_start, end - 1)
103        } else {
104            (new_start, end)
105        }
106    } else {
107        (start, end)
108    };
109    bytes.split_at(new_start).1.split_at(new_end - new_start).0
110}

112/// Parses a `loglevel` value.
113///
114/// Accepts `0..=8` or lowercase level names. Returns `None` for malformed values
115/// so they do not override earlier valid settings.
116const fn parse_loglevel_at(bytes: &[u8]) -> Option<LevelFilter> {
117    let bytes = strip_linux_double_quotes(bytes, 0, bytes.len());
118
119    if bytes.is_empty() {
120        return None;
121    }
122
123    if bytes[0].is_ascii_digit() {
124        if let Ok(result) = u8::from_ascii_radix(bytes, 10)
125            && result <= 8
126        {
127            return Some(LevelFilter::from_u8(result));
128        }
129        return None;
130    }
131
132    parse_loglevel_name_at(bytes)
133}
134
135/// Parses lowercase textual loglevel names.
136const fn parse_loglevel_name_at(bytes: &[u8]) -> Option<LevelFilter> {
137    match bytes {
138        b"off" => Some(LevelFilter::Off),
139        b"emerg" => Some(LevelFilter::Emerg),
140        b"alert" => Some(LevelFilter::Alert),
141        b"crit" => Some(LevelFilter::Crit),
142        b"error" | b"err" => Some(LevelFilter::Error),
143        b"warning" | b"warn" => Some(LevelFilter::Warning),
144        b"notice" => Some(LevelFilter::Notice),
145        b"info" => Some(LevelFilter::Info),
146        b"debug" => Some(LevelFilter::Debug),
147        _ => None,
148    }
149}

151#[cfg(ktest)]
152mod tests {
153    use ostd::prelude::*;
154
155    use super::*;
156
157    #[ktest]
158    fn is_early_param_recognizes_early_keys() {
159        assert!(is_early_param("earlycon"));
160        assert!(is_early_param("loglevel"));
161        assert!(!is_early_param("log_level"));
162        assert!(!is_early_param("unknown"));
163    }
164
165    #[ktest]
166    fn early_cmdline_parser_defaults() {
167        let result = early_cmdline_parser("");
168        assert_eq!(result.log_level, LevelFilter::Debug);
169        assert!(!result.has_early_console);
170    }
171
172    #[ktest]
173    fn early_cmdline_parser_earlycon() {
174        assert!(early_cmdline_parser("earlycon").has_early_console);
175        assert!(early_cmdline_parser("\"earlycon\"").has_early_console);
176        assert!(!early_cmdline_parser("\"  earlycon \"").has_early_console);
177        assert!(!early_cmdline_parser("earlyconxxxx").has_early_console);
178    }
179
180    #[ktest]
181    fn early_cmdline_parser_loglevel() {
182        assert_eq!(
183            early_cmdline_parser("loglevel=0").log_level,
184            LevelFilter::Off
185        );
186        assert_eq!(
187            early_cmdline_parser("loglevel=\"0\"").log_level,
188            LevelFilter::Off
189        );
190        assert_eq!(
191            early_cmdline_parser("\"loglevel=0\"").log_level,
192            LevelFilter::Off
193        );
194        assert_eq!(
195            early_cmdline_parser("loglevel=3").log_level,
196            LevelFilter::Crit
197        );
198        assert_eq!(
199            early_cmdline_parser("loglevel=3x").log_level,
200            LevelFilter::Debug
201        );
202        assert_eq!(
203            early_cmdline_parser("loglevel=9").log_level,
204            LevelFilter::Debug
205        );
206        assert_eq!(
207            early_cmdline_parser("loglevel=error").log_level,
208            LevelFilter::Error
209        );
210        assert_eq!(
211            early_cmdline_parser("loglevel=err").log_level,
212            LevelFilter::Error
213        );
214        assert_eq!(
215            early_cmdline_parser("loglevel=off").log_level,
216            LevelFilter::Off
217        );
218        assert_eq!(
219            early_cmdline_parser("loglevel=warn").log_level,
220            LevelFilter::Warning
221        );
222        assert_eq!(
223            early_cmdline_parser("loglevel=Warn").log_level,
224            LevelFilter::Debug
225        );
226    }
227
228    #[ktest]
229    fn early_cmdline_parser_last_loglevel_wins() {
230        assert_eq!(
231            early_cmdline_parser("loglevel=4 loglevel=0").log_level,
232            LevelFilter::Off
233        );
234    }
235
236    #[ktest]
237    fn early_cmdline_parser_invalid_loglevel_does_not_override() {
238        assert_eq!(
239            early_cmdline_parser("loglevel=0 loglevel=3x").log_level,
240            LevelFilter::Off
241        );
242        assert_eq!(
243            early_cmdline_parser("loglevel=error loglevel=").log_level,
244            LevelFilter::Error
245        );
246        assert_eq!(
247            early_cmdline_parser("loglevel=warn loglevel=9").log_level,
248            LevelFilter::Warning
249        );
250    }
251
252    #[ktest]
253    fn early_cmdline_parser_stops_at_double_dash() {
254        let result = early_cmdline_parser("loglevel=0 -- loglevel=4 earlycon");
255        assert_eq!(result.log_level, LevelFilter::Off);
256        assert!(!result.has_early_console);
257    }
258
259    #[ktest]
260    fn early_cmdline_parser_quotes_protect_spaces() {
261        let result = early_cmdline_parser("loglevel=0 \"ignored -- token\" earlycon");
262        assert_eq!(result.log_level, LevelFilter::Off);
263        assert!(result.has_early_console);
264    }
