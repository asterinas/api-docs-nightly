3//! Kernel command-line parameter dispatch and init forwarding.
4//!
5//! Dispatches kernel command-line parameters to registered handlers and
6//! forwards unrecognized parameters to the init process.
7
8use alloc::{
9    collections::BTreeMap,
10    ffi::CString,
11    string::{String, ToString},
12    vec::Vec,
13};
14
15use component::{ComponentInitError, init_component};
16use spin::Once;

18/// The arguments passed to the init process, extracted from the kernel command line.
19#[derive(Debug, PartialEq)]
20pub struct InitprocArgs {
21    argv: Vec<CString>,
22    envp: Vec<CString>,
23}
24
25impl InitprocArgs {
26    /// Returns the argument vector (`argv`) of the init process.
27    pub fn argv(&self) -> &[CString] {
28        &self.argv
29    }
30
31    /// Returns the environment vector (`envp`) of the init process.
32    pub fn envp(&self) -> &[CString] {
33        &self.envp
34    }
35}
36
37#[doc(hidden)]
38#[derive(Debug)]
39pub struct KernelParam {
40    name: &'static str,
41    setup_fn: fn(occurrences: &[Option<&str>]),
42    early: bool,
43}
44
45impl KernelParam {
46    #[doc(hidden)]
47    pub const fn new(
48        name: &'static str,
49        setup_fn: fn(occurrences: &[Option<&str>]),
50        early: bool,
51    ) -> KernelParam {
52        if Self::contains_hyphen(name) {
53            panic!("kernel param registration must not contain '-' (use '_')");
54        }
55        KernelParam {
56            name,
57            setup_fn,
58            early,
59        }
60    }
61
62    const fn contains_hyphen(s: &'static str) -> bool {
63        let bytes = s.as_bytes();
64        let mut i = 0;
65        while i < bytes.len() {
66            if bytes[i] == b'-' {
67                return true;
68            }
69            i += 1;
70        }
71        false
72    }
73}

75inventory::collect!(KernelParam);

77pub static INIT_PROC_ARGS: Once<InitprocArgs> = Once::new();
78
79#[init_component]
80fn init() -> Result<(), ComponentInitError> {
81    INIT_PROC_ARGS.call_once(|| dispatch_params(ostd::boot::boot_info().kernel_cmdline.as_str()));
82
83    Ok(())
84}

86// Splits the command line string by spaces but preserve
87// ones that are protected by double quotes(`"`).
88fn split_arg(input: &str) -> impl Iterator<Item = &str> {
89    let mut inside_quotes = false;
90
91    input.split(move |c: char| {
92        if c == '"' {
93            inside_quotes = !inside_quotes;
94        }
95
96        !inside_quotes && c.is_whitespace()
97    })
98}

100fn dispatch_params(cmdline: &str) -> InitprocArgs {
101    let mut result: InitprocArgs = InitprocArgs {
102        argv: Vec::new(),
103        envp: Vec::new(),
104    };
105
106    let mut kcmdline_end = false;
107
108    // Step 1: Build lookup from registered param name to handler.
109    let mut registry = BTreeMap::new();
110    for p in inventory::iter::<KernelParam> {
111        if let Some(prev) = registry.insert(p.name, p) {
112            ostd::warn!(
113                "duplicate kernel parameter '{}' registered; keeping last",
114                prev.name
115            );
116        }
117    }
118
119    // Step 2: Tokenize the kernel command line and group recognized param by normalized name.
120    let mut grouped: BTreeMap<String, Vec<Option<&str>>> = BTreeMap::new();
121    for arg in split_arg(cmdline) {
122        // Everything after "--" goes to init.
123        if kcmdline_end {
124            result.argv.push(CString::new(arg).unwrap());
125            continue;
126        }
127        if arg == "--" {
128            kcmdline_end = true;
129            continue;
130        }
131
132        let (key, value) = match arg.find('=') {
133            Some(pos) => (&arg[..pos], Some(&arg[pos + 1..])),
134            None => (arg, None),
135        };
136        // Normalize hyphens to underscores (Linux compatibility)
137        let normalized = key.replace('-', "_");
138
139        if registry.contains_key(normalized.as_str()) {
140            // Group by normalized name
141            grouped.entry(normalized).or_default().push(value);
142        } else {
143            // Unknown parameter: forward to init
144            if key.contains('.') {
145                // The entry contains a dot, which is treated as a module argument.
146                // Unrecognized module arguments are ignored.
147                continue;
148            } else if let Some(value) = value {
149                // If the entry is not recognized, it is passed to the init process.
150                // Pattern 'entry=value' is treated as the init environment.
151                let envp_entry = CString::new(key.to_string() + "=" + value).unwrap();
152                result.envp.push(envp_entry);
153            } else {
154                // If the entry is not recognized, it is passed to the init process.
155                // Pattern 'entry' without value is treated as the init argument.
156                let argv_entry = CString::new(key.to_string()).unwrap();
157                result.argv.push(argv_entry);
158            }
159        }
160    }
161
162    // Step 3: Dispatch each group to its handler.
163    let (early_params, params): (Vec<_>, Vec<_>) = grouped
164        .iter()
165        .filter_map(|(name, occurrences)| registry.get(name.as_str()).map(|p| (*p, occurrences)))
166        .partition(|(p, _)| p.early);
167
168    early_params
169        .into_iter()
170        .chain(params)
171        .for_each(|(param, occurrences)| (param.setup_fn)(occurrences));
172
173    result
174}

176#[cfg(ktest)]
177mod tests {
178    use ostd::prelude::*;
179
180    use super::*;
181
182    #[ktest]
183    fn unknown_kv_forwarded_to_init_env() {
184        let args = dispatch_params("unknown_key=1");
185        assert!(args.envp().iter().any(|e| e.to_bytes() == b"unknown_key=1"));
186    }
187
188    #[ktest]
189    fn dotted_unknown_param_not_forwarded() {
190        let args = dispatch_params("some_module.flag");
191        assert!(args.argv().is_empty());
192        assert!(args.envp().is_empty());
193    }
194
195    #[ktest]
196    fn repeated_unknown_kv_and_arg_forwarding() {
197        let args = dispatch_params("unknown_key=1 unknown_key unknown_key=2");
198
199        assert_eq!(args.envp.len(), 2);
200        assert_eq!(args.envp[0].to_bytes(), b"unknown_key=1");
201        assert_eq!(args.envp[1].to_bytes(), b"unknown_key=2");
202
203        assert_eq!(args.argv.len(), 1);
204        assert_eq!(args.argv[0].to_bytes(), b"unknown_key");
205    }
206
207    #[ktest]
208    fn registered_params_not_forwarded() {
209        static TEST_VARIABLE: Once<u32> = Once::new();
210        static TEST_VARIABLE_REPEATED: Once<Vec<String>> = Once::new();
211
212        crate::define_kv_param!("log_level", TEST_VARIABLE);
213        crate::define_repeatable_kv_param!("console", TEST_VARIABLE_REPEATED);
214
215        let args = dispatch_params("log_level=4 console=ttyS0 console=ttyS1");
216        assert!(args.argv().is_empty());
217        assert!(args.envp().is_empty());
218    }
219}
