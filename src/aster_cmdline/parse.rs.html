3//! Kernel command-line parameter parsing and storage.
4//!
5//! This module defines traits for parsing different types of parameters and
6//! provides implementations for common cases (e.g., `FromStr` types, flags).
7
8use alloc::vec::Vec;
9use core::{
10    str::FromStr,
11    sync::atomic::{AtomicBool, AtomicU8, AtomicU32, Ordering},
12};
13
14use spin::Once;
15
16/// Trait for types that can store a parsed parameter value.
17pub trait ParamStorage: Sync + 'static {
18    type Value;
19    fn store_param(&self, value: Self::Value);
20}
21
22impl<T: Send + Sync + 'static> ParamStorage for Once<T> {
23    type Value = T;
24    fn store_param(&self, value: T) {
25        self.call_once(|| value);
26    }
27}
28
29impl ParamStorage for AtomicU8 {
30    type Value = u8;
31    fn store_param(&self, value: u8) {
32        self.store(value, Ordering::Relaxed);
33    }
34}
35
36impl ParamStorage for AtomicU32 {
37    type Value = u32;
38    fn store_param(&self, value: u32) {
39        self.store(value, Ordering::Relaxed);
40    }
41}
42
43impl ParamStorage for AtomicBool {
44    type Value = bool;
45    fn store_param(&self, value: bool) {
46        self.store(value, Ordering::Relaxed);
47    }
48}
49
50/// Errors while parsing kernel command line parameters.
51#[derive(Clone, Copy, Debug, Eq, PartialEq)]
52pub enum ParamError {
53    InvalidValue,
54}
55
56/// Parses a single-value key-value parameter (e.g., log_level=3).
57///
58/// If the parameter appears multiple times, the framework applies last-wins
59/// semantics (matching Linux behavior) and passes only the final value.
60pub trait ParseParamValue: Sized {
61    fn parse_param(value: &str) -> Result<Self, ParamError>;
62}
63
64/// Parses a repeatable key-value parameter (e.g., "console=ttyS0 console=ttyS1").
65///
66/// The framework collects all occurrences and passes the full slice.
67pub trait ParseRepeatableParamValue: Sized {
68    fn parse_all(values: &[&str]) -> Result<Self, ParamError>;
69}
70
71/// Parses a flag parameter (e.g., "ro", "debug", "nokaslr").
72///
73/// Flags may optionally accept a value (e.g., "debug=1" in Linux).
74pub trait ParseFlag: Sized {
75    /// Prases the flag with an optional value.
76    ///
77    /// If the flag is specified as "flag=v", then `value` is `Some(v)`;
78    /// if the flag is bare, then `value` is `None`.
79    fn parse_flag(value: Option<&str>) -> Result<Self, ParamError>;
80}
81
82/// Any `FromStr` type can be a single-value parameter.
83impl<T: FromStr> ParseParamValue for T {
84    fn parse_param(value: &str) -> Result<Self, ParamError> {
85        value.parse().map_err(|_| ParamError::InvalidValue)
86    }
87}
88
89/// A `Vec<T>` where `T: FromStr` can be a repeatable parameter.
90impl<T: FromStr> ParseRepeatableParamValue for Vec<T> {
91    fn parse_all(values: &[&str]) -> Result<Self, ParamError> {
92        values
93            .iter()
94            .map(|v| v.parse().map_err(|_| ParamError::InvalidValue))
95            .collect()
96    }
97}
98
99/// `bool` as a flag: a bare flag means `true`;
100/// flags with typical values such as `1`/`0`, `on`/`off` are also accepted.
101impl ParseFlag for bool {
102    fn parse_flag(value: Option<&str>) -> Result<Self, ParamError> {
103        match value {
104            None | Some("1") | Some("on") | Some("yes") | Some("true") => Ok(true),
105            Some("0") | Some("off") | Some("no") | Some("false") => Ok(false),
106            _ => Err(ParamError::InvalidValue),
107        }
108    }
109}

111#[doc(hidden)]
112pub fn setup_kv_param<S: ParamStorage>(storage: &S, occurrences: &[Option<&str>], name: &str)
113where
114    S::Value: ParseParamValue,
115{
116    let Some(last) = occurrences.last() else {
117        return;
118    };
119    match last {
120        Some(value) => match S::Value::parse_param(value) {
121            Ok(v) => storage.store_param(v),
122            Err(_) => ostd::warn!("invalid value for kernel parameter '{}'", name),
123        },
124        None => ostd::warn!("kernel parameter '{}' requires a value", name),
125    }
126}
127
128#[doc(hidden)]
129pub fn setup_repeatable_kv_param<S: ParamStorage>(
130    storage: &S,
131    occurrences: &[Option<&str>],
132    name: &str,
133) where
134    S::Value: ParseRepeatableParamValue,
135{
136    let values: Vec<&str> = occurrences.iter().filter_map(|val| *val).collect();
137    if values.is_empty() {
138        ostd::warn!("repeatable parameter '{}' requires values", name);
139        return;
140    }
141    match S::Value::parse_all(&values) {
142        Ok(v) => storage.store_param(v),
143        Err(_) => ostd::warn!("invalid value for kernel parameter '{}'", name),
144    }
145}
146
147#[doc(hidden)]
148pub fn setup_flag_param<S: ParamStorage>(storage: &S, occurrences: &[Option<&str>], name: &str)
149where
150    S::Value: ParseFlag,
151{
152    let last = match occurrences.last() {
153        Some(val) => *val,
154        None => return,
155    };
156    match S::Value::parse_flag(last) {
157        Ok(v) => storage.store_param(v),
158        Err(_) => ostd::warn!("invalid value for flag '{}'", name),
159    }
160}
