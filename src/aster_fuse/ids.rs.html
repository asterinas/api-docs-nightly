5/// A FUSE request identifier (`unique`) exchanged on the protocol.
6///
7/// The client assigns one `FuseUnique` to each request and the server copies it
8/// into the matching reply. The value `0` is reserved for unsolicited
9/// notifications rather than ordinary request/reply matching.
10#[repr(transparent)]
11#[derive(Clone, Copy, Debug, Default, Eq, PartialEq, Pod)]
12pub struct FuseUnique(u64);
13
14impl FuseUnique {
15    pub const fn new(raw: u64) -> Self {
16        Self(raw)
17    }
18
19    pub const fn as_u64(self) -> u64 {
20        self.0
21    }
22}

39/// A FUSE inode identifier (`nodeid`) exchanged on the protocol.
40///
41/// A `FuseNodeId` identifies one server-side inode object. It appears in FUSE
42/// request headers and in lookup-like replies such as [`crate::EntryReply`].
43///
44/// For a cached VFS inode, this value is an immutable identity binding: if
45/// lookup revalidation later resolves the same pathname to a different
46/// `FuseNodeId`, the old cached inode must be treated as stale and replaced
47/// instead of being retargeted in place.
48#[repr(transparent)]
49#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq, Pod)]
50pub struct FuseNodeId(u64);
51
52impl FuseNodeId {
53    pub const fn new(raw: u64) -> Self {
54        Self(raw)
55    }
56
57    pub const fn as_u64(self) -> u64 {
58        self.0
59    }
60}

82/// Client-side mirror of the server's nlookup for an inode.
83///
84/// In the FUSE protocol, every reply that contains an [`crate::EntryReply`]
85/// increments the server-side nlookup by one. This includes `FUSE_LOOKUP`,
86/// `FUSE_CREATE`, `FUSE_MKDIR`, `FUSE_MKNOD`, and `FUSE_LINK`. This type
87/// tracks the same count on the client side so that the accumulated value can
88/// be sent back via `FUSE_FORGET` when the inode is dropped.
89#[repr(transparent)]
90pub struct LookupCount(AtomicU64);
91
92impl LookupCount {
93    /// Creates a counter seeded with the lookup reference carried by an
94    /// [`crate::EntryReply`] reply.
95    pub const fn initial() -> Self {
96        Self(AtomicU64::new(1))
97    }
98
99    /// Acquires one lookup reference after receiving an `EntryReply` reply.
100    pub fn acquire(&self) {
101        self.0.fetch_add(1, Ordering::AcqRel);
102    }
103
104    /// Returns the value to pass to `FUSE_FORGET` and clears the counter.
105    pub fn drain(&self) -> u64 {
106        self.0.swap(0, Ordering::AcqRel)
107    }
108}
