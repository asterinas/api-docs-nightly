7/// A raw directory entry in a `FUSE_READDIR` reply.
8#[repr(C)]
9#[derive(Clone, Copy, Debug, Pod)]
10pub struct Dirent {
11    /// Inode number of the directory entry.
12    ino: u64,
13    /// Offset cookie for continuing directory iteration.
14    off: u64,
15    /// Length of the entry name in bytes.
16    namelen: u32,
17    /// POSIX directory entry type.
18    typ: u32,
19}
20
21impl Dirent {
22    /// Returns the inode number of the directory entry.
23    pub fn ino(&self) -> u64 {
24        self.ino
25    }
26
27    /// Returns the offset cookie for continuing directory iteration.
28    pub fn off(&self) -> u64 {
29        self.off
30    }
31
32    /// Returns the length of the entry name in bytes.
33    pub fn namelen(&self) -> u32 {
34        self.namelen
35    }
36
37    /// Returns the POSIX directory entry type.
38    pub fn typ(&self) -> u32 {
39        self.typ
40    }
41}

43/// An opaque offset cookie for continuing directory iteration.
44///
45/// The offset cookie may differ between different servers.
46/// The FUSE protocol treats it as an opaque value, and the client should not interpret its value.
47//
48// FIXME: we currently treat the offset as a simple byte offset in the directory stream,
49// but some servers may use a different format for the offset cookie.
50#[repr(transparent)]
51#[derive(Clone, Copy, Debug, Eq, PartialEq)]
52pub struct DirOffset(u64);
53
54impl DirOffset {
55    /// Creates an offset cookie from its wire value.
56    pub const fn new(value: u64) -> Self {
57        Self(value)
58    }
59
60    /// Returns the wire value of this offset cookie.
61    pub const fn get(self) -> u64 {
62        self.0
63    }
64}

66/// A directory entry decoded from a `FUSE_READDIR` reply.
67#[derive(Debug, Clone)]
68pub struct FuseDirEntry {
69    ino: u64,
70    offset: DirOffset,
71    type_: DirentType,
72    name: String,
73}
74
75impl FuseDirEntry {
76    pub fn new(ino: u64, offset: DirOffset, type_: DirentType, name: String) -> Self {
77        Self {
78            ino,
79            offset,
80            type_,
81            name,
82        }
83    }
84
85    /// Returns the inode number of this directory entry.
86    pub fn ino(&self) -> u64 {
87        self.ino
88    }
89
90    /// Returns the offset cookie for continuing directory iteration.
91    pub fn offset(&self) -> DirOffset {
92        self.offset
93    }
94
95    /// Returns the POSIX directory entry type.
96    pub fn type_(&self) -> DirentType {
97        self.type_
98    }
99
100    /// Returns the entry name.
101    pub fn name(&self) -> &str {
102        &self.name
103    }
104}

106/// POSIX `d_type` values carried in [`Dirent::typ`].
107///
108/// See <https://www.man7.org/linux/man-pages/man3/readdir.3.html>.
109#[repr(u32)]
110#[derive(Clone, Copy, Debug, Eq, PartialEq, TryFromInt)]
111pub enum DirentType {
112    Unknown = 0,
113    Fifo = 1,
114    Char = 2,
115    Dir = 4,
116    Block = 6,
117    Regular = 8,
118    Link = 10,
119    Sock = 12,
120}
