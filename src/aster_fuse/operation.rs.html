12/// A FUSE protocol operation with typed request and reply bodies.
13///
14/// Each implementer represents one request/reply pair defined by the FUSE protocol.
15pub trait FuseOperation {
16    /// Describes the successful reply produced by that operation.
17    type Output;
18
19    /// Returns the opcode identifying this operation's wire format.
20    fn opcode(&self) -> FuseOpcode;
21
22    /// Returns the request body length, excluding the `ReqHeader`.
23    fn body_len(&self) -> usize {
24        0
25    }
26
27    /// Writes the request body bytes into the transport buffer.
28    ///
29    /// The writer is positioned immediately after the `ReqHeader`.
30    fn write_body(&mut self, _writer: &mut VmWriter<'_, Infallible>) -> FuseResult<()> {
31        Ok(())
32    }
33
34    /// Returns the reply shape expected for this operation.
35    fn reply_expectation(&self) -> ReplyExpectation;
36
37    /// Parses the reply payload into [`Self::Output`].
38    ///
39    /// The reader is positioned at the start of the typed reply payload, after
40    /// the `ReplyHeader`. Implementations must use `payload_len` to bound the
41    /// bytes accepted from the server.
42    fn parse_reply(
43        payload_len: usize,
44        reader: &mut VmReader<'_, Infallible>,
45    ) -> FuseResult<Self::Output>;
46}
47
48/// A reply shape expected for a FUSE operation.
49#[derive(Clone, Copy, Debug, Eq, PartialEq)]
50pub enum ReplyExpectation {
51    /// No reply is expected for a one-way request.
52    None,
53    /// A reply contains only a `ReplyHeader` and no payload bytes.
54    HeaderOnly,
55    /// A reply contains a `ReplyHeader` and a payload up to the given payload bytes.
56    Payload(NonZeroUsize),
57}
58
59impl ReplyExpectation {
60    /// Returns a payload-bearing reply expectation.
61    ///
62    /// A zero payload size is returned as [`Self::HeaderOnly`] so
63    /// [`Self::Payload`] cannot encode the same wire shape.
64    pub fn payload(payload_size: usize) -> Self {
65        let Some(payload_size) = NonZeroUsize::new(payload_size) else {
66            return Self::HeaderOnly;
67        };
68        Self::Payload(payload_size)
69    }
70}
