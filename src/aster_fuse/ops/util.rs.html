3//! Shared encoding and decoding helpers for FUSE operation implementations.
4
5use ostd::mm::{Infallible, VmReader, VmWriter};
6use ostd_pod::Pod;
7
8use crate::{FuseError, FuseResult};
9
10/// The trailing NULL byte required by FUSE request names.
11pub(super) const NAME_TERMINATOR: &[u8] = &[0];
12
13pub(super) fn name_body_len(prefix_len: usize, name: &str) -> usize {
14    prefix_len
15        .saturating_add(name.len())
16        .saturating_add(NAME_TERMINATOR.len())
17}
18
19pub(super) fn read_bytes(reader: &mut VmReader<'_, Infallible>, dst: &mut [u8]) -> FuseResult<()> {
20    if reader.remain() < dst.len() {
21        return Err(FuseError::BufferTooSmall);
22    }
23    reader.read(&mut VmWriter::from(dst));
24    Ok(())
25}
26
27pub(super) fn read_payload<T: Pod>(
28    payload_len: usize,
29    reader: &mut VmReader<'_, Infallible>,
30) -> FuseResult<T> {
31    if payload_len < size_of::<T>() {
32        return Err(FuseError::MalformedResponse);
33    }
34
35    Ok(reader.read_val().unwrap())
36}
