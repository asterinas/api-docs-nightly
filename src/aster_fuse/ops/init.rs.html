3//! `FUSE_INIT` negotiates the protocol version and capabilities for a FUSE
4//! connection.
5//!
6//! The request body contains [`InitReq`] with the client-supported version,
7//! readahead size, and feature flags. The reply body contains [`InitReply`] with
8//! the server-selected version, limits, and negotiated flags.
9
10use bitflags::bitflags;
11use ostd::mm::{Infallible, VmReader, VmWriter};
12use ostd_pod::{FromZeros, IntoBytes};
13
14use super::util::read_bytes;
15use crate::{FuseError, FuseOpcode, FuseOperation, FuseResult, ReplyExpectation};
16
17/// The minimum valid `FUSE_INIT` reply payload size for legacy servers.
18///
19/// Reference: <https://elixir.bootlin.com/linux/v6.18/source/include/uapi/linux/fuse.h#L911>
20const FUSE_COMPAT_INIT_OUT_SIZE: usize = 8;
21
22#[repr(C)]
23#[derive(Clone, Copy, Debug, Pod)]
24pub struct InitReq {
25    /// Major version of the FUSE protocol supported by the client.
26    major: u32,
27    /// Minor version of the FUSE protocol supported by the client.
28    minor: u32,
29    /// Maximum readahead size requested by the client.
30    max_readahead: u32,
31    /// Lower 32 bits of supported client capabilities.
32    flags: FuseInitFlags,
33    /// The following fields are extensions.
34    ///
35    /// Upper 32 bits of supported client capabilities.
36    flags2: FuseInitFlags2,
37    unused: [u32; 11],
38}
39
40impl InitReq {
41    pub const fn new(
42        major: u32,
43        minor: u32,
44        max_readahead: u32,
45        flags: FuseInitFlags,
46        flags2: FuseInitFlags2,
47    ) -> Self {
48        Self {
49            major,
50            minor,
51            max_readahead,
52            flags,
53            flags2,
54            unused: [0; 11],
55        }
56    }
57}
58
59#[repr(C)]
60#[derive(Clone, Copy, Debug, Pod)]
61pub struct InitReply {
62    /// Major version of the FUSE protocol selected by the server.
63    major: u32,
64    /// Minor version of the FUSE protocol selected by the server.
65    minor: u32,
66    /// Maximum readahead size accepted by the server.
67    max_readahead: u32,
68    /// Lower 32 bits of negotiated capabilities.
69    flags: FuseInitFlags,
70    /// Maximum number of background requests.
71    max_background: u16,
72    /// Background request threshold for congestion.
73    congestion_threshold: u16,
74    /// Maximum write size accepted by the server.
75    max_write: u32,
76    /// Timestamp granularity in nanoseconds.
77    time_gran: u32,
78    /// Maximum number of pages in a request.
79    max_pages: u16,
80    /// Mapping alignment requirement as a power-of-two page count.
81    map_alignment: u16,
82    /// Upper 32 bits of negotiated capabilities.
83    flags2: FuseInitFlags2,
84    /// Maximum stack depth for passthrough operations.
85    max_stack_depth: u32,
86    /// Request timeout in seconds.
87    request_timeout: u16,
88    unused: [u16; 11],
89}
90

138bitflags! {
139    /// FUSE capability and feature flags exchanged in `FUSE_INIT`.
140    ///
141    /// The client sends its supported set in [`InitReq::new`]; the server
142    /// responds with the subset it also supports in [`InitReply::flags`].
143    #[repr(C)]
144    #[derive(Pod)]
145    pub struct FuseInitFlags: u32 {
146        /// Supports asynchronous reads.
147        const ASYNC_READ          = 1 << 0;
148        /// Supports POSIX byte-range locks.
149        const POSIX_LOCKS         = 1 << 1;
150        /// Uses file-handle based operations.
151        const FILE_OPS            = 1 << 2;
152        /// Supports atomic `O_TRUNC` handling during open.
153        const ATOMIC_O_TRUNC      = 1 << 3;
154        /// Supports stable inode numbers for export.
155        const EXPORT_SUPPORT      = 1 << 4;
156        /// Supports writes larger than 4 KiB.
157        const BIG_WRITES          = 1 << 5;
158        /// Preserves mode bits instead of applying the process umask.
159        const DONT_MASK           = 1 << 6;
160        /// Supports splice-based writes.
161        const SPLICE_WRITE        = 1 << 7;
162        /// Supports splice move optimization.
163        const SPLICE_MOVE         = 1 << 8;
164        /// Supports splice-based reads.
165        const SPLICE_READ         = 1 << 9;
166        /// Supports BSD-style flock locks.
167        const FLOCK_LOCKS         = 1 << 10;
168        /// Supports ioctl requests on directories.
169        const HAS_IOCTL_DIR       = 1 << 11;
170        /// Invalidates cached file data automatically on attribute changes.
171        const AUTO_INVAL_DATA     = 1 << 12;
172        /// Supports `FUSE_READDIRPLUS`.
173        const DO_READDIRPLUS      = 1 << 13;
174        /// Allows the server to choose when to use `FUSE_READDIRPLUS`.
175        const READDIRPLUS_AUTO    = 1 << 14;
176        /// Supports asynchronous direct I/O.
177        const ASYNC_DIO           = 1 << 15;
178        /// Supports writeback caching.
179        const WRITEBACK_CACHE     = 1 << 16;
180        /// Allows `ENOSYS` from `FUSE_OPEN` to mean open is unsupported.
181        const NO_OPEN_SUPPORT     = 1 << 17;
182        /// Supports parallel directory operations.
183        const PARALLEL_DIROPS     = 1 << 18;
184        /// Lets the server clear privilege bits after writes and truncates.
185        const HANDLE_KILLPRIV     = 1 << 19;
186        /// Supports POSIX ACLs.
187        const POSIX_ACL           = 1 << 20;
188        /// Supports returning an error from abort handling.
189        const ABORT_ERROR         = 1 << 21;
190        /// Supports the `max_pages` field in [`InitReply`].
191        const MAX_PAGES           = 1 << 22;
192        /// Supports caching symbolic-link targets.
193        const CACHE_SYMLINKS      = 1 << 23;
194        /// Allows `ENOSYS` from `FUSE_OPENDIR` to mean opendir is unsupported.
195        const NO_OPENDIR_SUPPORT  = 1 << 24;
196        /// Supports explicit file-data invalidation.
197        const EXPLICIT_INVAL_DATA = 1 << 25;
198        /// Supports the `map_alignment` field in [`InitReply`].
199        const MAP_ALIGNMENT       = 1 << 26;
200        /// Supports submounts.
201        const SUBMOUNTS           = 1 << 27;
202        /// Supports the version-2 privilege-bit clearing protocol.
203        const HANDLE_KILLPRIV_V2  = 1 << 28;
204        /// Supports extended setxattr requests.
205        const SETXATTR_EXT        = 1 << 29;
206        /// Supports extended `FUSE_INIT` fields.
207        const INIT_EXT            = 1 << 30;
208        /// Reserved by the FUSE protocol.
209        const INIT_RESERVED       = 1 << 31;
210    }
211}

213bitflags! {
214    /// Upper FUSE capability and feature flags exchanged in `FUSE_INIT`.
215    ///
216    /// These flags encode bits 32 through 63 of the protocol capability set
217    /// after shifting them down into the `flags2` field.
218    #[repr(C)]
219    #[derive(Pod)]
220    pub struct FuseInitFlags2: u32 {
221        /// Supports security context information.
222        const SECURITY_CTX             = 1 << 0;
223        /// Supports inode DAX information.
224        const HAS_INODE_DAX           = 1 << 1;
225        /// Supports supplementary groups during create.
226        const CREATE_SUPP_GROUP       = 1 << 2;
227        /// Supports expire-only invalidation.
228        const HAS_EXPIRE_ONLY         = 1 << 3;
229        /// Allows memory maps on direct-I/O files.
230        const DIRECT_IO_ALLOW_MMAP    = 1 << 4;
231        /// Supports passthrough operations.
232        const PASSTHROUGH             = 1 << 5;
233        /// Supports disabling export support.
234        const NO_EXPORT_SUPPORT       = 1 << 6;
235        /// Supports resending requests.
236        const HAS_RESEND              = 1 << 7;
237        /// Allows ID-mapped mounts.
238        const ALLOW_IDMAP             = 1 << 8;
239        /// Supports FUSE over io_uring.
240        const OVER_IO_URING           = 1 << 9;
241        /// Supports request timeouts.
242        const REQUEST_TIMEOUT         = 1 << 10;
243    }
244}
245
246pub struct InitOperation {
247    init_req: InitReq,
248}
249
250impl InitOperation {
251    pub fn new(init_req: InitReq) -> Self {
252        Self { init_req }
253    }
254}
255
256impl FuseOperation for InitOperation {
257    type Output = InitReply;
258
259    fn opcode(&self) -> FuseOpcode {
260        FuseOpcode::Init
261    }
262
263    fn body_len(&self) -> usize {
264        size_of::<InitReq>()
265    }
266
267    fn write_body(&mut self, writer: &mut VmWriter<'_, Infallible>) -> FuseResult<()> {
268        writer
269            .write_val(&self.init_req)
270            .map_err(|_| FuseError::BufferTooSmall)
271    }
272
273    fn reply_expectation(&self) -> ReplyExpectation {
274        ReplyExpectation::payload(size_of::<InitReply>())
275    }
276
277    fn parse_reply(
278        payload_len: usize,
279        reader: &mut VmReader<'_, Infallible>,
280    ) -> FuseResult<Self::Output> {
281        if payload_len < FUSE_COMPAT_INIT_OUT_SIZE {
282            return Err(FuseError::MalformedResponse);
283        }
284
285        let mut init_reply = InitReply::new_zeroed();
286        let read_len = core::cmp::min(payload_len, size_of::<InitReply>());
287        read_bytes(reader, &mut init_reply.as_mut_bytes()[..read_len])?;
288
289        Ok(init_reply)
290    }
291}
