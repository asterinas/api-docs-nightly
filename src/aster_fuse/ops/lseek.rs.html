3//! `FUSE_LSEEK` computes a file offset for an open file handle.
4//!
5//! The request body contains [`LseekReq`] with the handle, base offset, and
6//! seek mode. The reply body contains [`LseekReply`].
7
8use ostd::mm::{Infallible, VmReader, VmWriter};
9
10use super::util::read_payload;
11use crate::{FuseError, FuseFileHandle, FuseOpcode, FuseOperation, FuseResult, ReplyExpectation};
12
13#[repr(C)]
14#[derive(Clone, Copy, Debug, Pod)]
15pub struct LseekReq {
16    fh: FuseFileHandle,
17    /// Base offset for the seek operation.
18    offset: i64,
19    /// Determines how `offset` is interpreted when computing the new file position.
20    whence: u32,
21    padding: u32,
22}
23
24impl LseekReq {
25    pub const fn new(fh: FuseFileHandle, offset: i64, whence: u32) -> Self {
26        Self {
27            fh,
28            offset,
29            whence,
30            padding: 0,
31        }
32    }
33}
34
35#[repr(C)]
36#[derive(Clone, Copy, Debug, Pod)]
37pub struct LseekReply {
38    offset: i64,
39}
40
41impl LseekReply {
42    /// Returns the file offset computed by the server.
43    pub fn offset(&self) -> i64 {
44        self.offset
45    }
46}
47
48pub struct LseekOperation {
49    lseek_req: LseekReq,
50}
51
52impl LseekOperation {
53    pub fn new(lseek_req: LseekReq) -> Self {
54        Self { lseek_req }
55    }
56}
57
58impl FuseOperation for LseekOperation {
59    type Output = LseekReply;
60
61    fn opcode(&self) -> FuseOpcode {
62        FuseOpcode::Lseek
63    }
64
65    fn body_len(&self) -> usize {
66        size_of::<LseekReq>()
67    }
68
69    fn write_body(&mut self, writer: &mut VmWriter<'_, Infallible>) -> FuseResult<()> {
70        writer
71            .write_val(&self.lseek_req)
72            .map_err(|_| FuseError::BufferTooSmall)
73    }
74
75    fn reply_expectation(&self) -> ReplyExpectation {
76        ReplyExpectation::payload(size_of::<LseekReply>())
77    }
78
79    fn parse_reply(
80        payload_len: usize,
81        reader: &mut VmReader<'_, Infallible>,
82    ) -> FuseResult<Self::Output> {
83        read_payload(payload_len, reader)
84    }
85}
