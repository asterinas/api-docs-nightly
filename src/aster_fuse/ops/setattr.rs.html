3//! `FUSE_SETATTR` updates selected attributes of an inode.
4//!
5//! The request body contains [`SetattrReq`], whose [`SetattrValid`] mask selects
6//! the fields to apply. The reply body contains [`FuseAttrReply`] with the
7//! updated attributes and their cache timeout.
8
9use bitflags::bitflags;
10use ostd::mm::{Infallible, VmReader, VmWriter};
11
12use super::util::read_payload;
13use crate::{
14    FuseAttrReply, FuseError, FuseFileHandle, FuseOpcode, FuseOperation, FuseResult,
15    ReplyExpectation,
16};
17
18bitflags! {
19    /// Selects which fields in [`SetattrReq`] are valid.
20    #[repr(C)]
21    #[derive(Pod, Default)]
22    pub struct SetattrValid: u32 {
23        const FATTR_MODE = 1 << 0;
24        const FATTR_UID = 1 << 1;
25        const FATTR_GID = 1 << 2;
26        const FATTR_SIZE = 1 << 3;
27        const FATTR_ATIME = 1 << 4;
28        const FATTR_MTIME = 1 << 5;
29        const FATTR_FH = 1 << 6;
30        const FATTR_ATIME_NOW = 1 << 7;
31        const FATTR_MTIME_NOW = 1 << 8;
32        const FATTR_LOCKOWNER = 1 << 9;
33        const FATTR_CTIME = 1 << 10;
34    }
35}
36
37#[repr(C)]
38#[derive(Clone, Copy, Debug, Default, Pod)]
39pub struct SetattrReq {
40    /// Bitmask selecting the attributes to update.
41    valid: SetattrValid,
42    padding: u32,
43    /// File handle used when [`SetattrValid::FATTR_FH`] is set.
44    fh: FuseFileHandle,
45    /// New file size.
46    size: u64,
47    /// Lock owner used when [`SetattrValid::FATTR_LOCKOWNER`] is set.
48    lock_owner: u64,
49    /// New access time in seconds since the Unix epoch.
50    atime: u64,
51    /// New modification time in seconds since the Unix epoch.
52    mtime: u64,
53    /// New status-change time in seconds since the Unix epoch.
54    ctime: u64,
55    /// Nanosecond component of [`SetattrReq::atime`].
56    atimensec: u32,
57    /// Nanosecond component of [`SetattrReq::mtime`].
58    mtimensec: u32,
59    /// Nanosecond component of [`SetattrReq::ctime`].
60    ctimensec: u32,
61    /// New file type and permission bits.
62    mode: u32,
63    unused4: u32,
64    /// New owner user ID.
65    uid: u32,
66    /// New owner group ID.
67    gid: u32,
68    unused5: u32,
69}
70
71impl SetattrReq {
72    /// Creates a `SetattrReq` with the selected valid-field mask.
73    pub const fn new(valid: SetattrValid) -> Self {
74        Self {
75            valid,
76            padding: 0,
77            fh: FuseFileHandle::new(0),
78            size: 0,
79            lock_owner: 0,
80            atime: 0,
81            mtime: 0,
82            ctime: 0,
83            atimensec: 0,
84            mtimensec: 0,
85            ctimensec: 0,
86            mode: 0,
87            unused4: 0,
88            uid: 0,
89            gid: 0,
90            unused5: 0,
91        }
92    }
93
94    const fn set_field_valid(mut self, valid: SetattrValid) -> Self {
95        self.valid = self.valid.union(valid);
96        self
97    }
98
99    /// Returns the bitmask selecting which attributes to update.
100    pub fn valid(&self) -> SetattrValid {
101        self.valid
102    }
103
104    /// Sets the file handle used when `FATTR_FH` is present.
105    pub const fn set_fh(mut self, fh: FuseFileHandle) -> Self {
106        self.fh = fh;
107        self.set_field_valid(SetattrValid::FATTR_FH)
108    }
109
110    /// Sets the new file size.
111    pub const fn set_size(mut self, size: u64) -> Self {
112        self.size = size;
113        self.set_field_valid(SetattrValid::FATTR_SIZE)
114    }
115
116    /// Sets the new file mode bits.
117    pub const fn set_mode(mut self, mode: u32) -> Self {
118        self.mode = mode;
119        self.set_field_valid(SetattrValid::FATTR_MODE)
120    }
121
122    /// Sets the new owner user ID.
123    pub const fn set_uid(mut self, uid: u32) -> Self {
124        self.uid = uid;
125        self.set_field_valid(SetattrValid::FATTR_UID)
126    }
127
128    /// Sets the new owner group ID.
129    pub const fn set_gid(mut self, gid: u32) -> Self {
130        self.gid = gid;
131        self.set_field_valid(SetattrValid::FATTR_GID)
132    }
133
134    /// Sets the new access time.
135    pub const fn set_atime(mut self, atime: u64, atimensec: u32) -> Self {
136        self.atime = atime;
137        self.atimensec = atimensec;
138        self.set_field_valid(SetattrValid::FATTR_ATIME)
139    }
140
141    /// Sets the new modification time.
142    pub const fn set_mtime(mut self, mtime: u64, mtimensec: u32) -> Self {
143        self.mtime = mtime;
144        self.mtimensec = mtimensec;
145        self.set_field_valid(SetattrValid::FATTR_MTIME)
146    }
147
148    /// Sets the new status-change time.
149    pub const fn set_ctime(mut self, ctime: u64, ctimensec: u32) -> Self {
150        self.ctime = ctime;
151        self.ctimensec = ctimensec;
152        self.set_field_valid(SetattrValid::FATTR_CTIME)
153    }
154}
155
156/// Encodes a `FUSE_SETATTR` request.
157pub struct SetattrOperation {
158    setattr_req: SetattrReq,
159}
160
161impl SetattrOperation {
162    /// Only the fields selected by [`SetattrReq::valid`] are applied;
163    pub fn new(setattr_req: SetattrReq) -> Self {
164        Self { setattr_req }
165    }
166}
167
168impl FuseOperation for SetattrOperation {
169    type Output = FuseAttrReply;
170
171    fn opcode(&self) -> FuseOpcode {
172        FuseOpcode::Setattr
173    }
174
175    fn body_len(&self) -> usize {
176        size_of::<SetattrReq>()
177    }
178
179    fn write_body(&mut self, writer: &mut VmWriter<'_, Infallible>) -> FuseResult<()> {
180        writer
181            .write_val(&self.setattr_req)
182            .map_err(|_| FuseError::BufferTooSmall)
183    }
184
185    fn reply_expectation(&self) -> ReplyExpectation {
186        ReplyExpectation::payload(size_of::<FuseAttrReply>())
187    }
188
189    fn parse_reply(
190        payload_len: usize,
191        reader: &mut VmReader<'_, Infallible>,
192    ) -> FuseResult<Self::Output> {
193        read_payload(payload_len, reader)
194    }
195}
