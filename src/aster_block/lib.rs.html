3//! The block devices of Asterinas.
4//！
5//！This crate provides a number of base components for block devices, including
6//! an abstraction of block devices, as well as the registration and lookup of block devices.
7//!
8//! Block devices use a queue-based model for asynchronous I/O operations. It is necessary
9//! for a block device to maintain a queue to handle I/O requests. The users (e.g., fs)
10//! submit I/O requests to this queue and wait for their completion. Drivers implementing
11//! block devices can create their own queues as needed, with the possibility to reorder
12//! and merge requests within the queue.
13//!
14//! This crate also offers the `Bio` related data structures and APIs to accomplish
15//! safe and convenient block I/O operations, for example:
16//!
17//! ```no_run
18//! // Creates a bio request.
19//! let bio = Bio::new(BioType::Write, sid, segments, None);
20//! // Submits to the block device.
21//! let bio_waiter = bio.submit(block_device)?;
22//! // Waits for the the completion.
23//! let Some(status) = bio_waiter.wait() else {
24//!     return Err(IoError);
25//! };
26//! assert!(status == BioStatus::Complete);
27//! ```
28//!

59pub trait BlockDevice: Send + Sync + Any + Debug {
60    /// Enqueues a new `SubmittedBio` to the block device.
61    fn enqueue(&self, bio: SubmittedBio) -> Result<(), BioEnqueueError>;
62
63    /// Returns the metadata of the block device.
64    fn metadata(&self) -> BlockDeviceMeta;
65
66    /// Returns the name of the block device.
67    fn name(&self) -> &str;
68
69    /// Returns the device ID of the block device.
70    fn id(&self) -> DeviceId;
71
72    /// Returns whether the block device is a partition.
73    fn is_partition(&self) -> bool {
74        false
75    }
76
77    /// Sets the partitions of the block device.
78    fn set_partitions(&self, _infos: Vec<Option<PartitionInfo>>) {}
79
80    /// Returns the partitions of the block device.
81    fn partitions(&self) -> Option<Vec<Arc<dyn BlockDevice>>> {
82        None
83    }
84}

106    Registered,
107    /// Device not found
108    NotFound,
109    /// Invalid arguments
110    InvalidArgs,
111    /// Id Acquired
112    IdAcquired,
113    /// Id Exhausted
114    IdExhausted,

117/// Registers a new block device.
118pub fn register(device: Arc<dyn BlockDevice>) -> Result<(), Error> {
119    let mut registry = DEVICE_REGISTRY.lock();
120    let id = device.id().to_raw();
121    if registry.contains_key(&id) {
122        return Err(Error::Registered);
123    }
124    registry.insert(id, device);
125
126    Ok(())
127}
128
129/// Unregisters an existing block device, returning the device if found.
130pub fn unregister(id: DeviceId) -> Result<Arc<dyn BlockDevice>, Error> {
131    DEVICE_REGISTRY
132        .lock()
133        .remove(&id.to_raw())
134        .ok_or(Error::NotFound)
135}
136
137/// Collects all block devices.
138pub fn collect_all() -> Vec<Arc<dyn BlockDevice>> {
139    DEVICE_REGISTRY.lock().values().cloned().collect()
140}
141
142/// Looks up a block device of a given device ID.
143pub fn lookup(id: DeviceId) -> Option<Arc<dyn BlockDevice>> {
144    DEVICE_REGISTRY.lock().get(&id.to_raw()).cloned()
145}
146
147static DEVICE_REGISTRY: Mutex<BTreeMap<u32, Arc<dyn BlockDevice>>> = Mutex::new(BTreeMap::new());

149#[init_component]

156#[init_component(process)]
157fn init_in_first_process() -> Result<(), component::ComponentInitError> {
158    let devices = collect_all();
159    for device in devices {
160        let Some(partition_info) = partition::parse(&device) else {
161            continue;
162        };
163
164        device.set_partitions(partition_info);
165    }
166
167    Ok(())
