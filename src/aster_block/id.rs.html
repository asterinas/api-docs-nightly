10/// The block index used in the filesystem.
11pub type Bid = BlockId<BLOCK_SIZE>;
12/// The sector index used in the device.
13pub type Sid = BlockId<SECTOR_SIZE>;
14
15impl From<Bid> for Sid {
16    fn from(bid: Bid) -> Self {
17        Self::new(bid.to_raw() * (BLOCK_SIZE / SECTOR_SIZE) as u64)
18    }
19}
20
21const BLOCK_SIZE: u16 = super::BLOCK_SIZE as u16;
22const SECTOR_SIZE: u16 = super::SECTOR_SIZE as u16;
23const_assert!(BLOCK_SIZE / SECTOR_SIZE >= 1);

25/// An index of a block.
26///
27/// The `BlockId<const N: u16>` is a generic type that is parameterized by a constant `N`, which
28/// represents the size of each block in bytes. The `BlockId<_>` provides a type-safe way of handling
29/// block indices.
30/// An Instance of `BlockId<_>` is guaranteed to represent valid block index, derived from byte offset
31/// and the specified block size `N`.
32///
33/// # Examples
34///
35/// ```rust
36/// const BLOCK_SIZE: u16 = 512;
37///
38/// let bytes_offset = 2048;
39/// let block_id = BlockId<BLOCK_SIZE>::from_offset(bytes_offset);
40/// assert!(block_id == (bytes_offset / BLOCK_SIZE));
41/// ```
42///
43/// # Limitation
44///
45/// Currently, the block size is expressed in `u16`. We choose `u16` because
46/// it is reasonably large to represent the common block size used in practice.
47#[repr(C)]
48#[derive(Copy, Clone, Debug, Hash, PartialEq, Eq, PartialOrd, Ord, Pod)]
49pub struct BlockId<const N: u16>(u64);
50
51impl<const N: u16> BlockId<N> {
52    /// Constructs an id from a raw id.
53    pub const fn new(raw_id: u64) -> Self {
54        Self(raw_id)
55    }
56
57    /// Constructs an id from a byte offset.
58    pub const fn from_offset(offset: usize) -> Self {
59        Self((offset / (N as usize)) as _)
60    }
61
62    /// Converts to a byte offset.
63    pub fn to_offset(self) -> usize {
64        (self.0 as usize) * (N as usize)
65    }
66
67    /// Converts to raw id.
68    pub fn to_raw(self) -> u64 {
69        self.0
70    }
71}
72
73impl<const N: u16> Add<u64> for BlockId<N> {
74    type Output = Self;
75
76    fn add(self, other: u64) -> Self::Output {
77        Self(self.0 + other)
78    }
79}
80
81impl<const N: u16> Sub<u64> for BlockId<N> {
82    type Output = Self;
83
84    fn sub(self, other: u64) -> Self::Output {
85        Self(self.0 - other)
86    }
87}
88
89/// Implements the `Step` trait to iterate over `Range<Id>`.
90impl<const N: u16> Step for BlockId<N> {
91    fn steps_between(start: &Self, end: &Self) -> (usize, Option<usize>) {
92        u64::steps_between(&start.0, &end.0)
93    }
94
95    fn forward_checked(start: Self, count: usize) -> Option<Self> {
96        u64::forward_checked(start.0, count).map(Self::new)
97    }
98
99    fn backward_checked(start: Self, count: usize) -> Option<Self> {
100        u64::backward_checked(start.0, count).map(Self::new)
101    }
102}
