13/// Represents a partition entry.
14#[derive(Debug, Copy, Clone, PartialEq, Eq)]
15pub enum PartitionInfo {
16    Mbr(MbrEntry),
17    Gpt(GptEntry),
18}
19
20impl PartitionInfo {
21    pub fn start_sector(&self) -> u64 {
22        match self {
23            PartitionInfo::Mbr(entry) => entry.start_sector as u64,
24            PartitionInfo::Gpt(entry) => entry.start_lba,
25        }
26    }
27
28    pub fn total_sectors(&self) -> u64 {
29        match self {
30            PartitionInfo::Mbr(entry) => entry.total_sectors as u64,
31            PartitionInfo::Gpt(entry) => entry.end_lba - entry.start_lba + 1,
32        }
33    }
34}

36/// A MBR (Master Boot Record) partition table header.
37///
38/// See <https://wiki.osdev.org/MBR_(x86)#MBR_Format>.
39#[repr(C)]
40#[derive(Debug, Copy, Clone, Pod)]
41struct MbrHeader {
42    bootstrap_code: [u8; 440],
43    id: u32,
44    reserved: u16,
45    entries: [MbrEntry; 4],
46    signature: u16,
47}
48
49impl MbrHeader {
50    fn check_signature(&self) -> bool {
51        self.signature == 0xAA55
52    }
53}
54
55/// A MBR (Master Boot Record) partition entry.
56///
57/// See <https://wiki.osdev.org/Partition_Table>.
58#[repr(C, packed)]
59#[derive(Debug, Copy, Clone, PartialEq, Eq, Pod)]
60pub struct MbrEntry {
61    flag: u8,
62    start_chs: ChsAddr,
63    type_: u8,
64    end_chs: ChsAddr,
65    start_sector: u32,
66    total_sectors: u32,
67}
68
69impl MbrEntry {
70    fn is_extended(&self) -> bool {
71        self.type_ == 0x05 || self.type_ == 0x0F
72    }
73
74    fn is_valid(&self) -> bool {
75        // A System ID byte value of 0 is the definitive indicator for an unused entry.
76        // Any other illegal value (CHS Sector = 0 or Total Sectors = 0) may also indicate an unused entry.
77        self.type_ != 0x00
78            && self.start_chs.0[1] != 0
79            && self.end_chs.0[1] != 0
80            && self.total_sectors != 0
81    }
82}

84/// A CHS (Cylinder-Head-Sector) address.
85///
86/// In CHS addressing, sector numbers always start at 1; there is no sector 0.
87///
88/// The CHS address is stored as a 3-byte field:
89/// - Byte 0: Head number (8 bits)
90/// - Byte 1: Bits 0–5 are the sector number (6 bits, valid values 1–63);
91///   bits 6–7 are the upper two bits of the cylinder number
92/// - Byte 2: Lower 8 bits of the cylinder number (bits 0–7)
93#[repr(C)]
94#[derive(Debug, Copy, Clone, PartialEq, Eq, Pod)]
95struct ChsAddr([u8; 3]);
96
97/// A GPT (GUID Partition Table) header.
98///
99/// See <https://wiki.osdev.org/GPT#LBA_1:_Partition_Table_Header>.
100#[repr(C)]
101#[derive(Debug, Copy, Clone, Pod)]
102struct GptHeader {
103    signature: u64,
104    revision: u32,
105    size: u32,
106    crc32: u32,
107    reserved: u32,
108    current_lba: u64,
109    backup_lba: u64,
110    first_usable_lba: u64,
111    last_usable_lba: u64,
112    guid: [u8; 16],
113    partition_entry_lba: u64,
114    nr_partition_entries: u32,
115    size_of_partition_entry: u32,
116    crc32_of_partition_entries: u32,
117    _padding: [u8; 420],
118}
119
120impl GptHeader {
121    fn check_signature(&self) -> bool {
122        &self.signature.to_le_bytes() == b"EFI PART"
123    }
124}

126/// A GPT (GUID Partition Table) partition entry.
127///
128/// See <https://wiki.osdev.org/GPT#LBA_2:_Partition_Entries>.
129#[repr(C)]
130#[derive(Debug, Copy, Clone, PartialEq, Eq, Pod)]
131pub struct GptEntry {
132    // Unique ID that defines the purpose and type of this Partition.
133    // A value of zero defines that this partition entry is not being used.
134    type_guid: [u8; 16],
135    // GUID that is unique for every partition entry.
136    guid: [u8; 16],
137    start_lba: u64,
138    end_lba: u64,
139    attributes: u64,
140    // Null-terminated string containing a human-readable name of the partition.
141    name: [u8; 72],
142}
143
144impl GptEntry {
145    fn is_valid(&self) -> bool {
146        self.type_guid != [0; 16]
147    }
148}

150pub(super) fn parse(device: &Arc<dyn BlockDevice>) -> Option<Vec<Option<PartitionInfo>>> {
151    let mbr = device.read_val::<MbrHeader>(0).unwrap();
152
153    // 0xEE indicates a GPT Protective MBR, a fake partition covering the entire disk.
154    let partitions = if mbr.check_signature() && mbr.entries[0].type_ != 0xEE {
155        parse_mbr(device, &mbr)
156    } else {
157        parse_gpt(device)
158    };
159
160    partitions.iter().any(|p| p.is_some()).then_some(partitions)
161}
162
163fn parse_mbr(device: &Arc<dyn BlockDevice>, mbr: &MbrHeader) -> Vec<Option<PartitionInfo>> {
164    let mut partitions = Vec::new();
165    let mut extended_partition = None;
166    for entry in mbr.entries {
167        if entry.is_extended() {
168            extended_partition = Some(entry.start_sector);
169        }
170
171        if entry.is_valid() {
172            partitions.push(Some(PartitionInfo::Mbr(entry)));
173        } else {
174            partitions.push(None);
175        }
176    }
177
178    if let Some(start_sector) = extended_partition {
179        parse_ebr(device, &mut partitions, start_sector, 0);
180    }
181
182    partitions
183}
184
185fn parse_ebr(
186    device: &Arc<dyn BlockDevice>,
187    partitions: &mut Vec<Option<PartitionInfo>>,
188    start_sector: u32,
189    offset: u32,
190) {
191    let ebr_sector = start_sector + offset;
192    let mut ebr = device
193        .read_val::<MbrHeader>(ebr_sector as usize * SECTOR_SIZE)
194        .unwrap();
195    if ebr.entries[0].is_valid() {
196        ebr.entries[0].start_sector += ebr_sector;
197        partitions.push(Some(PartitionInfo::Mbr(ebr.entries[0])));
198    }
199
200    if ebr.entries[1].is_extended() {
201        parse_ebr(
202            device,
203            partitions,
204            start_sector,
205            ebr.entries[1].start_sector,
206        );
207    }
208}

210fn parse_gpt(device: &Arc<dyn BlockDevice>) -> Vec<Option<PartitionInfo>> {
211    let mut partitions = Vec::new();
212
213    // The primary GPT Header must be located in LBA 1.
214    let gpt = device.read_val::<GptHeader>(SECTOR_SIZE).unwrap();
215
216    if !gpt.check_signature() {
217        return partitions;
218    }
219
220    // TODO: Check the CRC32 of the header and the partition entries, check the backup GPT header.
221
222    let entry_size = gpt.size_of_partition_entry as usize;
223    let entries_per_sector = SECTOR_SIZE / entry_size;
224    let total_sectors = gpt.nr_partition_entries as usize / entries_per_sector;
225    for i in 0..total_sectors {
226        let mut buf = [0u8; SECTOR_SIZE];
227        let offset = (gpt.partition_entry_lba as usize + i) * SECTOR_SIZE;
228        device.read_bytes(offset, buf.as_mut_slice()).unwrap();
229
230        for j in 0..entries_per_sector {
231            let entry_offset = j * gpt.size_of_partition_entry as usize;
232            let entry = GptEntry::from_first_bytes(&buf[entry_offset..entry_offset + entry_size]);
233            if entry.is_valid() {
234                partitions.push(Some(PartitionInfo::Gpt(entry)));
235            } else {
236                partitions.push(None);
237            }
238        }
239    }
240
241    partitions
242}

244#[derive(Debug)]
245pub struct PartitionNode {
246    id: DeviceId,
247    name: String,
248    device: Arc<dyn BlockDevice>,
249    info: PartitionInfo,
250}
251
252impl BlockDevice for PartitionNode {
253    fn enqueue(&self, bio: SubmittedBio) -> Result<(), BioEnqueueError> {
254        bio.set_sid_offset(self.info.start_sector());
255        self.device.enqueue(bio)
256    }
257
258    fn metadata(&self) -> BlockDeviceMeta {
259        let mut metadata = self.device.metadata();
260        metadata.nr_sectors = self.info.total_sectors() as usize;
261        metadata
262    }
263
264    fn name(&self) -> &str {
265        &self.name
266    }
267
268    fn id(&self) -> DeviceId {
269        self.id
270    }
271}
272
273impl PartitionNode {
274    pub fn new(
275        id: DeviceId,
276        name: String,
277        device: Arc<dyn BlockDevice>,
278        info: PartitionInfo,
279    ) -> Self {
280        Self {
281            id,
282            name,
283            device,
284            info,
285        }
286    }
