3//! Defines the main `SysTree` structure and its root node implementation.
4
5use alloc::sync::{Arc, Weak};
6
7use inherit_methods_macro::inherit_methods;
8
9use super::{
10    Result, SysStr,
11    attr::SysAttrSet,
12    node::{SysBranchNode, SysObj},
13};
14use crate::{BranchNodeFields, SysPerms, inherit_sys_branch_node};
15
16#[derive(Debug)]
17pub struct SysTree<Root: SysBranchNode> {
18    root: Arc<Root>,
19    // event_hub: SysEventHub,
20}
21
22impl SysTree<RootNode> {
23    /// Creates a new `SysTree` instance with a default root node.
24    pub(crate) fn new() -> Self {
25        let name = ""; // Only the root has an empty name
26        let attr_set = SysAttrSet::new_empty(); // The root has no attributes
27
28        let root_node = Arc::new_cyclic(|weak_self| {
29            let fields = BranchNodeFields::new(SysStr::from(name), attr_set, weak_self.clone());
30            RootNode { fields }
31        });
32
33        Self { root: root_node }
34    }
35}
36
37impl<Root: SysBranchNode> SysTree<Root> {
38    /// Creates a new `SysTree` instance with the provided root node.
39    pub fn new_with_root(root: Arc<Root>) -> Result<Self> {
40        if !root.is_root() {
41            return Err(crate::Error::InternalError(
42                "The provided root is not a root node",
43            ));
44        }
45
46        Ok(Self { root })
47    }
48
49    /// Returns a reference to the root node of the tree.
50    pub fn root(&self) -> &Arc<Root> {
51        &self.root
52    }
53}
54
55/// The default root node in the `SysTree`.
56///
57/// A `RootNode` can work like a branching node, allowing to add additional nodes
58/// as its children.
59#[derive(Debug)]
60pub struct RootNode {
61    fields: BranchNodeFields<dyn SysObj, Self>,
62}
63
64#[inherit_methods(from = "self.fields")]
65impl RootNode {
66    /// Adds a child node to this `RootNode`.
67    pub fn add_child(&self, new_child: Arc<dyn SysObj>) -> Result<()>;
68}
69
70inherit_sys_branch_node!(RootNode, fields, {
71    fn is_root(&self) -> bool {
72        true
73    }
74
75    fn init_parent(&self, _parent: Weak<dyn SysBranchNode>) {
76        // This method should be a no-op for `RootNode`.
77    }
78
79    fn perms(&self) -> SysPerms {
80        SysPerms::DEFAULT_RO_PERMS
81    }
82});
