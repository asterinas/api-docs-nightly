3//! `print` and `println` macros
4//!
5//! FIXME: It will print to all `virtio-console` devices, which is not a good choice.
6//!
7
8use alloc::{collections::btree_map::BTreeMap, fmt, string::String, sync::Arc};
9use core::fmt::Write;
10
11use aster_console::AnyConsoleDevice;
12use ostd::sync::{LocalIrqDisabled, SpinLockGuard};
13
14/// Prints the formatted arguments to the standard output.
15pub fn _print(args: fmt::Arguments) {
16    // We must call `all_devices_lock` instead of `all_devices` here, as `all_devices` invokes the
17    // `clone` method of `String` and `Arc`, which may lead to a deadlock when there is low memory
18    // in the heap. (The heap allocator will log a message when memory is low.)
19    //
20    // Also, holding the lock will prevent the logs from interleaving.
21    let devices = aster_console::all_devices_lock();
22
23    struct Printer<'a>(
24        SpinLockGuard<'a, BTreeMap<String, Arc<dyn AnyConsoleDevice>>, LocalIrqDisabled>,
25    );
26    impl Write for Printer<'_> {
27        fn write_str(&mut self, s: &str) -> fmt::Result {
28            self.0
29                .values()
30                .for_each(|console| console.send(s.as_bytes()));
31            Ok(())
32        }
33    }
34
35    Printer(devices).write_fmt(args).unwrap();
36}

38/// Copied from Rust std: <https://github.com/rust-lang/rust/blob/master/library/std/src/macros.rs>
39#[macro_export]
40macro_rules! print {
41    ($($arg:tt)*) => {{
42        $crate::_print(format_args!($($arg)*));
43    }};
44}
45
46/// Copied from Rust std: <https://github.com/rust-lang/rust/blob/master/library/std/src/macros.rs>
47#[macro_export]
48macro_rules! println {
49    () => {
50        $crate::print!("\n")
51    };
52    ($($arg:tt)*) => {{
53        $crate::_print(format_args_nl!($($arg)*));
54    }};
55}
