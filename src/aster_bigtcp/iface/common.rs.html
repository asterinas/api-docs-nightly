35pub struct IfaceCommon<E: Ext> {
36    index: u32,
37    name: String,
38    type_: InterfaceType,
39    flags: InterfaceFlags,
40
41    interface: SpinLock<PollableIface<E>, BottomHalfDisabled>,
42    used_ports: SpinLock<BTreeMap<u16, PortState>, BottomHalfDisabled>,
43    sockets: SpinLock<SocketTable<E>, BottomHalfDisabled>,
44    sched_poll: E::ScheduleNextPoll,
45}
46
47impl<E: Ext> IfaceCommon<E> {
48    pub(super) fn new(
49        name: String,
50        type_: InterfaceType,
51        flags: InterfaceFlags,
52        interface: smoltcp::iface::Interface,
53        sched_poll: E::ScheduleNextPoll,
54    ) -> Self {
55        let index = INTERFACE_INDEX_ALLOCATOR.fetch_add(1, Ordering::Relaxed);
56
57        Self {
58            index,
59            name,
60            type_,
61            flags,
62            interface: SpinLock::new(PollableIface::new(interface)),
63            used_ports: SpinLock::new(BTreeMap::new()),
64            sockets: SpinLock::new(SocketTable::new()),
65            sched_poll,
66        }
67    }
68
69    pub(super) fn index(&self) -> u32 {
70        self.index
71    }
72
73    pub(super) fn name(&self) -> &str {
74        &self.name
75    }
76
77    pub(super) fn type_(&self) -> InterfaceType {
78        self.type_
79    }
80
81    pub(super) fn flags(&self) -> InterfaceFlags {
82        self.flags
83    }
84
85    pub(super) fn ipv4_addr(&self) -> Option<Ipv4Address> {
86        self.interface.lock().ipv4_addr()
87    }
88
89    pub(super) fn prefix_len(&self) -> Option<u8> {
90        self.interface.lock().prefix_len()
91    }
92
93    pub(super) fn sched_poll(&self) -> &E::ScheduleNextPoll {
94        &self.sched_poll
95    }
96}

116const IP_LOCAL_PORT_START: u16 = 32768;
117const IP_LOCAL_PORT_END: u16 = 60999;
118
119impl<E: Ext> IfaceCommon<E> {
120    pub(super) fn bind(
121        &self,
122        iface: Arc<dyn Iface<E>>,
123        config: BindPortConfig,
124    ) -> core::result::Result<BoundPort<E>, BindError> {
125        let (port, can_reuse) = self.bind_port(config)?;
126        Ok(BoundPort {
127            iface,
128            port,
129            can_reuse: AtomicBool::new(can_reuse),
130        })
131    }
132
133    /// Allocates an ephemeral port.
134    ///
135    /// We follow the port range that many Linux kernels use by default, which is 32768-60999.
136    ///
137    /// See <https://en.wikipedia.org/wiki/Ephemeral_port>.
138    fn alloc_ephemeral_port(
139        used_ports: &mut BTreeMap<u16, PortState>,
140        _can_reuse: bool,
141    ) -> Option<u16> {
142        for port in IP_LOCAL_PORT_START..=IP_LOCAL_PORT_END {
143            if let Entry::Vacant(..) = used_ports.entry(port) {
144                return Some(port);
145            }
146        }
147
148        // FIXME: If `can_reuse` is `true`, we should also check all in-use ephemeral ports
149        // to see if any can be reused instead of directly returning `None`.
150
151        None
152    }
153
154    fn bind_port(&self, config: BindPortConfig) -> Result<(u16, bool), BindError> {
155        let mut used_ports = self.used_ports.lock();
156        let config_can_reuse = config.can_reuse();
157
158        let port = if let Some(port) = config.port() {
159            port
160        } else {
161            match Self::alloc_ephemeral_port(&mut used_ports, config_can_reuse) {
162                Some(port) => port,
163                None => return Err(BindError::Exhausted),
164            }
165        };
166
167        if let Some(port_state) = used_ports.get_mut(&port) {
168            // FIXME: If the socket is not a backlog socket,
169            // we should check whether there is a listening socket on the port.
170            // If there is, the socket cannot be bound to that port.
171            let can_reuse = matches!(config, BindPortConfig::Backlog(_))
172                || (port_state.can_reuse() & config_can_reuse);
173            if can_reuse {
174                port_state.nsocket += 1;
175                if config_can_reuse {
176                    port_state.nreuse += 1;
177                }
178            } else {
179                return Err(BindError::InUse);
180            }
181        } else {
182            let port_state = PortState::new(config_can_reuse);
183            used_ports.insert(port, port_state);
184        };
185
186        Ok((port, config_can_reuse))
187    }
188
189    /// Releases the port so that it can be used again.
190    fn release_port(&self, port: u16, can_reuse: bool) {
191        let mut used_ports = self.used_ports.lock();
192        if let Entry::Occupied(mut entry) = used_ports.entry(port) {
193            let port_state = entry.get_mut();
194            port_state.nsocket -= 1;
195            if can_reuse {
196                port_state.nreuse -= 1;
197            }
198            if port_state.nsocket == 0 {
199                entry.remove_entry();
200            }
201        }
202    }
203}

220        debug_assert!(removed.is_some());
221    }

224impl<E: Ext> IfaceCommon<E> {
225    pub(super) fn poll<D, P, Q>(
226        &self,
227        device: &mut D,
228        mut process_phy: P,
229        mut dispatch_phy: Q,
230    ) -> Option<u64>
231    where
232        D: Device + ?Sized,
233        P: for<'pkt, 'cx, 'tx> FnHelper<
234                &'pkt [u8],
235                &'cx mut Context,
236                D::TxToken<'tx>,
237                Option<(Ipv4Packet<&'pkt [u8]>, D::TxToken<'tx>)>,
238            >,
239        Q: FnMut(&Packet, &mut Context, D::TxToken<'_>),
240    {
241        let mut interface = self.interface();
242        interface.context_mut().now = get_network_timestamp();
243
244        let mut sockets = self.sockets.lock();
245        let mut socket_actions = Vec::new();
246
247        let mut context = PollContext::new(interface.as_mut(), &sockets, &mut socket_actions);
248        context.poll_ingress(device, &mut process_phy, &mut dispatch_phy);
249        context.poll_egress(device, &mut dispatch_phy);
250
251        // Insert new connections and remove dead connections.
252        for action in socket_actions.into_iter() {
253            match action {
254                SocketTableAction::AddTcpConn(new_tcp_conn) => {
255                    let res = sockets.insert_connection(new_tcp_conn);
256                    debug_assert!(res.is_ok());
257                }
258                SocketTableAction::DelTcpConn(dead_conn_key) => {
259                    sockets.remove_dead_tcp_connection(&dead_conn_key);
260                }
261            }
262        }
263
264        // Note that only TCP connections can have timers set, so as far as the time to poll is
265        // concerned, we only need to consider TCP connections.
266        interface.next_poll_at_ms()
267    }
268}

270/// A port bound to an iface.
271///
272/// When dropped, the port is automatically released.
273//
274// FIXME: TCP and UDP ports are independent. Find a way to track the protocol here.
275pub struct BoundPort<E: Ext> {
276    iface: Arc<dyn Iface<E>>,
277    port: u16,
278    can_reuse: AtomicBool,
279}
280
281impl<E: Ext> BoundPort<E> {
282    /// Returns a reference to the iface.
283    pub fn iface(&self) -> &Arc<dyn Iface<E>> {
284        &self.iface
285    }
286
287    /// Returns the port number.
288    pub fn port(&self) -> u16 {
289        self.port
290    }
291
292    /// Returns the bound endpoint.
293    pub fn endpoint(&self) -> Option<IpEndpoint> {
294        let ip_addr = {
295            let ipv4_addr = self.iface().ipv4_addr()?;
296            IpAddress::Ipv4(ipv4_addr)
297        };
298        Some(IpEndpoint::new(ip_addr, self.port))
299    }
300
301    /// Sets whether the port can be reused.
302    pub fn set_can_reuse(&self, can_reuse: bool) {
303        let iface_common = self.iface.common();
304        let mut used_ports = iface_common.used_ports.lock();
305
306        if self.can_reuse.load(Ordering::Relaxed) == can_reuse {
307            return;
308        }
309
310        if let Some(port_state) = used_ports.get_mut(&self.port) {
311            if can_reuse {
312                port_state.nreuse += 1;
313            } else {
314                port_state.nreuse -= 1;
315            }
316        }
317
318        self.can_reuse.store(can_reuse, Ordering::Relaxed);
319    }
320}
321
322impl<E: Ext> Drop for BoundPort<E> {
323    fn drop(&mut self) {
324        self.iface
325            .common()
326            .release_port(self.port, *self.can_reuse.get_mut());
327    }
328}

330struct PortState {
331    nsocket: usize,
332    /// The number of sockets that have enabled address reuse on this port.
333    nreuse: usize,
334}
335
336impl PortState {
337    pub(self) fn new(can_reuse: bool) -> Self {
338        let nreuse = if can_reuse { 1 } else { 0 };
339        Self { nsocket: 1, nreuse }
340    }
341
342    pub(self) fn can_reuse(&self) -> bool {
343        self.nsocket == self.nreuse
344    }
345}

347/// Interface type.
348///
349/// Reference: <https://elixir.bootlin.com/linux/v6.0.18/source/include/uapi/linux/if_arp.h#L30>
350#[repr(u16)]
351#[derive(Debug, Clone, Copy, TryFromInt, PartialEq, Eq)]
352pub enum InterfaceType {
353    // Arp protocol hardware identifiers
354    /// from KA9Q: NET/ROM pseudo
355    NETROM = 0,
356    /// Ethernet 10Mbps
357    ETHER = 1,
358    /// Experimental Ethernet
359    EETHER = 2,
360
361    // Dummy types for non ARP hardware
362    /// IPIP tunnel
363    TUNNEL = 768,
364    /// IP6IP6 tunnel
365    TUNNEL6 = 769,
366    /// Frame Relay Access Device
367    FRAD = 770,
368    /// SKIP vif
369    SKIP = 771,
370    /// Loopback device
371    LOOPBACK = 772,
372    /// Localtalk device
373    LOCALTALK = 773,
374    // TODO: This enum is not exhaustive
375}

377bitflags! {
378    /// Interface flags.
379    ///
380    /// Reference: <https://elixir.bootlin.com/linux/v6.0.18/source/include/uapi/linux/if.h#L82>
381    pub struct InterfaceFlags: u32 {
382        /// Interface is up
383        const UP				= 1<<0;
384        /// Broadcast address valid
385        const BROADCAST			= 1<<1;
386        /// Turn on debugging
387        const DEBUG			    = 1<<2;
388        /// Loopback net
389        const LOOPBACK			= 1<<3;
390        /// Interface is has p-p link
391        const POINTOPOINT		= 1<<4;
392        /// Avoid use of trailers
393        const NOTRAILERS		= 1<<5;
394        /// Interface RFC2863 OPER_UP
395        const RUNNING			= 1<<6;
396        /// No ARP protocol
397        const NOARP			    = 1<<7;
398        /// Receive all packets
399        const PROMISC			= 1<<8;
400        /// Receive all multicast packets
401        const ALLMULTI			= 1<<9;
402        /// Master of a load balancer
403        const MASTER			= 1<<10;
404        /// Slave of a load balancer
405        const SLAVE			    = 1<<11;
406        /// Supports multicast
407        const MULTICAST			= 1<<12;
408        /// Can set media type
409        const PORTSEL			= 1<<13;
410        /// Auto media select active
411        const AUTOMEDIA			= 1<<14;
412        /// Dialup device with changing addresses
413        const DYNAMIC			= 1<<15;
414        /// Driver signals L1 up
415        const LOWER_UP			= 1<<16;
416        /// Driver signals dormant
417        const DORMANT			= 1<<17;
418        /// Echo sent packets
419        const ECHO			    = 1<<18;
420    }
