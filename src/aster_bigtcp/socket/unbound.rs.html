3use alloc::{boxed::Box, vec};
4
5pub(super) type RawTcpSocket = smoltcp::socket::tcp::Socket<'static>;
6pub type RawUdpSocket = smoltcp::socket::udp::Socket<'static>;
7
8pub(super) fn new_tcp_socket() -> Box<RawTcpSocket> {
9    let raw_tcp_socket = {
10        let rx_buffer = smoltcp::socket::tcp::SocketBuffer::new(vec![0u8; TCP_RECV_BUF_LEN]);
11        let tx_buffer = smoltcp::socket::tcp::SocketBuffer::new(vec![0u8; TCP_SEND_BUF_LEN]);
12        RawTcpSocket::new(rx_buffer, tx_buffer)
13    };
14    Box::new(raw_tcp_socket)
15}
16
17pub(super) fn new_udp_socket() -> Box<RawUdpSocket> {
18    let raw_udp_socket = {
19        let metadata = smoltcp::socket::udp::PacketMetadata::EMPTY;
20        let rx_buffer = smoltcp::socket::udp::PacketBuffer::new(
21            vec![metadata; UDP_METADATA_LEN],
22            vec![0u8; UDP_RECV_PAYLOAD_LEN],
23        );
24        let tx_buffer = smoltcp::socket::udp::PacketBuffer::new(
25            vec![metadata; UDP_METADATA_LEN],
26            vec![0u8; UDP_SEND_PAYLOAD_LEN],
27        );
28        RawUdpSocket::new(rx_buffer, tx_buffer)
29    };
30    Box::new(raw_udp_socket)
31}

33// TCP socket buffer sizes:
34//
35// According to
36// <https://github.com/torvalds/linux/blob/9852d85ec9d492ebef56dc5f229416c925758edc/include/net/sock.h#L2798-L2806>
37// and
38// <https://github.com/torvalds/linux/blob/9852d85ec9d492ebef56dc5f229416c925758edc/net/core/sock.c#L286-L287>,
39// it seems that the socket buffer should be 256 packets * 256 bytes/packet = 65536 bytes by
40// default. However, the loopback MTU is also 65536 bytes, and having the same size for the socket
41// buffer and the MTU will cause the implementation of Nagle's algorithm in smoltcp to behave
42// abnormally (see <https://github.com/asterinas/asterinas/pull/1396>). So the socket buffer size
43// is increased from 64K to 128K.
44//
45// TODO: Consider allowing user programs to set the socket buffer length via `setsockopt` system calls.
46pub const TCP_RECV_BUF_LEN: usize = 65536 * 2;
47pub const TCP_SEND_BUF_LEN: usize = 65536 * 2;
48
49// UDP socket buffer sizes:
50pub const UDP_SEND_PAYLOAD_LEN: usize = 65536;
51pub const UDP_RECV_PAYLOAD_LEN: usize = 65536;
52const UDP_METADATA_LEN: usize = 256;
