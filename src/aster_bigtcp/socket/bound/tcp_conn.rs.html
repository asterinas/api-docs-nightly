67impl<E: Ext> RawTcpSocketExt<E> {
68    /// Checks if the socket may receive any new data.
69    ///
70    /// This is similar to [`RawTcpSocket::may_recv`]. However, this method checks if there can be
71    /// _new_ data. In other words, if there is already buffered data in the socket,
72    /// [`RawTcpSocket::may_recv`] will always return true since it is possible to receive the
73    /// buffered data, but this method may return false if the peer has closed its sending half (so
74    /// no new data can come in).
75    pub fn may_recv_new(&self) -> bool {
76        // See also the implementation of `RawTcpSocket::may_recv`.
77        match self.state() {
78            State::Established => true,
79            // Our sending half is closed, but the peer's sending half is still active.
80            State::FinWait1 | State::FinWait2 => true,
81            _ => false,
82        }
83    }
84
85    /// Checks if the socket is closing.
86    ///
87    /// More specifically, we say a socket is closing if and only if it has sent its FIN packet but
88    /// is still waiting for an ACK packet from the peer to acknowledge the FIN it sent.
89    pub fn is_closing(&self) -> bool {
90        let state = self.state();
91        matches!(state, State::FinWait1 | State::Closing | State::LastAck)
92    }
93
94    /// Returns whether the receiving half of the socket is shut down.
95    ///
96    /// This method will return true if and only if [`TcpConnection::shut_recv`] or
97    /// [`TcpConnection::close`] is called.
98    pub fn is_recv_shut(&self) -> bool {
99        self.is_recv_shut
100    }
101
102    /// Checks if the socket is closed by a RST packet.
103    ///
104    /// Note that the flag is automatically cleared when it is read by
105    /// [`TcpConnection::clear_rst_closed`], [`TcpConnection::send`], or [`TcpConnection::recv`].
106    pub fn is_rst_closed(&self) -> bool {
107        self.is_rst_closed
108    }
109}

112    /// Whether the TCP connection became dead.
113    TcpConnBecameDead
114);

116impl<E: Ext> RawTcpSocketExt<E> {
117    /// Checks the TCP state for additional events and whether the connection is dead.
118    fn check_state(
119        &mut self,
120        this: &Arc<TcpConnectionBg<E>>,
121        old_state: State,
122        old_recv_queue: usize,
123        is_rst: bool,
124    ) -> (SocketEvents, TcpConnBecameDead) {
125        let became_dead = if self.state() != State::Established {
126            // After the connection is closed by the user, no new data can be read, and such unread
127            // data will immediately cause the connection to be reset.
128            // Note that "closed" here means that either (1) `close()` or (2) both `shut_send()`
129            // and `shut_recv()` are called. In the latter case, there may be some buffered data.
130            if self.is_recv_shut
131                // These are states where the sending half is closed but new data can come in.
132                && matches!(old_state, State::FinWait1 | State::FinWait2)
133                && self.recv_queue() > old_recv_queue
134            {
135                // Strictly speaking, the socket isn't closed by an incoming RST packet in this
136                // situation. Instead, we reset the connection and _send_ an outgoing RST packet.
137                // However, Linux reports `ECONNRESET`, so we have to follow Linux.
138                self.is_rst_closed = true;
139                self.abort();
140            }
141            self.check_dead(this)
142        } else {
143            TcpConnBecameDead::FALSE
144        };
145
146        let events = if self.state() != old_state {
147            if self.state() == State::Closed && is_rst {
148                self.is_rst_closed = true;
149            }
150            self.on_new_state(this)
151        } else {
152            SocketEvents::empty()
153        };
154
155        (events, became_dead)
156    }
157
158    fn on_new_state(&mut self, this: &Arc<TcpConnectionBg<E>>) -> SocketEvents {
159        let may_send = self.may_send();
160
161        if may_send && !self.has_connected {
162            self.has_connected = true;
163
164            if let Some(ref listener) = self.listener {
165                let mut backlog = listener.inner.backlog.lock();
166                if let Some(value) = backlog.connecting.remove(this.connection_key()) {
167                    backlog.connected.push(value);
168                }
169                listener.notify_events(SocketEvents::CAN_RECV);
170            }
171        }
172
173        let mut events = SocketEvents::empty();
174        if !self.may_recv_new() {
175            events |= SocketEvents::CLOSED_RECV;
176        }
177        if !may_send {
178            events |= SocketEvents::CLOSED_SEND;
179        }
180
181        events
182    }
183
184    /// Checks whether the TCP connection becomes dead.
185    ///
186    /// A TCP connection is considered dead when and only when the TCP socket is in the closed
187    /// state, meaning it's no longer accepting packets from the network. This is different from
188    /// the socket file being closed, which only initiates the socket close process.
189    ///
190    /// This method must be called after handling network events. However, it is not necessary to
191    /// call this method after handling non-closing user events, because the socket can never be
192    /// dead if it is not closed.
193    fn check_dead(&self, this: &Arc<TcpConnectionBg<E>>) -> TcpConnBecameDead {
194        // According to the current smoltcp implementation, a socket in the CLOSED state with the
195        // remote endpoint set means that an outgoing RST packet is pending. We cannot simply mark
196        // such a socket as dead.
197        if self.state() == State::Closed && self.remote_endpoint().is_none() {
198            return TcpConnBecameDead::TRUE;
199        }
200
201        // According to the current smoltcp implementation, a backlog socket will return back to
202        // the `Listen` state if the connection is RSTed before its establishment.
203        if self.state() == State::Listen {
204            if let Some(ref listener) = self.listener {
205                let mut backlog = listener.inner.backlog.lock();
206                // This may fail due to race conditions, but it's fine.
207                let _ = backlog.connecting.remove(&this.inner.connection_key);
208            }
209
210            return TcpConnBecameDead::TRUE;
211        }
212
213        TcpConnBecameDead::FALSE
214    }
215}

218    pub(super) fn new(
219        socket: Box<RawTcpSocket>,
220        listener: Option<Arc<TcpListenerBg<E>>>,
221        weak_self: &Weak<TcpConnectionBg<E>>,
222    ) -> Self {
223        let connection_key = {
224            // Since the socket is connected, the following unwrap can never fail
225            let local_endpoint = socket.local_endpoint().unwrap();
226            let remote_endpoint = socket.remote_endpoint().unwrap();
227            ConnectionKey::from((local_endpoint, remote_endpoint))
228        };
229
230        let poll_key = PollKey::new(Weak::as_ptr(weak_self).addr());
231
232        let socket_ext = RawTcpSocketExt {
233            socket,
234            listener,
235            has_connected: false,
236            is_recv_shut: false,
237            is_rst_closed: false,
238        };
239
240        TcpConnectionInner {
241            socket: SpinLock::new(socket_ext),
242            poll_key,
243            connection_key,
244        }
245    }
246
247    pub(super) fn lock(&self) -> SpinLockGuard<'_, RawTcpSocketExt<E>, BottomHalfDisabled> {
248        self.socket.lock()
249    }

254
255    fn on_drop(this: &Arc<SocketBg<Self, E>>) {
256        debug_assert!(
257            {
258                let socket = this.inner.lock();
259                if socket.state() == State::Closed {
260                    // (1) The socket is fully closed.
261                    true
262                } else {
263                    // (2) The receiving half is closed and the sending half is closing.
264                    socket.is_recv_shut
265                        && !matches!(
266                            socket.state(),
267                            State::SynSent
268                                | State::SynReceived
269                                | State::Established
270                                | State::CloseWait,
271                        )
272                }
273            },
274            "a connection must be either closed or reset before dropping"
275        );
276    }

287impl<E: Ext> TcpConnection<E> {
288    /// Connects to a remote endpoint.
289    ///
290    /// Polling the iface is _always_ required after this method succeeds.
291    pub fn new_connect(
292        bound: BoundPort<E>,
293        remote_endpoint: IpEndpoint,
294        option: &RawTcpOption,
295        observer: E::TcpEventObserver,
296    ) -> Result<Self, (BoundPort<E>, ConnectError)> {
297        let Some(local_endpoint) = bound.endpoint() else {
298            return Err((bound, ConnectError::Unaddressable));
299        };
300
301        let iface = bound.iface().clone();
302        // We have to lock `interface` before locking `sockets`
303        // to avoid dead lock due to inconsistent lock orders.
304        let mut interface = iface.common().interface();
305        let mut sockets = iface.common().sockets();
306
307        let connection_key = ConnectionKey::from((local_endpoint, remote_endpoint));
308
309        if sockets.lookup_connection(&connection_key).is_some() {
310            return Err((bound, ConnectError::AddressInUse));
311        }
312
313        let socket = {
314            let mut socket = new_tcp_socket();
315
316            option.apply(&mut socket);
317
318            if let Err(err) = socket.connect(interface.context_mut(), remote_endpoint, bound.port())
319            {
320                return Err((bound, err.into()));
321            }
322
323            socket
324        };
325
326        let connection =
327            Self::new_cyclic(bound, |weak| TcpConnectionInner::new(socket, None, weak));
328        interface.update_next_poll_at_ms(&connection.0, PollAt::Now);
329        connection.init_observer(observer);
330
331        let res = sockets.insert_connection(connection.inner().clone());
332        debug_assert!(res.is_ok());
333
334        Ok(connection)
335    }
336
337    /// Returns the state of the connecting procedure.
338    pub fn connect_state(&self) -> ConnectState {
339        let socket = self.0.inner.lock();
340
341        if socket.state() == State::SynSent || socket.state() == State::SynReceived {
342            ConnectState::Connecting
343        } else if socket.has_connected {
344            ConnectState::Connected
345        } else if Arc::strong_count(self.0.as_ref()) > 1 {
346            // Now we should return `ConnectState::Refused`. However, when we do this, we must
347            // guarantee that `into_bound_port` can succeed (see the method's doc comments). We can
348            // only guarantee this after we have removed all `Arc<TcpConnectionBg>` in the iface's
349            // socket set.
350            //
351            // This branch serves to avoid a race condition: if the removal process hasn't
352            // finished, we will return `Connecting` so that the caller won't try to call
353            // `into_bound_port` (which may fail immediately).
354            ConnectState::Connecting
355        } else {
356            ConnectState::Refused
357        }
358    }
359
360    /// Converts back to the [`BoundPort`].
361    ///
362    /// This method will succeed if the connection is fully closed and no network events can reach
363    /// this connection. We guarantee that this method will always succeed if
364    /// [`Self::connect_state`] returns [`ConnectState::Refused`].
365    pub fn into_bound_port(mut self) -> Option<BoundPort<E>> {
366        let this: TcpConnectionBg<E> = Arc::into_inner(self.0.take())?;
367        Some(this.bound)
368    }

370    /// Sends some data.
371    ///
372    /// Polling the iface _may_ be required after this method succeeds.
373    pub fn send<F, R>(&self, f: F) -> Result<(R, NeedIfacePoll), SendError>
374    where
375        F: FnOnce(&mut [u8]) -> (usize, R),
376    {
377        let common = self.iface().common();
378        let mut iface = common.interface();
379
380        let mut socket = self.0.inner.lock();
381
382        if socket.is_rst_closed {
383            socket.is_rst_closed = false;
384            return Err(SendError::ConnReset);
385        }
386        let result = socket.send(f)?;
387
388        let poll_at = socket.poll_at(iface.context_mut());
389        let need_poll = iface.update_next_poll_at_ms(&self.0, poll_at);
390
391        Ok((result, need_poll))
392    }
393
394    /// Receives some data.
395    ///
396    /// Polling the iface _may_ be required after this method succeeds.
397    pub fn recv<F, R>(&self, f: F) -> Result<(R, NeedIfacePoll), RecvError>
398    where
399        F: FnOnce(&mut [u8]) -> (usize, R),
400    {
401        let common = self.iface().common();
402        let mut iface = common.interface();
403
404        let mut socket = self.0.inner.lock();
405
406        if socket.is_recv_shut && socket.recv_queue() == 0 {
407            return Err(RecvError::Finished);
408        }
409        let result = match socket.recv(f) {
410            Err(_) if socket.is_rst_closed => {
411                socket.is_rst_closed = false;
412                return Err(RecvError::ConnReset);
413            }
414            res => res,
415        }?;
416
417        let poll_at = socket.poll_at(iface.context_mut());
418        let need_poll = iface.update_next_poll_at_ms(&self.0, poll_at);
419
420        Ok((result, need_poll))
421    }
422
423    /// Checks if the socket is closed by a RST packet and clears the flag.
424    ///
425    /// This flag is set when the socket is closed by a RST packet, and cleared when the connection
426    /// reset error is reported via one of the [`Self::send`], [`Self::recv`], or this method.
427    pub fn clear_rst_closed(&self) -> bool {
428        let mut socket = self.0.inner.lock();
429
430        let is_rst = socket.is_rst_closed;
431        socket.is_rst_closed = false;
432        is_rst
433    }
434
435    /// Shuts down the sending half of the connection.
436    ///
437    /// This method will return `false` if the socket is in the CLOSED or TIME_WAIT state.
438    ///
439    /// Polling the iface is _always_ required after this method succeeds.
440    pub fn shut_send(&self) -> bool {
441        let mut iface = self.iface().common().interface();
442        let mut socket = self.0.inner.lock();
443
444        if matches!(socket.state(), State::Closed | State::TimeWait) {
445            return false;
446        }
447
448        socket.close();
449
450        let poll_at = socket.poll_at(iface.context_mut());
451        iface.update_next_poll_at_ms(&self.0, poll_at);
452
453        true
454    }
455
456    /// Shuts down the receiving half of the connection.
457    ///
458    /// This method will return `false` if the socket is in the CLOSED or TIME_WAIT state.
459    ///
460    /// Polling the iface is _not_ required after this method succeeds.
461    pub fn shut_recv(&self) -> bool {
462        let mut socket = self.0.inner.lock();
463
464        if matches!(socket.state(), State::Closed | State::TimeWait) {
465            return false;
466        }
467
468        socket.is_recv_shut = true;
469
470        true
471    }
472
473    /// Closes the connection.
474    ///
475    /// Polling the iface is _always_ required after this method succeeds.
476    ///
477    /// Note that either this method or [`Self::reset`] must be called before dropping the TCP
478    /// connection to avoid resource leakage.
479    pub fn close(&self) {
480        let mut iface = self.iface().common().interface();
481        let mut socket = self.0.inner.lock();
482
483        socket.is_recv_shut = true;
484
485        if socket.recv_queue() != 0 {
486            // If there is unread data, reset the connection immediately.
487            socket.abort();
488        } else {
489            socket.close();
490        }
491
492        let poll_at = socket.poll_at(iface.context_mut());
493        iface.update_next_poll_at_ms(&self.0, poll_at);
494    }
495
496    /// Resets the connection.
497    ///
498    /// Polling the iface is _always_ required after this method succeeds.
499    ///
500    /// Note that either this method or [`Self::close`] must be called before dropping the TCP
501    /// connection to avoid resource leakage.
502    pub fn reset(&self) {
503        let mut iface = self.iface().common().interface();
504        let mut socket = self.0.inner.lock();
505
506        socket.abort();
507
508        let poll_at = socket.poll_at(iface.context_mut());
509        iface.update_next_poll_at_ms(&self.0, poll_at);
510    }
511
512    /// Calls `f` with an immutable reference to the associated [`RawTcpSocket`].
513    //
514    // NOTE: If a mutable reference is required, add a method above that correctly updates the next
515    // polling time.
516    pub fn raw_with<F, R>(&self, f: F) -> R
517    where
518        F: FnOnce(&RawTcpSocketExt<E>) -> R,
519    {
520        let socket = self.0.inner.lock();
521        f(&socket)
522    }
523}

552#[derive(Debug, PartialEq, Eq, Clone)]
553pub(crate) enum TcpProcessResult {
554    NotProcessed,
555    Processed,
556    ProcessedWithReply(IpRepr, TcpRepr<'static>),
557}
558
559impl<E: Ext> TcpConnectionBg<E> {
560    /// Tries to process an incoming packet and returns whether the packet is processed.
561    pub(crate) fn process(
562        self: &Arc<Self>,
563        iface: &mut PollableIfaceMut<E>,
564        ip_repr: &IpRepr,
565        tcp_repr: &TcpRepr,
566    ) -> (TcpProcessResult, TcpConnBecameDead) {
567        let mut socket = self.inner.lock();
568
569        if !socket.accepts(iface.context_mut(), ip_repr, tcp_repr) {
570            return (TcpProcessResult::NotProcessed, TcpConnBecameDead::FALSE);
571        }
572
573        // If the socket is in the TIME-WAIT state and a new packet arrives that is a SYN packet
574        // without an ACK number, the TIME-WAIT socket will be marked as dead,
575        // and the packet will be passed on to any other listening sockets for processing.
576        //
577        // FIXME: According to the Linux implementation:
578        //  * Before marking the socket as dead, we should check some additional fields in the
579        //    packet (e.g., the timestamp options) to make sure that the packet is not an old
580        //    duplicate packet [1], and ensure that there is actually a listening socket that can
581        //    accept the SYN packet [2].
582        //  * After marking the socket as dead, we should choose a reasonable initial sequence
583        //    number for the newly accepted connection to avoid mixing old duplicate packets with
584        //    packets in the new connection [3].
585        // All of these detail mechanisms are not currently implemented.
586        //
587        // [1]: https://elixir.bootlin.com/linux/v6.0.9/source/net/ipv4/tcp_minisocks.c#L211-L214
588        // [2]: https://elixir.bootlin.com/linux/v6.0.9/source/net/ipv4/tcp_ipv4.c#L2138-L2145
589        // [3]: https://elixir.bootlin.com/linux/v6.0.9/source/net/ipv4/tcp_minisocks.c#L218
590        if socket.state() == State::TimeWait
591            && tcp_repr.control == TcpControl::Syn
592            && tcp_repr.ack_number.is_none()
593        {
594            // This is a very silly approach to force the socket to go dead. The first `abort` call
595            // changes the socket state to `CLOSED`, the second `dispatch` call sets the tuple
596            // (local/remote endpoint) to none. So this socket will not accept or send any packets
597            // in the future.
598            socket.abort();
599            socket
600                .dispatch(iface.context_mut(), |_, _| {
601                    Ok::<(), core::convert::Infallible>(())
602                })
603                .unwrap();
604
605            iface.update_next_poll_at_ms(self, PollAt::Ingress);
606            return (TcpProcessResult::NotProcessed, TcpConnBecameDead::TRUE);
607        }
608
609        let old_state = socket.state();
610        let old_recv_queue = socket.recv_queue();
611        let is_rst = tcp_repr.control == TcpControl::Rst;
612        // For TCP, receiving an ACK packet can free up space in the queue, allowing more packets
613        // to be queued.
614        let mut events = SocketEvents::CAN_RECV | SocketEvents::CAN_SEND;
615
616        let result = match socket.process(iface.context_mut(), ip_repr, tcp_repr) {
617            None => TcpProcessResult::Processed,
618            Some((ip_repr, tcp_repr)) => TcpProcessResult::ProcessedWithReply(ip_repr, tcp_repr),
619        };
620
621        let (state_events, became_dead) =
622            socket.check_state(self, old_state, old_recv_queue, is_rst);
623        events |= state_events;
624
625        self.notify_events(events);
626
627        let poll_at = socket.poll_at(iface.context_mut());
628        iface.update_next_poll_at_ms(self, poll_at);
629
630        (result, became_dead)
631    }
632
633    /// Tries to generate an outgoing packet and dispatches the generated packet.
634    pub(crate) fn dispatch<D>(
635        self: &Arc<Self>,
636        iface: &mut PollableIfaceMut<E>,
637        dispatch: D,
638    ) -> (Option<(IpRepr, TcpRepr<'static>)>, TcpConnBecameDead)
639    where
640        D: FnOnce(PollableIfaceMut<E>, &IpRepr, &TcpRepr) -> Option<(IpRepr, TcpRepr<'static>)>,
641    {
642        let mut socket = self.inner.lock();
643
644        let old_state = socket.state();
645        let old_recv_queue = socket.recv_queue();
646        let mut is_rst = false;
647        let mut events = SocketEvents::empty();
648
649        let mut reply = None;
650        let (cx, pending) = iface.inner_mut();
651        socket
652            .dispatch(cx, |cx, (ip_repr, tcp_repr)| {
653                reply = dispatch(PollableIfaceMut::new(cx, pending), &ip_repr, &tcp_repr);
654                Ok::<(), ()>(())
655            })
656            .unwrap();
657
658        // `dispatch` can return a packet in response to the generated packet. If the socket
659        // accepts the packet, we can process it directly.
660        while let Some((ref ip_repr, ref tcp_repr)) = reply {
661            if !socket.accepts(iface.context_mut(), ip_repr, tcp_repr) {
662                break;
663            }
664            is_rst |= tcp_repr.control == TcpControl::Rst;
665            events |= SocketEvents::CAN_RECV | SocketEvents::CAN_SEND;
666            reply = socket.process(iface.context_mut(), ip_repr, tcp_repr);
667        }
668
669        let (state_events, became_dead) =
670            socket.check_state(self, old_state, old_recv_queue, is_rst);
671        events |= state_events;
672
673        self.notify_events(events);
674
675        let poll_at = socket.poll_at(iface.context_mut());
676        iface.update_next_poll_at_ms(self, poll_at);
677
678        (reply, became_dead)
679    }
680}
