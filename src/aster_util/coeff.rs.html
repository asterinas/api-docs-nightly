3//! This module provides an abstraction `Coeff` to server for efficient and accurate calculation
4//! of fraction multiplication.
5
6use core::ops::Mul;
7
8/// A `Coeff` is used to do a fraction multiplication operation with an unsigned integer.
9/// It can achieve accurate and efficient calculation and avoid numeric overflow at the same time.
10///
11/// # Example
12///
13/// Let's say we want to multiply a fraction (23456 / 56789) with the target integer `a`,
14/// which will be no larger than `1_000_000_000`, we can use the following code snippets
15/// to get an accurate result.
16///
17/// ```
18/// let a = input();
19/// let coeff = Coeff::new(23456, 56789, 1_000_000_000);
20/// let result = coeff * a;
21/// ```
22///
23/// # How it works
24/// `Coeff` is used in the calculation of a fraction value multiplied by an integer.
25/// Here is a simple example of such calculation:
26///
27/// ```rust
28/// let result = (a / b) * c;
29/// ```
30///
31/// In this equation, `a`, `b`, `c` and `result` are all integers. To acquire a more precise result, we will
32/// generally calculate `a * c` first and then divide the multiplication result with `b`.
33/// However, this simple calculation above has two complications:
34/// - The calculation of `a * c` may overflow if they are too large.
35/// - The division operation is much more expensive than integer multiplication, which can easily create performance bottlenecks.
36///
37/// `Coeff` is implemented to address these two issues. It can be used to replace the fraction in this calculation.
38/// For example, a `Coeff` generated from (a / b) can modify the calculation above to ensure that:
39///
40/// ```
41/// coeff * c ~= (a / b) * c
42/// ```
43///
44/// In principle, `Coeff` actually turns the multiplication and division into a combination of multiplication and bit operation.
45/// When creating a `Coeff`, it needs to know the numerator and denominator of the represented fraction
46/// and the max multiplier it will be multiplied by. Then, a `mult` and a `shift` will be chosen to achieve the replacement of calculation.
47/// Taking the previous calculation as an example again, `coeff * c` will turn into `mult * c >> shift`, ensuring that:
48///
49/// ```
50/// mult * c >> shift ~= (a / b) * c
51/// ```
52///
53/// and
54///  
55/// `mult * c` will not result in numeric overflow (i.e., `mult * c` will stay below MAX_U64).
56///
57/// This is how `Coeff` achieves accuracy and efficiency at the same time.
58#[derive(Debug, Copy, Clone)]
59pub struct Coeff {
60    mult: u32,
61    shift: u32,
62    max_multiplier: u64,
63}
64
65impl Coeff {
66    /// Create a new coeff, which is essentially equivalent to ï¼ˆ`numerator` / `denominator`) when being multiplied to an integer;
67    /// Here users should make sure the multiplied integer should not be larger than `max_multiplier`.
68    pub fn new(numerator: u64, denominator: u64, max_multiplier: u64) -> Self {
69        let mut shift_acc: u32 = 32;
70        // Too large `max_multiplier` will make the generated coeff imprecise
71        debug_assert!(max_multiplier < (1 << 40));
72        let mut tmp = max_multiplier >> 32;
73        // Counts the number of 0 in front of the `max_multiplier`.
74        // `shift_acc` indicates the maximum number of bits `mult` can have.
75        while tmp > 0 {
76            tmp >>= 1;
77            shift_acc -= 1;
78        }
79
80        // Try the `shift` from 32 to 0.
81        let mut shift = 32;
82        let mut mult = 0;
83        while shift > 0 {
84            mult = numerator << shift;
85            mult += denominator / 2;
86            mult /= denominator;
87            if (mult >> shift_acc) == 0 {
88                break;
89            }
90            shift -= 1;
91        }
92        Self {
93            mult: mult as u32,
94            shift,
95            max_multiplier,
96        }
97    }
98
99    /// Return the `mult` of the Coeff.
100    /// Only used for the VdsoData and will be removed in the future.
101    pub fn mult(&self) -> u32 {
102        self.mult
103    }
104
105    /// Return the `shift` of the Coeff.
106    /// Only used for the VdsoData and will be removed in the future.
107    pub fn shift(&self) -> u32 {
108        self.shift
109    }
110}
111
112impl Mul<u64> for Coeff {
113    type Output = u64;
114    fn mul(self, rhs: u64) -> Self::Output {
115        debug_assert!(rhs <= self.max_multiplier);
116        (rhs * self.mult as u64) >> self.shift
117    }
118}
119
120impl Mul<u32> for Coeff {
121    type Output = u32;
122    fn mul(self, rhs: u32) -> Self::Output {
123        debug_assert!(rhs as u64 <= self.max_multiplier);
124        ((rhs as u64 * self.mult as u64) >> self.shift) as u32
125    }
126}
