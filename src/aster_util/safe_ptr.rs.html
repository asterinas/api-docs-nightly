17/// Safe pointers.
18///
19/// # Overview
20///
21/// Safe pointers allows using pointers to access memory without
22/// unsafe code, which is a key enabler for writing device drivers in safe Rust.
23///
24/// To ensure its soundness, safe pointers (`SafePtr<T, M, _>`) have to be
25/// more restricted than raw pointers (`*const T` or `*mut T`).
26/// More specifically, there are three major restrictions.
27///
28/// 1. A safe pointer can only refer to a value of a POD type `T: Pod`,
29///    while raw pointers can do to a value of any type `T`.
30/// 2. A safe pointer can only refer to an address within a virtual memory object
31///    of type `M: VmIo` (e.g., VMAR and VMO), while raw pointers can do to
32///    an address within any virtual memory space.
33/// 3. A safe pointer only allows one to copy values to/from the target address,
34///    while a raw pointer allows one to borrow an immutable or mutable reference
35///    to the target address.
36///
37/// The expressiveness of safe pointers, although being less than that of
38/// raw pointers, is sufficient for our purpose of writing an OS kernel in safe
39/// Rust.
40///
41/// In addition, safe pointers `SafePtr<T, M, R>` are associated with access
42/// rights, which are encoded statically with type `R: TRights`.
43///
44/// # Examples
45///
46/// ## Constructing a safe pointer
47///
48/// An instance of `SafePtr` can be created with a VM object and an address
49/// within the VM object.
50///
51/// ```
52/// let u32_ptr: SafePtr<u32, Vec<u8>, _> = {
53///     let vm_obj = VmoOptions::new(PAGE_SIZE).alloc().unwrap();
54///     let addr = 16;
55///     SafePtr::new(vm_obj, addr)
56/// };
57/// ```
58///
59/// The generic parameter `M` of `SafePtr<_, M, _>` must implement the `VmIo`
60/// trait. The most important `VmIo` types are `Vmar`, `Vmo`, `IoMem`, and
61/// `UFrame`. The blanket implementations of `VmIo` also include pointer-like
62/// types that refer to a `VmIo` type. Some examples are `&Vmo`, `Box<Vmar>`,
63/// and `Arc<IoMem>`.
64///
65/// The safe pointer itself does not and cannot guarantee that its address is valid.
66/// This is because different VM objects may interpret addresses differently
67/// and each VM object can have different restrictions for valid addresses.
68/// So the detection of invalid addresses is delayed to the time when the
69/// pointers are actually read from or written to.
70///
71/// Initially, a newly-created safe pointer has all access rights.
72///
73/// ## Reading and writing a safe pointer
74///
75/// The value pointed to by a safe pointer can be read or written with the
76/// `read` or `write` method. Both methods may return errors. The possible reasons
77/// of error are determined by the underlying VM objects.
78///
79/// ```
80/// u32_ptr.write(1234).unwrap();
81/// assert!(u32_ptr.read().unwrap() == 1234);
82/// ```
83///
84/// ## Manipulating a safe pointer
85///
86/// The address of a safe pointer can be obtained by the `addr` method.
87/// The address can be updated by assigning a new value with the `set_addr` method
88/// or updated incrementally through methods like `add`, `offset`, `byte_addr`,
89/// `byte_offset`.
90///  
91/// The VM object of a safe pointer can also be obtained or updated through the
92/// `vm` and `set_vm` methods. A new safe pointer that is backed by the same
93/// VM object of an existing safe pointer can be obtained through the `borrow_vm`
94/// method.
95///
96/// As an example, the code below shows how the `add` and `borrow_vm` methods
97/// can be used together to to iterate all values pointed to by an array pointer.
98///
99/// ```
100/// fn collect_values<T>(array_ptr: &SafePtr<T, M, _>, array_len: usize) -> Vec<T> {
101///     let mut curr_ptr: SafePtr<T, &M, _> = array_ptr.borrow_vm();
102///     (0..array_len)
103///         .iter()
104///         .map(|_| {
105///             let val = curr_ptr.read().unwrap();
106///             curr_ptr.add(1);
107///             val
108///         })
109///         .collect()
110/// }
111/// ```
112///
113/// The data type of a safe pointer can be converted with the `cast` method.
114///
115/// ```rust
116/// let u8_ptr: SafePtr<u8, _, _> = u32_ptr.cast();
117/// ```
118///
119/// ## Reading and writing the fields of a struct
120///
121/// Given a safe pointer that points to a struct (say, `Foo`), one can read
122/// the value of its field as follows.
123///
124/// ```
125/// pub struct Foo {
126///     first: u64,
127///     second: u32,
128/// }
129///
130/// fn read_second_field<M: VmIo>(ptr: &SafePtr<Foo, M, _>) -> u32 {
131///     let field_ptr = ptr
132///         .borrow_vm()
133///         .byte_add(offset_of!(Foo, second) as usize)
134///         .cast::<u32>();
135///     field_ptr.read().unwrap()
136/// }
137/// ```
138///
139/// But this coding pattern is too tedius for such a common task.
140/// To make the life of users easier, we provide a convenient macro named
141/// `field_ptr`, which can be used to obtain the safe pointer of a field from
142/// that of its containing struct.
143///
144/// ```
145/// fn read_second_field<M: VmIo>(ptr: &SafePtr<Foo, M, _>) -> u32 {
146///     let field_ptr = field_ptr!(ptr, Foo, second);
147///     field_ptr.read().unwrap()
148/// }
149/// ```
150///
151/// # Access rights
152///
153/// A safe pointer may have a combination of three access rights:
154/// Read, Write, and Dup.
155pub struct SafePtr<T, M, R = Full> {
156    offset: usize,
157    vm_obj: M,
158    rights: R,
159    phantom: PhantomData<T>,
160}

162impl<T, M> SafePtr<T, M> {
163    /// Create a new instance.
164    ///
165    /// # Access rights
166    ///
167    /// The default access rights of a new instance are `Read`, `Write`, and
168    /// `Dup`.
169    pub fn new(vm_obj: M, offset: usize) -> Self {
170        Self {
171            vm_obj,
172            offset,
173            rights: TRightSet(<TRights![Dup, Read, Write, Exec, Signal]>::new()),
174            phantom: PhantomData,
175        }
176    }
177}

179impl<T, M: HasPaddr, R> HasPaddr for SafePtr<T, M, R> {
180    fn paddr(&self) -> Paddr {
181        self.vm_obj.paddr() + self.offset
182    }

185// =============== Read and write methods ==============
186impl<T: Pod, M: VmIo, R: TRights> SafePtr<T, M, TRightSet<R>> {
187    /// Reads the value from the pointer.
188    ///
189    /// # Access rights
190    ///
191    /// This method requires the Read right.
192    #[require(R > Read)]
193    pub fn read(&self) -> Result<T> {
194        self.vm_obj.read_val(self.offset)
195    }
196
197    /// Reads a slice of values from the pointer.
198    ///
199    /// # Access rights
200    ///
201    /// This method requires the Read right.
202    #[require(R > Read)]
203    pub fn read_slice(&self, slice: &mut [T]) -> Result<()> {
204        self.vm_obj.read_slice(self.offset, slice)
205    }
206
207    /// Overwrites the value at the pointer.
208    ///
209    /// # Access rights
210    ///
211    /// This method requires the Write right.
212    #[require(R > Write)]
213    pub fn write(&self, val: &T) -> Result<()> {
214        self.vm_obj.write_val(self.offset, val)
215    }
216
217    /// Overwrites a slice of values at the pointer.
218    ///
219    /// # Access rights
220    ///
221    /// This method requires the Write right.
222    #[require(R > Write)]
223    pub fn write_slice(&self, slice: &[T]) -> Result<()> {
224        self.vm_obj.write_slice(self.offset, slice)
225    }
226
227    /// Copies the value from another pointer.
228    ///
229    /// # Errors
230    ///
231    /// This method will fail if
232    ///  - either pointer's underlying VM object is too short to hold `T` or
233    ///  - the memory copy between two pointers fails.
234    ///
235    /// # Access rights
236    ///
237    /// This method requires
238    ///  - the Write right for this pointer and
239    ///  - the Read right for another pointer.
240    #[require(R > Write)]
241    #[require(R1 > Read)]
242    pub fn copy_from<M1: HasVmReaderWriter, R1: TRights>(
243        &self,
244        ptr: &SafePtr<T, M1, TRightSet<R1>>,
245    ) -> Result<()> {
246        let mut reader = M1::Types::to_reader_result(ptr.vm_obj.reader())?.to_fallible();
247
248        if reader.remain() < size_of::<T>() {
249            return Err(Error::InvalidArgs);
250        }
251        reader.limit(size_of::<T>());
252
253        self.vm_obj.write(self.offset, &mut reader)
254    }
255}

280// =============== Address-related methods ==============
281impl<T, M, R> SafePtr<T, M, R> {
282    pub const fn is_aligned(&self) -> bool {
283        self.offset.is_multiple_of(align_of::<T>())
284    }
285
286    /// Increase the address in units of bytes occupied by the generic T.
287    pub fn add(&mut self, count: usize) {
288        let offset = count * size_of::<T>();
289        self.offset += offset;
290    }
291
292    /// Increase or decrease the address in units of bytes occupied by the generic T.
293    pub fn offset(&mut self, count: isize) {
294        let offset = count * size_of::<T>() as isize;
295        if count >= 0 {
296            self.offset += offset as usize;
297        } else {
298            self.offset -= offset as usize;
299        }
300    }
301
302    /// Increase the address in units of bytes.
303    pub fn byte_add(&mut self, bytes: usize) {
304        self.offset += bytes;
305    }
306
307    /// Increase or decrease the address in units of bytes.
308    pub fn byte_offset(&mut self, bytes: isize) {
309        if bytes >= 0 {
310            self.offset += bytes as usize;
311        } else {
312            self.offset -= (-bytes) as usize;
313        }
314    }
315}

449/// Create a safe pointer for the field of a struct.
450#[macro_export]
451macro_rules! field_ptr {
452    ($ptr:expr, $type:ty, $($field:tt)+) => {{
453        use aster_util::safe_ptr::SafePtr;
454
455        #[inline]
456        fn new_field_ptr<T, M, R: Clone, U>(
457            container_ptr: &SafePtr<T, M, R>,
458            field_offset: usize,
459            _type_infer: *const U,
460        ) -> SafePtr<U, &M, R>
461        {
462            let mut ptr = container_ptr.borrow_vm();
463            ptr.byte_add(field_offset);
464            ptr.cast()
465        }
466
467        let field_offset = core::mem::offset_of!($type, $($field)*);
468        let type_infer = ostd::ptr_null_of!({
469            let ptr: *const $type = core::ptr::null();
470            // This is not sound, but the code won't be executed.
471            &raw const (*ptr).$($field)*
472        });
473
474        new_field_ptr($ptr, field_offset, type_infer)
475    }}
