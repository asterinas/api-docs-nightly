3//! A sparse u32 ID allocator over a configurable range.
4//!
5//! The [`SparseIdAlloc`] type allocates the smallest available ID in an
6//! inclusive `u32` range and supports returning IDs later.
7
8#![cfg_attr(not(test), no_std)]
9#![deny(unsafe_code)]
10
11extern crate alloc;
12
13use alloc::collections::BTreeMap;
14
15/// A sparse `u32` ID allocator over a configurable `[min, max]` range.
16///
17/// # Allocation policy
18///
19/// Always returns the smallest unallocated ID in the range, or `None`
20/// when every ID is in use.
21#[derive(Clone, Debug)]
22pub struct SparseIdAlloc {
23    min: u32,
24    max: u32,
25    /// Maximal runs of allocated IDs: start -> inclusive end.
26    /// Runs are disjoint and non-adjacent.
27    allocated_runs: BTreeMap<u32, u32>,
28}
29
30impl SparseIdAlloc {
31    /// Creates a new allocator that hands out IDs from the inclusive range
32    /// `[min, max]`.
33    ///
34    /// # Panics
35    ///
36    /// Panics if `min > max`.
37    pub const fn new(min: u32, max: u32) -> Self {
38        assert!(min <= max, "min must be <= max");
39        Self {
40            min,
41            max,
42            allocated_runs: BTreeMap::new(),
43        }
44    }
45
46    /// Allocates and returns a new ID from the configured range.
47    ///
48    /// Returns `None` when every ID in `[min, max]` is in use.
49    pub fn alloc(&mut self) -> Option<u32> {
50        let first = self
51            .allocated_runs
52            .first_key_value()
53            .map(|(&start, &end)| (start, end));
54
55        match first {
56            // The range's lowest IDs are allocated; the smallest free ID
57            // is just past the first run (unless the range is exhausted).
58            Some((start, end)) if start == self.min => {
59                if end == self.max {
60                    return None;
61                }
62
63                let id = end + 1;
64                let new_end = id
65                    .checked_add(1)
66                    .and_then(|next_start| self.allocated_runs.remove(&next_start))
67                    .unwrap_or(id);
68                *self.allocated_runs.get_mut(&start).unwrap() = new_end;
69                Some(id)
70            }
71            // `min` itself is free.
72            _ => {
73                let id = self.min;
74                let end = id
75                    .checked_add(1)
76                    .and_then(|next_start| self.allocated_runs.remove(&next_start))
77                    .unwrap_or(id);
78                self.allocated_runs.insert(id, end);
79                Some(id)
80            }
81        }
82    }
83
84    /// Releases the given ID.
85    ///
86    /// # Panics
87    ///
88    /// Panics if `id` is not currently allocated (out of range, never
89    /// issued, or already freed).
90    pub fn free(&mut self, id: u32) {
91        let run = self
92            .allocated_runs
93            .range(..=id)
94            .next_back()
95            .map(|(&start, &end)| (start, end));
96        let Some((start, end)) = run.filter(|&(_, end)| id <= end) else {
97            panic!("free({id}) of unallocated ID");
98        };
99
100        if id == start {
101            self.allocated_runs.remove(&start);
102        } else {
103            *self.allocated_runs.get_mut(&start).unwrap() = id - 1;
104        }
105        if id < end {
106            self.allocated_runs.insert(id + 1, end);
107        }
108    }
109}

111#[cfg(test)]
112mod test {
113    use super::SparseIdAlloc;
114
115    #[test]
116    fn sparse_alloc_is_monotonic_from_min() {
117        let mut a = SparseIdAlloc::new(1, u32::MAX);
118        assert_eq!(a.alloc(), Some(1));
119        assert_eq!(a.alloc(), Some(2));
120        assert_eq!(a.alloc(), Some(3));
121    }
122
123    #[test]
124    fn sparse_alloc_reuses_freed_ids_immediately() {
125        let mut a = SparseIdAlloc::new(1, u32::MAX);
126        for _ in 0..3 {
127            let _ = a.alloc();
128        }
129        a.free(3);
130        a.free(1);
131        a.free(2);
132        assert_eq!(a.alloc(), Some(1));
133        assert_eq!(a.alloc(), Some(2));
134        assert_eq!(a.alloc(), Some(3));
135    }
136
137    #[test]
138    fn sparse_alloc_reuses_smallest_gap_first() {
139        let mut a = SparseIdAlloc::new(1, 4);
140        for _ in 0..4 {
141            let _ = a.alloc();
142        }
143        assert_eq!(a.alloc(), None);
144        a.free(4);
145        a.free(2);
146        assert_eq!(a.alloc(), Some(2));
147        assert_eq!(a.alloc(), Some(4));
148        assert_eq!(a.alloc(), None);
149    }
150
151    #[test]
152    fn sparse_alloc_handles_single_id_range() {
153        let mut a = SparseIdAlloc::new(u32::MAX, u32::MAX);
154        assert_eq!(a.alloc(), Some(u32::MAX));
155        assert_eq!(a.alloc(), None);
156    }
157
158    #[test]
159    #[should_panic]
160    fn sparse_alloc_panics_when_min_greater_than_max() {
161        let _ = SparseIdAlloc::new(5, 4);
162    }
163
164    #[test]
165    #[should_panic(expected = "unallocated")]
166    fn sparse_alloc_panics_on_free_never_issued() {
167        let mut a = SparseIdAlloc::new(1, 100);
168        let _ = a.alloc();
169        a.free(5);
170    }
171
172    #[test]
173    #[should_panic(expected = "unallocated")]
174    fn sparse_alloc_panics_on_double_free() {
175        let mut a = SparseIdAlloc::new(1, 100);
176        let id = a.alloc().unwrap();
177        a.free(id);
178        a.free(id);
179    }
180}
