24/// Raw PCI Capabilities.
25#[derive(Debug, Default)]
26pub(super) struct RawCapabilities {
27    msix: Option<RawCapabilityMsix>,
28    vndr: Vec<RawCapabilityVndr>,
29}
30
31/// PCI capability types.
32#[repr(u8)]
33#[derive(Clone, Copy, Debug, TryFromInt)]
34enum CapabilityType {
35    /// Power Management
36    Pm = 0x01,
37    /// Accelerated Graphics Part
38    Agp = 0x02,
39    /// Vital Product Data
40    Vpd = 0x03,
41    /// Slot Identification
42    SlotId = 0x04,
43    /// Message Signalled Interrupts
44    Msi = 0x05,
45    /// CompactPCI HotSwap
46    Chswp = 0x06,
47    /// PCI-X
48    PciX = 0x07,
49    /// HyperTransport
50    Hp = 0x08,
51    /// Vendor-Specific
52    Vndr = 0x09,
53    /// Debug port
54    Dbg = 0x0A,
55    /// CompactPCI Central Resource Control
56    Ccrc = 0x0B,
57    /// PCI Standard Hot-Plug Controller
58    Shpc = 0x0C,
59    /// Bridge subsystem vendor/device ID
60    Ssvid = 0x0D,
61    /// AGP Target PCI-PCI bridge
62    Agp3 = 0x0E,
63    /// Secure Device
64    Secdev = 0x0F,
65    /// PCI Express
66    Exp = 0x10,
67    /// MSI-X
68    Msix = 0x11,
69    /// SATA Data/Index Conf
70    Sata = 0x12,
71    /// PCI Advanced Features
72    Af = 0x13,
73    /// Enhanced Allocation
74    Ea = 0x14,
75}
76
77impl RawCapabilities {
78    /// The top of the capability position.
79    const CAPABILITY_TOP: u16 = 0xFC;
80
81    /// Parses the capabilities of the PCI device.
82    pub(super) fn parse(dev: &PciCommonDevice) -> Self {
83        if !dev.read_status().contains(Status::CAPABILITIES_LIST) {
84            return Self::default();
85        }
86
87        // The offset of the first capability pointer is the same for PCI general devices and PCI
88        // bridge devices.
89        const CAP_OFFSET: u16 = PciGeneralDeviceCfgOffset::CapabilitiesPointer as u16;
90        let mut cap_ptr =
91            (dev.location().read8(CAP_OFFSET) as u16).align_down(align_of::<u32>() as _);
92        let mut cap_ptr_vec = Vec::new();
93
94        // Read all capability pointers so that it is easy for us to get the length of each
95        // capability.
96        while cap_ptr > 0 {
97            cap_ptr_vec.push(cap_ptr);
98            cap_ptr = (dev.location().read8(cap_ptr + 1) as u16).align_down(align_of::<u32>() as _);
99        }
100        cap_ptr_vec.sort();
101
102        // Push the top position so that we can calculate the length of the last capability.
103        cap_ptr_vec.push(Self::CAPABILITY_TOP);
104
105        let mut caps = Self::default();
106
107        let length = cap_ptr_vec.len();
108        for i in 0..length - 1 {
109            let cap_ptr = cap_ptr_vec[i];
110            let next_ptr = cap_ptr_vec[i + 1];
111            let raw_cap_type = dev.location().read8(cap_ptr);
112
113            let Ok(cap_type) = CapabilityType::try_from(raw_cap_type) else {
114                continue;
115            };
116            match cap_type {
117                CapabilityType::Msix => {
118                    // "More than one MSI-X Capability structure per Function is prohibited."
119                    if caps.msix.is_some() {
120                        ostd::warn!(
121                            "superfluous MSI-X Capability structures at {:?} are ignored",
122                            dev.location()
123                        );
124                        continue;
125                    }
126                    caps.msix = Some(RawCapabilityMsix::parse(dev, cap_ptr));
127                }
128                CapabilityType::Vndr => {
129                    caps.vndr
130                        .push(RawCapabilityVndr::new(cap_ptr, next_ptr - cap_ptr));
131                }
132                _ => {}
133            }
134        }
135
136        caps
137    }
138
139    /// Acquires a new [`CapabilityMsixData`] instance.
140    pub(super) fn acquire_msix_data(
141        &self,
142        loc: &PciDeviceLocation,
143        bar_manager: &mut BarManager,
144    ) -> Result<Option<CapabilityMsixData>> {
145        let Some(raw_msix) = self.msix.as_ref() else {
146            return Ok(None);
147        };
148
149        Ok(Some(CapabilityMsixData::new(loc, bar_manager, raw_msix)?))
150    }
151
152    /// Iterates over [`CapabilityVndrData`] instances.
153    pub(super) fn iter_vndr_data(
154        &self,
155        loc: &PciDeviceLocation,
156    ) -> impl Iterator<Item = CapabilityVndrData> {
157        self.vndr
158            .iter()
159            .map(|raw_vndr| CapabilityVndrData::new(loc, raw_vndr))
160    }
161}
