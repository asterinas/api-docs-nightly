13/// PCI common device.
14///
15/// This type contains a range of information and functions common to PCI devices.
16#[derive(Debug)]
17pub struct PciCommonDevice {
18    device_id: PciDeviceId,
19    location: PciDeviceLocation,
20    header_type: PciHeaderType,
21    bar_manager: BarManager,
22    capabilities: RawCapabilities,
23}
24
25impl PciCommonDevice {
26    /// Returns the PCI device ID.
27    pub fn device_id(&self) -> &PciDeviceId {
28        &self.device_id
29    }
30
31    /// Returns the PCI device location.
32    pub fn location(&self) -> &PciDeviceLocation {
33        &self.location
34    }
35
36    /// Returns a reference to the PCI Base Address Register (BAR) manager.
37    pub fn bar_manager(&self) -> &BarManager {
38        &self.bar_manager
39    }
40
41    /// Returns a mutable reference to the PCI Base Address Register (BAR) manager.
42    pub fn bar_manager_mut(&mut self) -> &mut BarManager {
43        &mut self.bar_manager
44    }
45
46    /// Returns the PCI device type.
47    pub fn device_type(&self) -> PciDeviceType {
48        self.header_type.device_type()
49    }
50
51    /// Checks whether the device is a multi-function device.
52    pub fn has_multi_funcs(&self) -> bool {
53        self.header_type.has_multi_funcs()
54    }
55
56    /// Reads the PCI command.
57    pub fn read_command(&self) -> Command {
58        Command::from_bits_truncate(self.location.read16(PciCommonCfgOffset::Command as u16))
59    }
60
61    /// Writes the PCI command.
62    pub fn write_command(&self, command: Command) {
63        self.location
64            .write16(PciCommonCfgOffset::Command as u16, command.bits())
65    }
66
67    /// Reads the PCI status.
68    pub fn read_status(&self) -> Status {
69        Status::from_bits_truncate(self.location.read16(PciCommonCfgOffset::Status as u16))
70    }
71
72    /// Acquires necessary resources to build the MSI-X capability data, if the capability exists.
73    ///
74    /// Note that the MSI-X capability data occupies some memory BARs. Therefore, it will fail if
75    /// the necessary resources are not available.
76    pub fn acquire_msix_capability(&mut self) -> Result<Option<CapabilityMsixData>> {
77        self.capabilities
78            .acquire_msix_data(&self.location, &mut self.bar_manager)
79    }
80
81    /// Gets access to the vendor-specific capability data.
82    pub fn iter_vndr_capability(&self) -> impl Iterator<Item = CapabilityVndrData> {
83        self.capabilities.iter_vndr_data(&self.location)
84    }
85
86    /// Gets access to the vendor-specific capability data with a mutable reference to the BAR
87    /// manager.
88    pub fn iter_vndr_capability_with_bar_manager(
89        &mut self,
90    ) -> (impl Iterator<Item = CapabilityVndrData>, &mut BarManager) {
91        (
92            self.capabilities.iter_vndr_data(&self.location),
93            &mut self.bar_manager,
94        )
95    }
96
97    pub(super) fn new(location: PciDeviceLocation) -> Option<Self> {
98        if location.read16(0) == 0xFFFF {
99            // No device.
100            return None;
101        }
102
103        let device_id = PciDeviceId::new(location);
104
105        let mut header_type =
106            PciHeaderType::try_from_raw(location.read8(PciCommonCfgOffset::HeaderType as u16))?;
107        if let PciDeviceType::PciToPciBridge(primary_bus, secondary_bus, subordinate_bus) =
108            &mut header_type.device_type
109        {
110            *primary_bus = location.read8(PciBridgeCfgOffset::PrimaryBusNumber as u16);
111            *secondary_bus = location.read8(PciBridgeCfgOffset::SecondaryBusNumber as u16);
112            *subordinate_bus = location.read8(PciBridgeCfgOffset::SubordinateBusNumber as u16);
113        }
114
115        let bar_manager = BarManager {
116            bars: [const { None }; 6],
117        };
118        let capabilities = RawCapabilities::default();
119
120        let mut device = Self {
121            device_id,
122            location,
123            header_type,
124            bar_manager,
125            capabilities,
126        };
127
128        // While setting up the BARs, we need to ensure that
129        // "Decode (I/O or memory) of the appropriate address space is disabled via the Command
130        // Register before sizing a Base Address register."
131        let command_val = device.read_command() | Command::BUS_MASTER;
132        device.write_command(command_val - (Command::MEMORY_SPACE | Command::IO_SPACE));
133        device.bar_manager = BarManager::new(device.header_type.device_type(), location);
134        device.write_command(command_val | (Command::MEMORY_SPACE | Command::IO_SPACE));
135
136        device.capabilities = RawCapabilities::parse(&device);
137
138        Some(device)
139    }
140}

142/// The header type field of a PCI device struct in the PCI configuration space.
143///
144/// A header type is comprised of two pieces of information:
145/// 1. The device type ([`PciDeviceType`]);
146/// 2. Whether the device has multiple functions (`bool`).
147#[derive(Clone, Copy, Debug, Eq, PartialEq)]
148struct PciHeaderType {
149    device_type: PciDeviceType,
150    has_multi_funcs: bool,
151}
152
153impl PciHeaderType {
154    /// Converts a byte into a header type.
155    ///
156    /// According to the PCI specification, the encoding of a header type is as follows:
157    /// - Bit 0-6 encodes the raw value of `PciDeviceType`;
158    /// - Bit 7 indicates whether the PCI device has multiple functions.
159    pub fn try_from_raw(raw: u8) -> Option<Self> {
160        let device_type = PciDeviceType::try_from_raw(raw & 0x7F)?;
161        let has_multi_funcs = (raw & 0x80) != 0;
162
163        Some(Self {
164            device_type,
165            has_multi_funcs,
166        })
167    }
168
169    /// Returns the device type.
170    pub fn device_type(self) -> PciDeviceType {
171        self.device_type
172    }
173
174    /// Returns whether the device has multiple functions.
175    pub fn has_multi_funcs(self) -> bool {
176        self.has_multi_funcs
177    }
178}

180/// Represents the type of PCI device, determined by the device's header type.
181///
182/// Used to distinguish between general devices, PCI-to-PCI bridges, and PCI-to-Cardbus bridges.
183#[repr(u8)]
184#[derive(Clone, Copy, Debug, Eq, PartialEq)]
185pub enum PciDeviceType {
186    /// General PCI device (header type 0x00).
187    GeneralDevice,
188    /// PCI-to-PCI bridge (header type 0x01).
189    /// Contains the primary, secondary, and subordinate bus numbers.
190    PciToPciBridge(u8, u8, u8),
191    /// PCI-to-Cardbus bridge (header type 0x02).
192    PciToCardbusBridge,
193}
194
195impl PciDeviceType {
196    /// Converts a raw header type value into a `PciDeviceType`.
197    pub fn try_from_raw(raw: u8) -> Option<Self> {
198        match raw {
199            0x00 => Some(PciDeviceType::GeneralDevice),
200            0x01 => Some(PciDeviceType::PciToPciBridge(0, 0, 0)),
201            0x02 => Some(PciDeviceType::PciToCardbusBridge),
202            _ => None,
203        }
204    }
205}

207/// Base Address Register (BAR) manager.
208#[derive(Debug)]
209pub struct BarManager {
210    /// There are at most 6 BARs in PCI device.
211    bars: [Option<Bar>; 6],
212}
213
214impl BarManager {
215    /// Gains access to the BAR space and returns `None` if that BAR is absent.
216    pub fn bar(&self, idx: u8) -> Option<&Bar> {
217        self.bars[idx as usize].as_ref()
218    }
219
220    /// Gains mutable access to the BAR space and returns `None` if that BAR is absent.
221    pub fn bar_mut(&mut self, idx: u8) -> Option<&mut Bar> {
222        self.bars[idx as usize].as_mut()
223    }
224
225    /// Parses the BAR space by PCI device location.
226    fn new(device_type: PciDeviceType, location: PciDeviceLocation) -> Self {
227        let mut bars = [None, None, None, None, None, None];
228
229        // Determine the maximum number of BARs based on the device type.
230        let max = match device_type {
231            PciDeviceType::GeneralDevice => 6,
232            PciDeviceType::PciToPciBridge(_, _, _) => 2,
233            PciDeviceType::PciToCardbusBridge => 0,
234        };
235        let mut idx = 0;
236        while idx < max {
237            let mut idx_step = 1;
238            let Ok(bar) = Bar::new(location, idx) else {
239                idx += idx_step;
240                continue;
241            };
242
243            if let Bar::Memory(memory_bar) = &bar
244                && memory_bar.address_length() == AddrLen::Bits64
245            {
246                // 64-bit BAR occupies two BAR slots.
247                idx_step += 1;
248            }
249
250            bars[idx as usize] = Some(bar);
251            idx += idx_step;
252        }
253
254        Self { bars }
255    }
