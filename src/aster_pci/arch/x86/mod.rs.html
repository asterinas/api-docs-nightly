3//! PCI bus access
4
5use ostd::{
6    Error,
7    arch::device::io_port::{ReadWriteAccess, WriteOnlyAccess},
8    io::IoPort,
9};
10use spin::Once;
11
12use crate::device_info::PciDeviceLocation;
13
14static PCI_ADDRESS_PORT: Once<IoPort<u32, WriteOnlyAccess>> = Once::new();
15static PCI_DATA_PORT: Once<IoPort<u32, ReadWriteAccess>> = Once::new();
16
17const BIT32_ALIGN_MASK: u32 = 0xFFFC;
18
19pub(crate) fn write32(location: &PciDeviceLocation, offset: u32, value: u32) -> Result<(), Error> {
20    PCI_ADDRESS_PORT
21        .get()
22        .ok_or(Error::IoError)?
23        .write(encode_as_port(location) | (offset & BIT32_ALIGN_MASK));
24    PCI_DATA_PORT
25        .get()
26        .ok_or(Error::IoError)?
27        .write(value.to_le());
28    Ok(())
29}
30
31pub(crate) fn read32(location: &PciDeviceLocation, offset: u32) -> Result<u32, Error> {
32    PCI_ADDRESS_PORT
33        .get()
34        .ok_or(Error::IoError)?
35        .write(encode_as_port(location) | (offset & BIT32_ALIGN_MASK));
36    Ok(PCI_DATA_PORT.get().ok_or(Error::IoError)?.read().to_le())
37}
38
39/// Encodes the bus, device, and function into a port address for use with the PCI I/O port.
40fn encode_as_port(location: &PciDeviceLocation) -> u32 {
41    // 1 << 31: Configuration enable
42    (1 << 31)
43        | ((location.bus as u32) << 16)
44        | (((location.device as u32) & 0b11111) << 11)
45        | (((location.function as u32) & 0b111) << 8)
46}
47
48pub(crate) fn has_pci_bus() -> bool {
49    true
50}
51
52pub(crate) fn init() {
53    // We use `acquire_overlapping` to acquire the port at 0xCF8 because 0xCF9 may be used as a
54    // reset control register in the PIIX4. Although the two ports overlap in their I/O range, they
55    // serve completely different purposes. See
56    // <https://www.intel.com/Assets/PDF/datasheet/290562.pdf>.
57    PCI_ADDRESS_PORT.call_once(|| IoPort::acquire_overlapping(0xCF8).unwrap());
58    PCI_DATA_PORT.call_once(|| IoPort::acquire(0xCFC).unwrap());
59}

61pub(crate) const MSIX_DEFAULT_MSG_ADDR: u32 = 0xFEE0_0000;
62
63pub(crate) fn construct_remappable_msix_address(remapping_index: u32) -> u32 {
64    // Use remappable format. The bits[4:3] should be always set to 1 according to the manual.
65    let mut address = MSIX_DEFAULT_MSG_ADDR | 0b1_1000;
66
67    // Interrupt index[14:0] is on address[19:5] and interrupt index[15] is on address[2].
68    address |= (remapping_index & 0x7FFF) << 5;
69    address |= (remapping_index & 0x8000) >> 13;
70
71    address
72}
