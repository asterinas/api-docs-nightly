21struct AddressAndDataPort {
22    address_port: IoPort<u32, WriteOnlyAccess>,
23    data_port: IoPort<u32, ReadWriteAccess>,
24}
25
26static PCI_PIO_CFG_SPACE: Once<SpinLock<AddressAndDataPort>> = Once::new();
27
28static PCI_ECAM_CFG_SPACE: Once<IoMem> = Once::new();
29
30pub(crate) fn write32(location: &PciDeviceLocation, offset: u32, value: u32) -> Result<(), Error> {
31    if let Some(ecam) = PCI_ECAM_CFG_SPACE.get() {
32        if offset > PCI_ECAM_MAX_OFFSET {
33            return Err(Error::InvalidArgs);
34        }
35        ecam.write_once(
36            (encode_as_address_offset(location) | offset) as usize,
37            &value,
38        )?;
39        return Ok(());
40    }
41
42    if let Some(pio_ports) = PCI_PIO_CFG_SPACE.get() {
43        if offset > PCI_PIO_MAX_OFFSET {
44            return Err(Error::InvalidArgs);
45        }
46        let pio = pio_ports.lock();
47        pio.address_port.write(encode_as_port(location) | offset);
48        pio.data_port.write(value.to_le());
49        return Ok(());
50    }
51
52    Err(Error::IoError)
53}
54
55pub(crate) fn read32(location: &PciDeviceLocation, offset: u32) -> Result<u32, Error> {
56    if let Some(ecam) = PCI_ECAM_CFG_SPACE.get() {
57        if offset > PCI_ECAM_MAX_OFFSET {
58            return Err(Error::InvalidArgs);
59        }
60        return ecam.read_once((encode_as_address_offset(location) | offset) as usize);
61    }
62
63    if let Some(pio_ports) = PCI_PIO_CFG_SPACE.get() {
64        if offset > PCI_PIO_MAX_OFFSET {
65            return Err(Error::InvalidArgs);
66        }
67        let pio = pio_ports.lock();
68        pio.address_port.write(encode_as_port(location) | offset);
69        return Ok(pio.data_port.read().to_le());
70    }
71
72    Err(Error::IoError)
73}

75/// The maximum offset in the 12-bit configuration space when using [`encode_as_address_offset`].
76const PCI_ECAM_MAX_OFFSET: u32 = 0xffc;
77
78/// Encodes the bus, device, and function into an address offset in the PCI MMIO region.
79fn encode_as_address_offset(location: &PciDeviceLocation) -> u32 {
80    ((location.bus as u32) << 20)
81        | ((location.device as u32) << 15)
82        | ((location.function as u32) << 12)
83}
84
85/// The maximum offset in the 8-bit configuration space when using [`encode_as_port`].
86const PCI_PIO_MAX_OFFSET: u32 = 0xfc;
87
88/// Encodes the bus, device, and function into a port address for use with the PCI I/O port.
89fn encode_as_port(location: &PciDeviceLocation) -> u32 {
90    // 1 << 31: Configuration enable
91    (1 << 31)
92        | ((location.bus as u32) << 16)
93        | (((location.device as u32) & 0b11111) << 11)
94        | (((location.function as u32) & 0b111) << 8)
95}

97/// Initializes the platform-specific module for accessing the PCI configuration space.
98///
99/// Returns a range for the PCI bus number, or [`None`] if there is no PCI bus.
100pub(crate) fn init() -> Option<RangeInclusive<u8>> {
101    if let Some(ecam) = ACPI_INFO.get().unwrap().pci_ecam_region.as_ref() {
102        let bus_start = ecam.bus_start;
103        let bus_end = ecam.bus_end;
104
105        let addr_start = ecam.base_address as usize;
106        // Note that the base address always corresponds to the bus number 0, regardless of the
107        // actual value of `bus_start`.
108        let addr_end = addr_start + (bus_end as usize + 1) * (1 << 20);
109        PCI_ECAM_CFG_SPACE.call_once(|| IoMem::acquire(addr_start..addr_end).unwrap());
110
111        return Some(bus_start..=bus_end);
112    }
113
114    // We use `acquire_overlapping` to acquire the port at 0xCF8 because 0xCF9 may be used as a
115    // reset control register in the PIIX4. Although the two ports overlap in their I/O range, they
116    // serve completely different purposes. See
117    // <https://www.intel.com/Assets/PDF/datasheet/290562.pdf>.
118    let address_port = IoPort::acquire_overlapping(0xCF8).unwrap();
119    let data_port = IoPort::acquire(0xCFC).unwrap();
120    PCI_PIO_CFG_SPACE.call_once(move || {
121        SpinLock::new(AddressAndDataPort {
122            address_port,
123            data_port,
124        })
125    });
126
127    Some(0..=255)
128}

130pub(crate) const MSIX_DEFAULT_MSG_ADDR: u32 = 0xFEE0_0000;
131
132pub(crate) fn construct_remappable_msix_address(remapping_index: u32) -> u32 {
133    // Use remappable format. The bits[4:3] should be always set to 1 according to the manual.
134    let mut address = MSIX_DEFAULT_MSG_ADDR | 0b1_1000;
135
136    // Interrupt index[14:0] is on address[19:5] and interrupt index[15] is on address[2].
137    address |= (remapping_index & 0x7FFF) << 5;
138    address |= (remapping_index & 0x8000) >> 13;
139
140    address
141}
