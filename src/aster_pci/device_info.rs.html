3//! PCI device Information

5use crate::cfg_space::PciCommonCfgOffset;

7/// PCI device location.
8#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord)]
9pub struct PciDeviceLocation {
10    /// Bus number
11    pub bus: u8,
12    /// Device number with max 31
13    pub device: u8,
14    /// Device number with max 7
15    pub function: u8,
16}
17
18impl PciDeviceLocation {
19    pub const MIN_DEVICE: u8 = 0;
20    pub const MAX_DEVICE: u8 = 31;
21
22    pub const MIN_FUNCTION: u8 = 0;
23    pub const MAX_FUNCTION: u8 = 7;
24}
25
26impl PciDeviceLocation {
27    /// Reads an 8-bit value from the PCI configuration space at the specified offset.
28    pub fn read8(&self, offset: u16) -> u8 {
29        let val = self.read32(offset & !0b11);
30        ((val >> ((offset as usize & 0b11) << 3)) & 0xFF) as u8
31    }
32
33    /// Reads a 16-bit value from the PCI configuration space at the specified offset.
34    pub fn read16(&self, offset: u16) -> u16 {
35        debug_assert!(
36            (offset & 0b1) == 0,
37            "misaligned PCI configuration dword u16 read"
38        );
39
40        let val = self.read32(offset & !0b11);
41        ((val >> ((offset as usize & 0b10) << 3)) & 0xFFFF) as u16
42    }
43
44    /// Reads a 32-bit value from the PCI configuration space at the specified offset.
45    pub fn read32(&self, offset: u16) -> u32 {
46        debug_assert!(
47            (offset & 0b11) == 0,
48            "misaligned PCI configuration dword u32 read"
49        );
50
51        crate::arch::read32(self, offset as u32).unwrap()
52    }
53
54    /// Writes an 8-bit value to the PCI configuration space at the specified offset.
55    pub fn write8(&self, offset: u16, val: u8) {
56        let old = self.read32(offset & !0b11);
57        let dest = (offset as usize & 0b11) << 3;
58        let mask = (0xFF << dest) as u32;
59        self.write32(
60            offset & !0b11,
61            (((val as u32) << dest) | (old & !mask)).to_le(),
62        );
63    }
64
65    /// Writes a 16-bit value to the PCI configuration space at the specified offset.
66    pub fn write16(&self, offset: u16, val: u16) {
67        debug_assert!(
68            (offset & 0b1) == 0,
69            "misaligned PCI configuration dword u16 write"
70        );
71
72        let old = self.read32(offset & !0b11);
73        let dest = (offset as usize & 0b10) << 3;
74        let mask = (0xFFFF << dest) as u32;
75        self.write32(
76            offset & !0b11,
77            (((val as u32) << dest) | (old & !mask)).to_le(),
78        );
79    }
80
81    /// Writes a 32-bit value to the PCI configuration space at the specified offset.
82    pub fn write32(&self, offset: u16, val: u32) {
83        debug_assert!(
84            (offset & 0b11) == 0,
85            "misaligned PCI configuration dword u32 write"
86        );
87
88        crate::arch::write32(self, offset as u32, val).unwrap()
89    }
90}

92/// PCI device ID
93#[derive(Debug, Copy, Clone, PartialEq, Eq)]
94pub struct PciDeviceId {
95    /// Vendor ID
96    pub vendor_id: u16,
97    /// Device ID
98    pub device_id: u16,
99    /// Revision ID
100    pub revision_id: u8,
101    /// Programming Interface Byte
102    pub prog_if: u8,
103    /// Specifies the specific function the device performs.
104    pub subclass: u8,
105    /// Specifies the type of function the device performs.
106    pub class: u8,
107}
108
109impl PciDeviceId {
110    pub(super) fn new(location: PciDeviceLocation) -> Self {
111        let vendor_id = location.read16(PciCommonCfgOffset::VendorId as u16);
112        let device_id = location.read16(PciCommonCfgOffset::DeviceId as u16);
113        let revision_id = location.read8(PciCommonCfgOffset::RevisionId as u16);
114        let prog_if = location.read8(PciCommonCfgOffset::ProgIf as u16);
115        let subclass = location.read8(PciCommonCfgOffset::SubclassCode as u16);
116        let class = location.read8(PciCommonCfgOffset::BaseClassCode as u16);
117        Self {
118            vendor_id,
119            device_id,
120            revision_id,
121            prog_if,
122            subclass,
123            class,
124        }
125    }
126}
