28static ENTROPY_DEVICE_ID: AtomicUsize = AtomicUsize::new(0);
29
30/// The prefix of Linux-compatible device names (`virtio_rng.N`).
31//
32// TODO: Export the names in sysfs (`/sys/class/misc/hw_random/rng_available`)
33// and allow the user to select the current device.
34const ENTROPY_DEVICE_PREFIX: &str = "virtio_rng.";
35
36/// The queue size for in-flight entropy requests.
37///
38/// A ring depth of 1 is sufficient because `try_read` submits a new entropy
39/// request only when none is in flight.
40const ENTROPY_QUEUE_SIZE: u16 = 1;
41
42/// The buffer size of an entropy request.
43const ENTROPY_BUFFER_SIZE: usize = PAGE_SIZE;
44
45/// Entropy devices, which supply high-quality randomness for guest use.
46pub struct EntropyDevice {
47    transport: SpinLock<Box<dyn VirtioTransport>>,
48    inner: SpinLock<EntropyDeviceInner, LocalIrqDisabled>,
49    /// A filled DMA buffer drained by user-space reads.
50    cache: Mutex<EntropyCache>,
51    wait_queue: WaitQueue,
52}
53
54impl EntropyDevice {
55    pub(crate) fn init(mut transport: Box<dyn VirtioTransport>) -> Result<(), VirtioDeviceError> {
56        let queue = VirtQueue::new(0, ENTROPY_QUEUE_SIZE, transport.as_mut())?;
57        let device = Arc::new(EntropyDevice {
58            transport: SpinLock::new(transport),
59            inner: SpinLock::new(EntropyDeviceInner::new(queue)?),
60            cache: Mutex::new(EntropyCache::new()?),
61            wait_queue: WaitQueue::new(),
62        });
63
64        let mut transport = device.transport.lock();
65
66        // Register IRQ callbacks.
67        transport.register_queue_callback(
68            0,
69            Box::new({
70                let device = Arc::downgrade(&device);
71                move |_: &TrapFrame| {
72                    if let Some(device) = device.upgrade() {
73                        device.handle_recv_irq()
74                    }
75                }
76            }),
77            false,
78        )?;
79        // Virtio-rng has no configuration fields, so config-space change interrupts
80        // are not expected and no config callback is registered.
81
82        transport.finish_init();
83        drop(transport);
84
85        let device_id = ENTROPY_DEVICE_ID.fetch_add(1, Ordering::Relaxed);
86        let name = format!("{ENTROPY_DEVICE_PREFIX}{device_id}");
87
88        entropy::register_device(name, device);
89
90        Ok(())
91    }
92
93    /// Attempts to read random data from cache into the given writer without blocking.
94    ///
95    /// Returns `Ok(Some(n))` with `n > 0` on success, `Ok(Some(0))` iff
96    /// `writer.avail() == 0`, and `Ok(None)` when no entropy is currently
97    /// available. The caller may block on [`Self::wait_queue`] and retry.
98    pub fn try_read(&self, writer: &mut VmWriter<'_, Fallible>) -> Result<Option<usize>, Error> {
99        let mut cache = self.cache.lock();
100
101        // Fast path: drain the cache under the mutex (`cache`) only.
102        if !cache.is_empty() {
103            return Ok(Some(cache.drain_into(writer)?));
104        }
105
106        // Slow path: under `inner`, lift a ready buffer into the cache (by
107        // handle swap — no memcpy), or submit a new request.
108        {
109            let mut inner = self.inner.lock();
110            if inner.ready_len > 0 {
111                let len = core::mem::replace(&mut inner.ready_len, 0);
112                cache.swap_in(&mut inner.dma_buf, len);
113                // Fall through to drain after releasing `inner`.
114            } else {
115                if !inner.in_flight {
116                    inner.submit();
117                }
118                return Ok(None);
119            }
120        }
121
122        // The user-space copy runs under the mutex (`cache`) only.
123        Ok(Some(cache.drain_into(writer)?))
124    }
125
126    /// Returns the wait queue callers can wait on; the device wakes it when
127    /// fresh entropy arrives.
128    pub fn wait_queue(&self) -> &WaitQueue {
129        &self.wait_queue
130    }
131
132    fn handle_recv_irq(&self) {
133        let mut inner = self.inner.lock();
134
135        let Ok((_, used_len)) = inner.queue.pop_used() else {
136            // No completed request was queued by the device, so the current
137            // in-flight request, if any, must remain pending.
138            return;
139        };
140
141        inner.ready_len = used_len as usize;
142        inner.in_flight = false;
143
144        drop(inner);
145
146        // Wake the waiters up so they can try again and see the new entropy.
147        self.wait_queue.wake_all();
148    }
149}

151struct EntropyDeviceInner {
152    queue: VirtQueue,
153    dma_buf: DmaStream<FromDevice>,
154    ready_len: usize,
155    in_flight: bool,
156}
157
158impl EntropyDeviceInner {
159    fn new(queue: VirtQueue) -> Result<Self, VirtioDeviceError> {
160        let dma_buf = DmaStream::<FromDevice>::alloc_uninit(ENTROPY_BUFFER_SIZE / PAGE_SIZE, false)
161            .map_err(VirtioDeviceError::ResourceAlloc)?;
162        Ok(Self {
163            queue,
164            dma_buf,
165            ready_len: 0,
166            in_flight: false,
167        })
168    }
169
170    fn submit(&mut self) {
171        let Self {
172            queue,
173            dma_buf,
174            in_flight,
175            ..
176        } = self;
177
178        queue.add_output_bufs(&[dma_buf]).unwrap();
179        if queue.should_notify() {
180            queue.notify();
181        }
182        *in_flight = true;
183    }
184}

186/// A filled DMA buffer waiting to be drained into user space.
187///
188/// Valid bytes live in `dma_buf[0..avail]` and are consumed from the tail.
189struct EntropyCache {
190    dma_buf: DmaStream<FromDevice>,
191    avail: usize,
192}
193
194impl EntropyCache {
195    fn new() -> Result<Self, VirtioDeviceError> {
196        let dma_buf = DmaStream::<FromDevice>::alloc_uninit(ENTROPY_BUFFER_SIZE / PAGE_SIZE, false)
197            .map_err(VirtioDeviceError::ResourceAlloc)?;
198        Ok(Self { dma_buf, avail: 0 })
199    }
200
201    fn is_empty(&self) -> bool {
202        self.avail == 0
203    }
204
205    /// Copies random data from the tail of the valid region
206    /// into `writer`, and marks those bytes as consumed.
207    fn drain_into(&mut self, writer: &mut VmWriter<'_, Fallible>) -> Result<usize, Error> {
208        let to_copy = writer.avail().min(self.avail);
209        let start = self.avail - to_copy;
210        let copied = self
211            .dma_buf
212            .reader()
213            .unwrap()
214            .skip(start)
215            .limit(to_copy)
216            .read_fallible(writer)
217            .map_err(|(err, _)| err)?;
218        self.avail = start;
219        Ok(copied)
220    }
221
222    /// Swaps the cache's drained buffer with the other DMA buffer.
223    ///
224    /// The incoming DMA buffer's first `len` bytes are treated as valid entropy.
225    /// The cache must be empty; otherwise unread entropy would be discarded.
226    fn swap_in(&mut self, other: &mut DmaStream<FromDevice>, len: usize) {
227        debug_assert!(self.is_empty());
228        other.sync_from_device(0..len).unwrap();
229        core::mem::swap(&mut self.dma_buf, other);
230        self.avail = len;
231    }
232}
