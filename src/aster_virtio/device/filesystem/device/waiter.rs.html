3//! Waitable reply state for submitted FUSE requests.
4//!
5//! This module defines [`FuseWaiter`], which lets synchronous callers block for
6//! a FUSE reply and lets asynchronous users integrate a request into an
7//! [`io_util::batch::IoBatch`].
8
9use aster_fuse::{FuseCompletion, FuseError, FuseOperation, FuseStatus, ReplyHeader};
10use io_util::{IoError, batch::IoCompletion};
11use ostd::{
12    mm::io::util::HasVmReaderWriter,
13    sync::{LocalIrqDisabled, SpinLock, WaitQueue},
14};
15
16use crate::device::filesystem::pool::FuseReplyBuf;
17
18/// Reply buffers owned by one submitted FUSE request.
19pub(super) struct ReplyBufs {
20    header: Option<FuseReplyBuf>,
21    payload: Option<FuseReplyBuf>,
22}
23
24impl ReplyBufs {
25    /// Creates an empty buffer set for a request that expects no reply.
26    pub(super) fn new_none() -> Self {
27        Self {
28            header: None,
29            payload: None,
30        }
31    }
32
33    /// Creates a buffer set for a reply that only contains a `ReplyHeader`.
34    pub(super) fn new_header_only(header: FuseReplyBuf) -> Self {
35        Self {
36            header: Some(header),
37            payload: None,
38        }
39    }
40
41    /// Creates a buffer set for a reply with a `ReplyHeader` and payload.
42    pub(super) fn new_with_payload(header: FuseReplyBuf, payload: FuseReplyBuf) -> Self {
43        Self {
44            header: Some(header),
45            payload: Some(payload),
46        }
47    }
48
49    /// Returns whether no reply buffer is expected.
50    pub(super) fn is_empty(&self) -> bool {
51        self.header.is_none()
52    }
53
54    /// Returns the reply header buffer.
55    pub(super) fn header(&self) -> Option<&FuseReplyBuf> {
56        self.header.as_ref()
57    }
58
59    /// Returns the reply payload buffer.
60    pub(super) fn payload(&self) -> Option<&FuseReplyBuf> {
61        self.payload.as_ref()
62    }
63
64    /// Returns the reply buffers in virtqueue descriptor order.
65    pub(super) fn iter(&self) -> impl Iterator<Item = &FuseReplyBuf> {
66        self.header.iter().chain(self.payload.iter())
67    }
68}

70/// A waiter for one submitted FUSE request.
71#[must_use]
72pub struct FuseWaiter {
73    reply_bufs: ReplyBufs,
74    status: SpinLock<FuseStatus, LocalIrqDisabled>,
75    wait_queue: WaitQueue,
76}
77
78impl FuseWaiter {
79    /// Creates a waiter for a request's reply buffers.
80    pub(super) fn new(reply_bufs: ReplyBufs) -> Self {
81        Self {
82            reply_bufs,
83            status: SpinLock::new(FuseStatus::Pending),
84            wait_queue: WaitQueue::new(),
85        }
86    }
87
88    fn reply_header_buf(&self) -> Result<&FuseReplyBuf, FuseError> {
89        let Some(header_buf) = self.reply_bufs.header() else {
90            return Err(FuseError::MalformedResponse);
91        };
92        Ok(header_buf)
93    }
94
95    /// Parses a typed FUSE operation reply from the payload bytes.
96    pub(super) fn parse_reply<Op: FuseOperation>(
97        &self,
98        payload_len: usize,
99    ) -> Result<Op::Output, FuseError> {
100        let mut reader = if let Some(payload_buf) = self.reply_bufs.payload() {
101            payload_buf.reader().unwrap()
102        } else {
103            let header_buf = self.reply_header_buf()?;
104            let mut reader = header_buf.reader().unwrap();
105            reader.skip(size_of::<ReplyHeader>());
106            reader
107        };
108
109        Op::parse_reply(payload_len, &mut reader)
110    }
111
112    /// Waits until the FUSE request completes.
113    ///
114    /// # Locking
115    ///
116    /// This method may sleep. Callers must not call it while holding a
117    /// spinlock, IRQ-disabled guard, or any other lock that cannot be held
118    /// across sleep.
119    pub(super) fn wait(&self) -> FuseCompletion {
120        // FIXME: There is no timeout logic. If the host virtio-fs server stalls,
121        // the guest driver task will block indefinitely. Adding timeout support
122        // is non-trivial: simply dropping the in-flight request is not safe,
123        // because the host may still hold descriptors pointing to the guest's
124        // DMA buffers. A proper timeout path would require restoring the
125        // virtqueue state, which in turn likely necessitates a full device reset.
126        self.wait_queue.wait_until(|| {
127            let status = *self.status.lock();
128            status.has_completed()
129        })
130    }
131
132    /// Records completion and wakes waiters.
133    pub(super) fn wake_completed(&self, completion: FuseCompletion) {
134        let should_wake = {
135            let mut current_status = self.status.lock();
136            if !current_status.is_pending() {
137                false
138            } else {
139                *current_status = FuseStatus::Completed(completion);
140                true
141            }
142        };
143
144        if should_wake {
145            self.wait_queue.wake_all();
146        }
147    }
148
149    /// Returns reply DMA buffers that hold the FUSE reply.
150    pub(super) fn reply_bufs(&self) -> &ReplyBufs {
151        &self.reply_bufs
152    }
153}
154
155impl IoCompletion for FuseWaiter {
156    fn wait(&self) -> Result<(), IoError> {
157        match self.wait() {
158            FuseCompletion::Complete(_) => Ok(()),
159            FuseCompletion::MalformedResponse | FuseCompletion::RemoteError(_) => {
160                Err(IoError::Failed)
161            }
162        }
163    }
164}
