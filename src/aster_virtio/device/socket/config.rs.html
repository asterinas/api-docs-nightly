3use core::mem::offset_of;
4
5use aster_util::safe_ptr::SafePtr;
6use bitflags::bitflags;
7
8use crate::transport::{ConfigManager, VirtioTransport};
9
10#[repr(C)]
11#[derive(Clone, Copy, Debug, Pod)]
12pub(super) struct VirtioVsockConfig {
13    guest_cid_low: u32,
14    guest_cid_high: u32,
15}
16
17impl VirtioVsockConfig {
18    pub(super) fn new_manager(transport: &dyn VirtioTransport) -> ConfigManager<Self> {
19        ConfigManager::new(
20            transport
21                .device_config_mem()
22                .map(|memory| SafePtr::new(memory, 0)),
23            transport.device_config_bar(),
24        )
25    }
26
27    pub(super) fn read_guest_cid(config_manager: &ConfigManager<Self>) -> u64 {
28        // We have no way to read the 64-bit CID atomically because the virtio specification says
29        // "Drivers MUST NOT assume reads from fields greater than 32 bits wide are atomic." For
30        // now, race conditions do not matter because the high 32 bits are always zeros and are
31        // reserved. Even if the CID changes concurrently, we should also receive a `TransportReset`
32        // event so we will reload the CID later.
33        let guest_cid_low = config_manager
34            .read_once::<u32>(offset_of!(Self, guest_cid_low))
35            .unwrap();
36        let guest_cid_high = config_manager
37            .read_once::<u32>(offset_of!(Self, guest_cid_high))
38            .unwrap();
39        (u64::from(guest_cid_high) << 32) | u64::from(guest_cid_low)
40    }
41}
42
43bitflags! {
44    pub(super) struct VsockFeatures: u64 {
45        const VIRTIO_VSOCK_F_STREAM    = 1 << 0;
46        const VIRTIO_VSOCK_F_SEQPACKET = 1 << 1;
47    }
48}
49
50impl VsockFeatures {
51    pub(super) const fn supported_features() -> Self {
52        Self::VIRTIO_VSOCK_F_STREAM
53    }
54}
