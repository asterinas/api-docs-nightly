22pub struct ConsoleDevice {
23    config_manager: ConfigManager<VirtioConsoleConfig>,
24    transport: SpinLock<Box<dyn VirtioTransport>>,
25    receive_queue: SpinLock<VirtQueue>,
26    transmit_queue: SpinLock<VirtQueue>,
27    send_buffer: Arc<DmaStream>,
28    receive_buffer: Arc<DmaStream>,
29    #[expect(clippy::box_collection)]
30    callbacks: Rcu<Box<Vec<&'static ConsoleCallback>>>,
31}
32
33impl AnyConsoleDevice for ConsoleDevice {
34    fn send(&self, value: &[u8]) {
35        let mut transmit_queue = self.transmit_queue.disable_irq().lock();
36        let mut reader = VmReader::from(value);
37
38        while reader.remain() > 0 {
39            let mut writer = self.send_buffer.writer().unwrap();
40            let len = writer.write(&mut reader);
41            self.send_buffer.sync_to_device(0..len).unwrap();
42
43            let slice = Slice::new(&self.send_buffer, 0..len);
44            transmit_queue.add_dma_buf(&[&slice], &[]).unwrap();
45
46            if transmit_queue.should_notify() {
47                transmit_queue.notify();
48            }
49            while !transmit_queue.can_pop() {
50                spin_loop();
51            }
52            transmit_queue.pop_used().unwrap();
53        }
54    }
55
56    fn register_callback(&self, callback: &'static ConsoleCallback) {
57        loop {
58            let callbacks = self.callbacks.read();
59            let mut callbacks_cloned = callbacks.get().clone();
60            callbacks_cloned.push(callback);
61            if callbacks.compare_exchange(callbacks_cloned).is_ok() {
62                break;
63            }
64            // Contention on pushing, retry.
65            core::hint::spin_loop();
66        }
67    }
68}

81impl ConsoleDevice {
82    pub fn negotiate_features(features: u64) -> u64 {
83        let mut features = ConsoleFeatures::from_bits_truncate(features);
84        // A virtio console device may have multiple ports, but we only use one port to communicate now.
85        features.remove(ConsoleFeatures::VIRTIO_CONSOLE_F_MULTIPORT);
86        features.bits()
87    }
88
89    pub fn init(mut transport: Box<dyn VirtioTransport>) -> Result<(), VirtioDeviceError> {
90        let config_manager = VirtioConsoleConfig::new_manager(transport.as_ref());
91        debug!("virtio_console_config = {:?}", config_manager.read_config());
92
93        const RECV0_QUEUE_INDEX: u16 = 0;
94        const TRANSMIT0_QUEUE_INDEX: u16 = 1;
95        let receive_queue =
96            SpinLock::new(VirtQueue::new(RECV0_QUEUE_INDEX, 2, transport.as_mut()).unwrap());
97        let transmit_queue =
98            SpinLock::new(VirtQueue::new(TRANSMIT0_QUEUE_INDEX, 2, transport.as_mut()).unwrap());
99
100        let send_buffer = Arc::new(DmaStream::alloc(1, false).unwrap());
101
102        let receive_buffer = Arc::new(DmaStream::alloc(1, false).unwrap());
103
104        let device = Arc::new(Self {
105            config_manager,
106            transport: SpinLock::new(transport),
107            receive_queue,
108            transmit_queue,
109            send_buffer,
110            receive_buffer,
111            callbacks: Rcu::new(Box::new(Vec::new())),
112        });
113
114        device.activate_receive_buffer(&mut device.receive_queue.disable_irq().lock());
115
116        // Register irq callbacks
117        let mut transport = device.transport.disable_irq().lock();
118        let handle_console_input = {
119            let device = device.clone();
120            move |_: &TrapFrame| device.handle_recv_irq()
121        };
122        transport
123            .register_queue_callback(RECV0_QUEUE_INDEX, Box::new(handle_console_input), false)
124            .unwrap();
125        transport
126            .register_cfg_callback(Box::new(config_space_change))
127            .unwrap();
128        transport.finish_init();
129        drop(transport);
130
131        aster_console::register_device(DEVICE_NAME.to_string(), device);
132
133        Ok(())
134    }
135
136    fn handle_recv_irq(&self) {
137        let mut receive_queue = self.receive_queue.disable_irq().lock();
138
139        let Ok((_, len)) = receive_queue.pop_used() else {
140            return;
141        };
142        self.receive_buffer
143            .sync_from_device(0..len as usize)
144            .unwrap();
145
146        let callbacks = self.callbacks.read();
147        for callback in callbacks.get().iter() {
148            let mut reader = self.receive_buffer.reader().unwrap();
149            reader.limit(len as usize);
150            callback(reader);
151        }
152        drop(callbacks);
153
154        self.activate_receive_buffer(&mut receive_queue);
155    }
156
157    fn activate_receive_buffer(&self, receive_queue: &mut VirtQueue) {
158        receive_queue
159            // We limit the buffer length to one to work around a QEMU bug that causes incorrect
160            // results when pasting more than 32 bytes into the virtio console. This has no
161            // performance penalty, since QEMU always gets one byte at a time, regardless of
162            // whether we have this limit or not.
163            //
164            // For the QEMU bug, see details at
165            // <https://lore.kernel.org/qemu-devel/20240707111940.232549-3-lrh2000@pku.edu.cn/T/#u>.
166            .add_dma_buf(&[], &[&Slice::new(&self.receive_buffer, 0..1)])
167            .unwrap();
168
169        if receive_queue.should_notify() {
170            receive_queue.notify();
171        }
172    }
173}
