12bitflags! {
13    /// Virtio Net Feature bits.
14    pub struct NetworkFeatures: u64 {
15        const VIRTIO_NET_F_CSUM = 1 << 0;               // Device handles packets with partial checksum.
16        const VIRTIO_NET_F_GUEST_CSUM = 1 << 1;         // Driver handles packets with partial checksum
17        const VIRTIO_NET_F_CTRL_GUEST_OFFLOADS = 1 << 2;// Control channel offloads reconfiguration support
18        const VIRTIO_NET_F_MTU = 1 << 3;                // Device maximum MTU reporting is supported
19        const VIRTIO_NET_F_MAC = 1 << 5;                // Device has given MAC address.
20        const VIRTIO_NET_F_GUEST_TSO4 = 1 << 7;         // Driver can receive TSOv4.
21        const VIRTIO_NET_F_GUEST_TSO6 = 1 <<8;          // Driver can receive TSOv6.
22        const VIRTIO_NET_F_GUEST_ECN = 1 << 9;          // Driver can receive TSO with ECN.
23        const VIRTIO_NET_F_GUEST_UFO = 1 << 10;         // Driver can receive UFO.
24        const VIRTIO_NET_F_HOST_TSO4 = 1 << 11;         // Device can receive TSOv4.
25        const VIRTIO_NET_F_HOST_TSO6 = 1 <<12;          // Device can receive TSOv6.
26        const VIRTIO_NET_F_HOST_ECN = 1 << 13;          // Device can receive TSO with ECN.
27        const VIRTIO_NET_F_HOST_UFO = 1 << 14;          // Device can receive UFO.
28        const VIRTIO_NET_F_MRG_RXBUF = 1 << 15;         // Driver can merge receive buffers.
29        const VIRTIO_NET_F_STATUS = 1 << 16;            // Configuration status field is available.
30        const VIRTIO_NET_F_CTRL_VQ = 1 << 17;           // Control channel is available.
31        const VIRTIO_NET_F_CTRL_RX = 1 << 18;           // Control channel RX mode support.
32        const VIRTIO_NET_F_CTRL_VLAN = 1 << 19;         // Control channel VLAN filtering.
33        const VIRTIO_NET_F_EXTRA = 1 << 20;             //
34        const VIRTIO_NET_F_GUEST_ANNOUNCE = 1 << 21;    // Driver can send gratuitous packets.
35        const VIRTIO_NET_F_MQ = 1 << 22;                // Device supports multiqueue with automatic receive steering.
36        const VIRTIO_NET_F_CTRL_MAC_ADDR = 1 << 23;     // Set MAC address through control channel.
37        const VIRTIO_NET_F_HASH_TUNNEL = 1 << 51;       // Device supports inner header hash for encapsulated packets.
38        const VIRTIO_NET_F_VQ_NOTF_COAL = 1 << 52;      // Device supports virtqueue notification coalescing.
39        const VIRTIO_NET_F_NOTF_COAL = 1 << 53;         // Device supports notifications coalescing.
40        const VIRTIO_NET_F_GUEST_USO4 = 1 << 54;        // Driver can receive USOv4 packets.
41        const VIRTIO_NET_F_GUEST_USO6 = 1 << 55;        // Driver can receive USOv6 packets.
42        const VIRTIO_NET_F_HOST_USO = 1 << 56;          // Device can receive USO packets.
43        const VIRTIO_NET_F_HASH_REPORT = 1 << 57;       // Device can report per-packet hash value and a type of calculated hash.
44        const VIRTIO_NET_F_GUEST_HDRLEN = 1 << 59;      // Driver can provide the exact hdr_len value. Device benefits from knowing the exact header length.
45        const VIRTIO_NET_F_RSS = 1 << 60;               // Device supports RSS (receive-side scaling) with Toeplitz hash calculation and configurable hash parameters for receive steering.
46        const VIRTIO_NET_F_RSC_EXT = 1 << 61;           // DevicecanprocessduplicatedACKsandreportnumberofcoalescedseg- ments and duplicated ACKs.
47        const VIRTIO_NET_F_STANDBY = 1 << 62;           // Device may act as a standby for a primary device with the same MAC address.
48        const VIRTIO_NET_F_SPEED_DUPLEX = 1 << 63;      // Device reports speed and duplex.
49    }
50}
51
52impl NetworkFeatures {
53    pub fn support_features() -> Self {
54        NetworkFeatures::VIRTIO_NET_F_MAC | NetworkFeatures::VIRTIO_NET_F_STATUS
55    }

58bitflags! {
59    #[repr(C)]
60    #[derive(Pod)]
61    pub struct Status: u16 {
62        const VIRTIO_NET_S_LINK_UP = 1;
63        const VIRTIO_NET_S_ANNOUNCE = 2;
64    }
65}
66
67#[derive(Debug, Clone, Copy, Pod)]
68#[repr(C)]
69pub struct VirtioNetConfig {
70    pub mac: EthernetAddr,
71    pub status: Status,
72    max_virtqueue_pairs: u16,
73    pub mtu: u16,
74    speed: u32,
75    duplex: u8,
76    rss_max_key_size: u8,
77    rss_max_indirection_table_length: u16,
78    supported_hash_types: u32,
79}
80
81impl VirtioNetConfig {
82    pub(super) fn new_manager(transport: &dyn VirtioTransport) -> ConfigManager<Self> {
83        let safe_ptr = transport
84            .device_config_mem()
85            .map(|mem| SafePtr::new(mem, 0));
86        let bar_space = transport.device_config_bar();
87        ConfigManager::new(safe_ptr, bar_space)
88    }
89}
90
91impl ConfigManager<VirtioNetConfig> {
92    pub(super) fn read_config(&self) -> VirtioNetConfig {
93        let mut net_config = VirtioNetConfig::new_zeroed();
94        // Only following fields are defined in legacy interface.
95        for i in 0..6 {
96            net_config.mac.0[i] = self
97                .read_once::<u8>(offset_of!(VirtioNetConfig, mac) + i)
98                .unwrap();
99        }
100        net_config.status.bits = self
101            .read_once::<u16>(offset_of!(VirtioNetConfig, status))
102            .unwrap();
103
104        if self.is_modern() {
105            net_config.max_virtqueue_pairs = self
106                .read_once::<u16>(offset_of!(VirtioNetConfig, max_virtqueue_pairs))
107                .unwrap();
108            net_config.mtu = self
109                .read_once::<u16>(offset_of!(VirtioNetConfig, mtu))
110                .unwrap();
111            net_config.speed = self
112                .read_once::<u32>(offset_of!(VirtioNetConfig, speed))
113                .unwrap();
114            net_config.duplex = self
115                .read_once::<u8>(offset_of!(VirtioNetConfig, duplex))
116                .unwrap();
117            net_config.rss_max_key_size = self
118                .read_once::<u8>(offset_of!(VirtioNetConfig, rss_max_key_size))
119                .unwrap();
120            net_config.rss_max_indirection_table_length = self
121                .read_once::<u16>(offset_of!(
122                    VirtioNetConfig,
123                    rss_max_indirection_table_length
124                ))
125                .unwrap();
126            net_config.supported_hash_types = self
127                .read_once::<u32>(offset_of!(VirtioNetConfig, supported_hash_types))
128                .unwrap();
129        }
130
131        net_config
132    }
