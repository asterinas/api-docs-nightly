3use id_alloc::IdAlloc;
4use ostd::sync::{SpinLock, WaitQueue};
5
6/// A synchronized ID allocator that manages a fixed-size pool of IDs.
7///
8/// We say this ID allocator "synchronized" because
9/// its `alloc` method may be used by multiple threads to allocate IDs concurrently
10/// and it would block if no free IDs are available at the moment.
11/// Once an ID is no longer in use, the `dealloc` method may be called
12/// to return the ID to the ID allocator.
13///
14/// This ID allocator is designed for use by device drivers.
15/// The `alloc` method should only be called in the task context,
16/// whereas the `dealloc` method should only be used in the IRQ handler of a device driver.
17/// Failing to conform with the above requirement may result in deadlock.
18pub struct SyncIdAlloc {
19    wait_queue: WaitQueue,
20    id_allocator: SpinLock<IdAlloc>,
21}
22
23impl SyncIdAlloc {
24    /// Creates an allocator that may return IDs in the range of `0..capacity`.
25    pub fn with_capacity(capacity: usize) -> Self {
26        Self {
27            wait_queue: WaitQueue::new(),
28            id_allocator: SpinLock::new(IdAlloc::with_capacity(capacity)),
29        }
30    }
31
32    /// Allocates a new ID.
33    ///
34    /// This method must only be called in the task context.
35    /// It will block until an ID is free to allocate.
36    pub fn alloc(&self) -> usize {
37        self.wait_queue
38            .wait_until(|| self.id_allocator.disable_irq().lock().alloc())
39    }
40
41    /// Deallocates an ID.
42    ///
43    /// This method assumes that the caller is in the context of an IRQ handler.
44    ///
45    /// # Panics
46    ///
47    /// This method would panic if `id` is greater than or equal to the capacity of this ID allocator.
48    pub fn dealloc(&self, id: usize) {
49        self.id_allocator.disable_irq().lock().free(id);
50        self.wait_queue.wake_all();
51    }
52}
53
54impl core::fmt::Debug for SyncIdAlloc {
55    fn fmt(&self, f: &mut core::fmt::Formatter) -> core::fmt::Result {
56        f.debug_struct("SyncIdAlloc")
57            .field("id_allocator", &self.id_allocator.lock())
58            .finish_non_exhaustive()
59    }
60}
