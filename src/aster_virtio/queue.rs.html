26/// The mechanism for bulk data transport on virtio devices.
27///
28/// Each device can have zero or more virtqueues.
29#[derive(Debug)]
30pub struct VirtQueue {
31    /// Descriptor table
32    descs: Vec<DescriptorSlot>,
33    /// Available ring
34    avail: SafePtr<AvailRing, Arc<DmaCoherent>>,
35    /// Used ring
36    used: SafePtr<UsedRing, Arc<DmaCoherent>>,
37    /// Notify configuration manager
38    notify_config: ConfigManager<u32>,
39
40    /// The index of the queue.
41    queue_idx: u32,
42    /// The size of the queue as seen by the device.
43    ///
44    /// This is the number of slots in the available and used rings. It can be larger than the
45    /// number of descriptors if the device expects a larger queue, but the driver expects a smaller
46    /// one.
47    ///
48    /// This is _not_ the queue size specified by the driver, which is `desc.len()`.
49    device_queue_size: u16,
50    /// The number of used descriptors.
51    num_used: u16,
52    /// The head descriptor index of the free list.
53    free_head: Option<u16>,
54    /// The next avail ring index.
55    avail_idx: u16,
56    /// The last-served used ring index.
57    last_used_idx: u16,
58    /// Whether the callback of this queue is enabled.
59    is_callback_enabled: bool,
60}
61

83#[derive(Debug)]
84struct DescriptorSlot {
85    /// The device-visible descriptor stored in DMA-coherent memory.
86    ///
87    /// This memory is shared with the device, so descriptor contents read from it
88    /// may be stale, corrupted, or otherwise untrusted after device access.
89    ptr: SafePtr<Descriptor, Arc<DmaCoherent>>,
90
91    /// The next descriptor in the current driver-managed list.
92    ///
93    /// For an in-use descriptor, this links to the next descriptor in the same
94    /// device-visible buffer chain. For a free descriptor, this links to the next
95    /// descriptor in the free list.
96    next: Option<u16>,
97    /// The total output buffer length for an in-use descriptor chain.
98    ///
99    /// This is set only on the head descriptor of an in-use chain. It records the
100    /// total number of bytes covered by all descriptors in that chain.
101    len: Option<u32>,
102}

104impl VirtQueue {
105    /// Creates a new virtqueue.
106    pub(crate) fn new(
107        idx: u16,
108        size: u16,
109        transport: &mut dyn VirtioTransport,
110    ) -> Result<Self, CreationError> {
111        if !size.is_power_of_two() {
112            return Err(CreationError::InvalidArgs);
113        }
114
115        let (descriptor_ptr, avail_ring_ptr, used_ring_ptr, device_queue_size) = if transport
116            .is_legacy_version()
117        {
118            let device_queue_size = transport
119                .max_queue_size(idx)
120                .map_err(CreationError::Transport)? as usize;
121            let desc_size = size_of::<Descriptor>() * device_queue_size;
122
123            // We should establish a reasonable upper bound on the requested queue size from the
124            // device, but we are unsure of the exact value. We chose 1024 based on the current
125            // need, but it can be increased in the future if necessary.
126            if !device_queue_size.is_power_of_two()
127                || size as usize > device_queue_size
128                || device_queue_size > 1024
129            {
130                return Err(CreationError::InvalidArgs);
131            }
132
133            let (dma1, dma2) = {
134                let align_size = VirtioPciLegacyTransport::QUEUE_ALIGN_SIZE;
135                let total_frames =
136                    VirtioPciLegacyTransport::calc_virtqueue_size_aligned(device_queue_size)
137                        / align_size;
138                let dma =
139                    DmaCoherent::alloc(total_frames, true).map_err(CreationError::ResourceAlloc)?;
140
141                let avail_size = size_of::<u16>() * (3 + device_queue_size);
142                let seg1_frames = (desc_size + avail_size).div_ceil(align_size);
143
144                dma.split(seg1_frames * align_size)
145            };
146
147            let desc_frame_ptr: SafePtr<Descriptor, Arc<DmaCoherent>> =
148                SafePtr::new(Arc::new(dma1), 0);
149            let mut avail_frame_ptr: SafePtr<AvailRing, Arc<DmaCoherent>> =
150                desc_frame_ptr.clone().cast();
151            avail_frame_ptr.byte_add(desc_size);
152            let used_frame_ptr: SafePtr<UsedRing, Arc<DmaCoherent>> =
153                SafePtr::new(Arc::new(dma2), 0);
154
155            (
156                desc_frame_ptr,
157                avail_frame_ptr,
158                used_frame_ptr,
159                device_queue_size as u16,
160            )
161        } else {
162            let max_queue_size = transport
163                .max_queue_size(idx)
164                .map_err(CreationError::Transport)? as usize;
165
166            // There can be a maximum of 256 descriptors on one page.
167            if size as usize > max_queue_size || size > 256 {
168                return Err(CreationError::InvalidArgs);
169            }
170
171            let desc_frame = DmaCoherent::alloc(1, true).map_err(CreationError::ResourceAlloc)?;
172            let avail_frame = DmaCoherent::alloc(1, true).map_err(CreationError::ResourceAlloc)?;
173            let used_frame = DmaCoherent::alloc(1, true).map_err(CreationError::ResourceAlloc)?;
174
175            (
176                SafePtr::new(Arc::new(desc_frame), 0),
177                SafePtr::new(Arc::new(avail_frame), 0),
178                SafePtr::new(Arc::new(used_frame), 0),
179                size,
180            )
181        };
182
183        debug!("queue_desc start paddr: {:x?}", descriptor_ptr.paddr());
184        debug!("queue_driver start paddr: {:x?}", avail_ring_ptr.paddr());
185        debug!("queue_device start paddr: {:x?}", used_ring_ptr.paddr());
186
187        transport
188            .set_queue(
189                idx,
190                device_queue_size,
191                &descriptor_ptr,
192                &avail_ring_ptr,
193                &used_ring_ptr,
194            )
195            .map_err(CreationError::Transport)?;
196
197        let mut descs = Vec::with_capacity(size as usize);
198        descs.push(DescriptorSlot {
199            ptr: descriptor_ptr,
200            next: None,
201            len: None,
202        });
203        for i in 0..size - 1 {
204            let last_desc = &mut descs[i as usize];
205            let new_desc = {
206                let mut ptr = last_desc.ptr.clone();
207                ptr.add(1);
208                DescriptorSlot {
209                    ptr,
210                    next: None,
211                    len: None,
212                }
213            };
214            last_desc.next = Some(i + 1);
215            descs.push(new_desc);
216        }
217
218        let notify_config = transport.notify_config(idx as usize);
219
220        field_ptr!(&avail_ring_ptr, AvailRing, flags)
221            .write_once(&AvailFlags::empty())
222            .unwrap();
223
224        Ok(VirtQueue {
225            descs,
226            avail: avail_ring_ptr,
227            used: used_ring_ptr,
228            notify_config,
229            queue_idx: idx as u32,
230            device_queue_size,
231            num_used: 0,
232            free_head: Some(0),
233            avail_idx: 0,
234            last_used_idx: 0,
235            is_callback_enabled: true,
236        })
237    }

253    /// Adds input and output DMA buffers to the virtqueue and returns a token.
254    ///
255    /// When successful, the result token is guaranteed to be valid. It will not exceed the queue
256    /// size, and the same token will not be returned twice, unless it has been removed from the
257    /// queue by [`Self::pop_used`] in the meantime.
258    ///
259    /// # Errors
260    ///
261    /// This method will return an error if:
262    /// - both `inputs` and `outputs` are empty; or
263    /// - the queue does not have enough free descriptors, that is, [`Self::available_desc`] is
264    ///   smaller than `inputs.len() + outputs.len()`.
265    ///
266    /// If at least one buffer is provided and enough descriptors are available, this method is
267    /// guaranteed to succeed. Callers that have already checked these conditions may unwrap the
268    /// result.
269    ///
270    /// Ref: linux virtio_ring.c virtqueue_add
271    pub fn add_dma_bufs<I: DmaBuf, O: DmaBuf>(
272        &mut self,
273        inputs: &[&I],
274        outputs: &[&O],
275    ) -> Result<u16, AddBufsError> {
276        if inputs.is_empty() && outputs.is_empty() {
277            return Err(AddBufsError::InvalidArgs);
278        }
279        if inputs.len() + outputs.len() > self.available_desc() {
280            return Err(AddBufsError::BufferTooSmall);
281        }
282
283        let head = self.free_head.unwrap();
284        let mut output_len = 0;
285
286        // Allocate descriptors from the free list.
287        let mut last = self.free_head;
288        let mut current = self.free_head;
289        for input in inputs.iter() {
290            let desc = &self.descs[current.unwrap() as usize];
291            set_dma_buf(
292                &desc.ptr.borrow_vm().restrict::<TRights![Write, Dup]>(),
293                *input,
294            );
295            field_ptr!(&desc.ptr, Descriptor, flags)
296                .write_once(&DescFlags::NEXT)
297                .unwrap();
298            if let Some(next) = desc.next {
299                field_ptr!(&desc.ptr, Descriptor, next)
300                    .write_once(&next)
301                    .unwrap();
302            }
303            last = current;
304            current = desc.next;
305        }
306        for output in outputs.iter() {
307            let desc = &self.descs[current.unwrap() as usize];
308            output_len += set_dma_buf(
309                &desc.ptr.borrow_vm().restrict::<TRights![Write, Dup]>(),
310                *output,
311            );
312            field_ptr!(&desc.ptr, Descriptor, flags)
313                .write_once(&(DescFlags::NEXT | DescFlags::WRITE))
314                .unwrap();
315            if let Some(next) = desc.next {
316                field_ptr!(&desc.ptr, Descriptor, next)
317                    .write_once(&next)
318                    .unwrap();
319            }
320            last = current;
321            current = desc.next;
322        }
323        // Clear `DescFlags::NEXT` in the last descriptor.
324        {
325            let desc = &mut self.descs[last.unwrap() as usize];
326            self.free_head = desc.next;
327            desc.next = None;
328            let mut flags: DescFlags = field_ptr!(&desc.ptr, Descriptor, flags)
329                .read_once()
330                .unwrap();
331            flags.remove(DescFlags::NEXT);
332            field_ptr!(&desc.ptr, Descriptor, flags)
333                .write_once(&flags)
334                .unwrap();
335        }
336        // Update the number of used descriptors.
337        self.num_used += (inputs.len() + outputs.len()) as u16;
338        // Store the length of the DMA buffer in the first descriptor.
339        self.descs[head as usize].len = Some(output_len);
340
341        {
342            let avail_slot = self.avail_idx & (self.device_queue_size - 1);
343            let ring_ptr: SafePtr<[u16; 64], &Arc<DmaCoherent>> =
344                field_ptr!(&self.avail, AvailRing, ring);
345            let mut ring_slot_ptr = ring_ptr.cast::<u16>();
346            ring_slot_ptr.add(avail_slot as usize);
347            ring_slot_ptr.write_once(&head).unwrap();
348        }
349        // Write barrier.
350        fence(Ordering::SeqCst);
351
352        // Increase the head index of the avail ring.
353        self.avail_idx = self.avail_idx.wrapping_add(1);
354        field_ptr!(&self.avail, AvailRing, idx)
355            .write_once(&self.avail_idx)
356            .unwrap();
357        // Write barrier.
358        fence(Ordering::SeqCst);
359
360        Ok(head)
361    }

363    /// Returns whether there is a used element that can pop.
364    ///
365    /// Even if this method returns true, [`Self::pop_used`] can still return `None`. See the note
366    /// about malfunctioning devices in [`Self::pop_used`] for details.
367    pub fn can_pop(&self) -> bool {
368        // Read barrier.
369        fence(Ordering::SeqCst);
370
371        self.last_used_idx != field_ptr!(&self.used, UsedRing, idx).read_once().unwrap()
372    }

374    /// Returns the number of free descriptors.
375    pub fn available_desc(&self) -> usize {
376        self.descs.len() - self.num_used as usize
377    }

379    /// Pops a device-used buffer and returns the token and the buffer length.
380    ///
381    /// When successful, the following is guaranteed:
382    /// - The token is valid.  It will not exceed the queue size. It was previously returned by
383    ///   [`Self::add_dma_bufs`], [`Self::add_input_bufs`], or [`Self::add_output_bufs`], and it has
384    ///   not yet been removed from the queue by this method.
385    /// - The length is valid. It will not exceed the length of the original DMA buffer.
386    ///
387    /// If the device malfunctions, it may report a token or length that violates these guarantees.
388    /// Such reports are logged as errors and ignored; the reported token is not returned to the
389    /// caller, preventing an invalid token from corrupting upper-layer state. If the device
390    /// continues to malfunction, the queue may become stuck because the affected buffer cannot be
391    /// reclaimed.
392    ///
393    /// # Errors
394    ///
395    /// This method will return an error if no valid buffers can be popped. Note that this can occur
396    /// even if [`Self::can_pop`] returns true because [`Self::can_pop`] does not check validity.
397    ///
398    /// Ref: linux virtio_ring.c virtqueue_get_buf_ctx
399    pub fn pop_used(&mut self) -> Result<(u16, u32), PopUsedError> {
400        self.pop_used_with_min_bytes(0)
401    }
402
403    /// Pops a device-used buffer, which is expected to contain at least `min_bytes` bytes, and
404    /// returns the token and the buffer length.
405    ///
406    /// This is the same as [`Self::pop_used`], except it also guarantees that the used buffer is at
407    /// least `min_bytes` bytes. The note about malfunctioning devices in [`Self::pop_used`] applies
408    /// here as well, including extra cases where the used buffer is shorter than `min_bytes`.
409    ///
410    /// For more information, see [`Self::pop_used`].
411    pub fn pop_used_with_min_bytes(
412        &mut self,
413        min_bytes: usize,
414    ) -> Result<(u16, u32), PopUsedError> {
415        loop {
416            if !self.can_pop() {
417                return Err(PopUsedError::NotReady);
418            }
419
420            let last_used_slot = self.last_used_idx & (self.device_queue_size - 1);
421            let element_ptr = {
422                let mut ptr = self.used.borrow_vm();
423                ptr.byte_add(offset_of!(UsedRing, ring) + last_used_slot as usize * 8);
424                ptr.cast::<UsedElem>()
425            };
426            let index = field_ptr!(&element_ptr, UsedElem, id).read_once().unwrap();
427            let len = field_ptr!(&element_ptr, UsedElem, len).read_once().unwrap();
428            self.last_used_idx = self.last_used_idx.wrapping_add(1);
429
430            let (desc, dma_len) = if let Some(desc) = self.descs.get_mut(index as usize)
431                && let Some(dma_len) = desc.len
432            {
433                (desc, dma_len)
434            } else {
435                ostd::error!(
436                    "invalid used token: {} (queue size: {})",
437                    index,
438                    self.descs.len(),
439                );
440                continue;
441            };
442            if len > dma_len || (len as usize) < min_bytes {
443                ostd::error!(
444                    "invalid used length: {} (expected {}..={})",
445                    len,
446                    min_bytes,
447                    dma_len,
448                );
449                continue;
450            }
451            desc.len = None;
452            self.recycle_descriptors(index as u16);
453
454            return Ok((index as u16, len));
455        }
456    }

458    /// Recycles descriptors in the list specified by `head`.
459    ///
460    /// This will push all linked descriptors at the front of the free list.
461    fn recycle_descriptors(&mut self, mut head: u16) {
462        let origin_free_head = self.free_head;
463        self.free_head = Some(head);
464
465        loop {
466            let desc = &mut self.descs[head as usize];
467            // Set the buffer address and length to 0.
468            field_ptr!(&desc.ptr, Descriptor, addr)
469                .write_once(&(0u64))
470                .unwrap();
471            field_ptr!(&desc.ptr, Descriptor, len)
472                .write_once(&(0u32))
473                .unwrap();
474            self.num_used -= 1;
475
476            if let Some(next) = desc.next {
477                head = next;
478            } else {
479                desc.next = origin_free_head;
480                break;
481            }
482        }
483    }

485    /// Returns whether the driver should notify the device.
486    pub fn should_notify(&self) -> bool {
487        // Read barrier.
488        fence(Ordering::SeqCst);
489
490        let flags = field_ptr!(&self.used, UsedRing, flags).read_once().unwrap();
491        flags & 0x0001u16 == 0u16
492    }

494    /// Notifies the device that there are available elements.
495    pub fn notify(&mut self) {
496        if self.notify_config.is_modern() {
497            self.notify_config
498                .write_once::<u32>(0, self.queue_idx)
499                .unwrap();
500        } else {
501            self.notify_config
502                .write_once::<u16>(0, self.queue_idx as u16)
503                .unwrap();
504        }
505    }

507    /// Disables registered callbacks.
508    ///
509    /// That is to say, the queue won't generate interrupts after calling this method.
510    pub fn disable_callback(&mut self) {
511        if !self.is_callback_enabled {
512            return;
513        }
514
515        let flags_ptr = field_ptr!(&self.avail, AvailRing, flags);
516        let mut flags: AvailFlags = flags_ptr.read_once().unwrap();
517        debug_assert!(!flags.contains(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT));
518        flags.insert(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT);
519        flags_ptr.write_once(&flags).unwrap();
520
521        self.is_callback_enabled = false;
522    }
523
524    /// Enables registered callbacks.
525    ///
526    /// The queue will generate interrupts if any event comes after calling this method.
527    pub fn enable_callback(&mut self) {
528        if self.is_callback_enabled {
529            return;
530        }
531
532        let flags_ptr = field_ptr!(&self.avail, AvailRing, flags);
533        let mut flags: AvailFlags = flags_ptr.read_once().unwrap();
534        debug_assert!(flags.contains(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT));
535        flags.remove(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT);
536        flags_ptr.write_once(&flags).unwrap();
537
538        self.is_callback_enabled = true;
539    }

542#[repr(C, align(16))]
543#[derive(Clone, Copy, Debug, Default, Pod)]
544pub struct Descriptor {
545    addr: u64,
546    len: u32,
547    flags: DescFlags,
548    next: u16,
549}

551type DescriptorPtr<'a> = SafePtr<Descriptor, &'a Arc<DmaCoherent>, TRightSet<TRights![Dup, Write]>>;
552
553fn set_dma_buf<T: DmaBuf>(desc_ptr: &DescriptorPtr, buf: &T) -> u32 {
554    let daddr = buf.daddr();
555    let len = buf.len();
556
557    debug_assert!(len < (u32::MAX) as usize);
558    // TODO: Should we skip the empty DMA buffer or just return an error?
559    debug_assert_ne!(len, 0);
560
561    field_ptr!(desc_ptr, Descriptor, addr)
562        .write_once(&(daddr as u64))
563        .unwrap();
564    field_ptr!(desc_ptr, Descriptor, len)
565        .write_once(&(len as u32))
566        .unwrap();
567
568    len as u32
569}

584/// The driver uses the available ring to offer buffers to the device:
585/// each ring entry refers to the head of a descriptor chain.
586/// It is only written by the driver and read by the device.
587#[repr(C, align(2))]
588#[derive(Clone, Copy, Debug, Pod)]
589pub struct AvailRing {
590    flags: AvailFlags,
591    /// A driver MUST NOT decrement the idx.
592    idx: u16,
593    ring: [u16; 64], // actual size: queue_size
594    used_event: u16, // unused
595}
596
597/// The used ring is where the device returns buffers once it is done with them.
598/// It is only written to by the device and read by the driver.
599#[padding_struct]
600#[repr(C, align(4))]
601#[derive(Clone, Copy, Debug, Pod)]
602pub struct UsedRing {
603    flags: u16,
604    /// The next index of the used element in the ring array.
605    idx: u16,
606    ring: [UsedElem; 64], // actual size: queue_size
607    avail_event: u16,     // unused
608}
609
610#[repr(C)]
611#[derive(Clone, Copy, Debug, Default, Pod)]
612pub struct UsedElem {
613    id: u32,
614    len: u32,
615}
