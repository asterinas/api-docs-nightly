31/// The mechanism for bulk data transport on virtio devices.
32///
33/// Each device can have zero or more virtqueues.
34#[derive(Debug)]
35pub struct VirtQueue {
36    /// Descriptor table
37    descs: Vec<SafePtr<Descriptor, Arc<DmaCoherent>>>,
38    /// Available ring
39    avail: SafePtr<AvailRing, Arc<DmaCoherent>>,
40    /// Used ring
41    used: SafePtr<UsedRing, Arc<DmaCoherent>>,
42    /// Notify configuration manager
43    notify_config: ConfigManager<u32>,
44
45    /// The index of queue
46    queue_idx: u32,
47    /// The size of the queue.
48    ///
49    /// This is both the number of descriptors, and the number of slots in the available and used
50    /// rings.
51    queue_size: u16,
52    /// The number of used queues.
53    num_used: u16,
54    /// The head desc index of the free list.
55    free_head: u16,
56    /// the index of the next avail ring index
57    avail_idx: u16,
58    /// last service used index
59    last_used_idx: u16,
60    /// Whether the callback of this queue is enabled
61    is_callback_enabled: bool,
62}
63
64impl VirtQueue {
65    /// Create a new VirtQueue.
66    pub(crate) fn new(
67        idx: u16,
68        mut size: u16,
69        transport: &mut dyn VirtioTransport,
70    ) -> Result<Self, QueueError> {
71        if !size.is_power_of_two() {
72            return Err(QueueError::InvalidArgs);
73        }
74
75        let (descriptor_ptr, avail_ring_ptr, used_ring_ptr) = if transport.is_legacy_version() {
76            // Currently, we use one UFrame to place the descriptors and available rings, one UFrame to place used rings
77            // because the virtio-mmio legacy required the address to be continuous. The max queue size is 128.
78            if size > 128 {
79                return Err(QueueError::InvalidArgs);
80            }
81            let queue_size = transport.max_queue_size(idx).unwrap() as usize;
82            let desc_size = size_of::<Descriptor>() * queue_size;
83            size = queue_size as u16;
84
85            let (dma1, dma2) = {
86                let align_size = VirtioPciLegacyTransport::QUEUE_ALIGN_SIZE;
87                let total_frames =
88                    VirtioPciLegacyTransport::calc_virtqueue_size_aligned(queue_size) / align_size;
89                let dma = DmaCoherent::alloc(total_frames, true).unwrap();
90
91                let avial_size = size_of::<u16>() * (3 + queue_size);
92                let seg1_frames = (desc_size + avial_size).div_ceil(align_size);
93
94                dma.split(seg1_frames * align_size)
95            };
96            let desc_frame_ptr: SafePtr<Descriptor, Arc<DmaCoherent>> =
97                SafePtr::new(Arc::new(dma1), 0);
98            let mut avail_frame_ptr: SafePtr<AvailRing, Arc<DmaCoherent>> =
99                desc_frame_ptr.clone().cast();
100            avail_frame_ptr.byte_add(desc_size);
101            let used_frame_ptr: SafePtr<UsedRing, Arc<DmaCoherent>> =
102                SafePtr::new(Arc::new(dma2), 0);
103            (desc_frame_ptr, avail_frame_ptr, used_frame_ptr)
104        } else {
105            if size > 256 {
106                return Err(QueueError::InvalidArgs);
107            }
108            (
109                SafePtr::new(Arc::new(DmaCoherent::alloc(1, true).unwrap()), 0),
110                SafePtr::new(Arc::new(DmaCoherent::alloc(1, true).unwrap()), 0),
111                SafePtr::new(Arc::new(DmaCoherent::alloc(1, true).unwrap()), 0),
112            )
113        };
114        debug!("queue_desc start paddr:{:x?}", descriptor_ptr.paddr());
115        debug!("queue_driver start paddr:{:x?}", avail_ring_ptr.paddr());
116        debug!("queue_device start paddr:{:x?}", used_ring_ptr.paddr());
117
118        transport
119            .set_queue(idx, size, &descriptor_ptr, &avail_ring_ptr, &used_ring_ptr)
120            .unwrap();
121        let mut descs = Vec::with_capacity(size as usize);
122        descs.push(descriptor_ptr);
123        for i in 0..size {
124            let mut desc = descs.get(i as usize).unwrap().clone();
125            let next_i = i + 1;
126            if next_i != size {
127                field_ptr!(&desc, Descriptor, next)
128                    .write_once(&next_i)
129                    .unwrap();
130                desc.add(1);
131                descs.push(desc);
132            } else {
133                field_ptr!(&desc, Descriptor, next)
134                    .write_once(&(0u16))
135                    .unwrap();
136            }
137        }
138
139        let notify_config = transport.notify_config(idx as usize);
140        field_ptr!(&avail_ring_ptr, AvailRing, flags)
141            .write_once(&AvailFlags::empty())
142            .unwrap();
143        Ok(VirtQueue {
144            descs,
145            avail: avail_ring_ptr,
146            used: used_ring_ptr,
147            notify_config,
148            queue_size: size,
149            queue_idx: idx as u32,
150            num_used: 0,
151            free_head: 0,
152            avail_idx: 0,
153            last_used_idx: 0,
154            is_callback_enabled: true,
155        })
156    }

158    /// Add dma buffers to the virtqueue, return a token.
159    ///
160    /// Ref: linux virtio_ring.c virtqueue_add
161    pub fn add_dma_buf<T: DmaBuf>(
162        &mut self,
163        inputs: &[&T],
164        outputs: &[&T],
165    ) -> Result<u16, QueueError> {
166        if inputs.is_empty() && outputs.is_empty() {
167            return Err(QueueError::InvalidArgs);
168        }
169        if inputs.len() + outputs.len() + self.num_used as usize > self.queue_size as usize {
170            return Err(QueueError::BufferTooSmall);
171        }
172
173        // allocate descriptors from free list
174        let head = self.free_head;
175        let mut last = self.free_head;
176        for input in inputs.iter() {
177            let desc = &self.descs[self.free_head as usize];
178            set_dma_buf(&desc.borrow_vm().restrict::<TRights![Write, Dup]>(), *input);
179            field_ptr!(desc, Descriptor, flags)
180                .write_once(&DescFlags::NEXT)
181                .unwrap();
182            last = self.free_head;
183            self.free_head = field_ptr!(desc, Descriptor, next).read_once().unwrap();
184        }
185        for output in outputs.iter() {
186            let desc = &mut self.descs[self.free_head as usize];
187            set_dma_buf(
188                &desc.borrow_vm().restrict::<TRights![Write, Dup]>(),
189                *output,
190            );
191            field_ptr!(desc, Descriptor, flags)
192                .write_once(&(DescFlags::NEXT | DescFlags::WRITE))
193                .unwrap();
194            last = self.free_head;
195            self.free_head = field_ptr!(desc, Descriptor, next).read_once().unwrap();
196        }
197        // set last_elem.next = NULL
198        {
199            let desc = &mut self.descs[last as usize];
200            let mut flags: DescFlags = field_ptr!(desc, Descriptor, flags).read_once().unwrap();
201            flags.remove(DescFlags::NEXT);
202            field_ptr!(desc, Descriptor, flags)
203                .write_once(&flags)
204                .unwrap();
205        }
206        self.num_used += (inputs.len() + outputs.len()) as u16;
207
208        let avail_slot = self.avail_idx & (self.queue_size - 1);
209
210        {
211            let ring_ptr: SafePtr<[u16; 64], &Arc<DmaCoherent>> =
212                field_ptr!(&self.avail, AvailRing, ring);
213            let mut ring_slot_ptr = ring_ptr.cast::<u16>();
214            ring_slot_ptr.add(avail_slot as usize);
215            ring_slot_ptr.write_once(&head).unwrap();
216        }
217        // write barrier
218        fence(Ordering::SeqCst);
219
220        // increase head of avail ring
221        self.avail_idx = self.avail_idx.wrapping_add(1);
222        field_ptr!(&self.avail, AvailRing, idx)
223            .write_once(&self.avail_idx)
224            .unwrap();
225
226        fence(Ordering::SeqCst);
227        Ok(head)
228    }

230    /// Whether there is a used element that can pop.
231    pub fn can_pop(&self) -> bool {
232        // read barrier
233        fence(Ordering::SeqCst);
234
235        self.last_used_idx != field_ptr!(&self.used, UsedRing, idx).read_once().unwrap()
236    }
237
238    /// The number of free descriptors.
239    pub fn available_desc(&self) -> usize {
240        (self.queue_size - self.num_used) as usize
241    }
242
243    /// Recycle descriptors in the list specified by head.
244    ///
245    /// This will push all linked descriptors at the front of the free list.
246    fn recycle_descriptors(&mut self, mut head: u16) {
247        let origin_free_head = self.free_head;
248        self.free_head = head;
249        loop {
250            let desc = &mut self.descs[head as usize];
251            // Sets the buffer address and length to 0
252            field_ptr!(desc, Descriptor, addr)
253                .write_once(&(0u64))
254                .unwrap();
255            field_ptr!(desc, Descriptor, len)
256                .write_once(&(0u32))
257                .unwrap();
258            self.num_used -= 1;
259
260            let flags: DescFlags = field_ptr!(desc, Descriptor, flags).read_once().unwrap();
261            if flags.contains(DescFlags::NEXT) {
262                field_ptr!(desc, Descriptor, flags)
263                    .write_once(&DescFlags::empty())
264                    .unwrap();
265                head = field_ptr!(desc, Descriptor, next).read_once().unwrap();
266            } else {
267                field_ptr!(desc, Descriptor, next)
268                    .write_once(&origin_free_head)
269                    .unwrap();
270                break;
271            }
272        }
273    }

275    /// Get a token from device used buffers, return (token, len).
276    ///
277    /// Ref: linux virtio_ring.c virtqueue_get_buf_ctx
278    pub fn pop_used(&mut self) -> Result<(u16, u32), QueueError> {
279        if !self.can_pop() {
280            return Err(QueueError::NotReady);
281        }
282
283        let last_used_slot = self.last_used_idx & (self.queue_size - 1);
284        let element_ptr = {
285            let mut ptr = self.used.borrow_vm();
286            ptr.byte_add(offset_of!(UsedRing, ring) + last_used_slot as usize * 8);
287            ptr.cast::<UsedElem>()
288        };
289        let index = field_ptr!(&element_ptr, UsedElem, id).read_once().unwrap();
290        let len = field_ptr!(&element_ptr, UsedElem, len).read_once().unwrap();
291
292        self.recycle_descriptors(index as u16);
293        self.last_used_idx = self.last_used_idx.wrapping_add(1);
294
295        Ok((index as u16, len))
296    }

298    /// If the given token is next on the device used queue, pops it and returns the total buffer
299    /// length which was used (written) by the device.
300    ///
301    /// Ref: linux virtio_ring.c virtqueue_get_buf_ctx
302    pub fn pop_used_with_token(&mut self, token: u16) -> Result<u32, QueueError> {
303        if !self.can_pop() {
304            return Err(QueueError::NotReady);
305        }
306
307        let last_used_slot = self.last_used_idx & (self.queue_size - 1);
308        let element_ptr = {
309            let mut ptr = self.used.borrow_vm();
310            ptr.byte_add(offset_of!(UsedRing, ring) + last_used_slot as usize * 8);
311            ptr.cast::<UsedElem>()
312        };
313        let index = field_ptr!(&element_ptr, UsedElem, id).read_once().unwrap();
314        let len = field_ptr!(&element_ptr, UsedElem, len).read_once().unwrap();
315
316        if index as u16 != token {
317            return Err(QueueError::WrongToken);
318        }
319
320        self.recycle_descriptors(index as u16);
321        self.last_used_idx = self.last_used_idx.wrapping_add(1);
322
323        Ok(len)
324    }

326    /// Return size of the queue.
327    pub fn size(&self) -> u16 {
328        self.queue_size
329    }
330
331    /// whether the driver should notify the device
332    pub fn should_notify(&self) -> bool {
333        // read barrier
334        fence(Ordering::SeqCst);
335        let flags = field_ptr!(&self.used, UsedRing, flags).read_once().unwrap();
336        flags & 0x0001u16 == 0u16
337    }
338
339    /// notify that there are available rings
340    pub fn notify(&mut self) {
341        if self.notify_config.is_modern() {
342            self.notify_config
343                .write_once::<u32>(0, self.queue_idx)
344                .unwrap();
345        } else {
346            self.notify_config
347                .write_once::<u16>(0, self.queue_idx as u16)
348                .unwrap();
349        }
350    }
351
352    /// Disables registered callbacks.
353    ///
354    /// That is to say, the queue won't generate interrupts after calling this method.
355    pub fn disable_callback(&mut self) {
356        if !self.is_callback_enabled {
357            return;
358        }
359
360        let flags_ptr = field_ptr!(&self.avail, AvailRing, flags);
361        let mut flags: AvailFlags = flags_ptr.read_once().unwrap();
362        debug_assert!(!flags.contains(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT));
363        flags.insert(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT);
364        flags_ptr.write_once(&flags).unwrap();
365
366        self.is_callback_enabled = false;
367    }
368
369    /// Enables registered callbacks.
370    ///
371    /// The queue will generate interrupts if any event comes after calling this method.
372    pub fn enable_callback(&mut self) {
373        if self.is_callback_enabled {
374            return;
375        }
376
377        let flags_ptr = field_ptr!(&self.avail, AvailRing, flags);
378        let mut flags: AvailFlags = flags_ptr.read_once().unwrap();
379        debug_assert!(flags.contains(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT));
380        flags.remove(AvailFlags::VIRTQ_AVAIL_F_NO_INTERRUPT);
381        flags_ptr.write_once(&flags).unwrap();
382
383        self.is_callback_enabled = true;
384    }

387#[repr(C, align(16))]
388#[derive(Debug, Default, Copy, Clone, Pod)]
389pub struct Descriptor {
390    addr: u64,
391    len: u32,
392    flags: DescFlags,
393    next: u16,
394}
395
396type DescriptorPtr<'a> = SafePtr<Descriptor, &'a Arc<DmaCoherent>, TRightSet<TRights![Dup, Write]>>;
397
398fn set_dma_buf<T: DmaBuf>(desc_ptr: &DescriptorPtr, buf: &T) {
399    // TODO: skip the empty dma buffer or just return error?
400    debug_assert_ne!(buf.len(), 0);
401    let daddr = buf.daddr();
402    field_ptr!(desc_ptr, Descriptor, addr)
403        .write_once(&(daddr as u64))
404        .unwrap();
405    field_ptr!(desc_ptr, Descriptor, len)
406        .write_once(&(buf.len() as u32))
407        .unwrap();
408}

411    /// Descriptor flags
412    #[derive(Pod, Default)]
413    #[repr(C)]
414    struct DescFlags: u16 {
415        const NEXT = 1;
416        const WRITE = 2;
417        const INDIRECT = 4;
418    }

423/// The driver uses the available ring to offer buffers to the device:
424/// each ring entry refers to the head of a descriptor chain.
425/// It is only written by the driver and read by the device.
426#[repr(C, align(2))]
427#[derive(Debug, Copy, Clone, Pod)]
428pub struct AvailRing {
429    flags: AvailFlags,
430    /// A driver MUST NOT decrement the idx.
431    idx: u16,
432    ring: [u16; 64], // actual size: queue_size
433    used_event: u16, // unused
434}
435
436/// The used ring is where the device returns buffers once it is done with them:
437/// it is only written to by the device, and read by the driver.
438#[repr(C, align(4))]
439#[padding_struct]
440#[derive(Debug, Copy, Clone, Pod)]
441pub struct UsedRing {
442    // the flag in UsedRing
443    flags: u16,
444    // the next index of the used element in ring array
445    idx: u16,
446    ring: [UsedElem; 64], // actual size: queue_size
447    avail_event: u16,     // unused
448}
449
450#[repr(C)]
451#[derive(Debug, Default, Copy, Clone, Pod)]
452pub struct UsedElem {
453    id: u32,
454    len: u32,
455}
