12/// A DMA-capable buffer.
13///
14/// Any type implements this trait should also implements `HasDaddr` trait,
15/// and provides the exact length of DMA area.
16pub trait DmaBuf: HasDaddr {
17    /// The length of Dma area, in bytes
18    fn len(&self) -> usize;
19}
20
21macro_rules! impl_dma_buf_for {
22    (<D> $t:ty) => {
23        impl<D: DmaDirection> DmaBuf for $t {
24            fn len(&self) -> usize {
25                self.size()
26            }
27        }
28
29        impl<D: DmaDirection> DmaBuf for Slice<$t> {
30            fn len(&self) -> usize {
31                self.size()
32            }
33        }
34    };
35    ($t:ty) => {
36        impl DmaBuf for $t {
37            fn len(&self) -> usize {
38                self.size()
39            }
40        }
41
42        impl DmaBuf for Slice<$t> {
43            fn len(&self) -> usize {
44                self.size()
45            }
46        }
47    };
48}
49
50impl_dma_buf_for!(<D> DmaStream<D>);
51impl_dma_buf_for!(<D> &DmaStream<D>);
52impl_dma_buf_for!(<D> Arc<DmaStream<D>>);
53impl_dma_buf_for!(<D> &Arc<DmaStream<D>>);
54impl_dma_buf_for!(DmaCoherent);
55impl_dma_buf_for!(&DmaCoherent);
56impl_dma_buf_for!(Arc<DmaCoherent>);
57impl_dma_buf_for!(&Arc<DmaCoherent>);
58
