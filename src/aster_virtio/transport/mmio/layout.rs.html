3use core::fmt::Debug;
4
5#[derive(Clone, Copy, Pod)]
6#[repr(C)]
7pub struct VirtioMmioLayout {
8    /// Magic value: 0x74726976. **Read-only**
9    pub magic_value: u32,
10    /// Device version. 1 => Legacy, 2 => Normal. **Read-only**
11    pub version: u32,
12    /// Virtio Subsystem Device ID. **Read-only**
13    pub device_id: u32,
14    /// Virtio Subsystem Vendor ID. **Read-only**
15    pub vendor_id: u32,
16
17    /// Flags representing features the device supports.
18    /// Bits 0-31 if `device_features_sel`is 0,
19    /// bits 32-63 if `device_features_sel` is 1.
20    /// **Read-only**
21    pub device_features: u32,
22    /// **Write-only**
23    pub device_features_select: u32,
24
25    __r1: [u8; 8],
26
27    /// Flags representing device features understood and activated by the driver.
28    /// Bits 0-31 if `driver_features_sel`is 0,
29    /// bits 32-63 if `driver_features_sel` is 1.
30    /// **Write-only**
31    pub driver_features: u32,
32    /// **Write-only**
33    pub driver_features_select: u32,
34
35    /// Guest page size.
36    ///
37    /// The driver writes the guest page size in bytes
38    /// to the register during initialization, before any queues are used.
39    ///
40    /// This value should be a power of 2 and is used by the device to
41    /// calculate the Guest address of the first queue page (see legacy_queue_pfn).
42    /// **Write-only**
43    pub legacy_guest_page_size: u32,
44
45    __r2: [u8; 4],
46
47    /// Selected queue. **Write-only**
48    pub queue_sel: u32,
49    /// Maximum virtual queue size. **Read-only**
50    pub queue_num_max: u32,
51    /// Virtual queue size. **Write-only**
52    pub queue_num: u32,
53
54    pub legacy_queue_align: u32,
55
56    pub legacy_queue_pfn: u32,
57
58    /// Virtual queue ready bit.
59    ///
60    /// Write 1 to notifies the device that it can execute requests from this virtual queue.
61    /// **Read-Write**
62    pub queue_ready: u32,
63
64    __r3: [u8; 8],
65
66    /// Queue notifier.
67    ///
68    /// Writing a value to this register notifies the device
69    /// that there are new buffers to process in a queue. **Write-only**
70    pub queue_notify: u32,
71
72    __r4: [u8; 12],
73
74    /// Interrupt status.
75    ///
76    /// bit0 => Used Buffer Notification;
77    /// bit1 => Configuration Change Notification
78    /// **Read-only**
79    pub interrupt_status: u32,
80    /// Interrupt acknowledge. **Write-only**
81    pub interrupt_ack: u32,
82
83    __r5: [u8; 8],
84
85    /// Device status. **Read-Write**
86    pub status: u32,
87
88    __r6: [u8; 12],
89
90    /// Virtual queue’s Descriptor Area 64 bit long physical address. **Write-only**
91    pub queue_desc_low: u32,
92    pub queue_desc_high: u32,
93
94    __r7: [u8; 8],
95
96    /// Virtual queue’s Driver Area 64 bit long physical address. **Write-only**
97    pub queue_driver_low: u32,
98    pub queue_driver_high: u32,
99
100    __r8: [u8; 8],
101
102    /// Virtual queue’s Device Area 64 bit long physical address. **Write-only**
103    pub queue_device_low: u32,
104    pub queue_device_high: u32,
105
106    __r9: [u8; 84],
107
108    /// Configuration atomicity value. **Read-only**
109    pub config_generation: u32,
110}
111
112impl Debug for VirtioMmioLayout {
113    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
114        f.debug_struct("VirtioMmioLayout")
115            .field("magic_value", &self.magic_value)
116            .field("version", &self.version)
117            .field("device_id", &self.device_id)
118            .field("vendor_id", &self.vendor_id)
119            .field("device_features", &self.device_features)
120            .field("device_features_sel", &self.device_features_select)
121            .field("driver_features", &self.driver_features)
122            .field("driver_features_sel", &self.driver_features_select)
123            .field("legacy_guest_page_size", &self.legacy_guest_page_size)
124            .field("queue_sel", &self.queue_sel)
125            .field("queue_num_max", &self.queue_num_max)
126            .field("queue_num", &self.queue_num)
127            .field("legacy_queue_align", &self.legacy_queue_align)
128            .field("legacy_queue_pfn", &self.legacy_queue_pfn)
129            .field("queue_ready", &self.queue_ready)
130            .field("queue_notify", &self.queue_notify)
131            .field("interrupt_status", &self.interrupt_status)
132            .field("interrupt_ack", &self.interrupt_ack)
133            .field("status", &self.status)
134            .field("queue_desc_low", &self.queue_desc_low)
135            .field("queue_desc_high", &self.queue_desc_high)
136            .field("queue_driver_low", &self.queue_driver_low)
137            .field("queue_driver_high", &self.queue_driver_high)
138            .field("queue_device_low", &self.queue_device_low)
139            .field("queue_device_high", &self.queue_device_high)
140            .field("config_generation", &self.config_generation)
141            .finish()
142    }
143}
