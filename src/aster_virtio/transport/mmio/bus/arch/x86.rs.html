3use log::debug;
4use ostd::arch::irq::IRQ_CHIP;
5pub(super) use ostd::arch::irq::MappedIrqLine;
6
7use crate::transport::mmio::bus::MmioRegisterError;
8
9pub(super) fn probe_for_device() {
10    // TODO: The correct method for detecting VirtIO-MMIO devices on x86_64 systems is to parse the
11    // kernel command line if ACPI tables are absent [1], or the ACPI SSDT if ACPI tables are
12    // present [2]. Neither of them is supported for now. This function's approach of blindly
13    // scanning the MMIO region is only a workaround.
14    // [1]: https://github.com/torvalds/linux/blob/0ff41df1cb268fc69e703a08a57ee14ae967d0ca/drivers/virtio/virtio_mmio.c#L733
15    // [2]: https://github.com/torvalds/linux/blob/0ff41df1cb268fc69e703a08a57ee14ae967d0ca/drivers/virtio/virtio_mmio.c#L840
16
17    // Constants from QEMU MicroVM. We should remove them as they're QEMU's implementation details.
18    //
19    // https://github.com/qemu/qemu/blob/3c5a5e213e5f08fbfe70728237f7799ac70f5b99/hw/i386/microvm.c#L201
20    const QEMU_MMIO_BASE: usize = 0xFEB0_0000;
21    const QEMU_MMIO_SIZE: usize = 512;
22    // https://github.com/qemu/qemu/blob/3c5a5e213e5f08fbfe70728237f7799ac70f5b99/hw/i386/microvm.c#L196
23    const QEMU_IOAPIC1_GSI_BASE: u32 = 16;
24    const QEMU_IOAPIC1_NUM_TRANS: u32 = 8;
25    // https://github.com/qemu/qemu/blob/3c5a5e213e5f08fbfe70728237f7799ac70f5b99/hw/i386/microvm.c#L192
26    const QEMU_IOAPIC2_GSI_BASE: u32 = 24;
27    const QEMU_IOAPIC2_NUM_TRANS: u32 = 24;
28
29    let irq_chip = IRQ_CHIP.get().unwrap();
30    let (gsi_base, num_trans) = match irq_chip.count_io_apics() {
31        1 => (QEMU_IOAPIC1_GSI_BASE, QEMU_IOAPIC1_NUM_TRANS),
32        2.. => (QEMU_IOAPIC2_GSI_BASE, QEMU_IOAPIC2_NUM_TRANS),
33        0 => {
34            debug!("[Virtio]: Skip MMIO detection because there are no I/O APICs");
35            return;
36        }
37    };
38
39    for index in 0..num_trans {
40        let mmio_base = QEMU_MMIO_BASE + (index as usize) * QEMU_MMIO_SIZE;
41        match super::try_register_mmio_device(mmio_base..(mmio_base + QEMU_MMIO_SIZE), |irq_line| {
42            irq_chip.map_gsi_pin_to(irq_line, gsi_base + index)
43        }) {
44            Err(e) if e.is_fatal() => break,
45            _ => continue,
46        }
47    }
48}
49
50impl MmioRegisterError {
51    /// Returns `true` if it should terminate a linear MMIO scan.
52    fn is_fatal(self) -> bool {
53        matches!(self, Self::MmioUnavailable | Self::MagicMismatch)
54    }
55}
