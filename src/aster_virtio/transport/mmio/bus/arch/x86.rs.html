12pub(super) fn probe_for_device() {
13    probe_from_kernel_cmdline();
14    probe_from_microvm_constants();
15}

17static VIRTIO_MMIO_CMDLINE_DEVICES: Once<Vec<MmioDevice>> = Once::new();
18aster_cmdline::define_repeatable_kv_param!("virtio_mmio.device", VIRTIO_MMIO_CMDLINE_DEVICES);
19
20/// Probes Linux-compatible `virtio_mmio.device=<size>@<base>:<irq>[:<id>]` parameters.
21///
22/// This format follows Linux's `virtio_mmio.device` kernel parameter.
23fn probe_from_kernel_cmdline() {
24    let Some(devices) = VIRTIO_MMIO_CMDLINE_DEVICES.get() else {
25        return;
26    };
27
28    let irq_chip = IRQ_CHIP.get().unwrap();
29
30    for device in devices {
31        info!(
32            "Probe MMIO command-line device: base={:#x}, size={:#x}, irq={}",
33            device.base(),
34            device.size().get(),
35            device.irq().get()
36        );
37
38        let Some(mmio_end) = device.base().checked_add(device.size().get()) else {
39            warn!(
40                "Ignore MMIO command-line device at {:#x} because its range overflows",
41                device.base()
42            );
43            continue;
44        };
45
46        if let Err(err) = super::try_register_mmio_device(device.base()..mmio_end, |irq_line| {
47            irq_chip.map_gsi_pin_to(irq_line, device.irq().get())
48        }) {
49            warn!(
50                "Ignore MMIO command-line device at {:#x} due to an error ({:?})",
51                device.base(),
52                err,
53            );
54        }
55    }
56}

58fn probe_from_microvm_constants() {
59    // TODO: If ACPI tables are present, the correct method for detecting VirtIO-MMIO
60    // devices is to parse the ACPI SSDT [1]. It is not supported yet, so we fall
61    // back to blindly scanning QEMU MicroVM's fixed MMIO window as a workaround.
62    // [1]: https://github.com/torvalds/linux/blob/0ff41df1cb268fc69e703a08a57ee14ae967d0ca/drivers/virtio/virtio_mmio.c#L840
63
64    // Constants from QEMU MicroVM. We should remove them as they're QEMU's implementation details.
65    //
66    // https://github.com/qemu/qemu/blob/3c5a5e213e5f08fbfe70728237f7799ac70f5b99/hw/i386/microvm.c#L201
67    const QEMU_MMIO_BASE: usize = 0xFEB0_0000;
68    const QEMU_MMIO_SIZE: usize = 512;
69    // https://github.com/qemu/qemu/blob/3c5a5e213e5f08fbfe70728237f7799ac70f5b99/hw/i386/microvm.c#L196
70    const QEMU_IOAPIC1_GSI_BASE: u32 = 16;
71    const QEMU_IOAPIC1_NUM_TRANS: u32 = 8;
72    // https://github.com/qemu/qemu/blob/3c5a5e213e5f08fbfe70728237f7799ac70f5b99/hw/i386/microvm.c#L192
73    const QEMU_IOAPIC2_GSI_BASE: u32 = 24;
74    const QEMU_IOAPIC2_NUM_TRANS: u32 = 24;
75
76    let irq_chip = IRQ_CHIP.get().unwrap();
77    let (gsi_base, num_trans) = match irq_chip.count_io_apics() {
78        1 => (QEMU_IOAPIC1_GSI_BASE, QEMU_IOAPIC1_NUM_TRANS),
79        2.. => (QEMU_IOAPIC2_GSI_BASE, QEMU_IOAPIC2_NUM_TRANS),
80        0 => {
81            debug!("Skip MMIO detection because there are no I/O APICs");
82            return;
83        }
84    };
85
86    for index in 0..num_trans {
87        let mmio_base = QEMU_MMIO_BASE + (index as usize) * QEMU_MMIO_SIZE;
88        match super::try_register_mmio_device(mmio_base..(mmio_base + QEMU_MMIO_SIZE), |irq_line| {
89            irq_chip.map_gsi_pin_to(irq_line, gsi_base + index)
90        }) {
91            Err(e) if e.is_fatal() => break,
92            _ => continue,
93        }
94    }
95}

97impl MmioRegisterError {
98    /// Returns `true` if it should terminate a linear MMIO scan.
99    fn is_fatal(self) -> bool {
100        matches!(self, Self::MmioUnavailable | Self::MagicMismatch)
101    }
102}
