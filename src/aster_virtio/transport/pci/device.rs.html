69impl VirtioTransport for VirtioPciModernTransport {
70    fn device_type(&self) -> VirtioDeviceType {
71        self.device_type
72    }
73
74    fn set_queue(
75        &mut self,
76        idx: u16,
77        queue_size: u16,
78        descriptor_ptr: &SafePtr<Descriptor, Arc<DmaCoherent>>,
79        avail_ring_ptr: &SafePtr<AvailRing, Arc<DmaCoherent>>,
80        used_ring_ptr: &SafePtr<UsedRing, Arc<DmaCoherent>>,
81    ) -> Result<(), VirtioTransportError> {
82        if idx >= self.num_queues() {
83            return Err(VirtioTransportError::InvalidArgs);
84        }
85        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_select)
86            .write_once(&idx)
87            .unwrap();
88        debug_assert_eq!(
89            field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_select)
90                .read_once()
91                .unwrap(),
92            idx
93        );
94
95        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_size)
96            .write_once(&queue_size)
97            .unwrap();
98        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_desc)
99            .write_once(&(descriptor_ptr.daddr() as u64))
100            .unwrap();
101        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_driver)
102            .write_once(&(avail_ring_ptr.daddr() as u64))
103            .unwrap();
104        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_device)
105            .write_once(&(used_ring_ptr.daddr() as u64))
106            .unwrap();
107        // Enable queue
108        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_enable)
109            .write_once(&1u16)
110            .unwrap();
111        Ok(())
112    }
113
114    fn notify_config(&self, idx: usize) -> ConfigManager<u32> {
115        debug_assert!(idx < self.num_queues() as usize);
116        let safe_ptr = Some(SafePtr::new(
117            self.notify.io_memory.clone(),
118            (self.notify.offset + self.notify.offset_multiplier * idx as u32) as usize,
119        ));
120
121        ConfigManager::new(safe_ptr, None)
122    }
123
124    fn num_queues(&self) -> u16 {
125        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, num_queues)
126            .read_once()
127            .unwrap()
128    }
129
130    fn device_config_mem(&self) -> Option<IoMem> {
131        let offset = self.device_cfg.offset() as usize;
132        let length = self.device_cfg.length() as usize;
133        let io_mem = self
134            .device_cfg
135            .memory_bar()
136            .as_ref()
137            .unwrap()
138            .io_mem()
139            .slice(offset..offset + length);
140
141        Some(io_mem)
142    }
143
144    fn device_config_bar(&self) -> Option<(Bar, usize)> {
145        None
146    }
147
148    fn read_device_features(&self) -> u64 {
149        // select low
150        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, device_feature_select)
151            .write_once(&0u32)
152            .unwrap();
153        let device_feature_low = field_ptr!(&self.common_cfg, VirtioPciCommonCfg, device_features)
154            .read_once()
155            .unwrap();
156        // select high
157        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, device_feature_select)
158            .write_once(&1u32)
159            .unwrap();
160        let device_feature_high = field_ptr!(&self.common_cfg, VirtioPciCommonCfg, device_features)
161            .read_once()
162            .unwrap() as u64;
163        (device_feature_high << 32) | device_feature_low as u64
164    }
165
166    fn write_driver_features(&mut self, features: u64) -> Result<(), VirtioTransportError> {
167        let low = features as u32;
168        let high = (features >> 32) as u32;
169        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, driver_feature_select)
170            .write_once(&0u32)
171            .unwrap();
172        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, driver_features)
173            .write_once(&low)
174            .unwrap();
175        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, driver_feature_select)
176            .write_once(&1u32)
177            .unwrap();
178        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, driver_features)
179            .write_once(&high)
180            .unwrap();
181        Ok(())
182    }
183
184    fn read_device_status(&self) -> DeviceStatus {
185        let status = field_ptr!(&self.common_cfg, VirtioPciCommonCfg, device_status)
186            .read_once()
187            .unwrap();
188        DeviceStatus::from_bits(status).unwrap()
189    }
190
191    fn write_device_status(&mut self, status: DeviceStatus) -> Result<(), VirtioTransportError> {
192        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, device_status)
193            .write_once(&(status.bits()))
194            .unwrap();
195        Ok(())
196    }
197
198    fn max_queue_size(&self, idx: u16) -> Result<u16, crate::transport::VirtioTransportError> {
199        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_select)
200            .write_once(&idx)
201            .unwrap();
202        debug_assert_eq!(
203            field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_select)
204                .read_once()
205                .unwrap(),
206            idx
207        );
208
209        Ok(field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_size)
210            .read_once()
211            .unwrap())
212    }
213
214    fn register_queue_callback(
215        &mut self,
216        index: u16,
217        func: Box<IrqCallbackFunction>,
218        single_interrupt: bool,
219    ) -> Result<(), VirtioTransportError> {
220        if index >= self.num_queues() {
221            return Err(VirtioTransportError::InvalidArgs);
222        }
223        let (vector, irq) = if single_interrupt {
224            if let Some(unused_irq) = self.msix_manager.pop_unused_irq() {
225                unused_irq
226            } else {
227                warn!(
228                    "{:?}: `single_interrupt` ignored: no more IRQ lines available",
229                    self.device_type()
230                );
231                self.msix_manager.shared_irq_line()
232            }
233        } else {
234            self.msix_manager.shared_irq_line()
235        };
236        irq.on_active(func);
237        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_select)
238            .write_once(&index)
239            .unwrap();
240        debug_assert_eq!(
241            field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_select)
242                .read_once()
243                .unwrap(),
244            index
245        );
246        field_ptr!(&self.common_cfg, VirtioPciCommonCfg, queue_msix_vector)
247            .write_once(&vector)
248            .unwrap();
249        Ok(())
250    }
251
252    fn register_cfg_callback(
253        &mut self,
254        func: Box<IrqCallbackFunction>,
255    ) -> Result<(), VirtioTransportError> {
256        let (_, irq) = self.msix_manager.config_msix_irq();
257        irq.on_active(func);
258        Ok(())
259    }
260
261    fn is_legacy_version(&self) -> bool {
262        // TODO: Support legacy version
263        false
264    }
265}

267impl VirtioPciModernTransport {
268    #[expect(clippy::result_large_err)]
269    pub(super) fn new(
270        common_device: PciCommonDevice,
271    ) -> Result<Self, (BusProbeError, PciCommonDevice)> {
272        let device_id = common_device.device_id().device_id;
273        let device_type_value = if device_id <= 0x1040 {
274            device_id - 0x1000
275        } else {
276            device_id - 0x1040
277        };
278
279        let device_type = match VirtioDeviceType::try_from(device_type_value as u8) {
280            Ok(device) => device,
281            Err(_) => {
282                warn!("Unrecognized virtio-pci device id:{:x?}", device_id);
283                return Err((BusProbeError::DeviceNotMatch, common_device));
284            }
285        };
286
287        info!("[Virtio]: Found device:{:?}", device_type);
288
289        let mut msix = None;
290        let mut notify = None;
291        let mut common_cfg = None;
292        let mut device_cfg = None;
293        for cap in common_device.capabilities().iter() {
294            match cap.capability_data() {
295                CapabilityData::Vndr(vendor) => {
296                    let data = VirtioPciCapabilityData::new(common_device.bar_manager(), *vendor);
297                    match data.typ() {
298                        VirtioPciCpabilityType::CommonCfg => {
299                            common_cfg = Some(VirtioPciCommonCfg::new(&data));
300                        }
301                        VirtioPciCpabilityType::NotifyCfg => {
302                            notify = Some(VirtioPciNotify {
303                                offset_multiplier: data.option_value().unwrap(),
304                                offset: data.offset(),
305                                io_memory: data.memory_bar().as_ref().unwrap().io_mem().clone(),
306                            });
307                        }
308                        VirtioPciCpabilityType::IsrCfg => {}
309                        VirtioPciCpabilityType::DeviceCfg => {
310                            device_cfg = Some(data);
311                        }
312                        VirtioPciCpabilityType::PciCfg => {}
313                    }
314                }
315                CapabilityData::Msix(data) => {
316                    msix = Some(data.clone());
317                }
318                CapabilityData::Unknown(id) => {
319                    panic!("unknown capability: {}", id)
320                }
321                _ => {
322                    panic!("PCI Virtio device should not have other type of capability")
323                }
324            }
325        }
326        // TODO: Support interrupt without MSI-X
327        let msix = msix.unwrap();
328        let notify = notify.unwrap();
329        let common_cfg = common_cfg.unwrap();
330        let device_cfg = device_cfg.unwrap();
331        let msix_manager = VirtioMsixManager::new(msix);
332        Ok(Self {
333            common_device,
334            common_cfg,
335            device_cfg,
336            notify,
337            msix_manager,
338            device_type,
339        })
340    }
