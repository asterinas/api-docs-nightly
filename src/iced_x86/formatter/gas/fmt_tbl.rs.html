19lazy_static! {
20	pub(super) static ref ALL_INFOS: Box<[Box<dyn InstrInfo + Send + Sync>; IcedConstants::CODE_ENUM_COUNT]> = read();
21}
22
23fn add_suffix(s: &str, c: char) -> String {
24	if c == '\0' {
25		String::from(s)
26	} else {
27		let mut res = String::with_capacity(s.len() + 1);
28		res.push_str(s);
29		res.push(c);
30		res
31	}
32}

34fn read() -> Box<[Box<dyn InstrInfo + Send + Sync>; IcedConstants::CODE_ENUM_COUNT]> {
35	let mut infos: Vec<Box<dyn InstrInfo + Send + Sync>> = Vec::with_capacity(IcedConstants::CODE_ENUM_COUNT);
36	let mut reader = DataReader::new(FORMATTER_TBL_DATA);
37	let strings = get_strings_table_ref();
38	let mut prev_index = -1isize;
39	for i in 0..IcedConstants::CODE_ENUM_COUNT {
40		// SAFETY: generated (and immutable) data is valid
41
42		let f = reader.read_u8();
43		let mut ctor_kind: CtorKind = unsafe { mem::transmute((f & 0x7F) as u8) };
44		let current_index;
45		if ctor_kind == CtorKind::Previous {
46			current_index = reader.index() as isize;
47			reader.set_index(prev_index as usize);
48			ctor_kind = unsafe { mem::transmute((reader.read_u8() & 0x7F) as u8) };
49		} else {
50			current_index = -1;
51			prev_index = reader.index() as isize - 1;
52		}
53		let mut s = if (f & 0x80) != 0 {
54			let s = strings[reader.read_compressed_u32() as usize];
55			let mut res = String::with_capacity(s.len() + 1);
56			res.push('v');
57			res.push_str(s);
58			res
59		} else {
60			String::from(strings[reader.read_compressed_u32() as usize])
61		};
62
63		let c;
64		let v;
65		let v2;
66		let v3;
67		let info: Box<dyn InstrInfo + Send + Sync> = match ctor_kind {
68			CtorKind::Previous => unreachable!(),
69			CtorKind::Normal_1 => Box::new(SimpleInstrInfo::with_mnemonic(s)),
70
71			CtorKind::Normal_2a => {
72				c = reader.read_u8() as u8 as char;
73				let s2 = add_suffix(&s, c);
74				Box::new(SimpleInstrInfo::with_mnemonic_suffix(s, s2))
75			}
76
77			CtorKind::Normal_2b => {
78				v = reader.read_compressed_u32();
79				Box::new(SimpleInstrInfo::with_mnemonic_flags(s, v))
80			}
81
82			CtorKind::Normal_2c => {
83				c = reader.read_u8() as u8 as char;
84				s = add_suffix(&s, c);
85				Box::new(SimpleInstrInfo::with_mnemonic(s))
86			}
87
88			CtorKind::Normal_3 => {
89				c = reader.read_u8() as u8 as char;
90				let s2 = add_suffix(&s, c);
91				v = reader.read_compressed_u32();
92				Box::new(SimpleInstrInfo::new(s, s2, v))
93			}
94
95			CtorKind::AamAad => Box::new(SimpleInstrInfo_AamAad::new(s)),
96
97			CtorKind::asz => {
98				v = reader.read_compressed_u32();
99				Box::new(SimpleInstrInfo_as::new(v, s))
100			}
101
102			CtorKind::bnd => {
103				c = reader.read_u8() as u8 as char;
104				let s2 = add_suffix(&s, c);
105				v = reader.read_compressed_u32();
106				Box::new(SimpleInstrInfo_bnd::new(s, s2, v))
107			}
108
109			CtorKind::DeclareData => Box::new(SimpleInstrInfo_DeclareData::new(unsafe { mem::transmute(i as CodeUnderlyingType) }, s)),
110
111			CtorKind::er_2 => {
112				v = reader.read_compressed_u32();
113				Box::new(SimpleInstrInfo_er::with_mnemonic(v, s))
114			}
115
116			CtorKind::er_4 => {
117				c = reader.read_u8() as u8 as char;
118				let s2 = add_suffix(&s, c);
119				v = reader.read_compressed_u32();
120				v2 = reader.read_compressed_u32();
121				Box::new(SimpleInstrInfo_er::new(v, s, s2, v2))
122			}
123
124			CtorKind::far => {
125				c = reader.read_u8() as u8 as char;
126				let s2 = add_suffix(&s, c);
127				v = reader.read_compressed_u32();
128				Box::new(SimpleInstrInfo_far::new(v, s, s2))
129			}
130
131			CtorKind::imul => {
132				c = reader.read_u8() as u8 as char;
133				let s2 = add_suffix(&s, c);
134				Box::new(SimpleInstrInfo_imul::new(s, s2))
135			}
136
137			CtorKind::maskmovq => Box::new(SimpleInstrInfo_maskmovq::new(s)),
138
139			CtorKind::movabs => {
140				c = reader.read_u8() as u8 as char;
141				let s2 = add_suffix(&s, c);
142				let s3 = String::from(strings[reader.read_compressed_u32() as usize]);
143				let s4 = add_suffix(&s3, c);
144				Box::new(SimpleInstrInfo_movabs::new(s, s2, s3, s4))
145			}
146
147			CtorKind::nop => {
148				v = reader.read_compressed_u32();
149				v2 = reader.read_u8() as u32;
150				Box::new(SimpleInstrInfo_nop::new(v, s, unsafe { mem::transmute(v2 as RegisterUnderlyingType) }))
151			}
152
153			CtorKind::OpSize => {
154				v = reader.read_u8() as u32;
155				let s2 = add_suffix(&s, 'w');
156				let s3 = add_suffix(&s, 'l');
157				let s4 = add_suffix(&s, 'q');
158				Box::new(SimpleInstrInfo_OpSize::new(unsafe { mem::transmute(v as CodeSizeUnderlyingType) }, s, s2, s3, s4))
159			}
160
161			CtorKind::OpSize2_bnd => {
162				let s2 = String::from(strings[reader.read_compressed_u32() as usize]);
163				let s3 = String::from(strings[reader.read_compressed_u32() as usize]);
164				let s4 = String::from(strings[reader.read_compressed_u32() as usize]);
165				Box::new(SimpleInstrInfo_OpSize2_bnd::new(s, s2, s3, s4))
166			}
167
168			CtorKind::OpSize3 => {
169				c = reader.read_u8() as u8 as char;
170				let s2 = add_suffix(&s, c);
171				v = reader.read_compressed_u32();
172				Box::new(SimpleInstrInfo_OpSize3::new(v, s, s2))
173			}
174
175			CtorKind::os => {
176				v = reader.read_compressed_u32();
177				v2 = reader.read_u8() as u32;
178				debug_assert!(v2 <= 1);
179				v3 = reader.read_compressed_u32();
180				Box::new(SimpleInstrInfo_os::new(v, s, v2 != 0, v3))
181			}
182
183			CtorKind::CC_1 => {
184				c = reader.read_u8() as u8 as char;
185				let s2 = add_suffix(&s, c);
186				v = reader.read_compressed_u32();
187				Box::new(SimpleInstrInfo_cc::new(v, vec![s], vec![s2]))
188			}
189
190			CtorKind::CC_2 => {
191				let s2 = String::from(strings[reader.read_compressed_u32() as usize]);
192				c = reader.read_u8() as u8 as char;
193				let s3 = add_suffix(&s, c);
194				let s4 = add_suffix(&s2, c);
195				v = reader.read_compressed_u32();
196				Box::new(SimpleInstrInfo_cc::new(v, vec![s, s2], vec![s3, s4]))
197			}
198
199			CtorKind::CC_3 => {
200				let s2 = String::from(strings[reader.read_compressed_u32() as usize]);
201				let s3 = String::from(strings[reader.read_compressed_u32() as usize]);
202				c = reader.read_u8() as u8 as char;
203				let s4 = add_suffix(&s, c);
204				let s5 = add_suffix(&s2, c);
205				let s6 = add_suffix(&s3, c);
206				v = reader.read_compressed_u32();
207				Box::new(SimpleInstrInfo_cc::new(v, vec![s, s2, s3], vec![s4, s5, s6]))
208			}
209
210			CtorKind::os_jcc_1 => {
211				v2 = reader.read_compressed_u32();
212				v = reader.read_compressed_u32();
213				Box::new(SimpleInstrInfo_os_jcc::new(v, v2, vec![s]))
214			}
215
216			CtorKind::os_jcc_2 => {
217				let s2 = String::from(strings[reader.read_compressed_u32() as usize]);
218				v2 = reader.read_compressed_u32();
219				v = reader.read_compressed_u32();
220				Box::new(SimpleInstrInfo_os_jcc::new(v, v2, vec![s, s2]))
221			}
222
223			CtorKind::os_jcc_3 => {
224				let s2 = String::from(strings[reader.read_compressed_u32() as usize]);
225				let s3 = String::from(strings[reader.read_compressed_u32() as usize]);
226				v2 = reader.read_compressed_u32();
227				v = reader.read_compressed_u32();
228				Box::new(SimpleInstrInfo_os_jcc::new(v, v2, vec![s, s2, s3]))
229			}
230
231			CtorKind::os_loopcc => {
232				let s2 = String::from(strings[reader.read_compressed_u32() as usize]);
233				c = reader.read_u8() as u8 as char;
234				let s3 = add_suffix(&s, c);
235				let s4 = add_suffix(&s2, c);
236				v3 = reader.read_compressed_u32();
237				v = reader.read_compressed_u32();
238				v2 = reader.read_compressed_u32();
239				Box::new(SimpleInstrInfo_os_loop::new(v, v2, v3, vec![s, s2], vec![s3, s4]))
240			}
241
242			CtorKind::os_loop => {
243				c = reader.read_u8() as u8 as char;
244				let s2 = add_suffix(&s, c);
245				v = reader.read_compressed_u32();
246				v2 = reader.read_compressed_u32();
247				Box::new(SimpleInstrInfo_os_loop::new(v, v2, u32::MAX, vec![s], vec![s2]))
248			}
249
250			CtorKind::os_mem => {
251				c = reader.read_u8() as u8 as char;
252				let s2 = add_suffix(&s, c);
253				v = reader.read_compressed_u32();
254				Box::new(SimpleInstrInfo_os_mem::new(v, s, s2))
255			}
256
257			CtorKind::Reg16 => {
258				let s2 = add_suffix(&s, 'w');
259				Box::new(SimpleInstrInfo_Reg16::new(s, s2))
260			}
261
262			CtorKind::os_mem2 => {
263				c = reader.read_u8() as u8 as char;
264				let s2 = add_suffix(&s, c);
265				v = reader.read_compressed_u32();
266				Box::new(SimpleInstrInfo_os_mem2::new(v, s, s2))
267			}
268
269			CtorKind::os2_3 => {
270				c = reader.read_u8() as u8 as char;
271				let s2 = add_suffix(&s, c);
272				v = reader.read_compressed_u32();
273				v2 = reader.read_u8() as u32;
274				debug_assert!(v2 <= 1);
275				Box::new(SimpleInstrInfo_os2::new(v, s, s2, v2 != 0, 0))
276			}
277
278			CtorKind::os2_4 => {
279				c = reader.read_u8() as u8 as char;
280				let s2 = add_suffix(&s, c);
281				v = reader.read_compressed_u32();
282				v2 = reader.read_u8() as u32;
283				debug_assert!(v2 <= 1);
284				v3 = reader.read_compressed_u32();
285				Box::new(SimpleInstrInfo_os2::new(v, s, s2, v2 != 0, v3))
286			}
287
288			CtorKind::pblendvb => Box::new(SimpleInstrInfo_pblendvb::new(s)),
289
290			CtorKind::pclmulqdq => {
291				v = reader.read_u8() as u32;
292				Box::new(SimpleInstrInfo_pclmulqdq::new(s, get_pseudo_ops(unsafe { mem::transmute(v as u8) })))
293			}
294
295			CtorKind::pops => {
296				v = reader.read_u8() as u32;
297				v2 = reader.read_u8() as u32;
298				debug_assert!(v2 <= 1);
299				Box::new(SimpleInstrInfo_pops::new(s, get_pseudo_ops(unsafe { mem::transmute(v as u8) }), v2 != 0))
300			}
301
302			CtorKind::mem16 => {
303				c = reader.read_u8() as u8 as char;
304				let s2 = add_suffix(&s, c);
305				let s3 = add_suffix(&s, 'w');
306				Box::new(SimpleInstrInfo_mem16::new(s, s2, s3))
307			}
308
309			CtorKind::Reg32 => Box::new(SimpleInstrInfo_Reg32::new(s)),
310
311			CtorKind::sae => {
312				v = reader.read_compressed_u32();
313				Box::new(SimpleInstrInfo_sae::new(v, s))
314			}
315
316			CtorKind::ST_STi => Box::new(SimpleInstrInfo_ST_STi::new(s)),
317			CtorKind::STi_ST => {
318				v = reader.read_u8() as u32;
319				debug_assert!(v <= 1);
320				Box::new(SimpleInstrInfo_STi_ST::new(s, v != 0))
321			}
322
323			CtorKind::STIG1 => {
324				v = reader.read_u8() as u32;
325				debug_assert!(v <= 1);
326				Box::new(SimpleInstrInfo_STIG1::new(s, v != 0))
327			}
328		};
329
330		infos.push(info);
331		if current_index >= 0 {
332			reader.set_index(current_index as usize);
333		}
334	}
335	debug_assert!(!reader.can_read());
336
337	#[allow(clippy::unwrap_used)]
338	infos.into_boxed_slice().try_into().ok().unwrap()
339}
