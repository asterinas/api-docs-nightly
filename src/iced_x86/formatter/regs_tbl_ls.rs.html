4use crate::formatter::regs_tbl::{MAX_STRING_LENGTH, PADDING_SIZE, REGS_DATA};
5use crate::formatter::FormatterString;
6use crate::iced_constants::IcedConstants;
7use alloc::boxed::Box;
8use alloc::string::String;
9use alloc::vec::Vec;
10use core::convert::TryInto;
11use lazy_static::lazy_static;
12
13lazy_static! {
14	pub(super) static ref REGS_TBL: Box<[FormatterString; IcedConstants::REGISTER_ENUM_COUNT]> = {
15		let mut v = Vec::with_capacity(IcedConstants::REGISTER_ENUM_COUNT);
16		let mut s = String::with_capacity(MAX_STRING_LENGTH);
17		let mut data = &REGS_DATA[..];
18		for _ in 0..IcedConstants::REGISTER_ENUM_COUNT {
19			let len = data[0] as usize;
20			data = &data[1..];
21			for &c in &data[0..len] {
22				s.push(c as char);
23			}
24			data = &data[len..];
25			v.push(FormatterString::new(s.clone()));
26			s.clear();
27		}
28		debug_assert!(data.len() == PADDING_SIZE);
29		#[allow(clippy::unwrap_used)]
30		v.into_boxed_slice().try_into().ok().unwrap()
31	};
32}
