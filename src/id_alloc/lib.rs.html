6use core::{fmt::Debug, ops::Range};
7
8use bitvec::prelude::BitVec;
9
10/// An id allocator implemented by the bitmap.
11/// The true bit implies that the id is allocated, and vice versa.
12#[derive(Clone)]
13pub struct IdAlloc {
14    bitset: BitVec<u8>,
15    first_available_id: usize,
16}
17
18impl IdAlloc {
19    /// Constructs a new id allocator with a maximum capacity.
20    pub fn with_capacity(capacity: usize) -> Self {
21        let mut bitset = BitVec::with_capacity(capacity);
22        bitset.resize(capacity, false);
23        Self {
24            bitset,
25            first_available_id: 0,
26        }
27    }
28
29    /// Allocates and returns a new `id`.
30    ///
31    /// If allocation is not possible, it returns `None`.
32    pub fn alloc(&mut self) -> Option<usize> {
33        if self.first_available_id < self.bitset.len() {
34            let id = self.first_available_id;
35            self.bitset.set(id, true);
36            self.first_available_id = (id + 1..self.bitset.len())
37                .find(|&i| !self.bitset[i])
38                .map_or(self.bitset.len(), |i| i);
39            Some(id)
40        } else {
41            None
42        }
43    }
44
45    /// Allocates a consecutive range of new `id`s.
46    ///
47    /// The `count` is the number of consecutive `id`s to allocate. If it is 0, return `None`.
48    ///
49    /// If allocation is not possible, it returns `None`.
50    ///
51    /// TODO: Choose a more efficient strategy.
52    pub fn alloc_consecutive(&mut self, count: usize) -> Option<Range<usize>> {
53        if count == 0 {
54            return None;
55        }
56
57        let end = self.first_available_id.checked_add(count)?;
58        if end > self.bitset.len() {
59            return None;
60        }
61
62        // Scan the bitmap from the position `first_available_id`
63        // for the first `count` number of consecutive 0's.
64        let allocated_range = {
65            // Invariance: all bits within `curr_range` are 0's
66            let mut curr_range = self.first_available_id..self.first_available_id + 1;
67            while curr_range.len() < count && curr_range.end < self.bitset.len() {
68                if !self.is_allocated(curr_range.end) {
69                    curr_range.end += 1;
70                } else {
71                    curr_range = curr_range.end + 1..curr_range.end + 1;
72                }
73            }
74
75            if curr_range.len() < count {
76                return None;
77            }
78
79            curr_range
80        };
81
82        // Set every bit to 1 within the allocated range
83        for id in allocated_range.clone() {
84            self.bitset.set(id, true);
85        }
86
87        // In case we need to update first_available_id
88        if self.is_allocated(self.first_available_id) {
89            self.first_available_id = (allocated_range.end..self.bitset.len())
90                .find(|&i| !self.bitset[i])
91                .map_or(self.bitset.len(), |i| i);
92        }
93
94        Some(allocated_range)
95    }
96
97    /// Releases the consecutive range of allocated `id`s.
98    ///
99    /// # Panics
100    ///
101    /// If the `range` is out of bounds, this method will panic.
102    pub fn free_consecutive(&mut self, range: Range<usize>) {
103        if range.is_empty() {
104            return;
105        }
106
107        let range_start = range.start;
108        for id in range {
109            debug_assert!(self.is_allocated(id));
110            self.bitset.set(id, false);
111        }
112
113        if range_start < self.first_available_id {
114            self.first_available_id = range_start
115        }
116    }
117
118    /// Releases the allocated `id`.
119    ///
120    /// # Panics
121    ///
122    /// If the `id` is out of bounds, this method will panic.
123    pub fn free(&mut self, id: usize) {
124        debug_assert!(self.is_allocated(id));
125
126        self.bitset.set(id, false);
127        if id < self.first_available_id {
128            self.first_available_id = id;
129        }
130    }
131
132    /// Allocate a specific ID.
133    ///
134    /// If the ID is already allocated, it returns `None`, otherwise it
135    /// returns the allocated ID.
136    ///
137    /// # Panics
138    ///
139    /// If the `id` is out of bounds, this method will panic.
140    pub fn alloc_specific(&mut self, id: usize) -> Option<usize> {
141        if self.bitset[id] {
142            return None;
143        }
144        self.bitset.set(id, true);
145        if id == self.first_available_id {
146            self.first_available_id = (id + 1..self.bitset.len())
147                .find(|&i| !self.bitset[i])
148                .map_or(self.bitset.len(), |i| i);
149        }
150        Some(id)
151    }
152
153    /// Returns true if the `id` is allocated.
154    ///
155    /// # Panics
156    ///
157    /// If the `id` is out of bounds, this method will panic.
158    pub fn is_allocated(&self, id: usize) -> bool {
159        self.bitset[id]
160    }
161}
162
163impl Debug for IdAlloc {
164    fn fmt(&self, f: &mut core::fmt::Formatter) -> core::fmt::Result {
165        f.debug_struct("IdAlloc")
166            .field("len", &self.bitset.len())
167            .field("first_available_id", &self.first_available_id)
168            .finish()
169    }
170}

172#[cfg(test)]
173mod test {
174    use super::IdAlloc;
175
176    #[test]
177    fn bitmap_alloc_out_of_bounds() {
178        let capacity = 16;
179        let mut bitmap = IdAlloc::with_capacity(capacity);
180
181        for _ in 0..capacity {
182            assert!(bitmap.alloc().is_some());
183        }
184
185        // Allocating one more ID should fail since the
186        // bitmap's `first_available_id` + `count` is out of bounds.
187        assert!(bitmap.alloc_consecutive(1).is_none());
188    }
189}
