3//! NVMe Controller Register definitions.
4//!
5//! Refer to NVM Express Base Specification Revision 2.0, Section 3.1.
6
7/// 32-bit Controller Registers.
8#[repr(usize)]
9#[derive(Clone, Copy, Debug)]
10#[expect(dead_code)]
11pub(crate) enum NvmeRegs32 {
12    /// Version (VS): Indicates the NVMe specification version.
13    Vs = 0x8,
14    /// Interrupt Mask Set (INTMS): Used to set interrupt mask bits.
15    Intms = 0xC,
16    /// Interrupt Mask Clear (INTMC): Used to clear interrupt mask bits.
17    Intmc = 0x10,
18    /// Controller Configuration (CC): Used to configure the controller.
19    Cc = 0x14,
20    /// Controller Status (CSTS): Reports status of the controller.
21    Csts = 0x1C,
22    /// NVM Subsystem Reset (NSSR): Used to reset the NVM subsystem.
23    Nssr = 0x20,
24    /// Admin Queue Attributes (AQA): Defines the size of Admin Queues.
25    Aqa = 0x24,
26    /// Controller Memory Buffer Location (CMBLOC): Indicates the location of the Controller Memory Buffer.
27    Cmbloc = 0x38,
28    /// Controller Memory Buffer Size (CMBSZ): Indicates the size of the Controller Memory Buffer.
29    Cmbsz = 0x3C,
30    /// Boot Partition Information (BPINFO): Provides information about boot partitions.
31    Bpinfo = 0x40,
32    /// Boot Partition Read Select (BPRSEL): Selects which boot partition to read from.
33    Bprsel = 0x44,
34    /// Boot Partition Memory Buffer Location (BPMBL): Indicates the location of the boot partition memory buffer.
35    Bpmbl = 0x48,
36    /// Controller Memory Buffer Status (CMBSTS): Reports the status of the Controller Memory Buffer.
37    Cmbsts = 0x58,
38    /// Persistent Memory Region Capabilities (PMRCAP): Reports Persistent Memory Region capabilities.
39    Pmrcap = 0xE00,
40    /// Persistent Memory Region Control (PMRCTL): Controls the Persistent Memory Region.
41    Pmrctl = 0xE04,
42    /// Persistent Memory Region Status (PMRSTS): Reports the status of the Persistent Memory Region.
43    Pmrsts = 0xE08,
44    /// Persistent Memory Region Elasticity Buffer Size (PMREBS): Reports the size of the elasticity buffer.
45    Pmrebs = 0xE0C,
46    /// Persistent Memory Region Sustained Write Throughput (PMRSWTP): Reports sustained write throughput.
47    Pmrswtp = 0xE10,
48}

50/// 64-bit Controller Registers.
51#[repr(usize)]
52#[derive(Clone, Copy, Debug)]
53#[expect(dead_code)]
54pub(crate) enum NvmeRegs64 {
55    /// Controller Capabilities (CAP): Identifies basic capabilities.
56    Cap = 0x0,
57    /// Admin Submission Queue Base Address (ASQ): Base address of the Admin Submission Queue.
58    Asq = 0x28,
59    /// Admin Completion Queue Base Address (ACQ): Base address of the Admin Completion Queue.
60    Acq = 0x30,
61    /// Controller Memory Buffer Memory Space Control (CMBMSC): Controls the Controller Memory Buffer memory space.
62    Cmbmsc = 0x50,
63    /// Persistent Memory Region Memory Space Control (PMRMSC): Controls the Persistent Memory Region memory space.
64    Pmrmsc = 0xE14,
65}

89/// Doorbell Registers.
90///
91/// Doorbell registers are used to notify the controller of updates to submission
92/// and completion queues. Each queue pair has two doorbell registers:
93/// - Submission Queue y Tail Doorbell (SQyTDBL): offset 0x1000 + (2y * (4 << DSTRD))
94/// - Completion Queue y Head Doorbell (CQyHDBL): offset 0x1000 + ((2y+1) * (4 << DSTRD))
95///
96/// Where 'y' is the queue identifier (queue ID).
97#[repr(usize)]
98#[derive(Clone, Copy, Debug)]
99pub(crate) enum NvmeDoorbellRegs {
100    /// Submission Queue y Tail Doorbell (SQyTDBL).
101    Sqtdbl,
102    /// Completion Queue y Head Doorbell (CQyHDBL).
103    Cqhdbl,
104}
105
106impl NvmeDoorbellRegs {
107    /// Calculates the offset for this doorbell register.
108    ///
109    /// # Arguments
110    ///
111    /// * `qid` - Queue identifier (queue ID)
112    /// * `dstrd` - Doorbell Stride value from the CAP register
113    ///
114    /// # Returns
115    ///
116    /// The offset, in bytes, from the base of the controller registers.
117    pub(crate) fn offset(&self, qid: u16, dstrd: u16) -> usize {
118        const DOORBELL_BASE: usize = 0x1000;
119        let stride = 4usize << usize::from(dstrd);
120
121        match self {
122            NvmeDoorbellRegs::Sqtdbl => DOORBELL_BASE + (2 * usize::from(qid)) * stride,
123            NvmeDoorbellRegs::Cqhdbl => DOORBELL_BASE + ((2 * usize::from(qid)) + 1) * stride,
124        }
125    }
126}
