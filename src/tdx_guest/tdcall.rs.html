4//! The TDCALL instruction causes a VM exit to the Intel TDX module.
5//!
6//! It is used to call guest-side Intel TDX functions. For more information about
7//! TDCALL, please refer to the [Intel® TDX Module v1.5 ABI Specification](https://cdrdv2.intel.com/v1/dl/getContent/733579)

15#[derive(Debug, PartialEq)]
16pub enum TdCallError {
17    /// There is no valid #VE information.
18    TdxNoValidVeInfo,
19    /// Operand is invalid.
20    TdxOperandInvalid,
21    /// The operand is busy (e.g., it is locked in Exclusive mode).
22    TdxOperandBusy,
23    /// Page has already been accepted.
24    TdxPageAlreadyAccepted,
25    /// Requested page size does not match the current GPA mapping size.
26    TdxPageSizeMismatch,
27    /// The provided FIELD_ID is incorrect.
28    TdxMetadataFieldIdIncorrect,
29    /// Field code and write mask are for a read-only field.
30    TdxMetadataFieldNotWritable,
31    /// Field code is for an unreadable field.
32    TdxMetadataFieldNotReadable,
33    /// The provided field value is not valid.
34    TdxMetadataFieldValueNotValid,
35    /// The TD's OP_STATE is incorrect for the required operation.
36    TdxOpStateIncorrect,
37    /// Operand address is out of range (e.g., not in a TDMR).
38    TdxOperandAddrRangeError,
39    /// Physical page metadata (in PAMT) are incorrect for the requested operation.
40    TdxPageMetadataIncorrect,
41    /// Service TD hash of TDINFO_STRUCT does not match the currently bound hash.
42    TdxServtdInfoHashMismatch,
43    /// Service TD is not bound.
44    TdxServtdNotBound,
45    /// Service TD UUID does not match the currently bound UUID.
46    TdxServtdUuidMismatch,
47    /// Target TD UUID does not match the requested TD_UUID.
48    TdxTargetUuidMismatch,
49    /// Target TD UUID does not match the requested TD_UUID, but pre-migration target TD UUID does match it.
50    TdxTargetUuidUpdated,
51    /// TD is in a FATAL error state.
52    TdxTdFatal,
53    /// TD keys have not been configured on the hardware.
54    TdxTdKeysNotConfigured,
55    /// TDCS pages have not been allocated.
56    TdxTdcsNotAllocated,
57    Other,
58}

137/// Get guest TD execution environment information.
138pub fn get_tdinfo() -> Result<TdgVpInfo, TdCallError> {
139    let mut args = TdcallArgs {
140        rax: TdcallNum::VpInfo as u64,

142    };

144    Ok(TdgVpInfo {

148        max_vcpus: (args.r8 >> 32) as u32,
149        vcpu_index: args.r9 as u32,
150        sys_rd: args.r10 as u32,

218/// TDG.MR.REPORT creates a TDREPORT_STRUCT structure that contains the measurements/configuration
219/// information of the guest TD that called the function, measurements/configuration information
220/// of the Intel TDX module and a REPORTMACSTRUCT.

222/// # Safety
223///
224/// The caller must ensure that:
225/// - `report_gpa` is a valid Guest Physical Address pointing to a writable buffer

227/// - `data_gpa` is a valid Guest Physical Address pointing to a readable buffer
228///   containing additional data to be included in the report (typically 64 bytes)
229/// - Both addresses are properly aligned (typically 64-byte aligned)
230/// - The memory regions are accessible and not shared with untrusted code
231/// - The buffers remain valid and unmodified during the entire call
232/// - The caller has appropriate privileges to perform this operation

234    let mut args = TdcallArgs {
235        rax: TdcallNum::MrReport as u64,
236        rcx: report_gpa,
237        rdx: data_gpa,
238        ..Default::default()
239    };
240    td_call(&mut args)
241}

396/// As a service TD, read a metadata field (control structure field) of a target TD.
397///
398/// Inputs:
399/// - binding_handle: the binding handle of the target TD.
400/// - field_identifier: the identifier of the field to read.
401///   The `LAST_ELEMENT_IN_FIELD` and `LAST_FIELD_IN_SEQUENCE` components of the field identifier must be 0.
402///   `WRITE_MASK_VALID`, `INC_SIZE`, `CONTEXT_CODE` and `ELEMENT_SIZE_CODE` components of the field identifier are ignored.
403///   A value of -1 is a special case: it is not a valid field identifier; in this case the first readable field identifier is returned in `RDX`.
404/// - uuid: the TD_UUID of the target TD, using little-Endian.

406/// Outputs:
407/// - Next readable field identifier. A value of -1 indicates no next field identifier is available.
408///   If the input field identifier was -1, `RDX` returns the first readable field identifier.
409///   In case of another error, `RDX` returns -1.

411/// - Updated target TD’s TD_UUID, using little-Endian.

413    binding_handle: u64,
414    field_identifier: u64,
415    uuid: [u64; 4],
416) -> Result<(u64, u64, [u64; 4]), TdCallError> {
417    let mut args = TdcallArgs {
418        rax: TdcallNum::ServetdRd as u64,
419        rcx: binding_handle,
420        rdx: field_identifier,
421        r10: uuid[0],
422        r11: uuid[1],
423        r12: uuid[2],

425        ..Default::default()
426    };
427    td_call(&mut args).map(|_| (args.rdx, args.r8, [args.r10, args.r11, args.r12, args.r13]))
428}

476/// Enter L2 VCPU operation.
477///
478/// Inputs:
479/// - l2_vm_idx: L2 virtual machine index (must be 1 or higher).
480/// - invd_translations: Controls how enter_l2_vcpu flushes the TLB context and extended paging structure (EPxE) caches
481///   associated with the L2 VM before entering the L2 VCPU.
482/// - guest_state_gpa: The GPA of a 256-bytes aligned L2EnterGuestState structure.
483///
484/// Outputs:
485/// - Return registers status in L2EnterGuestState.
486pub fn enter_l2_vcpu(
487    l2_vm_idx: u64,
488    invd_translations: InvdTranslations,
489    guest_state_gpa: u64,
490) -> Result<(), TdCallError> {
491    if (l2_vm_idx > 3) | (invd_translations.clone() as u64 > 3) {
492        return Err(TdCallError::TdxOperandInvalid);
493    }
494
495    let rcx = (l2_vm_idx << 52) | (invd_translations as u64);
496    let mut args = TdcallArgs {
497        rax: TdcallNum::VpEnter as u64,
498        rcx: l2_vm_idx,
499        rdx: guest_state_gpa,
500        ..Default::default()
501    };
502    td_call(&mut args)
503}

509///   Bit 1: Invalidate EPT for L2 VM #1.

520    };
521    td_call(&mut args)
522}

524/// Invalidate Guest Linear Address (GLA) mappings in the translation lookaside buffers (TLBs) and paging-structure caches
525/// for a specified L2 VM and a specified list of 4KB-aligned linear addresses.
526///
527/// Inputs:
528/// - l2_vm_idx: the index of the L2 VM to invalidate.
529///   1: Invalidate EPT for L2 VM #1.
530///   2: Invalidate EPT for L2 VM #2.
531///   3: Invalidate EPT for L2 VM #3.
532/// - list: GlaListEntry or GlaListInfo Flags.
533///   0: gla contains a single GLA list entry.
534///   1: RDX contains the GPA and other information of a GLA list in memory.
535/// - gla: Depending on the list flag, it contains either of the following:
536///     - A single GlaListEntry, specifying up to 512 consecutive guest linear addresses, each aligned on 4KB.
537///     - GlaLstInfo, specifying the GPA of a guest linear address (GLA) list in private
538///       memory. Each entry in the GLA list specifies up to 512 consecutive guest linear
539///       addresses, each aligned on 4KB. GlaLstInfo also specifies the first and last GLA
540///       list entries to process.
541///
542/// Outputs:
543/// - RDX: Depending on the list flag, it contains either of the following:
544///     - If list was 0, RDX contains the single GlaListEntry provided as an input, unmodified.
545///     - If list was 1, RDX contains the GlaLstInfo provided as input,
546///       but with the FIRST_ENTRY and NUM_ENTRIES fields updated to reflect the number of entries processed so far.
547///       If all entries have been processed successfully, NUM_ENTRIES is set to 0.
548pub fn invalidate_l2_gla(l2_vm_idx: u64, list: bool, gla: u64) -> Result<u64, TdCallError> {
549    if l2_vm_idx > 3 {
550        return Err(TdCallError::TdxOperandInvalid);
551    }
552
553    let rcx = (l2_vm_idx << 52) | if list { 0b1 } else { 0 };
554    let rdx = if list {
555        Gla::ListEntry(GlaListEntry(gla))
556    } else {
557        Gla::ListInfo(GlaListInfo(gla))
558    }
559    .value();
560
561    let mut args = TdcallArgs {
562        rax: TdcallNum::VpInvgla as u64,
563        rcx,
564        rdx,
565        ..Default::default()
566    };
567    td_call(&mut args).map(|_| args.rdx)
568}

570fn td_call(args: &mut TdcallArgs) -> Result<(), TdCallError> {
571    let result = unsafe { asm_td_call(args) };
572    match result {
573        0 => Ok(()),
574        _ => Err((result >> 32).into()),
575    }
576}

781enum Gla {
782    ListEntry(GlaListEntry),
783    ListInfo(GlaListInfo),

786/// The `GlaListEntry` species a range of consecutive guest linear addresses, each aligned on 4KB.
787///
788/// The `GlaListEntry` consists of the following fields:
789/// - **Bit 0-11**  LAST_GLA_INDEX: Index of the last 4KB-aligned linear address to be processed.
790/// - **Bit 12-63** BASE_GLA: Bits 63:12 of the guest linear address of the first 4KB page to be processed.
791pub struct GlaListEntry(u64);
792
793/// The `GlaListInfo` is used as a GPR input and output operand of TDG.VP.INVGLA.
794///
795/// It provides the GPA of the GLA list page in private memory,
796/// the index of the first entry and the number of entries to be processed.
797///
798/// The `GlaListInfo` consists of the following fields:
799/// - **Bit 0-8**    FIRST_ENTRY: Index of the first entry of the list to be processed.
800/// - **Bit 9-11**   RESERVED: Reserved: must be 0.
801/// - **Bit 12-51**  LIST_GPA: Bits 51:12 of the guest physical address of the GLA list page, which must be a private GPA.
802/// - **Bit 52-61**  NUM_ENTRIES: Number of entries in the GLA list to be processed, must be between 0 through 512.
803/// - **Bit 62-63**  RESERVED: Reserved: must be 0.
804pub struct GlaListInfo(u64);

853            vm1_attr: GpaAttr::from_bits_truncate(((val >> 16) & 0xFFFF) as u16),
854            vm2_attr: GpaAttr::from_bits_truncate(((val >> 32) & 0xFFFF) as u16),

860impl From<GpaAttrAll> for u64 {
861    fn from(s: GpaAttrAll) -> Self {
862        let field1 = s.l1_attr.bits() as u64;
863        let field2 = (s.vm1_attr.bits() as u64) << 16;
864        let field3 = (s.vm2_attr.bits() as u64) << 32;
865        let field4 = (s.vm3_attr.bits() as u64) << 48;
866        field4 | field3 | field2 | field1
867    }
868}
