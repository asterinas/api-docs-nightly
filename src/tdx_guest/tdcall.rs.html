36bitflags! {
37    /// Controls whether CPUID executed by the guest TD will cause #VE unconditionally.
38    pub struct CpuidveFlag: u64 {
39        /// Flags that when CPL is 0, a CPUID executed
40        /// by the guest TD will cause a #VE unconditionally.
41        const SUPERVISOR = 1 << 0;

43        /// by the guest TD will cause a #VE unconditionally.
44        const USER = 1 << 1;
45    }

48bitflags! {
49    /// GPA Attributes (Single VM) Definition.
50    pub struct GpaAttr: u16 {
51        /// Read.
52        const R = 1;
53        /// Write.
54        const W = 1 << 1;
55        /// Execute (Supervisor).
56        const XS = 1 << 2;
57        /// Execute (User).
58        const XU = 1 << 3;
59        /// Verify Guest Paging.
60        const VGP = 1 << 4;
61        /// Paging-Write Access.
62        const PWA = 1 << 5;
63        /// Supervisor Shadow Stack.
64        const SSS = 1 << 6;
65        /// Suppress #VE.
66        const SVE = 1 << 7;
67        /// Indicates that the other bits are valid.
68        /// If its value is 0, other fields are reserved and must be 0.
69        const VALID = 1 << 15;
70    }

79/// GPA Attributes (all VMs) Definition.
80pub struct GpaAttrAll {
81    /// L1 GPA attributes.
82    l1_attr: GpaAttr,
83    /// GPA attributes for L2 VM #1.
84    vm1_attr: GpaAttr,
85    /// GPA attributes for L2 VM #2.
86    vm2_attr: GpaAttr,
87    /// GPA attributes for L2 VM #3.
88    vm3_attr: GpaAttr,
89}
90
91impl From<u64> for GpaAttrAll {
92    fn from(val: u64) -> Self {
93        GpaAttrAll {
94            l1_attr: GpaAttr::from_bits_truncate((val & 0xFFFF) as u16),
95            vm1_attr: GpaAttr::from_bits_truncate(((val >> 16) & 0xFFFF) as u16),
96            vm2_attr: GpaAttr::from_bits_truncate(((val >> 32) & 0xFFFF) as u16),
97            vm3_attr: GpaAttr::from_bits_truncate(((val >> 48) & 0xFFFF) as u16),
98        }
99    }
100}
101
102impl From<GpaAttrAll> for u64 {
103    fn from(s: GpaAttrAll) -> Self {
104        let field1 = s.l1_attr.bits() as u64;
105        let field2 = (s.vm1_attr.bits() as u64) << 16;
106        let field3 = (s.vm2_attr.bits() as u64) << 32;
107        let field4 = (s.vm3_attr.bits() as u64) << 48;
108        field4 | field3 | field2 | field1
109    }
110}

334/// The `GlaListEntry` species a range of consecutive guest linear addresses, each aligned on 4KB.
335///
336/// The `GlaListEntry` consists of the following fields:
337/// - **Bit 0-11**  LAST_GLA_INDEX: Index of the last 4KB-aligned linear address to be processed.
338/// - **Bit 12-63** BASE_GLA: Bits 63:12 of the guest linear address of the first 4KB page to be processed.
339pub struct GlaListEntry(u64);
340
341/// The `GlaListInfo` is used as a GPR input and output operand of TDG.VP.INVGLA.
342///
343/// It provides the GPA of the GLA list page in private memory,
344/// the index of the first entry and the number of entries to be processed.
345///
346/// The `GlaListInfo` consists of the following fields:
347/// - **Bit 0-8**    FIRST_ENTRY: Index of the first entry of the list to be processed.
348/// - **Bit 9-11**   RESERVED: Reserved: must be 0.
349/// - **Bit 12-51**  LIST_GPA: Bits 51:12 of the guest physical address of the GLA list page, which must be a private GPA.
350/// - **Bit 52-61**  NUM_ENTRIES: Number of entries in the GLA list to be processed, must be between 0 through 512.
351/// - **Bit 62-63**  RESERVED: Reserved: must be 0.
352pub struct GlaListInfo(u64);

354#[derive(Debug, PartialEq)]
355pub enum TdCallError {
356    /// There is no valid #VE information.
357    TdxNoValidVeInfo,
358    /// Operand is invalid.
359    TdxOperandInvalid,
360    /// The operand is busy (e.g., it is locked in Exclusive mode).
361    TdxOperandBusy,
362    /// Page has already been accepted.
363    TdxPageAlreadyAccepted,
364    /// Requested page size does not match the current GPA mapping size.
365    TdxPageSizeMismatch,
366    /// The provided FIELD_ID is incorrect.
367    TdxMetadataFieldIdIncorrect,
368    /// Field code and write mask are for a read-only field.
369    TdxMetadataFieldNotWritable,
370    /// Field code is for an unreadable field.
371    TdxMetadataFieldNotReadable,
372    /// The provided field value is not valid.
373    TdxMetadataFieldValueNotValid,
374    /// The TD's OP_STATE is incorrect for the required operation.
375    TdxOpStateIncorrect,
376    /// Operand address is out of range (e.g., not in a TDMR).
377    TdxOperandAddrRangeError,
378    /// Physical page metadata (in PAMT) are incorrect for the requested operation.
379    TdxPageMetadataIncorrect,
380    /// Service TD hash of TDINFO_STRUCT does not match the currently bound hash.
381    TdxServtdInfoHashMismatch,
382    /// Service TD is not bound.
383    TdxServtdNotBound,
384    /// Service TD UUID does not match the currently bound UUID.
385    TdxServtdUuidMismatch,
386    /// Target TD UUID does not match the requested TD_UUID.
387    TdxTargetUuidMismatch,
388    /// Target TD UUID does not match the requested TD_UUID, but pre-migration target TD UUID does match it.
389    TdxTargetUuidUpdated,
390    /// TD is in a FATAL error state.
391    TdxTdFatal,
392    /// TD keys have not been configured on the hardware.
393    TdxTdKeysNotConfigured,
394    /// TDCS pages have not been allocated.
395    TdxTdcsNotAllocated,
396    Other,
397}

399impl From<u64> for TdCallError {
400    fn from(val: u64) -> Self {
401        match val {
402            0x0000_0B0A => Self::TdxPageAlreadyAccepted,
403            0x8000_0200 => Self::TdxOperandBusy,
404            0x8000_0810 => Self::TdxTdKeysNotConfigured,
405            0xC000_0100 => Self::TdxOperandInvalid,
406            0xC000_0101 => Self::TdxOperandAddrRangeError,
407            0xC000_0300 => Self::TdxPageMetadataIncorrect,
408            0xC000_0606 => Self::TdxTdcsNotAllocated,
409            0xC000_0608 => Self::TdxOpStateIncorrect,
410            0xC000_0704 => Self::TdxNoValidVeInfo,
411            0xC000_0B0B => Self::TdxPageSizeMismatch,
412            0xC000_0C00 => Self::TdxMetadataFieldIdIncorrect,
413            0xC000_0C01 => Self::TdxMetadataFieldNotWritable,
414            0xC000_0C02 => Self::TdxMetadataFieldNotReadable,
415            0xC000_0C03 => Self::TdxMetadataFieldValueNotValid,
416            0xC000_0D03 => Self::TdxServtdInfoHashMismatch,
417            0xC000_0D04 => Self::TdxServtdUuidMismatch,
418            0xC000_0D05 => Self::TdxServtdNotBound,
419            0xC000_0D07 => Self::TdxTargetUuidMismatch,
420            0xC000_0D08 => Self::TdxTargetUuidUpdated,
421            0xE000_0604 => Self::TdxTdFatal,
422            _ => Self::Other,
423        }
424    }
425}

488/// Get guest TD execution environment information.
489pub fn get_tdinfo() -> Result<TdgVpInfo, TdCallError> {
490    let mut args = TdcallArgs {
491        rax: TdcallNum::VpInfo as u64,
492        ..Default::default()

495    Ok(TdgVpInfo {

498        num_vcpus: args.r8 as u32,
499        max_vcpus: (args.r8 >> 32) as u32,
500        vcpu_index: args.r9 as u32,
501        sys_rd: args.r10 as u32,
502    })
503}

647/// Enter L2 VCPU operation.
648///
649/// Inputs:
650/// - l2_vm_idx: L2 virtual machine index (must be 1 or higher).
651/// - invd_translations: Controls how enter_l2_vcpu flushes the TLB context and extended paging structure (EPxE) caches
652///   associated with the L2 VM before entering the L2 VCPU.
653/// - guest_state_gpa: The GPA of a 256-bytes aligned L2EnterGuestState structure.
654///
655/// Outputs:
656/// - Return registers status in L2EnterGuestState.
657pub fn enter_l2_vcpu(
658    l2_vm_idx: u64,
659    invd_translations: InvdTranslations,
660    guest_state_gpa: u64,
661) -> Result<(), TdCallError> {
662    if (l2_vm_idx > 3) | (invd_translations.clone() as u64 > 3) {
663        return Err(TdCallError::TdxOperandInvalid);
664    }
665
666    let rcx = (l2_vm_idx << 52) | (invd_translations as u64);
667    let mut args = TdcallArgs {
668        rax: TdcallNum::VpEnter as u64,
669        rcx: l2_vm_idx,
670        rdx: guest_state_gpa,
671        ..Default::default()
672    };
673    td_call(&mut args)
674}

677///
678/// Inputs:
679/// - l2_vm_idx_bitmap: the index of the L2 VM to invalidate.
680///     Bit 1: Invalidate EPT for L2 VM #1.

688        rax: TdcallNum::VpInvept as u64,
689        rcx: l2_vm_idx_bitmap,
690        ..Default::default()
691    };
692    td_call(&mut args)
693}

695/// Invalidate Guest Linear Address (GLA) mappings in the translation lookaside buffers (TLBs) and paging-structure caches
696/// for a specified L2 VM and a specified list of 4KB-aligned linear addresses.
697///
698/// Inputs:
699/// - l2_vm_idx: the index of the L2 VM to invalidate.
700///     1: Invalidate EPT for L2 VM #1.
701///     2: Invalidate EPT for L2 VM #2.
702///     3: Invalidate EPT for L2 VM #3.
703/// - list: GlaListEntry or GlaListInfo Flags.
704///     0: gla contains a single GLA list entry.
705///     1: RDX contains the GPA and other information of a GLA list in memory.
706/// - gla: Depending on the list flag, it contains either of the following:
707///     - A single GlaListEntry, specifying up to 512 consecutive guest linear addresses, each aligned on 4KB.
708///     - GlaLstInfo, specifying the GPA of a guest linear address (GLA) list in private
709///       memory. Each entry in the GLA list specifies up to 512 consecutive guest linear
710///       addresses, each aligned on 4KB. GlaLstInfo also specifies the first and last GLA
711///       list entries to process.
712///
713/// Outputs:
714/// - RDX: Depending on the list flag, it contains either of the following:
715///     - If list was 0, RDX contains the single GlaListEntry provided as an input, unmodified.
716///     - If list was 1, RDX contains the GlaLstInfo provided as input,
717///       but with the FIRST_ENTRY and NUM_ENTRIES fields updated to reflect the number of entries processed so far.
718///       If all entries have been processed successfully, NUM_ENTRIES is set to 0.
719pub fn invalidate_l2_gla(l2_vm_idx: u64, list: bool, gla: u64) -> Result<u64, TdCallError> {
720    if l2_vm_idx > 3 {
721        return Err(TdCallError::TdxOperandInvalid);
722    }
723
724    let rcx = (l2_vm_idx << 52) | if list { 0b1 } else { 0 };
725    let rdx = if list {
726        Gla::ListEntry(GlaListEntry(gla))
727    } else {
728        Gla::ListInfo(GlaListInfo(gla))
729    }
730    .value();
731
732    let mut args = TdcallArgs {
733        rax: TdcallNum::VpInvgla as u64,
734        rcx,
735        rdx,
736        ..Default::default()
737    };
738    td_call(&mut args).map(|_| args.rdx)
739}

741/// As a service TD, read a metadata field (control structure field) of a target TD.
742///
743/// Inputs:
744/// - binding_handle: the binding handle of the target TD.
745/// - field_identifier: the identifier of the field to read.
746///   The `LAST_ELEMENT_IN_FIELD` and `LAST_FIELD_IN_SEQUENCE` components of the field identifier must be 0.
747///   `WRITE_MASK_VALID`, `INC_SIZE`, `CONTEXT_CODE` and `ELEMENT_SIZE_CODE` components of the field identifier are ignored.
748///   A value of -1 is a special case: it is not a valid field identifier; in this case the first readable field identifier is returned in `RDX`.
749/// - uuid: the TD_UUID of the target TD, using little-Endian.
750///
751/// Outputs:
752/// - Next readable field identifier. A value of -1 indicates no next field identifier is available.
753///   If the input field identifier was -1, `RDX` returns the first readable field identifier.
754///   In case of another error, `RDX` returns -1.
755/// - Contents of the field. In case of an error, as indicated by `RAX`, `R8` returns 0.
756/// - Updated target TD’s TD_UUID, using little-Endian.
757pub fn read_servetd(
758    binding_handle: u64,
759    field_identifier: u64,
760    uuid: [u64; 4],
761) -> Result<(u64, u64, [u64; 4]), TdCallError> {
762    let mut args = TdcallArgs {
763        rax: TdcallNum::ServetdRd as u64,
764        rcx: binding_handle,
765        rdx: field_identifier,
766        r10: uuid[0],
767        r11: uuid[1],
768        r12: uuid[2],
769        r13: uuid[3],
770        ..Default::default()
771    };
772    td_call(&mut args).map(|_| (args.rdx, args.r8, [args.r10, args.r11, args.r12, args.r13]))
773}

775/// As a service TD, write a metadata field (control structure field) of a target TD.
776///
777/// Inputs:
778/// - binding_handle: the binding handle of the target TD.
779/// - field_identifier: the identifier of the field to read.
780///   The `LAST_ELEMENT_IN_FIELD` and `LAST_FIELD_IN_SEQUENCE` components of the field identifier must be 0.
781///   `WRITE_MASK_VALID`, `INC_SIZE`, `CONTEXT_CODE` and `ELEMENT_SIZE_CODE` components of the field identifier are ignored.
782///   A value of -1 is a special case: it is not a valid field identifier; in this case the first readable field identifier is returned in `RDX`.
783/// - data: Data to write to the field.
784/// - write_mask: A 64b write mask to indicate which bits of the value in `R8` are to be written to the field.
785/// - uuid: the TD_UUID of the target TD, using little-Endian.
786///
787/// Outputs:
788/// - Previous contents of the field. In case of an error, `R8` returns 0.
789/// - Updated target TD’s TD_UUID, using little-Endian.
790pub fn write_servetd(
791    binding_handle: u64,
792    field_identifier: u64,
793    data: u64,
794    mask: u64,
795    uuid: [u64; 4],
796) -> Result<(u64, [u64; 4]), TdCallError> {
797    let mut args = TdcallArgs {
798        rax: TdcallNum::ServetdWr as u64,
799        rcx: binding_handle,
800        rdx: field_identifier,
801        r8: data,
802        r9: mask,
803        r10: uuid[0],
804        r11: uuid[1],
805        r12: uuid[2],
806        r13: uuid[3],
807    };
808    td_call(&mut args).map(|_| (args.r8, [args.r10, args.r11, args.r12, args.r13]))
809}
810
811fn td_call(args: &mut TdcallArgs) -> Result<(), TdCallError> {
812    let result = unsafe { asm_td_call(args) };
813    match result {
814        0 => Ok(()),
815        _ => Err((result >> 32).into()),
816    }
