99/// Returns true if the system is identified as an Intel TDX guest during early boot.
100///
101/// This function is designed for use in environments like the EFI stub where
102/// complex initialization is not yet possible. It uses an internal atomic cache
103/// to ensure that the hardware CPUID check is performed only once.
104pub fn is_tdx_guest_early() -> bool {
105    match TdxEarlyState::from(TDX_EARLY_STATE.load(Acquire)) {
106        TdxEarlyState::Enabled => true,
107        TdxEarlyState::Disabled => false,
108        _ => {
109            let is_tdx = is_tdx_hardware_present();
110            let new_state = if is_tdx {
111                TdxEarlyState::Enabled
112            } else {
113                TdxEarlyState::Disabled
114            };
115
116            TDX_EARLY_STATE.store(new_state as u8, Release);
117            is_tdx
118        }
119    }
120}

122/// Performs full initialization of the Intel TDX guest environment.
123///
124/// This function validates the TDX hardware signature, invokes the `TDG.VP.INFO`
125/// TDCALL to retrieve Trust Domain environment information, and configures global
126/// state such as the shared memory mask.
127pub fn init_tdx() -> Result<TdgVpInfo, InitError> {
128    if tdx_is_enabled() {
129        return tdcall::get_tdinfo().map_err(InitError::TdxGetVpInfoError);
130    }
131
132    check_tdx_guest()?;
133
134    let info = tdcall::get_tdinfo().map_err(InitError::TdxGetVpInfoError)?;
135
136    let gpaw: u64 = info.gpaw.into();
137    let mask = 1u64 << (gpaw - 1);
138    SHARED_MASK.store(mask, Relaxed);
139
140    TDX_ENABLED.store(true, Relaxed);
141
142    Ok(info)
143}

145pub fn handle_virtual_exception(trapframe: &mut dyn TdxTrapFrame, ve_info: &TdgVeInfo) {
146    let mut instr_len = ve_info.exit_instruction_length;
147    match ve_info.exit_reason.into() {
148        TdxVirtualExceptionType::Hlt => {
149            hlt();
150        }
151        TdxVirtualExceptionType::Io => {
152            if !handle_io(trapframe, ve_info) {
153                serial_println!("Handle tdx ioexit errors, ready to halt");
154                halt_forever();
155            }
156        }
157        TdxVirtualExceptionType::MsrRead => {
158            let msr = unsafe { rdmsr(trapframe.rcx() as u32).unwrap() };
159            trapframe.set_rax((msr as u32) as usize);
160            trapframe.set_rdx(((msr >> 32) as u32) as usize);
161        }
162        TdxVirtualExceptionType::MsrWrite => {
163            let data = trapframe.rax() as u64 | ((trapframe.rdx() as u64) << 32);
164            unsafe { wrmsr(trapframe.rcx() as u32, data).unwrap() };
165        }
166        TdxVirtualExceptionType::CpuId => {
167            let cpuid_info = cpuid(trapframe.rax() as u32, trapframe.rcx() as u32).unwrap();
168            trapframe.set_rax(cpuid_info.eax);
169            trapframe.set_rbx(cpuid_info.ebx);
170            trapframe.set_rcx(cpuid_info.ecx);
171            trapframe.set_rdx(cpuid_info.edx);
172        }
173        TdxVirtualExceptionType::EptViolation => {
174            if is_protected_gpa(ve_info.guest_physical_address as TdxGpa) {
175                serial_println!("Unexpected EPT-violation on private memory");
176                halt_forever();
177            }
178            instr_len = handle_mmio(trapframe, ve_info).unwrap() as u32;
179        }
180        TdxVirtualExceptionType::Other => {
181            serial_println!("Unknown TDX virtual exception type");
182            halt_forever();
183        }
184        _ => return,
185    }
186    trapframe.set_rip(trapframe.rip() + instr_len as usize);
187}

203/// Accepts a range of physical memory to be used as TDX private memory.
204///
205/// # Safety
206///
207/// The caller must ensure the following invariants are met:
208/// - **Address Validity**: The GPA range `[gpa_start, gpa_end)` must represent a valid range.
209/// - **State Invariant**: The target memory pages must be in the `Pending` state.
210///   Accepting pages that are already `Accepted` or in an uninitialized state will
211///   result in a TDX instruction error.
212/// - **Exclusive Access**: The caller must ensure no other CPU context is
213///   simultaneously attempting to accept or access this specific GPA range to
214///   prevent race conditions in the TDX Module's metadata.
215/// - **Alignment**: While the function checks basic 4K alignment, the caller must ensure
216///   the range corresponds to actual physical backing store provided by the VMM.
217pub unsafe fn accept_memory(gpa_start: u64, gpa_end: u64) -> Result<(), AcceptError> {
218    if gpa_start >= gpa_end {
219        return Ok(());
220    }
221
222    if (gpa_start & (PageLevel::L1_4K.bytes() - 1)) != 0 {
223        return Err(AcceptError::InvalidAlignment);
224    }
225
226    let mut current_addr = gpa_start;
227
228    while current_addr < gpa_end {
229        let len = gpa_end - current_addr;
230        let mut accepted = false;
231
232        for &level in &PageLevel::PRIORITIES {
233            match try_accept_one(current_addr, len, level)? {
234                TryAcceptResult::Accepted(size) => {
235                    current_addr += size;
236                    accepted = true;
237                    break;
238                }
239                TryAcceptResult::SizeMismatch | TryAcceptResult::SkipLevel => {
240                    // Try next (smaller) page level
241                    continue;
242                }
243            }
244        }
245
246        if !accepted {
247            // Fails if even L1_4K cannot be accepted
248            return Err(AcceptError::InvalidAlignment);
249        }
250    }
251    Ok(())
252}

254pub fn enable_cpu_topology_enumeration() -> Result<(), TopologyError> {
255    let configured = tdcall::read_td_metadata(metadata::TOPOLOGY_ENUM_CONFIGURED)?;
256
257    if configured == 0 {
258        return Err(TopologyError::NotConfigured);
259    }
260
261    tdcall::write_td_metadata(
262        metadata::TD_CTLS,
263        metadata::TdCtls::ENUM_TOPOLOGY.bits(),
264        metadata::TdCtls::ENUM_TOPOLOGY.bits(),
265    )?;
266
267    Ok(())
268}

270pub fn disable_sept_ve(td_attr: TdAttributes) -> Result<(), SeptVeError> {
271    let debug = td_attr.contains(TdAttributes::DEBUG);
272
273    let config = ConfigFlags::from_bits_truncate(tdcall::read_td_metadata(metadata::CONFIG_FLAGS)?);
274
275    if !config.contains(ConfigFlags::FLEXIBLE_PENDING_VE) {
276        if td_attr.contains(TdAttributes::SEPT_VE_DISABLE) {
277            return Ok(());
278        }
279
280        if !debug {
281            return Err(SeptVeError::Misconfiguration);
282        }
283        return Ok(());
284    }
285
286    let controls =
287        metadata::TdCtls::from_bits_truncate(tdcall::read_td_metadata(metadata::TD_CTLS)?);
288
289    if controls.contains(metadata::TdCtls::PENDING_VE_DISABLE) {
290        return Ok(());
291    }
292
293    if debug {
294        return Ok(());
295    }
296
297    tdcall::write_td_metadata(
298        metadata::TD_CTLS,
299        metadata::TdCtls::PENDING_VE_DISABLE.bits(),
300        metadata::TdCtls::PENDING_VE_DISABLE.bits(),
301    )?;
302
303    Ok(())
304}

306bitflags! {
307    /// TdAttributes is defined as a 64b field that specifies various attested guest TD attributes.
308    pub struct TdAttributes: u64 {
309        /// Guest TD runs in off-TD debug mode. Its VCPU state and private memory are accessible by the host VMM.
310        /// DEBUG may not be set if MIGRATABLE is set.
311        const DEBUG = 1 << 0;
312        /// The TD is subject to HGS+ operation. HGS+ monitors the TD operation as part of the whole system.
313        /// This bit may be set, if supported by the TDX module, regardless of CPU support.
314        const HGS_PLUS_PROF = 1 << 4;
315        /// The TD is subject to system profiling using performance monitoring counters.
316        /// Those counters are not context-switched on TD entry and exit; they monitor the TD operation as part of the whole system.
317        /// This bit may be set, if supported by the TDX module, regardless of CPU support.
318        const PERF_PROF = 1 << 5;
319        /// The TD is subject to system profiling using core out-of-band telemetry.
320        /// Core telemetry monitors the TD operation as part of the whole system.
321        /// This bit may be set, if supported by the TDX module, regardless of CPU support.
322        const PMT_PROF = 1 << 6;
323        /// Indicates that the TDX module must use Instruction-Count based Single-Step Defense to protect against single-step attacks.
324        /// ICSSD may not be set if PERFMON is set.
325        /// This bit may only be set if the TDX module supports ICSSD.
326        const ICSSD = 1 << 16;
327        /// TD is allowed to use Linear Address Space Separation.
328        /// This bit may only be set if both the TDX module and the CPU support LASS.
329        const LASS = 1 << 27;
330        /// Disable EPT violation conversion to #VE(PENDING) on guest TD access of PENDING pages.
331        const SEPT_VE_DISABLE = 1 << 28;
332        /// TD is migratable (using a Migration TD).
333        /// MIGRATABLE may not be set if either DEBUG or PERFMON is set.
334        /// MIGRATABLE may not be set if CONFIG_FLAGS.TDX_CONNECT is set.
335        /// This bit may only be set if the TDX module supports TD Migration.
336        const MIGRATABLE = 1 << 29;
337        /// TD is allowed to use Supervisor Protection Keys.
338        /// This bit may only be set if both the TDX module and the CPU support PKS.
339        const PKS = 1 << 30;
340        /// TD is allowed to use Key Locker.
341        /// This bit may only be set if both the TDX module and the CPU support Key Locker.
342        const KL = 1 << 31;
343        /// The TD is a TDX Connect Provisioning Agent. This bit may only be set if both the TDX module and the CPU support TDX Connect.
344        const TPA = 1 << 62;
345        /// TD is allowed to use Perfmon and PERF_METRICS capabilities.
346        /// PERFMON may not be set if either MIGRATABLE or ICSSD is set.
347        /// This bit may only be set if the TDX Module supports Performance Monitoring virtualization.
348        const PERFMON = 1 << 63;
349    }
350}

383pub(crate) fn is_protected_gpa(gpa: TdxGpa) -> bool {
384    let mask = SHARED_MASK.load(Relaxed);
385    let gpa_u64 = u64::try_from(gpa).expect("TdxGpa must fit into u64 on x86_64");
386    (gpa_u64 & mask) == 0
387}

389fn check_tdx_guest() -> Result<(), InitError> {
390    let max_leaf = cpuid_count(0, 0).eax;
391    if max_leaf < TDX_CPUID_LEAF_ID {
392        return Err(InitError::TdxCpuLeafIdTooLow);
393    }
394    if !is_tdx_hardware_present() {
395        return Err(InitError::TdxVendorIdMismatch);
396    }
397
398    Ok(())
399}
400
401fn is_tdx_hardware_present() -> bool {
402    let res: CpuIdResult = cpuid_count(TDX_CPUID_LEAF_ID, 0);
403
404    let mut sig = [0u8; 12];
405    sig[0..4].copy_from_slice(&res.ebx.to_le_bytes());
406    sig[4..8].copy_from_slice(&res.edx.to_le_bytes());
407    sig[8..12].copy_from_slice(&res.ecx.to_le_bytes());
408
409    &sig == TDX_IDENT
410}

448/// Attempts to accept a single memory page at the specified level.
449fn try_accept_one(
450    start: u64,
451    len: u64,
452    page_level: PageLevel,
453) -> Result<TryAcceptResult, AcceptError> {
454    let size = page_level.bytes();
455
456    if (start & (size - 1)) != 0 || len < size {
457        return Ok(TryAcceptResult::SkipLevel);
458    }
459
460    match unsafe { accept_page(page_level as u64, start) } {
461        Ok(_) => Ok(TryAcceptResult::Accepted(size)),
462        Err(e) => match e {
463            // PageAlreadyAccepted: If 4K, it is fully accepted. If larger, earlier stages
464            // might have accepted only a subregion (e.g. 4K out of 2M), so we must split.
465            TdCallError::TdxPageAlreadyAccepted => {
466                if page_level == PageLevel::L1_4K {
467                    Ok(TryAcceptResult::Accepted(size))
468                } else {
469                    Ok(TryAcceptResult::SizeMismatch)
470                }
471            }
472            // PageSizeMismatch: VMM mapped it differently.
473            // OperandInvalid: Hardware doesn't support this size or address is rejected.
474            TdCallError::TdxPageSizeMismatch | TdCallError::TdxOperandInvalid => {
475                if page_level == PageLevel::L1_4K {
476                    // If the minimum architectural unit is rejected, it's a fatal error.
477                    Err(AcceptError::TdCall(e))
478                } else {
479                    // Fall back to a smaller page size.
480                    Ok(TryAcceptResult::SizeMismatch)
481                }
482            }
483            _ => Err(AcceptError::TdCall(e)),
484        },
485    }
486}

508#[derive(Clone, Copy, Debug, Eq, PartialEq)]
509#[repr(u8)]
510enum PageLevel {
511    L1_4K = 0,
512    L2_2M = 1,
513    L3_1G = 2,
514}
515
516impl PageLevel {
517    pub const PRIORITIES: [Self; 3] = [Self::L3_1G, Self::L2_2M, Self::L1_4K];
518
519    pub const fn bytes(self) -> u64 {
520        1 << (12 + (self as u32) * 9)
521    }
522}

524/// Represents the result of a single page acceptance attempt.
525enum TryAcceptResult {
526    /// Successfully accepted a page of the given size.
527    Accepted(u64),
528    /// Current address or length is not aligned/sufficient for this level.
529    SkipLevel,
530    /// Hardware/VMM reports a size mismatch or lack of support for this level.
531    SizeMismatch,
532}
