4//! Support for unaccepted memory in TDX guest environments.
5//! This module provides mechanisms to manage and accept unaccepted memory regions in TDX guests.
6//! The core data structure is `EfiUnacceptedMemory`, which represents the EFI table header and provides
7//! methods to manipulate the unaccepted memory bitmap and perform acceptance operations.
8
9use crate::{accept_memory, AcceptError};
10
11/// GUID of the Linux-compatible unaccepted-memory EFI table.
12pub const LINUX_EFI_UNACCEPTED_MEM_TABLE_GUID: uefi_raw::Guid =
13    uefi_raw::guid!("d5d1de3c-105c-44f9-9ea9-bcef98120031");
14
15/// Version of the Linux-compatible unaccepted-memory EFI table supported here.
16pub const LINUX_EFI_UNACCEPTED_MEM_TABLE_VERSION: u32 = 1;
17
18/// Unit size for unaccepted-memory bitmap entries (2 MiB).
19pub const EFI_UNACCEPTED_UNIT_SIZE: u64 = 2 * 1024 * 1024;
20
21/// Header of the Linux-compatible EFI unaccepted-memory table.
22///
23/// This type describes only the fixed-size header. The bitmap payload is stored
24/// immediately after the header in memory (C-style trailing data):
25///
26/// ### Memory Layout
27/// The total memory footprint is
28/// `size_of::<EfiUnacceptedMemory>() + self.bitmap_size_bytes`.
29/// The bitmap begins at the first byte following this structure.
30///
31/// ### Bitmap Semantics
32/// - Each bit in the trailing bitmap represents a memory region of
33///   `unit_size_bytes` bytes.
34/// - Bit 0 corresponds to the physical address specified by `phys_base`.
35/// - A **set bit (1)** indicates memory is unaccepted (pending);
36///   a **cleared bit (0)** indicates it has been accepted.
37///
38/// ### Concurrency Contract
39/// This type does not provide internal synchronization for bitmap mutation.
40/// Callers must serialize mutating operations (for example with a spinlock)
41/// before invoking methods like `register_range` or `accept_range`.
42#[derive(Debug)]
43#[repr(C, packed)]
44pub struct EfiUnacceptedMemory {
45    /// The version of the table. Currently, only version 1 is defined.
46    version: u32,
47    /// The size of the memory region represented by a single bit in the bitmap.
48    /// Typically set to 2MiB (0x200000) to align with huge page boundaries.
49    unit_size_bytes: u32,
50    /// The start physical address of the memory range covered by this bitmap.
51    /// Bit 0 of the bitmap corresponds to this address.
52    phys_base: u64,
53    /// The bitmap payload length in bytes, excluding this header.
54    bitmap_size_bytes: u64,
55}
56
57impl EfiUnacceptedMemory {
58    /// Initializes the table header fields for EFI installation.
59    pub fn init_header(
60        &mut self,
61        unit_size_bytes: u32,
62        phys_base: u64,
63        bitmap_size_bytes: u64,
64    ) -> Result<(), AcceptError> {
65        if unit_size_bytes == 0 || !unit_size_bytes.is_power_of_two() {
66            return Err(AcceptError::InvalidAlignment);
67        }
68
69        self.version = LINUX_EFI_UNACCEPTED_MEM_TABLE_VERSION;
70        self.unit_size_bytes = unit_size_bytes;
71        self.phys_base = phys_base;
72        self.bitmap_size_bytes = bitmap_size_bytes;
73        Ok(())
74    }

96    /// Returns whether `(start, size)` overlaps any pending bitmap unit.
97    ///
98    /// # Safety
99    ///
100    /// The caller must ensure this header is followed in memory by at least
101    /// `self.bitmap_size_bytes` readable bitmap bytes.
102    pub unsafe fn is_range_pending_by_size(
103        &self,
104        start: u64,
105        size: u64,
106    ) -> Result<bool, AcceptError> {
107        let Some(end) = start.checked_add(size) else {
108            return Err(AcceptError::ArithmeticOverflow);
109        };
110
111        // SAFETY: Caller guarantees the table header is followed by readable bitmap bytes.
112        unsafe { self.is_range_pending(start, end) }
113    }

115    /// Returns whether `[start, end)` overlaps any pending bitmap unit.
116    ///
117    /// # Safety
118    ///
119    /// The caller must ensure this header is followed in memory by at least
120    /// `self.bitmap_size_bytes` readable bitmap bytes.
121    pub unsafe fn is_range_pending(&self, start: u64, end: u64) -> Result<bool, AcceptError> {
122        let Some((range_start, range_end, unit_size)) =
123            self.clamp_gpa_range_to_bitmap_coverage(start, end)?
124        else {
125            return Ok(false);
126        };
127
128        let (first_bit, last_bit) = self.addr_to_bit_range(range_start, range_end, unit_size)?;
129        // SAFETY: Caller guarantees the table header is followed by readable bitmap bytes.
130        let bitmap = unsafe { self.as_bitmap_slice() };
131        BitmapRef::new(bitmap).has_set_bit(first_bit, last_bit)
132    }

134    /// Returns whether every bitmap unit overlapping `[start, end)` is accepted.
135    ///
136    /// Ranges outside bitmap coverage are considered accepted by definition,
137    /// because this table only tracks deferred acceptance inside its own coverage.
138    ///
139    /// # Safety
140    ///
141    /// The caller must ensure this header is followed in memory by at least
142    /// `self.bitmap_size_bytes` readable bitmap bytes.
143    pub unsafe fn is_fully_accepted(&self, start: u64, end: u64) -> Result<bool, AcceptError> {
144        // SAFETY: Caller guarantees the table header is followed by readable bitmap bytes.
145        Ok(!unsafe { self.is_range_pending(start, end) }?)
146    }

161    /// Returns [`AcceptError::ArithmeticOverflow`] if `start + size` overflows.

163    pub unsafe fn accept_by_size(&mut self, start: u64, size: u64) -> Result<(), AcceptError> {
164        let Some(end) = start.checked_add(size) else {
165            return Err(AcceptError::ArithmeticOverflow);
166        };
167
168        // SAFETY: Caller guarantees table/bitmap validity and target range correctness.

170    }

172    /// Accepts bitmap-marked units that overlap `start..end`, then clears accepted bits.
173    ///
174    /// The input is interpreted as a half-open GPA interval `[start, end)`.
175    ///
176    /// Behavior summary:
177    /// - The requested range is first clamped to bitmap coverage.
178    /// - Any bitmap bit set to `1` and overlapping the clamped range is accepted.
179    /// - Successfully accepted bits are cleared to `0` in-place.
180    /// - If the clamped range is empty, this is a no-op.
181    ///
182    /// # Safety
183    ///
184    /// The caller must ensure this table and bitmap describe pending private-memory units,
185    /// and the target GPA ranges are valid for TDX acceptance.
186    ///
187    /// # Errors
188    ///
189    /// Returns [`AcceptError::InvalidAlignment`] for invalid unit configuration.
190    /// Returns [`AcceptError::ArithmeticOverflow`] for address/index arithmetic overflow.
191    /// Returns [`AcceptError::OutOfBounds`] for bitmap index out-of-range accesses.
192    /// Returns hardware-originated failures from `accept_memory` via
193    /// [`AcceptError::TdCall`].
194    pub unsafe fn accept_range(&mut self, start: u64, end: u64) -> Result<(), AcceptError> {
195        // SAFETY: Caller guarantees table/bitmap validity and target range correctness.
196        let _ = unsafe { self.accept_if_needed_range(start, end) }?;
197        Ok(())
198    }

200    /// Returns the end GPA (exclusive) covered by the bitmap.
201    ///

203    pub fn bitmap_coverage_end(&self) -> Option<u64> {
204        let base = PhysAddr::new(self.phys_base);
205        Some(base.checked_add(self.total_coverage_size()?).ok()?.raw())

208    /// Returns an immutable slice view of the trailing bitmap payload.
209    ///
210    /// # Safety
211    ///
212    /// The caller must ensure that this header is followed by at least
213    /// `self.bitmap_size_bytes`
214    /// readable bytes in memory.
215    pub unsafe fn as_bitmap_slice(&self) -> &[u8] {
216        debug_assert!(self.byte_len().is_ok());
217        let bitmap_ptr = core::ptr::from_ref(self)
218            .cast::<u8>()
219            .wrapping_add(core::mem::size_of::<Self>());
220        let bitmap_len = self
221            .byte_len()
222            .expect("bitmap size must fit usize on this platform");
223        // SAFETY: `bitmap_ptr` points to the trailing bitmap bytes immediately
224        // after `self`; `bitmap_len` is validated from `self.bitmap_size_bytes`;
225        // caller guarantees
226        // readable backing memory for the returned slice.
227        unsafe { core::slice::from_raw_parts(bitmap_ptr, bitmap_len) }
228    }

230    /// Returns a mutable slice view of the trailing bitmap payload.
231    ///
232    /// # Safety
233    ///
234    /// The caller must ensure that this header is followed by at least
235    /// `self.bitmap_size_bytes`
236    /// writable bytes in memory, and that no aliased mutable reference exists
237    /// while the returned slice is in use.
238    pub unsafe fn as_bitmap_slice_mut(&mut self) -> &mut [u8] {
239        debug_assert!(self.byte_len().is_ok());
240        let bitmap_ptr_mut = core::ptr::from_mut(self)
241            .cast::<u8>()
242            .wrapping_add(core::mem::size_of::<Self>());
243        debug_assert!(!bitmap_ptr_mut.is_null());
244        let bitmap_len = self
245            .byte_len()
246            .expect("bitmap size must fit usize on this platform");
247        // SAFETY: `bitmap_ptr_mut` points to the trailing bitmap bytes immediately
248        // after `self`; `bitmap_len` is validated from `self.bitmap_size_bytes`;
249        // caller guarantees
250        // writable backing memory and unique mutable access for the returned slice.
251        unsafe { core::slice::from_raw_parts_mut(bitmap_ptr_mut, bitmap_len) }
252    }

254    /// Processes `start..end` by eagerly accepting required parts and deferring the rest in bitmap.
255    ///
256    /// This method applies a hybrid policy:
257    /// - edge fragments that are not `unit_size`-aligned are accepted immediately;
258    /// - aligned interior regions within bitmap coverage are marked as unaccepted bits;
259    /// - aligned regions outside bitmap coverage are accepted immediately.
260    ///
261    /// # Safety
262    ///
263    /// The caller must ensure the range is valid guest-private memory in pending/acceptable state.
264    ///
265    /// # Errors
266    ///
267    /// Returns [`AcceptError::InvalidAlignment`] for invalid unit configuration.
268    /// Returns [`AcceptError::ArithmeticOverflow`] for address arithmetic overflow.
269    /// Returns [`AcceptError::Overlap`] if the same bitmap unit is registered twice.
270    /// Returns hardware-originated errors from `accept_memory`
271    /// via [`AcceptError::TdCall`].
272    pub unsafe fn register_range(&mut self, start: u64, end: u64) -> Result<(), AcceptError> {
273        let table_phys_base = self.phys_base;
274        let unit_size = self.validated_unit_size()?;
275        if start >= end {
276            return Ok(());
277        }
278
279        let unit_mask = unit_size - 1;
280
281        if end - start < 2 * unit_size {
282            // SAFETY: Caller guarantees the physical range is valid for TDX acceptance.
283            return unsafe { Self::try_accept_range(start, end) };
284        }
285
286        let mut current_start = start;
287        let mut current_end = end;
288
289        if current_start & unit_mask != 0 {
290            let Some(aligned_start) = align_up(current_start, unit_size) else {
291                return Err(AcceptError::ArithmeticOverflow);
292            };
293            // SAFETY: Caller guarantees the physical subrange is valid for TDX acceptance.
294            unsafe { Self::try_accept_range(current_start, aligned_start)? };
295            current_start = aligned_start;
296        }
297
298        if current_end & unit_mask != 0 {
299            let aligned_end = align_down(current_end, unit_size);
300            // SAFETY: Caller guarantees the physical subrange is valid for TDX acceptance.
301            unsafe { Self::try_accept_range(aligned_end, current_end)? };
302            current_end = aligned_end;
303        }
304
305        let Some(bitmap_coverage) = self.total_coverage_size() else {
306            return Err(AcceptError::ArithmeticOverflow);
307        };
308
309        let Some(bitmap_end) = table_phys_base.checked_add(bitmap_coverage) else {
310            return Err(AcceptError::ArithmeticOverflow);
311        };
312
313        // 1) Process aligned range before bitmap coverage.
314        if current_start < table_phys_base {
315            let accept_end = current_end.min(table_phys_base);
316            // SAFETY: Caller guarantees the physical subrange is valid for TDX acceptance.
317            unsafe { Self::try_accept_range(current_start, accept_end)? };
318            current_start = accept_end;
319        }
320
321        if current_start >= current_end {
322            return Ok(());
323        }
324
325        // 2) Process aligned range within bitmap coverage.
326        if current_start < bitmap_end {
327            let bitmap_range_end = current_end.min(bitmap_end);
328            if current_start < bitmap_range_end {
329                // SAFETY: GPA range is unit-aligned and within bitmap coverage.
330                unsafe {
331                    self.mark_range_as_unaccepted(current_start, bitmap_range_end, unit_size)?
332                };
333            }
334            current_start = bitmap_range_end;
335        }
336
337        // 3) Process aligned range after bitmap coverage.
338        if current_start < current_end {
339            // SAFETY: Caller guarantees the physical subrange is valid for TDX acceptance.
340            unsafe { Self::try_accept_range(current_start, current_end)? };
341        }
342
343        Ok(())
344    }

346    pub fn total_coverage_size(&self) -> Option<u64> {
347        let unit_size = u64::from(self.unit_size_bytes);
348        self.bitmap_size_bytes
349            .checked_mul(unit_size)?
350            .checked_mul(8)
351    }

353    /// Conditionally accepts bitmap-marked units that overlap `start..end`.
354    ///
355    /// Returns [`AcceptOutcome::AlreadyAccepted`] when the range overlaps the
356    /// bitmap but there are no pending bits to process.
357    ///
358    /// # Safety
359    ///
360    /// Same requirements as [`Self::accept_range`].
361    unsafe fn accept_if_needed_range(
362        &mut self,
363        start: u64,
364        end: u64,
365    ) -> Result<AcceptOutcome, AcceptError> {
366        if start >= end {
367            return Ok(AcceptOutcome::AlreadyAccepted);
368        }
369
370        let (range_start, range_end, unit_size) =
371            match self.clamp_gpa_range_to_bitmap_coverage(start, end)? {
372                Some(vals) => vals,
373                None => return Ok(AcceptOutcome::OutOfCoverage),
374            };
375
376        let (first_bit, last_bit) = self.addr_to_bit_range(range_start, range_end, unit_size)?;
377        let phys_base = PhysAddr::new(self.phys_base);
378
379        let bit_to_gpa = |bit: BitIndex| -> Result<u64, AcceptError> {
380            Ok(phys_base.checked_add_units(bit, unit_size)?.raw())
381        };
382
383        // SAFETY: Caller guarantees table/bitmap validity and exclusive mutable access.
384        let mut bitmap = BitmapMut::new(unsafe { self.as_bitmap_slice_mut() });
385        let mut accepted_units = 0u64;
386
387        let mut scan = first_bit;
388        while let Some(run_start) = bitmap.find_next_set(scan, last_bit)? {
389            let run_end = bitmap
390                .find_next_zero(run_start, last_bit)?
391                .unwrap_or(last_bit);
392
393            let run_gpa_start = bit_to_gpa(run_start)?;
394            let run_gpa_end = bit_to_gpa(run_end)?;
395
396            // SAFETY: Caller guarantees bitmap/GPA mapping validity for pending private pages.
397            unsafe { accept_memory(run_gpa_start, run_gpa_end)? };
398            accepted_units = accepted_units
399                .checked_add(run_end.raw() - run_start.raw())
400                .ok_or(AcceptError::ArithmeticOverflow)?;
401
402            bitmap.clear_range(run_start, run_end)?;
403
404            scan = run_end;
405        }
406
407        match accepted_units {
408            0 => Ok(AcceptOutcome::AlreadyAccepted),
409            n => Ok(AcceptOutcome::AcceptedNow { accepted_units: n }),
410        }
411    }

413    /// Marks `start..end` (bitmap-relative) as unaccepted bits.
414    ///
415    /// # Safety
416    ///
417    /// The caller must ensure `start..end` is already converted to bitmap-relative offsets
418    /// (i.e., based on `self.phys_base`) and does not violate bitmap ownership/aliasing rules.
419    unsafe fn set_unaccepted_bits(&mut self, start: u64, end: u64) -> Result<(), AcceptError> {
420        let unit_size = self.validated_unit_size()?;
421
422        let abs_start = self
423            .phys_base
424            .checked_add(start)
425            .ok_or(AcceptError::ArithmeticOverflow)?;
426        let abs_end = self
427            .phys_base
428            .checked_add(end)
429            .ok_or(AcceptError::ArithmeticOverflow)?;
430
431        // SAFETY: Caller guarantees bitmap-relative range correctness and exclusive access.
432        unsafe { self.mark_range_as_unaccepted(abs_start, abs_end, unit_size) }
433    }

445    fn validated_unit_size(&self) -> Result<u64, AcceptError> {
446        let unit_size = u64::from(self.unit_size_bytes);
447        if unit_size == 0 || !unit_size.is_power_of_two() {
448            return Err(AcceptError::InvalidAlignment);
449        }
450        Ok(unit_size)

453    fn max_phys_addr_exclusive(&self, unit_size: u64) -> Result<u64, AcceptError> {
454        let total_bits = self.total_bits()?;
455        let coverage_len = total_bits
456            .checked_mul(unit_size)
457            .ok_or(AcceptError::ArithmeticOverflow)?;
458        self.phys_base
459            .checked_add(coverage_len)
460            .ok_or(AcceptError::ArithmeticOverflow)
461    }

463    fn clamp_gpa_range_to_bitmap_coverage(
464        &self,
465        start: u64,
466        end: u64,
467    ) -> Result<Option<(u64, u64, u64)>, AcceptError> {
468        if start >= end {
469            return Ok(None);
470        }
471
472        let unit_size = self.validated_unit_size()?;
473        let coverage_end = self.max_phys_addr_exclusive(unit_size)?;
474
475        let range_start = start.max(self.phys_base);
476        let range_end = end.min(coverage_end);
477        if range_start >= range_end {
478            return Ok(None);
479        }
480
481        Ok(Some((range_start, range_end, unit_size)))
482    }

484    fn addr_to_bit_range(
485        &self,
486        start: u64,
487        end: u64,
488        unit_size: u64,
489    ) -> Result<(BitIndex, BitIndex), AcceptError> {
490        debug_assert!(start >= self.phys_base);
491        debug_assert!(start < end);
492        debug_assert!(unit_size.is_power_of_two());
493
494        let rel_start = start - self.phys_base;
495        let rel_end = end - self.phys_base;
496
497        let first_bit = rel_start / unit_size;
498        // NOTE: last_bit is exclusive and uses ceil-div for overlap semantics.
499        // Any unit intersecting [start, end) is considered.
500        let last_bit = rel_end
501            .checked_add(unit_size - 1)
502            .ok_or(AcceptError::ArithmeticOverflow)?
503            / unit_size;
504
505        Ok((BitIndex::new(first_bit), BitIndex::new(last_bit)))
506    }

508    /// Marks unit-aligned bitmap-covered GPA range `start..end` as unaccepted bits.
509    ///
510    /// # Safety
511    ///
512    /// The caller must ensure:
513    /// - `self` points to a valid unaccepted-memory table whose trailing bitmap memory is writable;
514    /// - mutable access to `self`/bitmap is unique for the duration of this call (no aliasing);
515    /// - `start..end` is unit-aligned for `unit_size` and corresponds to this table's
516    ///   bitmap coverage semantics.
517    unsafe fn mark_range_as_unaccepted(
518        &mut self,
519        start: u64,
520        end: u64,
521        unit_size: u64,
522    ) -> Result<(), AcceptError> {
523        if start >= end {
524            return Ok(());
525        }
526
527        debug_assert_eq!(start % unit_size, 0);
528        debug_assert_eq!(end % unit_size, 0);
529
530        let start_bit = BitIndex::new((start - self.phys_base) / unit_size);
531        let end_bit = BitIndex::new((end - self.phys_base) / unit_size);
532        let total_bits = BitIndex::new(self.total_bits()?);
533
534        let clamped_start_bit = BitIndex::new(start_bit.raw().min(total_bits.raw()));
535        let clamped_end_bit = BitIndex::new(end_bit.raw().min(total_bits.raw()));
536        if clamped_start_bit >= clamped_end_bit {
537            return Ok(());
538        }
539
540        // SAFETY: Caller guarantees bitmap memory is writable and uniquely accessible.
541        let mut bitmap = BitmapMut::new(unsafe { self.as_bitmap_slice_mut() });
542        for bit in clamped_start_bit.raw()..clamped_end_bit.raw() {
543            bitmap.set_bit(BitIndex::new(bit))?;
544        }
545
546        Ok(())
547    }
548
549    /// Accepts physical memory in `start..end` if the range is non-empty.
550    ///
551    /// # Safety
552    ///
553    /// The caller must ensure `start..end` is a valid GPA range for TDX acceptance,
554    /// and that accepting this range does not race with other concurrent acceptance or
555    /// access operations on the same memory.
556    unsafe fn try_accept_range(start: u64, end: u64) -> Result<(), AcceptError> {
557        if start >= end {
558            return Ok(());
559        }
560        // SAFETY: Caller guarantees the physical range is valid for TDX acceptance.
561        unsafe { accept_memory(start, end) }
562    }

565/// Mutable bitmap view used by registration/acceptance paths.
566///
567/// This helper is intentionally non-atomic; callers must provide external
568/// synchronization when multiple CPUs could touch the same bitmap.
569struct BitmapMut<'a> {
570    bits: &'a mut [u8],
571}
572
573impl<'a> BitmapMut<'a> {
574    fn new(bits: &'a mut [u8]) -> Self {
575        Self { bits }
576    }
577
578    fn capacity(&self) -> Result<u64, AcceptError> {
579        let len = u64::try_from(self.bits.len()).map_err(|_| AcceptError::OutOfBounds)?;
580        len.checked_mul(8).ok_or(AcceptError::ArithmeticOverflow)
581    }
582
583    fn get_pos_mask(&self, bit_index: BitIndex) -> Result<(usize, u8), AcceptError> {
584        if bit_index.raw() >= self.capacity()? {
585            return Err(AcceptError::OutOfBounds);
586        }
587
588        let byte_index =
589            usize::try_from(bit_index.raw() >> 3).map_err(|_| AcceptError::OutOfBounds)?;
590        let mask = 1u8 << (bit_index.raw() & 7);
591        Ok((byte_index, mask))
592    }
593
594    fn is_set(&self, bit_index: BitIndex) -> Result<bool, AcceptError> {
595        let (byte_index, mask) = self.get_pos_mask(bit_index)?;
596        Ok((self.bits[byte_index] & mask) != 0)
597    }
598
599    fn set_bit(&mut self, bit_index: BitIndex) -> Result<(), AcceptError> {
600        if self.is_set(bit_index)? {
601            return Err(AcceptError::Overlap);
602        }
603        let (byte_index, mask) = self.get_pos_mask(bit_index)?;
604        self.bits[byte_index] |= mask;
605        Ok(())
606    }
607
608    fn clear_bit(&mut self, bit_index: BitIndex) -> Result<(), AcceptError> {
609        let (byte_index, mask) = self.get_pos_mask(bit_index)?;
610        self.bits[byte_index] &= !mask;
611        Ok(())
612    }
613
614    fn clear_range(&mut self, start_bit: BitIndex, end_bit: BitIndex) -> Result<(), AcceptError> {
615        let bit_len = self.capacity()?;
616        if start_bit.raw() > end_bit.raw() || end_bit.raw() > bit_len {
617            return Err(AcceptError::OutOfBounds);
618        }
619        if start_bit == end_bit {
620            return Ok(());
621        }
622
623        let start_byte =
624            usize::try_from(start_bit.raw() >> 3).map_err(|_| AcceptError::OutOfBounds)?;
625        let end_exclusive_byte =
626            usize::try_from((end_bit.raw() + 7) >> 3).map_err(|_| AcceptError::OutOfBounds)?;
627        let start_off = u8::try_from(start_bit.raw() & 7).map_err(|_| AcceptError::OutOfBounds)?;
628        let end_off = u8::try_from(end_bit.raw() & 7).map_err(|_| AcceptError::OutOfBounds)?;
629
630        if start_byte + 1 == end_exclusive_byte {
631            // Entire range is inside one byte.
632            let end_off_eff = if end_off == 0 { 8 } else { end_off };
633            let clear_mask = bit_range_mask(start_off, end_off_eff);
634            self.bits[start_byte] &= !clear_mask;
635            return Ok(());
636        }
637
638        // Leading partial byte.
639        if start_off != 0 {
640            self.bits[start_byte] &= low_bits_mask(start_off);
641        } else {
642            self.bits[start_byte] = 0;
643        }
644
645        // Middle full bytes.
646        let middle_start = start_byte + 1;
647        let middle_end = if end_off == 0 {
648            end_exclusive_byte
649        } else {
650            end_exclusive_byte - 1
651        };
652        if middle_start < middle_end {
653            self.bits[middle_start..middle_end].fill(0);
654        }
655
656        // Trailing partial byte.
657        if end_off != 0 {
658            let keep_high = !low_bits_mask(end_off);
659            let last = end_exclusive_byte - 1;
660            self.bits[last] &= keep_high;
661        }
662
663        Ok(())
664    }
665
666    fn find_next_set(
667        &self,
668        start_bit: BitIndex,
669        end_bit: BitIndex,
670    ) -> Result<Option<BitIndex>, AcceptError> {
671        self.find_next_matching(start_bit, end_bit, true)
672    }
673
674    fn find_next_zero(
675        &self,
676        start_bit: BitIndex,
677        end_bit: BitIndex,
678    ) -> Result<Option<BitIndex>, AcceptError> {
679        self.find_next_matching(start_bit, end_bit, false)
680    }
681
682    fn find_next_matching(
683        &self,
684        start_bit: BitIndex,
685        end_bit: BitIndex,
686        target: bool,
687    ) -> Result<Option<BitIndex>, AcceptError> {
688        let bit_len = self.capacity()?;
689        if start_bit.raw() > end_bit.raw() || end_bit.raw() > bit_len {
690            return Err(AcceptError::OutOfBounds);
691        }
692
693        if start_bit == end_bit {
694            return Ok(None);
695        }
696
697        let mut scan_bit = start_bit.raw();
698        let end_bit_raw = end_bit.raw();
699
700        // Scan leading bits until the index is 64-bit aligned.
701        while scan_bit < end_bit_raw && (scan_bit & 63) != 0 {
702            if self.is_set(BitIndex::new(scan_bit))? == target {
703                return Ok(Some(BitIndex::new(scan_bit)));
704            }
705            scan_bit += 1;
706        }
707
708        // Bulk scan by 64-bit words, then use trailing_zeros for first matching bit.
709        while end_bit_raw - scan_bit >= 64 {
710            let next = scan_bit + 64;
711
712            let byte_index =
713                usize::try_from(scan_bit >> 3).map_err(|_| AcceptError::OutOfBounds)?;
714
715            // SAFETY: `next <= end_bit <= bit_len` guarantees we can read exactly 8 bytes here.
716            let word = unsafe {
717                let ptr = self.bits.as_ptr().add(byte_index).cast::<u64>();
718                u64::from_le(ptr.read_unaligned())
719            };
720
721            let match_word = if target { word } else { !word };
722            if match_word != 0 {
723                let delta = u64::from(match_word.trailing_zeros());
724                let found = scan_bit + delta;
725                return Ok(Some(BitIndex::new(found)));
726            }
727
728            scan_bit = next;
729        }
730
731        // Scan remaining tail bits (< 64).
732        while scan_bit < end_bit_raw {
733            if self.is_set(BitIndex::new(scan_bit))? == target {
734                return Ok(Some(BitIndex::new(scan_bit)));
735            }
736            scan_bit += 1;
737        }
738
739        Ok(None)
740    }
741}

743struct BitmapRef<'a> {
744    bits: &'a [u8],
745}
746
747impl<'a> BitmapRef<'a> {
748    fn new(bits: &'a [u8]) -> Self {
749        Self { bits }
750    }
751
752    fn has_set_bit(&self, start_bit: BitIndex, end_bit: BitIndex) -> Result<bool, AcceptError> {
753        if start_bit >= end_bit {
754            return Ok(false);
755        }
756
757        let bit_len = self
758            .bits
759            .len()
760            .checked_mul(8)
761            .ok_or(AcceptError::ArithmeticOverflow)?;
762        let start_bit = usize::try_from(start_bit.raw()).map_err(|_| AcceptError::OutOfBounds)?;
763        let end_bit = usize::try_from(end_bit.raw()).map_err(|_| AcceptError::OutOfBounds)?;
764        if end_bit > bit_len {
765            return Err(AcceptError::OutOfBounds);
766        }
767
768        let mut bit = start_bit;
769
770        // Scan head until byte alignment.
771        while bit < end_bit && (bit & 7) != 0 {
772            let byte_idx = bit >> 3;
773            let mask = 1u8 << (bit & 7);
774            if (self.bits[byte_idx] & mask) != 0 {
775                return Ok(true);
776            }
777            bit += 1;
778        }
779
780        let mut byte_idx = bit >> 3;
781        let end_full_byte = end_bit >> 3;
782
783        // Bulk scan by u64 for full bytes.
784        while byte_idx + 8 <= end_full_byte {
785            // SAFETY: loop condition guarantees 8 readable bytes.
786            let word = unsafe {
787                let ptr = self.bits.as_ptr().add(byte_idx).cast::<u64>();
788                u64::from_le(ptr.read_unaligned())
789            };
790            if word != 0 {
791                return Ok(true);
792            }
793            byte_idx += 8;
794        }
795
796        while byte_idx < end_full_byte {
797            if self.bits[byte_idx] != 0 {
798                return Ok(true);
799            }
800            byte_idx += 1;
801        }
802
803        // Check tail bits in one masked-byte test.
804        let tail_bits = (end_bit & 7) as u8;
805        if tail_bits != 0 {
806            let tail_mask = low_bits_mask(tail_bits);
807            if (self.bits[end_full_byte] & tail_mask) != 0 {
808                return Ok(true);
809            }
810        }
811
812        Ok(false)
813    }
814}

817struct PhysAddr(u64);
818
819impl PhysAddr {
820    const fn new(raw: u64) -> Self {
821        Self(raw)
822    }
823
824    const fn raw(self) -> u64 {
825        self.0
826    }

828    fn checked_add(self, bytes: u64) -> Result<Self, AcceptError> {
829        self.0
830            .checked_add(bytes)
831            .map(Self)
832            .ok_or(AcceptError::ArithmeticOverflow)
833    }

836        let bytes = bits

839            .ok_or(AcceptError::ArithmeticOverflow)?;
840        self.checked_add(bytes)
841    }

844/// Strongly-typed bitmap index used to avoid mixing bit position with GPA.

846struct BitIndex(u64);
847
848impl BitIndex {

851    }

854        self.0
855    }

858fn low_bits_mask(count: u8) -> u8 {
859    debug_assert!(count <= 8);
860    if count == 0 {
861        0
862    } else {
863        u8::MAX >> (8 - count)
864    }
865}
866
867fn bit_range_mask(start_off: u8, end_off: u8) -> u8 {
868    debug_assert!(start_off <= end_off && end_off <= 8);
869    if start_off == end_off {
870        return 0;
871    }
872
873    let width = end_off - start_off;
874    low_bits_mask(width) << start_off
875}

877fn align_down(addr: u64, align: u64) -> u64 {
878    addr & !(align - 1)
879}
880
881fn align_up(addr: u64, align: u64) -> Option<u64> {
882    addr.checked_add(align - 1).map(|v| v & !(align - 1))
883}
