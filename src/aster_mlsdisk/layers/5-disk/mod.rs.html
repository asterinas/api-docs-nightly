3//! The layer of secure virtual disk.
4//!
5//! `MlsDisk` provides three block I/O interfaces, `read()`, `write()` and `sync()`.
6//! `MlsDisk` protects a logical block of user data using authenticated encryption.
7//! The metadata of the encrypted logical blocks are inserted into a secure index `TxLsmTree`.
8//!
9//! `MlsDisk`'s backed untrusted host disk space is managed in `BlockAlloc`. Block reclamation can be
10//! delayed to user-defined callbacks on `TxLsmTree`.
11//! `MlsDisk` supports buffering written logical blocks.
12//!
13//! # Usage Example
14//!
15//! Write, sync then read blocks from `MlsDisk`.
16//!
17//! ```
18//! let nblocks = 1024;
19//! let mem_disk = MemDisk::create(nblocks)?;
20//! let root_key = Key::random();
21//! let mlsdisk = MlsDisk::create(mem_disk.clone(), root_key)?;
22//!
23//! let num_rw = 128;
24//! let mut rw_buf = Buf::alloc(1)?;
25//! for i in 0..num_rw {
26//!     rw_buf.as_mut_slice().fill(i as u8);
27//!     mlsdisk.write(i as Lba, rw_buf.as_ref())?;
28//! }
29//! mlsdisk.sync()?;
30//! for i in 0..num_rw {
31//!     mlsdisk.read(i as Lba, rw_buf.as_mut())?;
32//!     assert_eq!(rw_buf.as_slice()[0], i as u8);
33//! }
34//! ```
