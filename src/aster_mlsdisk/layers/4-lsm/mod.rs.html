3//! The layer of transactional Lsm-Tree.
4//!
5//! This module provides the implementation for `TxLsmTree`.
6//! `TxLsmTree` is similar to general-purpose LSM-Tree, supporting `put()`, `get()`, `get_range()`
7//! key-value records, which are managed in MemTables and SSTables.
8//!
9//! `TxLsmTree` is transactional in the sense that
10//! 1) it supports `sync()` that guarantees changes are persisted atomically and irreversibly,
11//!    synchronized records and unsynchronized records can co-existed.
12//! 2) its internal data is securely stored in `TxLogStore` (L3) and updated in transactions for consistency,
13//!    WALs and SSTables are stored and managed in `TxLogStore`.
14//!
15//! `TxLsmTree` supports piggybacking callbacks during compaction and recovery.
16//!
17//! # Usage Example
18//!
19//! Create a `TxLsmTree` then put some records into it.
20//!
21//! ```
22//! // Prepare an underlying disk (implement `BlockSet`) first
23//! let nblocks = 1024;
24//! let mem_disk = MemDisk::create(nblocks)?;
25//!
26//! // Prepare an underlying `TxLogStore` (L3) for storing WALs and SSTs
27//! let tx_log_store = Arc::new(TxLogStore::format(mem_disk)?);
28//!
29//! // Create a `TxLsmTree` with the created `TxLogStore`
30//! let tx_lsm_tree: TxLsmTree<BlockId, String, MemDisk> =
31//!     TxLsmTree::format(tx_log_store, Arc::new(YourFactory), None)?;
32//!
33//! // Put some key-value records into the tree
34//! for i in 0..10 {
35//!     let k = i as BlockId;
36//!     let v = i.to_string();
37//!     tx_lsm_tree.put(k, v)?;
38//! }
39//!
40//! // Issue a sync operation to the tree to ensure persistency
41//! tx_lsm_tree.sync()?;
42//!
43//! // Use `get()` (or `get_range()`) to query the tree
44//! let target_value = tx_lsm_tree.get(&5).unwrap();
45//! // Check the previously put value
46//! assert_eq(target_value, "5");
47//!
48//! // `TxLsmTree` supports user-defined per-TX callbacks
49//! struct YourFactory;
50//! struct YourListener;
51//!
52//! impl<K, V> TxEventListenerFactory<K, V> for YourFactory {
53//!     // Support create per-TX (upon compaction or upon recovery) listener
54//!     fn new_event_listener(&self, tx_type: TxType) -> Arc<dyn TxEventListener<K, V>> {
55//!         Arc::new(YourListener::new(tx_type))
56//!     }
57//! }
58//!
59//! // Support defining callbacks when record is added or drop, or
60//! // at some critical points during a TX
61//! impl<K, V> TxEventListener<K, V> for YourListener {
62//!     /* details omitted, see the API for more */
63//! }
64//! ```
