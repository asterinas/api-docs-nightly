10/// A log of data blocks that can support random reads and append-only
11/// writes.
12///
13/// # Thread safety
14///
15/// `BlockLog` is a data structure of interior mutability.
16/// It is ok to perform I/O on a `BlockLog` concurrently in multiple threads.
17/// `BlockLog` promises the serialization of the append operations, i.e.,
18/// concurrent appends are carried out as if they are done one by one.
19pub trait BlockLog: Sync + Send {
20    /// Read one or multiple blocks at a specified position.
21    fn read(&self, pos: BlockId, buf: BufMut) -> Result<()>;
22
23    /// Append one or multiple blocks at the end,
24    /// returning the ID of the first newly-appended block.
25    fn append(&self, buf: BufRef) -> Result<BlockId>;
26
27    /// Ensure that blocks are persisted to the disk.
28    fn flush(&self) -> Result<()>;
29
30    /// Returns the number of blocks.
31    fn nblocks(&self) -> usize;
32}
33
34macro_rules! impl_blocklog_for {
35    ($typ:ty,$from:tt) => {
36        #[inherit_methods(from = $from)]
37        impl<T: BlockLog> BlockLog for $typ {
38            fn read(&self, pos: BlockId, buf: BufMut) -> Result<()>;
39            fn append(&self, buf: BufRef) -> Result<BlockId>;
40            fn flush(&self) -> Result<()>;
41            fn nblocks(&self) -> usize;
42        }
43    };
44}
45
46impl_blocklog_for!(&T, "(**self)");
47impl_blocklog_for!(&mut T, "(**self)");
48impl_blocklog_for!(Box<T>, "(**self)");
49impl_blocklog_for!(Arc<T>, "(**self)");

51/// An in-memory log that impls `BlockLog`.
52pub struct MemLog {
53    log: Mutex<Buf>,
54    append_pos: AtomicUsize,
55}
56
57impl BlockLog for MemLog {
58    fn read(&self, pos: BlockId, mut buf: BufMut) -> Result<()> {
59        let nblocks = buf.nblocks();
60        if pos + nblocks > self.nblocks() {
61            return_errno_with_msg!(InvalidArgs, "read range out of bound");
62        }
63        let log = self.log.lock();
64        let read_buf = &log.as_slice()[Self::offset(pos)..Self::offset(pos) + nblocks * BLOCK_SIZE];
65        buf.as_mut_slice().copy_from_slice(read_buf);
66        Ok(())
67    }
68
69    fn append(&self, buf: BufRef) -> Result<BlockId> {
70        let nblocks = buf.nblocks();
71        let mut log = self.log.lock();
72        let pos = self.append_pos.load(Ordering::Acquire);
73        if pos + nblocks > log.nblocks() {
74            return_errno_with_msg!(InvalidArgs, "append range out of bound");
75        }
76        let write_buf =
77            &mut log.as_mut_slice()[Self::offset(pos)..Self::offset(pos) + nblocks * BLOCK_SIZE];
78        write_buf.copy_from_slice(buf.as_slice());
79        self.append_pos.fetch_add(nblocks, Ordering::Release);
80        Ok(pos)
81    }
82
83    fn flush(&self) -> Result<()> {
84        Ok(())
85    }
86
87    fn nblocks(&self) -> usize {
88        self.append_pos.load(Ordering::Acquire)
89    }
90}
91
92impl MemLog {
93    pub fn create(num_blocks: usize) -> Result<Self> {
94        Ok(Self {
95            log: Mutex::new(Buf::alloc(num_blocks)?),
96            append_pos: AtomicUsize::new(0),
97        })
98    }
99
100    fn offset(pos: BlockId) -> usize {
101        pos * BLOCK_SIZE
102    }
103}

105#[cfg(test)]
106mod tests {
107    use super::*;
108
109    #[test]
110    fn mem_log() -> Result<()> {
111        let total_blocks = 64;
112        let append_nblocks = 8;
113        let mem_log = MemLog::create(total_blocks)?;
114        assert_eq!(mem_log.nblocks(), 0);
115
116        let mut append_buf = Buf::alloc(append_nblocks)?;
117        let content = 5_u8;
118        append_buf.as_mut_slice().fill(content);
119        let append_pos = mem_log.append(append_buf.as_ref())?;
120        assert_eq!(append_pos, 0);
121        assert_eq!(mem_log.nblocks(), append_nblocks);
122
123        mem_log.flush()?;
124        let mut read_buf = Buf::alloc(1)?;
125        let read_pos = 7 as BlockId;
126        mem_log.read(read_pos, read_buf.as_mut())?;
127        assert_eq!(
128            read_buf.as_slice(),
129            &append_buf.as_slice()[read_pos * BLOCK_SIZE..(read_pos + 1) * BLOCK_SIZE]
130        );
131        Ok(())
132    }
