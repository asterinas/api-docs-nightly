16/// A cryptographically-protected log of user data blocks.
17///
18/// `CryptoLog<L>`, which is backed by an untrusted block log (`L`),
19/// serves as a secure log file that supports random reads and append-only
20/// writes of data blocks. `CryptoLog<L>` encrypts the data blocks and
21/// protects them with a Merkle Hash Tree (MHT), which itself is also encrypted.
22///
23/// # Security
24///
25/// Each instance of `CryptoLog<L>` is assigned a randomly-generated root key
26/// upon its creation. The root key is used to encrypt the root MHT block only.
27/// Each new version of the root MHT block is encrypted with the same key, but
28/// different random IVs. This arrangement ensures the confidentiality of
29/// the root block.
30///
31/// After flushing a `CryptoLog<L>`, a new root MHT (as well as other MHT nodes)
32/// shall be appended to the backend block log (`L`).
33/// The metadata of the root MHT, including its position, encryption
34/// key, IV, and MAC, must be kept by the user of `CryptoLog<L>` so that
35/// he or she can use the metadata to re-open the `CryptoLog`.
36/// The information contained in the metadata is sufficient to verify the
37/// integrity and freshness of the root MHT node, and thus the whole `CryptoLog`.
38///
39/// Other MHT nodes as well as data nodes are encrypted with randomly-generated,
40/// unique keys. Their metadata, including its position, encryption key, IV, and
41/// MAC, are kept securely in their parent MHT nodes, which are also encrypted.
42/// Thus, the confidentiality and integrity of non-root nodes are protected.
43///
44/// # Performance
45///
46/// Thanks to its append-only nature, `CryptoLog<L>` avoids MHT's high
47/// performance overheads under the workload of random writes
48/// due to "cascades of updates".
49///
50/// Behind the scene, `CryptoLog<L>` keeps a cache for nodes so that frequently
51/// or lately accessed nodes can be found in the cache, avoiding the I/O
52/// and decryption cost incurred when re-reading these nodes.
53/// The cache is also used for buffering new data so that multiple writes to
54/// individual nodes can be merged into a large write to the underlying block log.
55/// Therefore, `CryptoLog<L>` is efficient for both reads and writes.
56///
57/// # Disk space
58///
59/// One consequence of using an append-only block log (`L`) as the backend is
60/// that `CryptoLog<L>` cannot do in-place updates to existing MHT nodes.
61/// This means the new version of MHT nodes are appended to the underlying block
62/// log and the invalid blocks occupied by old versions are not reclaimed.
63///
64/// But lucky for us, this block reclamation problem is not an issue in practice.
65/// This is because a `CryptoLog<L>` is created for one of the following two
66/// use cases.
67///
68/// 1. Write-once-then-read-many. In this use case, all the content of a
69///    `CryptoLog` is written in a single run.
70///    Writing in a single run won't trigger any updates to MHT nodes and thus
71///    no waste of disk space.
72///    After the writing is done, the `CryptoLog` becomes read-only.
73///
74/// 2. Write-many-then-read-once. In this use case, the content of a
75///    `CryptoLog` may be written in many runs. But the number of `CryptoLog`
76///    under such workloads is limited and their lengths are also limited.
77///    So the disk space wasted by such `CryptoLog` is bounded.
78///    And after such `CryptoLog`s are done writing, they will be read once and
79///    then discarded.
80pub struct CryptoLog<L> {
81    mht: RwLock<Mht<L>>,
82}

111/// The Merkle-Hash Tree (MHT) node (internal).
112/// It contains a header for node metadata and a bunch of entries for managing children nodes.
113#[repr(C)]
114#[derive(Clone, Copy, Pod)]
115struct MhtNode {
116    header: MhtNodeHeader,
117    entries: [MhtNodeEntry; MHT_NBRANCHES],
118}
119const_assert!(size_of::<MhtNode>() <= BLOCK_SIZE);
120
121/// The header contains metadata of the current MHT node.
122#[repr(C)]
123#[derive(Clone, Copy, Debug, Pod)]
124struct MhtNodeHeader {
125    // The height of the MHT whose root is this node
126    height: Height,
127    // The total number of valid data nodes covered by this node
128    num_data_nodes: u32,
129    // The number of valid entries (children) within this node
130    num_valid_entries: u16,
131}
132
133/// The entry of the MHT node, which contains the
134/// metadata of the child MHT/data node.
135#[repr(C)]
136#[derive(Clone, Copy, Debug, Pod)]
137struct MhtNodeEntry {
138    pos: Pbid,
139    key: Key,
140    mac: Mac,
141}
142
143// Number of branches of one MHT node. (102 for now)
144const MHT_NBRANCHES: usize = (BLOCK_SIZE - size_of::<MhtNodeHeader>()) / size_of::<MhtNodeEntry>();
145
146/// The data node (leaf). It contains a block of data.
147#[repr(C)]
148#[derive(Clone, Copy, Pod)]
149struct DataNode([u8; BLOCK_SIZE]);

204impl<L: BlockLog> CryptoLog<L> {
205    /// Creates a new `CryptoLog`.
206    ///
207    /// A newly-created instance won't occupy any space on the `block_log`
208    /// until the first flush, which triggers writing the root MHT node.
209    pub fn new(block_log: L, root_key: Key, node_cache: Arc<dyn NodeCache>) -> Self {
210        Self {
211            mht: RwLock::new(Mht::new(block_log, root_key, node_cache)),
212        }
213    }
214
215    /// Opens an existing `CryptoLog` backed by a `block_log`.
216    ///
217    /// The given key and the metadata of the root MHT are sufficient to
218    /// load and verify the root node of the `CryptoLog`.
219    pub fn open(
220        block_log: L,
221        root_key: Key,
222        root_meta: RootMhtMeta,
223        node_cache: Arc<dyn NodeCache>,
224    ) -> Result<Self> {
225        Ok(Self {
226            mht: RwLock::new(Mht::open(block_log, root_key, root_meta, node_cache)?),
227        })
228    }
229
230    /// Gets the root key.
231    pub fn root_key(&self) -> Key {
232        self.mht.read().root_key
233    }
234
235    /// Gets the metadata of the root MHT node.
236    ///
237    /// Returns `None` if there hasn't been any appends or flush.
238    pub fn root_meta(&self) -> Option<RootMhtMeta> {
239        self.mht.read().root_meta()
240    }
241
242    fn root_node(&self) -> Option<Arc<MhtNode>> {
243        self.mht.read().root_node().cloned()
244    }
245
246    /// Gets the number of data nodes (blocks).
247    pub fn nblocks(&self) -> usize {
248        self.mht.read().total_data_nodes()
249    }
250
251    /// Reads one or multiple data blocks at a specified position.
252    pub fn read(&self, pos: Lbid, buf: BufMut) -> Result<()> {
253        let mut search_ctx = SearchCtx::new(pos, buf);
254        self.mht.read().search(&mut search_ctx)?;
255
256        debug_assert!(search_ctx.is_completed);
257        Ok(())
258    }
259
260    /// Appends one or multiple data blocks at the end.
261    pub fn append(&self, buf: BufRef) -> Result<()> {
262        let data_nodes: Vec<Arc<DataNode>> = buf
263            .iter()
264            .map(|block_buf| {
265                let mut node = DataNode::new_uninit();
266                node.0.copy_from_slice(block_buf.as_slice());
267                Arc::new(node)
268            })
269            .collect();
270
271        self.mht.write().append_data_nodes(data_nodes)
272    }
273
274    /// Ensures that all new data are persisted.
275    ///
276    /// Each successful flush triggers writing a new version of the root MHT
277    /// node to the underlying block log. The metadata of the latest root MHT
278    /// can be obtained via the `root_meta` method.
279    pub fn flush(&self) -> Result<()> {
280        self.mht.write().flush()
281    }
282
283    pub fn display_mht(&self) {
284        self.mht.read().display();
285    }
286}

288impl<L: BlockLog> Mht<L> {
289    // Buffer capacity for appended data nodes.
290    const APPEND_BUF_CAPACITY: usize = 2048;
291
292    pub fn new(block_log: L, root_key: Key, node_cache: Arc<dyn NodeCache>) -> Self {
293        let storage = Arc::new(MhtStorage::new(block_log, node_cache));
294        let start_pos = 0 as Lbid;
295        Self {
296            root: None,
297            root_key,
298            data_buf: AppendDataBuf::new(Self::APPEND_BUF_CAPACITY, start_pos, storage.clone()),
299            storage,
300        }
301    }
302
303    pub fn open(
304        block_log: L,
305        root_key: Key,
306        root_meta: RootMhtMeta,
307        node_cache: Arc<dyn NodeCache>,
308    ) -> Result<Self> {
309        let storage = Arc::new(MhtStorage::new(block_log, node_cache));
310        let root_node = storage.root_mht_node(&root_key, &root_meta)?;
311        let start_pos = root_node.num_data_nodes() as Lbid;
312        Ok(Self {
313            root: Some((root_meta, root_node)),
314            root_key,
315            data_buf: AppendDataBuf::new(Self::APPEND_BUF_CAPACITY, start_pos, storage.clone()),
316            storage,
317        })
318    }
319
320    pub fn root_meta(&self) -> Option<RootMhtMeta> {
321        self.root.as_ref().map(|(root_meta, _)| root_meta.clone())
322    }
323
324    fn root_node(&self) -> Option<&Arc<MhtNode>> {
325        self.root.as_ref().map(|(_, root_node)| root_node)
326    }
327
328    pub fn total_data_nodes(&self) -> usize {
329        self.data_buf.num_append()
330            + self
331                .root
332                .as_ref()
333                .map_or(0, |(_, root_node)| root_node.num_data_nodes())
334    }
335
336    pub fn search(&self, search_ctx: &mut SearchCtx<'_>) -> Result<()> {
337        let root_node = self
338            .root_node()
339            .ok_or(Error::with_msg(NotFound, "root MHT node not found"))?;
340
341        if search_ctx.pos + search_ctx.num > self.total_data_nodes() {
342            return_errno_with_msg!(InvalidArgs, "search out of MHT capacity");
343        }
344
345        // Search the append data buffer first
346        self.data_buf.search_data_nodes(search_ctx)?;
347        if search_ctx.is_completed {
348            return Ok(());
349        }
350
351        // Search the MHT if needed
352        self.search_hierarchy(vec![root_node.clone()], root_node.height(), search_ctx)
353    }
354
355    fn search_hierarchy(
356        &self,
357        level_targets: Vec<Arc<MhtNode>>,
358        mut curr_height: Height,
359        search_ctx: &mut SearchCtx<'_>,
360    ) -> Result<()> {
361        debug_assert!(
362            !level_targets.is_empty() && curr_height == level_targets.first().unwrap().height()
363        );
364        let num_data_nodes = search_ctx.num;
365
366        // Calculate two essential values for searching the current level:
367        // how many nodes to skip and how many nodes we need
368        let (nodes_skipped, nodes_needed) = {
369            let pos = &mut search_ctx.pos;
370            let next_level_max_num_data_nodes = MhtNode::max_num_data_nodes(curr_height - 1);
371            let skipped = *pos / next_level_max_num_data_nodes;
372            *pos -= skipped * next_level_max_num_data_nodes;
373            let needed = align_up(num_data_nodes + *pos, next_level_max_num_data_nodes)
374                / next_level_max_num_data_nodes;
375            (skipped, needed)
376        };
377
378        let target_entries = level_targets
379            .iter()
380            .flat_map(|node| node.entries.iter())
381            .skip(nodes_skipped)
382            .take(nodes_needed);
383
384        // Search down to the leaves, ready to collect data nodes
385        if MhtNode::is_lowest_level(curr_height) {
386            debug_assert_eq!(num_data_nodes, nodes_needed);
387            for entry in target_entries {
388                self.storage
389                    .read_data_node(entry, search_ctx.node_buf(search_ctx.offset))?;
390                search_ctx.offset += 1;
391            }
392            search_ctx.is_completed = true;
393            return Ok(());
394        }
395
396        // Prepare target MHT nodes for the lower level
397        let next_level_targets = {
398            let mut targets = Vec::with_capacity(nodes_needed);
399            for entry in target_entries {
400                let target_node = self.storage.read_mht_node(
401                    entry.pos,
402                    &entry.key,
403                    &entry.mac,
404                    &Iv::new_zeroed(),
405                )?;
406                targets.push(target_node);
407            }
408            targets
409        };
410
411        // Search the lower level
412        curr_height -= 1;
413        self.search_hierarchy(next_level_targets, curr_height, search_ctx)
414    }
415
416    pub fn append_data_nodes(&mut self, data_nodes: Vec<Arc<DataNode>>) -> Result<()> {
417        self.data_buf.append_data_nodes(data_nodes)?;
418        if self.data_buf.is_full() {
419            let data_node_entries = self.data_buf.flush()?;
420            self.do_build(data_node_entries)?;
421        }
422        Ok(())
423    }
424
425    pub fn flush(&mut self) -> Result<()> {
426        let data_node_entries = self.data_buf.flush()?;
427        self.do_build(data_node_entries)?;
428        // FIXME: Should we sync the storage here?
429        // self.storage.flush()?;
430        Ok(())
431    }
432
433    fn do_build(&mut self, data_node_entries: Vec<MhtNodeEntry>) -> Result<()> {
434        let new_root_node = {
435            TreeBuilder::new(&self.storage)
436                .previous_built_root(self.root.as_ref().map(|(_, root_node)| root_node))
437                .build(data_node_entries)?
438        };
439        let root_meta = self
440            .storage
441            .append_root_mht_node(&self.root_key, &new_root_node)?;
442        let _ = self.root.insert((root_meta, new_root_node));
443        Ok(())
444    }
445
446    pub fn display(&self) {
447        info!("{:?}", MhtDisplayer(self));
448    }
449}

451impl<L: BlockLog> MhtStorage<L> {
452    pub fn new(block_log: L, node_cache: Arc<dyn NodeCache>) -> Self {
453        Self {
454            block_log,
455            node_cache,
456            crypt_buf: Mutex::new(CryptBuf::new()),
457        }
458    }
459
460    pub fn flush(&self) -> Result<()> {
461        self.block_log.flush()
462    }
463
464    pub fn root_mht_node(&self, root_key: &Key, root_meta: &RootMhtMeta) -> Result<Arc<MhtNode>> {
465        self.read_mht_node(root_meta.pos, root_key, &root_meta.mac, &root_meta.iv)
466    }
467
468    pub fn append_root_mht_node(&self, root_key: &Key, node: &Arc<MhtNode>) -> Result<RootMhtMeta> {
469        let mut crypt_buf = self.crypt_buf.lock();
470        let iv = Iv::random();
471        let mac = Aead::new().encrypt(
472            node.as_bytes(),
473            root_key,
474            &iv,
475            &[],
476            crypt_buf.cipher.as_mut_slice(),
477        )?;
478
479        let pos = self.block_log.append(crypt_buf.cipher.as_ref())?;
480        self.node_cache.put(pos, node.clone());
481        Ok(RootMhtMeta { pos, mac, iv })
482    }
483
484    fn append_mht_nodes(&self, nodes: &[Arc<MhtNode>]) -> Result<Vec<MhtNodeEntry>> {
485        let num_append = nodes.len();
486        let mut node_entries = Vec::with_capacity(num_append);
487        let mut cipher_buf = Buf::alloc(num_append)?;
488        let mut pos = self.block_log.nblocks() as BlockId;
489        let start_pos = pos;
490        for (i, node) in nodes.iter().enumerate() {
491            let plain = node.as_bytes();
492            let cipher = &mut cipher_buf.as_mut_slice()[i * BLOCK_SIZE..(i + 1) * BLOCK_SIZE];
493            let key = Key::random();
494            let mac = Aead::new().encrypt(plain, &key, &Iv::new_zeroed(), &[], cipher)?;
495
496            node_entries.push(MhtNodeEntry { pos, key, mac });
497            self.node_cache.put(pos, node.clone());
498            pos += 1;
499        }
500
501        let append_pos = self.block_log.append(cipher_buf.as_ref())?;
502        debug_assert_eq!(start_pos, append_pos);
503        Ok(node_entries)
504    }
505
506    fn append_data_nodes(&self, nodes: &[Arc<DataNode>]) -> Result<Vec<MhtNodeEntry>> {
507        let num_append = nodes.len();
508        let mut node_entries = Vec::with_capacity(num_append);
509        if num_append == 0 {
510            return Ok(node_entries);
511        }
512
513        let mut cipher_buf = Buf::alloc(num_append)?;
514        let mut pos = self.block_log.nblocks() as BlockId;
515        let start_pos = pos;
516        for (i, node) in nodes.iter().enumerate() {
517            let cipher = &mut cipher_buf.as_mut_slice()[i * BLOCK_SIZE..(i + 1) * BLOCK_SIZE];
518            let key = Key::random();
519            let mac = Aead::new().encrypt(&node.0, &key, &Iv::new_zeroed(), &[], cipher)?;
520
521            node_entries.push(MhtNodeEntry { pos, key, mac });
522            pos += 1;
523        }
524
525        let append_pos = self.block_log.append(cipher_buf.as_ref())?;
526        debug_assert_eq!(start_pos, append_pos);
527        Ok(node_entries)
528    }
529
530    fn read_mht_node(&self, pos: Pbid, key: &Key, mac: &Mac, iv: &Iv) -> Result<Arc<MhtNode>> {
531        if let Some(node) = self.node_cache.get(pos) {
532            return node.downcast::<MhtNode>().map_err(|_| {
533                Error::with_msg(InvalidArgs, "cache node downcasts to MHT node failed")
534            });
535        }
536
537        let mht_node = {
538            let mut crypt_buf = self.crypt_buf.lock();
539            self.block_log.read(pos, crypt_buf.cipher.as_mut())?;
540            let mut node = MhtNode::new_zeroed();
541            Aead::new().decrypt(
542                crypt_buf.cipher.as_slice(),
543                key,
544                iv,
545                &[],
546                mac,
547                node.as_bytes_mut(),
548            )?;
549            crypt_buf
550                .plain
551                .as_mut_slice()
552                .copy_from_slice(node.as_bytes());
553            Arc::new(node)
554        };
555
556        self.node_cache.put(pos, mht_node.clone());
557        Ok(mht_node)
558    }
559
560    fn read_data_node(&self, entry: &MhtNodeEntry, node_buf: &mut [u8]) -> Result<()> {
561        debug_assert_eq!(node_buf.len(), BLOCK_SIZE);
562        let mut crypt_buf = self.crypt_buf.lock();
563
564        self.block_log.read(entry.pos, crypt_buf.cipher.as_mut())?;
565        Aead::new().decrypt(
566            crypt_buf.cipher.as_slice(),
567            &entry.key,
568            &Iv::new_zeroed(),
569            &[],
570            &entry.mac,
571            node_buf,
572        )
573    }
574}

576impl MhtNode {
577    pub fn height(&self) -> Height {
578        self.header.height
579    }
580
581    pub fn num_data_nodes(&self) -> usize {
582        self.header.num_data_nodes as _
583    }
584
585    pub fn num_valid_entries(&self) -> usize {
586        self.header.num_valid_entries as _
587    }
588
589    // Lowest level MHT node's children are data nodes
590    pub fn is_lowest_level(height: Height) -> bool {
591        height == 1
592    }
593
594    pub fn max_num_data_nodes(height: Height) -> usize {
595        // Also correct when height equals 0
596        MHT_NBRANCHES.pow(height as _)
597    }
598
599    // A complete node indicates that all children are valid and
600    // all covered with maximum number of data nodes
601    pub fn is_incomplete(&self) -> bool {
602        self.num_data_nodes() != Self::max_num_data_nodes(self.height())
603    }
604
605    pub fn num_complete_children(&self) -> usize {
606        if self.num_data_nodes().is_multiple_of(MHT_NBRANCHES)
607            || Self::is_lowest_level(self.height())
608        {
609            self.num_valid_entries()
610        } else {
611            self.num_valid_entries() - 1
612        }
613    }
614}

616impl<'a, L: BlockLog> TreeBuilder<'a, L> {
617    pub fn new(storage: &'a MhtStorage<L>) -> Self {
618        Self {
619            previous_build: None,
620            storage,
621        }
622    }
623
624    pub fn previous_built_root(mut self, previous_built_root: Option<&Arc<MhtNode>>) -> Self {
625        if previous_built_root.is_none() {
626            return self;
627        }
628        self.previous_build = Some(PreviousBuild::new(
629            previous_built_root.unwrap(),
630            self.storage,
631        ));
632        self
633    }
634
635    pub fn build(&self, data_node_entries: Vec<MhtNodeEntry>) -> Result<Arc<MhtNode>> {
636        let total_data_nodes = data_node_entries.len()
637            + self
638                .previous_build
639                .as_ref()
640                .map_or(0, |pre| pre.num_data_nodes());
641
642        self.build_hierarchy(
643            data_node_entries,
644            total_data_nodes,
645            1 as Height,
646            self.calc_target_height(total_data_nodes),
647        )
648    }
649
650    fn build_hierarchy(
651        &self,
652        level_entries: Vec<MhtNodeEntry>,
653        total_data_nodes: usize,
654        mut curr_height: Height,
655        target_height: Height,
656    ) -> Result<Arc<MhtNode>> {
657        // Build the MHT nodes of current level
658        let mut new_mht_nodes = {
659            // Previous built incomplete node of same level should participate in the building
660            let previous_incomplete_node = self
661                .previous_build
662                .as_ref()
663                .and_then(|pre| pre.target_node(curr_height));
664
665            LevelBuilder::new(level_entries, total_data_nodes, curr_height)
666                .previous_incomplete_node(previous_incomplete_node)
667                .build()
668        };
669
670        if curr_height == target_height {
671            // The root MHT node has been built
672            debug_assert_eq!(new_mht_nodes.len(), 1);
673            return Ok(new_mht_nodes.pop().unwrap());
674        }
675
676        // Prepare MHT node entries for the higher level
677        let next_level_entries = self.storage.append_mht_nodes(&new_mht_nodes)?;
678        // Build the higher level
679        curr_height += 1;
680        self.build_hierarchy(
681            next_level_entries,
682            total_data_nodes,
683            curr_height,
684            target_height,
685        )
686    }
687
688    fn calc_target_height(&self, num_data_node_entries: usize) -> Height {
689        let target_height = num_data_node_entries.ilog(MHT_NBRANCHES);
690        if MHT_NBRANCHES.pow(target_height) < num_data_node_entries || target_height == 0 {
691            (target_height + 1) as Height
692        } else {
693            target_height as Height
694        }
695    }
696}

698impl LevelBuilder {
699    pub fn new(level_entries: Vec<MhtNodeEntry>, total_data_nodes: usize, height: Height) -> Self {
700        Self {
701            level_entries,
702            total_data_nodes,
703            height,
704            previous_incomplete_node: None,
705        }
706    }
707
708    pub fn previous_incomplete_node(
709        mut self,
710        previous_incomplete_node: Option<Arc<MhtNode>>,
711    ) -> Self {
712        self.previous_incomplete_node = previous_incomplete_node;
713        self
714    }
715
716    pub fn build(&self) -> Vec<Arc<MhtNode>> {
717        let all_level_entries: Vec<&MhtNodeEntry> =
718            if let Some(pre_node) = self.previous_incomplete_node.as_ref() {
719                // If there exists a previous built node (same height),
720                // its complete entries should participate in the building
721                pre_node
722                    .entries
723                    .iter()
724                    .take(pre_node.num_complete_children())
725                    .chain(self.level_entries.iter())
726                    .collect()
727            } else {
728                self.level_entries.iter().collect()
729            };
730
731        let num_build = align_up(all_level_entries.len(), MHT_NBRANCHES) / MHT_NBRANCHES;
732        let mut new_mht_nodes = Vec::with_capacity(num_build);
733        // Each iteration builds a new MHT node
734        for (nth, entries_per_node) in all_level_entries.chunks(MHT_NBRANCHES).enumerate() {
735            if nth == num_build - 1 {
736                let last_new_node = self.build_last_node(entries_per_node);
737                new_mht_nodes.push(last_new_node);
738                break;
739            }
740
741            let mut mht_node = MhtNode::new_zeroed();
742            mht_node.header = MhtNodeHeader {
743                height: self.height,
744                num_data_nodes: MhtNode::max_num_data_nodes(self.height) as _,
745                num_valid_entries: MHT_NBRANCHES as _,
746            };
747            for (i, entry) in mht_node.entries.iter_mut().enumerate() {
748                *entry = *entries_per_node[i];
749            }
750
751            new_mht_nodes.push(Arc::new(mht_node));
752        }
753        new_mht_nodes
754    }
755
756    // Building last MHT node of the level can be complicated, since
757    // the last node may be incomplete
758    fn build_last_node(&self, entries: &[&MhtNodeEntry]) -> Arc<MhtNode> {
759        let num_data_nodes = {
760            let max_data_nodes = MhtNode::max_num_data_nodes(self.height);
761            let incomplete_nodes = self.total_data_nodes % max_data_nodes;
762            if incomplete_nodes == 0 {
763                max_data_nodes
764            } else {
765                incomplete_nodes
766            }
767        };
768        let num_valid_entries = entries.len();
769
770        let mut last_mht_node = MhtNode::new_zeroed();
771        last_mht_node.header = MhtNodeHeader {
772            height: self.height,
773            num_data_nodes: num_data_nodes as _,
774            num_valid_entries: num_valid_entries as _,
775        };
776        for (i, entry) in last_mht_node.entries.iter_mut().enumerate() {
777            *entry = if i < num_valid_entries {
778                *entries[i]
779            } else {
780                // Padding invalid entries to the rest
781                MhtNodeEntry::new_uninit()
782            };
783        }
784
785        Arc::new(last_mht_node)
786    }
787}

789impl<'a, L: BlockLog> PreviousBuild<'a, L> {
790    pub fn new(previous_built_root: &Arc<MhtNode>, storage: &'a MhtStorage<L>) -> Self {
791        let mut new_self = Self {
792            root: previous_built_root.clone(),
793            height: previous_built_root.height(),
794            internal_incomplete_nodes: HashMap::new(),
795            storage,
796        };
797        new_self.collect_incomplete_nodes();
798        new_self
799    }
800
801    pub fn target_node(&self, target_height: Height) -> Option<Arc<MhtNode>> {
802        if target_height == self.height {
803            return Some(self.root.clone());
804        }
805        self.internal_incomplete_nodes.get(&target_height).cloned()
806    }
807
808    pub fn num_data_nodes(&self) -> usize {
809        self.root.num_data_nodes()
810    }
811
812    fn collect_incomplete_nodes(&mut self) {
813        let root_node = &self.root;
814        if !root_node.is_incomplete() || MhtNode::is_lowest_level(self.height) {
815            return;
816        }
817
818        let mut lookup_node = {
819            let entry = root_node.entries[root_node.num_valid_entries() - 1];
820            self.storage
821                .read_mht_node(entry.pos, &entry.key, &entry.mac, &Iv::new_zeroed())
822                .unwrap()
823        };
824
825        while lookup_node.is_incomplete() {
826            let height = lookup_node.height();
827            self.internal_incomplete_nodes
828                .insert(height, lookup_node.clone());
829
830            if MhtNode::is_lowest_level(height) {
831                break;
832            }
833
834            // Incomplete nodes only appear in the last node of each level
835            lookup_node = {
836                let entry = lookup_node.entries[lookup_node.num_valid_entries() - 1];
837                self.storage
838                    .read_mht_node(entry.pos, &entry.key, &entry.mac, &Iv::new_zeroed())
839                    .unwrap()
840            }
841        }
842    }
843}

871/// A buffer that contains appended data.
872struct AppendDataBuf<L> {
873    node_queue: Vec<Arc<DataNode>>,
874    node_queue_cap: usize,
875    entry_queue: Vec<MhtNodeEntry>, // Also cache the data node entries
876    entry_queue_cap: usize,
877    start_pos: Lbid,
878    storage: Arc<MhtStorage<L>>,
879}
880
881impl<L: BlockLog> AppendDataBuf<L> {
882    // Maximum capacity of entries indicates a complete MHT (height equals 3)
883    const MAX_ENTRY_QUEUE_CAP: usize = MHT_NBRANCHES.pow(3);
884
885    pub fn new(capacity: usize, start_pos: Lbid, storage: Arc<MhtStorage<L>>) -> Self {
886        let (node_queue_cap, entry_queue_cap) = Self::calc_queue_cap(capacity, start_pos);
887        Self {
888            node_queue: Vec::with_capacity(node_queue_cap),
889            node_queue_cap,
890            entry_queue: Vec::with_capacity(entry_queue_cap),
891            start_pos,
892            entry_queue_cap,
893            storage,
894        }
895    }
896
897    pub fn num_append(&self) -> usize {
898        self.node_queue.len() + self.entry_queue.len()
899    }
900
901    pub fn is_full(&self) -> bool {
902        // Returns whether the data node entry queue is at capacity
903        self.entry_queue.len() >= self.entry_queue_cap
904    }
905
906    pub fn append_data_nodes(&mut self, nodes: Vec<Arc<DataNode>>) -> Result<()> {
907        if self.is_full() {
908            return_errno_with_msg!(OutOfMemory, "cache out of capacity");
909        }
910
911        self.node_queue.extend(nodes);
912        if self.node_queue.len() >= self.node_queue_cap {
913            // If node queue is full, flush nodes to the entry queue
914            self.flush_node_queue()?;
915        }
916        Ok(())
917    }
918
919    pub fn search_data_nodes(&self, search_ctx: &mut SearchCtx) -> Result<()> {
920        let start_pos = self.start_pos;
921        let (pos, num) = (search_ctx.pos, search_ctx.num);
922        if pos + num <= start_pos {
923            return Ok(());
924        }
925
926        let (mut start_nth, mut end_nth, mut offset) = if pos >= start_pos {
927            let start = pos - start_pos;
928            (start, start + num, 0)
929        } else {
930            let end = pos + num - start_pos;
931            let offset = search_ctx.num - end;
932            search_ctx.num -= end;
933            (0, end, offset)
934        };
935        debug_assert!(end_nth <= self.num_append());
936
937        // Read from entry queue first if needed
938        for entry in self
939            .entry_queue
940            .iter()
941            .skip(start_nth)
942            .take(end_nth - start_nth)
943        {
944            self.storage
945                .read_data_node(entry, search_ctx.node_buf(offset))?;
946            start_nth = 0;
947            end_nth -= 1;
948            offset += 1;
949        }
950
951        // Read from node queue if needed
952        for node in self
953            .node_queue
954            .iter()
955            .skip(start_nth)
956            .take(end_nth - start_nth)
957        {
958            let node_buf = search_ctx.node_buf(offset);
959            node_buf.copy_from_slice(&node.0);
960            offset += 1;
961        }
962
963        if pos >= start_pos {
964            search_ctx.is_completed = true;
965        }
966        Ok(())
967    }
968
969    pub fn flush(&mut self) -> Result<Vec<MhtNodeEntry>> {
970        self.flush_node_queue()?;
971        debug_assert!(self.node_queue.is_empty());
972
973        let all_cached_entries: Vec<MhtNodeEntry> = self.entry_queue.drain(..).collect();
974        self.start_pos += all_cached_entries.len();
975        Ok(all_cached_entries)
976    }
977
978    fn flush_node_queue(&mut self) -> Result<()> {
979        let new_node_entries = self.storage.append_data_nodes(&self.node_queue)?;
980        self.entry_queue.extend_from_slice(&new_node_entries);
981        self.node_queue.clear();
982        Ok(())
983    }
984
985    fn calc_queue_cap(capacity: usize, append_pos: Lbid) -> (usize, usize) {
986        // Half for data nodes, half for data node entries
987        let node_queue_cap = capacity / 2;
988        let entry_queue_cap = {
989            let max_cap = Self::MAX_ENTRY_QUEUE_CAP - append_pos;
990            let remain_cap = (capacity - node_queue_cap) * BLOCK_SIZE / size_of::<MhtNodeEntry>();
991            max_cap.min(remain_cap)
992        };
993        (node_queue_cap, entry_queue_cap)
994    }
995}

1034
1035impl<L: BlockLog> Debug for MhtDisplayer<'_, L> {
1036    // A heavy implementation to display the whole MHT.
1037    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
1038        let mut debug_struct = f.debug_struct("Mht");
1039
1040        // Display root MHT node
1041        let root_meta = self.0.root_meta();
1042        debug_struct.field("\nroot_meta", &root_meta);
1043        if root_meta.is_none() {
1044            return debug_struct.finish();
1045        }
1046        let root_mht_node = self.0.root_node().unwrap();
1047        debug_struct.field("\n-> root_mht_node", &root_mht_node);
1048        let mut height = root_mht_node.height();
1049        if MhtNode::is_lowest_level(height) {
1050            return debug_struct.finish();
1051        }
1052
1053        // Display internal MHT nodes hierarchically
1054        let mut level_entries: Vec<MhtNodeEntry> = root_mht_node
1055            .entries
1056            .into_iter()
1057            .take(root_mht_node.num_valid_entries())
1058            .collect();
1059        'outer: loop {
1060            let level_size = level_entries.len();
1061            for i in 0..level_size {
1062                let entry = &level_entries[i];
1063                let node = self
1064                    .0
1065                    .storage
1066                    .read_mht_node(entry.pos, &entry.key, &entry.mac, &Iv::new_zeroed())
1067                    .unwrap();
1068                debug_struct.field("\n node_entry", entry);
1069                debug_struct.field("\n -> mht_node", &node);
1070                for i in 0..node.num_valid_entries() {
1071                    level_entries.push(node.entries[i]);
1072                }
1073            }
1074            level_entries.drain(..level_size);
1075            height -= 1;
1076            if MhtNode::is_lowest_level(height) {
1077                break 'outer;
1078            }
1079        }
1080        debug_struct.finish()
1081    }
1082}
