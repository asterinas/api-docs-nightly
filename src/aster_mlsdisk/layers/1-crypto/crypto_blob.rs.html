12/// A cryptographically-protected blob of user data.
13///
14/// `CryptoBlob<B>` allows a variable-length of user data to be securely
15/// written to and read from a fixed, pre-allocated block set
16/// (represented by `B: BlockSet`) on disk. Obviously, the length of user data
17/// must be smaller than that of the block set.
18///
19/// # On-disk format
20///
21/// The on-disk format of `CryptoBlob` is shown below.
22///
23/// ```
24/// ┌─────────┬─────────┬─────────┬──────────────────────────────┐
25/// │VersionId│   MAC   │  Length │       Encrypted Payload      │
26/// │  (8B)   │  (16B)  │   (8B)  │        (Length bytes)        │
27/// └─────────┴─────────┴─────────┴──────────────────────────────┘
28/// ```
29///
30/// The version ID increments by one each time the `CryptoBlob` is updated.
31/// The MAC protects the integrity of the length and the encrypted payload.
32///
33/// # Security
34///
35/// To ensure the confidentiality and integrity of user data, `CryptoBlob`
36/// takes several measures:
37///
38/// 1. Each instance of `CryptoBlob` is associated with a randomly-generated,
39///    unique encryption key.
40/// 2. Each instance of `CryptoBlob` maintains a version ID, which is
41///    automatically incremented by one upon each write.
42/// 3. The user data written to a `CryptoBlob` is protected with authenticated
43///    encryption before being persisted to the disk.
44///    The encryption takes the current version ID as the IV and generates a MAC
45///    as the output.
46/// 4. To read user data from a `CryptoBlob`, it first decrypts
47///    the untrusted on-disk data with the encryption key associated with this object
48///    and validating its integrity. Optimally, the user can check the version ID
49///    of the decrypted user data and see if the version ID is up-to-date.
50///
51pub struct CryptoBlob<B> {
52    block_set: B,
53    key: Key,
54    header: Mutex<Option<Header>>,
55}
56
57#[repr(C)]
58#[derive(Copy, Clone, Pod)]
59struct Header {
60    version: VersionId,
61    mac: Mac,
62    payload_len: usize,
63}
64
65impl<B: BlockSet> CryptoBlob<B> {
66    /// The size of the header of a crypto blob in bytes.
67    pub const HEADER_NBYTES: usize = size_of::<Header>();
68
69    /// Opens an existing `CryptoBlob`.
70    ///
71    /// The capacity of this `CryptoBlob` object is determined by the size
72    /// of `block_set: B`.
73    pub fn open(key: Key, block_set: B) -> Self {
74        Self {
75            block_set,
76            key,
77            header: Mutex::new(None),
78        }
79    }
80
81    /// Creates a new `CryptoBlob`.
82    ///
83    /// The encryption key of a `CryptoBlob` is generated randomly so that
84    /// no two `CryptoBlob` instances shall ever use the same key.
85    pub fn create(block_set: B, init_data: &[u8]) -> Result<Self> {
86        let capacity = block_set.nblocks() * BLOCK_SIZE - Self::HEADER_NBYTES;
87        if init_data.len() > capacity {
88            return_errno_with_msg!(OutOfDisk, "init_data is too large");
89        }
90        let nblocks = (Self::HEADER_NBYTES + init_data.len()).div_ceil(BLOCK_SIZE);
91        let mut block_buf = Buf::alloc(nblocks)?;
92
93        // Encrypt init_data.
94        let aead = Aead::new();
95        let key = Key::random();
96        let version: VersionId = 0;
97        let mut iv = Iv::new_zeroed();
98        iv.as_bytes_mut()[..version.as_bytes().len()].copy_from_slice(version.as_bytes());
99        let output = &mut block_buf.as_mut_slice()
100            [Self::HEADER_NBYTES..Self::HEADER_NBYTES + init_data.len()];
101        let mac = aead.encrypt(init_data, &key, &iv, &[], output)?;
102
103        // Store header.
104        let header = Header {
105            version,
106            mac,
107            payload_len: init_data.len(),
108        };
109        block_buf.as_mut_slice()[..Self::HEADER_NBYTES].copy_from_slice(header.as_bytes());
110
111        // Write to `BlockSet`.
112        block_set.write(0, block_buf.as_ref())?;
113        Ok(Self {
114            block_set,
115            key,
116            header: Mutex::new(Some(header)),
117        })
118    }
119
120    /// Write the buffer to the disk as the latest version of the content of
121    /// this `CryptoBlob`.
122    ///
123    /// The size of the buffer must not be greater than the capacity of this
124    /// `CryptoBlob`.
125    ///
126    /// Each successful write increments the version ID by one. If
127    /// there is no valid version ID, an `Error` will be returned.
128    /// User could get a version ID, either by a successful call to
129    /// `read`, or `recover_from` another valid `CryptoBlob`.
130    ///
131    /// # Security
132    ///
133    /// This content is guaranteed to be confidential as long as the key is not
134    /// known to an attacker.
135    pub fn write(&mut self, buf: &[u8]) -> Result<VersionId> {
136        if buf.len() > self.capacity() {
137            return_errno_with_msg!(OutOfDisk, "write data is too large");
138        }
139        let nblocks = (Self::HEADER_NBYTES + buf.len()).div_ceil(BLOCK_SIZE);
140        let mut block_buf = Buf::alloc(nblocks)?;
141
142        // Encrypt payload.
143        let aead = Aead::new();
144        let version = match self.version_id() {
145            Some(version) => version + 1,
146            None => return_errno_with_msg!(NotFound, "write with no valid version ID"),
147        };
148        let mut iv = Iv::new_zeroed();
149        iv.as_bytes_mut()[..version.as_bytes().len()].copy_from_slice(version.as_bytes());
150        let output =
151            &mut block_buf.as_mut_slice()[Self::HEADER_NBYTES..Self::HEADER_NBYTES + buf.len()];
152        let mac = aead.encrypt(buf, &self.key, &iv, &[], output)?;
153
154        // Store header.
155        let header = Header {
156            version,
157            mac,
158            payload_len: buf.len(),
159        };
160        block_buf.as_mut_slice()[..Self::HEADER_NBYTES].copy_from_slice(header.as_bytes());
161
162        // Write to `BlockSet`.
163        self.block_set.write(0, block_buf.as_ref())?;
164        *self.header.lock() = Some(header);
165        Ok(version)
166    }
167
168    /// Read the content of the `CryptoBlob` from the disk into the buffer.
169    ///
170    /// The given buffer must has a length that is no less than the size of
171    /// the plaintext content of this `CryptoBlob`.
172    ///
173    /// # Security
174    ///
175    /// This content, including its length, is guaranteed to be authentic.
176    pub fn read(&self, buf: &mut [u8]) -> Result<usize> {
177        let header = match *self.header.lock() {
178            Some(header) => header,
179            None => {
180                let mut header = Header::new_zeroed();
181                self.block_set.read_slice(0, header.as_bytes_mut())?;
182                header
183            }
184        };
185        if header.payload_len > self.capacity() {
186            return_errno_with_msg!(OutOfDisk, "payload_len is greater than the capacity");
187        }
188        if header.payload_len > buf.len() {
189            return_errno_with_msg!(OutOfDisk, "read_buf is too small");
190        }
191        let nblock = (Self::HEADER_NBYTES + header.payload_len).div_ceil(BLOCK_SIZE);
192        let mut block_buf = Buf::alloc(nblock)?;
193        self.block_set.read(0, block_buf.as_mut())?;
194
195        // Decrypt payload.
196        let aead = Aead::new();
197        let version = header.version;
198        let mut iv = Iv::new_zeroed();
199        iv.as_bytes_mut()[..version.as_bytes().len()].copy_from_slice(version.as_bytes());
200        let input =
201            &block_buf.as_slice()[Self::HEADER_NBYTES..Self::HEADER_NBYTES + header.payload_len];
202        let output = &mut buf[..header.payload_len];
203        aead.decrypt(input, &self.key, &iv, &[], &header.mac, output)?;
204        *self.header.lock() = Some(header);
205        Ok(header.payload_len)
206    }
207
208    /// Returns the key associated with this `CryptoBlob`.
209    pub fn key(&self) -> &Key {
210        &self.key
211    }
212
213    /// Returns the current version ID.
214    ///
215    /// # Security
216    ///
217    /// It is valid after a successful call to `create`, `read` or `write`.
218    /// User could also get a version ID from another valid `CryptoBlob`,
219    /// (usually a backup), through method `recover_from`.
220    pub fn version_id(&self) -> Option<VersionId> {
221        self.header.lock().map(|header| header.version)
222    }
223
224    /// Recover from another `CryptoBlob`.
225    ///
226    /// If `CryptoBlob` doesn't have a valid version ID, e.g., payload decryption
227    /// failed when `read`, user could call this method to recover version ID and
228    /// payload from another `CryptoBlob` (usually a backup).
229    pub fn recover_from(&mut self, other: &CryptoBlob<B>) -> Result<()> {
230        if self.capacity() != other.capacity() {
231            return_errno_with_msg!(InvalidArgs, "capacity not aligned, recover failed");
232        }
233        if self.header.lock().is_some() {
234            return_errno_with_msg!(InvalidArgs, "no need to recover");
235        }
236        let nblocks = self.block_set.nblocks();
237        // Read version ID and payload from another `CryptoBlob`.
238        let mut read_buf = Buf::alloc(nblocks)?;
239        let payload_len = other.read(read_buf.as_mut_slice())?;
240        let version = other.version_id().unwrap();
241
242        // Encrypt payload.
243        let aead = Aead::new();
244        let mut iv = Iv::new_zeroed();
245        iv.as_bytes_mut()[..version.as_bytes().len()].copy_from_slice(version.as_bytes());
246        let input = &read_buf.as_slice()[..payload_len];
247        let mut write_buf = Buf::alloc(nblocks)?;
248        let output =
249            &mut write_buf.as_mut_slice()[Self::HEADER_NBYTES..Self::HEADER_NBYTES + payload_len];
250        let mac = aead.encrypt(input, self.key(), &iv, &[], output)?;
251
252        // Store header.
253        let header = Header {
254            version,
255            mac,
256            payload_len,
257        };
258        write_buf.as_mut_slice()[..Self::HEADER_NBYTES].copy_from_slice(header.as_bytes());
259
260        // Write to `BlockSet`.
261        self.block_set.write(0, write_buf.as_ref())?;
262        *self.header.lock() = Some(header);
263        Ok(())
264    }
265
266    /// Returns the current MAC of encrypted payload.
267    ///
268    /// # Security
269    ///
270    /// It is valid after a successful call to `create`, `read` or `write`.
271    pub fn current_mac(&self) -> Option<Mac> {
272        self.header.lock().map(|header| header.mac)
273    }
274
275    /// Returns the capacity of this `CryptoBlob` in bytes.
276    pub fn capacity(&self) -> usize {
277        self.block_set.nblocks() * BLOCK_SIZE - Self::HEADER_NBYTES
278    }
279
280    /// Returns the number of blocks occupied by the underlying `BlockSet`.
281    pub fn nblocks(&self) -> usize {
282        self.block_set.nblocks()
283    }
284}

286#[cfg(test)]
287mod tests {
288    use super::CryptoBlob;
289    use crate::layers::bio::{BLOCK_SIZE, BlockSet, MemDisk};
290
291    #[test]
292    fn create() {
293        let disk = MemDisk::create(2).unwrap();
294        let init_data = [1u8; BLOCK_SIZE];
295        let blob = CryptoBlob::create(disk, &init_data).unwrap();
296
297        println!("blob key: {:?}", blob.key());
298        assert_eq!(blob.version_id(), Some(0));
299        assert_eq!(blob.nblocks(), 2);
300        assert_eq!(
301            blob.capacity(),
302            2 * BLOCK_SIZE - CryptoBlob::<MemDisk>::HEADER_NBYTES
303        );
304    }
305
306    #[test]
307    fn open_and_read() {
308        let disk = MemDisk::create(4).unwrap();
309        let key = {
310            let subset = disk.subset(0..2).unwrap();
311            let init_data = [1u8; 1024];
312            let blob = CryptoBlob::create(subset, &init_data).unwrap();
313            blob.key
314        };
315
316        let subset = disk.subset(0..2).unwrap();
317        let blob = CryptoBlob::open(key, subset);
318        assert_eq!(blob.version_id(), None);
319        assert_eq!(blob.nblocks(), 2);
320        let mut buf = [0u8; BLOCK_SIZE];
321        let payload_len = blob.read(&mut buf).unwrap();
322        assert_eq!(buf[..payload_len], [1u8; 1024]);
323    }
324
325    #[test]
326    fn write() {
327        let disk = MemDisk::create(2).unwrap();
328        let init_data = [0u8; BLOCK_SIZE];
329        let mut blob = CryptoBlob::create(disk, &init_data).unwrap();
330
331        let write_buf = [1u8; 1024];
332        blob.write(&write_buf).unwrap();
333        let mut read_buf = [0u8; 1024];
334        blob.read(&mut read_buf).unwrap();
335        assert_eq!(read_buf, [1u8; 1024]);
336        assert_eq!(blob.version_id(), Some(1));
337    }
338
339    #[test]
340    fn recover_from() {
341        let disk = MemDisk::create(2).unwrap();
342        let init_data = [1u8; 1024];
343        let subset0 = disk.subset(0..1).unwrap();
344        let mut blob0 = CryptoBlob::create(subset0, &init_data).unwrap();
345        assert_eq!(blob0.version_id(), Some(0));
346        blob0.write(&init_data).unwrap();
347        assert_eq!(blob0.version_id(), Some(1));
348
349        let subset1 = disk.subset(1..2).unwrap();
350        let mut blob1 = CryptoBlob::open(blob0.key, subset1);
351        assert_eq!(blob1.version_id(), None);
352        blob1.recover_from(&blob0).unwrap();
353        let mut read_buf = [0u8; BLOCK_SIZE];
354        let payload_len = blob1.read(&mut read_buf).unwrap();
355        assert_eq!(read_buf[..payload_len], [1u8; 1024]);
356        assert_eq!(blob1.version_id(), Some(1));
357    }
