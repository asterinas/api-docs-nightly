15/// A cryptographically-protected chain of blocks.
16///
17/// `CryptoChain<L>` allows writing and reading a sequence of
18/// consecutive blocks securely to and from an untrusted storage of data log
19/// `L: BlockLog`.
20/// The target use case of `CryptoChain` is to implement secure journals,
21/// where old data are scanned and new data are appended.
22///
23/// # On-disk format
24///
25/// The on-disk format of each block is shown below.
26///
27/// ```text
28/// ┌─────────────────────┬───────┬──────────┬──────────┬──────────┬─────────┐
29/// │  Encrypted payload  │  Gap  │  Length  │  PreMac  │  CurrMac │   IV    │
30/// │(Length <= 4KB - 48B)│       │   (4B)   │  (16B)   │   (16B)  │  (12B)  │
31/// └─────────────────────┴───────┴──────────┴──────────┴──────────┴─────────┘
32///
33/// ◄─────────────────────────── Block size (4KB) ──────────────────────────►
34/// ```
35///
36/// Each block begins with encrypted user payload. The size of payload
37/// must be smaller than that of block size as each block ends with a footer
38/// (in plaintext).
39/// The footer consists of fours parts: the length of the payload (in bytes),
40/// the MAC of the previous block, the MAC of the current block, the IV used
41/// for encrypting the current block.
42/// The MAC of a block protects the encrypted payload, its length, and the MAC
43/// of the previous block.
44///
45/// # Security
46///
47/// Each `CryptoChain` is assigned a randomly-generated encryption key.
48/// Each block is encrypted using this key and a randomly-generated IV.
49/// This setup ensures the confidentiality of payload and even the same payloads
50/// result in different ciphertexts.
51///
52/// `CryptoChain` is called a "chain" of blocks because each block
53/// not only stores its own MAC, but also the MAC of its previous block.
54/// This effectively forms a "chain" (much like a blockchain),
55/// ensuring the orderness and consecutiveness of the sequence of blocks.
56///
57/// Due to this chain structure, the integrity of a `CryptoChain` can be ensured
58/// by verifying the MAC of the last block. Once the integrity of the last block
59/// is verified, the integrity of all previous blocks can also be verified.
60pub struct CryptoChain<L> {
61    block_log: L,
62    key: Key,
63    block_range: Range<BlockId>,
64    block_macs: Vec<Mac>,
65}
66
67#[repr(C)]
68#[derive(Copy, Clone, Pod)]
69struct Footer {
70    len: u32,
71    pre_mac: Mac,
72    this_mac: Mac,
73    this_iv: Iv,
74}
75
76impl<L: BlockLog> CryptoChain<L> {
77    /// The available size in each chained block is smaller than that of
78    /// the block size.
79    pub const AVAIL_BLOCK_SIZE: usize = BLOCK_SIZE - size_of::<Footer>();
80
81    /// Construct a new `CryptoChain` using `block_log: L` as the storage.
82    pub fn new(block_log: L) -> Self {
83        Self {
84            block_log,
85            block_range: 0..0,
86            key: Key::random(),
87            block_macs: Vec::new(),
88        }
89    }
90
91    /// Recover an existing `CryptoChain` backed by `block_log: L`,
92    /// starting from its `from` block.
93    pub fn recover(key: Key, block_log: L, from: BlockId) -> Recovery<L> {
94        Recovery::new(block_log, key, from)
95    }
96
97    /// Read a block at a specified position.
98    ///
99    /// The length of the given buffer should not be smaller than payload_len
100    /// stored in `Footer`.
101    ///
102    /// # Security
103    ///
104    /// The authenticity of the block is guaranteed.
105    pub fn read(&self, pos: BlockId, buf: &mut [u8]) -> Result<usize> {
106        if !self.block_range().contains(&pos) {
107            return_errno_with_msg!(NotFound, "read position is out of range");
108        }
109
110        // Read block and get footer.
111        let mut block_buf = Buf::alloc(1)?;
112        self.block_log.read(pos, block_buf.as_mut())?;
113        let footer: Footer = Pod::from_bytes(&block_buf.as_slice()[Self::AVAIL_BLOCK_SIZE..]);
114
115        let payload_len = footer.len as usize;
116        if payload_len > Self::AVAIL_BLOCK_SIZE || payload_len > buf.len() {
117            return_errno_with_msg!(OutOfDisk, "wrong payload_len or the read_buf is too small");
118        }
119
120        // Check the footer MAC, to ensure the orderness and consecutiveness of blocks.
121        let this_mac = self.block_macs.get(pos - self.block_range.start).unwrap();
122        if footer.this_mac.as_bytes() != this_mac.as_bytes() {
123            return_errno_with_msg!(NotFound, "check footer MAC failed");
124        }
125
126        // Decrypt payload.
127        let aead = Aead::new();
128        aead.decrypt(
129            &block_buf.as_slice()[..payload_len],
130            self.key(),
131            &footer.this_iv,
132            &footer.pre_mac,
133            &footer.this_mac,
134            &mut buf[..payload_len],
135        )?;
136        Ok(payload_len)
137    }
138
139    /// Append a block at the end.
140    ///
141    /// The length of the given buffer must not be larger than `AVAIL_BLOCK_SIZE`.
142    ///
143    /// # Security
144    ///
145    /// The confidentiality of the block is guaranteed.
146    pub fn append(&mut self, buf: &[u8]) -> Result<()> {
147        if buf.len() > Self::AVAIL_BLOCK_SIZE {
148            return_errno_with_msg!(OutOfDisk, "append data is too large");
149        }
150        let mut block_buf = Buf::alloc(1)?;
151
152        // Encrypt payload.
153        let aead = Aead::new();
154        let this_iv = Iv::random();
155        let pre_mac = self.block_macs.last().copied().unwrap_or_default();
156        let output = &mut block_buf.as_mut_slice()[..buf.len()];
157        let this_mac = aead.encrypt(buf, self.key(), &this_iv, &pre_mac, output)?;
158
159        // Store footer.
160        let footer = Footer {
161            len: buf.len() as _,
162            pre_mac,
163            this_mac,
164            this_iv,
165        };
166        let buf = &mut block_buf.as_mut_slice()[Self::AVAIL_BLOCK_SIZE..];
167        buf.copy_from_slice(footer.as_bytes());
168
169        self.block_log.append(block_buf.as_ref())?;
170        self.block_range.end += 1;
171        self.block_macs.push(this_mac);
172        Ok(())
173    }
174
175    /// Ensures the persistence of data.
176    pub fn flush(&self) -> Result<()> {
177        self.block_log.flush()
178    }
179
180    /// Trim the blocks before a specified position (exclusive).
181    ///
182    /// The purpose of this method is to free some memory used for keeping the
183    /// MACs of accessible blocks. After trimming, the range of accessible
184    /// blocks is shrunk accordingly.
185    pub fn trim(&mut self, before_block: BlockId) {
186        // We must ensure the invariance that there is at least one valid block
187        // after trimming.
188        debug_assert!(before_block < self.block_range.end);
189
190        if before_block <= self.block_range.start {
191            return;
192        }
193
194        let num_blocks_trimmed = before_block - self.block_range.start;
195        self.block_range.start = before_block;
196        self.block_macs.drain(..num_blocks_trimmed);
197    }
198
199    /// Returns the range of blocks that are accessible through the `CryptoChain`.
200    pub fn block_range(&self) -> &Range<BlockId> {
201        &self.block_range
202    }
203
204    /// Returns the underlying block log.
205    pub fn inner_log(&self) -> &L {
206        &self.block_log
207    }
208
209    /// Returns the encryption key of the `CryptoChain`.
210    pub fn key(&self) -> &Key {
211        &self.key
212    }
213}

215/// `Recovery<L>` represents an instance `CryptoChain<L>` being recovered.
216///
217/// An object `Recovery<L>` attempts to recover as many valid blocks of
218/// a `CryptoChain` as possible. A block is valid if and only if its real MAC
219/// is equal to the MAC value recorded in its successor.
220///
221/// For the last block, which does not have a successor block, the user
222/// can obtain its MAC from `Recovery<L>` and verify the MAC by comparing it
223/// with an expected value from another trusted source.
224pub struct Recovery<L> {
225    block_log: L,
226    key: Key,
227    block_range: Range<BlockId>,
228    block_macs: Vec<Mac>,
229    read_buf: Buf,
230    payload: Buf,
231}
232
233impl<L: BlockLog> Recovery<L> {
234    /// Construct a new `Recovery` from the `first_block` of
235    /// `block_log: L`, using a cryptographic `key`.
236    pub fn new(block_log: L, key: Key, first_block: BlockId) -> Self {
237        Self {
238            block_log,
239            key,
240            block_range: first_block..first_block,
241            block_macs: Vec::new(),
242            read_buf: Buf::alloc(1).unwrap(),
243            payload: Buf::alloc(1).unwrap(),
244        }
245    }
246
247    /// Returns the number of valid blocks.
248    ///
249    /// Each success call to `next` increments the number of valid blocks.
250    pub fn num_blocks(&self) -> usize {
251        self.block_range.len()
252    }
253
254    /// Returns the range of valid blocks.
255    ///
256    /// Each success call to `next` increments the upper bound by one.
257    pub fn block_range(&self) -> &Range<BlockId> {
258        &self.block_range
259    }
260
261    /// Returns the MACs of valid blocks.
262    ///
263    /// Each success call to `next` pushes the MAC of the new valid block.
264    pub fn block_macs(&self) -> &[Mac] {
265        &self.block_macs
266    }
267
268    /// Open a `CryptoChain<L>` from the recovery object.
269    ///
270    /// User should call `next` to  retrieve valid blocks as much as possible.
271    pub fn open(self) -> CryptoChain<L> {
272        CryptoChain {
273            block_log: self.block_log,
274            key: self.key,
275            block_range: self.block_range,
276            block_macs: self.block_macs,
277        }
278    }
279}

281#[gat]
282impl<L: BlockLog> LendingIterator for Recovery<L> {
283    type Item<'a> = &'a [u8];
284
285    fn next(&mut self) -> Option<Self::Item<'_>> {
286        let next_block_id = self.block_range.end;
287        self.block_log
288            .read(next_block_id, self.read_buf.as_mut())
289            .ok()?;
290
291        // Deserialize footer.
292        let footer: Footer =
293            Pod::from_bytes(&self.read_buf.as_slice()[CryptoChain::<L>::AVAIL_BLOCK_SIZE..]);
294        let payload_len = footer.len as usize;
295        if payload_len > CryptoChain::<L>::AVAIL_BLOCK_SIZE {
296            return None;
297        }
298
299        // Decrypt payload.
300        let aead = Aead::new();
301        aead.decrypt(
302            &self.read_buf.as_slice()[..payload_len],
303            &self.key,
304            &footer.this_iv,
305            &footer.pre_mac,
306            &footer.this_mac,
307            &mut self.payload.as_mut_slice()[..payload_len],
308        )
309        .ok()?;
310
311        // Crypto blocks are chained: each block stores not only
312        // the MAC of its own, but also the MAC of its previous block.
313        // So we need to check whether the two MAC values are the same.
314        // There is one exception that the `pre_mac` of the first block
315        // is NOT checked.
316        if self
317            .block_macs()
318            .last()
319            .is_some_and(|mac| mac.as_bytes() != footer.pre_mac.as_bytes())
320        {
321            return None;
322        }
323
324        self.block_range.end += 1;
325        self.block_macs.push(footer.this_mac);
326        Some(&self.payload.as_slice()[..payload_len])
327    }
328}

331mod tests {
332    use lending_iterator::LendingIterator;

334    use super::CryptoChain;
335    use crate::layers::bio::{BlockLog, BlockRing, BlockSet, MemDisk};

337    #[test]
338    fn new() {
339        let disk = MemDisk::create(16).unwrap();
340        let block_ring = BlockRing::new(disk);
341        block_ring.set_cursor(0);
342        let chain = CryptoChain::new(block_ring);
343
344        assert_eq!(chain.block_log.nblocks(), 0);
345        assert_eq!(chain.block_range.start, 0);
346        assert_eq!(chain.block_range.end, 0);
347        assert_eq!(chain.block_macs.len(), 0);
348    }
349
350    #[test]
351    fn append_trim_and_read() {
352        let disk = MemDisk::create(16).unwrap();
353        let block_ring = BlockRing::new(disk);
354        block_ring.set_cursor(0);
355        let mut chain = CryptoChain::new(block_ring);
356
357        let data = [1u8; 1024];
358        chain.append(&data[..256]).unwrap();
359        chain.append(&data[..512]).unwrap();
360        assert_eq!(chain.block_range.end, 2);
361        assert_eq!(chain.block_macs.len(), 2);
362
363        chain.trim(1);
364
365        assert_eq!(chain.block_range.start, 1);
366        assert_eq!(chain.block_range.end, 2);
367        assert_eq!(chain.block_macs.len(), 1);
368
369        let mut buf = [0u8; 1024];
370        let len = chain.read(1, &mut buf).unwrap();
371        assert_eq!(len, 512);
372        assert_eq!(buf[..512], [1u8; 512]);
373    }
374
375    #[test]
376    fn recover() {
377        let disk = MemDisk::create(16).unwrap();
378        let key = {
379            let sub_disk = disk.subset(0..8).unwrap();
380            let block_ring = BlockRing::new(sub_disk);
381            block_ring.set_cursor(0);
382            let data = [1u8; 1024];
383            let mut chain = CryptoChain::new(block_ring);
384            for _ in 0..4 {
385                chain.append(&data).unwrap();
386            }
387            chain.flush().unwrap();
388            chain.key
389        };
390
391        let sub_disk = disk.subset(0..8).unwrap();
392        let block_ring = BlockRing::new(sub_disk);
393        let mut recover = CryptoChain::recover(key, block_ring, 2);
394        while let Some(payload) = recover.next() {
395            assert_eq!(payload.len(), 1024);
396        }
397        let chain = recover.open();
398        assert_eq!(chain.block_range(), &(2..4));
399        assert_eq!(chain.block_macs.len(), 2);
400    }
