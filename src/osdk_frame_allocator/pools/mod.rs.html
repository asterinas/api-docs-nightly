23/// The global free buddies.
24static GLOBAL_POOL: SpinLock<BuddySet<MAX_BUDDY_ORDER>, LocalIrqDisabled> =
25    SpinLock::new(BuddySet::new_empty());
26/// A snapshot of the total size of the global free buddies, not precise.
27static GLOBAL_POOL_SIZE: AtomicUsize = AtomicUsize::new(0);
28
29// CPU-local free buddies.
30cpu_local! {
31    static LOCAL_POOL: RefCell<BuddySet<MAX_LOCAL_BUDDY_ORDER>> = RefCell::new(BuddySet::new_empty());
32}
33
34/// Maximum supported order of the buddy system.
35///
36/// i.e., it is the number of classes of free blocks. It determines the
37/// maximum size of each allocation.
38///
39/// A maximum buddy order of 32 supports up to 4KiB*2^31 = 8 TiB of chunks.
40const MAX_BUDDY_ORDER: BuddyOrder = 32;
41
42/// Maximum supported order of the buddy system for CPU-local buddy system.
43///
44/// Since large blocks are rarely allocated, caching such blocks will lead
45/// to much fragmentation.
46///
47/// Lock guards are also allocated on stack. We can limit the stack usage
48/// for common paths in this way.
49///
50/// A maximum local buddy order of 18 supports up to 4KiB*2^17 = 512 MiB of
51/// chunks.
52const MAX_LOCAL_BUDDY_ORDER: BuddyOrder = 18;
53
54pub(super) fn alloc(guard: &DisabledLocalIrqGuard, layout: Layout) -> Option<Paddr> {
55    let local_pool_cell = LOCAL_POOL.get_with(guard);
56    let mut local_pool = local_pool_cell.borrow_mut();
57    let mut global_pool = OnDemandGlobalLock::new();
58
59    let size_order = greater_order_of(layout.size());
60    let align_order = greater_order_of(layout.align());
61    let order = size_order.max(align_order);
62
63    let mut chunk_addr = None;
64
65    if order < MAX_LOCAL_BUDDY_ORDER {
66        chunk_addr = local_pool.alloc_chunk(order);
67    }
68
69    // Fall back to the global free lists if the local free lists are empty.
70    if chunk_addr.is_none() {
71        chunk_addr = global_pool.get().alloc_chunk(order);
72    }
73    // TODO: On memory pressure the global pool may be not enough. We may need
74    // to merge all buddy chunks from the local pools to the global pool and
75    // try again.
76
77    // If the alignment order is larger than the size order, we need to split
78    // the chunk and return the rest part back to the free lists.
79    let allocated_size = size_of_order(order);
80    if allocated_size > layout.size()
81        && let Some(chunk_addr) = chunk_addr
82    {
83        do_dealloc(
84            &mut local_pool,
85            &mut global_pool,
86            [(chunk_addr + layout.size(), allocated_size - layout.size())].into_iter(),
87        );
88    }
89
90    balancing::balance(local_pool.deref_mut(), &mut global_pool);
91
92    chunk_addr
93}

95pub(super) fn dealloc(
96    guard: &DisabledLocalIrqGuard,
97    segments: impl Iterator<Item = (Paddr, usize)>,
98) {
99    let local_pool_cell = LOCAL_POOL.get_with(guard);
100    let mut local_pool = local_pool_cell.borrow_mut();
101    let mut global_pool = OnDemandGlobalLock::new();
102
103    do_dealloc(&mut local_pool, &mut global_pool, segments);
104
105    balancing::balance(local_pool.deref_mut(), &mut global_pool);
106}

108pub(super) fn add_free_memory(_guard: &DisabledLocalIrqGuard, addr: Paddr, size: usize) {
109    let mut global_pool = OnDemandGlobalLock::new();
110
111    split_to_chunks(addr, size).for_each(|(addr, order)| {
112        global_pool.get().insert_chunk(addr, order);
113    });
114}
115
116fn do_dealloc(
117    local_pool: &mut BuddySet<MAX_LOCAL_BUDDY_ORDER>,
118    global_pool: &mut OnDemandGlobalLock,
119    segments: impl Iterator<Item = (Paddr, usize)>,
120) {
121    segments.for_each(|(addr, size)| {
122        split_to_chunks(addr, size).for_each(|(addr, order)| {
123            if order >= MAX_LOCAL_BUDDY_ORDER {
124                global_pool.get().insert_chunk(addr, order);
125            } else {
126                local_pool.insert_chunk(addr, order);
127            }
128        });
129    });
130}

132type GlobalLockGuard = SpinLockGuard<'static, BuddySet<MAX_BUDDY_ORDER>, LocalIrqDisabled>;
133
134/// An on-demand guard that locks the global pool when needed.
135///
136/// It helps to avoid unnecessarily locking the global pool, and also avoids
137/// repeatedly locking the global pool when it is needed multiple times.
138struct OnDemandGlobalLock {
139    guard: Option<GlobalLockGuard>,
140}
141
142impl OnDemandGlobalLock {
143    fn new() -> Self {
144        Self { guard: None }
145    }
146
147    fn get(&mut self) -> &mut GlobalLockGuard {
148        self.guard.get_or_insert_with(|| GLOBAL_POOL.lock())
149    }
150
151    /// Returns the size of the global pool.
152    ///
153    /// If the global pool is locked, returns the actual size of the global pool.
154    /// Otherwise, returns the last snapshot of the global pool size by loading
155    /// [`GLOBAL_POOL_SIZE`].
156    fn get_global_size(&self) -> usize {
157        if let Some(guard) = self.guard.as_ref() {
158            guard.total_size()
159        } else {
160            GLOBAL_POOL_SIZE.load(Ordering::Relaxed)
161        }
162    }
163}
164
165impl Drop for OnDemandGlobalLock {
166    fn drop(&mut self) {
167        // Updates [`GLOBAL_POOL_SIZE`] if the global pool is locked.
168        if let Some(guard) = self.guard.as_ref() {
169            GLOBAL_POOL_SIZE.store(guard.total_size(), Ordering::Relaxed);
170        }
171    }
172}
