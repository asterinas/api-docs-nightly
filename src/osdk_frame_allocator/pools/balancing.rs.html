3//! Controlling the balancing between CPU-local free pools and the global free pool.
4
5use ostd::cpu::num_cpus;
6
7use super::{BuddyOrder, BuddySet, MAX_LOCAL_BUDDY_ORDER, OnDemandGlobalLock, lesser_order_of};
8
9use crate::chunk::split_to_order;
10
11/// Controls the expected size of cache for each CPU-local free pool.
12///
13/// The expected size will be the size of `GLOBAL_POOL` divided by the number
14/// of the CPUs, and then divided by this constant.
15const CACHE_EXPECTED_PORTION: usize = 2;
16
17/// Returns the expected size of cache for each CPU-local free pool.
18///
19/// It depends on the size of the global free pool.
20fn cache_expected_size(global_size: usize) -> usize {
21    global_size / num_cpus() / CACHE_EXPECTED_PORTION
22}
23
24/// Controls the minimal size of cache for each CPU-local free pool.
25///
26/// The minimal will be the expected size divided by this constant.
27const CACHE_MINIMAL_PORTION: usize = 8;
28
29/// Returns the minimal size of cache for each CPU-local free pool.
30///
31/// It depends on the size of the global free pool.
32fn cache_minimal_size(global_size: usize) -> usize {
33    cache_expected_size(global_size) / CACHE_MINIMAL_PORTION
34}
35
36/// Controls the maximal size of cache for each CPU-local free pool.
37///
38/// The maximal will be the expected size multiplied by this constant.
39const CACHE_MAXIMAL_MULTIPLIER: usize = 2;
40
41/// Returns the maximal size of cache for each CPU-local free pool.
42///
43/// It depends on the size of the global free pool.
44fn cache_maximal_size(global_size: usize) -> usize {
45    cache_expected_size(global_size) * CACHE_MAXIMAL_MULTIPLIER
46}
47
48/// Balances a local cache and the global free pool.
49pub fn balance(local: &mut BuddySet<MAX_LOCAL_BUDDY_ORDER>, global: &mut OnDemandGlobalLock) {
50    let global_size = global.get_global_size();
51
52    let minimal_local_size = cache_minimal_size(global_size);
53    let expected_local_size = cache_expected_size(global_size);
54    let maximal_local_size = cache_maximal_size(global_size);
55
56    let local_size = local.total_size();
57
58    if local_size >= maximal_local_size {
59        // Move local frames to the global pool.
60        if local_size == 0 {
61            return;
62        }
63
64        let expected_removal = local_size - expected_local_size;
65        let lesser_order = lesser_order_of(expected_removal);
66
67        balance_to(local, &mut *global.get(), lesser_order);
68    } else if local_size < minimal_local_size {
69        // Move global frames to the local pool.
70        if global_size == 0 {
71            return;
72        }
73
74        let expected_allocation = expected_local_size - local_size;
75        let lesser_order = lesser_order_of(expected_allocation);
76
77        balance_to(&mut *global.get(), local, lesser_order);
78    }
79}
80
81/// Balances from `a` to `b`.
82fn balance_to<const MAX_ORDER1: BuddyOrder, const MAX_ORDER2: BuddyOrder>(
83    a: &mut BuddySet<MAX_ORDER1>,
84    b: &mut BuddySet<MAX_ORDER2>,
85    order: BuddyOrder,
86) {
87    let allocated_from_a = a.alloc_chunk(order);
88
89    if let Some(addr) = allocated_from_a {
90        if order >= MAX_ORDER2 {
91            let inserted_order = MAX_ORDER2 - 1;
92            split_to_order(addr, order, inserted_order).for_each(|addr| {
93                b.insert_chunk(addr, inserted_order);
94            });
95        } else {
96            b.insert_chunk(addr, order);
97        }
98    } else {
99        // Maybe the chunk size is too large.
100        // Try to reduce the order and balance again.
101        if order > 1 {
102            balance_to(a, b, order - 1);
103            balance_to(a, b, order - 1);
104        }
105    }
106}
