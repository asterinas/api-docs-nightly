6//! An implementation of the global physical memory frame allocator for
7//! [OSTD](https://crates.io/crates/ostd) based kernels.
8//!
9//! # Background
10//!
11//! `OSTD` has provided a page allocator interface, namely [`GlobalFrameAllocator`]
12//! and [`global_frame_allocator`] procedure macro, allowing users to plug in
13//! their own frame allocator into the kernel safely. You can refer to the
14//! [`ostd::mm::frame::allocator`] module for detailed introduction.
15//!
16//! # Introduction
17//!
18//! This crate is an implementation of a scalable and efficient global frame
19//! allocator based on the buddy system. It is by default shipped with OSDK
20//! for users that don't have special requirements on the frame allocator.
21//!
22//! [`GlobalFrameAllocator`]: ostd::mm::frame::GlobalFrameAllocator
23//! [`global_frame_allocator`]: ostd::global_frame_allocator
24

48fast_smp_counter! {
49    /// The total size of free memory.
50    pub static TOTAL_FREE_SIZE: usize;
51}
52
53/// Loads the total size (in bytes) of free memory in the allocator.
54pub fn load_total_free_size() -> usize {
55    TOTAL_FREE_SIZE.get()
56}
57
58/// The global frame allocator provided by OSDK.
59///
60/// It is a singleton that provides frame allocation for the kernel. If
61/// multiple instances of this struct are created, all the member functions
62/// will eventually access the same allocator.
63pub struct FrameAllocator;
64
65impl GlobalFrameAllocator for FrameAllocator {
66    fn alloc(&self, layout: Layout) -> Option<Paddr> {
67        let guard = irq::disable_local();
68        let res = cache::alloc(&guard, layout);
69        if res.is_some() {
70            TOTAL_FREE_SIZE.sub(guard.current_cpu(), layout.size());
71        }
72        res
73    }
74
75    fn dealloc(&self, addr: Paddr, size: usize) {
76        let guard = irq::disable_local();
77        TOTAL_FREE_SIZE.add(guard.current_cpu(), size);
78        cache::dealloc(&guard, addr, size);
79    }
80
81    fn add_free_memory(&self, addr: Paddr, size: usize) {
82        let guard = irq::disable_local();
83        TOTAL_FREE_SIZE.add(guard.current_cpu(), size);
84        pools::add_free_memory(&guard, addr, size);
85    }
86}
