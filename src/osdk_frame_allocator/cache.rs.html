3//! A fixed-size local cache for frame allocation.
4
5use core::{alloc::Layout, cell::RefCell};
6
7use ostd::{
8    cpu_local,
9    irq::DisabledLocalIrqGuard,
10    mm::{PAGE_SIZE, Paddr},
11};
12
13cpu_local! {
14    static CACHE: RefCell<CacheOfSizes> = RefCell::new(CacheOfSizes::new());
15}
16
17struct CacheOfSizes {
18    cache1: CacheArray<1, 12>,
19    cache2: CacheArray<2, 6>,
20    cache3: CacheArray<3, 6>,
21    cache4: CacheArray<4, 6>,
22}
23
24/// A fixed-size local cache for frame allocation.
25///
26/// Each cache array contains at most `COUNT` segments. Each segment contains
27/// `NR_CONT_FRAMES` contiguous frames.
28struct CacheArray<const NR_CONT_FRAMES: usize, const COUNT: usize> {
29    inner: [Option<Paddr>; COUNT],
30    size: usize,
31}
32
33impl<const NR_CONT_FRAMES: usize, const COUNT: usize> CacheArray<NR_CONT_FRAMES, COUNT> {
34    const fn new() -> Self {
35        Self {
36            inner: [const { None }; COUNT],
37            size: 0,
38        }
39    }
40
41    /// The size of the segments that this cache manages.
42    const fn segment_size() -> usize {
43        NR_CONT_FRAMES * PAGE_SIZE
44    }
45
46    /// Allocates a segment of frames.
47    ///
48    /// It may allocate directly from this cache. If the cache is empty, it
49    /// will fill the cache.
50    fn alloc(&mut self, guard: &DisabledLocalIrqGuard) -> Option<Paddr> {
51        if let Some(frame) = self.pop_front() {
52            return Some(frame);
53        }
54
55        let nr_to_alloc = COUNT * 2 / 3;
56        let allocated = super::pools::alloc(
57            guard,
58            Layout::from_size_align(nr_to_alloc * Self::segment_size(), PAGE_SIZE).unwrap(),
59        )?;
60
61        for i in 1..nr_to_alloc {
62            self.push_front(allocated + i * Self::segment_size());
63        }
64
65        Some(allocated)
66    }
67
68    /// Deallocates a segment of frames.
69    ///
70    /// It may deallocate directly to this cache. If the cache is full, it will
71    /// deallocate to the global pool.
72    fn dealloc(&mut self, guard: &DisabledLocalIrqGuard, addr: Paddr) {
73        if self.push_front(addr).is_none() {
74            let nr_to_dealloc = COUNT * 2 / 3 + 1;
75
76            let segments = (0..nr_to_dealloc).map(|i| {
77                if i == 0 {
78                    (addr, Self::segment_size())
79                } else {
80                    (self.pop_front().unwrap(), Self::segment_size())
81                }
82            });
83
84            super::pools::dealloc(guard, segments);
85        };
86    }
87
88    fn push_front(&mut self, frame: Paddr) -> Option<()> {
89        if self.size == COUNT {
90            return None;
91        }
92
93        self.inner[self.size] = Some(frame);
94        self.size += 1;
95        Some(())
96    }
97
98    fn pop_front(&mut self) -> Option<Paddr> {
99        if self.size == 0 {
100            return None;
101        }
102
103        let frame = self.inner[self.size - 1].take().unwrap();
104        self.size -= 1;
105        Some(frame)
106    }
107}
108
109impl CacheOfSizes {
110    const fn new() -> Self {
111        Self {
112            cache1: CacheArray::new(),
113            cache2: CacheArray::new(),
114            cache3: CacheArray::new(),
115            cache4: CacheArray::new(),
116        }
117    }
118}
119
120pub(super) fn alloc(guard: &DisabledLocalIrqGuard, layout: Layout) -> Option<Paddr> {
121    let nr_frames = layout.size() / PAGE_SIZE;
122    if layout.align() > layout.size() {
123        return super::pools::alloc(guard, layout);
124    }
125
126    let cache_cell = CACHE.get_with(guard);
127    let mut cache = cache_cell.borrow_mut();
128
129    match nr_frames {
130        1 => cache.cache1.alloc(guard),
131        2 => cache.cache2.alloc(guard),
132        3 => cache.cache3.alloc(guard),
133        4 => cache.cache4.alloc(guard),
134        _ => super::pools::alloc(guard, layout),
135    }
136}
137
138pub(super) fn dealloc(guard: &DisabledLocalIrqGuard, addr: Paddr, size: usize) {
139    let nr_frames = size / PAGE_SIZE;
140    if nr_frames > 4 {
141        super::pools::dealloc(guard, [(addr, size)].into_iter());
142        return;
143    }
144
145    let cache_cell = CACHE.get_with(guard);
146    let mut cache = cache_cell.borrow_mut();
147
148    match nr_frames {
149        1 => cache.cache1.dealloc(guard, addr),
150        2 => cache.cache2.dealloc(guard, addr),
151        3 => cache.cache3.dealloc(guard, addr),
152        4 => cache.cache4.dealloc(guard, addr),
153        _ => super::pools::dealloc(guard, [(addr, size)].into_iter()),
154    }
155}
