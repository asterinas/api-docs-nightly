5/// An extension trait for Rust integer types, including `u8`, `u16`, `u32`,
6/// `u64`, and `usize`, to provide methods to make integers aligned to a
7/// power of two.
8pub trait AlignExt {
9    /// Returns to the smallest number that is greater than or equal to
10    /// `self` and is a multiple of the given power of two.
11    ///
12    /// The method panics if `power_of_two` is not a
13    /// power of two or is smaller than 2 or the calculation overflows
14    /// because `self` is too large.
15    ///
16    /// # Examples
17    ///
18    /// ```
19    /// use crate::align_ext::AlignExt;
20    /// assert_eq!(12usize.align_up(2), 12);
21    /// assert_eq!(12usize.align_up(4), 12);
22    /// assert_eq!(12usize.align_up(8), 16);
23    /// assert_eq!(12usize.align_up(16), 16);
24    /// ```
25    fn align_up(self, power_of_two: Self) -> Self;
26
27    /// Returns to the greatest number that is smaller than or equal to
28    /// `self` and is a multiple of the given power of two.
29    ///
30    /// The method panics if `power_of_two` is not a
31    /// power of two or is smaller than 2 or the calculation overflows
32    /// because `self` is too large. In release mode,
33    ///
34    /// # Examples
35    ///
36    /// ```
37    /// use crate::align_ext::AlignExt;
38    /// assert_eq!(12usize.align_down(2), 12);
39    /// assert_eq!(12usize.align_down(4), 12);
40    /// assert_eq!(12usize.align_down(8), 8);
41    /// assert_eq!(12usize.align_down(16), 0);
42    /// ```
43    fn align_down(self, power_of_two: Self) -> Self;
44}
45
46macro_rules! impl_align_ext {
47    ($( $uint_type:ty ),+,) => {
48        $(
49            impl AlignExt for $uint_type {
50                #[inline]
51                fn align_up(self, align: Self) -> Self {
52                    assert!(align.is_power_of_two() && align >= 2);
53                    self.checked_add(align - 1).unwrap() & !(align - 1)
54                }
55
56                #[inline]
57                fn align_down(self, align: Self) -> Self {
58                    assert!(align.is_power_of_two() && align >= 2);
59                    self & !(align - 1)
60                }
61            }
62        )*
63    }
64}
65
66impl_align_ext! {
67    u8,
68    u16,
69    u32,
70    u64,
71    usize,
72}

74#[cfg(test)]
75mod test {
76    use super::*;
77
78    #[test]
79    fn align_up() {
80        let input_ns = [0usize, 1, 2, 9, 15, 21, 32, 47, 50];
81        let input_as = [2usize, 2, 2, 2, 4, 4, 8, 8, 8];
82        let output_ns = [0usize, 2, 2, 10, 16, 24, 32, 48, 56];
83
84        for i in 0..input_ns.len() {
85            let n = input_ns[i];
86            let a = input_as[i];
87            let n2 = output_ns[i];
88            assert!(n.align_up(a) == n2);
89        }
90    }
91
92    #[test]
93    fn align_down() {
94        let input_ns = [0usize, 1, 2, 9, 15, 21, 32, 47, 50];
95        let input_as = [2usize, 2, 2, 2, 4, 4, 8, 8, 8];
96        let output_ns = [0usize, 0, 2, 8, 12, 20, 32, 40, 48];
97
98        for i in 0..input_ns.len() {
99            let n = input_ns[i];
100            let a = input_as[i];
101            let n2 = output_ns[i];
102            assert!(n.align_down(a) == n2);
103        }
104    }
105}
