3//! CMOS RTC.
4//!
5//! "CMOS" is a tiny bit of very low power static memory that lives on the same chip as the
6//! Real-Time Clock (RTC).
7//!
8//! According to the Linux implementation, in x86, if the CMOS/RTC is present at the legacy
9//! addresses (I/O Ports 0x70 and 0x71), then it is an MC146818 CMOS/RTC. Therefore, in this
10//! module, the register addresses and data interpretation are based on the MC146818 datasheet.
11//!
12//! Reference:
13//! <https://elixir.bootlin.com/linux/v6.17.5/source/arch/x86/kernel/rtc.c#L69>
14//! <https://www.scs.stanford.edu/23wi-cs212/pintos/specs/mc146818a.pdf>
15
16use core::num::NonZeroU8;
17
18use log::warn;
19use ostd::{arch::{device::io_port::{ReadWriteAccess, WriteOnlyAccess}, kernel::ACPI_INFO}, io::IoPort, sync::SpinLock};
20
21use crate::SystemTime;
22use super::Driver;

24pub struct RtcCmos {
25    access: SpinLock<CmosAccess>,
26    status_b: StatusB,
27}
28
29impl Driver for RtcCmos {
30    fn try_new() -> Option<Self> {
31        // TODO: Due to historical reasons, the "NMI Enable" bit (named `NMI_EN` in Intel's
32        // datasheet) and the "Real Time Clock Index" bits are assigned to the same I/O port
33        // (`IOPORT_SEL`). Currently, we do not support NMIs. However, once we add support, we
34        // should reconsider the safety impact to allow OSTD users to safely manipulate the NMI
35        // enablement.
36        //
37        // Reference:
38        // <https://edc.intel.com/content/www/id/id/design/publications/core-ultra-p200s-series-processors-soc-i-o-registers/nmi-enable-and-real-time-clock-index-nmi-en-offset-70/>
39        // <https://wiki.osdev.org/CMOS#Non-Maskable_Interrupts>
40        const IOPORT_SEL: u16 = 0x70;
41        const IOPORT_VAL: u16 = 0x71;
42
43        let acpi_info = ACPI_INFO.get().unwrap();
44
45        if acpi_info.boot_flags.is_some_and(|flags| flags.use_time_and_alarm_namespace_for_rtc()) {
46            // "If set, indicates that the CMOS RTC is either not implemented, or does not exist at
47            // the legacy addresses". See:
48            // <https://uefi.org/specs/ACPI/6.5/05_ACPI_Software_Programming_Model.html#ia-pc-boot-architecture-flags>.
49            return None;
50        }
51
52        let (io_sel, io_val) = match (IoPort::acquire(IOPORT_SEL), IoPort::acquire(IOPORT_VAL)) {
53            (Ok(io_sel), Ok(io_val)) => (io_sel, io_val),
54            _ => {
55                warn!("Failed to acquire CMOS RTC PIO region");
56                return None;
57            }
58        };
59
60        let century_register = acpi_info.century_register;
61
62        let mut access = CmosAccess {
63            io_sel,
64            io_val,
65            century_register,
66        };
67        let status_b = access.read_status_b();
68
69        // Ideally, the absence of the CMOS RTC should be reported in the ACPI tables (We've
70        // checked the flag above). However, the ACPI tables are sometimes unreliable (e.g., QEMU
71        // never sets the flag), so we need to perform additional checks.
72        if !access.check_presence() {
73            warn!("CMOS RTC reports unexpected status, ignoring this device");
74            return None;
75        }
76
77        Some(Self {
78            access: SpinLock::new(access),
79            status_b,
80        })
81    }
82
83    fn read_rtc(&self) -> SystemTime {
84        CmosData::read_rtc(self).into()
85    }
86}
87
88struct CmosAccess {
89    io_sel: IoPort<u8, WriteOnlyAccess>,
90    io_val: IoPort<u8, ReadWriteAccess>,
91    century_register: Option<NonZeroU8>,
92}
93
94#[repr(u8)]
95enum Register {
96    Second = 0x00,
97    Minute = 0x02,
98    Hour = 0x04,
99    Day = 0x07,
100    Month = 0x08,
101    Year = 0x09,
102
103    StatusA = 0x0A,
104    StatusB = 0x0B,
105    // `StatusC` is not used.
106    StatusD = 0x0D,
107}
108
109bitflags::bitflags! {
110    struct StatusA: u8 {
111        /// The update in progress (UIP) bit.
112        const UIP = 1 << 7;
113    }
114}
115
116bitflags::bitflags! {
117    struct StatusB: u8 {
118        /// The data mode (DM) bit.
119        ///
120        /// This bit is set when the binary format is used; otherwise, the BCD format is used.
121        const DM_BINARY = 1 << 2;
122        /// The clock mode (CM) bit.
123        ///
124        /// This bit is set when the 24-hour format is used; otherwise, the 12-hour format is used.
125        const CM_24HOUR = 1 << 1;
126    }
127}
128
129bitflags::bitflags! {
130    struct StatusD: u8 {
131        /// The valid RAM and time (VRT) bit.
132        ///
133        /// This bit is set when the RAM and time are valid.
134        const VRT = 1 << 7;
135    }
136}
137
138impl CmosAccess {
139    pub(self) fn read_register(&mut self, reg: Register) -> u8 {
140        self.read_register_impl(reg as u8)
141    }
142
143    pub(self) fn read_century(&mut self) -> Option<u8> {
144        self.century_register.map(|r| self.read_register_impl(r.get()))
145    }
146
147    pub(self) fn read_status_a(&mut self) -> StatusA {
148        StatusA::from_bits_truncate(self.read_register_impl(Register::StatusA as u8))
149    }
150
151    pub(self) fn read_status_b(&mut self) -> StatusB {
152        StatusB::from_bits_truncate(self.read_register_impl(Register::StatusB as u8))
153    }
154
155    pub(self) fn check_presence(&mut self) -> bool {
156        // If a working CMOS RTC is present, `VRT` should be set and all other reserved bits should
157        // not be set.
158        self.read_register_impl(Register::StatusD as u8) == StatusD::VRT.bits()
159    }
160
161    fn read_register_impl(&mut self, reg: u8) -> u8 {
162        self.io_sel.write(reg);
163        self.io_val.read()
164    }
165}

167#[derive(Debug, Clone, PartialEq, Eq)]
168struct CmosData {
169    century: Option<NonZeroU8>,
170    year: u16,
171    month: u8,
172    day: u8,
173    hour: u8,
174    minute: u8,
175    second: u8,
176}
177
178impl CmosData {
179    pub(self) fn read_rtc(rtc: &RtcCmos) -> Self {
180        let mut access = rtc.access.lock();
181
182        let mut now = Self::from_rtc_raw(&mut access);
183        // Retry if the new value differs from the old one. An RTC update may occur in the
184        // meantime, which would result in an invalid value.
185        while let new = Self::from_rtc_raw(&mut access) && now != new {
186            now = new;
187        }
188
189        drop(access);
190
191        if !rtc.status_b.contains(StatusB::DM_BINARY) {
192            now.convert_bcd_to_binary();
193        }
194        if !rtc.status_b.contains(StatusB::CM_24HOUR) {
195            now.convert_12_hour_to_24_hour();
196        }
197        now.modify_year();
198
199        now
200    }
201
202    fn from_rtc_raw(access: &mut CmosAccess) -> Self {
203        // Wait if the RTC updates are in progress.
204        while access.read_status_a().contains(StatusA::UIP) {
205            core::hint::spin_loop();
206        }
207
208        let second = access.read_register(Register::Second);
209        let minute = access.read_register(Register::Minute);
210        let hour = access.read_register(Register::Hour);
211        let day = access.read_register(Register::Day);
212        let month = access.read_register(Register::Month);
213        let year = access.read_register(Register::Year) as u16;
214        let century = access.read_century().and_then(NonZeroU8::new);
215
216        CmosData {
217            century,
218            year,
219            month,
220            day,
221            hour,
222            minute,
223            second,
224        }
225    }
226
227    /// Converts BCD values to binary values.
228    fn convert_bcd_to_binary(&mut self) {
229        fn bcd_to_binary(val: u8) -> u8 {
230            (val & 0xF) + (val >> 4) * 10
231        }
232
233        self.second = bcd_to_binary(self.second);
234        self.minute = bcd_to_binary(self.minute);
235        self.hour = bcd_to_binary(self.hour & !Self::HOUR_IS_AFTERNOON) | (self.hour & Self::HOUR_IS_AFTERNOON);
236        self.day = bcd_to_binary(self.day);
237        self.month = bcd_to_binary(self.month);
238        self.year = bcd_to_binary(self.year as u8) as u16;
239        self.century = self.century.and_then(|c| NonZeroU8::new(bcd_to_binary(c.get())));
240    }
241
242    const HOUR_IS_AFTERNOON: u8 = 0x80;
243
244    /// Converts the 12-hour clock to the 24-hour clock.
245    fn convert_12_hour_to_24_hour(&mut self) {
246        if self.hour & Self::HOUR_IS_AFTERNOON != 0 {
247            self.hour = (self.hour & !Self::HOUR_IS_AFTERNOON) + 12;
248        }
249    }
250
251    /// Converts the year without the century (e.g., 10) to the year with the century (e.g., 2010).
252    fn modify_year(&mut self) {
253        const DEFAULT_21_CENTURY: u8 = 20;
254
255        self.year += (self.century.map(NonZeroU8::get).unwrap_or(DEFAULT_21_CENTURY) as u16) * 100;
256    }
257}
258
259impl From<CmosData> for SystemTime {
260    fn from(cmos: CmosData) -> SystemTime {
261        SystemTime {
262            year: cmos.year,
263            month: cmos.month,
264            day: cmos.day,
265            hour: cmos.hour,
266            minute: cmos.minute,
267            second: cmos.second,
268            nanos: 0,
269        }
270    }
