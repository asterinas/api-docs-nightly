23/// A pool of workers.
24///
25/// The `WorkerPool` maintains workers created from different CPUs, while clustering workers
26/// from the same CPU into a `LocalWorkerPool` for better management.
27pub struct WorkerPool {
28    local_pools: Vec<Arc<LocalWorkerPool>>,
29    /// Monitor invokes `schedule()` in WorkerScheduler to determine whether there is a need for
30    /// adding or removing workers.
31    monitor: Arc<Monitor>,
32    priority: WorkPriority,
33    cpu_set: CpuSet,
34    scheduler: Arc<dyn WorkerScheduler>,
35    work_queues: SpinLock<Vec<Arc<WorkQueue>>>,
36}
37
38/// A set of workers for a specific CPU.
39pub struct LocalWorkerPool {
40    cpu_id: CpuId,
41    idle_wait_queue: WaitQueue,
42    parent: Weak<WorkerPool>,
43    /// A liveness check for LocalWorkerPool. The monitor periodically clears heartbeat,
44    /// and when a worker completes an item, it will be set to indicate that there is still
45    /// an active worker. If there is no heartbeats and there are still pending work items,
46    /// it suggests that more workers are needed.
47    heartbeat: AtomicBool,
48    workers: SpinLock<VecDeque<Arc<Worker>>>,
49}
50
51/// Schedule `Workers` for a `WorkerPool`.
52///
53/// Having an excessive number of Workers in WorkerPool may result in wastage of system
54/// resources, while a shortage of workers may lead to longer response time for workitems.
55/// A well-designed WorkerScheduler must strike a balance between resource utilization and response time.
56pub trait WorkerScheduler: Sync + Send {
57    /// Schedule workers in a worker pool. This needs to solve two problems: when to increase or decrease
58    /// workers, and how to add or remove workers to keep the number of workers in a reasonable range.
59    fn schedule(&self);
60}
61
62/// The `Monitor` is responsible for monitoring the `WorkerPool` for scheduling needs.
63///
64/// Currently, it only performs a liveness check, and attempts to schedule when no workers
65/// are found processing in the pool.
66pub struct Monitor {
67    worker_pool: Weak<WorkerPool>,
68    bound_task: Arc<Task>,
69}
70
71impl LocalWorkerPool {
72    fn new(worker_pool: Weak<WorkerPool>, cpu_id: CpuId) -> Self {
73        LocalWorkerPool {
74            cpu_id,
75            idle_wait_queue: WaitQueue::new(),
76            parent: worker_pool,
77            heartbeat: AtomicBool::new(false),
78            workers: SpinLock::new(VecDeque::new()),
79        }
80    }
81
82    fn add_worker(&self) {
83        let worker = Worker::new(self.parent.clone(), self.cpu_id);
84        self.workers.disable_irq().lock().push_back(worker.clone());
85        worker.bound_task().as_thread().unwrap().run();
86    }
87
88    fn remove_worker(&self) {
89        let mut workers = self.workers.disable_irq().lock();
90        for (index, worker) in workers.iter().enumerate() {
91            if worker.is_idle() {
92                worker.destroy();
93                workers.remove(index);
94                break;
95            }
96        }
97    }
98
99    fn wake_worker(&self) -> bool {
100        self.idle_wait_queue.wake_one()
101    }
102
103    fn has_pending_work_items(&self) -> bool {
104        self.parent
105            .upgrade()
106            .unwrap()
107            .has_pending_work_items(self.cpu_id)
108    }
109
110    fn heartbeat(&self) -> bool {
111        self.heartbeat.load(Ordering::Acquire)
112    }
113
114    fn set_heartbeat(&self, heartbeat: bool) {
115        self.heartbeat.store(heartbeat, Ordering::Release);
116    }
117
118    fn idle_current_worker(&self, worker: Arc<Worker>) {
119        self.idle_wait_queue
120            .wait_until(|| (worker.is_destroying() || self.has_pending_work_items()).then_some(0));
121    }
122
123    fn destroy_all_workers(&self) {
124        for worker in self.workers.disable_irq().lock().iter() {
125            worker.destroy();
126        }
127        self.idle_wait_queue.wake_all();
128    }
129}

131impl WorkerPool {
132    pub fn new(priority: WorkPriority, cpu_set: CpuSet) -> Arc<Self> {
133        Arc::new_cyclic(|pool_ref| {
134            let mut local_pools = Vec::new();
135            for cpu_id in cpu_set.iter() {
136                local_pools.push(Arc::new(LocalWorkerPool::new(pool_ref.clone(), cpu_id)));
137            }
138            WorkerPool {
139                local_pools,
140                monitor: Monitor::new(pool_ref.clone(), &priority),
141                priority,
142                cpu_set,
143                scheduler: Arc::new(SimpleScheduler::new(pool_ref.clone())),
144                work_queues: SpinLock::new(Vec::new()),
145            }
146        })
147    }
148
149    pub fn run(&self) {
150        self.monitor.run();
151    }
152
153    pub fn assign_work_queue(&self, work_queue: Arc<WorkQueue>) {
154        self.work_queues.disable_irq().lock().push(work_queue);
155    }
156
157    pub fn has_pending_work_items(&self, request_cpu: CpuId) -> bool {
158        self.work_queues
159            .disable_irq()
160            .lock()
161            .iter()
162            .any(|work_queue| work_queue.has_pending_work_items(request_cpu))
163    }
164
165    pub fn schedule(&self) {
166        self.scheduler.schedule();
167    }
168
169    pub fn num_workers(&self, cpu_id: CpuId) -> u16 {
170        self.local_pool(cpu_id).workers.disable_irq().lock().len() as u16
171    }
172
173    pub fn cpu_set(&self) -> &CpuSet {
174        &self.cpu_set
175    }
176
177    pub(super) fn fetch_pending_work_item(&self, request_cpu: CpuId) -> Option<Arc<WorkItem>> {
178        for work_queue in self.work_queues.disable_irq().lock().iter() {
179            let item = work_queue.dequeue(request_cpu);
180            if item.is_some() {
181                return item;
182            }
183        }
184        None
185    }
186
187    fn local_pool(&self, cpu_id: CpuId) -> &Arc<LocalWorkerPool> {
188        self.local_pools
189            .iter()
190            .find(|local_pool: &&Arc<LocalWorkerPool>| local_pool.cpu_id == cpu_id)
191            .unwrap()
192    }
193
194    pub(super) fn wake_worker(&self, cpu_id: CpuId) -> bool {
195        self.local_pool(cpu_id).wake_worker()
196    }
197
198    pub(super) fn add_worker(&self, cpu_id: CpuId) {
199        self.local_pool(cpu_id).add_worker();
200    }
201
202    pub(super) fn remove_worker(&self, cpu_id: CpuId) {
203        self.local_pool(cpu_id).remove_worker();
204    }
205
206    pub(super) fn is_high_priority(&self) -> bool {
207        self.priority == WorkPriority::High
208    }
209
210    pub(super) fn heartbeat(&self, cpu_id: CpuId) -> bool {
211        self.local_pool(cpu_id).heartbeat()
212    }
213
214    pub(super) fn set_heartbeat(&self, cpu_id: CpuId, heartbeat: bool) {
215        self.local_pool(cpu_id).set_heartbeat(heartbeat)
216    }
217
218    pub(super) fn idle_current_worker(&self, cpu_id: CpuId, worker: Arc<Worker>) {
219        self.local_pool(cpu_id).idle_current_worker(worker);
220    }
221}
222
223impl Drop for WorkerPool {
224    fn drop(&mut self) {
225        for local_pool in self.local_pools.iter() {
226            local_pool.destroy_all_workers();
227        }
228    }
229}

231impl Monitor {
232    pub fn new(worker_pool: Weak<WorkerPool>, priority: &WorkPriority) -> Arc<Self> {
233        Arc::new_cyclic(|monitor_ref| {
234            let weal_monitor = monitor_ref.clone();
235            let task_fn = Box::new(move || {
236                let current_monitor: Arc<Monitor> = weal_monitor.upgrade().unwrap();
237                current_monitor.run_monitor_loop();
238            });
239            let cpu_affinity = CpuSet::new_full();
240            let sched_policy = SchedPolicy::Fair(match priority {
241                WorkPriority::High => Nice::MIN,
242                WorkPriority::Normal => Nice::default(),
243            });
244            let bound_task = ThreadOptions::new(task_fn)
245                .cpu_affinity(cpu_affinity)
246                .sched_policy(sched_policy)
247                .build();
248            Self {
249                worker_pool,
250                bound_task,
251            }
252        })
253    }
254
255    pub fn run(&self) {
256        self.bound_task.as_thread().unwrap().run()
257    }
258
259    fn run_monitor_loop(self: &Arc<Self>) {
260        let sleep_queue = WaitQueue::new();
261        let sleep_duration = Duration::from_millis(100);
262        loop {
263            let worker_pool = self.worker_pool.upgrade();
264            let Some(worker_pool) = worker_pool else {
265                break;
266            };
267            worker_pool.schedule();
268            for local_pool in worker_pool.local_pools.iter() {
269                local_pool.set_heartbeat(false);
270            }
271            let _ = sleep_queue.wait_until_or_timeout(|| -> Option<()> { None }, &sleep_duration);
272        }
273    }
