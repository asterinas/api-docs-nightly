17/// A worker thread. A `Worker` will attempt to retrieve unfinished
18/// work items from its corresponding `WorkerPool`. If there are none,
19/// it will go to sleep and be rescheduled when a new work item is
20/// added to the `WorkerPool`.
21pub(super) struct Worker {
22    worker_pool: Weak<WorkerPool>,
23    bound_task: Arc<Task>,
24    bound_cpu: CpuId,
25    inner: SpinLock<WorkerInner>,
26}
27
28struct WorkerInner {
29    worker_status: WorkerStatus,
30}
31
32#[derive(PartialEq)]
33enum WorkerStatus {
34    Idle,
35    Running,
36    Exited,
37    /// This state only occurs when destructing the `WorkerPool`,
38    /// where workers will exit after processing the remaining work items.
39    Destroying,
40}
41
42impl Worker {
43    /// Creates a new `Worker` to the given `worker_pool`.
44    pub(super) fn new(worker_pool: Weak<WorkerPool>, bound_cpu: CpuId) -> Arc<Self> {
45        Arc::new_cyclic(|worker_ref| {
46            let weal_worker = worker_ref.clone();
47            let task_fn = Box::new(move || {
48                let current_worker: Arc<Worker> = weal_worker.upgrade().unwrap();
49                current_worker.run_worker_loop();
50            });
51            let mut cpu_affinity = CpuSet::new_empty();
52            cpu_affinity.add(bound_cpu);
53            let sched_policy =
54                SchedPolicy::Fair(if worker_pool.upgrade().unwrap().is_high_priority() {
55                    Nice::MIN
56                } else {
57                    Nice::default()
58                });
59            let bound_task = ThreadOptions::new(task_fn)
60                .cpu_affinity(cpu_affinity)
61                .sched_policy(sched_policy)
62                .build();
63            Self {
64                worker_pool,
65                bound_task,
66                bound_cpu,
67                inner: SpinLock::new(WorkerInner {
68                    worker_status: WorkerStatus::Running,
69                }),
70            }
71        })
72    }
73
74    pub(super) fn run(&self) {
75        let thread = self.bound_task.as_thread().unwrap();
76        thread.run();
77    }
78
79    /// The thread function bound to normal workers.
80    /// It pulls a work item from the work queue and sleeps if there is no more pending items.
81    fn run_worker_loop(self: &Arc<Self>) {
82        loop {
83            let worker_pool = self.worker_pool.upgrade();
84            let Some(worker_pool) = worker_pool else {
85                break;
86            };
87            if let Some(work_item) = worker_pool.fetch_pending_work_item(self.bound_cpu) {
88                work_item.set_processing();
89                work_item.call_work_func();
90                worker_pool.set_heartbeat(self.bound_cpu, true);
91            } else {
92                if self.is_destroying() {
93                    break;
94                }
95                self.inner.disable_irq().lock().worker_status = WorkerStatus::Idle;
96                worker_pool.idle_current_worker(self.bound_cpu, self.clone());
97                if !self.is_destroying() {
98                    self.inner.disable_irq().lock().worker_status = WorkerStatus::Running;
99                }
100            }
101        }
102        self.exit();
103    }
104
105    pub(super) fn bound_task(&self) -> &Arc<Task> {
106        &self.bound_task
107    }
108
109    pub(super) fn is_idle(&self) -> bool {
110        self.inner.disable_irq().lock().worker_status == WorkerStatus::Idle
111    }
112
113    pub(super) fn is_destroying(&self) -> bool {
114        self.inner.disable_irq().lock().worker_status == WorkerStatus::Destroying
115    }
116
117    pub(super) fn destroy(&self) {
118        self.inner.disable_irq().lock().worker_status = WorkerStatus::Destroying;
119    }
120
121    fn exit(&self) {
122        self.inner.disable_irq().lock().worker_status = WorkerStatus::Exited;
123    }
124
125    pub(super) fn is_exit(&self) -> bool {
126        self.inner.disable_irq().lock().worker_status == WorkerStatus::Exited
127    }
128}
