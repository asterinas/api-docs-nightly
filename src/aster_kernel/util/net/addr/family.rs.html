115const ADDR_MAX_LEN: usize = 128;
116
117/// Storage that can contain _any_ socket addresses.
118///
119/// The size and layout of this structure is specified by RFC 3493. For details, see
120/// <https://datatracker.ietf.org/doc/html/rfc3493#section-3.10>.
121#[repr(C)]
122#[derive(Debug, Clone, Copy, Pod)]
123struct Storage {
124    sa_family: u16,
125    bytes: [u8; ADDR_MAX_LEN - 2],
126    _align: [u64; 0],
127}
128
129/// Reads a socket address from userspace.
130///
131/// This method returns `Err(EINVAL)` for invalid socket address lengths and `Err(EAFNOSUPPORT)`
132/// for unsupported address families.
133///
134/// These error codes may be different from the Linux ones, but the difference is tricky and hard
135/// to fix. The main reason for this is that in Linux it's up to each protocol to decide how to
136/// intercept the bytes that represent socket addresses, but here this method is designed to parse
137/// socket addresses before diving deep into protocol-specific code.
138pub fn read_socket_addr_from_user(addr: Vaddr, addr_len: usize) -> Result<SocketAddr> {
139    if addr_len > ADDR_MAX_LEN {
140        return_errno_with_message!(Errno::EINVAL, "the socket address length is too large");
141    }
142
143    if addr_len < 2 {
144        return_errno_with_message!(Errno::EINVAL, "the socket address length is too small");
145    }
146
147    let mut storage = Storage::new_zeroed();
148    current_userspace!().read_bytes(addr, &mut storage.as_mut_bytes()[..addr_len])?;
149
150    let result = match CSocketAddrFamily::try_from(storage.sa_family as i32) {
151        Ok(CSocketAddrFamily::AF_INET) => {
152            if addr_len < size_of::<CSocketAddrInet>() {
153                return_errno_with_message!(Errno::EINVAL, "the socket address length is too small");
154            }
155            let (addr, port) = CSocketAddrInet::from_first_bytes(storage.as_bytes()).into();
156            SocketAddr::IPv4(addr, port)
157        }
158        Ok(CSocketAddrFamily::AF_UNIX) => {
159            let addr = unix::from_c_bytes(&storage.as_bytes()[..addr_len])?;
160            SocketAddr::Unix(addr)
161        }
162        Ok(CSocketAddrFamily::AF_NETLINK) => {
163            if addr_len < size_of::<CSocketAddrNetlink>() {
164                return_errno_with_message!(Errno::EINVAL, "the socket address length is too small");
165            }
166            let addr = CSocketAddrNetlink::from_first_bytes(storage.as_bytes());
167            SocketAddr::Netlink(addr.into())
168        }
169        Ok(CSocketAddrFamily::AF_VSOCK) => {
170            if addr_len < size_of::<CSocketAddrVm>() {
171                return_errno_with_message!(Errno::EINVAL, "the socket address length is too small");
172            }
173            let addr = CSocketAddrVm::from_first_bytes(storage.as_bytes());
174            SocketAddr::Vsock(addr.into())
175        }
176        _ => {
177            return_errno_with_message!(
178                Errno::EAFNOSUPPORT,
179                "the specified address family is not supported"
180            )
181        }
182    };
183
184    Ok(result)
185}

187/// Writes a socket address and its length to userspace.
188///
189/// Similar to [`write_socket_addr_with_max_len`], the socket address may be truncated if the
190/// buffer is not long enough. Even if truncation occurs, the actual length of the socket address
191/// is written to userspace. See <https://man7.org/linux/man-pages/man3/recvmsg.3p.html> for
192/// details on this behavior.
193///
194/// # Panics
195///
196/// This method will panic if the socket address cannot be validly mapped to the corresponding
197/// Linux C structures. Currently, the only possible example is that the pathname in the UNIX
198/// domain socket address is too long.
199///
200/// It is guaranteed that all socket addresses returned by [`read_socket_addr_from_user`] have
201/// valid representations for the corresponding C structures, so passing them to this method will
202/// not cause panic.
203pub fn write_socket_addr_to_user(
204    socket_addr: &SocketAddr,
205    dest: Vaddr,
206    max_len_ptr: Vaddr,
207) -> Result<()> {
208    let current_task = Task::current().unwrap();
209    let user_space = CurrentUserSpace::new(current_task.as_thread_local().unwrap());
210
211    let max_len = user_space.read_val::<i32>(max_len_ptr)?;
212
213    let actual_len = write_socket_addr_with_max_len(socket_addr, dest, max_len)?;
214
215    Ok(user_space.write_val(max_len_ptr, &actual_len)?)
216}

218/// Writes a socket address to the user space.
219///
220/// If the specified maximum length for the socket address is not enough, the socket address is
221/// truncated to the specified maximum length. This method returns the _actual_ length of the
222/// socket address, regardless of whether the truncation occurs or not.
223///
224/// # Panics
225///
226/// This method will panic if the socket address cannot be validly mapped to the corresponding
227/// Linux C structures. Currently, the only possible example is that the pathname in the UNIX
228/// domain socket address is too long.
229///
230/// It is guaranteed that all socket addresses returned by [`read_socket_addr_from_user`] have
231/// valid representations for the corresponding C structures, so passing them to this method will
232/// not cause panic.
233pub fn write_socket_addr_with_max_len(
234    socket_addr: &SocketAddr,
235    dest: Vaddr,
236    max_len: i32,
237) -> Result<i32> {
238    if max_len < 0 {
239        return_errno_with_message!(
240            Errno::EINVAL,
241            "the socket address length cannot be negative"
242        );
243    }
244
245    let actual_len = match socket_addr {
246        SocketAddr::IPv4(addr, port) => write_c_socket_address_util::<CSocketAddrInet, _>(
247            (*addr, *port),
248            dest,
249            max_len as usize,
250        )?,
251        SocketAddr::Unix(addr) => unix::into_c_bytes_and(addr, |bytes| {
252            let written_len = min(bytes.len(), max_len as _);
253            current_userspace!().write_bytes(dest, &bytes[..written_len])?;
254            Ok::<usize, Error>(bytes.len())
255        })?,
256        SocketAddr::Netlink(addr) => {
257            write_c_socket_address_util::<CSocketAddrNetlink, _>(*addr, dest, max_len as usize)?
258        }
259        SocketAddr::Vsock(addr) => {
260            write_c_socket_address_util::<CSocketAddrVm, _>(*addr, dest, max_len as usize)?
261        }
262    };
263
264    Ok(actual_len as i32)
265}

267// Utility function to write a C socket address to user space.
268fn write_c_socket_address_util<TCSockAddr, TSockAddr>(
269    addr: TSockAddr,
270    dest: Vaddr,
271    max_len: usize,
272) -> Result<usize>
273where
274    TCSockAddr: Pod + From<TSockAddr>,
275{
276    let c_socket_addr = TCSockAddr::from(addr);
277    let actual_len = size_of::<TCSockAddr>();
278    let written_len = min(actual_len, max_len);
279
280    current_userspace!().write_bytes(dest, &c_socket_addr.as_bytes()[..written_len])?;
281
282    Ok(actual_len)
