13/// Standard well-defined IP protocols.
14/// From <https://elixir.bootlin.com/linux/v6.0.9/source/include/uapi/linux/in.h>.
15#[repr(i32)]
16#[derive(Debug, Clone, Copy, TryFromInt)]
17#[expect(non_camel_case_types)]
18pub enum Protocol {
19    IPPROTO_IP = 0,         /* Dummy protocol for TCP		*/
20    IPPROTO_ICMP = 1,       /* Internet Control Message Protocol	*/
21    IPPROTO_IGMP = 2,       /* Internet Group Management Protocol	*/
22    IPPROTO_TCP = 6,        /* Transmission Control Protocol	*/
23    IPPROTO_EGP = 8,        /* Exterior Gateway Protocol		*/
24    IPPROTO_PUP = 12,       /* PUP protocol				*/
25    IPPROTO_UDP = 17,       /* User Datagram Protocol		*/
26    IPPROTO_IDP = 22,       /* XNS IDP protocol			*/
27    IPPROTO_TP = 29,        /* SO Transport Protocol Class 4	*/
28    IPPROTO_DCCP = 33,      /* Datagram Congestion Control Protocol */
29    IPPROTO_IPV6 = 41,      /* IPv6-in-IPv4 tunnelling		*/
30    IPPROTO_RSVP = 46,      /* RSVP Protocol			*/
31    IPPROTO_GRE = 47,       /* Cisco GRE tunnels (rfc 1701,1702)	*/
32    IPPROTO_ESP = 50,       /* Encapsulation Security Payload protocol */
33    IPPROTO_AH = 51,        /* Authentication Header protocol	*/
34    IPPROTO_MTP = 92,       /* Multicast Transport Protocol		*/
35    IPPROTO_BEETPH = 94,    /* IP option pseudo header for BEET	*/
36    IPPROTO_ENCAP = 98,     /* Encapsulation Header			*/
37    IPPROTO_PIM = 103,      /* Protocol Independent Multicast	*/
38    IPPROTO_COMP = 108,     /* Compression Header Protocol		*/
39    IPPROTO_SCTP = 132,     /* Stream Control Transport Protocol	*/
40    IPPROTO_UDPLITE = 136,  /* UDP-Lite (RFC 3828)			*/
41    IPPROTO_MPLS = 137,     /* MPLS in IP (RFC 4023)		*/
42    IPPROTO_ETHERNET = 143, /* Ethernet-within-IPv6 Encapsulation	*/
43    IPPROTO_RAW = 255,      /* Raw IP packets			*/
44    IPPROTO_MPTCP = 262,    /* Multipath TCP connection		*/
45}

47/// Socket types.
48/// From <https://elixir.bootlin.com/linux/v6.0.9/source/include/linux/net.h>.
49#[repr(i32)]
50#[expect(non_camel_case_types)]
51#[derive(Debug, Clone, Copy, TryFromInt)]
52pub enum SockType {
53    /// Stream socket
54    SOCK_STREAM = 1,
55    /// Datagram socket
56    SOCK_DGRAM = 2,
57    /// Raw socket
58    SOCK_RAW = 3,
59    /// Reliably-delivered message
60    SOCK_RDM = 4,
61    /// Sequential packet socket
62    SOCK_SEQPACKET = 5,
63    /// Datagram Congestion Control Protocol socket
64    SOCK_DCCP = 6,
65    /// Linux specific way of getting packets at the dev level
66    SOCK_PACKET = 10,
67}

80#[repr(C)]
81#[padding_struct]
82#[derive(Debug, Clone, Copy, Pod)]
83pub struct CUserMsgHdr {
84    /// Pointer to socket address structure
85    pub msg_name: Vaddr,
86    /// Size of socket address
87    pub msg_namelen: i32,
88    /// Scatter/Gather iov array
89    pub msg_iov: Vaddr,
90    /// The # of elements in msg_iov
91    pub msg_iovlen: usize,
92    /// Ancillary data
93    pub msg_control: Vaddr,
94    /// Ancillary data buffer length
95    pub msg_controllen: usize,
96    /// Flags on received message
97    pub msg_flags: u32,
98}
99
100impl CUserMsgHdr {
101    pub fn read_socket_addr_from_user(&self) -> Result<Option<SocketAddr>> {
102        if self.msg_name == 0 {
103            return Ok(None);
104        }
105
106        let socket_addr = read_socket_addr_from_user(self.msg_name, self.msg_namelen as usize)?;
107        Ok(Some(socket_addr))
108    }
109
110    pub fn write_socket_addr_to_user(&self, addr: Option<&SocketAddr>) -> Result<i32> {
111        if self.msg_name == 0 {
112            // The length field will not be touched if the name pointer is NULL.
113            // See <https://elixir.bootlin.com/linux/v6.15.6/source/net/socket.c#L2792>.
114            return Ok(self.msg_namelen);
115        }
116
117        let actual_len = if let Some(addr) = addr {
118            write_socket_addr_with_max_len(addr, self.msg_name, self.msg_namelen)?
119        } else {
120            0
121        };
122        Ok(actual_len)
123    }
124
125    pub fn read_control_messages_from_user(
126        &self,
127        user_space: &CurrentUserSpace,
128    ) -> Result<Vec<ControlMessage>> {
129        if self.msg_control == 0 {
130            return Ok(Vec::new());
131        }
132
133        let mut reader = user_space.reader(self.msg_control, self.msg_controllen)?;
134        let control_messages = ControlMessage::read_all_from(&mut reader)?;
135        Ok(control_messages)
136    }
137
138    pub fn write_control_messages_to_user(
139        &self,
140        control_messages: &[ControlMessage],
141        user_space: &CurrentUserSpace,
142    ) -> Result<u32> {
143        if self.msg_control == 0 {
144            if !control_messages.is_empty() {
145                warn!("setting MSG_CTRUNC is not supported");
146            }
147            // The length field will be set even if the control message pointer is NULL.
148            // See <https://elixir.bootlin.com/linux/v6.15.6/source/net/socket.c#L2807>.
149            return Ok(0);
150        }
151
152        let mut writer = user_space.writer(self.msg_control, self.msg_controllen)?;
153        let write_len = ControlMessage::write_all_to(control_messages, &mut writer) as u32;
154        Ok(write_len)
155    }
156
157    pub fn copy_reader_array_from_user<'a>(
158        &self,
159        user_space: &'a CurrentUserSpace<'a>,
160    ) -> Result<VmReaderArray<'a>> {
161        if self.msg_iovlen > MAX_IO_VECTOR_LENGTH {
162            return_errno_with_message!(Errno::EMSGSIZE, "the I/O vector contains too many buffers");
163        }
164
165        VmReaderArray::from_user_io_vecs(user_space, self.msg_iov, self.msg_iovlen)
166    }
167
168    pub fn copy_writer_array_from_user<'a>(
169        &self,
170        user_space: &'a CurrentUserSpace<'a>,
171    ) -> Result<VmWriterArray<'a>> {
172        if self.msg_iovlen > MAX_IO_VECTOR_LENGTH {
173            return_errno_with_message!(Errno::EMSGSIZE, "the I/O vector contains too many buffers");
174        }
175
176        VmWriterArray::from_user_io_vecs(user_space, self.msg_iov, self.msg_iovlen)
177    }
178}
