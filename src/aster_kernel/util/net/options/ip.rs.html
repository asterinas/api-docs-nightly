13/// Socket options for IP socket.
14///
15/// The raw definitions can be found at:
16/// <https://elixir.bootlin.com/linux/v6.0.19/source/include/uapi/linux/in.h#L94>.
17#[repr(i32)]
18#[derive(Debug, Clone, Copy, TryFromInt)]
19#[expect(non_camel_case_types)]
20#[expect(clippy::upper_case_acronyms)]
21pub enum CIpOptionName {
22    TOS = 1,
23    TTL = 2,
24    HDRINCL = 3,
25    OPTIONS = 4,
26    ROUTER_ALERT = 5,
27    RECVOPTS = 6,
28    RETOPTS = 7,
29    PKTINFO = 8,
30    PKTOPTIONS = 9,
31    MTU_DISCOVER = 10,
32    RECVERR = 11,
33    RECVTTL = 12,
34    RECVTOS = 13,
35    MTU = 14,
36    FREEBIND = 15,
37    IPSEC_POLICY = 16,
38    XFRM_POLICY = 17,
39    PASSSEC = 18,
40    TRANSPARENT = 19,
41    ORIGDSTADDR = 20,
42    MINTTL = 21,
43    NODEFRAG = 22,
44    CHECKSUM = 23,
45    BIND_ADDRESS_NO_PORT = 24,
46    RECVFRAGSIZE = 25,
47    RECVERR_RFC4884 = 26,
48    MULTICAST_IF = 32,
49    MULTICAST_TTL = 33,
50    MULTICAST_LOOP = 34,
51    ADD_MEMBERSHIP = 35,
52    DROP_MEMBERSHIP = 36,
53    UNBLOCK_SOURCE = 37,
54    BLOCK_SOURCE = 38,
55    ADD_SOURCE_MEMBERSHIP = 39,
56    DROP_SOURCE_MEMBERSHIP = 40,
57    MSFILTER = 41,
58    MCAST_JOIN_GROUP = 42,
59    MCAST_BLOCK_SOURCE = 43,
60    MCAST_UNBLOCK_SOURCE = 44,
61    MCAST_LEAVE_GROUP = 45,
62    MCAST_JOIN_SOURCE_GROUP = 46,
63    MCAST_LEAVE_SOURCE_GROUP = 47,
64    MCAST_MSFILTER = 48,
65    MULTICAST_ALL = 49,
66    UNICAST_IF = 50,
67}
68
69pub fn new_ip_option(name: i32) -> Result<Box<dyn RawSocketOption>> {
70    let name = CIpOptionName::try_from(name).map_err(|_| Errno::ENOPROTOOPT)?;
71    match name {
72        CIpOptionName::TOS => Ok(Box::new(Tos::new())),
73        CIpOptionName::TTL => Ok(Box::new(Ttl::new())),
74        CIpOptionName::HDRINCL => Ok(Box::new(Hdrincl::new())),
75        CIpOptionName::RECVERR => Ok(Box::new(Recverr::new())),
76        _ => return_errno_with_message!(Errno::ENOPROTOOPT, "unsupported ip level option"),
77    }
78}
