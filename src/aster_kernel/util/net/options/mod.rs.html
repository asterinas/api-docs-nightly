3//! This module introduces utilities to support Linux get/setsockopt syscalls.
4//!
5//! These two syscalls are used to get/set options for a socket. These options can be at different
6//! socket levels and of different types. To provide a unified interface, the `Socket` trait accepts
7//! a `dyn SocketOption` as a parameter. For each socket option, We define a struct that implements
8//! the `SocketOption` trait in net module.
9//!
10//! However, different socket options may have values of different types. For example, the values can
11//! be u32, C structs, or byte arrays. Furthermore, some values may have different formats in kernel
12//! space and user space. For example, for the option `ReusePort`, the user space may use an i32 while
13//! the kernel space may use a bool.
14//!
15//! We introduce the `RawSocketOption` trait for reading/writing socket options from/to user space. It
16//! can read/write values of different types and can convert the user type to the kernel type when
17//! reading from the user space and vice versa when writing to the user space. The `RawSocketOption`
18//! should not be implemented for a type by hand, and we provide macros to automatically implement the
19//! trait.
20//!
21//! # Example
22//!
23//! Suppose we want to add a new option `TcpNodelay`.
24//!
25//! First, the option should be added in the net module for the TCP socket.
26//!  
27//! ```rust no_run
28//! impl_socket_option!(TcpNodelay(bool));
29//! ```
30//!
31//! Then, we need to implement the `ReadFromUser` and `WriteFromUser` traits for the bool type
32//! in the utils module. These util functions can be shared if multiple options have the value
33//! of same type.
34//!
35//! ```rust compile_fail
36//! impl ReadFromUser for bool {
37//!     // content omitted here
38//! }
39//! impl WriteFromUser for bool {
40//!     // content omitted here
41//! }
42//! ```
43//!
44//! At last, we can implement `RawSocketOption` for `TcpNodelay` so that it can be read/from
45//! user space.
46//!
47//! ```rust no_run
48//! impl_raw_socket_option!(TcpNodeley);
49//! ```
50//!
51//! At the syscall level, the interface is unified for all options and does not need to be modified.
52//!

67pub trait RawSocketOption: SocketOption {
68    fn read_from_user(&mut self, addr: Vaddr, max_len: u32) -> Result<()>;
69
70    fn write_to_user(&self, addr: Vaddr, max_len: &mut u32) -> Result<usize>;
71
72    fn as_sock_option_mut(&mut self) -> &mut dyn SocketOption;
73
74    fn as_sock_option(&self) -> &dyn SocketOption;
75}
76
77/// Impl `RawSocketOption` for a struct which implements `SocketOption`.
78#[macro_export]
79macro_rules! impl_raw_socket_option {
80    ($option:ty) => {
81        impl RawSocketOption for $option {
82            fn read_from_user(&mut self, addr: Vaddr, max_len: u32) -> Result<()> {
83                use $crate::util::net::options::utils::ReadFromUser;
84
85                let input = ReadFromUser::read_from_user(addr, max_len)?;
86                self.set(input);
87                Ok(())
88            }
89
90            fn write_to_user(&self, addr: Vaddr, max_len: &mut u32) -> Result<usize> {
91                use $crate::util::net::options::utils::WriteToUser;
92
93                let output = self.get().unwrap();
94                output.write_to_user(addr, *max_len)
95            }
96
97            fn as_sock_option_mut(&mut self) -> &mut dyn SocketOption {
98                self
99            }
100
101            fn as_sock_option(&self) -> &dyn SocketOption {
102                self
103            }
104        }
105    };
106}

108/// Impl `RawSocketOption` for a struct which is for only `getsockopt` and implements `SocketOption`.
109#[macro_export]
110macro_rules! impl_raw_sock_option_get_only {
111    ($option:ty) => {
112        impl RawSocketOption for $option {
113            fn read_from_user(&mut self, _addr: Vaddr, _max_len: u32) -> Result<()> {
114                return_errno_with_message!(Errno::ENOPROTOOPT, "the option is getter-only");

163pub fn new_raw_socket_option(
164    level: CSocketOptionLevel,
165    name: i32,
166) -> Result<Box<dyn RawSocketOption>> {
167    match level {
168        CSocketOptionLevel::SOL_SOCKET => new_socket_option(name),
169        CSocketOptionLevel::SOL_IP => new_ip_option(name),
170        CSocketOptionLevel::SOL_TCP => new_tcp_option(name),
171        CSocketOptionLevel::SOL_NETLINK => new_netlink_option(name),
172        _ => return_errno_with_message!(Errno::EOPNOTSUPP, "unsupported option level"),
173    }
174}

176/// Socket level.
177///
178/// This can refer to either a socket option or a control message level.
179///
180/// Reference: <https://elixir.bootlin.com/linux/v6.0.9/source/include/linux/socket.h#L343>.
181#[repr(i32)]
182#[derive(Debug, Clone, Copy, TryFromInt, PartialEq, Eq)]
183#[expect(non_camel_case_types)]
184pub enum CSocketOptionLevel {
185    SOL_IP = 0,
186    SOL_SOCKET = 1,
187    SOL_TCP = 6,
188    SOL_UDP = 17,
189    SOL_IPV6 = 41,
190    SOL_RAW = 255,
191    SOL_NETLINK = 270,
192}
