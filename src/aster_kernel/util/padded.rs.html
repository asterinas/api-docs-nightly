3/// Creates a fixed-size byte array out of a byte slice.
4///
5/// # Example
6///
7/// ```
8/// let fixed_size_text: [u8; 128] = padded(b"Hello World");
9/// ```
10///
11/// Without this `padded` utility function,
12/// one would have to write a more lengthy but less efficient version.
13///
14/// ```
15/// let fixed_size_text: [u8; 128] = {
16///    const HELLO: &[u8] = b"Hello World";
17///    let mut buf = [0u8; 128];
18///    buf[..HELLO.len()].copy_from_slice(HELLO);
19///    buf
20/// };
21/// ```
22pub const fn padded<const N: usize>(s: &[u8]) -> [u8; N] {
23    let mut out = [0u8; N];
24    let mut i = 0;
25    while i < s.len() && i < N {
26        out[i] = s[i];
27        i += 1;
28    }
29    out
30}
