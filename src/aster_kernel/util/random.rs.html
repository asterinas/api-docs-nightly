3use rand::{RngCore, SeedableRng, rngs::StdRng};
4use spin::Once;

6use crate::prelude::*;

8static RNG: Once<SpinLock<StdRng>> = Once::new();
9
10/// Fill `dest` with random bytes.
11///
12/// It's cryptographically secure, as documented in [`rand::rngs::StdRng`].
13pub fn getrandom(dst: &mut [u8]) {
14    RNG.get().unwrap().lock().fill_bytes(dst);
15}
16
17pub fn init() {
18    // The seed used to initialize the RNG is required to be secure and unpredictable.
19    let seed = get_random_seed();
20
21    RNG.call_once(|| SpinLock::new(StdRng::from_seed(seed)));
22}

24#[cfg(target_arch = "x86_64")]
25fn get_random_seed() -> <StdRng as SeedableRng>::Seed {
26    use ostd::arch::read_random;
27
28    let mut seed = <StdRng as SeedableRng>::Seed::default();
29
30    // Notes for future refactorings: If hardware randomness cannot be generated (i.e., if
31    // `read_random` fails), we can usually continue with pseudorandomness. However, we should stop
32    // if we are TD guests. For more details, see
33    // <https://intel.github.io/ccc-linux-guest-hardening-docs/security-spec.html#randomness-inside-tdx-guest>.
34    let mut chunks = seed.as_mut().chunks_exact_mut(size_of::<u64>());
35    for chunk in chunks.by_ref() {
36        let src = read_random().expect("`read_random` failed").to_ne_bytes();
37        chunk.copy_from_slice(&src);
38    }
39    let tail = chunks.into_remainder();
40    let n = tail.len();
41    if n > 0 {
42        let src = read_random().expect("`read_random` failed").to_ne_bytes();
43        tail.copy_from_slice(&src[..n]);
44    }
45
46    seed
47}

49#[cfg(not(target_arch = "x86_64"))]
50fn get_random_seed() -> <StdRng as SeedableRng>::Seed {
51    use ostd::arch::boot::DEVICE_TREE;
52
53    let chosen = DEVICE_TREE.get().unwrap().find_node("/chosen").unwrap();
54    chosen
55        .property("rng-seed")
56        .unwrap()
57        .value
58        .try_into()
59        .unwrap()
60}
