16/// A lock-free SPSC FIFO ring buffer backed by a [`Segment<()>`].
17///
18/// The ring buffer supports `push`/`pop` any `T: Pod` items, also
19/// supports `write`/`read` any bytes data based on [`VmReader`]/[`VmWriter`].
20///
21/// The ring buffer returns immediately after processing without any blocking.
22/// The ring buffer can be shared between threads.
23///
24/// # Example
25///
26/// ```
27/// use ostd_pod::Pod;
28/// use ring_buffer::RingBuffer;
29///
30/// #[derive(Pod)]
31/// struct Item {
32///     a: u32,
33///     b: u32,
34/// }
35///
36/// let rb = RingBuffer::<Item>::new(10);
37/// let (producer, consumer) = rb.split();
38///
39/// for i in 0..10 {
40///     producer.push(Item { a: i, b: i }).unwrap();
41/// }
42///
43/// for _ in 0..10 {
44///     let item = consumer.pop().unwrap();
45///     assert_eq!(item.a, item.b);
46/// }
47/// ```
48pub struct RingBuffer<T> {
49    segment: Segment<()>,
50    capacity: usize,
51    tail: AtomicUsize,
52    head: AtomicUsize,
53    phantom: PhantomData<T>,
54}
55
56/// A producer of a [`RingBuffer`].
57pub struct Producer<T, R: Deref<Target = RingBuffer<T>>> {
58    rb: R,
59    phantom: PhantomData<T>,
60}
61/// A consumer of a [`RingBuffer`].
62pub struct Consumer<T, R: Deref<Target = RingBuffer<T>>> {
63    rb: R,
64    phantom: PhantomData<T>,
65}
66
67pub type RbProducer<T> = Producer<T, Arc<RingBuffer<T>>>;
68pub type RbConsumer<T> = Consumer<T, Arc<RingBuffer<T>>>;
69
70impl<T> RingBuffer<T> {
71    const T_SIZE: usize = size_of::<T>();
72
73    /// Creates a new [`RingBuffer`] with the given capacity.
74    pub fn new(capacity: usize) -> Self {
75        assert!(
76            capacity.is_power_of_two(),
77            "capacity must be a power of two"
78        );
79
80        let nframes = capacity
81            .checked_mul(Self::T_SIZE)
82            .unwrap()
83            .div_ceil(PAGE_SIZE);
84        let segment = FrameAllocOptions::new()
85            .zeroed(false)
86            .alloc_segment(nframes)
87            .unwrap();
88
89        Self {
90            segment,
91            capacity,
92            tail: AtomicUsize::new(0),
93            head: AtomicUsize::new(0),
94            phantom: PhantomData,
95        }
96    }
97
98    /// Splits the [`RingBuffer`] into a producer and a consumer.
99    pub fn split(self) -> (RbProducer<T>, RbConsumer<T>) {
100        let producer = Producer {
101            rb: Arc::new(self),
102            phantom: PhantomData,
103        };
104        let consumer = Consumer {
105            rb: Arc::clone(&producer.rb),
106            phantom: PhantomData,
107        };
108        (producer, consumer)
109    }
110
111    /// Gets the capacity of the `RingBuffer`.
112    pub fn capacity(&self) -> usize {
113        self.capacity
114    }
115
116    /// Checks if the `RingBuffer` is empty.
117    pub fn is_empty(&self) -> bool {
118        self.len() == 0
119    }
120
121    /// Checks if the `RingBuffer` is full.
122    pub fn is_full(&self) -> bool {
123        self.free_len() == 0
124    }
125
126    /// Gets the number of items in the `RingBuffer`.
127    pub fn len(&self) -> usize {
128        // Implementation notes: This subtraction only makes sense if either the head or the tail
129        // is considered frozen; if both are volatile, the number of the items may become negative
130        // due to race conditions. This is always true with a `RingBuffer` or a pair of
131        // `RbProducer` and `RbConsumer`.
132        (self.tail() - self.head()).0
133    }
134
135    /// Gets the number of free items in the `RingBuffer`.
136    pub fn free_len(&self) -> usize {
137        self.capacity - self.len()
138    }
139
140    /// Gets the head number of the `RingBuffer`.
141    ///
142    /// This is the number of items read from the ring buffer. The number wraps when crossing
143    /// [`usize`] boundaries.
144    pub fn head(&self) -> Wrapping<usize> {
145        Wrapping(self.head.load(Ordering::Acquire))
146    }
147
148    /// Gets the tail number of the `RingBuffer`.
149    ///
150    /// This is the number of items written into the ring buffer. The number wraps when crossing
151    /// [`usize`] boundaries.
152    pub fn tail(&self) -> Wrapping<usize> {
153        Wrapping(self.tail.load(Ordering::Acquire))
154    }
155
156    /// Clears the `RingBuffer`.
157    pub fn clear(&mut self) {
158        self.tail.store(0, Ordering::Release);
159        self.head.store(0, Ordering::Release);
160    }
161}

163impl<T: Pod> RingBuffer<T> {
164    /// Pushes an item to the `RingBuffer`.
165    ///
166    /// Returns `Some` on success. Returns `None` if
167    /// the ring buffer is full.
168    pub fn push(&mut self, item: T) -> Option<()> {
169        let mut producer = Producer {
170            rb: self,
171            phantom: PhantomData,
172        };
173        producer.push(item)
174    }
175
176    /// Pushes a slice of items to the `RingBuffer`.
177    ///
178    /// Returns `Some` on success, all items are pushed to the ring buffer.
179    /// Returns `None` if the ring buffer is full or cannot fit all items.
180    pub fn push_slice(&mut self, items: &[T]) -> Option<()> {
181        let mut producer = Producer {
182            rb: self,
183            phantom: PhantomData,
184        };
185        producer.push_slice(items)
186    }
187
188    /// Pops an item from the `RingBuffer`.
189    ///
190    /// Returns `Some` with the popped item on success.
191    /// Returns `None` if the ring buffer is empty.
192    pub fn pop(&mut self) -> Option<T> {
193        let mut consumer = Consumer {
194            rb: self,
195            phantom: PhantomData,
196        };
197        consumer.pop()
198    }
199
200    /// Pops a slice of items from the `RingBuffer`.
201    ///
202    /// Returns `Some` on success, all items are popped from the ring buffer.
203    /// Returns `None` if the ring buffer is empty or cannot fill all items.
204    pub fn pop_slice(&mut self, items: &mut [T]) -> Option<()> {
205        let mut consumer = Consumer {
206            rb: self,
207            phantom: PhantomData,
208        };
209        consumer.pop_slice(items)
210    }
211
212    pub(self) fn advance_tail(&self, mut tail: Wrapping<usize>, len: usize) {
213        tail += len;
214        self.tail.store(tail.0, Ordering::Release);
215    }
216
217    pub(self) fn advance_head(&self, mut head: Wrapping<usize>, len: usize) {
218        head += len;
219        self.head.store(head.0, Ordering::Release);
220    }
221
222    pub(self) fn reset_head(&self) {
223        let new_head = self.tail();
224        self.head.store(new_head.0, Ordering::Release);
225    }
226}

253impl<T: Pod, R: Deref<Target = RingBuffer<T>>> Producer<T, R> {
254    const T_SIZE: usize = size_of::<T>();
255
256    /// Pushes an item to the `RingBuffer`.
257    ///
258    /// Returns `Some` on success. Returns `None` if
259    /// the ring buffer is full.
260    pub fn push(&mut self, item: T) -> Option<()> {
261        let rb = &self.rb;
262        if rb.is_full() {
263            return None;
264        }
265
266        let tail = rb.tail();
267        let offset = tail.0 & (rb.capacity - 1);
268        let byte_offset = offset * Self::T_SIZE;
269
270        let mut writer = rb.segment.writer();
271        writer.skip(byte_offset);
272        writer.write_val(&item).unwrap();
273
274        rb.advance_tail(tail, 1);
275        Some(())
276    }
277
278    /// Pushes a slice of items to the `RingBuffer`.
279    ///
280    /// Returns `Some` on success, all items are pushed to the ring buffer.
281    /// Returns `None` if the ring buffer is full or cannot fit all items.
282    pub fn push_slice(&mut self, items: &[T]) -> Option<()> {
283        let rb = &self.rb;
284        let nitems = items.len();
285        if rb.free_len() < nitems {
286            return None;
287        }
288
289        let tail = rb.tail();
290        let offset = tail.0 & (rb.capacity - 1);
291        let byte_offset = offset * Self::T_SIZE;
292
293        if offset + nitems > rb.capacity {
294            // Write into two separate parts
295            rb.segment
296                .write_slice(byte_offset, &items[..rb.capacity - offset])
297                .unwrap();
298            rb.segment
299                .write_slice(0, &items[rb.capacity - offset..])
300                .unwrap();
301        } else {
302            rb.segment.write_slice(byte_offset, items).unwrap();
303        }
304
305        rb.advance_tail(tail, nitems);
306        Some(())
307    }
308
309    // There is no counterpart to `Consumer::skip` and `Consumer::clear`. They do not make sense
310    // for the producer.
311}

313impl<R: Deref<Target = RingBuffer<u8>>> Producer<u8, R> {
314    /// Writes data from the `VmReader` to the `RingBuffer`.
315    ///
316    /// Returns the number of bytes written.
317    pub fn write_fallible(&mut self, reader: &mut dyn MultiRead) -> Result<usize> {
318        self.write_fallible_with_max_len(reader, usize::MAX)
319    }
320
321    /// Writes data from the `VmReader` to the `RingBuffer` with the maximum length.
322    ///
323    /// Returns the number of bytes written.
324    pub fn write_fallible_with_max_len(
325        &mut self,
326        reader: &mut dyn MultiRead,
327        max_len: usize,
328    ) -> Result<usize> {
329        let rb = &self.rb;
330        let free_len = rb.free_len().min(max_len);
331
332        let tail = rb.tail();
333        let offset = tail.0 & (rb.capacity - 1);
334
335        let write_len = if offset + free_len > rb.capacity {
336            // Write into two separate parts
337            let mut write_len = 0;
338
339            let mut writer = rb.segment.writer();
340            writer.skip(offset).limit(rb.capacity - offset);
341            write_len += reader.read(&mut writer)?;
342
343            let mut writer = rb.segment.writer();
344            writer.limit(free_len - (rb.capacity - offset));
345            write_len += reader.read(&mut writer)?;
346
347            write_len
348        } else {
349            let mut writer = rb.segment.writer();
350            writer.skip(offset).limit(free_len);
351            reader.read(&mut writer)?
352        };
353
354        rb.advance_tail(tail, write_len);
355        Ok(write_len)
356    }
357}

370impl<T: Pod, R: Deref<Target = RingBuffer<T>>> Consumer<T, R> {
371    const T_SIZE: usize = size_of::<T>();
372
373    /// Pops an item from the `RingBuffer`.
374    ///
375    /// Returns `Some` with the popped item on success.
376    /// Returns `None` if the ring buffer is empty.
377    pub fn pop(&mut self) -> Option<T> {
378        let rb = &self.rb;
379        if rb.is_empty() {
380            return None;
381        }
382
383        let head = rb.head();
384        let offset = head.0 & (rb.capacity - 1);
385        let byte_offset = offset * Self::T_SIZE;
386
387        let mut reader = rb.segment.reader();
388        reader.skip(byte_offset);
389        let item = reader.read_val::<T>().unwrap();
390
391        rb.advance_head(head, 1);
392        Some(item)
393    }
394
395    /// Pops a slice of items from the `RingBuffer`.
396    ///
397    /// Returns `Some` on success, all items are popped from the ring buffer.
398    /// Returns `None` if the ring buffer is empty or cannot fill all items.
399    pub fn pop_slice(&mut self, items: &mut [T]) -> Option<()> {
400        let rb = &self.rb;
401        let nitems = items.len();
402        if rb.len() < nitems {
403            return None;
404        }
405
406        let head = rb.head();
407        let offset = head.0 & (rb.capacity - 1);
408        let byte_offset = offset * Self::T_SIZE;
409
410        if offset + nitems > rb.capacity {
411            // Read from two separate parts
412            rb.segment
413                .read_slice(byte_offset, &mut items[..rb.capacity - offset])
414                .unwrap();
415            rb.segment
416                .read_slice(0, &mut items[rb.capacity - offset..])
417                .unwrap();
418        } else {
419            rb.segment.read_slice(byte_offset, items).unwrap();
420        }
421
422        rb.advance_head(head, nitems);
423        Some(())
424    }
425
426    /// Skips `count` items in the `RingBuffer`.
427    ///
428    /// In other words, `count` items are popped from the `RingBuffer` and discarded.
429    ///
430    /// # Panics
431    ///
432    /// This method will panic if the number of the available items to pop is less than `count`.
433    pub fn skip(&mut self, count: usize) {
434        let rb = &self.rb;
435        let len = rb.len();
436        assert!(len >= count);
437
438        let head = rb.head();
439        rb.advance_head(head, count);
440    }
441
442    /// Clears the `RingBuffer`.
443    ///
444    /// In other words, all items are popped from the `RingBuffer` and discarded.
445    pub fn clear(&mut self) {
446        self.rb.reset_head();
447    }
448}

450impl<R: Deref<Target = RingBuffer<u8>>> Consumer<u8, R> {
451    /// Reads data from the `VmWriter` to the `RingBuffer`.
452    ///
453    /// Returns the number of bytes read.
454    pub fn read_fallible(&mut self, writer: &mut dyn MultiWrite) -> Result<usize> {
455        self.read_fallible_with_max_len(writer, usize::MAX)
456    }
457
458    /// Reads data from the `VmWriter` to the `RingBuffer` with the maximum length.
459    ///
460    /// Returns the number of bytes read.
461    pub fn read_fallible_with_max_len(
462        &mut self,
463        writer: &mut dyn MultiWrite,
464        max_len: usize,
465    ) -> Result<usize> {
466        let rb = &self.rb;
467        let len = rb.len().min(max_len);
468
469        let head = rb.head();
470        let offset = head.0 & (rb.capacity - 1);
471
472        let read_len = if offset + len > rb.capacity {
473            // Read from two separate parts
474            let mut read_len = 0;
475
476            let mut reader = rb.segment.reader();
477            reader.skip(offset).limit(rb.capacity - offset);
478            read_len += writer.write(&mut reader)?;
479
480            let mut reader = rb.segment.reader();
481            reader.limit(len - (rb.capacity - offset));
482            read_len += writer.write(&mut reader)?;
483
484            read_len
485        } else {
486            let mut reader = rb.segment.reader();
487            reader.skip(offset).limit(len);
488            writer.write(&mut reader)?
489        };
490
491        rb.advance_head(head, read_len);
492        Ok(read_len)
493    }
494}

513    #[ktest]
514    fn basics() {
515        let mut rb = RingBuffer::<i32>::new(4);
516        rb.push(-100).unwrap();
517        rb.push_slice(&[-1]).unwrap();
518        assert_eq!(rb.len(), 2);
519
520        let mut popped = [0i32; 2];
521        rb.pop_slice(&mut popped).unwrap();
522        assert_eq!(popped, [-100i32, -1]);
523        assert!(rb.is_empty());
524
525        rb.push_slice(&[i32::MAX, 1, -2, 100]).unwrap();
526        assert!(rb.is_full());
527
528        let popped = rb.pop();
529        assert_eq!(popped, Some(i32::MAX));
530        assert_eq!(rb.free_len(), 1);
531
532        let mut popped = [0i32; 3];
533        rb.pop_slice(&mut popped).unwrap();
534        assert_eq!(popped, [1i32, -2, 100]);
535        assert_eq!(rb.free_len(), 4);
536    }

538    #[ktest]
539    fn write_read_one() {
540        let rb = RingBuffer::<u8>::new(1);
541
542        let (mut prod, mut cons) = rb.split();
543        assert_eq!(prod.capacity(), 1);
544        assert_eq!(cons.capacity(), 1);
545
546        assert!(cons.pop().is_none());
547        assert!(prod.push(1).is_some());
548        assert!(prod.is_full());
549
550        assert!(prod.push(2).is_none());
551        assert!(prod.push_slice(&[2]).is_none());
552        assert_eq!(cons.pop().unwrap(), 1u8);
553        assert!(cons.is_empty());
554
555        let input = [u8::MAX];
556        assert_eq!(
557            prod.write_fallible(&mut reader_from(input.as_slice()))
558                .unwrap(),
559            1
560        );
561        assert_eq!(
562            prod.write_fallible(&mut reader_from(input.as_slice()))
563                .unwrap(),
564            0
565        );
566        assert_eq!(prod.len(), 1);
567
568        let mut output = [0u8];
569        assert_eq!(
570            cons.read_fallible(&mut writer_from(output.as_mut_slice()))
571                .unwrap(),
572            1
573        );
574        assert_eq!(
575            cons.read_fallible(&mut writer_from(output.as_mut_slice()))
576                .unwrap(),
577            0
578        );
579        assert_eq!(cons.free_len(), 1);
580
581        assert_eq!(output, input);
582    }

584    #[ktest]
585    fn write_read_all() {
586        let rb = RingBuffer::<u8>::new(4 * PAGE_SIZE);
587        assert_eq!(rb.capacity(), 4 * PAGE_SIZE);
588
589        let (mut prod, mut cons) = rb.split();
590        prod.push(u8::MIN).unwrap();
591        assert_eq!(cons.pop().unwrap(), u8::MIN);
592
593        prod.push_slice(&[u8::MAX]).unwrap();
594        let mut popped = [0u8];
595        cons.pop_slice(&mut popped).unwrap();
596        assert_eq!(popped, [u8::MAX]);
597
598        let step = 128;
599        let mut input = vec![0u8; step];
600        for i in (0..4 * PAGE_SIZE).step_by(step) {
601            input.fill(i as _);
602            let write_len = prod
603                .write_fallible(&mut reader_from(input.as_slice()))
604                .unwrap();
605            assert_eq!(write_len, step);
606        }
607        assert!(cons.is_full());
608
609        let mut output = vec![0u8; step];
610        for i in (0..4 * PAGE_SIZE).step_by(step) {
611            let read_len = cons
612                .read_fallible(&mut writer_from(output.as_mut_slice()))
613                .unwrap();
614            assert_eq!(read_len, step);
615            assert_eq!(output[0], i as u8);
616            assert_eq!(output[step - 1], i as u8);
617        }
618        assert!(prod.is_empty());
619    }
