3use core::sync::atomic::{AtomicUsize, Ordering};

5use keyable_arc::KeyableWeak;
6use ostd::sync::LocalIrqDisabled;

11/// A subject that notifies interesting events to registered observers.
12///
13/// This type does not have any inner locks. Therefore, users need to maintain an outer lock to
14/// obtain a mutable reference. Consequently, observers can break the atomic mode as long as the
15/// outer lock also permits it.
16pub struct Subject<E: Events>(BTreeSet<KeyableWeak<dyn Observer<E>>>);
17
18impl<E: Events> Subject<E> {
19    /// Creates an empty subject.
20    pub const fn new() -> Self {
21        Self(BTreeSet::new())
22    }
23
24    /// Registers an observer.
25    ///
26    /// A registered observer will get notified through its `on_events` method.
27    pub fn register_observer(&mut self, observer: Weak<dyn Observer<E>>) {
28        self.0.insert(KeyableWeak::from(observer));
29    }
30
31    /// Unregisters an observer.
32    ///
33    /// If such an observer is found, then the registered observer will be
34    /// removed from the set and this method will return `true`. Otherwise,
35    /// a `false` will be returned.
36    pub fn unregister_observer(&mut self, observer: &Weak<dyn Observer<E>>) -> bool {
37        self.0.remove(&KeyableWeak::from(observer.clone()))
38    }
39
40    /// Notifies events to all registered observers.
41    ///
42    /// It will remove the observers which have been freed.
43    pub fn notify_observers(&mut self, events: &E) {
44        self.0.retain(|observer| {
45            if let Some(observer) = observer.upgrade() {
46                observer.on_events(events);
47                true
48            } else {
49                false
50            }
51        });
52    }
53}
54
55impl<E: Events> Default for Subject<E> {
56    fn default() -> Self {
57        Self::new()
58    }
59}

61/// A synchronized subject that notifies interesting events to registered observers.
62///
63/// This type can be used via an immutable reference across threads. To enable this, the type
64/// maintains registered observers in a spin lock. As a result, when called on events, all
65/// registered observers should not break atomic mode. See also [`Subject`] if the condition may be
66/// violated.
67pub struct SyncSubject<E: Events, F: EventsFilter<E> = ()> {
68    // A table that maintains all interesting observers.
69    observers: SpinLock<BTreeMap<KeyableWeak<dyn Observer<E>>, F>, LocalIrqDisabled>,
70    // To reduce lock contentions, we maintain a counter for the size of the table
71    num_observers: AtomicUsize,
72}
73
74impl<E: Events, F: EventsFilter<E>> SyncSubject<E, F> {
75    /// Creates an empty subject.
76    pub const fn new() -> Self {
77        Self {
78            observers: SpinLock::new(BTreeMap::new()),
79            num_observers: AtomicUsize::new(0),
80        }
81    }
82
83    /// Registers an observer.
84    ///
85    /// A registered observer will get notified through its `on_events` method.
86    /// If events `filter` is provided, only filtered events will notify the observer.
87    ///
88    /// If the given observer has already been registered, then its registered events
89    /// filter will be updated.
90    pub fn register_observer(&self, observer: Weak<dyn Observer<E>>, filter: F) {
91        let mut observers = self.observers.lock();
92        let is_new = {
93            let observer: KeyableWeak<dyn Observer<E>> = observer.into();
94            observers.insert(observer, filter).is_none()
95        };
96        if is_new {
97            // This `Acquire` pairs with the `Release` in `notify_observers`.
98            self.num_observers.fetch_add(1, Ordering::Acquire);
99        }
100    }
101
102    /// Unregisters an observer.
103    ///
104    /// If such an observer is found, then the registered observer will be
105    /// removed from the subject and returned as the return value. Otherwise,
106    /// a `None` will be returned.
107    pub fn unregister_observer(
108        &self,
109        observer: &Weak<dyn Observer<E>>,
110    ) -> Option<Weak<dyn Observer<E>>> {
111        let observer: KeyableWeak<dyn Observer<E>> = observer.clone().into();
112        let mut observers = self.observers.lock();
113        let observer = observers
114            .remove_entry(&observer)
115            .map(|(observer, _)| observer.into());
116        if observer.is_some() {
117            self.num_observers.fetch_sub(1, Ordering::Relaxed);
118        }
119        observer
120    }
121
122    /// Notifies events to all registered observers.
123    ///
124    /// It will remove the observers which have been freed.
125    pub fn notify_observers(&self, events: &E) {
126        // Fast path.
127        //
128        // Note: This must use `Release`, which pairs with `Acquire` in `register_observer`, to
129        // ensure that even if this fast path is used, a concurrently registered observer will see
130        // the event we want to notify.
131        if self.num_observers.fetch_add(0, Ordering::Release) == 0 {
132            return;
133        }
134
135        // Slow path: broadcast the new events to all observers.
136        let mut num_freed = 0;
137        let mut observers = self.observers.lock();
138        observers.retain(|observer, filter| {
139            if let Some(observer) = observer.upgrade() {
140                if filter.filter(events) {
141                    observer.on_events(events);
142                }
143                true
144            } else {
145                num_freed += 1;
146                false
147            }
148        });
149        if num_freed > 0 {
150            self.num_observers.fetch_sub(num_freed, Ordering::Relaxed);
151        }
152    }
153}
154
155impl<E: Events> Default for SyncSubject<E> {
156    fn default() -> Self {
157        Self::new()
158    }
159}
