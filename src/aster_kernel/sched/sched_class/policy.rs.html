12/// The User-chosen scheduling policy.
13///
14/// The scheduling policies are specified by the user, usually through its priority.
15#[derive(Clone, Copy, Debug, Eq, Ord, PartialEq, PartialOrd)]
16pub enum SchedPolicy {
17    #[expect(dead_code)]
18    Stop,
19    RealTime {
20        rt_prio: RealTimePriority,
21        rt_policy: RealTimePolicy,
22    },
23    Fair(Nice),
24    Idle,
25}
26
27impl Default for SchedPolicy {
28    fn default() -> Self {
29        Self::Fair(Nice::default())
30    }
31}

33/// The Linux scheduling policy code.
34///
35/// Reference: <https://elixir.bootlin.com/linux/v6.17.7/source/include/uapi/linux/sched.h#L112>.
36#[repr(u32)]
37#[derive(Clone, Copy, Debug, Eq, Ord, PartialEq, PartialOrd, TryFromInt)]
38pub enum LinuxSchedPolicy {
39    Normal = 0,
40    Fifo = 1,
41    RoundRobin = 2,
42    Batch = 3, // Not supported.
43    Iso = 4,   // Reserved but not implemented yet on Linux.
44    Idle = 5,
45    Deadline = 6, // Not supported.
46    Ext = 7,      // Not supported.
47}
48
49/// Projects internal scheduling policies onto Linux's user-visible policy codes.
50///
51/// This projection is many-to-one: `Stop` is reported as `Fifo`, and the
52/// kernel-only `Idle` policy is reported as `Normal`.
53impl From<SchedPolicy> for LinuxSchedPolicy {
54    fn from(policy: SchedPolicy) -> Self {
55        match policy {
56            SchedPolicy::Stop => LinuxSchedPolicy::Fifo,
57            SchedPolicy::RealTime {
58                rt_policy: RealTimePolicy::Fifo,
59                ..
60            } => LinuxSchedPolicy::Fifo,
61            SchedPolicy::RealTime {
62                rt_policy: RealTimePolicy::RoundRobin { .. },
63                ..
64            } => LinuxSchedPolicy::RoundRobin,
65            SchedPolicy::Fair(_) | SchedPolicy::Idle => LinuxSchedPolicy::Normal,
66        }
67    }
68}

70#[repr(u8)]
71#[derive(Clone, Copy, Debug, Eq, Ord, PartialEq, PartialOrd, TryFromInt)]
72pub(super) enum SchedPolicyKind {
73    Stop = 0,
74    RealTime = 1,
75    Fair = 2,
76    Idle = 3,
77}
78
79impl SchedPolicy {
80    pub(super) fn kind(&self) -> SchedPolicyKind {
81        match self {
82            SchedPolicy::Stop => SchedPolicyKind::Stop,
83            SchedPolicy::RealTime { .. } => SchedPolicyKind::RealTime,
84            SchedPolicy::Fair(_) => SchedPolicyKind::Fair,
85            SchedPolicy::Idle => SchedPolicyKind::Idle,
86        }
87    }
88}

101#[derive(Debug)]
102pub(super) struct SchedPolicyState {
103    kind: AtomicSchedPolicyKind,
104    policy: SpinLock<SchedPolicy>,
105}
106
107impl SchedPolicyState {
108    pub fn new(policy: SchedPolicy) -> Self {
109        Self {
110            kind: AtomicSchedPolicyKind::new(policy.kind()),
111            policy: SpinLock::new(policy),
112        }
113    }
114
115    pub fn kind(&self) -> SchedPolicyKind {
116        self.kind.load(Relaxed)
117    }
118
119    pub fn get(&self) -> SchedPolicy {
120        *self.policy.disable_irq().lock()
121    }
122
123    pub fn set(&self, mut policy: SchedPolicy, update: impl FnOnce(SchedPolicy)) {
124        let mut this = self.policy.disable_irq().lock();
125
126        // Keep the old base slice factor if the new policy doesn't specify one.
127        if let (
128            SchedPolicy::RealTime {
129                rt_policy:
130                    RealTimePolicy::RoundRobin {
131                        base_slice_factor: slot,
132                    },
133                ..
134            },
135            SchedPolicy::RealTime {
136                rt_policy: RealTimePolicy::RoundRobin { base_slice_factor },
137                ..
138            },
139        ) = (*this, &mut policy)
140        {
141            *base_slice_factor = slot.or(*base_slice_factor);
142        }
143
144        update(policy);
145        self.kind.store(policy.kind(), Relaxed);
146        *this = policy;
147    }
148
149    pub fn update<T>(&self, update: impl FnOnce(&mut SchedPolicy) -> T) -> T {
150        update(&mut self.policy.disable_irq().lock())
151    }
152}
