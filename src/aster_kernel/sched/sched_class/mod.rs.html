47pub fn init() {
48    let scheduler = Box::leak(Box::new(ClassScheduler::new()));
49
50    // Inject the scheduler into the ostd for actual scheduling work.
51    inject_scheduler(scheduler);
52
53    // Set the scheduler into the system for statistics.
54    // We set this after injecting the scheduler into ostd,
55    // so that the loadavg statistics are updated after the scheduler is used.
56    set_stats_from_scheduler(scheduler);
57}

63/// Represents the middle layer between scheduling classes and generic scheduler
64/// traits. It consists of all the sets of run queues for CPU cores. Other global
65/// information may also be stored here.
66pub struct ClassScheduler {
67    /// The per-CPU runqueues.
68    ///
69    /// We use the `LocalIrqDisabled` marker for this spinlock to ensure local IRQs are always disabled,
70    /// preventing potential deadlocks due to the fact that
71    /// the runqueues may be accessed in both the task and interrupt context (L1 and L2).
72    rqs: Box<[SpinLock<PerCpuClassRqSet, LocalIrqDisabled>]>,
73    last_chosen_cpu: AtomicCpuId,
74}
75
76/// Represents the run queue for each CPU core. It stores a list of run queues for
77/// scheduling classes in its corresponding CPU core. The current task of this CPU
78/// core is also stored in this structure.
79struct PerCpuClassRqSet {
80    stop: stop::StopClassRq,
81    real_time: real_time::RealTimeClassRq,
82    fair: fair::FairClassRq,
83    idle: idle::IdleClassRq,
84    current: Option<(SchedEntity, CurrentRuntime)>,
85}
86
87/// Stores the runtime information of the current task.
88///
89/// This is used to calculate the time slice of the current task.
90///
91/// This struct is independent of the current `Arc<Task>` instead encapsulating the
92/// task, because the scheduling class implementations use `CurrentRuntime` and
93/// `SchedAttr` only.
94struct CurrentRuntime {
95    start: u64,
96    delta: u64,
97    period_delta: u64,
98}
99
100impl CurrentRuntime {
101    fn new() -> Self {
102        CurrentRuntime {
103            start: sched_clock(),
104            delta: 0,
105            period_delta: 0,
106        }
107    }
108
109    fn update(&mut self) {
110        let now = sched_clock();
111        self.delta = now - core::mem::replace(&mut self.start, now);
112        self.period_delta += self.delta;
113    }
114}

116/// The run queue for scheduling classes (the main trait). Scheduling classes
117/// should implement this trait to function as expected.
118trait SchedClassRq: Send + fmt::Debug {
119    /// Enqueues a task into the run queue.
120    fn enqueue(&mut self, task: Arc<Task>, flags: Option<EnqueueFlags>);
121
122    /// Returns the number of threads in the run queue.
123    fn len(&self) -> usize;
124
125    /// Checks if the run queue is empty.
126    fn is_empty(&self) -> bool {
127        self.len() == 0
128    }
129
130    /// Picks the next task for running.
131    fn pick_next(&mut self) -> Option<Arc<Task>>;
132
133    /// Update the information of the current task.
134    ///
135    /// The return value of this method indicates whether there is another task
136    /// **in this run queue** to replace the current one.
137    fn update_current(&mut self, rt: &CurrentRuntime, attr: &SchedAttr, flags: UpdateFlags)
138    -> bool;
139}

141/// The scheduling attribute for a thread.
142///
143/// This is used to store the scheduling policy and runtime parameters for each
144/// scheduling class.
145#[derive(Debug)]
146pub struct SchedAttr {
147    policy: SchedPolicyState,
148    last_cpu: AtomicCpuId,
149    real_time: real_time::RealTimeAttr,
150    fair: fair::FairAttr,
151}
152
153impl SchedAttr {
154    /// Constructs a new `SchedAttr` with the given scheduling policy.
155    pub fn new(policy: SchedPolicy) -> Self {
156        Self {
157            policy: SchedPolicyState::new(policy),
158            last_cpu: AtomicCpuId::default(),
159            real_time: {
160                let (prio, policy) = match policy {
161                    SchedPolicy::RealTime { rt_prio, rt_policy } => (rt_prio.get(), rt_policy),
162                    _ => (real_time::RealTimePriority::MAX.get(), Default::default()),
163                };
164                real_time::RealTimeAttr::new(prio, policy)
165            },
166            fair: fair::FairAttr::new(match policy {
167                SchedPolicy::Fair(nice) => nice,
168                _ => Nice::default(),
169            }),
170        }
171    }
172
173    /// Retrieves the current scheduling policy of the thread.
174    pub fn policy(&self) -> SchedPolicy {
175        self.policy.get()
176    }
177
178    fn policy_kind(&self) -> SchedPolicyKind {
179        self.policy.kind()
180    }
181
182    /// Updates the scheduling policy of the thread.
183    ///
184    /// Specifically for real-time policies, if the new policy doesn't
185    /// specify a base slice factor for RR, the old one will be kept.
186    pub fn set_policy(&self, policy: SchedPolicy) {
187        self.policy.set(policy, |policy| match policy {
188            SchedPolicy::RealTime { rt_prio, rt_policy } => {
189                self.real_time.update(rt_prio.get(), rt_policy);
190            }
191            SchedPolicy::Fair(nice) => self.fair.update(nice),
192            _ => {}
193        });
194    }
195
196    pub fn update_policy<T>(&self, f: impl FnOnce(&mut SchedPolicy) -> T) -> T {
197        self.policy.update(|policy| {
198            let ret = f(policy);
199            match *policy {
200                SchedPolicy::RealTime { rt_prio, rt_policy } => {
201                    self.real_time.update(rt_prio.get(), rt_policy);
202                }
203                SchedPolicy::Fair(nice) => self.fair.update(nice),
204                _ => {}
205            }
206            ret
207        })
208    }
209
210    fn last_cpu(&self) -> Option<CpuId> {
211        self.last_cpu.get()
212    }
213
214    fn set_last_cpu(&self, cpu_id: CpuId) {
215        self.last_cpu.set_anyway(cpu_id);
216    }
217}

219impl Scheduler for ClassScheduler {
220    fn enqueue(&self, task: Arc<Task>, flags: EnqueueFlags) -> Option<CpuId> {
221        let thread = task.as_thread()?.clone();
222
223        let (still_in_rq, cpu) = {
224            let selected_cpu_id = self.select_cpu(&thread, flags);
225
226            if let Err(task_cpu_id) = task.cpu().set_if_is_none(selected_cpu_id) {
227                debug_assert!(flags != EnqueueFlags::Spawn);
228                (true, task_cpu_id)
229            } else {
230                (false, selected_cpu_id)
231            }
232        };
233
234        let mut rq = self.rqs[cpu.as_usize()].lock();
235
236        // Note: call set_if_is_none again to prevent a race condition.
237        if still_in_rq && task.cpu().set_if_is_none(cpu).is_err() {
238            return None;
239        }
240
241        // Preempt if the new task has a higher priority.
242        let should_preempt = rq
243            .current
244            .as_ref()
245            .is_none_or(|((_, rq_current_thread), _)| {
246                thread.sched_attr().policy() < rq_current_thread.sched_attr().policy()
247            });
248
249        thread.sched_attr().set_last_cpu(cpu);
250        rq.enqueue_entity((task, thread), Some(flags));
251
252        should_preempt.then_some(cpu)
253    }
254
255    fn mut_local_rq_with(&self, f: &mut dyn FnMut(&mut dyn LocalRunQueue)) {
256        let guard = disable_local();
257        let mut lock = self.rqs[guard.current_cpu().as_usize()].lock();
258        f(&mut *lock)
259    }
260
261    fn local_rq_with(&self, f: &mut dyn FnMut(&dyn LocalRunQueue)) {
262        let guard = disable_local();
263        f(&*self.rqs[guard.current_cpu().as_usize()].lock())
264    }
265}

267impl ClassScheduler {
268    pub fn new() -> Self {
269        let class_rq = |cpu| {
270            SpinLock::new(PerCpuClassRqSet {
271                stop: stop::StopClassRq::new(),
272                real_time: real_time::RealTimeClassRq::new(cpu),
273                fair: fair::FairClassRq::new(cpu),
274                idle: idle::IdleClassRq::new(),
275                current: None,
276            })
277        };
278        ClassScheduler {
279            rqs: all_cpus().map(class_rq).collect(),
280            last_chosen_cpu: AtomicCpuId::default(),
281        }
282    }
283
284    // TODO: Implement a better algorithm and replace the current naive implementation.
285    fn select_cpu(&self, thread: &Thread, flags: EnqueueFlags) -> CpuId {
286        if let Some(last_cpu) = thread.sched_attr().last_cpu() {
287            return last_cpu;
288        }
289        debug_assert!(flags == EnqueueFlags::Spawn);
290
291        let guard = disable_local();
292
293        let mut selected = guard.current_cpu();
294        let mut minimum_load = u32::MAX;
295
296        // Set `selected` as `candidate` if the candidate's load is smaller.
297        let test_candidate = |candidate: CpuId| {
298            let PerCpuLoadStats { queue_len, .. } =
299                self.rqs[candidate.as_usize()].lock().load_stats();
300            let load = queue_len;
301            if load < minimum_load {
302                minimum_load = load;
303                selected = candidate;
304            }
305        };
306
307        let affinity = thread.atomic_cpu_affinity().load(Ordering::Relaxed);
308        match self.last_chosen_cpu.get() {
309            Some(cpu) => {
310                // Perform a round-robin selection starting after the last chosen CPU.
311                //
312                // It still checks every CPU in the affinity set to find the one with the
313                // minimum load, but avoids selecting the same CPU again in case of a tie.
314                Self::cycle_after(cpu, &affinity).for_each(test_candidate)
315            }
316            None => affinity.iter().for_each(test_candidate),
317        }
318
319        self.last_chosen_cpu.set_anyway(selected);
320        selected
321    }
322
323    /// Returns a cycling iterator over the CPUs in the [`CpuSet`], starting *after*
324    /// the given [`CpuId`].
325    ///
326    /// The iteration order is ascending up to the wrapping point, after which it
327    /// continues from the first CPU in the set in ascending order again.
328    ///
329    /// If the given [`CpuId`] is in the set, it will be the last element yielded.
330    fn cycle_after(cpu: CpuId, cpu_set: &CpuSet) -> impl Iterator<Item = CpuId> + '_ {
331        cpu_set
332            .iter_in((Bound::Excluded(cpu), Bound::Unbounded))
333            .chain(cpu_set.iter_in(..=cpu))
334    }
335}

337impl PerCpuClassRqSet {
338    fn pick_next_entity(&mut self) -> Option<SchedEntity> {
339        (self.stop.pick_next())
340            .or_else(|| self.real_time.pick_next())
341            .or_else(|| self.fair.pick_next())
342            .or_else(|| self.idle.pick_next())
343            .and_then(|task| {
344                let thread = task.as_thread()?.clone();
345                Some((task, thread))
346            })
347    }
348
349    fn enqueue_entity(&mut self, (task, thread): SchedEntity, flags: Option<EnqueueFlags>) {
350        match thread.sched_attr().policy_kind() {
351            SchedPolicyKind::Stop => self.stop.enqueue(task, flags),
352            SchedPolicyKind::RealTime => self.real_time.enqueue(task, flags),
353            SchedPolicyKind::Fair => self.fair.enqueue(task, flags),
354            SchedPolicyKind::Idle => self.idle.enqueue(task, flags),
355        }
356    }
357
358    fn load_stats(&self) -> PerCpuLoadStats {
359        let queue_len = (self.stop.len() + self.real_time.len() + self.fair.len()) as u32;
360        let is_idle = match &self.current {
361            Some(((_, thread), _)) => thread.sched_attr().policy_kind() == SchedPolicyKind::Idle,
362            None => true,
363        };
364        PerCpuLoadStats { queue_len, is_idle }
365    }
366}

368impl LocalRunQueue for PerCpuClassRqSet {
369    fn current(&self) -> Option<&Arc<Task>> {
370        self.current.as_ref().map(|((task, _), _)| task)
371    }
372
373    fn try_pick_next(&mut self) -> Option<&Arc<Task>> {
374        self.pick_next_entity().and_then(|next| {
375            // We guarantee that a task can appear at once in a `PerCpuClassRqSet`. So, the `next` cannot be the same
376            // as the current task here.
377            if let Some((old, _)) = self.current.replace((next, CurrentRuntime::new())) {
378                self.enqueue_entity(old, None);
379            }
380            self.current.as_ref().map(|((task, _), _)| task)
381        })
382    }
383
384    fn update_current(&mut self, flags: UpdateFlags) -> bool {
385        let (should_preempt, mut lookahead) = if let Some(((_, cur), rt)) = &mut self.current {
386            rt.update();
387            let attr = &cur.sched_attr();
388
389            match attr.policy_kind() {
390                SchedPolicyKind::Stop => (self.stop.update_current(rt, attr, flags), 0),
391                SchedPolicyKind::RealTime => (self.real_time.update_current(rt, attr, flags), 1),
392                SchedPolicyKind::Fair => (self.fair.update_current(rt, attr, flags), 2),
393                SchedPolicyKind::Idle => (self.idle.update_current(rt, attr, flags), 3),
394            }
395        } else {
396            (false, 4)
397        };
398
399        if matches!(flags, UpdateFlags::Wait | UpdateFlags::Exit) {
400            lookahead = 4;
401        }
402
403        should_preempt
404            || (lookahead >= 1 && !self.stop.is_empty())
405            || (lookahead >= 2 && !self.real_time.is_empty())
406            || (lookahead >= 3 && !self.fair.is_empty())
407            || (lookahead >= 4 && !self.idle.is_empty())
408    }
409
410    fn dequeue_current(&mut self) -> Option<Arc<Task>> {
411        self.current.take().map(|((cur_task, _), _)| {
412            cur_task.schedule_info().cpu.set_to_none();
413            cur_task
414        })
415    }
416}

418/// Holds per-CPU load information.
419struct PerCpuLoadStats {
420    /// The length of the run queue (excluding the idle task).
421    queue_len: u32,
422    /// If the CPU is currently idle.
423    ///
424    /// A CPU is said to be idle when it is running the idle task, or it is not
425    /// running any task at all. The latter case is very unlikely to happen
426    /// (almost a bug if it happens) as the idle task should always be runnable.
427    is_idle: bool,
428}
429
430impl SchedulerStats for ClassScheduler {
431    fn nr_queued_and_running(&self) -> (u32, u32) {
432        self.rqs.iter().fold((0, 0), |(queued, running), rq| {
433            let PerCpuLoadStats { queue_len, is_idle } = rq.lock().load_stats();
434            (queued + queue_len, running + u32::from(!is_idle))
435        })
436    }
437}
