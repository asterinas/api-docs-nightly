23pub type RealTimePriority = RangedU8<1, 99>;

25#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
26pub enum RealTimePolicy {
27    Fifo,
28    RoundRobin {
29        base_slice_factor: Option<NonZero<u32>>,
30    },
31}
32
33impl Default for RealTimePolicy {
34    fn default() -> Self {
35        Self::RoundRobin {
36            base_slice_factor: None,
37        }
38    }
39}
40
41impl RealTimePolicy {
42    fn to_time_slice(self) -> u64 {
43        match self {
44            RealTimePolicy::RoundRobin { base_slice_factor } => {
45                base_slice_clocks()
46                    * base_slice_factor
47                        .map_or(DEFAULT_BASE_SLICE_FACTOR, |factor| u64::from(factor.get()))
48            }
49            RealTimePolicy::Fifo => 0,
50        }
51    }
52}

54/// The scheduling attribute for the REAL-TIME scheduling class.
55///
56/// This structure provides not-only the priority of the thread,
57/// but also the time slice for the thread, measured in [`sched_clock`]s.
58///
59/// - If the time slice is not set, the thread is considered to be a FIFO
60///   thread, and will be executed to its end if there no thread with a
61///   lower priority.
62/// - If the time slice is set, the thread is considered to be an RR
63///   (round-robin) thread, and will be executed for the time slice, and
64///   then it will be put back to the inactive array.
65///
66/// [`sched_clock`]: super::sched_clock
67#[derive(Debug)]
68pub struct RealTimeAttr {
69    prio: AtomicU8,
70    time_slice: AtomicU64, // 0 for SCHED_FIFO; other for SCHED_RR
71}
72
73const DEFAULT_BASE_SLICE_FACTOR: u64 = 20;
74
75impl RealTimeAttr {
76    pub fn new(prio: u8, policy: RealTimePolicy) -> Self {
77        RealTimeAttr {
78            prio: prio.into(),
79            time_slice: AtomicU64::new(policy.to_time_slice()),
80        }
81    }
82
83    pub fn update(&self, prio: u8, policy: RealTimePolicy) {
84        self.prio.store(prio, Relaxed);
85        self.time_slice.store(policy.to_time_slice(), Relaxed);
86    }
87}

89struct PrioArray {
90    map: BitArr![for 100],
91    queue: [VecDeque<Arc<Task>>; 100],
92}
93
94impl core::fmt::Debug for PrioArray {
95    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
96        f.debug_struct("PrioArray")
97            .field_with("map", |f| {
98                f.debug_list().entries(self.map.iter_ones()).finish()
99            })
100            .field_with("queue", |f| {
101                f.debug_list()
102                    .entries(
103                        (self.queue.iter().flatten())
104                            .map(|task| task.as_thread().unwrap().sched_attr()),
105                    )
106                    .finish()
107            })
108            .finish()
109    }
110}
111
112impl PrioArray {
113    fn enqueue(&mut self, thread: Arc<Task>, prio: u8) {
114        let queue = &mut self.queue[usize::from(prio)];
115        let is_empty = queue.is_empty();
116        queue.push_back(thread);
117        if is_empty {
118            self.map.set(usize::from(prio), true);
119        }
120    }
121
122    fn pop(&mut self) -> Option<Arc<Task>> {
123        let mut iter = self.map.iter_ones();
124        let prio = iter.next()? as u8;
125
126        let queue = &mut self.queue[usize::from(prio)];
127        let thread = queue.pop_front().unwrap();
128
129        if queue.is_empty() {
130            self.map.set(usize::from(prio), false);
131        }
132        Some(thread)
133    }
134
135    fn peek_prio(&self) -> Option<u8> {
136        let prio = self.map.iter_ones().next()?;
137        Some(prio as u8)
138    }
139}

141/// The per-cpu run queue for the REAL-TIME scheduling class.
142///
143/// The REAL-TIME scheduling class is implemented as a classic O(1)
144/// priority algorithm.
145///
146/// It uses a bit array to track which priority levels have runnable
147/// threads, and a vector of queues to store the threads.
148///
149/// Threads are popped & dequeued from the active array (`array[index]`), and
150/// are enqueued into the inactive array (`array[!index]`). When the active array
151/// is empty, the 2 arrays are swapped by `index`.
152#[derive(Debug)]
153pub(super) struct RealTimeClassRq {
154    #[expect(unused)]
155    cpu: CpuId,
156    index: bool,
157    array: [PrioArray; 2],
158    nr_running: usize,
159}
160
161impl RealTimeClassRq {
162    pub fn new(cpu: CpuId) -> RealTimeClassRq {
163        RealTimeClassRq {
164            cpu,
165            index: false,
166            array: array::from_fn(|_| PrioArray {
167                map: bitarr![0; 100],
168                queue: array::from_fn(|_| VecDeque::new()),
169            }),
170            nr_running: 0,
171        }
172    }
173
174    fn active_array(&mut self) -> &mut PrioArray {
175        &mut self.array[usize::from(self.index)]
176    }
177
178    fn inactive_array(&mut self) -> &mut PrioArray {
179        &mut self.array[usize::from(!self.index)]
180    }
181
182    fn swap_arrays(&mut self) {
183        self.index = !self.index;
184    }
185}
186
187impl SchedClassRq for RealTimeClassRq {
188    fn enqueue(&mut self, entity: Arc<Task>, _: Option<EnqueueFlags>) {
189        let sched_attr = entity.as_thread().unwrap().sched_attr();
190        let prio = sched_attr.real_time.prio.load(Relaxed);
191        self.inactive_array().enqueue(entity, prio);
192        self.nr_running += 1;
193    }
194
195    fn len(&self) -> usize {
196        self.nr_running
197    }
198
199    fn is_empty(&self) -> bool {
200        self.nr_running == 0
201    }
202
203    fn pick_next(&mut self) -> Option<Arc<Task>> {
204        if self.nr_running == 0 {
205            return None;
206        }
207
208        (self.active_array().pop())
209            .or_else(|| {
210                self.swap_arrays();
211                self.active_array().pop()
212            })
213            .inspect(|_| self.nr_running -= 1)
214    }
215
216    fn update_current(
217        &mut self,
218        rt: &CurrentRuntime,
219        attr: &SchedAttr,
220        flags: UpdateFlags,
221    ) -> bool {
222        let attr = &attr.real_time;
223
224        match flags {
225            UpdateFlags::Tick | UpdateFlags::Yield => match attr.time_slice.load(Relaxed) {
226                0 => {
227                    self.active_array().peek_prio().is_some()
228                        || self
229                            .inactive_array()
230                            .peek_prio()
231                            .is_some_and(|prio| prio < attr.prio.load(Relaxed))
232                }
233                ts => ts <= rt.period_delta && !self.is_empty(),
234            },
235            UpdateFlags::Wait | UpdateFlags::Exit => !self.is_empty(),
236        }
237    }
238}
