3use ostd::timer;
4use spin::Once;
5
6use super::loadavg;
7
8/// The global scheduler statistic singleton
9static SCHEDULER_STATS: Once<&'static dyn SchedulerStats> = Once::new();
10
11/// Set the global scheduler statistics singleton.
12///
13/// This function should be called once to set the scheduler statistics system.
14/// It is used to get running stats from the scheduler and to periodically
15/// calculate the system load average.
16pub fn set_stats_from_scheduler(scheduler: &'static dyn SchedulerStats) {
17    SCHEDULER_STATS.call_once(|| scheduler);
18
19    // Register a callback to update the load average periodically
20    timer::register_callback_on_cpu(|| {
21        loadavg::update_loadavg(|| nr_queued_and_running().0);
22    });
23}
24
25/// The trait for the scheduler statistics.
26pub trait SchedulerStats: Sync + Send {
27    /// Returns a tuple with the number of tasks in the runqueues and the number of running tasks.
28    ///
29    /// We decided to return a tuple instead of having two separate functions to
30    /// avoid the overhead of disabling the preemption twice to inspect the scheduler.
31    fn nr_queued_and_running(&self) -> (u32, u32);
32}
33
34/// Get the amount of tasks in the runqueues and the amount of running tasks.
35pub fn nr_queued_and_running() -> (u32, u32) {
36    SCHEDULER_STATS.get().unwrap().nr_queued_and_running()
37}
