3//! This module implements the CPU load average calculation.
4//!
5//! Reference: <https://github.com/torvalds/linux/blob/46132e3/kernel/sched/loadavg.c>
6
7use core::sync::atomic::{AtomicU64, Ordering::Relaxed};
8
9use aster_util::fixed_point::FixedU32;
10use ostd::{
11    sync::RwLock,
12    timer::{self, TIMER_FREQ},
13};
14
15/// Fixed-point representation of the load average.
16///
17/// This is an equivalent of an u32 with 21 bits for the integer part and 11 bits for the fractional part.
18pub type LoadAvgFixed = FixedU32<11>;
19
20/// 5 sec intervals
21const LOAD_FREQ: u64 = 5 * TIMER_FREQ + 1;
22/// 1/exp(5sec/1min) as fixed-point
23const EXP_1: LoadAvgFixed = LoadAvgFixed::from_raw(1884);
24/// 1/exp(5sec/5min)
25const EXP_5: LoadAvgFixed = LoadAvgFixed::from_raw(2014);
26/// 1/exp(5sec/15min)
27const EXP_15: LoadAvgFixed = LoadAvgFixed::from_raw(2037);
28
29/// Load average of all CPU cores.
30///
31/// The load average is calculated as an exponential moving average of the load
32/// over the last 1, 5, and 15 minutes.
33static LOAD_AVG: RwLock<[LoadAvgFixed; 3]> = RwLock::new([LoadAvgFixed::ZERO; 3]);
34
35/// Next time the load average will be updated (in jiffies).
36static LOAD_AVG_NEXT_UPDATE: AtomicU64 = AtomicU64::new(0);
37
38/// Returns the calculated load average of the system.
39pub fn get_loadavg() -> [LoadAvgFixed; 3] {
40    *LOAD_AVG.read()
41}
42
43/// Updates the load average of the system.
44///
45/// This function should be called periodically to update the load average.
46/// The `get_load` function should return the load (the number of queued tasks) of the system.
47/// See `sched::stats::scheduler_stats::set_stats_from_scheduler()` for an example.
48pub fn update_loadavg<F>(get_load: F)
49where
50    F: Fn() -> u32,
51{
52    let jiffies = timer::Jiffies::elapsed().as_u64();
53
54    // Return if the load average was updated less than 5 seconds ago.
55    if jiffies < LOAD_AVG_NEXT_UPDATE.load(Relaxed) {
56        return;
57    }
58
59    // Update the next time the load average will be updated (now + 5sec)
60    LOAD_AVG_NEXT_UPDATE.store(jiffies + LOAD_FREQ, Relaxed);
61
62    // Get the fixed-point representation of the load
63    let new_load = LoadAvgFixed::saturating_from_num(get_load());
64
65    let mut load = LOAD_AVG.write();
66
67    // Calculate the new load average
68    load[0] = calc_loadavg(load[0], EXP_1, new_load);
69    load[1] = calc_loadavg(load[1], EXP_5, new_load);
70    load[2] = calc_loadavg(load[2], EXP_15, new_load);
71}
72
73fn calc_loadavg(old_load: LoadAvgFixed, exp: LoadAvgFixed, new_load: LoadAvgFixed) -> LoadAvgFixed {
74    old_load * exp + new_load * (LoadAvgFixed::ONE - exp)
75}
