3use alloc::string::ToString;
4
5use aster_framebuffer::{CONSOLE_NAME, FRAMEBUFFER_CONSOLE};
6use log::info;
7
8pub fn init() {
9    for device in aster_input::all_devices() {
10        info!("Found an input device, name:{}", device.name());
11    }
12
13    // FIXME: Currently, we have to do this manually to ensure the crates containing the input
14    // devices are linked and their `#[init_component]` hooks can run to register the devices with
15    // the input core. We should find a way to avoid this in the future.
16    #[expect(unused_imports)]
17    use aster_i8042::*;
18
19    if let Some(console) = FRAMEBUFFER_CONSOLE.get() {
20        aster_console::register_device(CONSOLE_NAME.to_string(), console.clone());
21    }
22}
