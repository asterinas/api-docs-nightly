20pub fn geturandom(writer: &mut VmWriter) -> Result<usize> {
21    const IO_CAPABILITY: usize = 4096;
22
23    if !writer.has_avail() {
24        return Ok(0);
25    }
26
27    let mut buffer = vec![0; writer.avail().min(IO_CAPABILITY)];
28    let mut written_bytes = 0;
29
30    while writer.has_avail() {
31        random::getrandom(&mut buffer[..writer.avail().min(IO_CAPABILITY)]);
32        match writer.write_fallible(&mut VmReader::from(buffer.as_slice())) {
33            Ok(len) => written_bytes += len,
34            Err((err, 0)) if written_bytes == 0 => return Err(err.into()),
35            Err((_, len)) => return Ok(written_bytes + len),
36        }
37    }
38
39    Ok(written_bytes)
40}

105impl InodeIo for MemFile {
106    fn read_at(
107        &self,
108        _offset: usize,
109        writer: &mut VmWriter,
110        _status_flags: StatusFlags,
111    ) -> Result<usize> {
112        match self {
113            MemFile::Full | MemFile::Zero => {
114                let len = writer.avail();
115                writer.fill_zeros(len)?;
116                Ok(len)
117            }
118            MemFile::Null => Ok(0),
119            MemFile::Random => getrandom(writer),
120            MemFile::Urandom => geturandom(writer),
121            _ => return_errno_with_message!(Errno::EINVAL, "read is not supported yet"),
122        }
123    }
124
125    fn write_at(
126        &self,
127        _offset: usize,
128        reader: &mut VmReader,
129        _status_flags: StatusFlags,
130    ) -> Result<usize> {
131        match self {
132            MemFile::Null | MemFile::Random | MemFile::Urandom | MemFile::Zero => {
133                let len = reader.remain();
134                reader.skip(len);
135                Ok(len)
136            }
137            MemFile::Full => {
138                return_errno_with_message!(Errno::ENOSPC, "no space left on /dev/full")
139            }
140            _ => return_errno_with_message!(Errno::EINVAL, "write is not supported yet"),
141        }
142    }
143}
