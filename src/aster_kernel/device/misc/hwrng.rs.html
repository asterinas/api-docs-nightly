25const HWRNG_MINOR: u32 = 183;

27/// The currently in-use hardware RNG device.
28//
29// TODO: Users can select a device by writing its name to `/sys/class/misc/hw_random/rng_current`,
30// which is not supported yet.
31static RNG_CURRENT: Mutex<Option<Arc<EntropyDevice>>> = Mutex::new(None);
32
33/// The `/dev/hwrng` device.
34#[derive(Debug)]
35struct HwRngDevice {
36    id: DeviceId,
37}
38
39impl HwRngDevice {
40    fn new() -> Arc<Self> {
41        let major = super::MISC_MAJOR.get().unwrap().get();
42        let minor = MinorId::new(HWRNG_MINOR);
43
44        let id = DeviceId::new(major, minor);
45        Arc::new(Self { id })
46    }
47}
48
49impl Device for HwRngDevice {
50    fn type_(&self) -> DeviceType {
51        DeviceType::Char
52    }
53
54    fn id(&self) -> DeviceId {
55        self.id
56    }
57
58    fn devtmpfs_meta(&self) -> Option<DevtmpfsInodeMeta<'_>> {
59        Some(DevtmpfsInodeMeta::new("hwrng"))
60    }
61
62    fn open(&self) -> Result<Box<dyn FileIo>> {
63        // TODO: Reject non-read-only opens with `EINVAL`
64        // once device `open` callbacks receive `AccessMode`.
65        // Reference: <https://elixir.bootlin.com/linux/v6.18/source/drivers/char/hw_random/core.c#L169>.
66        Ok(Box::new(HwRngFile))
67    }
68}

70/// A file handle opened from `/dev/hwrng`.
71struct HwRngFile;
72
73impl Pollable for HwRngFile {
74    fn poll(&self, mask: IoEvents, _poller: Option<&mut PollHandle>) -> IoEvents {
75        // Linux's `/dev/hwrng` does not implement `.poll`, so userspace sees the VFS
76        // default ("always ready").
77        // Reference: <https://elixir.bootlin.com/linux/v6.18/source/drivers/char/hw_random/core.c#L287-L292>.
78        mask & (IoEvents::IN | IoEvents::OUT)
79    }
80}
81
82impl InodeIo for HwRngFile {
83    fn read_at(
84        &self,
85        _offset: usize,
86        writer: &mut VmWriter,
87        status_flags: StatusFlags,
88    ) -> Result<usize> {
89        // Linux looks up the selected device at `read()`.
90        // Reference: <https://elixir.bootlin.com/linux/v6.18/source/drivers/char/hw_random/core.c#L215>.
91        let dev = current_device()?;
92        let is_nonblocking = status_flags.contains(StatusFlags::O_NONBLOCK);
93
94        let mut total_copied: usize = 0;
95        while writer.avail() > 0 {
96            // Clone the writer to keep the cursor at the correct position in case a page fault
97            // occurs later.
98            let mut new_writer = writer.clone_exclusive();
99
100            let res = if is_nonblocking {
101                match dev.try_read(&mut new_writer)? {
102                    Some(copied) => Ok(copied),
103                    None => return_errno_with_message!(Errno::EAGAIN, "no entropy is available"),
104                }
105            } else {
106                dev.wait_queue()
107                    .wait_until(|| match dev.try_read(&mut new_writer) {
108                        Ok(Some(copied)) => Some(Ok(copied)),
109                        Ok(None) => None,
110                        Err(err) => Some(Err(err)),
111                    })
112            };
113
114            match res {
115                Ok(copied) => {
116                    writer.skip(copied);
117                    total_copied += copied;
118                }
119                Err(err) => {
120                    if total_copied == 0 {
121                        return Err(err.into());
122                    }
123                    break;
124                }
125            }
126        }
127
128        Ok(total_copied)
129    }
130
131    fn write_at(
132        &self,
133        _offset: usize,
134        _reader: &mut VmReader,
135        _status_flags: StatusFlags,
136    ) -> Result<usize> {
137        // FIXME: Opening this device with `O_WRONLY` or `O_RDWR` fails on Linux. Therefore, the
138        // write operation should never be reached. See the TODO in `HwRngDevice::open`.
139        return_errno_with_message!(
140            Errno::EBADF,
141            "the hardware RNG device does not support writing"
142        );
143    }
144}

146impl FileIo for HwRngFile {
147    fn check_seekable(&self) -> Result<()> {
148        Ok(())
149    }

152        false

156fn current_device() -> Result<Arc<EntropyDevice>> {
157    let Some(rng) = RNG_CURRENT.lock().clone() else {
158        return_errno_with_message!(Errno::ENODEV, "no current hardware RNG device is selected");
159    };
160    Ok(rng)
161}
162
163pub(super) fn init_in_first_kthread() {
164    if let Some(device) = entropy::first_device() {
165        *RNG_CURRENT.lock() = Some(device);
166    }
167
168    char::register(HwRngDevice::new()).unwrap();
169}
