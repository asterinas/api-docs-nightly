32/// Bitfields describing the color channel layout; `struct fb_bitfield` in Linux.
33///
34/// Reference: <https://elixir.bootlin.com/linux/v6.17/source/include/uapi/linux/fb.h#L189>.
35#[repr(C)]
36#[derive(Debug, Clone, Copy, Pod, Default)]
37struct FbBitfield {
38    /// Bit offset of the field
39    pub offset: u32,
40    /// Length of the field in bits
41    pub length: u32,
42    /// Most significant bit position (0 = left, 1 = right)
43    pub msb_right: u32,
44}
45
46impl FbBitfield {
47    /// Converts pixel format to framebuffer bitfields for Linux compatibility.
48    #[rustfmt::skip]
49    fn from_pixel_format(pixel_format: PixelFormat) -> (Self, Self, Self, Self) {
50        match pixel_format {
51            PixelFormat::Grayscale8 => (
52                Self { offset: 0, length: 8, msb_right: 0 },
53                Self { offset: 0, length: 8, msb_right: 0 },
54                Self { offset: 0, length: 8, msb_right: 0 },
55                Self::default(),
56            ),
57            PixelFormat::Rgb565 => (
58                Self { offset: 11, length: 5, msb_right: 0 },
59                Self { offset: 5, length: 6, msb_right: 0 },
60                Self { offset: 0, length: 5, msb_right: 0 },
61                Self::default(),
62            ),
63            PixelFormat::Rgb888 => (
64                Self { offset: 16, length: 8, msb_right: 0 },
65                Self { offset: 8, length: 8, msb_right: 0 },
66                Self { offset: 0, length: 8, msb_right: 0 },
67                Self::default(),
68            ),
69            PixelFormat::BgrReserved => (
70                Self { offset: 16, length: 8, msb_right: 0 },
71                Self { offset: 8, length: 8, msb_right: 0 },
72                Self { offset: 0, length: 8, msb_right: 0 },
73                Self { offset: 24, length: 8, msb_right: 0 },
74            ),
75        }
76    }
77}

232        Some("fb0".into())

235    fn open(&self) -> Result<Box<dyn FileIo>> {
236        let Some(framebuffer) = FRAMEBUFFER.get() else {
237            return Err(Error::with_message(
238                Errno::ENODEV,
239                "the framebuffer device is not present",
240            ));
241        };
242        let framebuffer = framebuffer.clone();
243        Ok(Box::new(FbHandle { framebuffer }))
244    }

247impl FbHandle {
248    /// Reads an array of `u16` color map values from userspace.
249    fn read_color_maps_from_user(addr: usize, data: &mut [u16]) -> Result<()> {
250        for (i, item) in data.iter_mut().enumerate() {
251            let user_addr = addr + i * size_of::<u16>();
252            *item = current_userspace!().read_val(user_addr)?;
253        }
254        Ok(())
255    }
256
257    /// Writes an array of `u16` color map values to userspace.
258    fn write_color_maps_to_user(addr: usize, data: &[u16]) -> Result<()> {
259        for (i, &value) in data.iter().enumerate() {
260            let user_addr = addr + i * size_of::<u16>();
261            current_userspace!().write_val(user_addr, &value)?;
262        }
263        Ok(())
264    }
265
266    /// Collects the information in the [`FbVarScreenInfo`].
267    fn collect_var_screen_info(&self) -> FbVarScreenInfo {
268        /// Default pixel clock calculation for efifb compatibility
269        const DEFAULT_PIXEL_CLOCK_DIVISOR: u32 = 10_000_000;
270
271        /// Default timing parameters for efifb compatibility
272        const DEFAULT_RIGHT_MARGIN: u32 = 32;
273        const DEFAULT_UPPER_MARGIN: u32 = 16;
274        const DEFAULT_LOWER_MARGIN: u32 = 4;
275        const DEFAULT_VSYNC_LEN: u32 = 4;
276
277        let pixel_format = self.framebuffer.pixel_format();
278        let (red, green, blue, transp) = FbBitfield::from_pixel_format(pixel_format);
279
280        FbVarScreenInfo {
281            xres: self.framebuffer.width() as u32,
282            yres: self.framebuffer.height() as u32,
283            xres_virtual: self.framebuffer.width() as u32,
284            yres_virtual: self.framebuffer.height() as u32,
285            bits_per_pixel: (8 * pixel_format.nbytes()) as u32,
286            red,
287            green,
288            blue,
289            transp,
290            pixclock: DEFAULT_PIXEL_CLOCK_DIVISOR / self.framebuffer.width() as u32 * 1000
291                / self.framebuffer.height() as u32,
292            left_margin: (self.framebuffer.width() as u32 / 8) & 0xf8,
293            right_margin: DEFAULT_RIGHT_MARGIN,
294            upper_margin: DEFAULT_UPPER_MARGIN,
295            lower_margin: DEFAULT_LOWER_MARGIN,
296            vsync_len: DEFAULT_VSYNC_LEN,
297            hsync_len: (self.framebuffer.width() as u32 / 8) & 0xf8,
298            ..Default::default()
299        }
300    }
301
302    /// Collects the information in the [`FbFixScreenInfo`].
303    fn collect_fix_screen_info(&self) -> FbFixScreenInfo {
304        FbFixScreenInfo {
305            smem_start: self.framebuffer.io_mem().paddr() as u64,
306            smem_len: self.framebuffer.io_mem().size() as u32,
307            line_length: self.framebuffer.line_size() as u32,
308            ..Default::default()
309        }
310    }
311
312    /// Handles the [`ioctl_defs::GetColorMap`] ioctl command.
313    ///
314    /// Arguments:
315    ///  - Input: [`FbCmapUser`] (specifying the range).
316    ///  - Output: [`FbCmapUser`] (filled with color palette data).
317    fn handle_get_cmap(&self, cmap_user: &FbCmapUser) -> Result<()> {
318        if cmap_user.len == 0 {
319            return Ok(());
320        }
321
322        let start = cmap_user.start as usize;
323        let len = cmap_user.len as usize;
324
325        // Get color map entries from framebuffer
326        let entries = self.framebuffer.get_color_map(start, len).ok_or_else(|| {
327            Error::with_message(Errno::EINVAL, "the color map index is out of bounds")
328        })?;
329
330        // Extract color channels and write to userspace
331        let red: Vec<u16> = entries.iter().map(|e| e.red).collect();
332        let green: Vec<u16> = entries.iter().map(|e| e.green).collect();
333        let blue: Vec<u16> = entries.iter().map(|e| e.blue).collect();
334        let transp: Vec<u16> = entries.iter().map(|e| e.transp).collect();
335
336        Self::write_color_maps_to_user(cmap_user.red, &red)?;
337        Self::write_color_maps_to_user(cmap_user.green, &green)?;
338        Self::write_color_maps_to_user(cmap_user.blue, &blue)?;
339        if cmap_user.transp != 0 {
340            Self::write_color_maps_to_user(cmap_user.transp, &transp)?;
341        }
342
343        Ok(())
344    }
345
346    /// Handles the [`ioctl_defs::PutColorMap`] ioctl command.
347    ///
348    /// Arguments:
349    ///  - Input: [`FbCmapUser`] (with color palette data).
350    ///  - Output: None.
351    fn handle_set_cmap(&self, cmap_user: &FbCmapUser) -> Result<()> {
352        if cmap_user.len == 0 {
353            return Ok(());
354        }
355
356        let start = cmap_user.start as usize;
357        let len = cmap_user.len as usize;
358
359        // Check the size to prevent excessive memory allocation
360        if start > MAX_CMAP_SIZE || len > MAX_CMAP_SIZE - start {
361            return_errno_with_message!(
362                Errno::EINVAL,
363                "the color map range exceeds its maximum size"
364            );
365        }
366
367        // Read color data from userspace
368        let mut red = vec![0u16; len];
369        let mut green = vec![0u16; len];
370        let mut blue = vec![0u16; len];
371        let mut transp = vec![0u16; len];
372
373        Self::read_color_maps_from_user(cmap_user.red, &mut red)?;
374        Self::read_color_maps_from_user(cmap_user.green, &mut green)?;
375        Self::read_color_maps_from_user(cmap_user.blue, &mut blue)?;
376        if cmap_user.transp != 0 {
377            Self::read_color_maps_from_user(cmap_user.transp, &mut transp)?;
378        }
379
380        // Build color map entries
381        let entries: Vec<ColorMapEntry> = (0..len)
382            .map(|i| ColorMapEntry {
383                red: red[i],
384                green: green[i],
385                blue: blue[i],
386                transp: transp[i],
387            })
388            .collect();
389
390        // Set color map entries in framebuffer
391        self.framebuffer.set_color_map(start, &entries)?;
392
393        Ok(())
394    }
395}

404impl InodeIo for FbHandle {
405    fn read_at(
406        &self,
407        offset: usize,
408        writer: &mut VmWriter,
409        _status_flags: StatusFlags,
410    ) -> Result<usize> {
411        if !writer.has_avail() {
412            return Ok(0);
413        }
414
415        let mut reader = self.framebuffer.io_mem().reader();
416
417        if offset >= reader.remain() {
418            return Ok(0);
419        }
420        reader.skip(offset);
421
422        let mut reader = reader.to_fallible();
423        let len = match reader.read_fallible(writer) {
424            Ok(len) => len,
425            Err((err, 0)) => return Err(err.into()),
426            Err((_err, len)) => len,
427        };
428
429        Ok(len)
430    }
431
432    fn write_at(
433        &self,
434        offset: usize,
435        reader: &mut VmReader,
436        _status_flags: StatusFlags,
437    ) -> Result<usize> {
438        if !reader.has_remain() {
439            return Ok(0);
440        }
441
442        let mut writer = self.framebuffer.io_mem().writer();
443        if offset >= writer.avail() {
444            return_errno_with_message!(
445                Errno::ENOSPC,
446                "the write offset is beyond the framebuffer size"
447            );
448        }
449        writer.skip(offset);
450
451        let mut writer = writer.to_fallible();
452        let len = match writer.write_fallible(reader) {
453            Ok(len) => len,
454            Err((err, 0)) => return Err(err.into()),
455            Err((_err, len)) => len,
456        };
457
458        Ok(len)
459    }
460}

462impl FileIo for FbHandle {
463    fn check_seekable(&self) -> Result<()> {
464        Ok(())
465    }
466
467    fn is_offset_aware(&self) -> bool {
468        true
469    }
470
471    fn mappable(&self) -> Result<Mappable> {
472        let iomem = self.framebuffer.io_mem();
473        Ok(Mappable::IoMem(iomem.clone()))
474    }
475
476    fn ioctl(&self, raw_ioctl: RawIoctl) -> Result<i32> {
477        use ioctl_defs::*;
478
479        dispatch_ioctl!(match raw_ioctl {
480            cmd @ GetVarScreenInfo => {
481                cmd.write(&self.collect_var_screen_info())?;
482                Ok(0)
483            }
484            cmd @ PutVarScreenInfo => {
485                // EFI framebuffers do not support changing settings. Linux
486                // will return the old settings to user space and succeed.
487                // Reference: <https://elixir.bootlin.com/linux/v6.17/source/drivers/video/fbdev/core/fbmem.c#L276-L279>.
488                cmd.write(&self.collect_var_screen_info())?;
489                Ok(0)
490            }
491            cmd @ GetFixScreenInfo => {
492                cmd.write(&self.collect_fix_screen_info())?;
493                Ok(0)
494            }
495            cmd @ GetColorMap => {
496                self.handle_get_cmap(&cmd.read()?)?;
497                Ok(0)
498            }
499            cmd @ PutColorMap => {
500                self.handle_set_cmap(&cmd.read()?)?;
501                Ok(0)
502            }
503            PanDisplay | Blank => {
504                // These commands are not supported by efifb.
505                // We return errors according to the Linux behavior.
506                return_errno_with_message!(
507                    Errno::EINVAL,
508                    "the ioctl command is not supported by efifb devices"
509                )
510            }
511            _ => {
512                log::debug!(
513                    "the ioctl command {:#x} is unknown for framebuffer devices",
514                    raw_ioctl.cmd()
515                );
516                return_errno_with_message!(Errno::ENOTTY, "the ioctl command is unknown");
517            }
518        })
519    }
520}
