3use core::sync::atomic::{AtomicBool, Ordering};
4
5use crate::prelude::*;
6
7/// Control states related to packet mode.
8///
9/// Reference: <https://man7.org/linux/man-pages/man2/TIOCPKT.2const.html>.
10pub(super) struct PacketCtrl {
11    mode: AtomicBool,
12    status: SpinLock<PacketStatus>,
13}
14
15impl PacketCtrl {
16    // Creates a new `PacketCtrl`.
17    pub(super) fn new() -> Self {
18        Self {
19            mode: AtomicBool::new(false),
20            status: SpinLock::new(PacketStatus::empty()),
21        }
22    }
23
24    /// Returns whether packet mode is enabled.
25    pub(super) fn mode(&self) -> bool {
26        self.mode.load(Ordering::Relaxed)
27    }
28
29    /// Sets whether packet mode is enabled.
30    pub(super) fn set_mode(&self, mode: bool) {
31        // Reference: <https://elixir.bootlin.com/linux/v6.17/source/drivers/tty/pty.c#L158>.
32        let mut status = self.status.lock();
33
34        let old_mode = self.mode.load(Ordering::Relaxed);
35        self.mode.store(mode, Ordering::Relaxed);
36
37        if mode && !old_mode {
38            *status = PacketStatus::empty();
39        }
40    }
41
42    /// Sets packet status if packet mode is enabled.
43    pub(super) fn set_status(&self, set_packet_status: impl FnOnce(&mut PacketStatus)) -> bool {
44        // Fast path: Packet mode is disabled.
45        if !self.mode.load(Ordering::Relaxed) {
46            return false;
47        }
48
49        // Packet mode is enabled.
50        let mut packet_status = self.status.lock();
51        set_packet_status(&mut packet_status);
52        true
53    }
54
55    /// Checks if packet mode is enabled and there is pending packet status.
56    pub(super) fn has_status(&self) -> bool {
57        // Fast path: Packet mode is disabled.
58        if !self.mode.load(Ordering::Relaxed) {
59            return false;
60        }
61
62        // Packet mode is enabled.
63        !self.status.lock().is_empty()
64    }
65
66    /// Takes out packet status if packet mode is enabled.
67    pub(super) fn take_status(&self) -> Option<PacketStatus> {
68        // Fast path: Packed mode is disabled.
69        if !self.mode.load(Ordering::Relaxed) {
70            return None;
71        }
72
73        // Packet mode is enabled.
74        let mut status = self.status.lock();
75        let old_status = *status;
76        *status = PacketStatus::empty();
77        Some(old_status)
78    }
79}
