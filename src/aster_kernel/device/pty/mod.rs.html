21static DEV_PTS: Once<Path> = Once::new();
22
23pub fn init_in_first_process(path_resolver: &PathResolver, ctx: &Context) -> Result<()> {
24    let dev = path_resolver.lookup(&FsPath::try_from("/dev")?)?;
25    // Create the "pts" directory and mount devpts on it.
26    let devpts_path = dev.new_fs_child("pts", InodeType::Dir, mkmod!(a+rx, u+w))?;
27    let devpts_mount = devpts_path.mount(
28        DevPts::new(),
29        PerMountFlags::default(),
30        Some("devpts".to_string()),
31        ctx,
32    )?;
33
34    DEV_PTS.call_once(|| Path::new_fs_root(devpts_mount));
35
36    // Create the "ptmx" symlink.
37    let ptmx = dev.new_fs_child("ptmx", InodeType::SymLink, mkmod!(a+rwx))?;
38    ptmx.inode().write_link("pts/ptmx")?;
39    Ok(())
40}

42pub fn new_pty_pair(index: u32, ptmx: Arc<Ptmx>) -> Result<(Box<PtyMaster>, Arc<PtySlave>)> {
43    debug!("pty index = {}", index);
44    let master = PtyMaster::new(ptmx, index);
45    let slave = master.slave().clone();
46    Ok((master, slave))
