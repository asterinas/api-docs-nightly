23pub(super) fn init_in_first_kthread() {
24    for device in aster_block::collect_all() {
25        if device.is_partition() {
26            continue;
27        }
28
29        // Spawn threads for virtio block devices
30        if device.downcast_ref::<VirtIoBlockDevice>().is_some() {
31            let device_clone = device.clone();
32            let task_fn = move || {
33                info!("spawn the virtio-block thread");
34                let virtio_block_device = device_clone.downcast_ref::<VirtIoBlockDevice>().unwrap();
35                loop {
36                    virtio_block_device.handle_requests();
37                }
38            };
39            ThreadOptions::new(task_fn).spawn();
40        }
41        // Spawn threads for NVMe block devices
42        else if device.downcast_ref::<NvmeBlockDevice>().is_some() {
43            let device_clone = device.clone();
44            let task_fn = move || {
45                info!("spawn the nvme-block thread");
46                let nvme_block_device = device_clone.downcast_ref::<NvmeBlockDevice>().unwrap();
47                loop {
48                    nvme_block_device.handle_requests();
49                }
50            };
51            ThreadOptions::new(task_fn).spawn();
52        }
53    }
54}

56pub(super) fn init_in_first_process(path_resolver: &PathResolver) -> Result<()> {
57    for device in aster_block::collect_all() {
58        let device = Arc::new(BlockFile::new(device));
59        if let Some(devtmpfs_meta) = device.devtmpfs_meta() {
60            let dev_id = device.id().as_encoded_u64();
61            add_node(DeviceType::Block, dev_id, &devtmpfs_meta, path_resolver)?;
62        }
63    }
64
65    Ok(())
66}

68mod ioctl_defs {
69    use crate::util::ioctl::{NoData, OutData, ioc};
70
71    // Reference: <https://elixir.bootlin.com/linux/v6.18/source/include/uapi/linux/fs.h>
72
73    /// Returns the device size in bytes.
74    pub(super) type BlkGetSize64 = ioc!(BLKGETSIZE64, 0x12, 114, OutData<u64>);
75
76    /// Returns the logical sector size of the block device.
77    ///
78    /// This is the smallest unit of I/O the device can address and,
79    /// importantly, the minimum alignment required for `O_DIRECT` I/O on
80    /// files backed by this device. Both buffer address and offset must be a
81    /// multiple of this value.
82    ///
83    /// Benchmarks and filesystem tests (for example, `xfstests`, LTP
84    /// `preadv03`/`pwritev03`) rely on this ioctl to size `O_DIRECT` buffers.
85    /// If the effective alignment enforced by the filesystem layered on top
86    /// is larger than the hardware sector, such as `ext2`'s 4 KiB block, this
87    /// ioctl must return that larger value. Otherwise user programs will align
88    /// correctly for the device but still hit `EINVAL` at the filesystem.
89    pub(super) type BlkGetSectorSize = ioc!(BLKSSZGET, 0x12, 104, NoData);
90}

92/// Represents a block device inode in the filesystem.
93//
94// TODO: This type wraps an `Arc<dyn BlockDevice>` in another `Arc` just to implement the `Device`
95// trait. It leads to redundant vtable dispatch, reference counting, and heap allocation. We should
96// devise a better strategy to eliminate the unnecessary intermediate `Arc`.
97#[derive(Debug)]
98struct BlockFile(Arc<dyn BlockDevice>);

124/// Represents an opened block device file ready for I/O operations.
125//
126// TODO: This type wraps an `Arc<dyn BlockDevice>` in another `Box` just to implement the
127// `PerOpenFileOps` trait. It leads to redundant vtable dispatch and heap allocation. We should
128// devise a better strategy to eliminate the unnecessary intermediate `Box`.
129struct OpenBlockFile(Arc<dyn BlockDevice>);
130
131impl FileOps for OpenBlockFile {
132    fn read_at(
133        &self,
134        offset: usize,
135        writer: &mut VmWriter,
136        _status_flags: StatusFlags,
137    ) -> Result<usize> {
138        let total = writer.avail();
139        if total == 0 {
140            return Ok(0);
141        }
142
143        let device_size = self.0.metadata().nr_sectors * SECTOR_SIZE;
144        if offset >= device_size {
145            return Ok(0);
146        }
147
148        let read_len = total.min(device_size - offset);
149        {
150            // `VmIo::read` does not allow short writes,
151            // so the writer must be precisely limited here.
152            let mut limited_writer = writer.clone_exclusive();
153            limited_writer.limit(read_len);
154            self.0.read(offset, &mut limited_writer)?;
155        }
156        writer.skip(read_len);
157        Ok(read_len)
158    }
159
160    fn write_at(
161        &self,
162        offset: usize,
163        reader: &mut VmReader,
164        _status_flags: StatusFlags,
165    ) -> Result<usize> {
166        let total = reader.remain();
167        if total == 0 {
168            return Ok(0);
169        }
170
171        let device_size = self.0.metadata().nr_sectors * SECTOR_SIZE;
172        if offset >= device_size {
173            return_errno_with_message!(
174                Errno::ENOSPC,
175                "the write offset is beyond the block device"
176            );
177        }
178
179        let write_len = total.min(device_size - offset);
180        {
181            // `VmIo::write` does not allow short writes,
182            // so the reader must be precisely limited here.
183            let mut limited_reader = reader.clone();
184            limited_reader.limit(write_len);
185            self.0.write(offset, &mut limited_reader)?;
186        }
187        reader.skip(write_len);
188        Ok(write_len)
189    }
190}

192impl Pollable for OpenBlockFile {
193    fn poll(&self, mask: IoEvents, _: Option<&mut PollHandle>) -> IoEvents {
194        let events = IoEvents::IN | IoEvents::OUT;
195        events & mask
196    }
197}

199impl PerOpenFileOps for OpenBlockFile {
200    fn check_seekable(&self) -> Result<()> {
201        Ok(())
202    }
203
204    fn is_offset_aware(&self) -> bool {
205        true
206    }
207
208    fn seek_end(&self) -> Result<Option<usize>> {
209        Ok(Some(self.0.metadata().nr_sectors * SECTOR_SIZE))
210    }
211
212    fn ioctl(&self, raw_ioctl: RawIoctl) -> Result<i32> {
213        use ioctl_defs::*;
214
215        dispatch_ioctl!(match raw_ioctl {
216            _cmd @ BlkGetSectorSize => {
217                // TODO: Query the per-device logical block size once block device metadata
218                // exposes it. For now, report the effective minimum I/O granularity enforced
219                // by Asterinas filesystems so userspace can use `BLKSSZGET` for `O_DIRECT`
220                // alignment.
221                let sector_size = SECTOR_SIZE.max(BLOCK_SIZE) as i32;
222                current_userspace!().write_val(raw_ioctl.arg(), &sector_size)?;
223                Ok(0)
224            }
225            cmd @ BlkGetSize64 => {
226                let size = (self.0.metadata().nr_sectors * SECTOR_SIZE) as u64;
227                cmd.write(&size)?;
228                Ok(0)
229            }
230            _ => return_errno_with_message!(
231                Errno::ENOTTY,
232                "the ioctl command is not supported by block devices"
233            ),
234        })
235    }
236
237    fn settable_status_flags(&self) -> SettableStatusFlags {
238        SettableStatusFlags::minimal().with_o_direct()
239    }
240}

242pub(super) fn lookup(id: DeviceId) -> Option<Arc<dyn Device>> {
243    let block_device = aster_block::lookup(id)?;
244
245    let mut registry = DEVICE_REGISTRY.lock();
246    let block_device_file = registry
247        .entry(id.to_raw())
248        .or_insert_with(move || Arc::new(BlockFile::new(block_device)))
249        .clone();
250    Some(block_device_file)
251}
