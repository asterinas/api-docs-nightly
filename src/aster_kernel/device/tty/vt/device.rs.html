3use alloc::sync::Arc;
4
5use aster_console::{AnyConsoleDevice, ConsoleCallback};
6
7use crate::device::tty::vt::manager::{VtConsole, default_vt};
8
9const DEFAULT_VT_CONSOLE_NAME: &str = "Default-VT-Console";
10
11/// A console device that forwards output to the default VT console.
12#[derive(Debug)]
13struct DefaultVtConsoleDevice;
14
15impl DefaultVtConsoleDevice {
16    fn vt_console(&self) -> &VtConsole {
17        // Keep kernel logs on the default VT console so log visibility is
18        // predictable for debugging.
19        default_vt().driver().vt_console()
20    }
21}
22
23impl AnyConsoleDevice for DefaultVtConsoleDevice {
24    fn send(&self, buf: &[u8]) {
25        self.vt_console().send(buf);
26    }
27
28    fn register_callback(&self, _callback: &'static ConsoleCallback) {}
29}
30
31pub(super) fn init_in_first_process() {
32    aster_console::register_device(
33        DEFAULT_VT_CONSOLE_NAME.into(),
34        Arc::new(DefaultVtConsoleDevice),
35    );
36}
