13/// Gets the keyboard type.
14pub(super) type GetKeyboardType = ioc!(KDGKBTYPE,       0x4B33,     OutData<u8>);
15
16/// Sets the display mode.
17pub(super) type SetGraphicsMode = ioc!(KDSETMODE,       0x4B3A,     InData<i32, PassByVal>);
18/// Gets the display mode.
19pub(super) type GetGraphicsMode = ioc!(KDGETMODE,       0x4B3B,     OutData<i32>);
20
21/// Gets the keyboard mode.
22pub(super) type GetKeyboardMode = ioc!(KDGKBMODE,       0x4B44,     OutData<i32>);
23/// Sets the keyboard mode.
24pub(super) type SetKeyboardMode = ioc!(KDSKBMODE,       0x4B45,     InData<i32, PassByVal>);

26/// Sets or gets font data.

29// Reference: <https://elixir.bootlin.com/linux/v6.17/source/include/uapi/linux/vt.h>
30
31/// Returns the first available (non-opened) VT number.
32/// If all VTs are in use, returns -1.
33///
34/// Valid VT numbers start from 1.
35pub(super) type GetAvailableVt  = ioc!(VT_OPENQRY,      0x5600,     OutData<i32>);
36
37/// Gets the VT mode.
38pub(super) type GetVtMode       = ioc!(VT_GETMODE,      0x5601,     OutData<CVtMode>);
39/// Sets the VT mode.
40pub(super) type SetVtMode       = ioc!(VT_SETMODE,      0x5602,     InData<CVtMode>);
41
42/// Gets the global VT state.
43///
44/// Note:
45/// - VT0 is always open (alias for active VT).
46/// - At most 16 VT states can be returned due to ABI constraints.
47pub(super) type GetVtState      = ioc!(VT_GETSTATE,     0x5603,     OutData<CVtState>);
48
49/// Used in process-controlled VT switching to allow or deny
50/// VT release, or to acknowledge VT acquisition.
51pub(super) type ReleaseDisplay  = ioc!(VT_RELDISP,      0x5605,     InData<i32, PassByVal>);
52
53/// Activates the specified VT (VT numbers start from 1).
54///
55/// Switching to VT0 is not allowed.
56pub(super) type ActivateVt      = ioc!(VT_ACTIVATE,     0x5606,     InData<i32, PassByVal>);
57/// Blocks until the specified VT becomes active.
58pub(super) type WaitForVtActive = ioc!(VT_WAITACTIVE,   0x5607,     InData<i32, PassByVal>);
59
60/// Disallocates the specified VT if argument is non-zero, or releases all VTs if argument
61/// is zero. But we don't disallocate VT1 since it's the default VT and always open.
62pub(super) type DisallocateVt   = ioc!(VT_DISALLOCATE,  0x5608,     InData<i32, PassByVal>);
