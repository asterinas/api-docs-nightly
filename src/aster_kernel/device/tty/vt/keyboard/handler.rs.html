23impl InputHandlerClass for VtKeyboardHandlerClass {
24    fn name(&self) -> &str {
25        "vt_keyboard"
26    }

28    fn connect(&self, dev: Arc<dyn InputDevice>) -> Result<Arc<dyn InputHandler>, ConnectError> {
29        let capability = dev.capability();
30        if !capability.look_like_keyboard() {
31            return Err(ConnectError::IncompatibleDevice);
32        }
33        ostd::info!(
34            "Virtual terminal keyboard handler connected to device: {}",
35            dev.name()
36        );
37        Ok(Arc::new(VtKeyboardHandler))
38    }
39
40    fn disconnect(&self, dev: &Arc<dyn InputDevice>) {
41        ostd::info!(
42            "Virtual terminal keyboard handler disconnected from device: {}",
43            dev.name()
44        );
45    }
46}

48/// The state of modifier keys (Shift, Ctrl, Alt).
49///
50/// It is shared across all virtual terminals.
51static MODIFIER_KEYS_STATE: ModifierKeysState = ModifierKeysState::new();
52
53/// The state of lock keys (Caps Lock, Num Lock, Scroll Lock).
54///
55/// We should have one for each virtual terminal, but now
56/// we only have one virtual terminal, so we just use a global state.
57static LOCK_KEYS_STATE: LockKeysState = LockKeysState::new();
58
59/// A virtual terminal keyboard handler that handles input from a specific device.
60#[derive(Debug)]
61struct VtKeyboardHandler;
62
63impl VtKeyboardHandler {
64    fn handle_key_event(&self, keycode: KeyCode, key_status: KeyStatus) {
65        ostd::debug!(
66            "Virtual terminal keyboard handler received key event: keycode={:?}, status={:?}",
67            keycode,
68            key_status
69        );
70
71        let Some(console) = FRAMEBUFFER_CONSOLE.get() else {
72            return;
73        };
74
75        let console_callbacks = console.lock_callbacks();
76        let keyboard_mode = console_callbacks.keyboard_mode();
77
78        if keyboard_mode == KeyboardMode::MediumRaw {
79            self.handle_medium_mode(keycode, key_status, &console_callbacks);
80        }
81
82        // Modifier keys are handled based on the keycode instead of the keymap.
83        //
84        // This is important because the keymap lookup depends on the current
85        // modifier state. If a modifier is released while multiple modifiers
86        // are active (e.g. Shift+Alt), the keymap may not contain an entry for
87        // that modifier combination. In that case the lookup would return
88        // `KeySym::NoOp`, and the release event would be ignored, causing the
89        // modifier state to become stuck.
90        //
91        // By detecting modifiers directly from the keycode, we ensure that
92        // modifier press/release events are always processed correctly,
93        // regardless of the current modifier combination.
94        let key_sym = if let Some(modifier) = self.keycode_to_modifier_key(keycode) {
95            KeySym::Modifier(modifier)
96        } else {
97            get_keysym(MODIFIER_KEYS_STATE.flags(), keycode)
98        };
99
100        // In Medium Raw, Raw, or Off mode, ignore non-modifier keys
101        if (keyboard_mode == KeyboardMode::MediumRaw
102            || keyboard_mode == KeyboardMode::Raw
103            || keyboard_mode == KeyboardMode::Off)
104            && !matches!(key_sym, KeySym::Modifier(_))
105        {
106            // FIXME: For now, we don't support SAK (secure attention key).
107            // SAK is allowed even in Raw mode.
108            return;
109        }
110
111        // Ignore key release events except for modifier keys
112        if key_status == KeyStatus::Released && !matches!(key_sym, KeySym::Modifier(_)) {
113            return;
114        }
115
116        match key_sym {
117            KeySym::Char(ch) => self.push_input(ch, &console_callbacks),
118            KeySym::Letter(ch) => self.handle_letter(ch, &console_callbacks),
119            KeySym::Meta(ch) => self.handle_meta(ch, &console_callbacks),
120            KeySym::Modifier(modifier_key) => {
121                self.handle_modifier(modifier_key, key_status == KeyStatus::Pressed)
122            }
123            KeySym::Numpad(numpad_key) => self.handle_numpad(numpad_key, &console_callbacks),
124            KeySym::Function(id) => self.handle_function(id, &console_callbacks),
125            KeySym::Cursor(cursor_key) => self.handle_cursor(cursor_key, &console_callbacks),
126            KeySym::Special(handler) => self.handle_special(handler, &console_callbacks),
127            KeySym::SwitchVt(_) => {
128                // TODO: Implement virtual terminal switching
129                ostd::warn!("Switching virtual terminal is not implemented yet");
130            }
131            KeySym::AltNumpad(_) => {
132                // TODO: Implement Alt+Numpad input
133                ostd::warn!("Alt+Numpad input is not implemented yet");
134            }
135            KeySym::NoOp => {
136                // Do nothing
137            }
138        }
139    }
140
141    fn keycode_to_modifier_key(&self, keycode: KeyCode) -> Option<ModifierKey> {
142        match keycode {
143            KeyCode::LeftShift | KeyCode::RightShift => Some(ModifierKey::Shift),
144            KeyCode::LeftCtrl | KeyCode::RightCtrl => Some(ModifierKey::Ctrl),
145            KeyCode::LeftAlt | KeyCode::RightAlt => Some(ModifierKey::Alt),
146            _ => None,
147        }
148    }

150    /// Handles key events in Medium Raw mode.
151    ///
152    /// Reference: <https://elixir.bootlin.com/linux/v6.13/source/drivers/tty/vt/keyboard.c#L1436-L1454>
153    fn handle_medium_mode(
154        &self,
155        keycode: KeyCode,
156        key_status: KeyStatus,
157        console_callbacks: &ConsoleCallbacks,
158    ) {
159        const UP_FLAG: u8 = 0x80;
160
161        let up_bit = if matches!(key_status, KeyStatus::Pressed) {
162            0
163        } else {
164            UP_FLAG
165        };
166        let kc = keycode as u16;
167
168        if kc < 128 {
169            console_callbacks.trigger_callbacks(&[(kc as u8) | up_bit]);
170        } else {
171            let b0 = up_bit;
172            let b1 = ((kc >> 7) as u8) | UP_FLAG;
173            let b2 = (kc as u8) | UP_FLAG;
174            console_callbacks.trigger_callbacks(&[b0, b1, b2]);
175        }
176    }
177
178    fn handle_letter(&self, ch: char, console_callbacks: &ConsoleCallbacks) {
179        let caps_on = LOCK_KEYS_STATE.flags().contains(LockKeyFlags::CAPS_LOCK);
180
181        let out = if caps_on && ch.is_ascii_alphabetic() {
182            if ch.is_ascii_lowercase() {
183                ch.to_ascii_uppercase()
184            } else {
185                ch.to_ascii_lowercase()
186            }
187        } else {
188            ch
189        };
190
191        self.push_input(out, console_callbacks);
192    }
193
194    fn handle_meta(&self, ch: char, console_callbacks: &ConsoleCallbacks) {
195        debug_assert!(ch.is_ascii());
196
197        let mode_flags = console_callbacks.keyboard_mode_flags();
198        if mode_flags.contains(KeyboardModeFlags::META) {
199            console_callbacks.trigger_callbacks(&[0x1b, ch as u8]);
200        } else {
201            console_callbacks.trigger_callbacks(&[ch as u8]);
202        }
203    }
204
205    fn handle_modifier(&self, modifier_key: ModifierKey, is_pressed: bool) {
206        match modifier_key {
207            ModifierKey::Shift => {
208                if is_pressed {
209                    MODIFIER_KEYS_STATE.press(ModifierKeyFlags::SHIFT);
210                } else {
211                    MODIFIER_KEYS_STATE.release(ModifierKeyFlags::SHIFT);
212                }
213            }
214            ModifierKey::Ctrl => {
215                if is_pressed {
216                    MODIFIER_KEYS_STATE.press(ModifierKeyFlags::CTRL);
217                } else {
218                    MODIFIER_KEYS_STATE.release(ModifierKeyFlags::CTRL);
219                }
220            }
221            ModifierKey::Alt => {
222                if is_pressed {
223                    MODIFIER_KEYS_STATE.press(ModifierKeyFlags::ALT);
224                } else {
225                    MODIFIER_KEYS_STATE.release(ModifierKeyFlags::ALT);
226                }
227            }
228        }
229    }
230
231    fn handle_numpad(&self, numpad_key: NumpadKey, console_callbacks: &ConsoleCallbacks) {
232        let mode_flags = console_callbacks.keyboard_mode_flags();
233
234        let shift_down = MODIFIER_KEYS_STATE
235            .flags()
236            .contains(ModifierKeyFlags::SHIFT);
237        let num_lock_on = LOCK_KEYS_STATE.flags().contains(LockKeyFlags::NUM_LOCK);
238        let application_mode = mode_flags.contains(KeyboardModeFlags::APPLICATION);
239
240        if application_mode && !shift_down {
241            match numpad_key {
242                NumpadKey::Num0 => console_callbacks.trigger_callbacks(b"\x1bOp"), // p
243                NumpadKey::Num1 => console_callbacks.trigger_callbacks(b"\x1bOq"), // q
244                NumpadKey::Num2 => console_callbacks.trigger_callbacks(b"\x1bOr"), // r
245                NumpadKey::Num3 => console_callbacks.trigger_callbacks(b"\x1bOs"), // s
246                NumpadKey::Num4 => console_callbacks.trigger_callbacks(b"\x1bOt"), // t
247                NumpadKey::Num5 => console_callbacks.trigger_callbacks(b"\x1bOu"), // u
248                NumpadKey::Num6 => console_callbacks.trigger_callbacks(b"\x1bOv"), // v
249                NumpadKey::Num7 => console_callbacks.trigger_callbacks(b"\x1bOw"), // w
250                NumpadKey::Num8 => console_callbacks.trigger_callbacks(b"\x1bOx"), // x
251                NumpadKey::Num9 => console_callbacks.trigger_callbacks(b"\x1bOy"), // y
252                NumpadKey::Plus => console_callbacks.trigger_callbacks(b"\x1bOl"), // l
253                NumpadKey::Minus => console_callbacks.trigger_callbacks(b"\x1bOS"), // S
254                NumpadKey::Asterisk => console_callbacks.trigger_callbacks(b"\x1bOR"), // R
255                NumpadKey::Slash => console_callbacks.trigger_callbacks(b"\x1bOQ"), // Q
256                NumpadKey::Enter => console_callbacks.trigger_callbacks(b"\x1bOM"), // M
257                NumpadKey::Dot => console_callbacks.trigger_callbacks(b"\x1bOn"),  // n
258            }
259            return;
260        }
261
262        if !num_lock_on {
263            match numpad_key {
264                NumpadKey::Num0 => self.handle_function(FuncId::Insert, console_callbacks),
265                NumpadKey::Num1 => self.handle_function(FuncId::Select, console_callbacks),
266                NumpadKey::Num2 => self.handle_cursor(CursorKey::Down, console_callbacks),
267                NumpadKey::Num3 => self.handle_function(FuncId::Next, console_callbacks),
268                NumpadKey::Num4 => self.handle_cursor(CursorKey::Left, console_callbacks),
269                NumpadKey::Num5 => {
270                    if application_mode {
271                        console_callbacks.trigger_callbacks(b"\x1bOG");
272                    } else {
273                        console_callbacks.trigger_callbacks(b"\x1b[G");
274                    }
275                }
276                NumpadKey::Num6 => self.handle_cursor(CursorKey::Right, console_callbacks),
277                NumpadKey::Num7 => self.handle_function(FuncId::Find, console_callbacks),
278                NumpadKey::Num8 => self.handle_cursor(CursorKey::Up, console_callbacks),
279                NumpadKey::Num9 => self.handle_function(FuncId::Prior, console_callbacks),
280                NumpadKey::Dot => self.handle_function(FuncId::Remove, console_callbacks),
281                _ => {}
282            }
283            return;
284        }
285
286        let ch = match numpad_key {
287            NumpadKey::Num0 => '0',
288            NumpadKey::Num1 => '1',
289            NumpadKey::Num2 => '2',
290            NumpadKey::Num3 => '3',
291            NumpadKey::Num4 => '4',
292            NumpadKey::Num5 => '5',
293            NumpadKey::Num6 => '6',
294            NumpadKey::Num7 => '7',
295            NumpadKey::Num8 => '8',
296            NumpadKey::Num9 => '9',
297            NumpadKey::Dot => '.',
298            NumpadKey::Enter => '\r',
299            NumpadKey::Plus => '+',
300            NumpadKey::Minus => '-',
301            NumpadKey::Asterisk => '*',
302            NumpadKey::Slash => '/',
303        };
304
305        console_callbacks.trigger_callbacks(&[ch as u8]);
306
307        if matches!(numpad_key, NumpadKey::Enter) && mode_flags.contains(KeyboardModeFlags::CRLF) {
308            console_callbacks.trigger_callbacks(b"\n");
309        }
310    }
311
312    fn handle_function(&self, id: FuncId, console_callbacks: &ConsoleCallbacks) {
313        if let Some(seq) = id.to_function_string() {
314            console_callbacks.trigger_callbacks(seq);
315        }
316    }
317
318    fn handle_cursor(&self, cursor_key: CursorKey, console_callbacks: &ConsoleCallbacks) {
319        let cursor_key_mode = console_callbacks
320            .keyboard_mode_flags()
321            .contains(KeyboardModeFlags::CURSOR_KEY);
322
323        if cursor_key_mode {
324            match cursor_key {
325                CursorKey::Up => console_callbacks.trigger_callbacks(b"\x1bOA"),
326                CursorKey::Down => console_callbacks.trigger_callbacks(b"\x1bOB"),
327                CursorKey::Left => console_callbacks.trigger_callbacks(b"\x1bOD"),
328                CursorKey::Right => console_callbacks.trigger_callbacks(b"\x1bOC"),
329            }
330        } else {
331            match cursor_key {
332                CursorKey::Up => console_callbacks.trigger_callbacks(b"\x1b[A"),
333                CursorKey::Down => console_callbacks.trigger_callbacks(b"\x1b[B"),
334                CursorKey::Left => console_callbacks.trigger_callbacks(b"\x1b[D"),
335                CursorKey::Right => console_callbacks.trigger_callbacks(b"\x1b[C"),
336            }
337        }
338    }
339
340    fn handle_special(&self, handler: SpecialHandler, console_callbacks: &ConsoleCallbacks) {
341        match handler {
342            SpecialHandler::ToggleCapsLock => {
343                LOCK_KEYS_STATE.toggle(LockKeyFlags::CAPS_LOCK);
344            }
345            SpecialHandler::ToggleNumLock => {
346                let application_mode = console_callbacks
347                    .keyboard_mode_flags()
348                    .contains(KeyboardModeFlags::APPLICATION);
349                if application_mode {
350                    console_callbacks.trigger_callbacks(b"\x1bOP");
351                } else {
352                    LOCK_KEYS_STATE.toggle(LockKeyFlags::NUM_LOCK);
353                }
354            }
355            SpecialHandler::ToggleBareNumLock => {
356                LOCK_KEYS_STATE.toggle(LockKeyFlags::NUM_LOCK);
357            }
358            SpecialHandler::ToggleScrollLock => {
359                // TODO: Scroll Lock will affect the scrolling behavior of the console.
360                // For now, we just toggle the state.
361                LOCK_KEYS_STATE.toggle(LockKeyFlags::SCROLL_LOCK);
362            }
363            SpecialHandler::Enter => {
364                console_callbacks.trigger_callbacks(b"\r");
365                if console_callbacks
366                    .keyboard_mode_flags()
367                    .contains(KeyboardModeFlags::CRLF)
368                {
369                    console_callbacks.trigger_callbacks(b"\n");
370                }
371            }
372            SpecialHandler::DecreaseConsole
373            | SpecialHandler::IncreaseConsole
374            | SpecialHandler::ScrollBackward
375            | SpecialHandler::ScrollForward
376            | SpecialHandler::ShowMem
377            | SpecialHandler::ShowState
378            | SpecialHandler::Compose
379            | SpecialHandler::Reboot => {
380                ostd::warn!("VT keyboard action {:?} is not implemented yet", handler);
381            }
382        }
383    }

385    /// Adds the character to the input buffer of the virtual terminal.
386    ///
387    /// Reference: <https://elixir.bootlin.com/linux/v6.13/source/drivers/tty/vt/keyboard.c#L675>.
388    fn push_input(&self, ch: char, console_callbacks: &ConsoleCallbacks) {
389        let keyboard_mode = console_callbacks.keyboard_mode();
390
391        if keyboard_mode == KeyboardMode::Unicode {
392            let mut buf = [0u8; 4];
393            let s = ch.encode_utf8(&mut buf);
394            console_callbacks.trigger_callbacks(s.as_bytes());
395        } else if ch.is_ascii() {
396            console_callbacks.trigger_callbacks(&[ch as u8]);
397        }
398    }
399}

401impl InputHandler for VtKeyboardHandler {
402    fn handle_events(&self, events: &[InputEvent]) {
403        for event in events {
404            match event {
405                InputEvent::Key(keycode, key_status) => {
406                    self.handle_key_event(*keycode, *key_status)
407                }
408                InputEvent::Sync(_) => {
409                    // Nothing to do
410                }
411                _ => {
412                    ostd::warn!(
413                        "Virtual terminal keyboard handler received unsupported event: {:?}",
414                        event
415                    );
416                }
417            }
418        }
419    }
420}

422static REGISTERED_INPUT_HANDLER_CLASS: Once<RegisteredInputHandlerClass> = Once::new();
423
424pub(super) fn init_in_first_process() {
425    if FRAMEBUFFER_CONSOLE.get().is_none() {
426        return;
427    }
428
429    REGISTERED_INPUT_HANDLER_CLASS.call_once(|| {
430        let handler_class = Arc::new(VtKeyboardHandlerClass);
431        aster_input::register_handler_class(handler_class)
432    });
