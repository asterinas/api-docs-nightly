13/// The symbolic representation of a key under a given modifier state.
14///
15/// Reference: <https://elixir.bootlin.com/linux/v6.17.4/source/drivers/tty/vt/keyboard.c#L69-L77>
16#[derive(Clone, Copy, Debug, Eq, PartialEq)]
17pub(in crate::device::tty::vt::keyboard) enum KeySym {
18    /// A direct character output (similar to Linux `k_self`).
19    ///
20    /// It is used for characters that do not depend on Caps Lock state
21    /// (digits, punctuation, etc.).
22    Char(char),
23    /// An alphabetic character whose case may be affected by Caps Lock
24    /// (similar to Linux `k_lowercase`).
25    Letter(char),
26    /// A key with Meta (usually Alt) pressed (similar to Linux `k_meta`).
27    ///
28    /// The handler may emit an ESC-prefixed sequence depending on
29    /// the VT keyboard mode flags.
30    Meta(char),
31    /// A pure modifier key (similar to Linux `k_shift`).
32    Modifier(ModifierKey),
33    /// A numpad key (similar to Linux `k_pad`).
34    Numpad(NumpadKey),
35    /// A function key that may emit an ESC-prefixed function-string
36    /// (similar to Linux `k_fn`).
37    Function(FuncId),
38    /// A numpad digit key with Alt pressed, used for ASCII code input
39    /// (similar to Linux `k_ascii`).
40    ///
41    /// The handler is responsible for accumulating digits while Alt
42    /// is held and emitting the final character when the sequence
43    /// is committed (typically on Alt release).
44    AltNumpad(u8),
45    /// A cursor key (similar to Linux `k_cur`).
46    Cursor(CursorKey),
47    /// A special action (similar to Linux `k_spec`).
48    Special(SpecialHandler),
49    /// A virtual terminal switch (similar to Linux `k_cons`).
50    SwitchVt(VtIndex),
51    /// No operation.
52    NoOp,
53}

55/// Special key actions that are handled by the VT keyboard layer.
56///
57/// Reference: <https://elixir.bootlin.com/linux/v6.17.4/source/drivers/tty/vt/keyboard.c#L84-L89>
58#[derive(Clone, Copy, Debug, Eq, PartialEq)]
59pub(in crate::device::tty::vt::keyboard) enum SpecialHandler {
60    /// Toggle Caps Lock state (similar to Linux `fn_caps_toggle`).
61    ToggleCapsLock,
62    /// Toggle Num Lock state, possibly with special handling in
63    /// application-keypad mode (similar to Linux `fn_num`).
64    ToggleNumLock,
65    /// Toggle Num Lock state without any application-keypad side
66    /// effects (similar to Linux `fn_bare_num`).
67    ToggleBareNumLock,
68    /// Toggle Scroll Lock state (similar to Linux `fn_hold`).
69    ToggleScrollLock,
70    /// Print memory usage information to the console
71    /// (similar to Linux `fn_show_mem`).
72    ShowMem,
73    /// Print the current process state to the console
74    /// (similar to Linux `fn_show_state`).
75    ShowState,
76    /// Start a compose sequence (similar to Linux `fn_compose`).
77    Compose,
78    /// Trigger a reboot of the system (similar to Linux `fn_boot_it`).
79    Reboot,
80    /// Scroll the console history backward (similar to Linux
81    /// `fn_scroll_back`).
82    ScrollBackward,
83    /// Scroll the console history forward (similar to Linux
84    /// `fn_scroll_forward`).
85    ScrollForward,
86    /// Switch to the previous VT (similar to Linux `fn_dec_console`).
87    DecreaseConsole,
88    /// Switch to the next VT (similar to Linux `fn_inc_console`).
89    IncreaseConsole,
90    /// Handle the Enter key (similar to Linux `fn_enter`).
91    Enter,
92}

94/// The key mapping tables for different modifier key combinations.
95struct KeyMaps {
96    maps: Box<[Box<[KeySym]>]>,
97}
98
99impl Default for KeyMaps {
100    /// Creates a default table with standard key mappings.
101    ///
102    /// Reference: <https://elixir.bootlin.com/linux/v6.13/source/drivers/tty/vt/defkeymap.map>
103    /// and <https://elixir.bootlin.com/linux/v6.13/source/drivers/tty/vt/defkeymap.c_shipped>
104    fn default() -> Self {
105        let mut key_maps = Self::new(Self::MOD_COMBINATIONS, Self::KEY_COUNT);
106        key_maps.map_letter();
107        key_maps.map_char();
108        key_maps.map_function();
109        key_maps.map_switch_vt();
110        key_maps.map_meta();
111        key_maps.map_numpad_keys();
112        key_maps.map_enter();
113        key_maps.map_compose();
114        key_maps.map_modifier();
115        key_maps.map_reboot();
116        key_maps.map_scroll();
117        key_maps.map_cursor_keys();
118        key_maps.map_lock_keys();
119        key_maps
120    }
121}

123type KeyBind = (KeyCode, KeySym);
124
125impl KeyMaps {
126    /// Number of keycodes handled by the default VT keymap.
127    ///
128    /// Note:
129    /// - This table indexes by the numeric value of `KeyCode`.
130    /// - Key codes outside this range will map to `KeySym::NoOp`.
131    const KEY_COUNT: usize = 128;
132
133    /// Number of modifier combinations supported by the default VT keymap.
134    ///
135    /// We index keymaps by the raw 3-bit modifier mask (SHIFT|ALT|CTRL),
136    /// thus the table has 2^3 = 8 layers.
137    const MOD_COMBINATIONS: usize = 8;
138
139    fn new(mods_num: usize, keycode_num: usize) -> Self {
140        Self {
141            maps: (0..mods_num)
142                .map(|_| vec![KeySym::NoOp; keycode_num].into_boxed_slice())
143                .collect::<Vec<_>>()
144                .into_boxed_slice(),
145        }
146    }
147
148    fn get_keysym(&self, mods: ModifierKeyFlags, key_code: KeyCode) -> KeySym {
149        let mods_index = mods.bits() as usize;
150        let key_index = key_code as usize;
151        self.maps
152            .get(mods_index)
153            .and_then(|map| map.get(key_index).copied())
154            .unwrap_or(KeySym::NoOp)
155    }
156
157    fn set_keysym(&mut self, mods: ModifierKeyFlags, key_code: KeyCode, key_sym: KeySym) {
158        let mods_index = mods.bits() as usize;
159        let key_index = key_code as usize;
160        if let Some(entry) = self
161            .maps
162            .get_mut(mods_index)
163            .and_then(|map| map.get_mut(key_index))
164        {
165            *entry = key_sym;
166        }
167    }
168}

794    fn apply_key_binds(&mut self, mods_list: &[ModifierKeyFlags], binds: &[KeyBind]) {
795        for &mods in mods_list {

798            }

805/// Looks up the [`KeySym`] for a given modifier mask and key code.
806pub(in crate::device::tty::vt::keyboard) fn get_keysym(
807    mods: ModifierKeyFlags,
808    key_code: KeyCode,
809) -> KeySym {
810    KEYMAPS
811        .get()
812        .expect("`KEYMAPS` is not initialized")
813        .get_keysym(mods, key_code)
814}
