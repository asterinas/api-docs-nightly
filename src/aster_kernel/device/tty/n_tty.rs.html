24/// The driver for VT (virtual terminal) devices.
25//
26// TODO: This driver needs to support more features for future VT management.
27#[derive(Clone)]
28pub struct VtDriver {
29    console: Arc<dyn AnyConsoleDevice>,
30}
31
32impl TtyDriver for VtDriver {
33    // Reference: <https://elixir.bootlin.com/linux/v6.17/source/include/uapi/linux/major.h#L18>.
34    const DEVICE_MAJOR_ID: u32 = 4;
35
36    fn devtmpfs_path(&self, index: u32) -> Option<String> {
37        Some(format!("tty{}", index))
38    }
39
40    fn open(tty: Arc<Tty<Self>>) -> Result<Box<dyn FileIo>> {
41        Ok(Box::new(TtyFile(tty)))
42    }
43
44    fn push_output(&self, chs: &[u8]) -> Result<usize> {
45        self.console.send(chs);
46        Ok(chs.len())
47    }
48
49    fn echo_callback(&self) -> impl FnMut(&[u8]) + '_ {
50        |chs| self.console.send(chs)
51    }
52
53    fn can_push(&self) -> bool {
54        true
55    }
56
57    fn notify_input(&self) {}
58
59    fn console(&self) -> Option<&dyn AnyConsoleDevice> {
60        Some(&*self.console)
61    }
62
63    fn on_termios_change(&self, _old_termios: &CTermios, _new_termios: &CTermios) {}
64}

148static HVC0: Once<Arc<Tty<HvcDriver>>> = Once::new();

150/// Returns the `tty1` device.
151///
152/// # Panics
153///
154/// This function will panic if the `tty1` device has not been initialized.
155pub fn tty1_device() -> &'static Arc<Tty<VtDriver>> {
156    TTY1.get().unwrap()
157}

159/// Returns the `hvc0` device.
160///
161/// Returns `None` if the device is not found nor initialized.
162pub fn hvc0_device() -> Option<&'static Arc<Tty<HvcDriver>>> {
163    HVC0.get()
164}

166pub(super) fn init_in_first_process() -> Result<()> {
167    let devices = aster_console::all_devices();
168
169    // Initialize the `tty1` device.
170
171    let fb_console = devices
172        .iter()
173        .find(|(name, _)| name.as_str() == aster_framebuffer::CONSOLE_NAME)
174        .map(|(_, device)| device.clone())
175        .unwrap_or_else(|| Arc::new(DummyFramebufferConsole));
176
177    let driver = VtDriver {
178        console: fb_console.clone(),
179    };
180    let tty1 = Tty::new(1, driver);
181
182    TTY1.call_once(|| tty1.clone());
183    char::register(tty1.clone())?;
184
185    fb_console.register_callback(Box::leak(Box::new(
186        move |mut reader: VmReader<Infallible>| {
187            let mut chs = vec![0u8; reader.remain()];
188            reader.read(&mut VmWriter::from(chs.as_mut_slice()));
189            let _ = tty1.push_input(chs.as_slice());
190        },
191    )));
192
193    // Initialize the `hvc0` device if the virtio console is available.
194
195    let virtio_console = devices
196        .iter()
197        .find(|(name, _)| name.as_str() == aster_virtio::device::console::DEVICE_NAME)
198        .map(|(_, device)| device.clone());
199
200    if let Some(virtio_console) = virtio_console {
201        let driver = HvcDriver {
202            console: virtio_console.clone(),
203        };
204        let hvc0 = Tty::new(0, driver);
205
206        HVC0.call_once(|| hvc0.clone());
207        char::register(hvc0.clone())?;
208
209        virtio_console.register_callback(Box::leak(Box::new(
210            move |mut reader: VmReader<Infallible>| {
211                let mut chs = vec![0u8; reader.remain()];
212                reader.read(&mut VmWriter::from(chs.as_mut_slice()));
213                let _ = hvc0.push_input(chs.as_slice());
214            },
215        )));
216    }
217
218    Ok(())
219}
