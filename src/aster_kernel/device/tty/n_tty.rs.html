24/// The driver for VT (virtual terminal) devices.
25//
26// TODO: This driver needs to support more features for future VT management.
27#[derive(Clone)]
28pub struct VtDriver {
29    console: Arc<dyn AnyConsoleDevice>,
30}
31
32impl TtyDriver for VtDriver {
33    // Reference: <https://elixir.bootlin.com/linux/v6.17/source/include/uapi/linux/major.h#L18>.
34    const DEVICE_MAJOR_ID: u32 = 4;
35
36    fn devtmpfs_path(&self, index: u32) -> Option<String> {
37        Some(format!("tty{}", index))
38    }
39
40    fn open(tty: Arc<Tty<Self>>) -> Result<Box<dyn FileIo>> {
41        Ok(Box::new(TtyFile(tty)))
42    }
43
44    fn push_output(&self, chs: &[u8]) -> Result<usize> {
45        self.console.send(chs);
46        Ok(chs.len())
47    }
48
49    fn echo_callback(&self) -> impl FnMut(&[u8]) + '_ {
50        |chs| self.console.send(chs)
51    }
52
53    fn can_push(&self) -> bool {
54        true
55    }
56
57    fn notify_input(&self) {}
58
59    fn console(&self) -> Option<&dyn AnyConsoleDevice> {
60        Some(&*self.console)
61    }
62
63    fn on_termios_change(&self, _old_termios: &CTermios, _new_termios: &CTermios) {}
64}

157impl<D: TtyDriver> InodeIo for TtyFile<D> {
158    fn read_at(
159        &self,
160        _offset: usize,
161        writer: &mut VmWriter,
162        status_flags: StatusFlags,
163    ) -> Result<usize> {
164        self.0.read(writer, status_flags)
165    }
166
167    fn write_at(
168        &self,
169        _offset: usize,
170        reader: &mut VmReader,
171        status_flags: StatusFlags,
172    ) -> Result<usize> {
173        self.0.write(reader, status_flags)
174    }
175}

219pub(super) fn init_in_first_process() -> Result<()> {
220    let devices = aster_console::all_devices();
221
222    // Initialize the `tty1` device.
223
224    let fb_console = devices
225        .iter()
226        .find(|(name, _)| name.as_str() == aster_framebuffer::CONSOLE_NAME)
227        .map(|(_, device)| device.clone())
228        .unwrap_or_else(|| Arc::new(DummyFramebufferConsole));
229
230    let driver = VtDriver {
231        console: fb_console.clone(),
232    };
233    let tty1 = Tty::new(1, driver);
234
235    TTY1.call_once(|| tty1.clone());
236    char::register(tty1.clone())?;
237
238    fb_console.register_callback(Box::leak(Box::new(
239        move |mut reader: VmReader<Infallible>| {
240            let mut chs = vec![0u8; reader.remain()];
241            reader.read(&mut VmWriter::from(chs.as_mut_slice()));
242            let _ = tty1.push_input(chs.as_slice());
243        },
244    )));
245
246    // Initialize the `ttyS0` device if the serial console is available.
247
248    let serial_console = devices
249        .iter()
250        .find(|(name, _)| name.as_str() == aster_uart::CONSOLE_NAME)
251        .map(|(_, device)| device.clone());
252
253    if let Some(serial_console) = serial_console {
254        let driver = SerialDriver {
255            console: serial_console.clone(),
256        };
257        let serial0 = Tty::new(SerialDriver::MINOR_ID_BASE, driver);
258
259        SERIAL0.call_once(|| serial0.clone());
260        char::register(serial0.clone())?;
261
262        serial_console.register_callback(Box::leak(Box::new(
263            move |mut reader: VmReader<Infallible>| {
264                let mut chs = vec![0u8; reader.remain()];
265                reader.read(&mut VmWriter::from(chs.as_mut_slice()));
266                let _ = serial0.push_input(chs.as_slice());
267            },
268        )));
269    }
270
271    // Initialize the `hvc0` device if the virtio console is available.
272
273    let virtio_console = devices
274        .iter()
275        .find(|(name, _)| name.as_str() == aster_virtio::device::console::DEVICE_NAME)
276        .map(|(_, device)| device.clone());
277
278    if let Some(virtio_console) = virtio_console {
279        let driver = HvcDriver {
280            console: virtio_console.clone(),
281        };
282        let hvc0 = Tty::new(0, driver);
283
284        HVC0.call_once(|| hvc0.clone());
285        char::register(hvc0.clone())?;
286
287        virtio_console.register_callback(Box::leak(Box::new(
288            move |mut reader: VmReader<Infallible>| {
289                let mut chs = vec![0u8; reader.remain()];
290                reader.read(&mut VmWriter::from(chs.as_mut_slice()));
291                let _ = hvc0.push_input(chs.as_slice());
292            },
293        )));
294    }
295
296    Ok(())
