30/// Corresponds to `/dev/tty0` in the file system. This device represents the active virtual
31/// terminal.
32pub struct Tty0Device;
33
34impl Tty0Device {
35    fn active_vt(&self) -> &Arc<Tty<VtDriver>> {
36        // Currently there is only one virtual terminal `tty1`.
37        tty1_device()
38    }
39}
40
41impl Device for Tty0Device {
42    fn type_(&self) -> DeviceType {
43        DeviceType::Char
44    }
45
46    fn id(&self) -> DeviceId {
47        DeviceId::new(MajorId::new(4), MinorId::new(0))
48    }
49
50    fn devtmpfs_path(&self) -> Option<String> {
51        Some("tty0".into())
52    }
53
54    fn open(&self) -> Result<Box<dyn FileIo>> {
55        self.active_vt().open()
56    }
57}
58
59impl Terminal for Tty0Device {
60    fn job_control(&self) -> &JobControl {
61        self.active_vt().job_control()
62    }
63}

65/// Corresponds to `/dev/tty` in the file system. This device represents the controlling terminal
66/// of the session of the current process.
67pub struct TtyDevice;
68
69impl Device for TtyDevice {
70    fn type_(&self) -> DeviceType {
71        DeviceType::Char
72    }
73
74    fn id(&self) -> DeviceId {
75        DeviceId::new(MajorId::new(5), MinorId::new(0))
76    }
77
78    fn devtmpfs_path(&self) -> Option<String> {
79        Some("tty".into())
80    }
81
82    fn open(&self) -> Result<Box<dyn FileIo>> {
83        let Some(terminal) = current!().terminal() else {
84            return_errno_with_message!(
85                Errno::ENOTTY,
86                "the process does not have a controlling terminal"
87            );
88        };
89
90        terminal.open()
91    }
92}

94/// Corresponds to `/dev/console` in the file system. This device represents a console to which
95/// system messages will be sent.
96pub struct SystemConsole {
97    inner: Arc<dyn Terminal>,
98}
99
100impl SystemConsole {
101    /// Returns the singleton instance of the console device.
102    pub fn singleton() -> &'static Arc<SystemConsole> {
103        static INSTANCE: Once<Arc<SystemConsole>> = Once::new();
104
105        INSTANCE.call_once(|| {
106            // TODO: Support specifying multiple TTY devices, e.g., "console=hvc0 console=tty0".
107            let console_name = KCMDLINE
108                .get()
109                .unwrap()
110                .get_console_names()
111                .first()
112                .map(String::as_str)
113                .unwrap_or("tty0");
114
115            let device = match console_name {
116                "tty0" => Some(Arc::new(Tty0Device) as _),
117                "ttyS0" => serial0_device().cloned().map(|device| device as _),
118                "hvc0" => hvc0_device().cloned().map(|device| device as _),
119                _ => None,
120            };
121            let inner = device.unwrap_or_else(|| {
122                warn!(
123                    "'{}' console not found, falling back to 'tty0'",
124                    console_name
125                );
126                Arc::new(Tty0Device) as _
127            });
128
129            Arc::new(Self { inner })
130        })
131    }
132
133    /// Returns the terminal associated with the console device.
134    pub fn terminal(&self) -> &Arc<dyn Terminal> {
135        &self.inner
136    }
137}
138
139impl Device for SystemConsole {
140    fn type_(&self) -> DeviceType {
141        DeviceType::Char
142    }
143
144    fn id(&self) -> DeviceId {
145        DeviceId::new(MajorId::new(5), MinorId::new(1))
146    }
147
148    fn devtmpfs_path(&self) -> Option<String> {
149        Some("console".into())
150    }
151
152    fn open(&self) -> Result<Box<dyn FileIo>> {
153        self.inner.open()
154    }
155}
156
157pub(super) fn init_in_first_process() -> Result<()> {
158    char::register(Arc::new(Tty0Device))?;
159    char::register(Arc::new(TtyDevice))?;
160    char::register(SystemConsole::singleton().clone())?;
161
162    Ok(())
163}
