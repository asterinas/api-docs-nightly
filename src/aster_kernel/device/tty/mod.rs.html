47const IO_CAPACITY: usize = 4096;

49/// A teletyper (TTY).
50///
51/// This abstracts the general functionality of a TTY in a way that
52///  - Any input device driver can use [`Tty::push_input`] to push input characters, and users can
53///    [`Tty::read`] from the TTY;
54///  - Users can also [`Tty::write`] output characters to the TTY and the output device driver will
55///    receive the characters from [`TtyDriver::push_output`] where the generic parameter `D` is
56///    the [`TtyDriver`].
57///
58/// ```text
59/// +------------+     +-------------+
60/// |input device|     |output device|
61/// |   driver   |     |   driver    |
62/// +-----+------+     +------^------+
63///       |                   |
64///       |     +-------+     |
65///       +----->  TTY  +-----+
66///             +-------+
67/// Tty::push_input   D::push_output
68/// ```
69pub struct Tty<D> {
70    index: u32,
71    driver: D,
72    ldisc: SpinLock<LineDiscipline, LocalIrqDisabled>,
73    job_control: JobControl,
74    pollee: Pollee,
75    tty_flags: TtyFlags,
76    weak_self: Weak<Self>,
77}
78
79impl<D> Tty<D> {
80    pub(super) fn new(index: u32, driver: D) -> Arc<Self> {
81        Arc::new_cyclic(move |weak_ref| Tty {
82            index,
83            driver,
84            ldisc: SpinLock::new(LineDiscipline::new()),
85            job_control: JobControl::new(),
86            pollee: Pollee::new(),
87            tty_flags: TtyFlags::new(),
88            weak_self: weak_ref.clone(),
89        })
90    }
91
92    pub fn index(&self) -> u32 {
93        self.index
94    }
95
96    pub(super) fn driver(&self) -> &D {
97        &self.driver
98    }
99
100    /// Returns whether new characters can be pushed into the input buffer.
101    ///
102    /// This method should return `false` if the input buffer is full.
103    pub(super) fn can_push(&self) -> bool {
104        !self.ldisc.lock().is_full()
105    }
106
107    /// Notifies that the output buffer now has room for new characters.
108    ///
109    /// This method should be called when the state of [`TtyDriver::can_push`] changes from `false`
110    /// to `true`.
111    pub(super) fn notify_output(&self) {
112        self.pollee.notify(IoEvents::OUT);
113    }
114
115    /// Notifies that the other end has been closed.
116    pub(super) fn notify_hup(&self) {
117        self.pollee.notify(IoEvents::ERR | IoEvents::HUP);
118    }
119
120    /// Returns the TTY flags.
121    pub(super) fn tty_flags(&self) -> &TtyFlags {
122        &self.tty_flags
123    }
124}

126impl<D: TtyDriver> Tty<D> {
127    /// Pushes characters into the output buffer.
128    ///
129    /// This method returns the number of bytes pushed or fails with an error if no bytes can be
130    /// pushed because the buffer is full.
131    pub fn push_input(&self, chs: &[u8]) -> Result<usize> {
132        let mut ldisc = self.ldisc.lock();
133        let mut echo = self.driver.echo_callback();
134
135        let mut len = 0;
136        for ch in chs {
137            let res = ldisc.push_char(
138                *ch,
139                |signum| {
140                    if let Some(foreground) = self.job_control.foreground() {
141                        broadcast_signal_async(Arc::downgrade(&foreground), signum);
142                    }
143                },
144                &mut echo,
145            );
146            if res.is_err() && len == 0 {
147                return_errno_with_message!(Errno::EAGAIN, "the line discipline is full");
148            } else if res.is_err() {
149                break;
150            } else {
151                len += 1;
152            }
153        }
154
155        self.pollee.notify(IoEvents::IN | IoEvents::RDNORM);
156        Ok(len)
157    }
158
159    fn check_io_events(&self) -> IoEvents {
160        let mut events = IoEvents::empty();
161
162        if self.ldisc.lock().buffer_len() > 0 {
163            events |= IoEvents::IN | IoEvents::RDNORM;
164        }
165
166        if self.driver.can_push() {
167            events |= IoEvents::OUT;
168        }
169
170        if self.tty_flags.is_other_closed() {
171            events |= IoEvents::ERR | IoEvents::HUP;
172        }
173
174        events
175    }
176}

178impl<D: TtyDriver> Tty<D> {
179    fn console(&self) -> Result<&dyn AnyConsoleDevice> {
180        self.driver.console().ok_or_else(|| {
181            Error::with_message(Errno::ENOTTY, "the TTY is not connected to a console")
182        })
183    }
184
185    fn handle_set_font(&self, font_op: &CFontOp) -> Result<()> {
186        let console = self.console()?;
187
188        let CFontOp {
189            op,
190            flags: _,
191            width,
192            height,
193            charcount,
194            data,
195            ..
196        } = font_op;
197
198        let vpitch = match *op {
199            CFontOp::OP_SET => CFontOp::NONTALL_VPITCH,
200            CFontOp::OP_SET_TALL => font_op.height,
201            CFontOp::OP_SET_DEFAULT => {
202                return console_set_font(console, BitmapFont::new_basic8x8());
203            }
204            _ => return_errno_with_message!(Errno::EINVAL, "the font operation is invalid"),
205        };
206
207        if *width == 0
208            || *height == 0
209            || *width > CFontOp::MAX_WIDTH
210            || *height > CFontOp::MAX_HEIGHT
211            || *charcount > CFontOp::MAX_CHARCOUNT
212            || *height > vpitch
213        {
214            return_errno_with_message!(Errno::EINVAL, "the font is invalid or too large");
215        }
216
217        let font_size = width.div_ceil(u8::BITS) * vpitch * charcount;
218        let mut font_data = vec![0; font_size as usize];
219        current_userspace!().read_bytes(*data as Vaddr, &mut font_data[..])?;
220
221        // In Linux, the most significant bit represents the first pixel, but `BitmapFont` requires
222        // the least significant bit to represent the first pixel. So now we reverse the bits.
223        font_data
224            .iter_mut()
225            .for_each(|byte| *byte = byte.reverse_bits());
226
227        let font = BitmapFont::new_with_vpitch(
228            *width as usize,
229            *height as usize,
230            vpitch as usize,
231            font_data,
232        );
233        console_set_font(console, font)?;
234
235        Ok(())
236    }
237}
238
239fn console_set_font(console: &dyn AnyConsoleDevice, font: BitmapFont) -> Result<()> {
240    use aster_console::ConsoleSetFontError;
241
242    match console.set_font(font) {
243        Ok(()) => Ok(()),
244        Err(ConsoleSetFontError::InappropriateDevice) => {
245            return_errno_with_message!(Errno::ENOTTY, "the console has no support for font setting")
246        }
247        Err(ConsoleSetFontError::InvalidFont) => {
248            return_errno_with_message!(Errno::EINVAL, "the font is invalid for the console")
249        }
250    }
251}

253impl<D: TtyDriver> Pollable for Tty<D> {
254    fn poll(&self, mask: IoEvents, poller: Option<&mut PollHandle>) -> IoEvents {
255        self.pollee
256            .poll_with(mask, poller, || self.check_io_events())
257    }

260impl<D: TtyDriver> Tty<D> {
261    pub fn read(&self, writer: &mut VmWriter, status_flags: StatusFlags) -> Result<usize> {
262        if self.tty_flags.is_other_closed() {
263            return Ok(0);
264        }
265
266        self.job_control.wait_until_in_foreground()?;
267
268        // TODO: Add support for timeout.
269        let mut buf = vec![0u8; writer.avail().min(IO_CAPACITY)];
270        let is_nonblocking = status_flags.contains(StatusFlags::O_NONBLOCK);
271        let read_len = if is_nonblocking {
272            self.ldisc.lock().try_read(&mut buf)?
273        } else {
274            self.wait_events(IoEvents::IN, None, || self.ldisc.lock().try_read(&mut buf))?
275        };
276        self.pollee.invalidate();
277        self.driver.notify_input();
278
279        // TODO: Confirm what we should do if `write_fallible` fails in the middle.
280        writer.write_fallible(&mut buf[..read_len].into())?;
281        Ok(read_len)
282    }
283
284    pub fn write(&self, reader: &mut VmReader, status_flags: StatusFlags) -> Result<usize> {
285        if self.tty_flags.is_other_closed() {
286            return_errno_with_message!(Errno::EIO, "the TTY is closed");
287        }
288
289        let mut buf = vec![0u8; reader.remain().min(IO_CAPACITY)];
290        let write_len = reader.read_fallible(&mut buf.as_mut_slice().into())?;
291
292        // TODO: Add support for timeout.
293        let is_nonblocking = status_flags.contains(StatusFlags::O_NONBLOCK);
294        let len = if is_nonblocking {
295            self.driver.push_output(&buf[..write_len])?
296        } else {
297            self.wait_events(IoEvents::OUT, None, || {
298                self.driver.push_output(&buf[..write_len])
299            })?
300        };
301        self.pollee.invalidate();
302        Ok(len)
303    }
304
305    pub fn ioctl(&self, raw_ioctl: RawIoctl) -> Result<i32> {
306        use ioctl_defs::*;
307
308        use crate::fs::utils::ioctl_defs::GetNumBytesToRead;
309
310        dispatch_ioctl!(match raw_ioctl {
311            cmd @ GetTermios => {
312                let termios = *self.ldisc.lock().termios();
313
314                cmd.write(&termios)?;
315            }
316            cmd @ SetTermios => {
317                let termios = cmd.read()?;
318
319                let mut ldisc = self.ldisc.lock();
320                let old_termios = ldisc.termios();
321                self.driver().on_termios_change(old_termios, &termios);
322                ldisc.set_termios(termios);
323            }
324            cmd @ SetTermiosWait => {
325                let termios = cmd.read()?;
326
327                // TODO: If applicable, wait for the output buffer to drain. For now, we don't need
328                // to do anything here because:
329                //  - Linux does not consider a pty to have an output buffer, so it does not drain
330                //    it. See
331                //    <https://elixir.bootlin.com/linux/v5.10.247/source/drivers/tty/pty.c#L137-L148>.
332                //  - We don't currently have an output buffer for other TTYs.
333                let mut ldisc = self.ldisc.lock();
334                let old_termios = ldisc.termios();
335                self.driver().on_termios_change(old_termios, &termios);
336                ldisc.set_termios(termios);
337            }
338            cmd @ SetTermiosFlush => {
339                let termios = cmd.read()?;
340
341                // TODO: If applicable, wait for the output buffer to drain. (See comments above.)
342                let mut ldisc = self.ldisc.lock();
343                let old_termios = ldisc.termios();
344                self.driver().on_termios_change(old_termios, &termios);
345                ldisc.set_termios(termios);
346                ldisc.drain_input();
347
348                self.pollee.invalidate();
349            }
350            cmd @ GetWinSize => {
351                let winsize = self.ldisc.lock().window_size();
352
353                cmd.write(&winsize)?;
354            }
355            cmd @ SetWinSize => {
356                let winsize = cmd.read()?;
357
358                self.ldisc.lock().set_window_size(winsize);
359            }
360            cmd @ GetPtyNumber => {
361                let idx = self.index;
362
363                cmd.write(&idx)?;
364            }
365            cmd @ GetNumBytesToRead => {
366                if self.tty_flags.is_other_closed() {
367                    return_errno_with_message!(Errno::EIO, "the TTY is closed");
368                }
369
370                let buffer_len = self.ldisc.lock().buffer_len() as i32;
371
372                cmd.write(&buffer_len)?;
373            }
374            cmd @ SetOrGetFont => {
375                let font_op = cmd.read()?;
376
377                self.handle_set_font(&font_op)?;
378            }
379            cmd @ SetGraphicsMode => {
380                let console = self.console()?;
381
382                let mode = ConsoleMode::try_from(cmd.get())?;
383                if !console.set_mode(mode) {
384                    return_errno_with_message!(Errno::EINVAL, "the console mode is not supported");
385                }
386            }
387            cmd @ GetGraphicsMode => {
388                let console = self.console()?;
389
390                let mode = console.mode().unwrap_or(ConsoleMode::Text);
391                cmd.write(&(mode as i32))?;
392            }
393            cmd @ SetKeyboardMode => {
394                let console = self.console()?;
395
396                let mode = KeyboardMode::try_from(cmd.get())?;
397                if !console.set_keyboard_mode(mode) {
398                    return_errno_with_message!(Errno::EINVAL, "the keyboard mode is not supported");
399                }
400            }
401            cmd @ GetKeyboardMode => {
402                let console = self.console()?;
403
404                let mode = console.keyboard_mode().unwrap_or(KeyboardMode::Xlate);
405                cmd.write(&(mode as i32))?;
406            }
407
408            _ => (self.weak_self.upgrade().unwrap() as Arc<dyn Terminal>)
409                .job_ioctl(raw_ioctl, false)?,
410        });
411
412        Ok(0)
413    }
414}
