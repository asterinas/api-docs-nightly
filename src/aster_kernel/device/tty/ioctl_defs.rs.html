3use super::{
4    CFontOp,
5    termio::{CTermios, CWinSize},
6};
7use crate::util::ioctl::{InData, OutData, PassByVal, ioc};
8
9// Reference: <https://elixir.bootlin.com/linux/v6.18/source/include/uapi/asm-generic/ioctls.h>
10
11pub type GetTermios      = ioc!(TCGETS,     0x5401,     OutData<CTermios>);
12pub type SetTermios      = ioc!(TCSETS,     0x5402,     InData<CTermios>);
13pub type SetTermiosWait  = ioc!(TCSETSW,    0x5403,     InData<CTermios>);
14pub type SetTermiosFlush = ioc!(TCSETSF,    0x5404,     InData<CTermios>);
15
16pub type GetWinSize      = ioc!(TIOCGWINSZ, 0x5413,     OutData<CWinSize>);
17pub type SetWinSize      = ioc!(TIOCSWINSZ, 0x5414,     InData<CWinSize>);
18
19// TODO: Consider moving this to the `pty` module.
20pub type GetPtyNumber    = ioc!(TIOCGPTN,   b'T', 0x30, OutData<u32>);
21
22pub type SetGraphicsMode = ioc!(KDSETMODE,  0x4B3A,     InData<i32, PassByVal>);
23pub type GetGraphicsMode = ioc!(KDGETMODE,  0x4B3B,     OutData<i32>);
24
25pub type GetKeyboardMode = ioc!(KDGKBMODE,  0x4B44,     OutData<i32>);
26pub type SetKeyboardMode = ioc!(KDSKBMODE,  0x4B45,     InData<i32, PassByVal>);
27
28pub type SetOrGetFont    = ioc!(KDFONTOP,   0x4B72,     InData<CFontOp>);
