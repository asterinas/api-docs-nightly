3use super::termio::{CTermios, CTermios2, CWinSize};
4use crate::util::ioctl::{InData, OutData, ioc};
5
6// Reference: <https://elixir.bootlin.com/linux/v6.18/source/include/uapi/asm-generic/ioctls.h>
7
8pub type GetTermios      = ioc!(TCGETS,     0x5401,     OutData<CTermios>);
9pub type SetTermios      = ioc!(TCSETS,     0x5402,     InData<CTermios>);
10pub type SetTermiosWait  = ioc!(TCSETSW,    0x5403,     InData<CTermios>);
11pub type SetTermiosFlush = ioc!(TCSETSF,    0x5404,     InData<CTermios>);
12
13pub type GetTermios2     = ioc!(TCGETS2,    b'T', 0x2A, OutData<CTermios2>);
14pub type SetTermios2     = ioc!(TCSETS2,    b'T', 0x2B, InData<CTermios2>);
15
16pub type GetWinSize      = ioc!(TIOCGWINSZ, 0x5413,     OutData<CWinSize>);
17pub type SetWinSize      = ioc!(TIOCSWINSZ, 0x5414,     InData<CWinSize>);
