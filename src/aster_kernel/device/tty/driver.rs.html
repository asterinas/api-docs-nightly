11/// A TTY driver.
12///
13/// A driver exposes some device-specific behavior to [`Tty`]. For example, a device provides
14/// methods to write to the output buffer (see [`Self::push_output`]), where the output buffer can
15/// be the monitor if the underlying device is framebuffer, or just a ring buffer if the underlying
16/// device is pseduoterminal).
17///
18/// [`Tty`]: super::Tty
19pub trait TtyDriver: Send + Sync + 'static {
20    /// The device major ID.
21    const DEVICE_MAJOR_ID: u32;
22
23    /// Returns the path where the TTY should appear under devtmpfs, if any.
24    fn devtmpfs_path(&self, index: u32) -> Option<String>;
25
26    /// Opens the TTY.
27    ///
28    /// This function will be called when opening `/dev/tty`.
29    fn open(tty: Arc<Tty<Self>>) -> Result<Box<dyn FileIo>>
30    where
31        Self: Sized;
32
33    /// Pushes characters into the output buffer.
34    ///
35    /// This method returns the number of bytes pushed or fails with an error if no bytes can be
36    /// pushed.
37    fn push_output(&self, chs: &[u8]) -> Result<usize>;
38
39    /// Returns a callback function that echoes input characters to the output buffer.
40    ///
41    /// Note that the implementation may choose to hold a lock during the life of the callback.
42    /// During this time, calls to other methods such as [`Self::push_output`] may cause deadlocks.
43    fn echo_callback(&self) -> impl FnMut(&[u8]) + '_;
44
45    /// Returns whether new characters can be pushed into the output buffer.
46    ///
47    /// This method should return `false` if the output buffer is full.
48    fn can_push(&self) -> bool;
49
50    /// Notifies that the input buffer now has room for new characters.
51    ///
52    /// This method should be called when the state of [`Tty::can_push`] changes from `false` to
53    /// `true`.
54    ///
55    /// [`Tty::can_push`]: super::Tty::can_push
56    fn notify_input(&self);
57
58    /// Returns the console device associated with the TTY.
59    ///
60    /// If the TTY is not associated with any console device, this method will return `None`.
61    fn console(&self) -> Option<&dyn AnyConsoleDevice>;
62
63    /// Notifies that the TTY termios is changed.
64    ///
65    /// This method will be called with a spin lock held, so it cannot break atomic mode.
66    fn on_termios_change(&self, old_termios: &CTermios, new_termios: &CTermios);
67}
