20// The definition of WaitOptions is from Occlum
21bitflags! {
22    pub struct WaitOptions: u32 {
23        const WNOHANG = 0x1;
24        const WSTOPPED = 0x2; // Same as WUNTRACED
25        const WEXITED = 0x4;
26        const WCONTINUED = 0x8;
27        const WNOWAIT = 0x01000000;
28        const WNOTHREAD = 0x20000000;
29        const WALL = 0x40000000;
30        const WCLONE = 0x80000000;
31    }
32}
33
34impl WaitOptions {
35    pub fn check(&self) -> Result<()> {
36        // FIXME: The syscall `waitid` allows using WNOWAIT with
37        // WSTOPPED or WCONTINUED
38        if self.intersects(WaitOptions::WSTOPPED | WaitOptions::WCONTINUED)
39            && self.contains(WaitOptions::WNOWAIT)
40        {
41            return_errno_with_message!(
42                Errno::EINVAL,
43                "WNOWAIT cannot be used toghther with WSTOPPED or WCONTINUED"
44            );
45        }
46
47        let supported_args = WaitOptions::WNOHANG
48            | WaitOptions::WSTOPPED
49            | WaitOptions::WCONTINUED
50            | WaitOptions::WNOWAIT;
51        if !supported_args.contains(*self) {
52            warn!(
53                "unsupported wait options are found: {:?}",
54                *self - supported_args
55            );
56        }
57
58        Ok(())
59    }
60}

62pub fn do_wait(
63    child_filter: ProcessFilter,
64    wait_options: WaitOptions,
65    ctx: &Context,
66) -> Result<Option<WaitStatus>> {
67    wait_options.check()?;
68
69    let is_nonblocking = if let ProcessFilter::WithPidfd(pid_file) = &child_filter {
70        pid_file.is_nonblocking()
71    } else {
72        false
73    };
74
75    let zombie_child = with_sigmask_changed(
76        ctx,
77        |sigmask| sigmask + SIGCHLD,
78        || {
79            ctx.process.children_wait_queue().pause_until(|| {
80                // Acquire the children lock at first to prevent race conditions.
81                // We want to ensure that multiple waiting threads
82                // do not return the same waited process status.
83                let mut children_lock = ctx.process.children().lock();
84                let children_mut = children_lock.as_mut().unwrap();
85
86                let unwaited_children = children_mut
87                    .values()
88                    .filter(|child| match &child_filter {
89                        ProcessFilter::Any => true,
90                        ProcessFilter::WithPid(pid) => child.pid() == *pid,
91                        ProcessFilter::WithPgid(pgid) => child.pgid() == *pgid,
92                        ProcessFilter::WithPidfd(pid_file) => match pid_file.process_opt() {
93                            Some(process) => Arc::ptr_eq(&process, child),
94                            None => false,
95                        },
96                    })
97                    .collect::<Box<_>>();
98
99                if unwaited_children.is_empty() {
100                    return Some(Err(Error::with_message(
101                        Errno::ECHILD,
102                        "the process has no child to wait",
103                    )));
104                }
105
106                if let Some(status) = wait_zombie(&unwaited_children) {
107                    if !wait_options.contains(WaitOptions::WNOWAIT) {
108                        reap_zombie_child(
109                            status.pid(),
110                            children_mut,
111                            ctx.process.reaped_children_stats(),
112                        );
113                    }
114                    return Some(Ok(Some(status)));
115                }
116
117                if let Some(status) = wait_stopped_or_continued(&unwaited_children, wait_options) {
118                    return Some(Ok(Some(status)));
119                }
120
121                if wait_options.contains(WaitOptions::WNOHANG) {
122                    return Some(Ok(None));
123                }
124
125                if is_nonblocking {
126                    return Some(Err(Error::with_message(
127                        Errno::EAGAIN,
128                        "the PID file is nonblocking and the child has not terminated",
129                    )));
130                }
131
132                // wait
133                None
134            })
135        },
136    )??;
137
138    Ok(zombie_child)
139}

141pub enum WaitStatus {
142    Zombie(Arc<Process>),
143    Stop(Arc<Process>, SigNum),
144    Continue(Arc<Process>),
145}
146
147impl WaitStatus {
148    pub fn pid(&self) -> Pid {
149        self.process().pid()
150    }
151
152    pub fn uid(&self) -> Uid {
153        self.process()
154            .main_thread()
155            .as_posix_thread()
156            .unwrap()
157            .credentials()
158            .ruid()
159    }
160
161    pub fn prof_clock(&self) -> &Arc<ProfClock> {
162        self.process().prof_clock()
163    }
164
165    fn process(&self) -> &Arc<Process> {
166        match self {
167            WaitStatus::Zombie(process)
168            | WaitStatus::Stop(process, _)
169            | WaitStatus::Continue(process) => process,
170        }
171    }
172}
173
174fn wait_zombie(unwaited_children: &[&Arc<Process>]) -> Option<WaitStatus> {
175    unwaited_children
176        .iter()
177        .find(|child| child.status().is_zombie())
178        .map(|child| WaitStatus::Zombie((*child).clone()))
179}
180
181fn wait_stopped_or_continued(
182    unwaited_children: &[&Arc<Process>],
183    wait_options: WaitOptions,
184) -> Option<WaitStatus> {
185    if !wait_options.intersects(WaitOptions::WSTOPPED | WaitOptions::WCONTINUED) {
186        return None;
187    }
188
189    // Lock order: children of process -> tasks of process
190    for process in unwaited_children.iter() {
191        let Some(stop_wait_status) = process.wait_stopped_or_continued(wait_options) else {
192            continue;
193        };
194
195        let wait_status = match stop_wait_status {
196            StopWaitStatus::Stopped(sig_num) => WaitStatus::Stop((*process).clone(), sig_num),
197            StopWaitStatus::Continue => WaitStatus::Continue((*process).clone()),
198        };
199        return Some(wait_status);
200    }
201
202    None
203}

205/// Free zombie child with `child_pid`, returns the exit code of child process.
206fn reap_zombie_child(
207    child_pid: Pid,
208    children_lock: &mut BTreeMap<Pid, Arc<Process>>,
209    reaped_children_stats: &Mutex<ReapedChildrenStats>,
210) -> ExitCode {
211    let child_process = children_lock.remove(&child_pid).unwrap();
212    assert!(child_process.status().is_zombie());
213
214    for task in child_process.tasks().lock().as_slice() {
215        thread_table::remove_thread(task.as_posix_thread().unwrap().tid());
216    }
217
218    // Lock order: children of process -> session table -> group table
219    // -> process table -> group of process -> group inner -> session inner
220    let mut session_table_mut = process_table::session_table_mut();
221    let mut group_table_mut = process_table::group_table_mut();
222
223    // Remove the process from the global table
224    let mut process_table_mut = process_table::process_table_mut();
225    process_table_mut.remove(child_process.pid());
226
227    // Remove the process group and the session from global table, if necessary
228    let mut child_group_mut = child_process.process_group.lock();
229    child_process.clear_old_group_and_session(
230        &mut child_group_mut,
231        &mut session_table_mut,
232        &mut group_table_mut,
233    );
234    *child_group_mut = Weak::new();
235
236    let (mut user_time, mut kernel_time) = child_process.reaped_children_stats().lock().get();
237    user_time += child_process.prof_clock().user_clock().read_time();
238    kernel_time += child_process.prof_clock().kernel_clock().read_time();
239    reaped_children_stats.lock().add(user_time, kernel_time);
240
241    child_process.status().exit_code()
242}
