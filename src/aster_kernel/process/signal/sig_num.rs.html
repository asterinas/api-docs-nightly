10#[derive(Debug, Clone, Copy, PartialEq, Eq)]
11pub struct SigNum {
12    sig_num: u8,
13}
14
15impl TryFrom<u8> for SigNum {
16    type Error = Error;
17
18    fn try_from(sig_num: u8) -> Result<Self> {
19        if !(MIN_STD_SIG_NUM..=MAX_RT_SIG_NUM).contains(&sig_num) {
20            return_errno_with_message!(Errno::EINVAL, "invalid signal number");
21        }
22        Ok(SigNum { sig_num })
23    }
24}
25
26impl SigNum {
27    /// Caller must ensure the sig_num is valid. Otherwise, use try_from will check sig_num and does not panic.
28    pub const fn from_u8(sig_num: u8) -> Self {
29        if sig_num > MAX_RT_SIG_NUM || sig_num < MIN_STD_SIG_NUM {
30            panic!("invalid signal number")
31        }
32        SigNum { sig_num }
33    }
34
35    pub const fn as_u8(&self) -> u8 {
36        self.sig_num
37    }
38
39    pub fn is_std(&self) -> bool {
40        self.sig_num <= MAX_STD_SIG_NUM
41    }
42
43    pub fn is_real_time(&self) -> bool {
44        self.sig_num >= MIN_RT_SIG_NUM
45    }
46
47    pub const fn sig_name(&self) -> &'static str {
48        match *self {
49            SIGHUP => "SIGHUP",
50            SIGINT => "SIGINT",
51            SIGQUIT => "SIGQUIT",
52            SIGILL => "SIGILL",
53            SIGTRAP => "SIGTRAP",
54            SIGABRT => "SIGABRT",
55            SIGBUS => "SIGBUS",
56            SIGFPE => "SIGFPE",
57            SIGKILL => "SIGKILL",
58            SIGUSR1 => "SIGUSR1",
59            SIGSEGV => "SIGSEGV",
60            SIGUSR2 => "SIGUSR2",
61            SIGPIPE => "SIGPIPE",
62            SIGALRM => "SIGALRM",
63            SIGTERM => "SIGTERM",
64            SIGSTKFLT => "SIGSTKFLT",
65            SIGCHLD => "SIGCHLD",
66            SIGCONT => "SIGCONT",
67            SIGSTOP => "SIGSTOP",
68            SIGTSTP => "SIGTSTP",
69            SIGTTIN => "SIGTTIN",
70            SIGTTOU => "SIGTTOU",
71            SIGURG => "SIGURG",
72            SIGXCPU => "SIGXCPU",
73            SIGXFSZ => "SIGXFSZ",
74            SIGVTALRM => "SIGVTALRM",
75            SIGPROF => "SIGPROF",
76            SIGWINCH => "SIGWINCH",
77            SIGIO => "SIGIO",
78            SIGPWR => "SIGPWR",
79            SIGSYS => "SIGSYS",
80            _ => "Realtime Signal",
81        }
82    }
83}

85/// Atomic signal number.
86///
87/// This struct represents a signal number and is different from [SigNum]
88/// in that it allows for an empty signal number.
89pub struct AtomicSigNum(AtomicU8);
90
91impl AtomicSigNum {
92    /// Creates a new empty atomic signal number
93    pub const fn new_empty() -> Self {
94        Self(AtomicU8::new(0))
95    }
96
97    /// Creates a new signal number with the specified value
98    pub const fn new(sig_num: SigNum) -> Self {
99        Self(AtomicU8::new(sig_num.as_u8()))
100    }
101
102    /// Determines whether the signal number is empty
103    pub fn is_empty(&self) -> bool {
104        self.0.load(Ordering::Relaxed) == 0
105    }
106
107    /// Returns the corresponding [`SigNum`]
108    pub fn as_sig_num(&self) -> Option<SigNum> {
109        let sig_num = self.0.load(Ordering::Relaxed);
110        if sig_num == 0 {
111            return None;
112        }
113
114        Some(SigNum::from_u8(sig_num))
115    }
116
117    /// Sets the new `sig_num`
118    pub fn set(&self, sig_num: SigNum) {
119        self.0.store(sig_num.as_u8(), Ordering::Relaxed)
120    }
121
122    /// Clears the signal number
123    pub fn clear(&self) {
124        self.0.store(0, Ordering::Relaxed)
125    }
126}
