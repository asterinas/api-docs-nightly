18/// Trait for handling pending signals.
19pub trait HandlePendingSignal {
20    /// Returns the thread's pending signal set.
21    ///
22    /// This includes signals that are currently blocked or ignored.
23    fn pending_signals(&self) -> SigSet;
24
25    /// Returns if there are pending signals that are neither blocked nor ignored.
26    ///
27    /// Note that ignored but not blocked signals may be dequeued silently.
28    fn has_pending(&self) -> bool;
29
30    /// Returns if a SIGKILL signal is pending.
31    fn has_pending_sigkill(&self) -> bool;
32
33    /// Dequeues the next pending signal that is not masked by `mask`.
34    ///
35    /// Returns `None` if no such signal is available.
36    fn dequeue_signal(&self, mask: &SigMask) -> Option<Box<dyn Signal>>;
37}
38
39impl HandlePendingSignal for Context<'_> {
40    fn pending_signals(&self) -> SigSet {
41        self.posix_thread.sig_queues().sig_pending() | self.process.sig_queues().sig_pending()
42    }
43
44    fn has_pending(&self) -> bool {
45        let posix_thread = self.posix_thread;
46        let process = self.process.as_ref();
47        has_pending_signal(posix_thread, process)
48    }
49
50    fn has_pending_sigkill(&self) -> bool {
51        self.posix_thread.sig_queues().has_pending_signal(SIGKILL)
52            || self.process.sig_queues().has_pending_signal(SIGKILL)
53    }
54
55    fn dequeue_signal(&self, mask: &SigMask) -> Option<Box<dyn Signal>> {
56        self.posix_thread
57            .sig_queues()
58            .dequeue(mask)
59            .or_else(|| self.process.sig_queues().dequeue(mask))
60    }
61}
62
63impl HandlePendingSignal for PosixThread {
64    fn pending_signals(&self) -> SigSet {
65        self.sig_queues().sig_pending() | self.process().sig_queues().sig_pending()
66    }
67
68    fn has_pending(&self) -> bool {
69        let process = self.process();
70        has_pending_signal(self, process.as_ref())
71    }
72
73    fn has_pending_sigkill(&self) -> bool {
74        self.sig_queues().has_pending_signal(SIGKILL)
75            || self.process().sig_queues().has_pending_signal(SIGKILL)
76    }
77
78    fn dequeue_signal(&self, mask: &SigMask) -> Option<Box<dyn Signal>> {
79        self.sig_queues()
80            .dequeue(mask)
81            .or_else(|| self.process().sig_queues().dequeue(mask))
82    }
83}
84
85fn has_pending_signal(posix_thread: &PosixThread, process: &Process) -> bool {
86    // Fast path: No signals are pending.
87    if posix_thread.sig_queues().is_empty() && process.sig_queues().is_empty() {
88        return false;
89    }
90
91    // Slow path: Some signals are pending.
92    let sig_dispositions = process.sig_dispositions().lock();
93    let sig_dispositions = sig_dispositions.lock();
94    let blocked = posix_thread.sig_mask().load(Ordering::Relaxed);
95
96    posix_thread
97        .sig_queues()
98        .has_pending(blocked, &sig_dispositions)
99        || process.sig_queues().has_pending(blocked, &sig_dispositions)
100}
