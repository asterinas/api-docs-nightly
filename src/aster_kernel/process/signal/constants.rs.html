21macro_rules! define_std_signums {
22    ( $( $name: ident = $num: expr ),+, ) => {
23        $(
24            pub const $name : SigNum = SigNum::from_u8($num);
25        )*
26    }
27}

30    SIGHUP    = 1, // Hangup detected on controlling terminal or death of controlling process
31    SIGINT    = 2, // Interrupt from keyboard
32    SIGQUIT   = 3, // Quit from keyboard
33    SIGILL    = 4, // Illegal Instruction

37    SIGFPE    = 8, // Floating-point exception
38    SIGKILL   = 9, // Kill signal

40    SIGSEGV   = 11, // Invalid memory reference
41    SIGUSR2   = 12, // User-defined signal 2
42    SIGPIPE   = 13, // Broken pipe: write to pipe with no readers; see pipe(7)
43    SIGALRM   = 14, // Timer signal from alarm(2)
44    SIGTERM   = 15, // Termination signal
45    SIGSTKFLT = 16, // Stack fault on coprocessor (unused)
46    SIGCHLD   = 17, // Child stopped or terminated
47    SIGCONT   = 18, // Continue if stopped
48    SIGSTOP   = 19, // Stop process
49    SIGTSTP   = 20, // Stop typed at terminal
50    SIGTTIN   = 21, // Terminal input for background process
51    SIGTTOU   = 22, // Terminal output for background process
52    SIGURG    = 23, // Urgent condition on socket (4.2BSD)
53    SIGXCPU   = 24, // CPU time limit exceeded (4.2BSD); see setrlimit(2)
54    SIGXFSZ   = 25, // File size limit exceeded (4.2BSD); see setrlimit(2)
55    SIGVTALRM = 26, // Virtual alarm clock (4.2BSD)
56    SIGPROF   = 27, // Profiling timer expired
57    SIGWINCH  = 28, // Window resize signal (4.3BSD, Sun)
58    SIGIO     = 29, // I/O now possible (4.2BSD)
59    SIGPWR    = 30, // Power failure (System V)
60    SIGSYS    = 31, // Bad system call (SVr4); see also seccomp(2)
61}
