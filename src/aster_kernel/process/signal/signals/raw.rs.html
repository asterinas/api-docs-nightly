3use core::fmt::Debug;
4
5use crate::process::signal::{c_types::siginfo_t, sig_num::SigNum, signals::Signal};
6
7/// A signal that carries raw [`siginfo_t`] information.
8#[derive(Clone, Copy)]
9pub struct RawSignal {
10    info: siginfo_t,
11}
12
13impl Debug for RawSignal {
14    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
15        f.debug_struct("RawSignal")
16            .field("signo", &self.info.si_signo)
17            .field("errno", &self.info.si_errno)
18            .field("code", &self.info.si_code)
19            .finish_non_exhaustive()
20    }
21}
22
23impl RawSignal {
24    /// Creates a signal that carries raw [`siginfo_t`] information.
25    ///
26    /// The caller must ensure that the `info.si_signo` is a valid signal number.
27    pub fn new(info: siginfo_t) -> Self {
28        Self { info }
29    }
30}
31
32impl Signal for RawSignal {
33    fn num(&self) -> SigNum {
34        SigNum::from_u8(self.info.si_signo as u8)
35    }
36
37    fn to_info(&self) -> siginfo_t {
38        self.info
39    }
40}
