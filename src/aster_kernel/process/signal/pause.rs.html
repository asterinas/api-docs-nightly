18/// `Pause` is an extension trait to make [`Waiter`] and [`WaitQueue`] signal aware.
19///
20/// The original methods of `Waiter` and `WaitQueue` only allow a thread
21/// to wait (via `wait`) until it is woken up (via `wake_up`, `wake_one`, or `wake_all`)
22/// or a condition is met (via `wait_until`).
23/// The `WaitTimeout` extension trait grants the extra ability
24/// to wait until a timeout (via `wait_until_or_timeout`).
25/// On top of `WaitTimeout`, this `Pause` trait provides the `pause`-family methods,
26/// which are similar to the `wait`-family methods except that the methods also return
27/// when the waiting thread is interrupted by a POSIX signal.
28/// When this happens, the `pause`-family methods return `Err(EINTR)`.
29pub trait Pause: WaitTimeout {
30    /// Pauses until the condition is met or a signal interrupts.
31    ///
32    /// # Errors
33    ///
34    /// This method will return an error with [`EINTR`] if a signal is received before the
35    /// condition is met.
36    ///
37    /// [`EINTR`]: crate::error::Errno::EINTR
38    #[track_caller]
39    fn pause_until<F, R>(&self, cond: F) -> Result<R>
40    where
41        F: FnMut() -> Option<R>,
42    {
43        self.pause_until_or_timeout_impl(cond, None, PauseReason::Sleep)
44    }
45
46    /// Pauses until the condition is met or a signal interrupts.
47    ///
48    /// This pause happens due to the reason specified. If the reason is `PauseReason::Sleep`,
49    /// the caller should use `pause_until` straightforwardly.
50    ///
51    /// # Errors
52    ///
53    /// This method will return an error with [`EINTR`] if a signal is received before the
54    /// condition is met.
55    ///
56    /// [`EINTR`]: crate::error::Errno::EINTR
57    #[track_caller]
58    fn pause_until_by<F, R>(&self, cond: F, reason: PauseReason) -> Result<R>
59    where
60        F: FnMut() -> Option<R>,
61    {
62        self.pause_until_or_timeout_impl(cond, None, reason)
63    }
64
65    /// Pauses until the condition is met, the timeout is reached, or a signal interrupts.
66    ///
67    /// # Errors
68    ///
69    /// Before the condition is met, this method will return an error with
70    ///  - [`EINTR`] if a signal is received;
71    ///  - [`ETIME`] if the timeout is reached.
72    ///
73    /// [`ETIME`]: crate::error::Errno::ETIME
74    /// [`EINTR`]: crate::error::Errno::EINTR
75    #[track_caller]
76    fn pause_until_or_timeout<'a, F, T, R>(&self, mut cond: F, timeout: T) -> Result<R>
77    where
78        F: FnMut() -> Option<R>,
79        T: Into<TimeoutExt<'a>>,
80    {
81        let timeout = timeout.into();
82        let timeout_inner = match timeout.check_expired() {
83            Ok(inner) => inner,
84            Err(err) => return cond().ok_or(err),
85        };
86
87        self.pause_until_or_timeout_impl(cond, timeout_inner, PauseReason::Sleep)
88    }
89
90    /// Pauses until the condition is met, the timeout is reached, or a signal interrupts.
91    ///
92    /// # Errors
93    ///
94    /// Before the condition is met, this method will return an error with
95    ///  - [`EINTR`] if a signal is received;
96    ///  - [`ETIME`] if the timeout is reached.
97    ///
98    /// [`ETIME`]: crate::error::Errno::ETIME
99    /// [`EINTR`]: crate::error::Errno::EINTR
100    #[doc(hidden)]
101    #[track_caller]
102    fn pause_until_or_timeout_impl<F, R>(
103        &self,
104        cond: F,
105        timeout: Option<&ManagedTimeout>,
106        reason: PauseReason,
107    ) -> Result<R>
108    where
109        F: FnMut() -> Option<R>;
110
111    /// Pauses until the thread is woken up, the timeout is reached, or a signal interrupts.
112    ///
113    /// # Errors
114    ///
115    /// This method will return an error with
116    ///  - [`EINTR`] if a signal is received;
117    ///  - [`ETIME`] if the timeout is reached.
118    ///
119    /// [`ETIME`]: crate::error::Errno::ETIME
120    /// [`EINTR`]: crate::error::Errno::EINTR
121    #[track_caller]
122    fn pause_timeout(&self, timeout: &TimeoutExt<'_>) -> Result<()>;
123}
124
125impl Pause for Waiter {
126    fn pause_until_or_timeout_impl<F, R>(
127        &self,
128        cond: F,
129        timeout: Option<&ManagedTimeout>,
130        reason: PauseReason,
131    ) -> Result<R>
132    where
133        F: FnMut() -> Option<R>,
134    {
135        // No fast paths for `Waiter`. If the caller wants a fast path, it should do so _before_
136        // the waiter is created.
137
138        let Some(posix_thread) = self
139            .task()
140            .as_thread()
141            .and_then(|thread| thread.as_posix_thread())
142        else {
143            return self.wait_until_or_timeout_cancelled(cond, || Ok(()), timeout);
144        };
145
146        let cancel_cond = || {
147            if posix_thread.has_pending() {
148                return Err(Error::with_message(
149                    Errno::EINTR,
150                    "the current thread is interrupted by a signal",
151                ));
152            }
153            Ok(())
154        };
155
156        posix_thread.set_signalled_waker(self.waker(), reason);
157        let res = self.wait_until_or_timeout_cancelled(cond, cancel_cond, timeout);
158        posix_thread.clear_signalled_waker();
159
160        res
161    }
162
163    fn pause_timeout(&self, timeout: &TimeoutExt<'_>) -> Result<()> {
164        let timer = timeout.check_expired()?.map(|timeout| {
165            let waker = self.waker();
166            timeout.create_timer(move |_guard: TimerGuard| {
167                waker.wake_up();
168            })
169        });
170
171        let posix_thread_opt = self
172            .task()
173            .as_thread()
174            .and_then(|thread| thread.as_posix_thread());
175
176        if let Some(posix_thread) = posix_thread_opt {
177            posix_thread.set_signalled_waker(self.waker(), PauseReason::Sleep);
178            // Check `has_pending` after `set_signalled_waker` to avoid race conditions.
179            if posix_thread.has_pending() {
180                posix_thread.clear_signalled_waker();
181                return_errno_with_message!(
182                    Errno::EINTR,
183                    "the current thread is interrupted by a signal"
184                );
185            }
186
187            self.wait();
188
189            posix_thread.clear_signalled_waker();
190        } else {
191            self.wait();
192        }
193
194        if let Some(timer) = timer {
195            let timer_guard = timer.lock();
196            if timer_guard.remain().is_zero() {
197                return_errno_with_message!(Errno::ETIME, "the time limit is reached");
198            }
199            // If the timeout is not expired, cancel the timer manually.
200            timer_guard.cancel();
201        }
202
203        if posix_thread_opt
204            .as_ref()
205            .is_some_and(|posix_thread| posix_thread.has_pending())
206        {
207            return_errno_with_message!(
208                Errno::EINTR,
209                "the current thread is interrupted by a signal"
210            );
211        }
212
213        Ok(())
214    }
215}

217impl Pause for WaitQueue {
218    fn pause_until_or_timeout_impl<F, R>(
219        &self,
220        mut cond: F,
221        timeout: Option<&ManagedTimeout>,
222        reason: PauseReason,
223    ) -> Result<R>
224    where
225        F: FnMut() -> Option<R>,
226    {
227        // Fast path:
228        if let Some(res) = cond() {
229            return Ok(res);
230        }
231
232        let (waiter, _) = Waiter::new_pair();
233        let cond = || {
234            self.enqueue(waiter.waker());
235            cond()
236        };
237        waiter.pause_until_or_timeout_impl(cond, timeout, reason)
238    }
239
240    fn pause_timeout(&self, _timeout: &TimeoutExt<'_>) -> Result<()> {
241        panic!("`pause_timeout` can only be used on `Waiter`");
242    }
243}

245/// The reason why a process is paused by a `pause`-family method.
246pub enum PauseReason {
247    Sleep,
248    StopBySignal,
249    #[expect(dead_code)]
250    StopByPtrace,
251}

253/// Executes a closure after temporarily adjusting the signal mask of the current POSIX thread.
254pub fn with_sigmask_changed<R>(
255    ctx: &Context,
256    mask_op: impl FnOnce(SigMask) -> SigMask,
257    operate: impl FnOnce() -> R,
258) -> R {
259    let sig_mask = ctx.posix_thread.sig_mask();
260
261    // Save the original signal mask and apply the mask updates.
262    let old_mask = sig_mask.load(Ordering::Relaxed);
263    sig_mask.store(mask_op(old_mask), Ordering::Relaxed);
264
265    // Perform the operation.
266    let res = operate();
267
268    // Restore the original signal mask.
269    sig_mask.store(old_mask, Ordering::Relaxed);
270
271    res
272}
