16/// A wrapper for the [`xmas_elf`] ELF parser.
17pub struct ElfHeaders {
18    elf_header: ElfHeader,
19    loadable_phdrs: Vec<LoadablePhdr>,
20    max_load_align: usize,
21    interp_phdr: Option<InterpPhdr>,
22}
23
24impl ElfHeaders {
25    /// The minimized length of a valid ELF header.
26    ///
27    /// [`Self::parse`] will fail with [`ENOEXEC`] if the slice is shorter than this constant. This
28    /// can also be checked manually if a different error code is expected.
29    ///
30    /// [`ENOEXEC`]: Errno::ENOEXEC
31    pub(super) const LEN: usize = size_of::<HeaderPt1>() + size_of::<HeaderPt2_64>();
32
33    pub fn parse(input: &[u8]) -> Result<Self> {
34        // Parse the ELF header.
35        let header = xmas_elf::header::parse_header(input)
36            .map_err(|_| Error::with_message(Errno::ENOEXEC, "the ELF header is invalid"))?;
37        let elf_header = ElfHeader::parse(header)?;
38        check_elf_header(&elf_header)?;
39
40        // Validate the placement of the ELF program headers.
41        let ph_count = elf_header.pt2.ph_count;
42        let ph_entry_size = elf_header.pt2.ph_entry_size;
43        let ph_offset = elf_header.pt2.ph_offset;
44        if ph_entry_size as usize != size_of::<ProgramHeader64>() {
45            return_errno_with_message!(
46                Errno::ENOEXEC,
47                "the size of ELF program headers is invalid"
48            );
49        }
50        if ph_offset >= input.len() as u64
51            || (input.len() as u64 - ph_offset) / (ph_entry_size as u64) < (ph_count as u64)
52        {
53            // TODO: Currently, the program headers must follow the ELF header immediately (they
54            // must reside in `input`, the first page of the ELF file). This requirement may be
55            // relaxed in the future.
56            return_errno_with_message!(
57                Errno::ENOEXEC,
58                "the placement of ELF program headers is not supported"
59            );
60        }
61
62        // Parse the ELF program headers.
63        let mut loadable_phdrs = Vec::with_capacity(ph_count as usize);
64        let mut max_load_align = PAGE_SIZE;
65        let mut interp_phdr = None;
66        for index in 0..ph_count {
67            let program_header = xmas_elf::program::parse_program_header(input, header, index)
68                .map_err(|_| {
69                    Error::with_message(Errno::ENOEXEC, "the ELF program header is invalid")
70                })?;
71            let ph64 = match program_header {
72                xmas_elf::program::ProgramHeader::Ph64(ph64) => *ph64,
73                xmas_elf::program::ProgramHeader::Ph32(_) => {
74                    return_errno_with_message!(
75                        Errno::ENOEXEC,
76                        "the ELF program header is not 64-bit"
77                    )
78                }
79            };
80            match ph64.get_type() {
81                Ok(program::Type::Load) => {
82                    loadable_phdrs.push(LoadablePhdr::parse(&ph64)?);
83                    // Like Linux, we ignore any invalid alignment requirements that are not a
84                    // power of two.
85                    if ph64.align.is_power_of_two() {
86                        max_load_align = max_load_align.max(ph64.align as usize);
87                    }
88                }
89                Ok(program::Type::Interp) if interp_phdr.is_none() => {
90                    // Like Linux, we only handle the first interpreter program header.
91                    interp_phdr = Some(InterpPhdr::parse(&ph64)?);
92                }
93                _ => (),
94            }
95        }
96        if loadable_phdrs.is_empty() {
97            return_errno_with_message!(Errno::ENOEXEC, "there are no loadable ELF program headers");
98        }
99
100        Ok(Self {
101            elf_header,
102            loadable_phdrs,
103            max_load_align,
104            interp_phdr,
105        })
106    }
107
108    /// Returns whether the ELF is a shared object.
109    pub(super) fn is_shared_object(&self) -> bool {
110        self.elf_header.pt2.type_.as_type() == header::Type::SharedObject
111    }
112
113    /// Returns the address of the entry point.
114    pub(super) fn entry_point(&self) -> Vaddr {
115        self.elf_header.pt2.entry_point as Vaddr
116    }
117
118    /// Returns the number of the program headers.
119    pub(super) fn ph_count(&self) -> u16 {
120        self.elf_header.pt2.ph_count
121    }
122
123    /// Returns the size of a program header.
124    pub(super) fn ph_ent(&self) -> u16 {
125        self.elf_header.pt2.ph_entry_size
126    }
127
128    /// Returns a reference to the loadable program headers.
129    ///
130    /// It is guaranteed that there is at least one loadable program header.
131    pub(super) fn loadable_phdrs(&self) -> &[LoadablePhdr] {
132        self.loadable_phdrs.as_slice()
133    }
134
135    /// Returns the maximum alignment of the loadable program headers.
136    ///
137    /// It is guaranteed that the alignment is a power of two and is at least [`PAGE_SIZE`].
138    pub(super) fn max_load_align(&self) -> usize {
139        self.max_load_align
140    }
141
142    /// Returns a reference to the interpreter program header.
143    pub(super) fn interp_phdr(&self) -> Option<&InterpPhdr> {
144        self.interp_phdr.as_ref()
145    }
146
147    /// Finds the virtual address of the program headers.
148    pub(super) fn find_vaddr_of_phdrs(&self) -> Result<Vaddr> {
149        let ph_offset = self.elf_header.pt2.ph_offset as usize;
150        for loadable_phdrs in self.loadable_phdrs.iter() {
151            if loadable_phdrs.file_range().contains(&ph_offset) {
152                return Ok(loadable_phdrs.virt_range().start
153                    + (ph_offset - loadable_phdrs.file_range().start));
154            }
155        }
156        return_errno_with_message!(
157            Errno::ENOEXEC,
158            "the ELF program headers are not located in any segments"
159        );
160    }
161
162    /// Calculates the virtual address bounds of all segments as a range.
163    pub(super) fn calc_total_vaddr_bounds(&self) -> Range<Vaddr> {
164        self.loadable_phdrs
165            .iter()
166            .map(LoadablePhdr::virt_range)
167            .cloned()
168            .reduce(|r1, r2| r1.start.min(r2.start)..r1.end.max(r2.end))
169            .unwrap()
170    }
171
172    /// Finds the last loadable segment and returns its virtual address bounds.
173    pub(super) fn find_last_vaddr_bound(&self) -> Option<Range<Vaddr>> {
174        self.loadable_phdrs
175            .iter()
176            .max_by_key(|phdr| phdr.virt_range().end)
177            .map(|phdr| phdr.virt_range().clone())
178    }
179}

181struct ElfHeader {
182    pt1: HeaderPt1,
183    pt2: HeaderPt2_64,
184}
185
186impl ElfHeader {
187    pub(self) fn parse(header: Header) -> Result<Self> {
188        let pt1 = *header.pt1;
189        let pt2 = match header.pt2 {
190            HeaderPt2::Header64(header_pt2) => {
191                let HeaderPt2_ {
192                    type_,
193                    machine,
194                    version,
195                    entry_point,
196                    ph_offset,
197                    sh_offset,
198                    flags,
199                    header_size,
200                    ph_entry_size,
201                    ph_count,
202                    sh_entry_size,
203                    sh_count,
204                    sh_str_index,
205                } = header_pt2;
206                HeaderPt2_64 {
207                    type_: *type_,
208                    machine: *machine,
209                    version: *version,
210                    entry_point: *entry_point,
211                    ph_offset: *ph_offset,
212                    sh_offset: *sh_offset,
213                    flags: *flags,
214                    header_size: *header_size,
215                    ph_entry_size: *ph_entry_size,
216                    ph_count: *ph_count,
217                    sh_entry_size: *sh_entry_size,
218                    sh_count: *sh_count,
219                    sh_str_index: *sh_str_index,
220                }
221            }
222            _ => return_errno_with_message!(Errno::ENOEXEC, "the ELF file is not 64-bit"),
223        };
224        Ok(ElfHeader { pt1, pt2 })
225    }
226}

228struct HeaderPt2_64 {
229    type_: Type_,
230    machine: Machine_,
231    #[expect(dead_code)]
232    version: u32,
233    entry_point: u64,
234    ph_offset: u64,
235    #[expect(dead_code)]
236    sh_offset: u64,
237    #[expect(dead_code)]
238    flags: u32,
239    #[expect(dead_code)]
240    header_size: u16,
241    ph_entry_size: u16,
242    ph_count: u16,
243    #[expect(dead_code)]
244    sh_entry_size: u16,
245    #[expect(dead_code)]
246    sh_count: u16,
247    #[expect(dead_code)]
248    sh_str_index: u16,
249}

251fn check_elf_header(elf_header: &ElfHeader) -> Result<()> {
252    #[cfg(target_arch = "x86_64")]
253    const EXPECTED_ELF_MACHINE: header::Machine = header::Machine::X86_64;
254    #[cfg(target_arch = "riscv64")]
255    const EXPECTED_ELF_MACHINE: header::Machine = header::Machine::RISC_V;
256    // Reference: <https://loongson.github.io/LoongArch-Documentation/LoongArch-ELF-ABI-EN.html#_e_machine_identifies_the_machine>
257    #[cfg(target_arch = "loongarch64")]
258    const EXPECTED_ELF_MACHINE: header::Machine = header::Machine::Other(258);
259
260    if elf_header.pt1.class() != header::Class::SixtyFour {
261        return_errno_with_message!(Errno::ENOEXEC, "the ELF file is not 64-bit");
262    }
263
264    if elf_header.pt1.data() != header::Data::LittleEndian {
265        return_errno_with_message!(Errno::ENOEXEC, "the ELF file is not in little endian");
266    }
267
268    // TODO: Should we check `pt1.os_abi()` or `pt1.version()`?
269
270    if elf_header.pt2.machine.as_machine() != EXPECTED_ELF_MACHINE {
271        return_errno_with_message!(
272            Errno::ENOEXEC,
273            "the ELF file is of a different architecture"
274        );
275    }
276
277    let elf_type = elf_header.pt2.type_.as_type();
278    if elf_type != header::Type::Executable && elf_type != header::Type::SharedObject {
279        return_errno_with_message!(Errno::ENOEXEC, "the ELF file is not an executable");
280    }
281
282    Ok(())
283}

285/// A ELF program header of the type [`program::Type::Load`].
286pub(super) struct LoadablePhdr {
287    virt_range: Range<Vaddr>,
288    file_range: Range<usize>,
289    vm_perms: VmPerms,
290}
291
292impl LoadablePhdr {
293    pub(self) fn parse(phdr: &ProgramHeader64) -> Result<Self> {
294        debug_assert_eq!(phdr.get_type(), Ok(program::Type::Load));
295
296        let virt_start = phdr.virtual_addr;
297        let virt_end = if let Some(virt_end) = virt_start.checked_add(phdr.mem_size)
298            && virt_end <= VMAR_CAP_ADDR as u64
299        {
300            virt_end
301        } else {
302            return_errno_with_message!(Errno::ENOMEM, "the mapping address is too large");
303        };
304
305        let file_start = phdr.offset;
306        let Some(file_end) = file_start.checked_add(phdr.file_size) else {
307            return_errno_with_message!(Errno::EINVAL, "the mapping offset overflows");
308        };
309        if file_end >= isize::MAX as u64 {
310            return_errno_with_message!(Errno::EOVERFLOW, "the mapping offset overflows");
311        }
312
313        if phdr.mem_size == 0 {
314            return_errno_with_message!(Errno::EINVAL, "the mapping length is zero");
315        }
316        if phdr.mem_size < phdr.file_size {
317            return_errno_with_message!(
318                Errno::EINVAL,
319                "the mapping length is smaller than the file length"
320            );
321        }
322        if virt_start % (PAGE_SIZE as u64) != file_start % (PAGE_SIZE as u64) {
323            return_errno_with_message!(Errno::EINVAL, "the mapping address is not aligned");
324        }
325
326        Ok(Self {
327            virt_range: (virt_start as Vaddr)..(virt_end as Vaddr),
328            file_range: (file_start as usize)..(file_end as usize),
329            vm_perms: parse_segment_perm(phdr.flags),
330        })
331    }
332
333    /// Returns the virtual address range.
334    ///
335    /// The range is guaranteed to be below [`VMAR_CAP_ADDR`] and non-empty.
336    pub(super) fn virt_range(&self) -> &Range<Vaddr> {
337        &self.virt_range
338    }
339
340    /// Returns the file offset range.
341    ///
342    /// The range is guaranteed to be below [`i64::MAX`]. It will also be shorter than the virtual
343    /// address range ([`Self::virt_range`]) and have the same offset within a page as that range.
344    pub(super) fn file_range(&self) -> &Range<usize> {
345        &self.file_range
346    }
347
348    /// Returns the permission to map the virtual memory.
349    pub(super) fn vm_perms(&self) -> VmPerms {
350        self.vm_perms
351    }
352}
353
354fn parse_segment_perm(flags: xmas_elf::program::Flags) -> VmPerms {
355    let mut vm_perm = VmPerms::empty();
356    if flags.is_read() {
357        vm_perm |= VmPerms::READ;
358    }
359    if flags.is_write() {
360        vm_perm |= VmPerms::WRITE;
361    }
362    if flags.is_execute() {
363        vm_perm |= VmPerms::EXEC;
364    }
365    vm_perm
366}

368/// A ELF program header of the type [`program::Type::Interp`].
369pub(super) struct InterpPhdr {
370    file_offset: usize,
371    file_size: u16,
372}
373
374impl InterpPhdr {
375    pub(self) fn parse(phdr: &ProgramHeader64) -> Result<Self> {
376        debug_assert_eq!(phdr.get_type(), Ok(program::Type::Interp));
377
378        if phdr
379            .offset
380            .checked_add(phdr.file_size)
381            .is_none_or(|file_end| file_end > isize::MAX as u64)
382        {
383            return_errno_with_message!(Errno::EINVAL, "the interpreter offset overflows");
384        }
385
386        if phdr.file_size >= PATH_MAX as u64 {
387            return_errno_with_message!(Errno::ENOEXEC, "the interpreter path is too long");
388        }
389
390        const { assert!(PATH_MAX as u64 <= u16::MAX as u64) };
391
392        Ok(Self {
393            file_offset: phdr.offset as usize,
394            file_size: phdr.file_size as u16,
395        })
396    }
397
398    /// Reads the LDSO path from the ELF inode.
399    pub(super) fn read_ldso_path(&self, elf_inode: &Arc<dyn Inode>) -> Result<CString> {
400        // Note that `self.file_size` is at most `PATH_SIZE`.
401        let file_size = self.file_size as usize;
402        let mut buffer = vec![0; file_size];
403        if elf_inode.read_bytes_at(self.file_offset, &mut buffer)? != file_size {
404            return_errno_with_message!(Errno::EIO, "the interpreter path cannot be fully read");
405        }
406
407        let ldso_path = CString::from_vec_with_nul(buffer).map_err(|_| {
408            Error::with_message(
409                Errno::ENOEXEC,
410                "the interpreter path is not a valid C string",
411            )
412        })?;
413        Ok(ldso_path)
414    }
