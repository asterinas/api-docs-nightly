3//! Task sets.
4
5use ostd::{
6    sync::Waker,
7    task::{CurrentTask, Task},
8};
9
10use crate::prelude::*;
11
12/// A task set that maintains all tasks in a POSIX process.
13pub struct TaskSet {
14    tasks: Vec<Arc<Task>>,
15    has_exited_main: bool,
16    has_exited_group: bool,
17    in_execve: bool,
18    execve_waker: Option<Arc<Waker>>,
19}
20
21impl TaskSet {
22    /// Creates a new task set.
23    pub(super) fn new() -> Self {
24        Self {
25            tasks: Vec::new(),
26            has_exited_main: false,
27            has_exited_group: false,
28            in_execve: false,
29            execve_waker: None,
30        }
31    }
32
33    /// Inserts a new task to the task set.
34    ///
35    /// This method will fail if [`Self::set_exited_group`] or [`Self::start_execve`]
36    /// has been called before.
37    pub(super) fn insert(&mut self, task: Arc<Task>) -> core::result::Result<(), Arc<Task>> {
38        if self.has_exited_group || self.in_execve {
39            return Err(task);
40        }
41
42        self.tasks.push(task);
43        Ok(())
44    }
45
46    /// Removes the exited task from the task set if necessary.
47    ///
48    /// The task will be removed from the task set if the corresponding thread is not the main
49    /// thread.
50    ///
51    /// This method will return true if there are no more alive tasks in the task set.
52    ///
53    /// # Panics
54    ///
55    /// This method will panic if the task is not in the task set.
56    pub(super) fn remove_exited(&mut self, task: &CurrentTask) -> bool {
57        let position = self
58            .tasks
59            .iter()
60            .position(|some_task| core::ptr::eq(some_task.as_ref(), task.as_ref()))
61            .unwrap();
62
63        if position == 0 {
64            assert!(!self.has_exited_main);
65            self.has_exited_main = true;
66        } else {
67            self.tasks.swap_remove(position);
68        }
69
70        if let Some(waker) = self.execve_waker.as_ref() {
71            waker.wake_up();
72        }
73
74        self.has_exited_main && self.tasks.len() == 1
75    }
76
77    /// Return whether the main thread has exited.
78    pub(super) fn has_exited_main(&self) -> bool {
79        self.has_exited_main
80    }
81
82    /// Removes the main task and makes the remaining task become the main task.
83    ///
84    /// The method should only be calling when doing execve.
85    pub(super) fn swap_main(&mut self) {
86        // This is an extremely internal method. The caller must uphold certain invariants, update
87        // the thread status, modify the PID table entry, etc.
88
89        self.tasks.swap_remove(0);
90        self.has_exited_main = false;
91    }
92
93    /// Sets a flag that denotes that an `exit_group` has been initiated.
94    pub(super) fn set_exited_group(&mut self) {
95        debug_assert!(!self.in_execve);
96        self.has_exited_group = true;
97    }
98
99    /// Returns whether an `exit_group` has been initiated.
100    pub(super) fn has_exited_group(&self) -> bool {
101        self.has_exited_group
102    }
103
104    /// Sets a flag that denotes that an `execve` has been initiated.
105    pub(super) fn start_execve(&mut self) {
106        debug_assert!(!self.has_exited_group);
107        self.in_execve = true;
108    }
109
110    /// Resets a flag to indicate an `execve` has finished.
111    pub(super) fn finish_execve(&mut self) {
112        self.in_execve = false;
113    }
114
115    /// Returns whether an `execve` has been initiated.
116    pub(super) fn in_execve(&self) -> bool {
117        self.in_execve
118    }
119
120    /// Registers a waker to be notified when any thread exits.
121    ///
122    /// Only a thread performing execve should set this waker; it is used to
123    /// wake the execve-ing thread while it waits for other threads to exit.
124    pub(super) fn set_execve_waker(&mut self, waker: Arc<Waker>) {
125        debug_assert!(self.execve_waker.is_none());
126        self.execve_waker = Some(waker);
127    }
128
129    /// Clears the waker previously set by [`Self::set_execve_waker`].
130    pub(super) fn clear_execve_waker(&mut self) {
131        self.execve_waker = None;
132    }
133}
134
135impl TaskSet {
136    /// Returns a slice of the tasks in the task set.
137    pub fn as_slice(&self) -> &[Arc<Task>] {
138        self.tasks.as_slice()
139    }
140
141    /// Returns the main task/thread.
142    pub fn main(&self) -> &Arc<Task> {
143        &self.tasks[0]
144    }
145}
