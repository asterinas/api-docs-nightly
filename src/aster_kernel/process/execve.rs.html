35pub fn do_execve(
36    elf_file: Path,
37    argv_ptr_ptr: Vaddr,
38    envp_ptr_ptr: Vaddr,
39    ctx: &Context,
40    user_context: &mut UserContext,
41) -> Result<()> {
42    // FIXME: A malicious user could cause a kernel panic by exhausting available memory.
43    // Currently, the implementation reads up to `MAX_NR_STRING_ARGS` arguments, each up to
44    // `MAX_LEN_STRING_ARG` in length, without first verifying the total combined size.
45    // To prevent excessive memory allocation, a preliminary check should sum the lengths
46    // of all strings to enforce a sensible overall limit.
47    let argv = read_cstring_vec(argv_ptr_ptr, MAX_NR_STRING_ARGS, MAX_LEN_STRING_ARG, ctx)?;
48    let envp = read_cstring_vec(envp_ptr_ptr, MAX_NR_STRING_ARGS, MAX_LEN_STRING_ARG, ctx)?;
49
50    let fs_ref = ctx.thread_local.borrow_fs();
51    let path_resolver = fs_ref.resolver().read();
52
53    debug!(
54        "file path: {:?}, argv = {:?}, envp = {:?}",
55        path_resolver.make_abs_path(&elf_file).into_string(),
56        argv,
57        envp
58    );
59
60    let program_to_load =
61        ProgramToLoad::build_from_file(elf_file.clone(), &path_resolver, argv, envp)?;
62
63    let new_vmar = Vmar::new(ProcessVm::new(elf_file.clone()));
64    let elf_load_info = program_to_load.load_to_vmar(new_vmar.as_ref(), &path_resolver)?;
65
66    // Ensure no other thread is concurrently performing exit_group or execve.
67    // If such an operation is in progress, return EAGAIN.
68    let mut task_set = ctx.process.tasks().lock();
69    if task_set.has_exited_group() || task_set.in_execve() {
70        return_errno_with_message!(
71            Errno::EAGAIN,
72            "the process has exited or has already executed a new program"
73        );
74    }
75    task_set.start_execve();
76
77    // Terminate all other threads
78    sigkill_other_threads(ctx.task, &task_set);
79    drop(task_set);
80
81    let former_tid = ctx.posix_thread.tid();
82
83    // After this point, failures in subsequent operations are fatal: the process
84    // state may be left inconsistent and it can never return to user mode.
85
86    let res = do_execve_no_return(
87        ctx,
88        user_context,
89        &path_resolver,
90        elf_file,
91        new_vmar,
92        &elf_load_info,
93    );
94
95    if res.is_ok() {
96        ctx.posix_thread
97            .ptrace_may_stop_on(PtraceEvent::Exec(former_tid), ctx, user_context);
98    } else {
99        ctx.posix_thread
100            .enqueue_signal(Box::new(KernelSignal::new(SIGKILL)));
101    }
102
103    ctx.process.tasks().lock().finish_execve();
104
105    res
106}

108fn read_cstring_vec(
109    array_ptr: Vaddr,
110    max_string_number: usize,
111    max_string_len: usize,
112    ctx: &Context,
113) -> Result<Vec<CString>> {
114    // On Linux, argv pointer and envp pointer can be specified as NULL.
115    if array_ptr == 0 {
116        return Ok(Vec::new());
117    }
118
119    let mut res = Vec::new();
120    let mut read_addr = array_ptr;
121
122    let user_space = ctx.user_space();
123    for _ in 0..max_string_number {
124        let cstring_ptr = user_space.read_val::<usize>(read_addr)?;
125        read_addr += 8;
126
127        if cstring_ptr == 0 {
128            return Ok(res);
129        }
130
131        let cstring = user_space
132            .read_cstring(cstring_ptr, max_string_len)
133            .map_err(|err| {
134                if err.error() == Errno::ENAMETOOLONG {
135                    Error::with_message(Errno::E2BIG, "there are too many bytes in the argument")
136                } else {
137                    err
138                }
139            })?;
140        res.push(cstring);
141    }
142
143    return_errno_with_message!(Errno::E2BIG, "there are too many arguments");
144}

146fn do_execve_no_return(
147    ctx: &Context,
148    user_context: &mut UserContext,
149    path_resolver: &PathResolver,
150    elf_file: Path,
151    new_vmar: Arc<Vmar>,
152    elf_load_info: &ElfLoadInfo,
153) -> Result<()> {
154    let Context {
155        process,
156        thread_local,
157        posix_thread,
158        ..
159    } = ctx;
160
161    // Wait for all other threads to terminate,
162    // then promote the current thread to be the process's main thread if necessary.
163    wait_other_threads_exit(ctx)?;
164    make_current_main_thread(ctx);
165
166    // Activate the new VMAR in the current context and apply file-capability changes,
167    // while holding the process VMAR lock.
168    // This prevents race conditions when checking access permissions while opening
169    // `/proc/[pid]/mem` or `/proc/[pid]/maps`.
170    let vmar_guard = activate_vmar(ctx, new_vmar);
171    apply_caps_from_exec(process, ctx.credentials_mut(), elf_file.inode())?;
172    drop(vmar_guard);
173
174    // After the program has been successfully loaded, the virtual memory of the current process
175    // is initialized. Hence, it is necessary to clear the previously recorded robust list.
176    *thread_local.robust_list().borrow_mut() = None;
177    thread_local.clear_child_tid().set(0);
178
179    // Set up the CPU context.
180    set_cpu_context(thread_local, user_context, elf_load_info);
181
182    // If this was a vfork child, reset vfork-specific state.
183    reset_vfork_child(process);
184
185    // Unshare file descriptor table and close files with O_CLOEXEC flag.
186    unshare_and_close_files(ctx);
187
188    // Update the process's executable path and set the thread name
189    let executable_path = path_resolver.make_abs_path(&elf_file).into_string();
190    *posix_thread.thread_name().lock() = ThreadName::new_from_executable_path(&executable_path);
191
192    // Unshare and reset signal dispositions to their default actions.
193    unshare_and_reset_sigdispositions(process);
194    // Reset the alternate signal stack to its default state.
195    *thread_local.sig_stack().borrow_mut() = SigStack::default();
196    // Restore the process exit signal to SIGCHLD.
197    process.set_exit_signal(SIGCHLD);
198
199    Ok(())
200}

202fn wait_other_threads_exit(ctx: &Context) -> Result<()> {
203    let is_main_thread = ctx.posix_thread.tid() == ctx.process.pid();
204
205    let mut tasks = ctx.process.tasks().lock();
206    loop {
207        if is_main_thread {
208            if tasks.as_slice().len() == 1 {
209                return Ok(());
210            }
211        } else if tasks.as_slice().len() == 2 && tasks.has_exited_main() {
212            return Ok(());
213        }
214
215        // Wait until any signal comes or any other thread exits.
216        let (waiter, waker) = Waiter::new_pair();
217
218        ctx.posix_thread
219            .set_signalled_waker(waker.clone(), PauseReason::Sleep);
220        if ctx.has_pending_sigkill() {
221            ctx.posix_thread.clear_signalled_waker();
222            return_errno_with_message!(Errno::EAGAIN, "the current thread has received SIGKILL");
223        }
224
225        tasks.set_execve_waker(waker);
226        drop(tasks);
227
228        waiter.wait();
229
230        ctx.posix_thread.clear_signalled_waker();
231
232        tasks = ctx.process.tasks().lock();
233        tasks.clear_execve_waker();
234    }
235}

237fn make_current_main_thread(ctx: &Context) {
238    let pid = ctx.process.pid();
239    let old_tid = ctx.posix_thread.tid();
240
241    // The current thread is already the main thread.
242    if old_tid == pid {
243        return;
244    }
245
246    // The current thread is not the main thread.
247
248    // Lock order: PID table -> tracer.tracees -> tasks of process
249    let mut pid_table = pid_table::pid_table_mut();
250    let tracer = ctx.posix_thread.tracer();
251    let mut tracer_tracees = tracer
252        .as_ref()
253        .map(|tracer| tracer.as_posix_thread().unwrap().tracees().unwrap().lock());
254    let mut tasks = ctx.process.tasks().lock();
255
256    assert!(tasks.has_exited_main());
257    assert!(tasks.in_execve());
258    assert_eq!(tasks.as_slice().len(), 2);
259    assert!(core::ptr::eq(ctx.task, tasks.as_slice()[1].as_ref()));
260
261    tasks.swap_main();
262    ctx.posix_thread.set_main(pid);
263
264    if let Some(tracer_tracees) = tracer_tracees.as_mut()
265        && let Some(tracee) = tracer_tracees.remove(&old_tid)
266    {
267        tracer_tracees.insert(pid, tracee);
268    }
269
270    drop(tasks);
271    drop(tracer_tracees);
272
273    let thread = pid_table.take_thread(old_tid).unwrap();
274    pid_table.replace_thread(pid, &thread);
275}

277fn set_cpu_context(
278    thread_local: &ThreadLocal,
279    user_context: &mut UserContext,
280    elf_load_info: &ElfLoadInfo,
281) {
282    // Reset FPU context.
283    thread_local.fpu().set_context(FpuContext::new());
284
285    // Reset general-purpose registers.
286    *user_context.general_regs_mut() = GeneralRegs::default();
287    // Clear the TLS pointer.
288    user_context.set_tls_pointer(0);
289    // Set the new instruction pointer to the ELF entry point.
290    user_context.set_instruction_pointer(elf_load_info.entry_point as _);
291    debug!("entry_point: 0x{:x}", elf_load_info.entry_point);
292    // Set the new user-space stack pointer.
293    user_context.set_stack_pointer(elf_load_info.user_stack_top as _);
294    debug!("user stack top: 0x{:x}", elf_load_info.user_stack_top);
295}

297/// Sets the UID and GID in the credentials according to the ELF inode.
298///
299/// The capabilities will be updated accordingly.
300fn apply_caps_from_exec(
301    process: &Process,
302    credentials: Credentials<ReadWriteOp>,
303    elf_inode: &Arc<dyn Inode>,
304) -> Result<()> {
305    set_uid_from_elf(process, &credentials, elf_inode)?;
306    set_gid_from_elf(process, &credentials, elf_inode)?;
307    credentials.set_keep_capabilities(false)?;
308
309    Ok(())
310}

312/// Sets the UID in the credentials according to the ELF inode.
313///
314/// If the ELF inode has the `set_uid` bit, the effective UID is set to the same value as the ELF
315/// inode's UID.
316fn set_uid_from_elf(
317    current: &Process,
318    credentials: &Credentials<ReadWriteOp>,
319    elf_inode: &Arc<dyn Inode>,
320) -> Result<()> {
321    if elf_inode.mode()?.has_set_uid() {
322        let uid = elf_inode.owner()?;
323        credentials.set_euid(uid);
324
325        current.clear_parent_death_signal();
326    }
327
328    // No matter whether the ELF inode has `set_uid` bit, SUID should be reset.
329    credentials.reset_suid();
330    Ok(())
331}
332
333/// Sets the GID in the credentials according to the ELF inode.
334///
335/// If the ELF inode has the `set_gid` bit, the effective GID is set to the same value as the ELF
336/// inode's GID.
337fn set_gid_from_elf(
338    current: &Process,
339    credentials: &Credentials<ReadWriteOp>,
340    elf_inode: &Arc<dyn Inode>,
341) -> Result<()> {
342    if elf_inode.mode()?.has_set_gid() {
343        let gid = elf_inode.group()?;
344        credentials.set_egid(gid);
345
346        current.clear_parent_death_signal();
347    }
348
349    // No matter whether the ELF inode has `set_gid` bit, SGID should be reset.
350    credentials.reset_sgid();
351    Ok(())
352}

354fn reset_vfork_child(process: &Process) {
355    if process.status().is_vfork_child() {
356        // Resumes the parent process.
357        process.status().set_vfork_child(false);
358        let parent = process.parent().lock().process().upgrade().unwrap();
359        parent.children_wait_queue().wake_all();
360    }
361}

363fn unshare_and_close_files(ctx: &Context) {
364    ctx.unshare_files();
365
366    ctx.thread_local
367        .borrow_file_table()
368        .unwrap()
369        .write()
370        .close_files_on_exec();
371}

373fn unshare_and_reset_sigdispositions(process: &Process) {
374    let mut sig_dispositions = process.sig_dispositions().lock();
375
376    let mut new = *sig_dispositions.lock();
377    new.inherit();
378
379    *sig_dispositions = Arc::new(Mutex::new(new));
