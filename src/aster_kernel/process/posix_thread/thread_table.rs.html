3use super::{Thread, Tid};
4use crate::{prelude::*, process::posix_thread::AsPosixThread};
5
6static THREAD_TABLE: SpinLock<BTreeMap<Tid, Arc<Thread>>> = SpinLock::new(BTreeMap::new());
7
8/// Adds a posix thread to global thread table
9pub fn add_thread(tid: Tid, thread: Arc<Thread>) {
10    debug_assert_eq!(tid, thread.as_posix_thread().unwrap().tid());
11    THREAD_TABLE.lock().insert(tid, thread);
12}
13
14/// Removes a posix thread to global thread table
15pub fn remove_thread(tid: Tid) {
16    THREAD_TABLE.lock().remove(&tid);
17}
18
19/// Gets a posix thread from the global thread table
20pub fn get_thread(tid: Tid) -> Option<Arc<Thread>> {
21    THREAD_TABLE.lock().get(&tid).cloned()
22}
23
24/// Makes the current thread become the main thread if necessary.
25pub(in crate::process) fn make_current_main_thread(ctx: &Context) {
26    let pid = ctx.process.pid();
27    let old_tid = ctx.posix_thread.tid();
28
29    // The current thread is already the main thread.
30    if old_tid == pid {
31        return;
32    }
33
34    // The current thread is not the main thread.
35    let mut tasks = ctx.process.tasks().lock();
36    let mut thread_table = THREAD_TABLE.lock();
37
38    assert!(tasks.has_exited_main());
39    assert!(tasks.in_execve());
40    assert_eq!(tasks.as_slice().len(), 2);
41    assert!(core::ptr::eq(ctx.task, tasks.as_slice()[1].as_ref()));
42
43    tasks.swap_main(pid, old_tid);
44    ctx.posix_thread.set_main(pid);
45
46    thread_table.remove(&pid).unwrap();
47    let thread = thread_table.remove(&old_tid).unwrap();
48    thread_table.insert(pid, thread);
49}
