3use crate::prelude::*;
4
5pub const MAX_THREAD_NAME_LEN: usize = 16;
6
7#[derive(Debug, Clone)]
8pub struct ThreadName([u8; MAX_THREAD_NAME_LEN]);
9
10impl ThreadName {
11    fn new() -> Self {
12        ThreadName([0; MAX_THREAD_NAME_LEN])
13    }
14
15    pub fn new_from_executable_path(executable_path: &str) -> Self {
16        let mut thread_name = ThreadName::new();
17        let Some(file_name) = executable_path.split('/').next_back() else {
18            return thread_name;
19        };
20
21        thread_name.set_name_as_bytes(file_name.as_bytes());
22        thread_name
23    }
24
25    pub fn set_name(&mut self, name: &CStr) {
26        self.set_name_as_bytes(name.to_bytes());
27    }
28
29    fn set_name_as_bytes(&mut self, name_as_bytes: &[u8]) {
30        let name_len = name_as_bytes.len().min(MAX_THREAD_NAME_LEN - 1);
31        self.0[..name_len].copy_from_slice(&name_as_bytes[..name_len]);
32        self.0[name_len..].fill(0);
33    }
34
35    pub fn name(&self) -> &CStr {
36        CStr::from_bytes_until_nul(&self.0).unwrap()
37    }
38}
