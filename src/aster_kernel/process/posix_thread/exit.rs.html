21/// Exits the current POSIX thread.
22///
23/// # Panics
24///
25/// If the current thread is not a POSIX thread, this method will panic.
26pub fn do_exit(term_status: TermStatus) {
27    exit_internal(term_status, false);
28}

39/// Exits the current POSIX thread or process.
40fn exit_internal(term_status: TermStatus, is_exiting_group: bool) {
41    let current_task = Task::current().unwrap();
42    let current_thread = current_task.as_thread().unwrap();
43    let posix_thread = current_thread.as_posix_thread().unwrap();
44    let thread_local = current_task.as_thread_local().unwrap();
45    let posix_process = posix_thread.process();
46
47    let is_last_thread = {
48        let mut tasks = posix_process.tasks().lock();
49        let has_exited_group = tasks.has_exited_group();
50        let in_evecve = tasks.in_execve();
51
52        if is_exiting_group && !has_exited_group && !in_evecve {
53            sigkill_other_threads(&current_task, &tasks);
54            tasks.set_exited_group();
55        }
56
57        // According to Linux's behavior, the last thread's exit code will become the process's
58        // exit code, so here we should just overwrite the old value (if any).
59        if !has_exited_group && !in_evecve {
60            posix_process.status().set_exit_code(term_status.as_u32());
61        }
62
63        // We should only change the thread status when running as the thread, so no race
64        // conditions can occur in between.
65        if current_thread.is_exited() {
66            return;
67        }
68        current_thread.exit();
69
70        tasks.remove_exited(&current_task, posix_thread.tid())
71    };
72
73    wake_clear_ctid(thread_local);
74
75    wake_robust_list(thread_local, posix_thread.tid());
76
77    // According to Linux behavior, the main thread shouldn't be removed from the table until the
78    // process is reaped by its parent.
79    if posix_thread.tid() != posix_process.pid() {
80        thread_table::remove_thread(posix_thread.tid());
81    }
82
83    // Drop fields in `PosixThread`.
84    *posix_thread.file_table().lock() = None;
85    *posix_thread.ns_proxy().lock() = None;
86
87    // Drop fields in `ThreadLocal`.
88    *thread_local.vmar().borrow_mut() = None;
89    thread_local.borrow_file_table_mut().remove();
90    thread_local.borrow_ns_proxy_mut().remove();
91
92    if is_last_thread {
93        exit_process(&posix_process);
94    }
95}

97/// Sends `SIGKILL` to all other threads in the current process.
98pub(in crate::process) fn sigkill_other_threads(current_task: &Task, task_set: &TaskSet) {
99    debug_assert!(
100        task_set
101            .as_slice()
102            .iter()
103            .any(|task| core::ptr::eq(current_task, task.as_ref()))
104    );
105
106    for task in task_set.as_slice() {
107        if core::ptr::eq(current_task, task.as_ref()) {
108            continue;
109        }
110        task.as_posix_thread()
111            .unwrap()
112            .enqueue_signal(Box::new(KernelSignal::new(SIGKILL)));
113    }
114}

116/// Writes zero to `clear_child_tid` and performs a futex wake.
117fn wake_clear_ctid(thread_local: &ThreadLocal) {
118    let clear_ctid = thread_local.clear_child_tid().get();
119
120    if clear_ctid == 0 {
121        return;
122    }
123
124    let _ = current_userspace!()
125        .write_val(clear_ctid, &0u32)
126        .inspect_err(|err| debug!("exit: cannot clear the child TID: {:?}", err));
127    let _ = futex_wake(clear_ctid, 1, None)
128        .inspect_err(|err| debug!("exit: cannot wake the futex on the child TID: {:?}", err));
129
130    thread_local.clear_child_tid().set(0);
131}

133/// Walks the robust futex list, marking futex dead and waking waiters.
134///
135/// This corresponds to Linux's `exit_robust_list`. Errors are silently ignored.
136fn wake_robust_list(thread_local: &ThreadLocal, tid: Tid) {
137    let list_head = match thread_local.robust_list().borrow_mut().take() {
138        Some(robust_list_head) => robust_list_head,
139        None => return,
140    };
141
142    trace!("exit: wake up the robust list: {:?}", list_head);
143    for futex_addr in list_head.futexes() {
144        if let Err(err) = wake_robust_futex(futex_addr, tid) {
145            debug!(
146                "exit: cannot wake the robust futex at {:#x}: {:?}",
147                futex_addr, err
148            );
149            return;
150        }
151    }
152}
