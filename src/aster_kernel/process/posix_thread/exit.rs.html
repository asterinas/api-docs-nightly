24/// Exits the current POSIX thread.
25///
26/// # Panics
27///
28/// If the current thread is not a POSIX thread, this method will panic.
29pub fn do_exit(term_status: TermStatus, ctx: &Context, user_ctx: &mut UserContext) {
30    exit_internal(term_status, false, ctx, user_ctx);
31}

33/// Kills all threads and exits the current POSIX process.
34///
35/// # Panics
36///
37/// If the current thread is not a POSIX thread, this method will panic.
38pub fn do_exit_group(term_status: TermStatus, ctx: &Context, user_ctx: &mut UserContext) {
39    exit_internal(term_status, true, ctx, user_ctx);

42/// Exits the current POSIX thread or process.
43//
44// A mutable reference to `user_ctx` is needed because exiting a traced thread
45// may trigger a "exit" ptrace-event-stop, which snapshots the current user
46// register state and may later restore tracer-updated registers.
47fn exit_internal(
48    term_status: TermStatus,
49    is_exiting_group: bool,
50    ctx: &Context,
51    user_ctx: &mut UserContext,
52) {
53    let exit_code = term_status.as_u32();
54    ctx.posix_thread
55        .ptrace_may_stop_on(PtraceEvent::Exit(exit_code), ctx, user_ctx);
56
57    let current_task = Task::current().unwrap();
58    let current_thread = current_task.as_thread().unwrap();
59    let posix_thread = current_thread.as_posix_thread().unwrap();
60    let thread_local = current_task.as_thread_local().unwrap();
61    let posix_process = posix_thread.process();
62
63    let is_last_thread = {
64        let mut tasks = posix_process.tasks().lock();
65        let has_exited_group = tasks.has_exited_group();
66        let in_evecve = tasks.in_execve();
67
68        if is_exiting_group && !has_exited_group && !in_evecve {
69            sigkill_other_threads(&current_task, &tasks);
70            tasks.set_exited_group();
71        }
72
73        // According to Linux's behavior, the last thread's exit code will become the process's
74        // exit code, so here we should just overwrite the old value (if any).
75        if !has_exited_group && !in_evecve {
76            posix_process.status().set_exit_code(exit_code);
77        }
78
79        posix_thread.set_exit_code(exit_code);
80
81        // We should only change the thread status when running as the thread, so no race
82        // conditions can occur in between.
83        if current_thread.is_exited() {
84            return;
85        }
86        current_thread.exit();
87
88        tasks.remove_exited(&current_task)
89    };
90
91    // This is put after `current_thread.exit()`,
92    // so `attach_tracee` will observe that the tracer has exited while
93    // holding the `tracees` lock, and can not race with `clear_tracees`.
94    posix_thread.clear_tracees();
95
96    if let Some(tracer) = posix_thread.tracer() {
97        let tracer = tracer.as_posix_thread().unwrap();
98        tracer.process().children_wait_queue().wake_all();
99    }
100
101    wake_clear_ctid(thread_local);
102
103    wake_robust_list(thread_local, posix_thread.tid());
104
105    // According to Linux behavior, the main thread shouldn't be removed from the table until the
106    // process is reaped by its parent.
107    if posix_thread.tid() != posix_process.pid() {
108        pid_table::pid_table_mut().remove_thread(posix_thread.tid());
109    }
110
111    // Drop fields in `PosixThread`.
112    *posix_thread.file_table().lock() = None;
113    *posix_thread.ns_proxy().lock() = None;
114
115    // Drop fields in `ThreadLocal`.
116    *thread_local.vmar().borrow_mut() = None;
117    thread_local.borrow_file_table_mut().remove();
118    thread_local.borrow_ns_proxy_mut().remove();
119
120    if is_last_thread {
121        exit_process(&posix_process);
122    }
123}

125/// Sends `SIGKILL` to all other threads in the current process.
126pub(in crate::process) fn sigkill_other_threads(current_task: &Task, task_set: &TaskSet) {
127    debug_assert!(
128        task_set
129            .as_slice()
130            .iter()
131            .any(|task| core::ptr::eq(current_task, task.as_ref()))
132    );
133
134    for task in task_set.as_slice() {
135        if core::ptr::eq(current_task, task.as_ref()) {
136            continue;
137        }
138        task.as_posix_thread()
139            .unwrap()
140            .enqueue_signal(Box::new(KernelSignal::new(SIGKILL)));
141    }
142}

144/// Writes zero to `clear_child_tid` and performs a futex wake.
145fn wake_clear_ctid(thread_local: &ThreadLocal) {
146    let clear_ctid = thread_local.clear_child_tid().get();
147
148    if clear_ctid == 0 {
149        return;
150    }
151
152    let _ = current_userspace!()
153        .write_val(clear_ctid, &0u32)
154        .inspect_err(|err| debug!("exit: cannot clear the child TID: {:?}", err));
155    let _ = futex_wake(clear_ctid, 1, FutexVisibility::Shared)
156        .inspect_err(|err| debug!("exit: cannot wake the futex on the child TID: {:?}", err));
157
158    thread_local.clear_child_tid().set(0);
159}

161/// Walks the robust futex list, marking futex dead and waking waiters.
162///
163/// This corresponds to Linux's `exit_robust_list`. Errors are silently ignored.
164fn wake_robust_list(thread_local: &ThreadLocal, tid: Tid) {
165    let list_head = match thread_local.robust_list().borrow_mut().take() {
166        Some(robust_list_head) => robust_list_head,
167        None => return,
168    };
169
170    debug!("exit: wake up the robust list: {:?}", list_head);
171    for futex_addr in list_head.futexes() {
172        if let Err(err) = wake_robust_futex(futex_addr, tid) {
173            debug!(
174                "exit: cannot wake the robust futex at {:#x}: {:?}",
175                futex_addr, err
176            );
177            return;
178        }
179    }
