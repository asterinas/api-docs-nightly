155    /// Sets the input [`Waker`] as the signalled waker of this thread,
156    /// along with the reason why the thread is paused.
157    ///
158    /// This approach can collaborate with signal-aware wait methods.
159    /// Once a signalled waker is set for a thread, it cannot be reset until it is cleared.
160    ///
161    /// # Panics
162    ///
163    /// If setting a new waker before clearing the current thread's signalled waker
164    /// this method will panic.
165    pub fn set_signalled_waker(&self, waker: Arc<Waker>, reason: PauseReason) {
166        let mut signalled_waker = self.signalled_waker.lock();
167        assert!(signalled_waker.is_none());
168        *signalled_waker = Some((waker, reason));
169    }
170
171    /// Clears the signalled waker of this thread.
172    pub fn clear_signalled_waker(&self) {
173        *self.signalled_waker.lock() = None;
174    }

176    /// Returns the sleeping state of this thread.
177    pub fn sleeping_state(&self) -> SleepingState {
178        // This implementation prevents a thread (let's call it `threadA`) that is
179        // sleeping in an interruptible wait from being mistakenly reported as
180        // sleeping in an uninterruptible wait due to a race condition, where another
181        // thread (`threadB`) may observe that its `task.schedule_info().cpu` is
182        // `AtomicCpuId::NONE` and its `signalled_waker` is `None` (not set yet or
183        // already cleared).
184        //
185        // When `threadA` enters an interruptible wait, it executes the following steps:
186        // ```
187        // A1: Acquire signalled_waker.lock |
188        // A2: set signalled_waker to Some  |-- critical section #1
189        // A3: Release signalled_waker.lock |
190        // A4: cpu.set_to_none(Relaxed)
191        // A5: cpu.set_if_is_none(cpuid, Relaxed)
192        // A6: Acquire signalled_waker.lock |
193        // A7: set signalled_waker to None  |-- critical section #2
194        // A8: Release signalled_waker.lock |
195        // ```
196        //
197        // When `threadB` calls `threadA.sleeping_state()`, it executes the following steps:
198        // ```
199        // B1: Acquire threadA.signalled_waker.lock |
200        // B2: check threadA.signalled_waker        |-- critical section #3
201        // B3: check threadA.cpu.get(Relaxed)       |
202        // B4: Release threadA.signalled_waker.lock |
203        // ```
204        //
205        // We can see that:
206        //  - If #3 happens before #1, B3 can not observe the effect of A4 due to the
207        //    release-acquire pair B4-A1.
208        //  - If #3 happens between #1 and #2, B2 will always see a `Some`.
209        //  - If #3 happens after #2, B3 can observe the effect of A5 due to the
210        //    release-acquire pair A8-B1.
211        // Therefore, the condition where both B2 and B3 see `None` will never happen.
212        //
213        // Similarly, this implementation prevents a process that has been stopped by
214        // a signal or ptrace from being incorrectly reported as sleeping in an
215        // (un)interruptible wait.
216        //
217        // FIXME: This implementation cannot prevent a stopped process from being
218        // reported as running when `crate::process::signal::handle_pending_signal`
219        // is called, but the pending signal is not a `SIGCONT`. However, is this
220        // actually a problem? We considered an approach to fix this issue, but it
221        // does not fully resolve it and has some drawbacks. For more details, see
222        // <https://github.com/asterinas/asterinas/pull/2491#issuecomment-3527958970>.
223        let signalled_waker = self.signalled_waker.lock();
224        let task = self.task.upgrade().unwrap();
225        match (
226            signalled_waker.as_ref(),
227            task.schedule_info().cpu.get().is_none(),
228        ) {
229            (Some((_, PauseReason::Sleep)), true) => SleepingState::Interruptible,
230            (Some((_, PauseReason::StopBySignal)), true) => SleepingState::StopBySignal,
231            (Some((_, PauseReason::StopByPtrace)), true) => SleepingState::StopByPtrace,
232            (None, true) => SleepingState::Uninterruptible,
233            (_, false) => SleepingState::Running,
234        }
235    }

237    /// Wakes up the signalled waker.
238    pub fn wake_signalled_waker(&self) {
239        if let Some((waker, _)) = &*self.signalled_waker.lock() {
240            waker.wake_up();
241        }
242    }

244    /// Enqueues a thread-directed signal.
245    ///
246    /// This method does not perform permission checks on user signals.
247    /// Therefore, unless the caller can ensure that there are no permission issues,
248    /// this method should be used to enqueue kernel signals or fault signals.
249    pub fn enqueue_signal(&self, signal: Box<dyn Signal>) {
250        self.sig_queues.enqueue(signal);
251        self.wake_signalled_waker();
252    }

254    pub fn register_signalfd_poller(&self, poller: &mut PollHandle, mask: IoEvents) {
255        self.sig_queues.register_signalfd_poller(poller, mask);
256        self.process()
257            .sig_queues()
258            .register_signalfd_poller(poller, mask);
259    }

338static POSIX_TID_ALLOCATOR: AtomicU32 = AtomicU32::new(1);
339
340/// Allocates a new tid for the new posix thread
341pub fn allocate_posix_tid() -> Tid {
342    let tid = POSIX_TID_ALLOCATOR.fetch_add(1, Ordering::SeqCst);
343    if tid >= PID_MAX {
344        // When the kernel's next PID value reaches `PID_MAX`,
345        // it should wrap back to a minimum PID value.
346        // PIDs with a value of `PID_MAX` or larger should not be allocated.
347        // Reference: <https://docs.kernel.org/admin-guide/sysctl/kernel.html#pid-max>.
348        //
349        // FIXME: Currently, we cannot determine which PID is recycled,
350        // so we are unable to allocate smaller PIDs.
351        warn!("the allocated ID is greater than the maximum allowed PID");
352    }
353    tid
354}
355
356/// Returns the last allocated tid
357pub fn last_tid() -> Tid {
358    POSIX_TID_ALLOCATOR.load(Ordering::SeqCst) - 1
359}

361/// The maximum allowed process ID.
362//
363// FIXME: The current value is chosen arbitrarily.
364// This value can be modified by the user by writing to `/proc/sys/kernel/pid_max`.
365pub const PID_MAX: u32 = u32::MAX / 2;
