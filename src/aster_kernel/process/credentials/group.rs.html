3use core::sync::atomic::{AtomicU32, Ordering};
4
5use atomic_integer_wrapper::define_atomic_version_of_integer_like_type;
6
7use crate::prelude::*;
8
9#[derive(Debug, Clone, Copy, Pod, Default, PartialEq, Eq, PartialOrd, Ord)]
10#[repr(C)]
11pub struct Gid(u32);
12
13impl Gid {
14    /// The invalid GID, typically used to indicate that no valid GID is found when returning to user space.
15    ///
16    /// Reference: <https://elixir.bootlin.com/linux/v6.15/source/include/linux/uidgid.h#L51>.
17    pub const INVALID: Gid = Gid(u32::MAX);
18
19    /// The overflow GID, typically used to indicate that group mappings between namespaces fail.
20    ///
21    /// This is currently a constant (65534 is usually the "nobody" group), but it should be
22    /// configured via `/proc/sys/kernel/overflowgid`.
23    ///
24    /// Reference: <https://elixir.bootlin.com/linux/v6.15/source/kernel/sys.c#L167>.
25    pub const OVERFLOW: Gid = Self::new(65534);
26
27    pub const fn new(gid: u32) -> Self {
28        Self(gid)
29    }
30
31    pub const fn new_root() -> Self {
32        Self(ROOT_GID)
33    }
34
35    pub const fn is_root(&self) -> bool {
36        self.0 == ROOT_GID
37    }
38}
39
40const ROOT_GID: u32 = 0;
41
42impl From<u32> for Gid {
43    fn from(value: u32) -> Self {
44        Self::new(value)
45    }
46}
47
48impl From<Gid> for u32 {
49    fn from(value: Gid) -> Self {
50        value.0
51    }
52}
53
54define_atomic_version_of_integer_like_type!(Gid, {
55    #[derive(Debug)]
56    pub(super) struct AtomicGid(AtomicU32);
57});
58
59impl Clone for AtomicGid {
60    fn clone(&self) -> Self {
61        Self::new(self.load(Ordering::Relaxed))
62    }
63}
