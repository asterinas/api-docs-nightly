10impl<R: TRights> Credentials<R> {
11    /// Creates a root `Credentials`.
12    ///
13    /// This method can only be used when creating the init process.
14    pub fn new_root() -> Self {
15        let uid = Uid::new_root();
16        let gid = Gid::new_root();
17        let cap = CapSet::new_root();
18        let credentials_ = Arc::new(Credentials_::new(uid, gid, cap));
19        Self(credentials_, R::new())
20    }
21
22    /// Clones a new `Credentials` from an existing `Credentials`.
23    ///
24    /// This method requires the `Read` right.
25    #[require(R1 > Read)]
26    pub fn new_from<R1: TRights>(credentials: &Credentials<R1>) -> Self {
27        let credentials_ = Arc::new(credentials.0.as_ref().clone());
28
29        Self(credentials_, R::new())
30    }
31
32    /// Duplicates the capabilities.
33    ///
34    /// This method requires the `Dup` right.
35    #[require(R > Dup)]
36    pub fn dup(&self) -> Self {
37        Self(self.0.clone(), self.1)
38    }
39
40    /// Restricts capabilities to a smaller set.
41    #[require(R > R1)]
42    pub fn restrict<R1: TRights>(self) -> Credentials<R1> {
43        let Credentials(credentials_, _) = self;
44
45        Credentials(credentials_, R1::new())
46    }

93    /// Sets the real, effective user IDs as `ruid`, `euid` respectively.

97    /// This method requires the `Write` right.

107    /// This method requires the `Write` right.
108    #[require(R > Write)]
