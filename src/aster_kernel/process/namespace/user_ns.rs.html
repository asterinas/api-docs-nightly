3use spin::Once;
4
5use crate::{
6    fs::pseudofs::{NsCommonOps, NsType, StashedDentry},
7    prelude::*,
8    process::{Uid, credentials::capabilities::CapSet, posix_thread::PosixThread},
9};
10
11/// The user namespace.
12pub struct UserNamespace {
13    stashed_dentry: StashedDentry,
14}
15
16impl UserNamespace {
17    /// Returns a reference to the singleton initial user namespace.
18    pub fn get_init_singleton() -> &'static Arc<UserNamespace> {
19        static INIT: Once<Arc<UserNamespace>> = Once::new();
20
21        INIT.call_once(|| {
22            Arc::new(Self {
23                stashed_dentry: StashedDentry::new(),
24            })
25        })
26    }
27
28    /// Checks whether the thread has the required capability in this user namespace.
29    pub fn check_cap(&self, required: CapSet, posix_thread: &PosixThread) -> Result<()> {
30        // Since creating new user namespaces is not supported at the moment,
31        // there is effectively only one user namespace in the entire system.
32        // Therefore, the thread has a single set of capabilities used for permission checks.
33        // FIXME: Once support for creating new user namespaces is added,
34        // we should verify the thread's capabilities within the relevant user namespace.
35        let cap_set = posix_thread.credentials().effective_capset();
36        if cap_set.contains(required) {
37            return Ok(());
38        }
39
40        return_errno_with_message!(
41            Errno::EPERM,
42            "the thread does not have the required capability"
43        )
44    }
45
46    /// Returns the owner UID of the user namespace.
47    pub fn owner_uid(&self) -> Result<Uid> {
48        // FIXME: The owner of the user namespace is not yet tracked.
49        // Return the correct user ID once ownership tracking is implemented.
50        Ok(Uid::new_root())
51    }
52
53    /// Returns whether this namespace is the same as, or an ancestor of, the other namespace.
54    pub fn is_same_or_ancestor_of(self: &Arc<Self>, other: &Arc<Self>) -> bool {
55        // FIXME: Creating new user namespaces is not yet supported,
56        // so we simply check pointer equality.
57        // Once user namespace creation is implemented,
58        // this should walk up the ancestor chain to verify
59        // whether `self` is an ancestor of `other`.
60        Arc::ptr_eq(self, other)
61    }
62}
63
64impl NsCommonOps for UserNamespace {
65    const TYPE: NsType = NsType::User;
66
67    fn owner_user_ns(&self) -> Option<&Arc<UserNamespace>> {
68        // For user namespaces, `NS_GET_USERNS` returns the parent user namespace
69        // rather than an "owner". The initial user namespace has no parent.
70        // Reference: <https://elixir.bootlin.com/linux/v6.19/source/kernel/user_namespace.c#L1406>
71        None
72    }
73
74    fn parent(&self) -> Result<&Arc<Self>> {
75        // User namespaces do not support `NS_GET_PARENT`.
76        // Reference: <https://elixir.bootlin.com/linux/v6.19/source/kernel/user_namespace.c#L1407>
77        return_errno_with_message!(Errno::EPERM, "user namespaces do not support NS_GET_PARENT");
78    }
79
80    fn stashed_dentry(&self) -> &StashedDentry {
81        &self.stashed_dentry
82    }
83}
