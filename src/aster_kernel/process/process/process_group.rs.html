3use alloc::collections::btree_map::Values;
4
5use super::{BOOTSTRAP_PGID, BOOTSTRAP_SID, Pgid, Pid, Process, Session};
6use crate::{prelude::*, process::signal::signals::Signal};

8/// A process group.
9///
10/// A process group represents a set of processes,
11/// which has a unique identifier PGID (i.e., [`Pgid`]).
12pub struct ProcessGroup {
13    pgid: Pgid,
14    session: Arc<Session>,
15    inner: Mutex<Inner>,
16}
17
18struct Inner {
19    processes: BTreeMap<Pid, Weak<Process>>,
20}
21
22impl ProcessGroup {
23    /// Creates a new process group with one process.
24    ///
25    /// The PGID is the same as the process ID, which means that the process will become the leader
26    /// process of the new process group.
27    ///
28    /// The caller needs to ensure that the process does not belong to other process group.
29    pub(super) fn new(process: &Arc<Process>, session: Arc<Session>) -> Arc<Self> {
30        Self::new_with(process.pid(), process, session)
31    }
32
33    /// Creates the bootstrap process group for the init process.
34    ///
35    /// The caller needs to ensure that the process is the init process,
36    /// the process does not belong to other process group, and
37    /// the session is the bootstrap session.
38    pub(super) fn new_bootstrap(process: &Arc<Process>, session: Arc<Session>) -> Arc<Self> {
39        debug_assert!(process.is_init_process());
40        debug_assert_eq!(session.sid(), BOOTSTRAP_SID);
41
42        Self::new_with(BOOTSTRAP_PGID, process, session)
43    }
44
45    fn new_with(pgid: Pgid, process: &Arc<Process>, session: Arc<Session>) -> Arc<Self> {
46        let inner = {
47            let mut processes = BTreeMap::new();
48            processes.insert(process.pid(), Arc::downgrade(process));
49            Inner { processes }
50        };
51
52        Arc::new(ProcessGroup {
53            pgid,
54            session,
55            inner: Mutex::new(inner),
56        })
57    }
58
59    /// Returns the process group identifier.
60    pub fn pgid(&self) -> Pgid {
61        self.pgid
62    }
63
64    /// Returns the session to which the process group belongs.
65    pub fn session(&self) -> &Arc<Session> {
66        &self.session
67    }
68
69    /// Acquires a lock on the process group.
70    pub fn lock(&self) -> ProcessGroupGuard<'_> {
71        ProcessGroupGuard {
72            inner: self.inner.lock(),
73        }
74    }
75
76    /// Broadcasts the signal to all processes in the process group.
77    ///
78    /// This method should only be used to broadcast fault signals and kernel signals.
79    //
80    // TODO: Do some checks to forbid user signals.
81    pub fn broadcast_signal(&self, signal: impl Signal + Clone + 'static) {
82        for process in self
83            .inner
84            .lock()
85            .processes
86            .values()
87            .filter_map(Weak::upgrade)
88        {
89            process.enqueue_signal(Box::new(signal.clone()));
90        }
91    }
92}
93
94/// A scoped lock guard for a process group.
95///
96/// It provides some public methods to prevent the exposure of the inner type.
97#[clippy::has_significant_drop]
98#[must_use]
99pub struct ProcessGroupGuard<'a> {
100    inner: MutexGuard<'a, Inner>,
101}
102
103impl ProcessGroupGuard<'_> {
104    /// Returns an iterator over the processes in the process group.
105    pub fn iter(&self) -> ProcessGroupIter<'_> {
106        ProcessGroupIter {
107            inner: self.inner.processes.values(),
108        }
109    }
110
111    /// Inserts a process into the process group.
112    ///
113    /// The caller needs to ensure that the process didn't previously belong to the process group,
114    /// but now does.
115    pub(in crate::process) fn insert_process(&mut self, process: &Arc<Process>) {
116        let old_process = self
117            .inner
118            .processes
119            .insert(process.pid(), Arc::downgrade(process));
120        debug_assert!(old_process.is_none());
121    }
122
123    /// Removes a process from the process group.
124    ///
125    /// The caller needs to ensure that the process previously belonged to the process group, but
126    /// now doesn't.
127    pub(in crate::process) fn remove_process(&mut self, pid: &Pid) {
128        let process = self.inner.processes.remove(pid);
129        debug_assert!(process.is_some());
130    }
131
132    /// Returns whether the process group is empty.
133    pub(in crate::process) fn is_empty(&self) -> bool {
134        self.inner.processes.is_empty()
135    }
136}
137
138/// An iterator over the processes of the process group.
139pub struct ProcessGroupIter<'a> {
140    inner: Values<'a, Pid, Weak<Process>>,
141}
142
143impl<'a> Iterator for ProcessGroupIter<'a> {
144    type Item = Arc<Process>;
145
146    fn next(&mut self) -> Option<Self::Item> {
147        self.inner.by_ref().find_map(Weak::upgrade)
148    }
149}
