3use alloc::collections::btree_map::Values;
4
5use super::{Pgid, Pid, Process, Session};
6use crate::{prelude::*, process::signal::signals::Signal};

8/// A process group.
9///
10/// A process group represents a set of processes,
11/// which has a unique identifier PGID (i.e., [`Pgid`]).
12pub struct ProcessGroup {
13    pgid: Pgid,
14    session: Weak<Session>,
15    inner: Mutex<Inner>,
16}
17
18struct Inner {
19    processes: BTreeMap<Pid, Arc<Process>>,
20}
21
22impl ProcessGroup {
23    /// Creates a new process group with one process.
24    ///
25    /// The PGID is the same as the process ID, which means that the process will become the leader
26    /// process of the new process group.
27    ///
28    /// The caller needs to ensure that the process does not belong to other process group.
29    pub(super) fn new(process: Arc<Process>, session: Weak<Session>) -> Arc<Self> {
30        let pid = process.pid();
31
32        let inner = {
33            let mut processes = BTreeMap::new();
34            processes.insert(pid, process);
35            Inner { processes }
36        };
37
38        Arc::new(ProcessGroup {
39            pgid: pid,
40            session,
41            inner: Mutex::new(inner),
42        })
43    }
44
45    /// Returns the process group identifier.
46    pub fn pgid(&self) -> Pgid {
47        self.pgid
48    }
49
50    /// Returns the session to which the process group belongs.
51    pub fn session(&self) -> Option<Arc<Session>> {
52        self.session.upgrade()
53    }
54
55    /// Acquires a lock on the process group.
56    pub fn lock(&self) -> ProcessGroupGuard<'_> {
57        ProcessGroupGuard {
58            inner: self.inner.lock(),
59        }
60    }
61
62    /// Broadcasts the signal to all processes in the process group.
63    ///
64    /// This method should only be used to broadcast fault signals and kernel signals.
65    //
66    // TODO: Do some checks to forbid user signals.
67    pub fn broadcast_signal(&self, signal: impl Signal + Clone + 'static) {
68        for process in self.inner.lock().processes.values() {
69            process.enqueue_signal(Box::new(signal.clone()));
70        }
71    }
72}
73
74/// A scoped lock guard for a process group.
75///
76/// It provides some public methods to prevent the exposure of the inner type.
77#[clippy::has_significant_drop]
78#[must_use]
79pub struct ProcessGroupGuard<'a> {
80    inner: MutexGuard<'a, Inner>,
81}
82
83impl ProcessGroupGuard<'_> {
84    /// Returns an iterator over the processes in the process group.
85    pub fn iter(&self) -> ProcessGroupIter<'_> {
86        ProcessGroupIter {
87            inner: self.inner.processes.values(),
88        }
89    }
90
91    /// Inserts a process into the process group.
92    ///
93    /// The caller needs to ensure that the process didn't previously belong to the process group,
94    /// but now does.
95    pub(in crate::process) fn insert_process(&mut self, process: Arc<Process>) {
96        let old_process = self.inner.processes.insert(process.pid(), process);
97        debug_assert!(old_process.is_none());
98    }
99
100    /// Removes a process from the process group.
101    ///
102    /// The caller needs to ensure that the process previously belonged to the process group, but
103    /// now doesn't.
104    pub(in crate::process) fn remove_process(&mut self, pid: &Pid) {
105        let process = self.inner.processes.remove(pid);
106        debug_assert!(process.is_some());
107    }
108
109    /// Returns whether the process group is empty.
110    pub(in crate::process) fn is_empty(&self) -> bool {
111        self.inner.processes.is_empty()
112    }
113}
114
115/// An iterator over the processes of the process group.
116pub struct ProcessGroupIter<'a> {
117    inner: Values<'a, Pid, Arc<Process>>,
118}
119
120impl<'a> Iterator for ProcessGroupIter<'a> {
121    type Item = &'a Arc<Process>;
122
123    fn next(&mut self) -> Option<Self::Item> {
124        self.inner.next()
125    }
126}
