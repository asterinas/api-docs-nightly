161/// Representing a parent process by holding a weak reference to it and its PID.
162///
163/// This type caches the value of the PID so that it can be retrieved cheaply.
164///
165/// The benefit of using `ParentProcess` over `(Mutex<Weak<Process>>, AtomicPid,)` is to
166/// enforce the invariant that the cached PID and the weak reference are always kept in sync.
167pub struct ParentProcess {
168    process: Mutex<Weak<Process>>,
169    pid: AtomicPid,
170}

172impl ParentProcess {
173    pub fn new(process: Weak<Process>) -> Self {
174        let pid = match process.upgrade() {
175            Some(process) => process.pid(),
176            None => 0,
177        };
178
179        Self {
180            process: Mutex::new(process),
181            pid: AtomicPid::new(pid),
182        }
183    }

187    }

190        ParentProcessGuard {
191            guard: self.process.lock(),

267    /// Runs the process.
268    pub(super) fn run(&self) {
269        let tasks = self.tasks.lock();
270        // when run the process, the process should has only one thread
271        debug_assert!(tasks.as_slice().len() == 1);
272        debug_assert!(!self.status().is_zombie());
273        let task = tasks.main().clone();
274        // should not hold the lock when run thread
275        drop(tasks);
276        let thread = task.as_thread().unwrap();
277        thread.run();
278    }

372    /// Moves the process to the new session.
373    ///
374    /// This method will create a new process group in a new session, move the process to the new
375    /// session, and return the session ID (which is equal to the process ID and the process group
376    /// ID).
377    ///
378    /// # Errors
379    ///
380    /// This method will return `EPERM` if an existing process group has the same identifier as the
381    /// process ID. This means that the process is or was a process group leader and that the
382    /// process group is still alive.
383    pub fn to_new_session(self: &Arc<Self>) -> Result<Sid> {
384        // Lock order: session table -> group table -> group of process
385        // -> group inner -> session inner
386        let mut session_table_mut = process_table::session_table_mut();
387        let mut group_table_mut = process_table::group_table_mut();
388
389        if group_table_mut.contains_key(&self.pid) {
390            return_errno_with_message!(
391                Errno::EPERM,
392                "a process group leader cannot be moved to a new session"
393            );
394        }
395
396        let mut process_group_mut = self.process_group.lock();
397
398        self.clear_old_group_and_session(
399            &mut process_group_mut,
400            &mut session_table_mut,
401            &mut group_table_mut,
402        );
403
404        Ok(self.set_new_session(
405            &mut process_group_mut,
406            &mut session_table_mut,
407            &mut group_table_mut,
408        ))
409    }
410
411    pub(super) fn clear_old_group_and_session(
412        &self,
413        process_group_mut: &mut MutexGuard<Weak<ProcessGroup>>,
414        session_table_mut: &mut MutexGuard<BTreeMap<Sid, Arc<Session>>>,
415        group_table_mut: &mut MutexGuard<BTreeMap<Pgid, Arc<ProcessGroup>>>,
416    ) {
417        let process_group = process_group_mut.upgrade().unwrap();
418        let mut process_group_inner = process_group.lock();
419        let session = process_group.session().unwrap();
420        let mut session_inner = session.lock();
421
422        // Remove the process from the process group.
423        process_group_inner.remove_process(&self.pid);
424        if process_group_inner.is_empty() {
425            group_table_mut.remove(&process_group.pgid());
426
427            // Remove the process group from the session.
428            session_inner.remove_process_group(&process_group.pgid());
429            if session_inner.is_empty() {
430                session_table_mut.remove(&session.sid());
431            }
432        }
433
434        **process_group_mut = Weak::new();
435    }

437    fn set_new_session(
438        self: &Arc<Self>,
439        process_group_mut: &mut MutexGuard<Weak<ProcessGroup>>,
440        session_table_mut: &mut MutexGuard<BTreeMap<Sid, Arc<Session>>>,
441        group_table_mut: &mut MutexGuard<BTreeMap<Pgid, Arc<ProcessGroup>>>,
442    ) -> Sid {
443        let (session, process_group) = Session::new_pair(self.clone());
444        let sid = session.sid();
445
446        **process_group_mut = Arc::downgrade(&process_group);
447
448        // Insert the new session and the new process group to the global table.
449        session_table_mut.insert(session.sid(), session);
450        group_table_mut.insert(process_group.pgid(), process_group);
451
452        sid
453    }

455    /// Moves the process itself or its child process to another process group.
456    ///
457    /// The process to be moved is specified with the process ID `pid`; `self` is used only for
458    /// permission checking purposes (see the Errors section below), which is typically
459    /// `current!()` when implementing system calls.
460    ///
461    /// If `pgid` is equal to the process ID, a new process group with the given PGID will be
462    /// created (if it does not already exist). Then, the process will be moved to the process
463    /// group with the given PGID, if the process group exists and belongs to the same session as
464    /// the given process.
465    ///
466    /// # Errors
467    ///
468    /// This method will return `ESRCH` in following cases:
469    ///  * The process specified by `pid` does not exist;
470    ///  * The process specified by `pid` is neither `self` or a child process of `self`.
471    ///
472    /// This method will return `EPERM` in following cases:
473    ///  * The process is not in the same session as `self`;
474    ///  * The process is a session leader, but the given PGID is not the process's PID/PGID;
475    ///  * The process group already exists, but the group does not belong to the same session;
476    ///  * The process group does not exist, but `pgid` is not equal to the process ID.
477    pub fn move_process_to_group(&self, pid: Pid, pgid: Pgid) -> Result<()> {
478        // Lock order: group table -> process table -> group of process
479        // -> group inner -> session inner
480        let group_table_mut = process_table::group_table_mut();
481        let process_table_mut = process_table::process_table_mut();
482
483        let process = process_table_mut.get(pid).ok_or(Error::with_message(
484            Errno::ESRCH,
485            "the process to set the PGID does not exist",
486        ))?;
487
488        let current_session = if self.pid == process.pid() {
489            // There is no need to check if the session is the same in this case.
490            None
491        } else if self.pid == process.parent().pid() {
492            // FIXME: If the child process has called `execve`, we should fail with `EACCESS`.
493
494            // Immediately release the `self.process_group` lock to avoid deadlocks. Race
495            // conditions don't matter because this is used for comparison purposes only.
496            Some(
497                self.process_group
498                    .lock()
499                    .upgrade()
500                    .unwrap()
501                    .session()
502                    .unwrap(),
503            )
504        } else {
505            return_errno_with_message!(
506                Errno::ESRCH,
507                "the process to set the PGID is neither the current process nor its child process"
508            );
509        };
510
511        if let Some(new_process_group) = group_table_mut.get(&pgid).cloned() {
512            process.to_existing_group(current_session, group_table_mut, new_process_group)
513        } else if pgid == process.pid() {
514            process.to_new_group(current_session, group_table_mut)
515        } else {
516            return_errno_with_message!(Errno::EPERM, "the new process group does not exist");
517        }
518    }

520    /// Moves the process to an existing group.
521    fn to_existing_group(
522        self: &Arc<Self>,
523        current_session: Option<Arc<Session>>,
524        mut group_table_mut: MutexGuard<BTreeMap<Pgid, Arc<ProcessGroup>>>,
525        new_process_group: Arc<ProcessGroup>,
526    ) -> Result<()> {
527        let mut process_group_mut = self.process_group.lock();
528        let process_group = process_group_mut.upgrade().unwrap();
529
530        let session = process_group.session().unwrap();
531        if session.sid() == self.pid {
532            return_errno_with_message!(
533                Errno::EPERM,
534                "a session leader cannot be moved to a new process group"
535            );
536        }
537        if !Arc::ptr_eq(&session, &new_process_group.session().unwrap()) {
538            return_errno_with_message!(
539                Errno::EPERM,
540                "the new process group does not belong to the same session"
541            );
542        }
543        if current_session.is_some_and(|current| !Arc::ptr_eq(&current, &session)) {
544            return_errno_with_message!(Errno::EPERM, "the process belongs to a different session");
545        }
546
547        // Lock order: group with a smaller PGID -> group with a larger PGID
548        let (mut process_group_inner, mut new_group_inner) =
549            match process_group.pgid().cmp(&new_process_group.pgid()) {
550                core::cmp::Ordering::Less => {
551                    let process_group_inner = process_group.lock();
552                    let new_group_inner = new_process_group.lock();
553                    (process_group_inner, new_group_inner)
554                }
555                core::cmp::Ordering::Greater => {
556                    let new_group_inner = new_process_group.lock();
557                    let process_group_inner = process_group.lock();
558                    (process_group_inner, new_group_inner)
559                }
560                core::cmp::Ordering::Equal => return Ok(()),
561            };
562        let mut session_inner = session.lock();
563
564        // Remove the process from the old process group
565        process_group_inner.remove_process(&self.pid);
566        if process_group_inner.is_empty() {
567            group_table_mut.remove(&process_group.pgid());
568            session_inner.remove_process_group(&process_group.pgid());
569        }
570
571        // Insert the process to the new process group
572        new_group_inner.insert_process(self.clone());
573        *process_group_mut = Arc::downgrade(&new_process_group);
574
575        Ok(())
576    }

578    /// Creates a new process group and moves the process to the group.
579    fn to_new_group(
580        self: &Arc<Self>,
581        current_session: Option<Arc<Session>>,
582        mut group_table_mut: MutexGuard<BTreeMap<Pgid, Arc<ProcessGroup>>>,
583    ) -> Result<()> {
584        let mut process_group_mut = self.process_group.lock();
585
586        let process_group = process_group_mut.upgrade().unwrap();
587        let session = process_group.session().unwrap();
588
589        if current_session.is_some_and(|current| !Arc::ptr_eq(&current, &session)) {
590            return_errno_with_message!(Errno::EPERM, "the process belongs to a different session");
591        }
592        if process_group.pgid() == self.pid {
593            // We'll hit this if the process is a session leader. There is no need to check below.
594            return Ok(());
595        }
596
597        let mut process_group_inner = process_group.lock();
598        let mut session_inner = session.lock();
599
600        // Remove the process from the old process group
601        process_group_inner.remove_process(&self.pid);
602        if process_group_inner.is_empty() {
603            group_table_mut.remove(&process_group.pgid());
604            session_inner.remove_process_group(&process_group.pgid());
605        }
606
607        // Create a new process group and insert the process to it
608        let new_process_group = ProcessGroup::new(self.clone(), Arc::downgrade(&session));
609        *process_group_mut = Arc::downgrade(&new_process_group);
610        group_table_mut.insert(new_process_group.pgid(), new_process_group.clone());
611        session_inner.insert_process_group(new_process_group);
612
613        Ok(())
614    }

632    /// Enqueues a process-directed signal.
633    ///
634    /// This method does not perform permission checks on user signals.
635    /// Therefore, unless the caller can ensure that there are no permission issues,
636    /// this method should be used to enqueue kernel signals or fault signals.
637    pub fn enqueue_signal(&self, signal: Box<dyn Signal>) {
638        if self.status.is_zombie() {
639            return;
640        }
641
642        self.sig_queues.enqueue(signal);
643
644        for task in self.tasks.lock().as_slice() {
645            let posix_thread = task.as_posix_thread().unwrap();
646            // FIXME: This behavior differs a bit from Linux.
647            // Linux wakes up a single thread that neither blocks the signal
648            // nor already has the same pending signal;
649            // for simplicity we wake up all threads.
650            // Reference: <https://elixir.bootlin.com/linux/v6.17/source/kernel/signal.c#L969>.
651            posix_thread.wake_signalled_waker();
652        }
653    }

688    /// Stops the process.
689    //
690    // FIXME: `ptrace` is another reason that can cause a process to stop.
691    // Consider extending the method signature to support `ptrace` if necessary.
692    pub fn stop(&self, sig_num: SigNum) {
693        if self.status.stop_status().stop(sig_num) {
694            self.wake_up_parent();
695        }
696    }

698    /// Resumes the stopped process.
699    pub fn resume(&self) {
700        if self.status.stop_status().resume() {
701            self.wake_up_parent();
702
703            // Note that the resume function is called by the thread which deals with SIGCONT,
704            // since SIGCONT is handled by any thread in this process, we need to wake
705            // up other stopped threads in the same process.
706            for task in self.tasks.lock().as_slice() {
707                let posix_thread = task.as_posix_thread().unwrap();
708                posix_thread.wake_signalled_waker();
709            }
710        }
711    }

724        let parent_guard = self.parent.lock();
725        let parent = parent_guard.process().upgrade().unwrap();
726        parent.children_wait_queue.wake_all();

731    /// Sets the child subreaper attribute of the current process.
732    ///
733    /// # Panics
734    ///
735    /// This method may panic if the process is a zombie process.
736    pub fn set_child_subreaper(&self) {
737        self.is_child_subreaper.store(true, Ordering::Relaxed);
738        let has_child_subreaper = self.has_child_subreaper.fetch_or(true, Ordering::Release);
739        if !has_child_subreaper {
740            self.propagate_has_child_subreaper();
741        }
742    }
743
744    /// Unsets the child subreaper attribute of the current process.
745    pub fn unset_child_subreaper(&self) {
746        self.is_child_subreaper.store(false, Ordering::Relaxed);
747    }
748
749    /// Returns whether this process is a child subreaper.
750    pub fn is_child_subreaper(&self) -> bool {
751        self.is_child_subreaper.load(Ordering::Relaxed)
752    }
753
754    /// Sets all descendants of the current process as having child subreaper.
755    fn propagate_has_child_subreaper(&self) {
756        let mut process_queue = VecDeque::new();
757        let children = self.children().lock();
758        for child_process in children.as_ref().unwrap().values() {
759            let has_child_subreaper = child_process
760                .has_child_subreaper
761                .fetch_or(true, Ordering::Release);
762            if !has_child_subreaper {
763                process_queue.push_back(child_process.clone());
764            }
765        }
766
767        while let Some(process) = process_queue.pop_front() {
768            let children = process.children().lock();
769            let Some(children_ref) = children.as_ref() else {
770                // The process is exiting group at the same time.
771                continue;
772            };
773            for child_process in children_ref.values() {
774                let has_child_subreaper = child_process
775                    .has_child_subreaper
776                    .fetch_or(true, Ordering::Release);
777                if !has_child_subreaper {
778                    process_queue.push_back(child_process.clone());
779                }
780            }
781        }
782    }

811/// Enqueues a process-directed kernel signal asynchronously.
812///
813/// This is the asynchronous version of [`Process::enqueue_signal`]. By asynchronous, this method
814/// submits a work item and returns, so this method doesn't sleep and can be used in atomic mode.
815pub fn enqueue_signal_async(process: Weak<Process>, signum: SigNum) {
816    use super::signal::signals::kernel::KernelSignal;
817    use crate::thread::work_queue;
818
819    work_queue::submit_work_func(
820        move || {
821            if let Some(process) = process.upgrade() {
822                process.enqueue_signal(Box::new(KernelSignal::new(signum)));
823            }
824        },
825        work_queue::WorkPriority::High,
826    );
827}
