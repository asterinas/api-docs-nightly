3use super::{Pgid, Pid};
4use crate::{fs::file_table::get_file_fast, prelude::*, process::PidFile};
5
6#[derive(Debug, Clone)]
7pub enum ProcessFilter {
8    Any,
9    WithPid(Pid),
10    WithPgid(Pgid),
11    WithPidfd(Arc<PidFile>),
12}
13
14impl ProcessFilter {
15    // For `waitpid`.
16    pub fn from_which_and_id(which: u64, id: u32, ctx: &Context) -> Result<Self> {
17        // Reference:
18        // <https://elixir.bootlin.com/linux/v6.14.4/source/include/uapi/linux/wait.h#L16-L20>
19        const P_ALL: u64 = 0;
20        const P_PID: u64 = 1;
21        const P_PGID: u64 = 2;
22        const P_PIDFD: u64 = 3;
23
24        match which {
25            P_ALL => Ok(ProcessFilter::Any),
26            P_PID => Ok(ProcessFilter::WithPid(id)),
27            P_PGID => Ok(ProcessFilter::WithPgid(id)),
28            P_PIDFD => {
29                let fd = {
30                    let fd = id.cast_signed();
31                    if fd < 0 {
32                        return_errno_with_message!(Errno::EINVAL, "the pidfd is invalid");
33                    }
34                    fd
35                };
36                let file = {
37                    let mut file_table = ctx.thread_local.borrow_file_table_mut();
38                    get_file_fast!(&mut file_table, fd).into_owned()
39                };
40                let pid_file = Arc::downcast(file).map_err(|_| {
41                    Error::with_message(Errno::EINVAL, "the file is not a PID file")
42                })?;
43                Ok(ProcessFilter::WithPidfd(pid_file))
44            }
45            _ => return_errno_with_message!(Errno::EINVAL, "the process filter is invalid"),
46        }
47    }
48
49    // For `wait4` and `kill`.
50    pub fn from_id(wait_pid: i32) -> Result<Self> {
51        // Reference:
52        // <https://man7.org/linux/man-pages/man2/waitpid.2.html>
53        // <https://man7.org/linux/man-pages/man2/kill.2.html>
54        if wait_pid == i32::MIN {
55            // Note that performing `-wait_pid` will overflow, so we return an error directly.
56            return_errno_with_message!(Errno::ESRCH, "the target group does not exist");
57        } else if wait_pid < -1 {
58            // "wait for any child process whose process group ID is equal to the absolute value of
59            // `pid`"
60            Ok(ProcessFilter::WithPgid((-wait_pid).cast_unsigned()))
61        } else if wait_pid == -1 {
62            // "wait for any child process"
63            Ok(ProcessFilter::Any)
64        } else if wait_pid == 0 {
65            // "wait for any child process whose process group ID is equal to that of the calling
66            // process at the time of the call to `waitpid()`"
67            let pgid = current!().pgid();
68            Ok(ProcessFilter::WithPgid(pgid))
69        } else {
70            // "wait for the child whose process ID is equal to the value of `pid`"
71            Ok(ProcessFilter::WithPid(wait_pid.cast_unsigned()))
72        }
73    }
74}
