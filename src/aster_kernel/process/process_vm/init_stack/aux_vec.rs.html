5/// Auxiliary Vector.
6///
7/// # What is Auxiliary Vector?
8///
9/// Here is a concise description of Auxiliary Vector from GNU's manual:
10///
11///  > When a program is executed, it receives information from the operating system
12///  > about the environment in which it is operating. The form of this information
13///  > is a table of key-value pairs, where the keys are from the set of ‘AT_’
14///  > values in elf.h.
15#[expect(non_camel_case_types)]
16#[expect(dead_code)]
17#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, PartialOrd, Ord)]
18#[repr(u8)]
19pub enum AuxKey {
20    AT_IGNORE = 1,    /* entry should be ignored */
21    AT_EXECFD = 2,    /* file descriptor of program */
22    AT_PHDR = 3,      /* program headers for program */
23    AT_PHENT = 4,     /* size of program header entry */
24    AT_PHNUM = 5,     /* number of program headers */
25    AT_PAGESZ = 6,    /* system page size */
26    AT_BASE = 7,      /* base address of interpreter */
27    AT_FLAGS = 8,     /* flags */
28    AT_ENTRY = 9,     /* entry point of program */
29    AT_NOTELF = 10,   /* program is not ELF */
30    AT_UID = 11,      /* real uid */
31    AT_EUID = 12,     /* effective uid */
32    AT_GID = 13,      /* real gid */
33    AT_EGID = 14,     /* effective gid */
34    AT_PLATFORM = 15, /* string identifying CPU for optimizations */
35    AT_HWCAP = 16,    /* arch dependent hints at CPU capabilities */
36    AT_CLKTCK = 17,   /* frequency at which times() increments */
37
38    /* 18...22 not used */
39    AT_SECURE = 23, /* secure mode boolean */
40    AT_BASE_PLATFORM = 24, /* string identifying real platform, may
41                     * differ from AT_PLATFORM. */
42    AT_RANDOM = 25, /* address of 16 random bytes */
43    AT_HWCAP2 = 26, /* extension of AT_HWCAP */
44
45    /* 28...30 not used */
46    AT_EXECFN = 31, /* filename of program */
47    AT_SYSINFO = 32,
48    AT_SYSINFO_EHDR = 33, /* the start address of the page containing the VDSO */
49}
50
51impl AuxKey {
52    /// A special auxiliary key that denotes the end of the auxiliary vector.
53    pub(super) const AT_NULL: u8 = 0;
54}
55
56#[derive(Clone, Default, Debug)]
57pub struct AuxVec {
58    table: BTreeMap<AuxKey, u64>,
59}
60
61impl AuxVec {
62    pub const fn new() -> AuxVec {
63        AuxVec {
64            table: BTreeMap::new(),
65        }
66    }
67
68    pub fn set(&mut self, key: AuxKey, val: u64) {
69        self.table
70            .entry(key)
71            .and_modify(|val_mut| *val_mut = val)
72            .or_insert(val);
73    }
74
75    pub(super) fn table(&self) -> &BTreeMap<AuxKey, u64> {
76        &self.table
77    }
78}
