3//! The init stack for the process.
4//!
5//! The init stack is used to store `argv`, `envp`, and auxiliary vectors.
6//! We can read `argv` and `envp` of a process from the init stack.
7//! Usually, the lowest address of the init stack is
8//! the highest address of the user stack of the main thread.
9//!
10//! However, the init stack will be mapped to user space
11//! and the user process can write the content of the init stack,
12//! so the content reading from the init stack may not be the same as the initial one.
13//!
14

50/*
51 * Illustration of the virtual memory space containing the processes' init stack:
52 *
53 *  (high address)
54 *  +---------------------+ <------+ Highest address
55 *  |                     |          Random stack paddings
56 *  +---------------------+ <------+ The base of stack (stack grows down)
57 *  |                     |
58 *  | Null-terminated     |
59 *  | strings referenced  |
60 *  | by variables below  |
61 *  |                     |
62 *  +---------------------+
63 *  | AT_NULL             |
64 *  +---------------------+
65 *  | AT_NULL             |
66 *  +---------------------+
67 *  | ...                 |
68 *  +---------------------+
69 *  | aux_val[0]          |
70 *  +---------------------+
71 *  | aux_key[0]          | <------+ Auxiliary table
72 *  +---------------------+
73 *  | NULL                |
74 *  +---------------------+
75 *  | ...                 |
76 *  +---------------------+
77 *  | char* envp[0]       | <------+ Environment variables
78 *  +---------------------+
79 *  | NULL                |
80 *  +---------------------+
81 *  | char* argv[argc-1]  |
82 *  +---------------------+
83 *  | ...                 |
84 *  +---------------------+
85 *  | char* argv[0]       |
86 *  +---------------------+
87 *  | long argc           | <------+ Program arguments
88 *  +---------------------+
89 *  |                     |
90 *  |                     |
91 *  +---------------------+
92 *  |                     |
93 *  +---------------------+ <------+ User stack default rlimit
94 *  (low address)
95 */
96
97/// The initial portion of the main stack of a process.
98pub struct InitStack {
99    /// The top address of the init stack.
100    ///
101    /// The stack grows down from this address.
102    initial_top: Vaddr,
103    /// The maximum size of the stack.
104    max_size: usize,
105    /// The current stack pointer.
106    ///
107    /// Before initialization, `pos` points to `initial_top`.
108    /// After initialization, `pos` points to the top of the process stack.
109    pos: AtomicUsize,
110    argv_range: SpinLock<Range<Vaddr>>,
111    envp_range: SpinLock<Range<Vaddr>>,
112}
113
114impl Clone for InitStack {
115    fn clone(&self) -> Self {
116        Self {
117            initial_top: self.initial_top,
118            max_size: self.max_size,
119            pos: AtomicUsize::new(self.pos.load(Ordering::Relaxed)),
120            argv_range: SpinLock::new(self.argv_range.lock().clone()),
121            envp_range: SpinLock::new(self.envp_range.lock().clone()),
122        }
123    }
124}
125
126impl InitStack {
127    pub fn new() -> Self {
128        let nr_pages_padding = {
129            // We do not want the stack top too close to `VMAR_CAP_ADDR`.
130            // So we add this fixed padding. Any small value greater than zero will do.
131            const NR_FIXED_PADDING_PAGES: usize = 7;
132
133            // Some random padding pages are added as a simple measure to
134            // make the stack values of a buggy user program harder
135            // to be exploited by attackers.
136            let mut nr_random_padding_pages: u8 = 0;
137            getrandom(nr_random_padding_pages.as_mut_bytes());
138
139            nr_random_padding_pages as usize + NR_FIXED_PADDING_PAGES
140        };
141        let initial_top = VMAR_CAP_ADDR - PAGE_SIZE * nr_pages_padding;
142        let max_size = INIT_STACK_SIZE;
143
144        Self {
145            initial_top,
146            max_size,
147            pos: AtomicUsize::new(initial_top),
148            argv_range: SpinLock::new(0..0),
149            envp_range: SpinLock::new(0..0),
150        }
151    }
152
153    /// Returns the top address of the user stack.
154    ///
155    /// This method should only be called after the stack is initialized.
156    pub fn user_stack_top(&self) -> Vaddr {
157        debug_assert!(self.is_initialized());
158
159        self.pos()
160    }
161
162    /// Maps the VMO of the init stack and constructs a writer to initialize its content.
163    pub(super) fn map_and_write(
164        &self,
165        vmar: &Vmar,
166        argv: Vec<CString>,
167        envp: Vec<CString>,
168        auxvec: AuxVec,
169    ) -> Result<()> {
170        self.set_uninitialized();
171
172        let vmo = {
173            let vmo_options = VmoOptions::new(self.max_size);
174            vmo_options.alloc()?
175        };
176        let vmar_map_options = {
177            let perms = VmPerms::READ | VmPerms::WRITE;
178            let map_addr = self.initial_top - self.max_size;
179            debug_assert!(map_addr.is_multiple_of(PAGE_SIZE));
180            vmar.new_map(self.max_size, perms)?
181                .offset(map_addr)
182                .vmo(vmo.clone())
183        };
184        vmar_map_options.build()?;
185
186        let writer = InitStackWriter {
187            pos: &self.pos,
188            vmo,
189            argv,
190            envp,
191            auxvec,
192            map_addr: self.initial_top - self.max_size,
193        };
194        let (argv_range, envp_range) = writer.write()?;
195
196        *self.argv_range.lock() = argv_range;
197        *self.envp_range.lock() = envp_range;
198
199        Ok(())
200    }
201
202    /// Constructs a reader to parse the content of an `InitStack`.
203    ///
204    /// This method should only be called after the stack is initialized.
205    pub(super) fn reader<'a>(&self, vmar: &'a Vmar) -> InitStackReader<'a> {
206        debug_assert!(self.is_initialized());
207
208        InitStackReader {
209            vmar,
210            argv_range: self.argv_range.lock().clone(),
211            envp_range: self.envp_range.lock().clone(),
212        }
213    }
214
215    fn is_initialized(&self) -> bool {
216        self.pos() != self.initial_top
217    }
218
219    fn set_uninitialized(&self) {
220        self.pos.store(self.initial_top, Ordering::Relaxed);
221    }
222
223    fn pos(&self) -> Vaddr {
224        self.pos.load(Ordering::Relaxed)
225    }
226}

228/// A writer to initialize the content of an `InitStack`.
229struct InitStackWriter<'a> {
230    pos: &'a AtomicUsize,
231    vmo: Arc<Vmo>,
232    argv: Vec<CString>,
233    envp: Vec<CString>,
234    auxvec: AuxVec,
235    /// The mapping address of the `InitStack`.
236    map_addr: usize,
237}
238
239impl InitStackWriter<'_> {
240    /// Writes the content to the init stack.
241    ///
242    /// Returns the range of argv and envp in the init stack.
243    fn write(mut self) -> Result<(Range<Vaddr>, Range<Vaddr>)> {
244        // FIXME: Some OSes may put the first page of the executable file here
245        // for interpreting ELF headers.
246
247        // Write envp strings.
248        let envp_end = self.pos();
249        let envp_pointers = self.write_envp_strings()?;
250        let envp_start = self.pos();
251
252        // Write argv strings.
253        let argv_end = self.pos();
254        let argv_pointers = self.write_argv_strings()?;
255        let argv_start = self.pos();
256
257        // Generate random values for the auxiliary vector.
258        let random_value_pointer = {
259            let random_value = generate_random_for_aux_vec();
260            self.write_bytes(&random_value)?
261        };
262        self.auxvec.set(AuxKey::AT_RANDOM, random_value_pointer);
263
264        self.adjust_stack_alignment(&envp_pointers, &argv_pointers)?;
265        self.write_aux_vec()?;
266        self.write_envp_pointers(envp_pointers)?;
267        self.write_argv_pointers(argv_pointers)?;
268
269        // Write argc.
270        let argc = self.argv.len();
271        self.write_u64(argc as u64)?;
272
273        // Ensure the stack top is 16-byte aligned.
274        debug_assert_eq!(self.pos() & !0xf, self.pos());
275
276        Ok((argv_start..argv_end, envp_start..envp_end))
277    }
278
279    fn write_envp_strings(&self) -> Result<Vec<u64>> {
280        let mut envp_pointers = Vec::with_capacity(self.envp.len());
281        for envp in self.envp.iter() {
282            let pointer = self.write_cstring(envp)?;
283            envp_pointers.push(pointer);
284        }
285        Ok(envp_pointers)
286    }
287
288    fn write_argv_strings(&self) -> Result<Vec<u64>> {
289        let mut argv_pointers = Vec::with_capacity(self.argv.len());
290        for argv in self.argv.iter().rev() {
291            let pointer = self.write_cstring(argv)?;
292            debug!("argv address = 0x{:x}", pointer);
293            argv_pointers.push(pointer);
294        }
295        argv_pointers.reverse();
296        Ok(argv_pointers)
297    }
298
299    /// Ensures that the top address of the user stack is 16-byte aligned.
300    ///
301    /// The 16-byte alignment is required by x86-64 System V ABI.
302    /// To meet that requirement, this method may write some extra 8-byte `u64`s.
303    fn adjust_stack_alignment(&self, envp_pointers: &[u64], argv_pointers: &[u64]) -> Result<()> {
304        // Ensure 8-byte alignment.
305        self.write_u64(0)?;
306        let auxvec_size = (self.auxvec.table().len() + 1) * (size_of::<u64>() * 2);
307        let envp_pointers_size = (envp_pointers.len() + 1) * size_of::<u64>();
308        let argv_pointers_size = (argv_pointers.len() + 1) * size_of::<u64>();
309        let argc_size = size_of::<u64>();
310        let to_write_size = auxvec_size + envp_pointers_size + argv_pointers_size + argc_size;
311        if !(self.pos() - to_write_size).is_multiple_of(16) {
312            self.write_u64(0)?;
313        }
314        Ok(())
315    }
316
317    fn write_aux_vec(&self) -> Result<()> {
318        // Write a NULL auxiliary entry.
319        self.write_u64(0)?;
320        self.write_u64(AuxKey::AT_NULL as u64)?;
321        // Write the auxiliary vector.
322        let aux_vec: Vec<_> = self
323            .auxvec
324            .table()
325            .iter()
326            .map(|(aux_key, aux_value)| (*aux_key, *aux_value))
327            .collect();
328        for (aux_key, aux_value) in aux_vec.iter() {
329            self.write_u64(*aux_value)?;
330            self.write_u64(*aux_key as u64)?;
331        }
332        Ok(())
333    }
334
335    fn write_envp_pointers(&self, mut envp_pointers: Vec<u64>) -> Result<()> {
336        // Write a NULL pointer.
337        self.write_u64(0)?;
338        // Write envp pointers.
339        envp_pointers.reverse();
340        for envp_pointer in envp_pointers {
341            self.write_u64(envp_pointer)?;
342        }
343        Ok(())
344    }
345
346    fn write_argv_pointers(&self, mut argv_pointers: Vec<u64>) -> Result<()> {
347        // Write a NULL pointer.
348        self.write_u64(0)?;
349        // Write argv pointers.
350        argv_pointers.reverse();
351        for argv_pointer in argv_pointers {
352            self.write_u64(argv_pointer)?;
353        }
354        Ok(())
355    }
356
357    /// Writes a `u64` to the stack.
358    /// Returns the writing address.
359    fn write_u64(&self, val: u64) -> Result<u64> {
360        let new_pos = self.reserve_pos(size_of::<u64>(), align_of::<u64>())?;
361        self.vmo.write_val(new_pos - self.map_addr, &val)?;
362        Ok(new_pos as u64)
363    }
364
365    /// Writes a `CString` including the nul byte to the stack.
366    /// Returns the writing address.
367    fn write_cstring(&self, val: &CString) -> Result<u64> {
368        let bytes = val.as_bytes_with_nul();
369        self.write_bytes(bytes)
370    }
371
372    /// Writes bytes to the stack.
373    /// Returns the writing address.
374    fn write_bytes(&self, bytes: &[u8]) -> Result<u64> {
375        let new_pos = self.reserve_pos(bytes.len(), align_of::<u8>())?;
376        self.vmo.write_bytes(new_pos - self.map_addr, bytes)?;
377        Ok(new_pos as u64)
378    }
379
380    fn reserve_pos(&self, size: usize, align: usize) -> Result<Vaddr> {
381        if let Some(pos) = self.pos().checked_sub(size)
382            && let new_pos = pos & !(align - 1)
383            && new_pos >= self.map_addr
384        {
385            self.pos.store(new_pos, Ordering::Relaxed);
386            return Ok(new_pos);
387        }
388        return_errno_with_message!(Errno::E2BIG, "the init stack overflows");
389    }
390
391    fn pos(&self) -> Vaddr {
392        self.pos.load(Ordering::Relaxed)
393    }
394}
395
396fn generate_random_for_aux_vec() -> [u8; 16] {
397    let mut rand_val = [0; 16];
398    getrandom(&mut rand_val);
399    rand_val
400}

402/// A reader to parse the content of an `InitStack`.
403pub struct InitStackReader<'a> {
404    vmar: &'a Vmar,
405    argv_range: Range<Vaddr>,
406    envp_range: Range<Vaddr>,
407}
408
409impl InitStackReader<'_> {
410    /// Reads argv at the `offset` from the process init stack.
411    pub fn argv(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
412        if offset >= self.argv_range.end - self.argv_range.start {
413            return Ok(0);
414        }
415
416        let read_at = self.argv_range.start + offset;
417        writer.limit(self.argv_range.end - read_at);
418        let bytes_read = self.vmar.read_remote(read_at, writer)?;
419
420        Ok(bytes_read)
421    }
422
423    /// Reads envp at the `offset` from the process init stack.
424    pub fn envp(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
425        if offset >= self.envp_range.end - self.envp_range.start {
426            return Ok(0);
427        }
428
429        let read_at = self.envp_range.start + offset;
430        writer.limit(self.envp_range.end - read_at);
431        let bytes_read = self.vmar.read_remote(read_at, writer)?;
432
433        Ok(bytes_read)
434    }
