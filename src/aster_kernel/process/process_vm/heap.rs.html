7use crate::{

17#[derive(Debug)]
18pub struct Heap {
19    inner: Mutex<Option<HeapInner>>,
20}
21
22#[derive(Debug)]
23pub struct LockedHeap<'a> {
24    inner: MutexGuard<'a, Option<HeapInner>>,
25}
26
27#[derive(Clone, Debug)]
28struct HeapInner {
29    /// The size of the data segment, used for rlimit checking.
30    data_segment_size: usize,
31    /// The heap range.
32    // NOTE: `heap_range.end` is decided by user input and may not be page-aligned.
33    heap_range: Range<Vaddr>,
34}
35
36impl Heap {
37    pub(super) const fn new_uninitialized() -> Self {
38        Self {
39            inner: Mutex::new(None),
40        }
41    }
42
43    /// Creates a new `Heap` with identical contents of an existing one.
44    pub(super) fn fork_from(heap_guard: &LockedHeap) -> Self {
45        let inner = heap_guard.inner.as_ref().expect("Heap is not initialized");
46
47        Self {
48            inner: Mutex::new(Some(inner.clone())),
49        }
50    }
51
52    /// Initializes and maps the heap virtual memory.
53    pub(super) fn map_and_init_heap(
54        &self,
55        vmar: &Vmar,
56        data_segment_size: usize,
57        heap_base: Vaddr,
58    ) -> Result<()> {
59        let mut inner = self.inner.lock();
60
61        let nr_pages_padding = {
62            // Some random padding pages are added as a simple measure to
63            // make the heap values of a buggy user program harder
64            // to be exploited by attackers.
65            let mut nr_random_padding_pages: u8 = 0;
66            getrandom(nr_random_padding_pages.as_mut_bytes());
67
68            nr_random_padding_pages as usize
69        };
70
71        let heap_start = heap_base.align_up(PAGE_SIZE) + nr_pages_padding * PAGE_SIZE;
72        let heap_end = heap_start + PAGE_SIZE;
73        if heap_end > VMAR_CAP_ADDR {
74            return_errno_with_message!(Errno::ENOMEM, "the mapping address is too large");
75        }
76
77        let vmar_map_options = {
78            let perms = VmPerms::READ | VmPerms::WRITE;
79            vmar.new_map(PAGE_SIZE, perms).unwrap().offset(heap_start)
80        };
81        vmar_map_options.build()?;
82
83        debug_assert!(inner.is_none());
84        *inner = Some(HeapInner {
85            data_segment_size,
86            heap_range: heap_start..heap_end,
87        });
88
89        Ok(())
90    }
91
92    /// Locks the heap and returns a guard to access the heap information.
93    pub fn lock(&self) -> LockedHeap<'_> {
94        LockedHeap {
95            inner: self.inner.lock(),
96        }
97    }
98
99    /// Modifies the end address of the heap.
100    ///
101    /// Returns the new heap end on success, or the current heap end on failure.
102    /// This behavior is consistent with the Linux `brk` syscall.
103    //
104    // Reference: <https://elixir.bootlin.com/linux/v6.16.9/source/mm/mmap.c#L115>
105    pub fn modify_heap_end(
106        &self,
107        new_heap_end: Vaddr,
108        ctx: &Context,
109    ) -> core::result::Result<Vaddr, Vaddr> {
110        let user_space = ctx.user_space();
111        let vmar = user_space.vmar();
112
113        let mut inner = self.inner.lock();
114        let inner = inner.as_mut().expect("Heap is not initialized");
115
116        let heap_start = inner.heap_range.start;
117        let current_heap_end = inner.heap_range.end;
118        let new_heap_range = heap_start..new_heap_end;
119
120        // Check if the new heap end is valid.
121        if new_heap_end < heap_start
122            || check_data_rlimit(inner.data_segment_size, &new_heap_range, ctx).is_err()
123            || new_heap_end.checked_add(PAGE_SIZE).is_none()
124        {
125            return Err(current_heap_end);
126        }
127
128        let current_heap_end_aligned = current_heap_end.align_up(PAGE_SIZE);
129        let new_heap_end_aligned = new_heap_end.align_up(PAGE_SIZE);
130
131        let old_size = current_heap_end_aligned - heap_start;
132        let new_size = new_heap_end_aligned - heap_start;
133
134        // No change in the heap mapping.
135        if old_size == new_size {
136            inner.heap_range = new_heap_range;
137            return Ok(new_heap_end);
138        }
139
140        // Because the mapped heap region may contain multiple mappings, which can be
141        // done by `mmap` syscall or other ways, we need to be careful when modifying
142        // the heap mapping.
143        // For simplicity, we set `check_single_mapping` to `true` to ensure that the
144        // heap region contains only a single mapping.
145        vmar.resize_mapping(heap_start, old_size, new_size, true)
146            .map_err(|_| current_heap_end)?;
147
148        inner.heap_range = new_heap_range;
149        Ok(new_heap_end)
150    }
151}
152
153impl LockedHeap<'_> {
154    /// Returns the current heap range.
155    pub fn heap_range(&self) -> &Range<Vaddr> {
156        let inner = self.inner.as_ref().expect("Heap is not initialized");
157
158        &inner.heap_range
159    }
160}
161
162/// Checks whether the new heap range exceeds the data segment size limit.
163// Reference: <https://elixir.bootlin.com/linux/v6.16.9/source/include/linux/mm.h#L3287>
164fn check_data_rlimit(
165    data_segment_size: usize,
166    new_heap_range: &Range<Vaddr>,
167    ctx: &Context,
168) -> Result<()> {
169    let rlimit_data = ctx
170        .process
171        .resource_limits()
172        .get_rlimit(ResourceType::RLIMIT_DATA)
173        .get_cur();
174
175    if rlimit_data.saturating_sub(data_segment_size as u64) < new_heap_range.len() as u64 {
176        return_errno_with_message!(Errno::ENOSPC, "the data segment size limit is reached");
177    }
178    Ok(())
179}
