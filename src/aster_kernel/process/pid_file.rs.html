24pub struct PidFile {
25    process: Weak<Process>,
26    is_nonblocking: AtomicBool,
27    /// The pseudo path associated with this pid file.
28    pseudo_path: Path,
29}
30
31impl Debug for PidFile {
32    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
33        let pid = self.process.upgrade().map_or(u32::MAX, |p| p.pid());
34        f.debug_struct("PidFile")
35            .field("process", &pid)
36            .field(
37                "is_nonblocking",
38                &self.is_nonblocking.load(Ordering::Relaxed),
39            )
40            .finish_non_exhaustive()
41    }
42}
43
44impl PidFile {
45    pub fn new(process: Arc<Process>, is_nonblocking: bool) -> Self {
46        let pseudo_path = PidfdFs::new_path(|_| "anon_inode:[pidfd]".to_string());
47
48        Self {
49            process: Arc::downgrade(&process),
50            is_nonblocking: AtomicBool::new(is_nonblocking),
51            pseudo_path,
52        }
53    }
54
55    fn check_io_events(&self) -> IoEvents {
56        // "A PID file descriptor can be monitored using poll(2), select(2),
57        // and epoll(7).  When the process that it refers to terminates, these
58        // interfaces indicate the file descriptor as readable."
59        // Reference: <https://man7.org/linux/man-pages/man2/pidfd_open.2.html>.
60        let Some(process) = self.process.upgrade() else {
61            // The process has been reaped.
62            return IoEvents::IN | IoEvents::HUP;
63        };
64        if process.status().is_zombie() {
65            IoEvents::IN
66        } else {
67            IoEvents::empty()
68        }
69    }
70
71    pub(super) fn is_nonblocking(&self) -> bool {
72        self.is_nonblocking.load(Ordering::Relaxed)
73    }
74
75    pub fn process_opt(&self) -> Option<Arc<Process>> {
76        self.process.upgrade()
77    }
78}

80impl FileLike for PidFile {
81    fn read(&self, _writer: &mut VmWriter) -> Result<usize> {
82        return_errno_with_message!(Errno::EINVAL, "PID file cannot be read");
83    }
84
85    fn write(&self, _reader: &mut VmReader) -> Result<usize> {
86        return_errno_with_message!(Errno::EINVAL, "PID file cannot be written");
87    }
88
89    fn read_at(&self, _offset: usize, _writer: &mut VmWriter) -> Result<usize> {
90        return_errno_with_message!(
91            Errno::EINVAL,
92            "PID file cannot be read at a specific offset"
93        );
94    }
95
96    fn write_at(&self, _offset: usize, _reader: &mut VmReader) -> Result<usize> {
97        return_errno_with_message!(
98            Errno::EINVAL,
99            "PID file cannot be written at a specific offset"
100        );
101    }
102
103    fn set_status_flags(&self, new_flags: StatusFlags) -> Result<()> {
104        if new_flags.contains(StatusFlags::O_NONBLOCK) {
105            self.is_nonblocking.store(true, Ordering::Relaxed);
106        } else {
107            self.is_nonblocking.store(false, Ordering::Relaxed);
108        }
109
110        Ok(())
111    }
112
113    fn status_flags(&self) -> StatusFlags {
114        if self.is_nonblocking() {
115            StatusFlags::O_NONBLOCK
116        } else {
117            StatusFlags::empty()
118        }
119    }
120
121    fn path(&self) -> &Path {
122        &self.pseudo_path
123    }
124
125    fn dump_proc_fdinfo(self: Arc<Self>, fd_flags: FdFlags) -> Box<dyn Display> {
126        struct FdInfo {
127            flags: u32,
128            pid: u32,
129        }
130
131        impl Display for FdInfo {
132            fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
133                writeln!(f, "pos:\t{}", 0)?;
134                writeln!(f, "flags:\t0{:o}", self.flags)?;
135                writeln!(f, "mnt_id:\t{}", PidfdFs::mount_node().id())?;
136                writeln!(f, "ino:\t{}", PidfdFs::shared_inode().ino())?;
137                writeln!(f, "Pid:\t{}", self.pid)?;
138                // TODO: Currently we do not support PID namespaces. Just print the PID once.
139                writeln!(f, "NSpid:\t{}", self.pid)
140            }
141        }
142
143        let mut flags = self.status_flags().bits() | self.access_mode() as u32;
144        if fd_flags.contains(FdFlags::CLOEXEC) {
145            flags |= CreationFlags::O_CLOEXEC.bits();
146        }
147        let pid = self.process.upgrade().map_or(u32::MAX, |p| p.pid());
148
149        Box::new(FdInfo { flags, pid })
150    }
151}

153impl Pollable for PidFile {
154    fn poll(&self, mask: IoEvents, poller: Option<&mut PollHandle>) -> IoEvents {
155        let Some(process) = self.process.upgrade() else {
156            // The process has been reaped.
157            return mask & (IoEvents::IN | IoEvents::HUP);
158        };
159        process
160            .pidfile_pollee
161            .poll_with(mask, poller, || self.check_io_events())
162    }
