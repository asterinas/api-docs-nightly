18pub struct PidFile {
19    process: Weak<Process>,
20    common: FileCommon,
21}
22
23impl Debug for PidFile {
24    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
25        let pid = self.process.upgrade().map_or(u32::MAX, |p| p.pid());
26        f.debug_struct("PidFile")
27            .field("process", &pid)
28            .field("is_nonblocking", &self.common.is_nonblocking())
29            .finish_non_exhaustive()
30    }
31}
32
33impl PidFile {
34    pub fn new(process: Arc<Process>, is_nonblocking: bool) -> Self {
35        let pseudo_path = PidfdFs::new_path(|_| "anon_inode:[pidfd]".to_string());
36        let status_flags = if is_nonblocking {
37            StatusFlags::O_NONBLOCK
38        } else {
39            StatusFlags::empty()
40        };
41
42        Self {
43            process: Arc::downgrade(&process),
44            common: FileCommon::new(pseudo_path, status_flags),
45        }
46    }
47
48    fn check_io_events(&self) -> IoEvents {
49        // "A PID file descriptor can be monitored using poll(2), select(2),
50        // and epoll(7).  When the process that it refers to terminates, these
51        // interfaces indicate the file descriptor as readable."
52        // Reference: <https://man7.org/linux/man-pages/man2/pidfd_open.2.html>.
53        let Some(process) = self.process.upgrade() else {
54            // The process has been reaped.
55            return IoEvents::IN | IoEvents::HUP;
56        };
57        if process.status().is_zombie() {
58            IoEvents::IN
59        } else {
60            IoEvents::empty()
61        }
62    }
63
64    pub(super) fn is_nonblocking(&self) -> bool {
65        self.common.is_nonblocking()
66    }
67
68    pub fn process_opt(&self) -> Option<Arc<Process>> {
69        self.process.upgrade()
70    }
71}

73impl FileLike for PidFile {
74    fn read_at(&self, _offset: usize, _writer: &mut VmWriter) -> Result<usize> {
75        return_errno_with_message!(
76            Errno::EINVAL,
77            "PID file cannot be read at a specific offset"
78        );
79    }
80
81    fn write_at(&self, _offset: usize, _reader: &mut VmReader) -> Result<usize> {
82        return_errno_with_message!(
83            Errno::EINVAL,
84            "PID file cannot be written at a specific offset"
85        );
86    }
87
88    fn access_mode(&self) -> AccessMode {
89        // Reference: <https://elixir.bootlin.com/linux/v7.0/source/kernel/fork.c#L1898>.
90        AccessMode::O_RDWR
91    }
92
93    fn common(&self) -> &FileCommon {
94        &self.common
95    }
96
97    fn dump_proc_fdinfo(self: Arc<Self>, fd_flags: FdFlags) -> Box<dyn Display> {
98        struct FdInfo {
99            flags: u32,
100            pid: u32,
101        }
102
103        impl Display for FdInfo {
104            fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
105                writeln!(f, "pos:\t{}", 0)?;
106                writeln!(f, "flags:\t0{:o}", self.flags)?;
107                writeln!(f, "mnt_id:\t{}", PidfdFs::mount_node().id())?;
108                writeln!(f, "ino:\t{}", PidfdFs::shared_inode().ino())?;
109                writeln!(f, "Pid:\t{}", self.pid)?;
110                // TODO: Currently we do not support PID namespaces. Just print the PID once.
111                writeln!(f, "NSpid:\t{}", self.pid)
112            }
113        }
114
115        let mut flags = self.status_flags().bits() | self.access_mode() as u32;
116        if fd_flags.contains(FdFlags::CLOEXEC) {
117            flags |= CreationFlags::O_CLOEXEC.bits();
118        }
119        let pid = self.process.upgrade().map_or(u32::MAX, |p| p.pid());
120
121        Box::new(FdInfo { flags, pid })
122    }
123}
124
125impl Pollable for PidFile {
126    fn poll(&self, mask: IoEvents, poller: Option<&mut PollHandle>) -> IoEvents {
127        let Some(process) = self.process.upgrade() else {
128            // The process has been reaped.
129            return mask & (IoEvents::IN | IoEvents::HUP);
130        };
131        process
132            .pidfile_pollee
133            .poll_with(mask, poller, || self.check_io_events())
134    }
