38bitflags! {
39    #[derive(Default)]
40    pub struct CloneFlags: u32 {
41        const CLONE_NEWTIME = 0x00000080;       /* New time namespace */
42        const CLONE_VM      = 0x00000100;       /* Set if VM shared between processes.  */
43        const CLONE_FS      = 0x00000200;       /* Set if fs info shared between processes.  */
44        const CLONE_FILES   = 0x00000400;       /* Set if open files shared between processes.  */
45        const CLONE_SIGHAND = 0x00000800;       /* Set if signal handlers shared.  */
46        const CLONE_PIDFD   = 0x00001000;       /* Set if a pidfd should be placed in parent.  */
47        const CLONE_PTRACE  = 0x00002000;       /* Set if tracing continues on the child.  */
48        const CLONE_VFORK   = 0x00004000;       /* Set if the parent wants the child to wake it up on mm_release.  */
49        const CLONE_PARENT  = 0x00008000;       /* Set if we want to have the same parent as the cloner.  */
50        const CLONE_THREAD  = 0x00010000;       /* Set to add to same thread group.  */
51        const CLONE_NEWNS   = 0x00020000;       /* Set to create new namespace.  */
52        const CLONE_SYSVSEM = 0x00040000;       /* Set to shared SVID SEM_UNDO semantics.  */
53        const CLONE_SETTLS  = 0x00080000;       /* Set TLS info.  */
54        const CLONE_PARENT_SETTID = 0x00100000; /* Store TID in userlevel buffer before MM copy.  */
55        const CLONE_CHILD_CLEARTID = 0x00200000;/* Register exit futex and memory location to clear.  */
56        const CLONE_DETACHED = 0x00400000;      /* Create clone detached.  */
57        const CLONE_UNTRACED = 0x00800000;      /* Set if the tracing process can't force CLONE_PTRACE on this clone.  */
58        const CLONE_CHILD_SETTID = 0x01000000;  /* Store TID in userlevel buffer in the child.  */
59        const CLONE_NEWCGROUP   = 0x02000000;	/* New cgroup namespace.  */
60        const CLONE_NEWUTS	= 0x04000000;	    /* New utsname group.  */
61        const CLONE_NEWIPC	= 0x08000000;	    /* New ipcs.  */
62        const CLONE_NEWUSER	= 0x10000000;	    /* New user namespace.  */
63        const CLONE_NEWPID	= 0x20000000;	    /* New pid namespace.  */
64        const CLONE_NEWNET	= 0x40000000;	    /* New network namespace.  */
65        const CLONE_IO	= 0x80000000;	        /* Clone I/O context.  */
66
67        /// A bitmask of all `CloneFlags` related to namespace creation.
68        const CLONE_NS_FLAGS = Self::CLONE_NEWTIME.bits() |
69            Self::CLONE_NEWNS.bits() |
70            Self::CLONE_NEWCGROUP.bits() |
71            Self::CLONE_NEWUTS.bits() |
72            Self::CLONE_NEWIPC.bits() |
73            Self::CLONE_NEWUSER.bits() |
74            Self::CLONE_NEWPID.bits() |
75            Self::CLONE_NEWNET.bits();
76    }

79/// An internal structure to homogenize the arguments for `clone` and
80/// `clone3`.
81///
82/// From the clone(2) man page:
83///
84/// ```
85/// The following table shows the equivalence between the arguments
86/// of clone() and the fields in the clone_args argument supplied to
87/// clone3():
88///     clone()         clone3()        Notes
89///                     cl_args field
90///     flags & ~0xff   flags           For most flags; details
91///                                     below
92///     parent_tid      pidfd           See CLONE_PIDFD
93///     child_tid       child_tid       See CLONE_CHILD_SETTID
94///     parent_tid      parent_tid      See CLONE_PARENT_SETTID
95///     flags & 0xff    exit_signal
96///     stack           stack
97///     ---             stack_size
98///     tls             tls             See CLONE_SETTLS
99///     ---             set_tid         See below for details
100///     ---             set_tid_size
101///     ---             cgroup          See CLONE_INTO_CGROUP
102/// ```
103#[derive(Debug, Clone, Copy, Default)]
104pub struct CloneArgs {
105    pub flags: CloneFlags,
106    pub pidfd: Option<Vaddr>,
107    pub child_tid: Vaddr,
108    pub parent_tid: Option<Vaddr>,
109    pub exit_signal: Option<SigNum>,
110    pub stack: Option<NonZeroU64>,
111    pub stack_size: Option<NonZeroU64>,
112    pub tls: u64,
113    pub _set_tid: Option<u64>,
114    pub _set_tid_size: Option<u64>,
115    pub _cgroup: Option<u64>,
116}
117
118impl CloneArgs {
119    /// Prepares a new [`CloneArgs`] based on the arguments for clone(2).
120    pub fn for_clone(
121        raw_flags: u64,
122        parent_tid: Vaddr,
123        child_tid: Vaddr,
124        tls: u64,
125        stack: u64,
126    ) -> Result<Self> {
127        const FLAG_MASK: u64 = 0xff;
128        let flags = CloneFlags::from(raw_flags & !FLAG_MASK);
129        let exit_signal = raw_flags & FLAG_MASK;
130
131        // Disambiguate the `parent_tid` parameter. The field is used
132        // both for `CLONE_PIDFD` and `CLONE_PARENT_SETTID`, so at
133        // most only one can be specified.
134        let (pidfd, parent_tid) = match (
135            flags.contains(CloneFlags::CLONE_PIDFD),
136            flags.contains(CloneFlags::CLONE_PARENT_SETTID),
137        ) {
138            (false, false) => (None, None),
139            (true, false) => (Some(parent_tid), None),
140            (false, true) => (None, Some(parent_tid)),
141            (true, true) => {
142                return_errno_with_message!(
143                    Errno::EINVAL,
144                    "`CLONE_PIDFD` and `CLONE_PARENT_SETTID` cannot be specified together"
145                );
146            }
147        };
148
149        Ok(Self {
150            flags,
151            pidfd,
152            child_tid,
153            parent_tid,
154            exit_signal: (exit_signal != 0).then(|| SigNum::from_u8(exit_signal as u8)),
155            stack: NonZeroU64::new(stack),
156            tls,
157            ..Default::default()
158        })
159    }
160
161    pub fn for_fork() -> Self {
162        Self {
163            exit_signal: Some(SIGCHLD),
164            ..Default::default()
165        }
166    }
167
168    pub fn for_vfork() -> Self {
169        Self {
170            flags: CloneFlags::CLONE_VFORK | CloneFlags::CLONE_VM,
171            exit_signal: Some(SIGCHLD),
172            ..Default::default()
173        }
174    }
175
176    pub(self) fn check(&self, ctx: &Context) -> Result<()> {
177        let clone_flags = self.flags;
178        clone_flags.check_unsupported_flags()?;
179
180        // This checks the arguments for all clone-related system calls.
181        // Reference: <https://elixir.bootlin.com/linux/v6.16.9/source/kernel/fork.c#L1926-L1978>.
182
183        // Reject invalid argument combinations related to the CLONE_FS flag.
184        if clone_flags.contains(CloneFlags::CLONE_FS)
185            && clone_flags.intersects(CloneFlags::CLONE_NEWNS | CloneFlags::CLONE_NEWUSER)
186        {
187            return_errno_with_message!(
188                Errno::EINVAL,
189                "`CLONE_FS` cannot be used together with `CLONE_NEWNS` or `CLONE_NEWUSER`"
190            );
191        }
192
193        // Reject invalid argument combinations related to the CLONE_PARENT flag.
194        if clone_flags.contains(CloneFlags::CLONE_PARENT) {
195            if clone_flags.intersects(CloneFlags::CLONE_NEWUSER | CloneFlags::CLONE_NEWPID) {
196                return_errno_with_message!(
197                    Errno::EINVAL,
198                    "`CLONE_PARENT` cannot be used together with `CLONE_NEWUSER` or `CLONE_NEWPID`"
199                );
200            }
201
202            if ctx.process.is_init_process() {
203                return_errno_with_message!(
204                    Errno::EINVAL,
205                    "`CLONE_PARENT` cannot be used if the process is the init process"
206                )
207            }
208        }
209
210        // Reject invalid argument combinations related to the CLONE_THREAD flag.
211        if clone_flags.contains(CloneFlags::CLONE_THREAD) {
212            if !clone_flags.contains(CloneFlags::CLONE_VM | CloneFlags::CLONE_SIGHAND) {
213                return_errno_with_message!(
214                    Errno::EINVAL,
215                    "`CLONE_THREAD` without `CLONE_VM` and `CLONE_SIGHAND` is not valid"
216                );
217            }
218
219            if clone_flags.intersects(CloneFlags::CLONE_PIDFD | CloneFlags::CLONE_NEWUSER) {
220                return_errno_with_message!(
221                    Errno::EINVAL,
222                    "`CLONE_THREAD` cannot be used together with `CLONE_PIDFD` or `CLONE_NEWUSER`"
223                );
224            }
225        }
226
227        // Reject invalid argument combinations related to the CLONE_SIGHAND flag.
228        if clone_flags.contains(CloneFlags::CLONE_SIGHAND)
229            && !clone_flags.contains(CloneFlags::CLONE_VM)
230        {
231            return_errno_with_message!(
232                Errno::EINVAL,
233                "`CLONE_SIGHAND` without `CLONE_VM` is not valid"
234            );
235        }
236
237        Ok(())
238    }
239}

241impl From<u64> for CloneFlags {
242    fn from(flags: u64) -> Self {
243        // We use the lower 32 bits
244        let clone_flags = (flags & 0xffff_ffff) as u32;
245        CloneFlags::from_bits_truncate(clone_flags)
246    }
247}

249impl CloneFlags {
250    fn check_unsupported_flags(&self) -> Result<()> {
251        let supported_flags = CloneFlags::CLONE_VM
252            | CloneFlags::CLONE_FS
253            | CloneFlags::CLONE_FILES
254            | CloneFlags::CLONE_SIGHAND
255            | CloneFlags::CLONE_PIDFD
256            | CloneFlags::CLONE_THREAD
257            | CloneFlags::CLONE_SYSVSEM
258            | CloneFlags::CLONE_SETTLS
259            | CloneFlags::CLONE_PARENT_SETTID
260            | CloneFlags::CLONE_CHILD_SETTID
261            | CloneFlags::CLONE_CHILD_CLEARTID
262            | CloneFlags::CLONE_VFORK
263            | CloneFlags::CLONE_NEWNS
264            | CloneFlags::CLONE_PARENT;
265        let unsupported_flags = *self - supported_flags;
266        if !unsupported_flags.is_empty() {
267            warn!("contains unsupported clone flags: {:?}", unsupported_flags);
268        }
269        Ok(())
270    }
271}

273/// Clone a child thread or child process.
274///
275/// FIXME: currently, the child process or thread will be scheduled to run at once,
276/// but this may not be the expected behavior.
277pub fn clone_child(
278    ctx: &Context,
279    parent_context: &UserContext,
280    clone_args: CloneArgs,
281) -> Result<Tid> {
282    clone_args.check(ctx)?;
283
284    if clone_args.flags.contains(CloneFlags::CLONE_THREAD) {
285        let child_task = clone_child_task(ctx, parent_context, clone_args)?;
286        let child_thread = child_task.as_thread().unwrap();
287        child_thread.run();
288
289        let child_tid = child_thread.as_posix_thread().unwrap().tid();
290        Ok(child_tid)
291    } else {
292        let child_process = clone_child_process(ctx, parent_context, clone_args)?;
293
294        let mut cgroup_guard = CgroupMembership::lock();
295        let cgroup = ctx.process.cgroup().get().map(|cgroup| cgroup.clone());
296        if let Some(cgroup) = cgroup {
297            cgroup_guard
298                .move_process_to_node(child_process.clone(), &cgroup)
299                .unwrap();
300        }
301        drop(cgroup_guard);
302
303        if clone_args.flags.contains(CloneFlags::CLONE_VFORK) {
304            child_process.status().set_vfork_child(true);
305        }
306
307        child_process.run();
308
309        PROCESS_CREATION_COUNTER
310            .get()
311            .unwrap()
312            // Race conditions are fine as we don't really care which CPU creates a process.
313            .add_on_cpu(CpuId::current_racy(), 1);
314
315        if child_process.status().is_vfork_child() {
316            let cond = || (!child_process.status().is_vfork_child()).then_some(());
317            let current = ctx.process.as_ref();
318            current.children_wait_queue().wait_until(cond);
319        }
320
321        let child_pid = child_process.pid();
322        Ok(child_pid)
323    }
324}

326fn clone_child_task(
327    ctx: &Context,
328    parent_context: &UserContext,
329    clone_args: CloneArgs,
330) -> Result<Arc<Task>> {
331    let clone_flags = clone_args.flags;
332
333    let Context {
334        process,
335        thread_local,
336        posix_thread,
337        ..
338    } = ctx;
339
340    // Clone system V semaphore
341    clone_sysvsem(clone_flags)?;
342
343    // Clone file table
344    let child_file_table = clone_files(thread_local.borrow_file_table().unwrap(), clone_flags);
345
346    // Clone fs
347    let child_fs = clone_fs(&thread_local.borrow_fs(), clone_flags);
348
349    // Clone FPU context
350    let child_fpu_context = thread_local.fpu().clone_context();
351
352    // Clone namespaces
353    let child_user_ns = thread_local.borrow_user_ns().clone();
354    let child_ns_proxy = clone_ns_proxy(
355        thread_local.borrow_ns_proxy().unwrap(),
356        &child_user_ns,
357        clone_flags,
358        posix_thread,
359    )?;
360
361    // Clone default timer slack
362    let default_timer_slack_ns = posix_thread.timer_slack_ns();
363
364    if clone_flags.contains(CloneFlags::CLONE_NEWNS) {
365        child_fs
366            .resolver()
367            .write()
368            .switch_to_mnt_ns(child_ns_proxy.mnt_ns())?;
369    }
370
371    let child_user_ctx = Box::new(clone_user_ctx(
372        parent_context,
373        clone_args.stack,
374        clone_args.stack_size,
375        clone_args.tls,
376        clone_flags,
377    ));
378
379    // Inherit sigmask from current thread
380    let sig_mask = posix_thread.sig_mask().load(Ordering::Relaxed).into();
381
382    // Inherit the thread name.
383    let thread_name = posix_thread.thread_name().lock().clone();
384
385    let child_tid = allocate_posix_tid();
386    let child_task = {
387        let credentials = {
388            let credentials = ctx.posix_thread.credentials();
389            Credentials::new_from(&credentials)
390        };
391
392        let mut thread_builder =
393            PosixThreadBuilder::new(child_tid, thread_name, child_user_ctx, credentials)
394                .process(posix_thread.weak_process())
395                .sig_mask(sig_mask)
396                .file_table(child_file_table)
397                .fs(child_fs)
398                .fpu_context(child_fpu_context)
399                .user_ns(child_user_ns)
400                .ns_proxy(child_ns_proxy)
401                .default_timer_slack_ns(default_timer_slack_ns);
402
403        // Deal with SETTID/CLEARTID flags
404        clone_parent_settid(child_tid, clone_args.parent_tid, clone_flags)?;
405        thread_builder = clone_child_cleartid(thread_builder, clone_args.child_tid, clone_flags);
406        thread_builder = clone_child_settid(thread_builder, clone_args.child_tid, clone_flags);
407
408        thread_builder.build()
409    };
410
411    process
412        .tasks()
413        .lock()
414        .insert(child_task.clone())
415        .map_err(|_| {
416            Error::with_message(
417                Errno::EINTR,
418                "the process has exited or has already executed a new program",
419            )
420        })?;
421
422    Ok(child_task)
423}

425fn clone_child_process(
426    ctx: &Context,
427    parent_context: &UserContext,
428    clone_args: CloneArgs,
429) -> Result<Arc<Process>> {
430    let Context {
431        process,
432        thread_local,
433        posix_thread,
434        ..
435    } = ctx;
436
437    let clone_flags = clone_args.flags;
438
439    // Clone the virtual memory space
440    let child_vmar = clone_vmar(thread_local.vmar().borrow().as_ref().unwrap(), clone_flags)?;
441
442    // Clone the user context
443    let child_user_ctx = Box::new(clone_user_ctx(
444        parent_context,
445        clone_args.stack,
446        clone_args.stack_size,
447        clone_args.tls,
448        clone_flags,
449    ));
450
451    // Clone the file table
452    let child_file_table = clone_files(thread_local.borrow_file_table().unwrap(), clone_flags);
453
454    // Clone the filesystem information
455    let child_fs = clone_fs(&thread_local.borrow_fs(), clone_flags);
456
457    // Clone signal dispositions
458    let child_sig_dispositions = clone_sighand(process.sig_dispositions(), clone_flags);
459
460    // Clone System V semaphore
461    clone_sysvsem(clone_flags)?;
462
463    // Clone FPU context
464    let child_fpu_context = thread_local.fpu().clone_context();
465
466    // Clone the namespaces
467    let child_user_ns = clone_user_ns(clone_flags, thread_local)?;
468    let child_ns_proxy = clone_ns_proxy(
469        thread_local.borrow_ns_proxy().unwrap(),
470        &child_user_ns,
471        clone_flags,
472        posix_thread,
473    )?;
474
475    // Clone default timer slack
476    let default_timer_slack_ns = posix_thread.timer_slack_ns();
477
478    if clone_flags.contains(CloneFlags::CLONE_NEWNS) {
479        child_fs
480            .resolver()
481            .write()
482            .switch_to_mnt_ns(child_ns_proxy.mnt_ns())?;
483    }
484
485    // Inherit the parent's signal mask
486    let child_sig_mask = posix_thread.sig_mask().load(Ordering::Relaxed).into();
487
488    // Inherit the parent's resource limits
489    let child_resource_limits = process.resource_limits().clone();
490
491    // Inherit the parent's nice value
492    let child_nice = process.nice().load(Ordering::Relaxed);
493
494    // Inherit the parent's OOM score adjustment
495    let child_oom_score_adj = process.oom_score_adj().load(Ordering::Relaxed);
496
497    let child_tid = allocate_posix_tid();
498
499    let child = {
500        let mut child_thread_builder = {
501            let thread_name = {
502                let executable_path = child_vmar.process_vm().executable_file();
503                thread_local
504                    .borrow_fs()
505                    .resolver()
506                    .read()
507                    .make_abs_path(executable_path)
508                    .into_string()
509            };
510            let child_thread_name = ThreadName::new_from_executable_path(&thread_name);
511
512            let credentials = {
513                let credentials = ctx.posix_thread.credentials();
514                Credentials::new_from(&credentials)
515            };
516
517            PosixThreadBuilder::new(child_tid, child_thread_name, child_user_ctx, credentials)
518                .sig_mask(child_sig_mask)
519                .file_table(child_file_table)
520                .fs(child_fs)
521                .fpu_context(child_fpu_context)
522                .user_ns(child_user_ns.clone())
523                .ns_proxy(child_ns_proxy)
524                .default_timer_slack_ns(default_timer_slack_ns)
525        };
526
527        // Deal with SETTID/CLEARTID flags
528        clone_parent_settid(child_tid, clone_args.parent_tid, clone_flags)?;
529        child_thread_builder =
530            clone_child_cleartid(child_thread_builder, clone_args.child_tid, clone_flags);
531        child_thread_builder =
532            clone_child_settid(child_thread_builder, clone_args.child_tid, clone_flags);
533
534        create_child_process(
535            child_tid,
536            child_vmar,
537            child_resource_limits,
538            child_nice,
539            child_oom_score_adj,
540            child_sig_dispositions,
541            child_user_ns,
542            child_thread_builder,
543        )
544    };
545
546    clone_pidfd(ctx, &child, clone_flags, clone_args.pidfd)?;
547
548    if let Some(sig) = clone_args.exit_signal {
549        child.set_exit_signal(sig);
550    };
551
552    // Sets parent process and group for child process.
553    set_parent_and_group(clone_flags, process, &child);
554
555    Ok(child)
556}

558fn clone_child_cleartid(
559    child_builder: PosixThreadBuilder,
560    child_tidptr: Vaddr,
561    clone_flags: CloneFlags,
562) -> PosixThreadBuilder {
563    if clone_flags.contains(CloneFlags::CLONE_CHILD_CLEARTID) {
564        child_builder.clear_child_tid(child_tidptr)
565    } else {
566        child_builder
567    }
568}
569
570fn clone_child_settid(
571    child_builder: PosixThreadBuilder,
572    child_tidptr: Vaddr,
573    clone_flags: CloneFlags,
574) -> PosixThreadBuilder {
575    if clone_flags.contains(CloneFlags::CLONE_CHILD_SETTID) {
576        child_builder.set_child_tid(child_tidptr)
577    } else {
578        child_builder
579    }
580}

582fn clone_parent_settid(
583    child_tid: Tid,
584    parent_tidptr: Option<Vaddr>,
585    clone_flags: CloneFlags,
586) -> Result<()> {
587    if let Some(addr) =
588        parent_tidptr.filter(|_| clone_flags.contains(CloneFlags::CLONE_PARENT_SETTID))
589    {
590        current_userspace!().write_val(addr, &child_tid)?;
591    }
592    Ok(())
593}

595fn clone_vmar(parent_vmar: &Arc<Vmar>, clone_flags: CloneFlags) -> Result<Arc<Vmar>> {
596    // If CLONE_VM is set, the child and parent share the same VMAR.
597    // Otherwise, the child has a copy of the parent's VMAR.
598    if clone_flags.contains(CloneFlags::CLONE_VM) {
599        Ok(parent_vmar.clone())
600    } else {
601        Ok(Vmar::fork_from(parent_vmar)?)
602    }
603}

605fn clone_user_ctx(
606    parent_context: &UserContext,
607    new_sp: Option<NonZeroU64>,
608    stack_size: Option<NonZeroU64>,
609    tls: u64,
610    clone_flags: CloneFlags,
611) -> UserContext {
612    let mut child_context = parent_context.clone();
613    // The return value in the child thread is zero.
614    child_context.set_syscall_ret(0);
615
616    if let Some(new_sp) = new_sp {
617        if let Some(stack_size) = stack_size {
618            // `new_sp` is the stack bottom if the stack size is specified.
619            child_context.set_stack_pointer((new_sp.get() + stack_size.get()) as usize);
620        } else {
621            // `new_sp` is the stack top if the stack size is not specified.
622            child_context.set_stack_pointer(new_sp.get() as usize);
623        }
624    }
625    if clone_flags.contains(CloneFlags::CLONE_SETTLS) {
626        child_context.set_tls_pointer(tls as usize);
627    }
628
629    child_context
630}

632fn clone_fs(parent_fs: &Arc<ThreadFsInfo>, clone_flags: CloneFlags) -> Arc<ThreadFsInfo> {
633    // If CLONE_FS is set, the child and parent share the same filesystem information.
634    // Otherwise, the child has a copy of the parent's filesystem information.
635    if clone_flags.contains(CloneFlags::CLONE_FS) {
636        parent_fs.clone()
637    } else {
638        Arc::new(parent_fs.as_ref().clone())
639    }
640}

642fn clone_files(parent_file_table: &RwArc<FileTable>, clone_flags: CloneFlags) -> RwArc<FileTable> {
643    // If CLONE_FILES is set, the child and parent share the same file table.
644    // Otherwise, the child has a copy of the parent's file table.
645    if clone_flags.contains(CloneFlags::CLONE_FILES) {
646        parent_file_table.clone()
647    } else {
648        RwArc::new(parent_file_table.read().clone())
649    }
650}

652fn clone_sighand(
653    parent_sig_dispositions: &Mutex<Arc<Mutex<SigDispositions>>>,
654    clone_flags: CloneFlags,
655) -> Arc<Mutex<SigDispositions>> {
656    // If CLONE_SIGHAND is set, the child and parent shares the same signal handlers.
657    // Otherwise, the child has a copy of the parent's signal handlers.
658    if clone_flags.contains(CloneFlags::CLONE_SIGHAND) {
659        parent_sig_dispositions.lock().clone()
660    } else {
661        let sig_dispositions = parent_sig_dispositions.lock();
662        let sig_dispositions = sig_dispositions.lock();
663        Arc::new(Mutex::new(*sig_dispositions))
664    }
665}

674fn clone_pidfd(
675    ctx: &Context,
676    child: &Arc<Process>,
677    clone_flags: CloneFlags,
678    pidfd_addr: Option<Vaddr>,
679) -> Result<()> {
680    if !clone_flags.contains(CloneFlags::CLONE_PIDFD) {
681        return Ok(());
682    }
683
684    let pidfd_addr = pidfd_addr.unwrap();
685
686    let fd = {
687        let pid_file = PidFile::new(child.clone(), false);
688        let file_table = ctx.thread_local.borrow_file_table();
689        let mut file_table_locked = file_table.unwrap().write();
690        file_table_locked.insert(Arc::new(pid_file), FdFlags::CLOEXEC)
691    };
692
693    // Since `write_val` may sleep, we cannot hold the file table lock during its execution.
694    // FIXME: Should we remove the file from the file table if the write operation fails?
695    match ctx.user_space().write_val(pidfd_addr, &fd) {
696        Ok(()) => Ok(()),
697        Err(err) => {
698            let file_table = ctx.thread_local.borrow_file_table();
699            let mut file_table_locked = file_table.unwrap().write();
700            file_table_locked.close_file(fd);
701            Err(err.into())
702        }
703    }
704}

730fn create_child_process(
731    pid: Pid,
732    vmar: Arc<Vmar>,
733    resource_limits: ResourceLimits,
734    nice: Nice,
735    oom_score_adj: i16,
736    sig_dispositions: Arc<Mutex<SigDispositions>>,
737    user_ns: Arc<UserNamespace>,
738    thread_builder: PosixThreadBuilder,
739) -> Arc<Process> {
740    let child_proc = Process::new(
741        pid,
742        vmar,
743        resource_limits,
744        nice,
745        oom_score_adj,
746        sig_dispositions,
747        user_ns,
748    );
749
750    let child_task = thread_builder.process(Arc::downgrade(&child_proc)).build();
751    child_proc.tasks().lock().insert(child_task).unwrap();
752
753    child_proc
754}

756fn set_parent_and_group(clone_flags: CloneFlags, parent: &Arc<Process>, child: &Arc<Process>) {
757    loop {
758        let real_parent = clone_parent(clone_flags, parent);
759
760        // Lock the parent's children before checking its status.
761        let mut children_mut = real_parent.children().lock();
762
763        let Some(children_mut) = children_mut.as_mut() else {
764            // The real parent is concurrently exiting group.
765            // The children have been cleared,
766            // so the retrial will see an up-to-date real parent.
767            continue;
768        };
769
770        // Lock order: children of process -> parent of process
771        child.parent().lock().set_process(&real_parent);
772
773        // Update `has_child_subreaper` for the child process to
774        // make sure the `has_child_subreaper` state of the child process
775        // will be consistent with its parent.
776        if real_parent.has_child_subreaper.load(Ordering::Acquire) {
777            child.has_child_subreaper.store(true, Ordering::Release);
778        }
779
780        // Lock order: children of process -> process table
781        // -> group of process -> group inner
782
783        let mut process_table_mut = process_table::process_table_mut();
784
785        let process_group_mut = parent.process_group.lock();
786
787        let process_group = process_group_mut.upgrade().unwrap();
788        let mut process_group_inner = process_group.lock();
789
790        // Put the child process in the parent's process group
791        process_group_inner.insert_process(child.clone());
792        *child.process_group.lock() = Arc::downgrade(&process_group);
793
794        // Put the child process in the parent's `children` field
795        children_mut.insert(child.pid(), child.clone());
796
797        // Put the child process in the global table
798        process_table_mut.insert(child.pid(), child.clone());
799
800        return;
801    }
802}

804fn clone_parent(clone_flags: CloneFlags, current: &Arc<Process>) -> Cow<'_, Arc<Process>> {
805    if clone_flags.contains(CloneFlags::CLONE_PARENT) {
806        // The parent process of the current process cannot be `None`, since we have checked that
807        // the current process is not the init process.
808        Cow::Owned(current.parent().lock().process().upgrade().unwrap())
809    } else {
810        Cow::Borrowed(current)
811    }
812}
