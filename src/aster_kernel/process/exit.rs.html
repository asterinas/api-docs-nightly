11/// Exits the current POSIX process.
12///
13/// This is for internal use. Do NOT call this directly. When the last thread in the process exits,
14/// [`do_exit`] or [`do_exit_group`] will invoke this method automatically.
15///
16/// [`do_exit`]: crate::process::posix_thread::do_exit
17/// [`do_exit_group`]: crate::process::posix_thread::do_exit_group
18pub(super) fn exit_process(current_process: &Process) {
19    current_process.status().set_vfork_child(false);
20
21    // Drop fields in `Process`.
22    drop_after!(current_process.lock_vmar().set_vmar(None));
23
24    // Move the children to the reaper process and send them signals. The children should see a new
25    // parent when they receive the signal.
26    let children = move_children_to_reaper_process(current_process);
27    send_parent_death_signal(&children);
28    drop(children);
29
30    // Mark the current process as a zombie. After this point, it may be reaped concurrently and
31    // removed from the process tree. For example, the process may no longer have a live parent if
32    // the parent exits after reaping the process.
33    //
34    // This must happen after `move_children_to_reaper_process`. See the comments in
35    // `find_reaper_process` for details.
36    current_process.status().set_zombie();
37
38    current_process.pidfile_pollee.notify(IoEvents::IN);
39
40    send_child_death_signal(current_process);
41
42    // Remove the process from the cgroup.
43    let mut cgroup_guard = CgroupMembership::write_lock();
44    cgroup_guard.move_process_to_root(current_process);
45    drop(cgroup_guard);
46}

48/// Drops a value after releasing the lock or borrow guard.
49///
50/// For example, [`ThreadLocal::vmar`] needs to be borrowed during a context switch. This means that
51/// the kernel may panic if we borrow it as mutable and tries to acquire a mutex while dropping the
52/// last [`VmarHandle`]. This utility macro ensures that the value is dropped only after the borrow
53/// guard has been released, preventing all these issues.
54///
55/// [`ThreadLocal::vmar`]: super::posix_thread::ThreadLocal::vmar
56/// [`VmarHandle`]: crate::vm::vmar::VmarHandle
57macro_rules! drop_after {
58    ($expr:expr) => {
59        let val = $expr;
60        drop(val);
61    };
62}
63
64pub(super) use drop_after;

66/// Moves the children to a reaper process.
67///
68/// Returns the moved children. Note that no new processes can become this process's children after
69/// this point, so the returned children reliably represent the set of children at the time the
70/// process exits.
71fn move_children_to_reaper_process(current_process: &Process) -> BTreeMap<Pid, Arc<Process>> {
72    if current_process.is_init_process() {
73        return BTreeMap::new();
74    }
75
76    while let Some(reaper_process) = find_reaper_process(current_process) {
77        if let Ok(children) = move_process_children(current_process, &reaper_process) {
78            reaper_process.children_wait_queue().wake_all();
79            return children;
80        }
81    }
82
83    let init_process = pid_table::pid_table_mut()
84        .get_process(INIT_PROCESS_PID)
85        .unwrap();
86    let children = move_process_children(current_process, &init_process).unwrap();
87    init_process.children_wait_queue().wake_all();
88
89    children
90}

92/// Finds a reaper process for `current_process`.
93///
94/// If there is no reaper process for `current_process`, returns `None`.
95fn find_reaper_process(current_process: &Process) -> Option<Arc<Process>> {
96    // The current process is not yet zombie (see `exit_process`), so it cannot have been reaped,
97    // and it is still present in its parent's children map. An exiting parent will move the
98    // current process to a new reaper (updating this `Weak` reference) before becoming zombie
99    // itself. Therefore, the parent is always alive and `upgrade` cannot fail.
100    let mut parent = current_process.parent().lock().process().upgrade().unwrap();
101
102    loop {
103        if parent.is_init_process() {
104            return Some(parent);
105        }
106
107        if !parent.has_child_subreaper.load(Ordering::Acquire) {
108            return None;
109        }
110
111        let is_reaper = parent.is_child_subreaper();
112        let is_zombie = parent.status().is_zombie();
113        if is_reaper && !is_zombie {
114            return Some(parent);
115        }
116
117        let grandparent = parent.parent().lock().process().upgrade();
118        if let Some(grandparent) = grandparent {
119            parent = grandparent;
120        } else {
121            // The grandparent was concurrently reaped and dropped, so we lost the path up the
122            // ancestor chain. Retry from the current process's parent to pick up the latest reaper
123            // assignment. This `upgrade` won't fail for the same reason explained above.
124            parent = current_process.parent().lock().process().upgrade().unwrap();
125        }
126    }
127}

129/// Moves the children of `current_process` to be the children of `reaper_process`.
130///
131/// Returns the moved children on success. Otherwise, if the `reaper_process` is zombie, returns
132/// `Err(())`.
133fn move_process_children(
134    current_process: &Process,
135    reaper_process: &Arc<Process>,
136) -> Result<BTreeMap<Pid, Arc<Process>>, ()> {
137    // Lock order: children of process -> parent of process
138    let mut reaper_process_children = reaper_process.children().lock();
139    let Some(reaper_process_children) = reaper_process_children.as_mut() else {
140        // The reaper process has exited, and it is not the init process
141        // (since we never clear the init process's children).
142        return Err(());
143    };
144
145    // We hold the lock of children while updating the children's parents.
146    // This ensures when dealing with CLONE_PARENT,
147    // the retrial will see an up-to-date real parent.
148    let mut current_children = current_process.children().lock();
149    let children = current_children.take().unwrap();
150
151    for child_process in children.values() {
152        let mut parent = child_process.parent.lock();
153        reaper_process_children.insert(child_process.pid(), child_process.clone());
154        parent.set_process(reaper_process);
155    }
156
157    Ok(children)
158}

160/// Sends parent-death signals to the children.
161//
162// FIXME: According to the Linux implementation, the signal should be sent when the POSIX thread
163// that created the child exits, not when the whole process exits. For more details, see the
164// "CAVEATS" section in <https://man7.org/linux/man-pages/man2/pr_set_pdeathsig.2const.html>.
165fn send_parent_death_signal(children: &BTreeMap<Pid, Arc<Process>>) {
166    for child in children.values() {
167        let Some(signum) = child.parent_death_signal() else {
168            continue;
169        };
170
171        // FIXME: Set `si_pid` in the `siginfo_t` argument.
172        let signal = Box::new(KernelSignal::new(signum));
173        child.enqueue_signal(signal);
174    }
175}

177/// Sends a child-death signal to the parent.
178fn send_child_death_signal(current_process: &Process) {
179    let Some(parent) = current_process.parent().lock().process().upgrade() else {
180        return;
181    };
182
183    if let Some(signal) = current_process.exit_signal().map(KernelSignal::new) {
184        parent.enqueue_signal(Box::new(signal));
185    };
186    parent.children_wait_queue().wake_all();
