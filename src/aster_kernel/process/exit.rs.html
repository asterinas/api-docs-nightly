11/// Exits the current POSIX process.
12///
13/// This is for internal use. Do NOT call this directly. When the last thread in the process exits,
14/// [`do_exit`] or [`do_exit_group`] will invoke this method automatically.
15///
16/// [`do_exit`]: crate::process::posix_thread::do_exit
17/// [`do_exit_group`]: crate::process::posix_thread::do_exit_group
18pub(super) fn exit_process(current_process: &Process) {
19    current_process.status().set_zombie();
20    current_process.status().set_vfork_child(false);
21
22    // Drop fields in `Process`.
23    current_process.lock_vmar().set_vmar(None);
24
25    current_process.pidfile_pollee.notify(IoEvents::IN);
26
27    send_parent_death_signal(current_process);
28
29    move_children_to_reaper_process(current_process);
30
31    send_child_death_signal(current_process);
32
33    // Remove the process from the cgroup.
34    let mut cgroup_guard = CgroupMembership::lock();
35    cgroup_guard.move_process_to_root(current_process);
36    drop(cgroup_guard);
37}

39/// Sends parent-death signals to the children.
40//
41// FIXME: According to the Linux implementation, the signal should be sent when the POSIX thread
42// that created the child exits, not when the whole process exits. For more details, see the
43// "CAVEATS" section in <https://man7.org/linux/man-pages/man2/pr_set_pdeathsig.2const.html>.
44fn send_parent_death_signal(current_process: &Process) {
45    let current_children = current_process.children().lock();
46    for child in current_children.as_ref().unwrap().values() {
47        let Some(signum) = child.parent_death_signal() else {
48            continue;
49        };
50
51        // FIXME: Set `si_pid` in the `siginfo_t` argument.
52        let signal = Box::new(KernelSignal::new(signum));
53        child.enqueue_signal(signal);
54    }
55}

57/// Finds a reaper process for `current_process`.
58///
59/// If there is no reaper process for `current_process`, returns `None`.
60fn find_reaper_process(current_process: &Process) -> Option<Arc<Process>> {
61    let mut parent = current_process.parent().lock().process().upgrade().unwrap();
62
63    loop {
64        if parent.is_init_process() {
65            return Some(parent);
66        }
67
68        if !parent.has_child_subreaper.load(Ordering::Acquire) {
69            return None;
70        }
71
72        let is_reaper = parent.is_child_subreaper();
73        let is_zombie = parent.status().is_zombie();
74        if is_reaper && !is_zombie {
75            return Some(parent);
76        }
77
78        let grandparent = parent.parent().lock().process().upgrade();
79        if let Some(grandparent) = grandparent {
80            parent = grandparent;
81        } else {
82            // If both the parent and grandparent have exited concurrently, we will lose the clue
83            // about the ancestor processes. Therefore, we have to retry.
84            parent = current_process.parent().lock().process().upgrade().unwrap();
85        }
86    }
87}

89/// Moves the children of `current_process` to be the children of `reaper_process`.
90///
91/// If the `reaper_process` is zombie, returns `Err(())`.
92fn move_process_children(
93    current_process: &Process,
94    reaper_process: &Arc<Process>,
95) -> core::result::Result<(), ()> {
96    // Lock order: children of process -> parent of process
97    let mut reaper_process_children = reaper_process.children().lock();
98    let Some(reaper_process_children) = reaper_process_children.as_mut() else {
99        // The reaper process has exited, and it is not the init process
100        // (since we never clear the init process's children).
101        return Err(());
102    };
103
104    // We hold the lock of children while updating the children's parents.
105    // This ensures when dealing with CLONE_PARENT,
106    // the retrial will see an up-to-date real parent.
107    let mut current_children = current_process.children().lock();
108    for child_process in current_children.as_mut().unwrap().values() {
109        let mut parent = child_process.parent.lock();
110        reaper_process_children.insert(child_process.pid(), child_process.clone());
111        parent.set_process(reaper_process);
112    }
113    *current_children = None;
114
115    Ok(())
116}

118/// Moves the children to a reaper process.
119fn move_children_to_reaper_process(current_process: &Process) {
120    if current_process.is_init_process() {
121        return;
122    }
123
124    while let Some(reaper_process) = find_reaper_process(current_process) {
125        if move_process_children(current_process, &reaper_process).is_ok() {
126            reaper_process.children_wait_queue().wake_all();
127            return;
128        }
129    }
130
131    const INIT_PROCESS_PID: Pid = 1;
132
133    let init_process = process_table::get_process(INIT_PROCESS_PID).unwrap();
134    move_process_children(current_process, &init_process).unwrap();
135    init_process.children_wait_queue().wake_all();
136}

138/// Sends a child-death signal to the parent.
139fn send_child_death_signal(current_process: &Process) {
140    let Some(parent) = current_process.parent().lock().process().upgrade() else {
141        return;
142    };
143
144    if let Some(signal) = current_process.exit_signal().map(KernelSignal::new) {
145        parent.enqueue_signal(Box::new(signal));
146    };
147    parent.children_wait_queue().wake_all();
148}
