20#[derive(Debug)]
21pub struct UnixControlMessage(Message);
22
23#[derive(Debug)]
24enum Message {
25    Files(FileMessage),
26    Cred(CredMessage),
27}
28
29impl UnixControlMessage {
30    pub fn read_from(header: &CControlHeader, reader: &mut VmReader) -> Result<Option<Self>> {
31        debug_assert_eq!(header.level(), Some(CSocketOptionLevel::SOL_SOCKET));
32
33        let Ok(type_) = CControlType::try_from(header.type_()) else {
34            warn!("unsupported control message type in {:?}", header);
35            reader.skip(header.payload_len());
36            return Ok(None);
37        };
38
39        match type_ {
40            CControlType::SCM_RIGHTS => {
41                let msg = FileMessage::read_from(header, reader)?;
42                Ok(Some(Self(Message::Files(msg))))
43            }
44            CControlType::SCM_CREDENTIALS => {
45                let msg = CredMessage::read_from(header, reader)?;
46                Ok(Some(Self(Message::Cred(msg))))
47            }
48            _ => {
49                warn!("unsupported control message type in {:?}", header);
50                reader.skip(header.payload_len());
51                Ok(None)
52            }
53        }
54    }
55
56    pub fn write_to(&self, writer: &mut VmWriter) -> Result<CControlHeader> {
57        match &self.0 {
58            Message::Files(msg) => msg.write_to(writer),
59            Message::Cred(msg) => msg.write_to(writer),
60        }
61    }
62}

64struct FileMessage {
65    files: Vec<Arc<dyn FileLike>>,
66}
67
68impl fmt::Debug for FileMessage {
69    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
70        f.debug_struct("FileMessage")
71            .field("len", &self.files.len())
72            .finish_non_exhaustive()
73    }
74}

76/// The maximum number of the file descriptors in the control messages.
77///
78/// Reference: <https://elixir.bootlin.com/linux/v6.15/source/include/net/scm.h#L18>.
79const MAX_NR_FILES: usize = 253;
80
81impl FileMessage {
82    fn read_from(header: &CControlHeader, reader: &mut VmReader) -> Result<Self> {
83        let payload_len = header.payload_len();
84        if !payload_len.is_multiple_of(size_of::<i32>()) {
85            return_errno_with_message!(Errno::EINVAL, "the SCM_RIGHTS message is invalid");
86        }
87        let nfiles = payload_len / size_of::<i32>();
88
89        // "Attempting to send an array larger than this limit causes sendmsg(2) to fail with the
90        // error EINVAL." -- Reference: <https://man7.org/linux/man-pages/man7/unix.7.html>.
91        if nfiles > MAX_NR_FILES {
92            return_errno_with_message!(Errno::EINVAL, "the SCM_RIGHTS message is too large");
93        }
94        // TODO: "[the ETOOMANYREFS error] occurs if the number of "in-flight" file descriptors
95        // exceeds the RLIMIT_NOFILE resource limit and the caller does not have the
96        // CAP_SYS_RESOURCE capability."
97
98        let mut files = Vec::with_capacity(nfiles);
99
100        let current = Task::current().unwrap();
101        let mut file_table = current.as_thread_local().unwrap().borrow_file_table_mut();
102        for _ in 0..nfiles {
103            let fd = reader.read_val::<i32>()?;
104            let file = get_file_fast!(&mut file_table, fd).into_owned();
105            files.push(file);
106        }
107
108        Ok(FileMessage { files })
109    }
110
111    fn write_to(&self, writer: &mut VmWriter) -> Result<CControlHeader> {
112        let nfiles = self
113            .files
114            .len()
115            .min(CControlHeader::payload_len_from_total(writer.avail())? / size_of::<i32>());
116        if nfiles == 0 {
117            return_errno_with_message!(Errno::EINVAL, "the control message buffer is too small");
118        }
119        if nfiles < self.files.len() {
120            warn!("setting MSG_CTRUNC is not supported");
121        }
122
123        let header = CControlHeader::new(
124            CSocketOptionLevel::SOL_SOCKET,
125            CControlType::SCM_RIGHTS as i32,
126            nfiles * size_of::<i32>(),
127        );
128        writer.write_val::<CControlHeader>(&header)?;
129
130        let current = Task::current().unwrap();
131        let file_table = current.as_thread_local().unwrap().borrow_file_table();
132        for file in self.files[..nfiles].iter() {
133            // TODO: Deal with the `O_CLOEXEC` flag.
134            let fd = file_table
135                .unwrap()
136                .write()
137                .insert(file.clone(), FdFlags::empty());
138            // Perhaps we should remove the inserted files from the file table if we cannot write
139            // the file descriptor back to user space? However, even Linux cannot handle every
140            // corner case (https://elixir.bootlin.com/linux/v6.15.2/source/net/core/scm.c#L357).
141            writer.write_val::<i32>(&fd)?;
142        }
143
144        Ok(header)
145    }
146}

148#[derive(Debug)]
149struct CredMessage {
150    cred: CUserCred,
151}
152
153impl CredMessage {
154    fn read_from(header: &CControlHeader, reader: &mut VmReader) -> Result<Self> {
155        if header.payload_len() != size_of::<CUserCred>() {
156            return_errno_with_message!(Errno::EINVAL, "the SCM_CREDENTIALS message is invalid");
157        }
158
159        let cred = reader.read_val()?;
160
161        Ok(Self { cred })
162    }
163
164    fn write_to(&self, writer: &mut VmWriter) -> Result<CControlHeader> {
165        let payload_len =
166            size_of::<CUserCred>().min(CControlHeader::payload_len_from_total(writer.avail())?);
167        if payload_len != size_of::<CUserCred>() {
168            warn!("setting MSG_CTRUNC is not supported");
169        }
170
171        let header = CControlHeader::new(
172            CSocketOptionLevel::SOL_SOCKET,
173            CControlType::SCM_CREDENTIALS as i32,
174            payload_len,
175        );
176        writer.write_val(&header)?;
177        writer.write_fallible(&mut VmReader::from(self.cred.as_bytes()))?;
178
179        Ok(header)
180    }
181}

183/// Control message types.
184///
185/// Reference: <https://elixir.bootlin.com/linux/v6.13/source/include/linux/socket.h#L178>.
186#[repr(i32)]
187#[derive(Debug, Clone, Copy, TryFromInt, PartialEq, Eq)]
188#[expect(non_camel_case_types)]
189enum CControlType {
190    SCM_RIGHTS = 1,
191    SCM_CREDENTIALS = 2,
192    SCM_SECURITY = 3,
193    SCM_PIDFD = 4,
194}

196/// Auxiliary data associated with UNIX messages.
197///
198/// In UNIX sockets, one can send payload bytes with multiple control messages. If these control
199/// messages need to be sent to a remote endpoint, they are packaged in this type and transmitted.
200///
201/// We use this type instead of transmitting control messages directly to the remote endpoint
202/// because control messages of the same type (e.g., files) can be merged and missing control
203/// messages of certain types (e.g., credentials) can be supplied automatically according to socket
204/// option settings.
205#[derive(Default)]
206pub(super) struct AuxiliaryData {
207    files: Vec<Arc<dyn FileLike>>,
208    cred: Option<SocketCred>,
209}
210
211impl AuxiliaryData {
212    /// Builds the auxiliary data from the control messages.
213    pub(super) fn from_control(ctrl_msgs: Vec<ControlMessage>) -> Result<Self> {
214        let mut files = Vec::new();
215        let mut cred = None;
216
217        for ctrl_msg in ctrl_msgs.into_iter() {
218            let ControlMessage::Unix(unix_ctrl_msg) = ctrl_msg;
219            // TODO: What should we do if there are control messages of other protocols?
220
221            match unix_ctrl_msg.0 {
222                Message::Files(FileMessage {
223                    files: mut msg_files,
224                }) => {
225                    if msg_files.len() > MAX_NR_FILES - files.len() {
226                        return_errno_with_message!(
227                            Errno::EINVAL,
228                            "the SCM_RIGHTS message is too large"
229                        );
230                    }
231                    files.append(&mut msg_files);
232                }
233                Message::Cred(CredMessage { cred: msg_cred }) => {
234                    let cur_cred = SocketCred::<ReadOp>::new_current();
235                    if cur_cred.to_real_c_cred() != msg_cred {
236                        // FIXME: Allow this if we're root or have the CAP_SYS_ADMIN capability.
237                        return_errno_with_message!(
238                            Errno::EPERM,
239                            "setting others' credentials is not allowed"
240                        );
241                    }
242                    cred = Some(cur_cred);
243                }
244            }
245        }
246
247        // FIXME: Sending UNIX sockets over UNIX sockets can easily lead to circular references and
248        // memory leaks. Linux uses a complex garbage collection algorithm to address these issues.
249        // See also <https://elixir.bootlin.com/linux/v6.15/source/net/unix/garbage.c#L592>.
250        if files
251            .iter()
252            .any(|file| (&**file as &dyn Any).is::<UnixStreamSocket>())
253        {
254            warn!("UNIX sockets in SCM_RIGHTS messages can leak kernel resource");
255
256            let credentials = current_thread!().as_posix_thread().unwrap().credentials();
257            if !credentials.effective_capset().contains(CapSet::SYS_ADMIN) {
258                return_errno_with_message!(
259                    Errno::EPERM,
260                    "UNIX sockets in SCM_RIGHTS messages can leak kernel resource"
261                )
262            }
263        }
264
265        Ok(Self { files, cred })
266    }
267
268    /// Fills the current credentials if there are no credentials.
269    pub(super) fn fill_cred(&mut self) {
270        if self.cred.is_none() {
271            self.cred = Some(SocketCred::<ReadOp>::new_current());
272        }
273    }
274
275    /// Generates the control messages from the auxiliary data.
276    pub(super) fn generate_control(&mut self, is_pass_cred: bool) -> Vec<ControlMessage> {
277        let mut ctrl_msgs = Vec::new();
278
279        let Self { files, cred } = self;
280
281        if is_pass_cred {
282            let unix_ctrl_msg = UnixControlMessage(Message::Cred(CredMessage {
283                cred: cred
284                    .as_ref()
285                    .map(SocketCred::to_real_c_cred)
286                    .unwrap_or_else(CUserCred::new_overflow),
287            }));
288            ctrl_msgs.push(ControlMessage::Unix(unix_ctrl_msg));
289        }
290
291        if !files.is_empty() {
292            let unix_ctrl_msg = UnixControlMessage(Message::Files(FileMessage {
293                files: core::mem::take(files),
294            }));
295            ctrl_msgs.push(ControlMessage::Unix(unix_ctrl_msg));
296        }
297
298        ctrl_msgs
299    }
300
301    /// Returns whether the auxiliary data contains nothing.
302    pub(super) fn is_empty(&self) -> bool {
303        self.files.is_empty() && self.cred.is_none()
304    }
305
306    /// Returns whether the auxiliary data can be treated as a subset of the other one.
307    ///
308    /// In stream sockets, we can receive more bytes at once if the current auxiliary data is a
309    /// subset of the subsequent auxiliary data.
310    pub(super) fn is_subset_of(&self, other: &Self, is_pass_cred: bool) -> bool {
311        if !self.files.is_empty() {
312            return false;
313        }
314
315        if is_pass_cred
316            && self.cred.as_ref().map(SocketCred::to_real_c_cred)
317                != other.cred.as_ref().map(SocketCred::to_real_c_cred)
318        {
319            return false;
320        }
321
322        true
323    }
