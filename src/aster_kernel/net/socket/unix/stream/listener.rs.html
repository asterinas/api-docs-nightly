30pub(super) struct Listener {
31    backlog: Arc<Backlog>,
32    is_write_shutdown: AtomicBool,
33}
34
35impl Listener {
36    pub(super) fn new(
37        addr: UnixSocketAddrBound,
38        backlog: usize,
39        is_read_shutdown: bool,
40        is_write_shutdown: bool,
41        pollee: Pollee,
42        is_seqpacket: bool,
43    ) -> Self {
44        let backlog = BACKLOG_TABLE
45            .add_backlog(addr, pollee, backlog, is_read_shutdown, is_seqpacket)
46            .unwrap();
47
48        Self {
49            backlog,
50            is_write_shutdown: AtomicBool::new(is_write_shutdown),
51        }
52    }
53
54    pub(super) fn addr(&self) -> &UnixSocketAddrBound {
55        self.backlog.addr()
56    }
57
58    pub(super) fn try_accept(&self, is_seqpacket: bool) -> Result<(Arc<dyn FileLike>, SocketAddr)> {
59        let connected = self.backlog.pop_incoming()?;
60
61        let peer_addr = connected.peer_addr().into();
62        let options = OptionSet::new_accepted(connected.is_pass_cred());
63
64        let socket = UnixStreamSocket::new_connected(connected, options, false, is_seqpacket);
65        Ok((socket, peer_addr))
66    }
67
68    pub(super) fn listen(&self, backlog: usize) {
69        self.backlog.set_backlog(backlog);
70    }
71
72    pub(super) fn shutdown(&self, cmd: SockShutdownCmd, pollee: &Pollee) {
73        if cmd.shut_read() {
74            self.backlog.shutdown();
75        }
76
77        if cmd.shut_write() {
78            self.is_write_shutdown.store(true, Ordering::Relaxed);
79            pollee.notify(SHUT_WRITE_EVENTS);
80        }
81    }
82
83    pub(super) fn is_read_shutdown(&self) -> bool {
84        self.backlog.is_shutdown()
85    }
86
87    pub(super) fn is_write_shutdown(&self) -> bool {
88        self.is_write_shutdown.load(Ordering::Relaxed)
89    }
90
91    pub(super) fn set_pass_cred(&self, is_pass_cred: bool) {
92        self.backlog
93            .is_pass_cred
94            .store(is_pass_cred, Ordering::Relaxed);
95    }
96
97    pub(super) fn check_io_events(&self) -> IoEvents {
98        self.backlog.check_io_events()
99    }
100
101    pub(super) fn cred(&self) -> &SocketCred<ReadDupOp> {
102        &self.backlog.listener_cred
103    }
104}
105
106impl Drop for Listener {
107    fn drop(&mut self) {
108        self.backlog.shutdown();
109
110        BACKLOG_TABLE.remove_backlog(&self.backlog.addr().to_key());
111    }
112}

114static BACKLOG_TABLE: BacklogTable = BacklogTable::new();
115
116struct BacklogTable {
117    backlog_sockets: RwLock<BTreeMap<UnixSocketAddrKey, Arc<Backlog>>>,
118}
119
120impl BacklogTable {
121    const fn new() -> Self {
122        Self {
123            backlog_sockets: RwLock::new(BTreeMap::new()),
124        }
125    }
126
127    fn add_backlog(
128        &self,
129        addr: UnixSocketAddrBound,
130        pollee: Pollee,
131        backlog: usize,
132        is_shutdown: bool,
133        is_seqpacket: bool,
134    ) -> Option<Arc<Backlog>> {
135        let addr_key = addr.to_key();
136
137        let mut backlog_sockets = self.backlog_sockets.write();
138
139        if backlog_sockets.contains_key(&addr_key) {
140            return None;
141        }
142
143        let new_backlog = Arc::new(Backlog::new(
144            addr,
145            pollee,
146            backlog,
147            is_shutdown,
148            is_seqpacket,
149        ));
150        let old_backlog = backlog_sockets.insert(addr_key, new_backlog.clone());
151        debug_assert!(old_backlog.is_none());
152
153        Some(new_backlog)
154    }
155
156    fn get_backlog(&self, addr_key: &UnixSocketAddrKey) -> Option<Arc<Backlog>> {
157        self.backlog_sockets.read().get(addr_key).cloned()
158    }
159
160    fn remove_backlog(&self, addr_key: &UnixSocketAddrKey) {
161        let old_backlog = self.backlog_sockets.write().remove(addr_key);
162        debug_assert!(old_backlog.is_some());
163    }
164}

166pub(super) struct Backlog {
167    addr: UnixSocketAddrBound,
168    pollee: Pollee,
169    backlog: AtomicUsize,
170    incoming_conns: SpinLock<Option<VecDeque<Connected>>>,
171    connect_wait_queue: WaitQueue,
172    listener_cred: SocketCred<ReadDupOp>,
173    is_pass_cred: AtomicBool,
174    is_seqpacket: bool,
175}
176
177impl Backlog {
178    fn new(
179        addr: UnixSocketAddrBound,
180        pollee: Pollee,
181        backlog: usize,
182        is_shutdown: bool,
183        is_seqpacket: bool,
184    ) -> Self {
185        let incoming_sockets = if is_shutdown {
186            None
187        } else {
188            Some(VecDeque::with_capacity(backlog))
189        };
190
191        Self {
192            addr,
193            pollee,
194            backlog: AtomicUsize::new(backlog),
195            incoming_conns: SpinLock::new(incoming_sockets),
196            connect_wait_queue: WaitQueue::new(),
197            listener_cred: SocketCred::<ReadDupOp>::new_current(),
198            is_pass_cred: AtomicBool::new(false),
199            is_seqpacket,
200        }
201    }
202
203    fn addr(&self) -> &UnixSocketAddrBound {
204        &self.addr
205    }
206
207    fn pop_incoming(&self) -> Result<Connected> {
208        let mut locked_incoming_conns = self.incoming_conns.lock();
209
210        let Some(incoming_conns) = &mut *locked_incoming_conns else {
211            return_errno_with_message!(Errno::EINVAL, "the socket is shut down for reading");
212        };
213        let conn = incoming_conns.pop_front();
214
215        drop(locked_incoming_conns);
216
217        if conn.is_some() {
218            self.pollee.invalidate();
219            self.connect_wait_queue.wake_one();
220        }
221
222        conn.ok_or_else(|| Error::with_message(Errno::EAGAIN, "no pending connection is available"))
223    }
224
225    fn set_backlog(&self, backlog: usize) {
226        let old_backlog = self.backlog.swap(backlog, Ordering::Relaxed);
227
228        if old_backlog < backlog {
229            self.connect_wait_queue.wake_all();
230        }
231    }
232
233    fn shutdown(&self) {
234        *self.incoming_conns.lock() = None;
235
236        self.pollee.notify(SHUT_READ_EVENTS);
237        self.connect_wait_queue.wake_all();
238    }
239
240    fn is_shutdown(&self) -> bool {
241        self.incoming_conns.lock().is_none()
242    }
243
244    fn check_io_events(&self) -> IoEvents {
245        if self
246            .incoming_conns
247            .lock()
248            .as_ref()
249            .is_some_and(|conns| !conns.is_empty())
250        {
251            IoEvents::IN
252        } else {
253            IoEvents::empty()
254        }
255    }
256}
257
258impl Backlog {
259    pub(super) fn push_incoming(
260        &self,
261        init: Init,
262        pollee: Pollee,
263        options: &OptionSet,
264        is_seqpacket: bool,
265    ) -> core::result::Result<Connected, (Error, Init)> {
266        if is_seqpacket != self.is_seqpacket {
267            // FIXME: According to the Linux implementation, we should avoid this error by
268            // maintaining two socket tables for SOCK_STREAM sockets and SOCK_SEQPACKET sockets
269            // separately.
270            return Err((
271                Error::with_message(
272                    Errno::ECONNREFUSED,
273                    "the listening socket has a different socket type",
274                ),
275                init,
276            ));
277        }
278
279        let mut locked_incoming_conns = self.incoming_conns.lock();
280
281        let Some(incoming_conns) = &mut *locked_incoming_conns else {
282            return Err((
283                Error::with_message(
284                    Errno::ECONNREFUSED,
285                    "the listening socket is shut down for reading",
286                ),
287                init,
288            ));
289        };
290
291        if incoming_conns.len() >= self.backlog.load(Ordering::Relaxed) {
292            return Err((
293                Error::with_message(
294                    Errno::EAGAIN,
295                    "the pending connection queue on the listening socket is full",
296                ),
297                init,
298            ));
299        }
300
301        let (client_conn, server_conn) = init.into_connected(
302            self.addr.clone(),
303            pollee,
304            self.listener_cred.dup().restrict(),
305        );
306        options.apply_to_connected(&client_conn);
307        if self.is_pass_cred.load(Ordering::Relaxed) {
308            server_conn.set_pass_cred(true);
309        }
310
311        incoming_conns.push_back(server_conn);
312        self.pollee.notify(IoEvents::IN);
313
314        Ok(client_conn)
315    }
316
317    /// Blocks until the backlogs are free and the `try_connect` succeeds, or until interrupted.
318    pub(super) fn block_connect<F>(&self, mut try_connect: F) -> Result<()>
319    where
320        F: FnMut() -> Result<()>,
321    {
322        self.connect_wait_queue
323            .pause_until(|| match try_connect() {
324                Err(err) if err.error() == Errno::EAGAIN => None,
325                result => Some(result),
326            })?
327    }
328}

330pub(super) fn get_backlog(server_key: &UnixSocketAddrKey) -> Result<Arc<Backlog>> {
331    BACKLOG_TABLE.get_backlog(server_key).ok_or_else(|| {
332        Error::with_message(
333            Errno::ECONNREFUSED,
334            "no socket is listening at the remote address",
335        )
336    })
