3use aster_bigtcp::wire::{IpAddress, IpEndpoint, Ipv4Address};
4
5use crate::{net::socket::util::SocketAddr, prelude::*, return_errno_with_message};
6
7impl TryFrom<SocketAddr> for IpEndpoint {
8    type Error = Error;
9
10    fn try_from(value: SocketAddr) -> Result<Self> {
11        match value {
12            SocketAddr::IPv4(addr, port) => Ok(IpEndpoint::new(addr.into(), port)),
13            _ => return_errno_with_message!(
14                Errno::EAFNOSUPPORT,
15                "the address is in an unsupported address family"
16            ),
17        }
18    }
19}
20
21impl From<IpEndpoint> for SocketAddr {
22    fn from(endpoint: IpEndpoint) -> Self {
23        let port = endpoint.port;
24        match endpoint.addr {
25            IpAddress::Ipv4(addr) => SocketAddr::IPv4(addr, port),
26            // TODO: support IPv6
27        }
28    }
29}

31/// A local endpoint, which indicates that the local endpoint is unspecified.
32///
33/// According to the Linux man pages and the Linux implementation, `getsockname()` will _not_ fail
34/// even if the socket is unbound. Instead, it will return an unspecified socket address. This
35/// unspecified endpoint helps with that.
36pub(super) const UNSPECIFIED_LOCAL_ENDPOINT: IpEndpoint =
37    IpEndpoint::new(IpAddress::Ipv4(Ipv4Address::UNSPECIFIED), 0);
