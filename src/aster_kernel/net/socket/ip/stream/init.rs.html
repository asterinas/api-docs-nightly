26pub(super) struct InitStream {
27    bound_port: Option<BoundTcpPort>,
28    /// The address family of this socket (IPv4 or IPv6).
29    //
30    // TODO: Encode the address family in the type system (e.g., `InitStream<IPv4Family>`)
31    // so that type mismatches are caught at compile time instead of runtime.
32    family: IpAddressFamily,
33    /// Indicates if the last `connect()` is considered to be done.
34    ///
35    /// If `connect()` is called but we're still in the `InitStream`, this means that the
36    /// connection is refused.
37    ///
38    ///  * If the connection is refused synchronously, the error code is returned by the
39    ///    `connect()` system call, and after that we always consider the `connect()` to be already
40    ///    done.
41    ///
42    ///  * If the connection is refused asynchronously (e.g., non-blocking sockets or interrupted
43    ///    `connect()`), the last `connect()` is not considered to have been done until another
44    ///    `connect()`, which checks and resets the boolean value and returns an appropriate error
45    ///    code.
46    is_connect_done: bool,
47    /// Indicates whether the socket error is `ECONNREFUSED`.
48    ///
49    /// This boolean value is set to true when the connection is refused and set to false when the
50    /// error code is reported via `getsockopt(SOL_SOCKET, SO_ERROR)`, `send()`, `recv()`, or
51    /// `connect()`.
52    is_conn_refused: AtomicBool,
53}
54
55impl InitStream {
56    pub(super) fn new(family: IpAddressFamily) -> Self {
57        Self {
58            bound_port: None,
59            family,
60            is_connect_done: true,
61            is_conn_refused: AtomicBool::new(false),
62        }
63    }
64
65    pub(super) fn new_bound(bound_port: BoundTcpPort, family: IpAddressFamily) -> Self {
66        Self {
67            bound_port: Some(bound_port),
68            family,
69            is_connect_done: true,
70            is_conn_refused: AtomicBool::new(false),
71        }
72    }
73
74    pub(super) fn new_refused(bound_port: BoundTcpPort, family: IpAddressFamily) -> Self {
75        Self {
76            bound_port: Some(bound_port),
77            family,
78            is_connect_done: false,
79            is_conn_refused: AtomicBool::new(true),
80        }
81    }
82
83    /// Returns the address family of this socket.
84    pub(super) fn family(&self) -> IpAddressFamily {
85        self.family
86    }
87
88    pub(super) fn bind(&mut self, endpoint: &IpEndpoint, can_reuse: bool) -> Result<()> {
89        if self.bound_port.is_some() {
90            return_errno_with_message!(Errno::EINVAL, "the socket is already bound to an address");
91        }
92
93        // When we support `IPV6_V6ONLY` and if it is set, we should also reject IPv4-mapped
94        // IPv6 addresses.
95        if IpAddressFamily::from(endpoint.addr) != self.family {
96            return_errno_with_message!(
97                Errno::EAFNOSUPPORT,
98                "the protocol family does not match the address family"
99            );
100        }
101
102        self.bound_port = Some(bind_port(endpoint, can_reuse)?);
103
104        Ok(())
105    }
106
107    pub(super) fn bound_port(&self) -> Option<&BoundTcpPort> {
108        self.bound_port.as_ref()
109    }
110
111    pub(super) fn connect(
112        self,
113        remote_endpoint: &IpEndpoint,
114        option: &RawTcpOption,
115        can_reuse: bool,
116        observer: StreamObserver,
117    ) -> Result<ConnectingStream, (Error, Self)> {
118        debug_assert!(
119            self.is_connect_done,
120            "`finish_last_connect()` should be called before calling `connect()`"
121        );
122
123        // When we support `IPV6_V6ONLY` and if it is set, we should also reject IPv4-mapped
124        // IPv6 addresses.
125        if IpAddressFamily::from(remote_endpoint.addr) != self.family {
126            return Err((
127                Error::with_message(
128                    Errno::EAFNOSUPPORT,
129                    "the protocol family does not match the address family",
130                ),
131                self,
132            ));
133        }
134
135        let bound_port = if let Some(bound_port) = self.bound_port {
136            bound_port
137        } else {
138            let endpoint = match get_ephemeral_endpoint(remote_endpoint) {
139                Some(ep) => ep,
140                None => {
141                    return Err((
142                        Error::with_message(
143                            Errno::EADDRNOTAVAIL,
144                            "no interface has an address for the specified family",
145                        ),
146                        self,
147                    ));
148                }
149            };
150            match bind_port(&endpoint, can_reuse) {
151                Ok(bound_port) => bound_port,
152                Err(err) => return Err((err, self)),
153            }
154        };
155
156        ConnectingStream::new(bound_port, *remote_endpoint, option, observer).map_err(
157            |(err, bound_port)| {
158                if err.error() == Errno::ECONNREFUSED {
159                    (err, InitStream::new_refused(bound_port, self.family))
160                } else {
161                    (err, InitStream::new_bound(bound_port, self.family))
162                }
163            },
164        )
165    }
166
167    pub(super) fn finish_last_connect(&mut self) -> Result<()> {
168        if self.is_connect_done {
169            return Ok(());
170        }
171
172        self.is_connect_done = true;
173
174        let is_conn_refused = self.is_conn_refused.get_mut();
175        if *is_conn_refused {
176            *is_conn_refused = false;
177            return_errno_with_message!(Errno::ECONNREFUSED, "the connection is refused");
178        } else {
179            return_errno_with_message!(
180                Errno::ECONNABORTED,
181                "the error code for the connection failure is not available"
182            );
183        }
184    }
185
186    pub(super) fn listen(
187        self,
188        backlog: usize,
189        option: &RawTcpOption,
190        observer: StreamObserver,
191    ) -> Result<ListenStream, (Error, Self)> {
192        if !self.is_connect_done {
193            // See the comments of `is_connect_done`.
194            // `listen()` is also not allowed until the second `connect()`.
195            return Err((
196                Error::with_message(
197                    Errno::EINVAL,
198                    "the connection is refused, but the connecting phase is not done",
199                ),
200                self,
201            ));
202        }
203
204        let Some(bound_port) = self.bound_port else {
205            // FIXME: The socket should be bound to INADDR_ANY (i.e., 0.0.0.0) with an ephemeral
206            // port. However, INADDR_ANY is not yet supported, so we need to return an error first.
207            warn!("listen() without bind() is not implemented");
208            return Err((
209                Error::with_message(Errno::EINVAL, "listen() without bind() is not implemented"),
210                self,
211            ));
212        };
213
214        match ListenStream::new(bound_port, backlog, option, observer) {
215            Ok(listen_stream) => Ok(listen_stream),
216            Err((bound_port, error)) => Err((error, Self::new_bound(bound_port, self.family))),
217        }
218    }
219
220    pub(super) fn try_recv(&self) -> Result<(usize, SocketAddr)> {
221        // FIXME: Linux does not return addresses for `recvfrom` on connection-oriented sockets.
222        // This is a placeholder that has no Linux equivalent. (Note also that in this case
223        // `getpeeraddr` will simply fail with `ENOTCONN`).
224        const UNSPECIFIED_SOCKET_ADDR: SocketAddr = SocketAddr::IPv4(Ipv4Addr::UNSPECIFIED, 0);
225
226        // Below are some magic checks to make our behavior identical to Linux.
227
228        if self.is_connect_done {
229            return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected");
230        }
231
232        if let Some(err) = self.test_and_clear_error() {
233            return Err(err);
234        }
235
236        Ok((0, UNSPECIFIED_SOCKET_ADDR))
237    }
238
239    pub(super) fn try_send(&self) -> Result<usize> {
240        if let Some(err) = self.test_and_clear_error() {
241            return Err(err);
242        }
243
244        return_errno_with_message!(Errno::EPIPE, "the socket is not connected");
245    }
246
247    pub(super) fn local_endpoint(&self) -> Option<IpEndpoint> {
248        self.bound_port
249            .as_ref()
250            .map(|bound_port| bound_port.endpoint())
251    }
252
253    pub(super) fn check_io_events(&self) -> IoEvents {
254        // Linux adds OUT and HUP events for a newly created socket
255        let mut events = IoEvents::OUT | IoEvents::HUP;
256
257        if self.is_conn_refused.load(Ordering::Relaxed) {
258            events |= IoEvents::ERR;
259        }
260
261        events
262    }
263
264    pub(super) fn test_and_clear_error(&self) -> Option<Error> {
265        if self.is_conn_refused.swap(false, Ordering::Relaxed) {
266            Some(Error::with_message(
267                Errno::ECONNREFUSED,
268                "the connection is refused",
269            ))
270        } else {
271            None
272        }
273    }
274}

276fn bind_port(endpoint: &IpEndpoint, can_reuse: bool) -> Result<BoundTcpPort> {
277    let (iface, config) = resolve_bind_iface_and_config(endpoint, can_reuse)?;
278    Ok(iface.bind_tcp(config)?)
