3use aster_bigtcp::{
4    errors::tcp::ListenError,
5    socket::{RawTcpOption, RawTcpSetOption},
6    wire::IpEndpoint,
7};
8
9use super::{connected::ConnectedStream, observer::StreamObserver};
10use crate::{
11    events::IoEvents,
12    net::iface::{BoundPort, Iface, TcpListener},
13    prelude::*,
14};
15
16pub(super) struct ListenStream {
17    tcp_listener: TcpListener,
18}
19
20impl ListenStream {
21    pub(super) fn new(
22        bound_port: BoundPort,
23        backlog: usize,
24        option: &RawTcpOption,
25        observer: StreamObserver,
26    ) -> core::result::Result<Self, (BoundPort, Error)> {
27        const SOMAXCONN: usize = 4096;
28        let max_conn = SOMAXCONN.min(backlog);
29
30        match TcpListener::new_listen(bound_port, max_conn, option, observer) {
31            Ok(tcp_listener) => Ok(Self { tcp_listener }),
32            Err((bound_port, ListenError::AddressInUse)) => Err((
33                bound_port,
34                Error::with_message(Errno::EADDRINUSE, "listener key conflicts"),
35            )),
36            Err((_, err)) => {
37                unreachable!("`new_listen` fails with {:?}, which should not happen", err)
38            }
39        }
40    }
41
42    pub(super) fn try_accept(&self) -> Result<ConnectedStream> {
43        let (new_conn, remote_endpoint) = self.tcp_listener.accept().ok_or_else(|| {
44            Error::with_message(Errno::EAGAIN, "no pending connection is available")
45        })?;
46
47        Ok(ConnectedStream::new(new_conn, remote_endpoint, false))
48    }
49
50    pub(super) fn local_endpoint(&self) -> IpEndpoint {
51        self.tcp_listener.local_endpoint().unwrap()
52    }
53
54    pub(super) fn iface(&self) -> &Arc<Iface> {
55        self.tcp_listener.iface()
56    }
57
58    pub(super) fn check_io_events(&self) -> IoEvents {
59        let can_accept = self.tcp_listener.can_accept();
60
61        // If network packets come in simultaneously, the socket state may change in the middle.
62        // However, the current pollee implementation should be able to handle this race condition.
63        if can_accept {
64            IoEvents::IN
65        } else {
66            IoEvents::empty()
67        }
68    }
69
70    pub(super) fn set_raw_option<R>(
71        &self,
72        set_option: impl FnOnce(&dyn RawTcpSetOption) -> R,
73    ) -> R {
74        set_option(&self.tcp_listener)
75    }
76
77    pub(super) fn into_listener(self) -> TcpListener {
78        self.tcp_listener
79    }
80}
