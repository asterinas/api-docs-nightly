57pub struct StreamSocket {
58    // Lock order: `state` first, `options` second
59    // FIXME: We perform userspace reads/writes when holding the spin locks (e.g., this state lock
60    // and other locks in `aster-bigtcp`), which will break the atomic mode.
61    state: RwLock<Takeable<State>>,
62    options: RwLock<OptionSet>,
63
64    is_nonblocking: AtomicBool,
65    pollee: Pollee,
66    pseudo_path: Path,
67}
68
69enum State {
70    // Start state
71    Init(InitStream),
72    // Intermediate state
73    Connecting(ConnectingStream),
74    // Final State 1
75    Connected(ConnectedStream),
76    // Final State 2
77    Listen(ListenStream),
78}
79
80#[derive(Debug, Clone)]
81struct OptionSet {
82    socket: SocketOptionSet,
83    ip: IpOptionSet,
84    tcp: TcpOptionSet,
85}

87impl OptionSet {
88    fn new() -> Self {
89        let socket = SocketOptionSet::new_tcp();
90        let ip = IpOptionSet::new_tcp();
91        let tcp = TcpOptionSet::new();
92        OptionSet { socket, ip, tcp }
93    }

95    fn raw(&self) -> RawTcpOption {
96        RawTcpOption {
97            keep_alive: self.socket.keep_alive().then_some(KEEPALIVE_INTERVAL),
98            is_nagle_enabled: !self.tcp.no_delay(),
99        }
100    }

103impl StreamSocket {
104    pub fn new(is_nonblocking: bool) -> Arc<Self> {
105        let init_stream = InitStream::new();
106        Arc::new(Self {
107            state: RwLock::new(Takeable::new(State::Init(init_stream))),
108            options: RwLock::new(OptionSet::new()),
109            is_nonblocking: AtomicBool::new(is_nonblocking),
110            pollee: Pollee::new(),
111            pseudo_path: SockFs::new_path(),
112        })
113    }
114
115    fn new_accepted(connected_stream: ConnectedStream) -> Arc<Self> {
116        let options = connected_stream.raw_with(|raw_tcp_socket| {
117            let mut options = OptionSet::new();
118
119            if raw_tcp_socket.keep_alive().is_some() {
120                options.socket.set_keep_alive(true);
121            }
122
123            if !raw_tcp_socket.nagle_enabled() {
124                options.tcp.set_no_delay(true);
125            }
126
127            // TODO: Update other options for a newly-accepted socket
128
129            options
130        });
131
132        let pollee = Pollee::new();
133        connected_stream.init_observer(StreamObserver::new(pollee.clone()));
134
135        Arc::new(Self {
136            options: RwLock::new(options),
137            state: RwLock::new(Takeable::new(State::Connected(connected_stream))),
138            is_nonblocking: AtomicBool::new(false),
139            pollee,
140            pseudo_path: SockFs::new_path(),
141        })
142    }

144    /// Ensures that the socket state is up to date and obtains a read lock on it.
145    ///
146    /// For a description of what "up-to-date" means, see [`Self::write_updated_state`].
147    fn read_updated_state(&self) -> RwLockReadGuard<'_, Takeable<State>, PreemptDisabled> {
148        loop {
149            let state = self.state.read();
150            match state.as_ref() {
151                State::Connecting(connecting_stream) if connecting_stream.has_result() => (),
152                _ => return state,
153            };
154            drop(state);
155
156            self.write_updated_state();
157        }
158    }
159
160    /// Ensures that the socket state is up to date and obtains a write lock on it.
161    ///
162    /// A connecting socket can become obsolete because some network events can set the socket to
163    /// connected state (if the connection succeeds) or initial state (if the connection is
164    /// refused) in [`Self::check_io_events`], but the state transition is delayed until the user
165    /// operates on the socket to avoid too many locks in the interrupt handler.
166    ///
167    /// This method performs the delayed state transition to ensure that the state is up to date
168    /// and returns the guard of the write-locked state.
169    fn write_updated_state(&self) -> RwLockWriteGuard<'_, Takeable<State>, PreemptDisabled> {
170        let mut state = self.state.write();
171
172        match state.as_ref() {
173            State::Connecting(connection_stream) if connection_stream.has_result() => (),
174            _ => return state,
175        }
176
177        state.borrow(|owned_state| {
178            let State::Connecting(connecting_stream) = owned_state else {
179                unreachable!("`State::Connecting` is checked before calling `borrow_result`");
180            };
181
182            match connecting_stream.into_result() {
183                ConnResult::Connecting(connecting_stream) => State::Connecting(connecting_stream),
184                ConnResult::Connected(connected_stream) => State::Connected(connected_stream),
185                ConnResult::Refused(init_stream) => State::Init(init_stream),
186            }
187        });
188
189        state
190    }

192    // Returns `None` to block the task and wait for the connection to be established, and returns
193    // `Some(_)` if blocking is not necessary or not allowed.
194    fn start_connect(&self, remote_endpoint: &IpEndpoint) -> Option<Result<()>> {
195        let is_nonblocking = self.is_nonblocking();
196
197        let mut state = self.write_updated_state();
198
199        let options = self.options.read();
200        let raw_option = options.raw();
201
202        let (result_or_block, iface_to_poll) = state.borrow_result(|mut owned_state| {
203            let mut init_stream = match owned_state {
204                State::Init(init_stream) => init_stream,
205                State::Connecting(_) if is_nonblocking => {
206                    return (
207                        owned_state,
208                        (
209                            Some(Err(Error::with_message(
210                                Errno::EALREADY,
211                                "the socket is connecting",
212                            ))),
213                            None,
214                        ),
215                    );
216                }
217                State::Connecting(_) => return (owned_state, (None, None)),
218                State::Connected(ref mut connected_stream) => {
219                    let err = connected_stream.finish_last_connect();
220                    return (owned_state, (Some(err), None));
221                }
222                State::Listen(_) => {
223                    return (
224                        owned_state,
225                        (
226                            Some(Err(Error::with_message(
227                                Errno::EISCONN,
228                                "the socket is listening",
229                            ))),
230                            None,
231                        ),
232                    );
233                }
234            };
235
236            if let Err(err) = init_stream.finish_last_connect() {
237                return (State::Init(init_stream), (Some(Err(err)), None));
238            }
239
240            let (target_state, iface_to_poll) = match init_stream.connect(
241                remote_endpoint,
242                &raw_option,
243                options.socket.reuse_addr(),
244                StreamObserver::new(self.pollee.clone()),
245            ) {
246                Ok(connecting_stream) => {
247                    let iface_to_poll = connecting_stream.iface().clone();
248                    (State::Connecting(connecting_stream), Some(iface_to_poll))
249                }
250                Err((err, init_stream)) if err.error() == Errno::ECONNREFUSED => {
251                    // `ECONNREFUSED` should be reported asynchronously, i.e., we need to return
252                    // `EINPROGRESS` first for non-blocking sockets.
253                    (State::Init(init_stream), None)
254                }
255                Err((err, init_stream)) => {
256                    return (State::Init(init_stream), (Some(Err(err)), None));
257                }
258            };
259
260            let result_or_block = if is_nonblocking {
261                Some(Err(Error::with_message(
262                    Errno::EINPROGRESS,
263                    "the socket is connecting",
264                )))
265            } else {
266                None
267            };
268
269            (target_state, (result_or_block, iface_to_poll))
270        });
271
272        drop(state);
273        drop(options);
274        self.pollee.invalidate();
275        if let Some(iface) = iface_to_poll {
276            iface.poll();
277        }
278
279        result_or_block
280    }
281
282    fn check_connect(&self) -> Result<()> {
283        let mut state = self.write_updated_state();
284
285        match state.as_mut() {
286            State::Init(init_stream) => init_stream.finish_last_connect(),
287            State::Connecting(_) => {
288                return_errno_with_message!(Errno::EAGAIN, "the connection is pending")
289            }
290            State::Connected(connected_stream) => connected_stream.finish_last_connect(),
291            State::Listen(_) => {
292                return_errno_with_message!(Errno::EISCONN, "the socket is listening")
293            }
294        }
295    }
296
297    fn try_accept(&self) -> Result<(Arc<dyn FileLike>, SocketAddr)> {
298        let state = self.read_updated_state();
299
300        let State::Listen(listen_stream) = state.as_ref() else {
301            return_errno_with_message!(Errno::EINVAL, "the socket is not listening");
302        };
303
304        let accepted = listen_stream.try_accept().map(|connected_stream| {
305            let remote_endpoint = connected_stream.remote_endpoint();
306            let accepted_socket = Self::new_accepted(connected_stream);
307            (accepted_socket as _, remote_endpoint.into())
308        });
309        let iface_to_poll = listen_stream.iface().clone();
310
311        drop(state);
312        self.pollee.invalidate();
313        iface_to_poll.poll();
314
315        accepted
316    }
317
318    fn try_recv(
319        &self,
320        writer: &mut dyn MultiWrite,
321        flags: SendRecvFlags,
322    ) -> Result<(usize, SocketAddr)> {
323        let state = self.read_updated_state();
324
325        let connected_stream = match state.as_ref() {
326            State::Connected(connected_stream) => connected_stream,
327            State::Init(init_stream) => {
328                let result = init_stream.try_recv();
329                self.pollee.invalidate();
330                return result;
331            }
332            State::Listen(_) => {
333                return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected")
334            }
335            State::Connecting(_) => {
336                return_errno_with_message!(Errno::EAGAIN, "the socket is connecting")
337            }
338        };
339
340        let result = connected_stream.try_recv(writer, flags);
341        self.pollee.invalidate();
342
343        let (recv_bytes, need_poll) = result?;
344        let iface_to_poll = need_poll.then(|| connected_stream.iface().clone());
345        let remote_endpoint = connected_stream.remote_endpoint();
346
347        drop(state);
348        if let Some(iface) = iface_to_poll {
349            iface.poll();
350        }
351
352        Ok((recv_bytes, remote_endpoint.into()))
353    }
354
355    fn try_send(&self, reader: &mut dyn MultiRead, flags: SendRecvFlags) -> Result<usize> {
356        let state = self.read_updated_state();
357
358        let connected_stream = match state.as_ref() {
359            State::Connected(connected_stream) => connected_stream,
360            State::Init(init_stream) => {
361                let result = init_stream.try_send();
362                self.pollee.invalidate();
363                return result;
364            }
365            State::Listen(_) => {
366                return_errno_with_message!(Errno::EPIPE, "the socket is not connected");
367            }
368            State::Connecting(_) => {
369                // FIXME: Linux indeed allows data to be buffered at this point. Can we do
370                // something similar?
371                return_errno_with_message!(Errno::EAGAIN, "the socket is connecting")
372            }
373        };
374
375        let result = connected_stream.try_send(reader, flags);
376        self.pollee.invalidate();
377
378        let (sent_bytes, need_poll) = result?;
379        let iface_to_poll = need_poll.then(|| connected_stream.iface().clone());
380
381        drop(state);
382        if let Some(iface) = iface_to_poll {
383            iface.poll();
384        }
385
386        Ok(sent_bytes)
387    }
388
389    fn check_io_events(&self) -> IoEvents {
390        let state = self.read_updated_state();
391
392        match state.as_ref() {
393            State::Init(init_stream) => init_stream.check_io_events(),
394            State::Connecting(connecting_stream) => connecting_stream.check_io_events(),
395            State::Listen(listen_stream) => listen_stream.check_io_events(),
396            State::Connected(connected_stream) => connected_stream.check_io_events(),
397        }
398    }
399
400    fn test_and_clear_error(&self) -> Option<Error> {
401        let state = self.read_updated_state();
402
403        let error = match state.as_ref() {
404            State::Init(init_stream) => init_stream.test_and_clear_error(),
405            State::Connected(connected_stream) => connected_stream.test_and_clear_error(),
406            State::Connecting(_) | State::Listen(_) => None,
407        };
408        self.pollee.invalidate();
409        error
410    }

430impl Socket for StreamSocket {
431    fn bind(&self, socket_addr: SocketAddr) -> Result<()> {
432        let endpoint = socket_addr.try_into()?;
433
434        let mut state = self.write_updated_state();
435        let State::Init(init_stream) = state.as_mut() else {
436            return_errno_with_message!(Errno::EINVAL, "the socket is already bound to an address");
437        };
438
439        let can_reuse = self.options.read().socket.reuse_addr();
440        init_stream.bind(&endpoint, can_reuse)
441    }
442
443    fn connect(&self, socket_addr: SocketAddr) -> Result<()> {
444        let remote_endpoint = socket_addr.try_into()?;
445
446        if let Some(result) = self.start_connect(&remote_endpoint) {
447            return result;
448        }
449
450        self.wait_events(IoEvents::OUT, None, || self.check_connect())
451    }
452
453    fn listen(&self, backlog: usize) -> Result<()> {
454        let mut state = self.write_updated_state();
455
456        let options = self.options.read();
457        let raw_option = options.raw();
458
459        state.borrow_result(|owned_state| {
460            let init_stream = match owned_state {
461                State::Init(init_stream) => init_stream,
462                State::Listen(listen_stream) => {
463                    return (State::Listen(listen_stream), Ok(()));
464                }
465                State::Connecting(_) | State::Connected(_) => {
466                    return (
467                        owned_state,
468                        Err(Error::with_message(
469                            Errno::EINVAL,
470                            "the socket is already connected",
471                        )),
472                    );
473                }
474            };
475
476            let listen_stream = match init_stream.listen(
477                backlog,
478                &raw_option,
479                StreamObserver::new(self.pollee.clone()),
480            ) {
481                Ok(listen_stream) => listen_stream,
482                Err((err, init_stream)) => {
483                    return (State::Init(init_stream), Err(err));
484                }
485            };
486
487            self.pollee.invalidate();
488            (State::Listen(listen_stream), Ok(()))
489        })
490    }
491
492    fn accept(&self) -> Result<(Arc<dyn FileLike>, SocketAddr)> {
493        self.block_on(IoEvents::IN, || self.try_accept())
494    }
495
496    fn shutdown(&self, cmd: SockShutdownCmd) -> Result<()> {
497        let state = self.read_updated_state();
498
499        let (result, iface_to_poll) = match state.as_ref() {
500            State::Connected(connected_stream) => (
501                connected_stream.shutdown(cmd, &self.pollee),
502                connected_stream.iface().clone(),
503            ),
504            // TODO: shutdown listening stream
505            _ => return_errno_with_message!(Errno::EINVAL, "cannot shutdown"),
506        };
507
508        drop(state);
509        // No need to call `Pollee::invalidate` because `ConnectedStream::shutdown` will call
510        // `Pollee::notify`.
511        iface_to_poll.poll();
512
513        result
514    }
515
516    fn addr(&self) -> Result<SocketAddr> {
517        let state = self.read_updated_state();
518        let local_endpoint = match state.as_ref() {
519            State::Init(init_stream) => init_stream
520                .local_endpoint()
521                .unwrap_or(UNSPECIFIED_LOCAL_ENDPOINT),
522            State::Connecting(connecting_stream) => connecting_stream.local_endpoint(),
523            State::Listen(listen_stream) => listen_stream.local_endpoint(),
524            State::Connected(connected_stream) => connected_stream.local_endpoint(),
525        };
526        Ok(local_endpoint.into())
527    }
528
529    fn peer_addr(&self) -> Result<SocketAddr> {
530        let state = self.read_updated_state();
531        let remote_endpoint = match state.as_ref() {
532            State::Init(_) | State::Listen(_) => {
533                return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected")
534            }
535            State::Connecting(connecting_stream) => connecting_stream.remote_endpoint(),
536            State::Connected(connected_stream) => connected_stream.remote_endpoint(),
537        };
538        Ok(remote_endpoint.into())
539    }

541    fn sendmsg(
542        &self,
543        reader: &mut dyn MultiRead,
544        message_header: MessageHeader,
545        flags: SendRecvFlags,
546    ) -> Result<usize> {
547        // TODO: Deal with flags
548        if !flags.is_all_supported() {
549            warn!("unsupported flags: {:?}", flags);
550        }
551
552        let MessageHeader {
553            control_messages, ..
554        } = message_header;
555
556        // According to the Linux man pages, `EISCONN` _may_ be returned when the destination
557        // address is specified for a connection-mode socket. In practice, the destination address
558        // is simply ignored. We follow the same behavior as the Linux implementation to ignore it.
559
560        if !control_messages.is_empty() {
561            // TODO: Support sending control message
562            warn!("sending control message is not supported");
563        }
564
565        self.block_on(IoEvents::OUT, || self.try_send(reader, flags))
566
567        // TODO: Trigger `SIGPIPE` if the error code is `EPIPE` and `MSG_NOSIGNAL` is not specified
568    }

570    fn recvmsg(
571        &self,
572        writer: &mut dyn MultiWrite,
573        flags: SendRecvFlags,
574    ) -> Result<(usize, MessageHeader)> {
575        // TODO: Deal with flags
576        if !flags.is_all_supported() {
577            warn!("unsupported flags: {:?}", flags);
578        }
579
580        let (received_bytes, _) = self.block_on(IoEvents::IN, || self.try_recv(writer, flags))?;
581
582        // TODO: Receive control message
583
584        // According to <https://elixir.bootlin.com/linux/v6.0.9/source/net/ipv4/tcp.c#L2645>,
585        // peer address is ignored for connected socket.
586        let message_header = MessageHeader::new(None, Vec::new());
587
588        Ok((received_bytes, message_header))
589    }

591    fn get_option(&self, option: &mut dyn SocketOption) -> Result<()> {
592        sock_option_mut!(match option {
593            socket_errors @ SocketError => {
594                socket_errors.set(self.test_and_clear_error());
595                return Ok(());
596            }
597            _ => (),
598        });
599
600        let state = self.read_updated_state();
601        let options = self.options.read();
602
603        // Deal with socket-level options
604        match options.socket.get_option(option, state.as_ref()) {
605            Err(err) if err.error() == Errno::ENOPROTOOPT => (),
606            res => return res,
607        }
608
609        // Deal with IP-level options
610        match options.ip.get_option(option) {
611            Err(err) if err.error() == Errno::ENOPROTOOPT => (),
612            res => return res,
613        }
614
615        // Deal with TCP-level options
616        // FIXME: Here we only return the previously set values, without actually
617        // asking the underlying sockets for the real, effective values.
618        sock_option_mut!(match option {
619            tcp_no_delay @ NoDelay => {
620                let no_delay = options.tcp.no_delay();
621                tcp_no_delay.set(no_delay);
622            }
623            tcp_maxseg @ MaxSegment => {
624                const DEFAULT_MAX_SEGMEMT: u32 = 536;
625                // For an unconnected socket,
626                // older Linux versions (e.g., v6.0) return
627                // the default MSS value defined above.
628                // However, newer Linux versions (e.g., v6.11)
629                // return the user-set MSS value if it is set.
630                // Here, we adopt the behavior of the latest Linux versions.
631                let maxseg = options.tcp.maxseg();
632                if maxseg == 0 {
633                    tcp_maxseg.set(DEFAULT_MAX_SEGMEMT);
634                } else {
635                    tcp_maxseg.set(maxseg);
636                }
637            }
638            tcp_keep_idle @ KeepIdle => {
639                let keep_idle = options.tcp.keep_idle();
640                tcp_keep_idle.set(keep_idle);
641            }
642            tcp_syn_cnt @ SynCnt => {
643                let syn_cnt = options.tcp.syn_cnt();
644                tcp_syn_cnt.set(syn_cnt);
645            }
646            tcp_defer_accept @ DeferAccept => {
647                let defer_accept = options.tcp.defer_accept();
648                let seconds = defer_accept.to_secs();
649                tcp_defer_accept.set(seconds);
650            }
651            tcp_window_clamp @ WindowClamp => {
652                let window_clamp = options.tcp.window_clamp();
653                tcp_window_clamp.set(window_clamp);
654            }
655            tcp_congestion @ Congestion => {
656                let congestion = options.tcp.congestion();
657                tcp_congestion.set(congestion);
658            }
659            tcp_user_timeout @ UserTimeout => {
660                let user_timeout = options.tcp.user_timeout();
661                tcp_user_timeout.set(user_timeout);
662            }
663            tcp_inq @ Inq => {
664                let inq = options.tcp.receive_inq();
665                tcp_inq.set(inq);
666            }
667            _ => return_errno_with_message!(
668                Errno::ENOPROTOOPT,
669                "the socket option to get is unknown"
670            ),
671        });
672
673        Ok(())
674    }

676    fn set_option(&self, option: &dyn SocketOption) -> Result<()> {
677        let mut state = self.write_updated_state();
678        let mut options = self.options.write();
679
680        // Deal with socket-level options
681        let need_iface_poll = match options.socket.set_option(option, state.as_ref()) {
682            Err(err) if err.error() == Errno::ENOPROTOOPT => {
683                // Deal with IP-level options
684                match options.ip.set_option(option, state.as_ref()) {
685                    Err(err) if err.error() == Errno::ENOPROTOOPT => {
686                        // Deal with TCP-level options
687                        do_tcp_setsockopt(option, &mut options, state.as_mut())?
688                    }
689                    Err(err) => return Err(err),
690                    Ok(need_iface_poll) => need_iface_poll,
691                }
692            }
693            Err(err) => return Err(err),
694            Ok(need_iface_poll) => need_iface_poll,
695        };
696
697        let iface_to_poll = need_iface_poll.then(|| state.iface().cloned()).flatten();
698
699        drop(state);
700        drop(options);
701
702        if let Some(iface) = iface_to_poll {
703            iface.poll();
704        }
705
706        Ok(())
707    }
708
709    fn pseudo_path(&self) -> &Path {
710        &self.pseudo_path
711    }
712}

714fn do_tcp_setsockopt(
715    option: &dyn SocketOption,
716    options: &mut OptionSet,
717    state: &mut State,
718) -> Result<NeedIfacePoll> {
719    sock_option_ref!(match option {
720        tcp_no_delay @ NoDelay => {
721            let no_delay = tcp_no_delay.get().unwrap();
722            options.tcp.set_no_delay(*no_delay);
723            state.set_raw_option(|raw_socket: &dyn RawTcpSetOption| {
724                raw_socket.set_nagle_enabled(!no_delay)
725            });
726        }
727        tcp_maxseg @ MaxSegment => {
728            const MIN_MAXSEG: u32 = 536;
729            const MAX_MAXSEG: u32 = 65535;
730
731            let maxseg = tcp_maxseg.get().unwrap();
732            if *maxseg < MIN_MAXSEG || *maxseg > MAX_MAXSEG {
733                return_errno_with_message!(
734                    Errno::EINVAL,
735                    "the maximum segment size is out of bounds"
736                );
737            }
738            options.tcp.set_maxseg(*maxseg);
739        }
740        tcp_keep_idle @ KeepIdle => {
741            const MIN_KEEP_IDLE: u32 = 1;
742            const MAX_KEEP_IDLE: u32 = 32767;
743
744            let keepidle = tcp_keep_idle.get().unwrap();
745            if *keepidle < MIN_KEEP_IDLE || *keepidle > MAX_KEEP_IDLE {
746                return_errno_with_message!(Errno::EINVAL, "the keep idle time is out of bounds");
747            }
748            options.tcp.set_keep_idle(*keepidle);
749
750            // TODO: Track when the socket becomes idle to actually support keep idle.
751        }
752        tcp_syn_cnt @ SynCnt => {
753            const MAX_TCP_SYN_CNT: u8 = 127;
754
755            let syncnt = tcp_syn_cnt.get().unwrap();
756            if *syncnt < 1 || *syncnt > MAX_TCP_SYN_CNT {
757                return_errno_with_message!(Errno::EINVAL, "the SYN count is out of bounds");
758            }
759            options.tcp.set_syn_cnt(*syncnt);
760        }
761        tcp_defer_accept @ DeferAccept => {
762            let mut seconds = *(tcp_defer_accept.get().unwrap());
763            if (seconds as i32) < 0 {
764                seconds = 0;
765            }
766            let retrans = Retrans::from_secs(seconds);
767            options.tcp.set_defer_accept(retrans);
768        }
769        tcp_window_clamp @ WindowClamp => {
770            let window_clamp = tcp_window_clamp.get().unwrap();
771            let half_recv_buf = options.socket.recv_buf() / 2;
772            if *window_clamp <= half_recv_buf {
773                options.tcp.set_window_clamp(half_recv_buf);
774            } else {
775                options.tcp.set_window_clamp(*window_clamp);
776            }
777        }
778        tcp_congestion @ Congestion => {
779            let congestion = tcp_congestion.get().unwrap();
780            options.tcp.set_congestion(*congestion);
781        }
782        tcp_user_timeout @ UserTimeout => {
783            let user_timeout = tcp_user_timeout.get().unwrap();
784            if (*user_timeout as i32) < 0 {
785                return_errno_with_message!(Errno::EINVAL, "the user timeout cannot be negative");
786            }
787            options.tcp.set_user_timeout(*user_timeout);
788        }
789        tcp_inq @ Inq => {
790            let inq = tcp_inq.get().unwrap();
791            options.tcp.set_receive_inq(*inq);
792        }
793        _ =>
794            return_errno_with_message!(Errno::ENOPROTOOPT, "the socket option to be set is unknown"),
795    });
796
797    Ok(NeedIfacePoll::FALSE)
798}

800impl State {
801    /// Calls `f` to set raw socket option.
802    ///
803    /// For listening sockets, socket options are inherited by new connections. However, they are
804    /// not updated for connections in the backlog queue.
805    fn set_raw_option<R>(&self, set_option: impl FnOnce(&dyn RawTcpSetOption) -> R) -> Option<R> {
806        match self {
807            State::Init(_) => None,
808            State::Connecting(connecting_stream) => {
809                Some(connecting_stream.set_raw_option(set_option))
810            }
811            State::Connected(connected_stream) => Some(connected_stream.set_raw_option(set_option)),
812            State::Listen(listen_stream) => Some(listen_stream.set_raw_option(set_option)),
813        }
814    }
815
816    fn iface(&self) -> Option<&Arc<Iface>> {
817        match self {
818            State::Init(_) => None,
819            State::Connecting(connecting_stream) => Some(connecting_stream.iface()),
820            State::Connected(connected_stream) => Some(connected_stream.iface()),
821            State::Listen(listen_stream) => Some(listen_stream.iface()),
822        }
823    }
824}

826impl GetSocketLevelOption for State {
827    fn is_listening(&self) -> bool {
828        matches!(self, Self::Listen(_))
829    }
830}

832impl SetSocketLevelOption for State {
833    fn set_reuse_addr(&self, reuse_addr: bool) {
834        let bound_port = match self {
835            State::Init(init_stream) => {
836                if let Some(bound_port) = init_stream.bound_port() {
837                    bound_port
838                } else {
839                    return;
840                }
841            }
842            State::Connecting(connecting_stream) => connecting_stream.bound_port(),
843            State::Connected(connected_stream) => connected_stream.bound_port(),
844            // Setting a listening address as reusable has no effect,
845            // since no other sockets can bind to a listening port.
846            State::Listen(_) => return,
847        };
848
849        bound_port.set_can_reuse(reuse_addr);
850    }
851
852    fn set_keep_alive(&self, keep_alive: bool) -> NeedIfacePoll {
853        let interval = if keep_alive {
854            Some(KEEPALIVE_INTERVAL)
855        } else {
856            None
857        };
858
859        let set_keepalive = |raw_socket: &dyn RawTcpSetOption| raw_socket.set_keep_alive(interval);
860
861        self.set_raw_option(set_keepalive)
862            .unwrap_or(NeedIfacePoll::FALSE)
863    }
864}

875impl Drop for StreamSocket {
876    fn drop(&mut self) {
877        let state = self.state.get_mut().take();
878
879        let conn = match state {
880            State::Init(_) => return,
881            State::Connecting(connecting_stream) => connecting_stream.into_connection(),
882            State::Connected(connected_stream) => connected_stream.into_connection(),
883            State::Listen(listen_stream) => {
884                let listener = listen_stream.into_listener();
885                listener.close();
886                listener.iface().poll();
887                return;
888            }
889        };
890
891        let linger = self.options.get_mut().socket.linger();
892        close_and_linger(conn, linger, &self.pollee);
893    }
894}
