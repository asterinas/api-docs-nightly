14pub(super) fn get_iface_to_bind(ip_addr: &IpAddress) -> Option<Arc<Iface>> {
15    let IpAddress::Ipv4(ipv4_addr) = ip_addr;
16    iter_all_ifaces()
17        .find(|iface| {
18            if let Some(iface_ipv4_addr) = iface.ipv4_addr() {
19                iface_ipv4_addr == *ipv4_addr
20            } else {
21                false
22            }
23        })
24        .map(Clone::clone)
25}

27/// Get a suitable iface to deal with sendto/connect request if the socket is not bound to an iface.
28/// If the remote address is the same as that of some iface, we will use the iface.
29/// Otherwise, we will use a default interface.
30fn get_ephemeral_iface(remote_ip_addr: &IpAddress) -> Arc<Iface> {
31    let IpAddress::Ipv4(remote_ipv4_addr) = remote_ip_addr;
32    if let Some(iface) = iter_all_ifaces().find(|iface| {
33        if let Some(iface_ipv4_addr) = iface.ipv4_addr() {
34            iface_ipv4_addr == *remote_ipv4_addr
35        } else {
36            false
37        }
38    }) {
39        return iface.clone();
40    }
41
42    // FIXME: Instead of hardcoding the rules here, we should choose the
43    // default interface according to the routing table.
44    if let Some(virtio_iface) = virtio_iface() {
45        virtio_iface.clone()
46    } else {
47        loopback_iface().clone()
48    }
49}

51pub(super) fn bind_port(endpoint: &IpEndpoint, can_reuse: bool) -> Result<BoundPort> {
52    let iface = match get_iface_to_bind(&endpoint.addr) {
53        Some(iface) => iface,
54        None => {
55            return_errno_with_message!(
56                Errno::EADDRNOTAVAIL,
57                "the address is not available from the local machine"
58            );
59        }
60    };
61
62    let bind_port_config = BindPortConfig::new(endpoint.port, can_reuse);
63
64    Ok(iface.bind(bind_port_config)?)
65}

67impl From<BindError> for Error {
68    fn from(value: BindError) -> Self {
69        match value {
70            BindError::Exhausted => {
71                Error::with_message(Errno::EAGAIN, "no ephemeral port is available")
72            }
73            BindError::InUse => {
74                Error::with_message(Errno::EADDRINUSE, "the address is already in use")
75            }
76        }
77    }
78}

80pub(super) fn get_ephemeral_endpoint(remote_endpoint: &IpEndpoint) -> IpEndpoint {
81    let iface = get_ephemeral_iface(&remote_endpoint.addr);
82    let ip_addr = iface.ipv4_addr().unwrap();
83    IpEndpoint::new(IpAddress::Ipv4(ip_addr), 0)
84}
