3//! This module defines vsock sockets.
4//!
5//! The vsock address family facilitates communication between virtual machines and the host they
6//! are running on. This address family is used by guest agents and hypervisor services that need a
7//! communications channel that is independent of virtual machine network configuration.
8//!
9//! The implementation is organized into three layers:
10//! - The [_device layer_](`aster_virtio::device::socket`) provides the basic packet transmit and
11//!   receive primitives.
12//! - The [_transport layer_](`self::transport`) implements protocol logic such as connection and
13//!   listener management.
14//! - The [_socket layer_](`self::stream`) builds the Linux-compatible socket interface used by
15//!   userspace-facing system calls.
16//!
17
18mod addr;
19mod stream;
20mod transport;
21
22pub use addr::VsockSocketAddr;
23pub use stream::VsockStreamSocket;
24
25pub(in crate::net) fn init() {
26    transport::init();
27}
