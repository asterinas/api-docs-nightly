31pub struct VsockStreamSocket {
32    state: Mutex<Takeable<State>>,
33    is_nonblocking: AtomicBool,
34    // Note that for vsock, all pollee notifications and invalidations live in the transport module
35    // (e.g., `super::transport`) rather than in this module.
36    pollee: Pollee,
37    pseudo_path: Path,
38}
39
40enum State {
41    Init(InitStream),
42    Connecting(ConnectingStream),
43    Connected(ConnectedStream),
44    Listen(ListenStream),
45}
46
47impl VsockStreamSocket {
48    pub fn new(is_nonblocking: bool) -> Result<Arc<Self>> {
49        Ok(Arc::new(Self {
50            state: Mutex::new(Takeable::new(State::Init(InitStream::new()))),
51            is_nonblocking: AtomicBool::new(is_nonblocking),
52            pollee: Pollee::new(),
53            pseudo_path: SockFs::new_path(),
54        }))
55    }
56
57    fn start_connect(&self, remote_addr: VsockSocketAddr) -> Option<Result<()>> {
58        let mut state = self.lock_updated_state();
59
60        state.borrow_result(|owned_state| {
61            let init_stream = match owned_state {
62                State::Init(init_stream) => init_stream,
63                State::Connecting(_) if self.is_nonblocking() => {
64                    return (
65                        owned_state,
66                        Some(Err(Error::with_message(
67                            Errno::EALREADY,
68                            "the socket is connecting",
69                        ))),
70                    );
71                }
72                State::Connecting(_) => {
73                    return (owned_state, None);
74                }
75                State::Connected(mut connected_stream) => {
76                    let result = connected_stream.finish_last_connect();
77                    return (State::Connected(connected_stream), Some(result));
78                }
79                State::Listen(_) => {
80                    return (
81                        owned_state,
82                        Some(Err(Error::with_message(
83                            Errno::EINVAL,
84                            "the socket is listening",
85                        ))),
86                    );
87                }
88            };
89
90            if !init_stream.is_connect_done() {
91                return (
92                    State::Init(init_stream),
93                    Some(Err(Error::with_message(
94                        Errno::EALREADY,
95                        "a previous connection attempt exists",
96                    ))),
97                );
98            }
99
100            match init_stream.connect(remote_addr, &self.pollee) {
101                Ok(connecting_stream) if self.is_nonblocking() => (
102                    State::Connecting(connecting_stream),
103                    Some(Err(Error::with_message(
104                        Errno::EINPROGRESS,
105                        "the socket is connecting",
106                    ))),
107                ),
108                Ok(connecting_stream) => (State::Connecting(connecting_stream), None),
109                Err((error, init_stream)) => (State::Init(init_stream), Some(Err(error))),
110            }
111        })
112
113        // The pollee should have already been invalidated in the transport module.
114    }
115
116    fn check_connect(&self) -> Result<()> {
117        let mut state = self.lock_updated_state();
118
119        match state.as_mut() {
120            State::Init(init_stream) => {
121                if let Some(error) = init_stream.test_and_clear_error(&self.pollee) {
122                    return Err(error);
123                }
124                return_errno_with_message!(
125                    Errno::ECONNABORTED,
126                    "the error code for the connection failure is not available"
127                );
128            }
129            State::Connecting(_) => {
130                return_errno_with_message!(Errno::EAGAIN, "the socket is connecting")
131            }
132            State::Connected(connected_stream) => connected_stream.finish_last_connect(),
133            State::Listen(_) => {
134                return_errno_with_message!(Errno::EISCONN, "the socket is listening")
135            }
136        }
137    }
138
139    fn try_accept(&self) -> Result<(Arc<dyn FileLike>, SocketAddr)> {
140        let state = self.lock_updated_state();
141        let State::Listen(listen_stream) = state.as_ref() else {
142            return_errno_with_message!(Errno::EINVAL, "the socket is not listening");
143        };
144
145        let connected = listen_stream.try_accept()?;
146
147        let peer_addr = connected.remote_addr().into();
148        let pollee = connected.pollee().clone();
149
150        let accepted = Arc::new(Self {
151            state: Mutex::new(Takeable::new(State::Connected(connected))),
152            is_nonblocking: AtomicBool::new(false),
153            pollee,
154            pseudo_path: SockFs::new_path(),
155        });
156
157        Ok((accepted, peer_addr))
158    }
159
160    fn try_send(&self, reader: &mut dyn MultiRead, flags: SendRecvFlags) -> Result<usize> {
161        let mut state = self.lock_updated_state();
162        let State::Connected(connected_stream) = state.as_mut() else {
163            return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected");
164        };
165
166        connected_stream.try_send(reader, flags)
167    }
168
169    fn try_recv(&self, writer: &mut dyn MultiWrite, flags: SendRecvFlags) -> Result<usize> {
170        let mut state = self.lock_updated_state();
171        let State::Connected(connected_stream) = state.as_mut() else {
172            return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected");
173        };
174
175        connected_stream.try_recv(writer, flags)
176    }
177
178    fn test_and_clear_error(&self) -> Option<Error> {
179        let mut state = self.lock_updated_state();
180
181        match state.as_mut() {
182            State::Init(init_stream) => init_stream.test_and_clear_error(&self.pollee),
183            State::Connecting(_) => None,
184            State::Connected(connected_stream) => connected_stream.test_and_clear_error(),
185            State::Listen(_) => None,
186        }
187    }
188
189    fn check_io_events(&self) -> IoEvents {
190        let state = self.lock_updated_state();
191
192        match state.as_ref() {
193            State::Init(init_stream) => init_stream.check_io_events(),
194            State::Connecting(connecting_stream) => connecting_stream.check_io_events(),
195            State::Connected(connected_stream) => connected_stream.check_io_events(),
196            State::Listen(listen_stream) => listen_stream.check_io_events(),
197        }
198    }
199
200    /// Locks the state and updates it if needed.
201    ///
202    /// If the current state is [`State::Connecting`] and the connect attempt has finished, this
203    /// method transitions the state to the latest one before returning the lock guard.
204    ///
205    /// Callers should always use this method instead of locking the state directly.
206    fn lock_updated_state(&self) -> MutexGuard<'_, Takeable<State>> {
207        let mut state = self.state.lock();
208
209        let State::Connecting(connecting_stream) = state.as_ref() else {
210            return state;
211        };
212        if !connecting_stream.has_result() {
213            return state;
214        }
215
216        state.borrow(|owned_state| {
217            let State::Connecting(connection_stream) = owned_state else {
218                unreachable!();
219            };
220            match connection_stream.into_result() {
221                ConnResult::Connecting(connection_stream) => State::Connecting(connection_stream),
222                ConnResult::Connected(connected_stream) => State::Connected(connected_stream),
223                ConnResult::Failed(init_stream) => State::Init(init_stream),
224            }
225        });
226
227        state
228    }
229}
230
231impl Pollable for VsockStreamSocket {
232    fn poll(&self, mask: IoEvents, poller: Option<&mut PollHandle>) -> IoEvents {
233        self.pollee
234            .poll_with(mask, poller, || self.check_io_events())
235    }
236}

238impl SocketPrivate for VsockStreamSocket {
239    fn is_nonblocking(&self) -> bool {
240        self.is_nonblocking.load(Ordering::Relaxed)
241    }
242
243    fn set_nonblocking(&self, nonblocking: bool) {
244        self.is_nonblocking.store(nonblocking, Ordering::Relaxed);
245    }
246}
247
248impl Socket for VsockStreamSocket {
249    fn bind(&self, socket_addr: SocketAddr) -> Result<()> {
250        let addr = VsockSocketAddr::try_from(socket_addr)?;
251
252        let mut state = self.lock_updated_state();
253        let State::Init(init_stream) = state.as_mut() else {
254            return_errno_with_message!(Errno::EINVAL, "the socket is already bound to an address");
255        };
256
257        init_stream.bind(addr)
258    }
259
260    fn connect(&self, socket_addr: SocketAddr) -> Result<()> {
261        let remote_addr = VsockSocketAddr::try_from(socket_addr)?;
262
263        if let Some(result) = self.start_connect(remote_addr) {
264            return result;
265        }
266
267        // FIXME: Linux cancels pending packets and aborts the connection if a blocking `connect()`
268        // is interrupted by a signal. Here, we follow the behavior of IP sockets by continuing to
269        // establish the connection in the background.
270        self.wait_events(IoEvents::OUT, None, || self.check_connect())
271    }
272
273    fn listen(&self, backlog: usize) -> Result<()> {
274        let mut state = self.lock_updated_state();
275
276        state.borrow_result(|owned_state| {
277            let init_stream = match owned_state {
278                State::Init(init_stream) => init_stream,
279                State::Listen(ref listen_stream) => {
280                    listen_stream.set_backlog(backlog);
281                    return (owned_state, Ok(()));
282                }
283                State::Connecting(_) | State::Connected(_) => {
284                    return (
285                        owned_state,
286                        Err(Error::with_message(
287                            Errno::EINVAL,
288                            "the socket is already connected",
289                        )),
290                    );
291                }
292            };
293
294            match init_stream.listen(backlog, &self.pollee) {
295                Ok(listen_stream) => (State::Listen(listen_stream), Ok(())),
296                Err((error, init_stream)) => (State::Init(init_stream), Err(error)),
297            }
298        })
299
300        // The pollee should have already been invalidated in the transport module.
301    }
302
303    fn accept(&self) -> Result<(Arc<dyn FileLike>, SocketAddr)> {
304        self.block_on(IoEvents::IN, || self.try_accept())
305    }
306
307    fn shutdown(&self, cmd: SockShutdownCmd) -> Result<()> {
308        let mut state = self.lock_updated_state();
309
310        match state.as_mut() {
311            State::Init(init_stream) => init_stream.shutdown(cmd),
312            State::Connecting(_) => {
313                return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected")
314            }
315            State::Connected(connected_stream) => connected_stream.shutdown(cmd),
316            State::Listen(_) => {
317                return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected")
318            }
319        }
320    }
321
322    fn addr(&self) -> Result<SocketAddr> {
323        let state = self.lock_updated_state();
324
325        let local_addr = match state.as_ref() {
326            State::Init(init_stream) => init_stream.local_addr(),
327            State::Connecting(connecting_stream) => Some(connecting_stream.local_addr()),
328            State::Connected(connected_stream) => Some(connected_stream.local_addr()),
329            State::Listen(listen_stream) => Some(listen_stream.local_addr()),
330        };
331
332        Ok(local_addr.unwrap_or(UNSPECIFIED_VSOCK_ADDR).into())
333    }
334
335    fn peer_addr(&self) -> Result<SocketAddr> {
336        let state = self.lock_updated_state();
337        let State::Connected(connected_stream) = state.as_ref() else {
338            return_errno_with_message!(Errno::ENOTCONN, "the socket is not connected");
339        };
340
341        Ok(connected_stream.remote_addr().into())
342    }

344    // TODO: Support setting socket options

346    fn get_option(&self, option: &mut dyn SocketOption) -> Result<()> {
347        sock_option_mut!(match option {
348            socket_errors @ SocketError => {
349                socket_errors.set(self.test_and_clear_error());
350                return Ok(());
351            }
352            _ => {}
353        });
354
355        // TODO: Support getting other socket options
356        return_errno_with_message!(Errno::EOPNOTSUPP, "the socket option to be get is unknown");
357    }
358
359    fn sendmsg(
360        &self,
361        reader: &mut dyn MultiRead,
362        message_header: MessageHeader,
363        flags: SendRecvFlags,
364    ) -> Result<usize> {
365        // TODO: Deal with flags
366        if !flags.is_all_supported() {
367            warn!("unsupported flags: {:?}", flags);
368        }
369
370        let MessageHeader {
371            control_messages,
372            addr,
373        } = message_header;
374
375        // According to the Linux man pages, `EISCONN` _may_ be returned when the destination
376        // address is specified for a connection-mode socket. In practice, `sendmsg` on vosck
377        // stream sockets will fail due to that. We follow the same behavior as the Linux
378        // implementation.
379        if addr.is_some() {
380            let state = self.lock_updated_state();
381            match state.as_ref() {
382                State::Init(_) | State::Listen(_) | State::Connecting(_) => {
383                    return_errno_with_message!(
384                        Errno::EOPNOTSUPP,
385                        "sending to a specific address is not allowed on vsock stream sockets"
386                    );
387                }
388                State::Connected(_) => {
389                    return_errno_with_message!(
390                        Errno::EISCONN,
391                        "sending to a specific address is not allowed on vsock stream sockets"
392                    );
393                }
394            }
395        }
396
397        if !control_messages.is_empty() {
398            // TODO: Support sending control message
399            warn!("sending control message is not supported");
400        }
401
402        self.block_on(IoEvents::OUT, || self.try_send(reader, flags))
403
404        // TODO: Trigger `SIGPIPE` if the error code is `EPIPE` and `MSG_NOSIGNAL` is not specified
405    }
406
407    fn recvmsg(
408        &self,
409        writer: &mut dyn MultiWrite,
410        flags: SendRecvFlags,
411    ) -> Result<(usize, MessageHeader)> {
412        // TODO: Deal with flags
413        if !flags.is_all_supported() {
414            warn!("unsupported flags: {:?}", flags);
415        }
416
417        let received_bytes = self.block_on(IoEvents::IN, || self.try_recv(writer, flags))?;
418
419        // TODO: Receive control message
420        let message_header = MessageHeader::new(None, Vec::new());
421
422        Ok((received_bytes, message_header))
423    }
424
425    fn pseudo_path(&self) -> &Path {
426        &self.pseudo_path
427    }
