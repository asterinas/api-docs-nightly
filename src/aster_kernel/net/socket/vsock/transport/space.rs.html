25// We currently support only one vsock device.
26// TODO: Add support for multiple vsock devices and the loopback vsock device.
27pub(super) struct VsockSpace {
28    device: Arc<SocketDevice>,
29    ports: SpinLock<PortTable>,
30    sockets: SpinLock<SocketTable, BottomHalfDisabled>,
31}
32
33struct SocketTable {
34    connections: BTreeMap<ConnId, Arc<ConnectionInner>>,
35    listeners: BTreeMap<u32, Arc<ListenerInner>>,
36}
37
38impl VsockSpace {
39    fn new(device: Arc<SocketDevice>) -> Self {
40        Self {
41            device,
42            ports: SpinLock::new(PortTable::new()),
43            sockets: SpinLock::new(SocketTable {
44                connections: BTreeMap::new(),
45                listeners: BTreeMap::new(),
46            }),
47        }
48    }
49
50    pub(super) fn device(&self) -> &SocketDevice {
51        &self.device
52    }
53
54    pub(super) fn guest_cid(&self) -> u64 {
55        self.device.guest_cid()
56    }
57
58    pub(super) fn lock_ports(&self) -> SpinLockGuard<'_, PortTable, PreemptDisabled> {
59        self.ports.lock()
60    }
61}

63// Connection and listener management.
64impl VsockSpace {
65    pub(super) fn new_connection(
66        &self,
67        bound_port: BoundPort,
68        remote_addr: VsockSocketAddr,
69        pollee: &Pollee,
70    ) -> core::result::Result<Connection, (Error, BoundPort)> {
71        use alloc::collections::btree_map::Entry;
72
73        let mut sockets = self.sockets.lock();
74
75        // Note that we should query the guest CID (part of `from_port_and_remote`) after locking
76        // `sockets` to avoid race conditions with `process_transport_event`.
77        let conn_id = ConnId::from_port_and_remote(&bound_port, remote_addr);
78        let Entry::Vacant(entry) = sockets.connections.entry(conn_id) else {
79            return Err((
80                Error::with_message(Errno::EADDRINUSE, "the vsock connection already exists"),
81                bound_port,
82            ));
83        };
84
85        let inner = ConnectionInner::new_connecting(bound_port, &conn_id, pollee.clone());
86        entry.insert(inner.clone());
87
88        Ok(Connection::new(inner))
89    }
90
91    /// Removes a connection.
92    ///
93    /// This method should not be called with the socket state lock held. We need to lock the
94    /// sockets before locking the socket state.
95    pub(super) fn remove_connection(&self, connection: &Arc<ConnectionInner>) {
96        use alloc::collections::btree_map::Entry;
97
98        let mut sockets = self.sockets.lock();
99
100        let conn_id = connection.conn_id();
101
102        // The following early returns may reach due to some race conditions between user
103        // operations and incoming packets. But we can safely ignore this case if it happens.
104        let Entry::Occupied(occupied) = sockets.connections.entry(conn_id) else {
105            return;
106        };
107        if !Arc::ptr_eq(connection, occupied.get()) {
108            return;
109        }
110
111        occupied.remove();
112    }
113
114    pub(super) fn new_listener(
115        &self,
116        bound_port: BoundPort,
117        backlog: usize,
118        pollee: &Pollee,
119    ) -> core::result::Result<Listener, (Error, BoundPort)> {
120        use alloc::collections::btree_map::Entry;
121
122        let mut sockets = self.sockets.lock();
123
124        let port = bound_port.port();
125        let Entry::Vacant(entry) = sockets.listeners.entry(port) else {
126            return Err((
127                Error::with_message(Errno::EADDRINUSE, "the vsock listener already exists"),
128                bound_port,
129            ));
130        };
131
132        let inner = ListenerInner::new(bound_port, backlog, pollee.clone());
133        entry.insert(inner.clone());
134
135        Ok(Listener::new(inner))
136    }
137
138    /// Removes a listener.
139    ///
140    /// This method should not be called with the socket state lock held. We need to lock the
141    /// sockets before locking the socket state.
142    pub(super) fn remove_listener(&self, listener: &Arc<ListenerInner>) {
143        use alloc::collections::btree_map::Entry;
144
145        let mut sockets = self.sockets.lock();
146
147        let port = listener.bound_port().port();
148        let removed = sockets.listeners.remove(&port);
149        debug_assert!(
150            removed
151                .as_ref()
152                .is_some_and(|removed| Arc::ptr_eq(removed, listener))
153        );
154
155        let connections = listener.take_incoming_on_removal();
156        for connection in connections.into_iter() {
157            let conn_id = connection.conn_id();
158
159            // The following `continue`s may reach due to some race conditions between user
160            // operations and incoming packets. But we can safely ignore this case if it happens.
161            let Entry::Occupied(occupied) = sockets.connections.entry(conn_id) else {
162                continue;
163            };
164            if !Arc::ptr_eq(&connection, occupied.get()) {
165                continue;
166            }
167
168            occupied.remove();
169
170            connection.active_rst();
171            // No need to notify the pollee since the connection isn't even accepted.
172        }
173    }
174}

176// RX packet and transport event processing.
177impl VsockSpace {
178    pub(super) fn process_rx(&self) {
179        // Lock order: device RX -> sockets -> socket state -> device TX
180
181        let mut rx = self.device.lock_rx();
182        let mut sockets = self.sockets.lock();
183
184        while let Some(packet) = rx.recv() {
185            self.process_rx_packet(&mut sockets, packet);
186        }
187    }
188
189    fn process_rx_packet(&self, sockets: &mut SocketTable, packet: RxPacket) {
190        use alloc::collections::btree_map::Entry;
191
192        let header = packet.header();
193
194        let conn_id = ConnId::from_incoming_header(&header);
195        let entry = sockets.connections.entry(conn_id);
196
197        match entry {
198            Entry::Vacant(vacant) => {
199                self.process_rx_with_listener(&sockets.listeners, vacant, &header, packet)
200            }
201            Entry::Occupied(occupied) => self.process_rx_with_connection(occupied, &header, packet),
202        }
203    }
204
205    fn process_rx_with_listener(
206        &self,
207        listeners: &BTreeMap<u32, Arc<ListenerInner>>,
208        vacant_conn: alloc::collections::btree_map::VacantEntry<'_, ConnId, Arc<ConnectionInner>>,
209        header: &VirtioVsockHdr,
210        packet: RxPacket,
211    ) {
212        if header.op() != Some(VirtioVsockOp::Request)
213            || !self.validate_rx_header(VirtioVsockOp::Request, header, &packet)
214        {
215            self.send_raw_rst(header);
216            return;
217        }
218
219        let dst_port = header.dst_port;
220        let listener = if let Some(listener) = listeners.get(&dst_port)
221            && !listener.is_full()
222        {
223            listener
224        } else {
225            self.send_raw_rst(header);
226            return;
227        };
228
229        let bound_port = BoundPort::new_shared(listener.bound_port());
230        let conn_id = vacant_conn.key();
231
232        let inner = ConnectionInner::new_connected(bound_port, conn_id, header);
233        vacant_conn.insert(inner.clone());
234
235        listener.push_incoming(inner.clone());
236    }
237
238    fn process_rx_with_connection(
239        &self,
240        occupied_conn: alloc::collections::btree_map::OccupiedEntry<
241            '_,
242            ConnId,
243            Arc<ConnectionInner>,
244        >,
245        header: &VirtioVsockHdr,
246        packet: RxPacket,
247    ) {
248        let op = if let Some(op) = header.op()
249            && self.validate_rx_header(op, header, &packet)
250        {
251            op
252        } else {
253            Self::reset_removed_connection(occupied_conn.remove());
254            return;
255        };
256
257        let connection = occupied_conn.get();
258
259        let should_remove = match op {
260            VirtioVsockOp::Request => {
261                connection.active_rst();
262                true
263            }
264            VirtioVsockOp::Response => connection.on_response(header).is_err(),
265            VirtioVsockOp::Rst => {
266                connection.on_rst();
267                true
268            }
269            VirtioVsockOp::Shutdown => connection.on_shutdown(header),
270            VirtioVsockOp::Rw => connection.on_rw(header, packet).is_err(),
271            VirtioVsockOp::CreditUpdate => connection.on_credit_update(header).is_err(),
272            VirtioVsockOp::CreditRequest => connection.on_credit_request(header).is_err(),
273        };
274
275        if should_remove {
276            Self::notify_removed_connection(occupied_conn.remove());
277        }
278    }
279
280    fn send_raw_rst(&self, header: &VirtioVsockHdr) {
281        if header.op == VirtioVsockOp::Rst as u16 {
282            // Do not send an RST packet in response to an RST packet. Otherwise, we may loop.
283            return;
284        }
285
286        // We do not use `VirtioVsockHdr::new` here because we want to specify the `type_` field. It
287        // may not be `VirtioVsockType::Stream`.
288        let rst_header = VirtioVsockHdr {
289            src_cid: header.dst_cid,
290            dst_cid: header.src_cid,
291            src_port: header.dst_port,
292            dst_port: header.src_port,
293            len: 0,
294            type_: header.type_,
295            op: VirtioVsockOp::Rst as u16,
296            flags: 0,
297            buf_alloc: 0,
298            fwd_cnt: 0,
299        };
300        let _ = self.send_packet(&rst_header);
301    }
302
303    pub(super) fn process_transport_event(&self) {
304        // Lock order: sockets -> socket state
305
306        let mut sockets = self.sockets.lock();
307
308        // As stated in the specification, we only need to deal with the connections:
309        // "The driver shuts down established connections and the guest_cid configuration field is
310        // fetched again. Existing listen sockets remain but their CID is updated to reflect the
311        // current guest_cid."
312
313        let connections = core::mem::take(&mut sockets.connections);
314        for connection in connections.into_values() {
315            connection.on_rst();
316            Self::notify_removed_connection(connection);
317        }
318
319        // The reload of the guest CID is protectd by the `sockets` lock.
320        self.device.reload_guest_id();
321    }
322
323    pub(super) fn process_timer_events(&self, events: Vec<TimerEvent>) {
324        use alloc::collections::btree_map::Entry;
325
326        // Lock order: sockets -> socket state
327
328        let mut sockets = self.sockets.lock();
329
330        for event in events.into_iter() {
331            let Entry::Occupied(entry) = sockets.connections.entry(event.conn_id) else {
332                continue;
333            };
334
335            if !entry.get().on_timeout(event.generation) {
336                continue;
337            }
338
339            Self::notify_removed_connection(entry.remove());
340        }
341    }
342
343    fn reset_removed_connection(connection: Arc<ConnectionInner>) {
344        connection.active_rst();
345        Self::notify_removed_connection(connection);
346    }
347
348    fn notify_removed_connection(connection: Arc<ConnectionInner>) {
349        // A reset connection may still be in the listener's accept queue. This is a deliberate
350        // design choice, as we currently lack the means to efficiently locate the listener. This
351        // should be harmless to the user space because a connection can always be reset just after
352        // being accepted.
353        //
354        // FIXME: This may not be consistent with Linux behavior.
355
356        let pollee = connection.pollee().clone();
357        drop(connection);
358
359        // Notify the pollee after dropping the connection. This ensures the connection's reference
360        // count is one, allowing the socket layer to use the `Connection::into_result` method.
361        pollee
362            .notify(IoEvents::IN | IoEvents::OUT | IoEvents::RDHUP | IoEvents::HUP | IoEvents::ERR);
363    }
364
365    /// Sends a control packet which does not carry payload.
366    ///
367    /// This method may be called while holding the socket state lock.
368    //
369    // TODO: This method may fail if memory allocation fails. For now, we will ignore the error in
370    // most cases. If possible, we should find better ways to handle the error.
371    #[must_use]
372    pub(super) fn send_packet(&self, header: &VirtioVsockHdr) -> bool {
373        let Ok(builder) = TxPacket::new_builder() else {
374            warn!("failed to allocate vsock packet: {:?}", header);
375            return false;
376        };
377        let packet = builder.build(header);
378
379        // Lock order: socket state -> device TX
380
381        let mut tx = self.device.lock_tx();
382        match tx.try_send(packet) {
383            Ok(()) => (),
384            Err(pending) => {
385                // TODO: Ideally, we should limit the number of in-flight packets. This will prevent
386                // the peer endpoint from exhausting a large amount of memory by triggering too many
387                // packets but never processing them.
388                pending.push_pending(None);
389            }
390        }
391
392        true
393    }
394
395    fn validate_rx_header(
396        &self,
397        op: VirtioVsockOp,
398        header: &VirtioVsockHdr,
399        packet: &RxPacket,
400    ) -> bool {
401        if header.type_ != VirtioVsockType::Stream as u16 {
402            return false;
403        }
404
405        if header.dst_cid != self.guest_cid() {
406            return false;
407        }
408
409        let payload_len = packet.payload_len();
410        if payload_len != header.len as usize {
411            return false;
412        }
413
414        match op {
415            VirtioVsockOp::Request
416            | VirtioVsockOp::Response
417            | VirtioVsockOp::Rst
418            | VirtioVsockOp::CreditUpdate
419            | VirtioVsockOp::CreditRequest => payload_len == 0 && header.flags == 0,
420            VirtioVsockOp::Shutdown => {
421                payload_len == 0 && VirtioVsockShutdownFlags::from_bits(header.flags).is_some()
422            }
423            VirtioVsockOp::Rw => header.flags == 0,
424        }
425    }
426}

428static VSOCK_SPACE: Once<VsockSpace> = Once::new();
429
430pub(super) fn vsock_space() -> Result<&'static VsockSpace> {
431    VSOCK_SPACE
432        .get()
433        .ok_or_else(|| Error::with_message(Errno::ENODEV, "no virtio-vsock device is available"))
434}
435
436pub(super) fn init(device: Arc<SocketDevice>) {
437    VSOCK_SPACE.call_once(move || VsockSpace::new(device));
