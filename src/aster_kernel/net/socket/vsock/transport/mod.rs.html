3//! The virtio-vsock transport protocol.
4//!
5//! Built on top of the [_basic virtio-vsock device support_](`aster_virtio::device::socket`), this
6//! module manages connections and listeners, handles connection establishment, data transfer, and
7//! shutdown, and implements credit-based flow control plus I/O event checking and notification.
8//! The socket layer is expected to build on these APIs to provide the user-visible socket
9//! interface.
10//!
11//! For a quick start, bind to a port by creating a [`BoundPort`] instance.
12//!  - To connect to a remote address, use the [`BoundPort::connect`] method and get a
13//!    [`Connection`] instance. Data can be transmitted or received via the [`Connection::try_send`]
14//!    and [`Connection::try_recv`] methods.
15//!  - To listen to the local address, use the [`BoundPort::listen`] method and get a [`Listener`]
16//!    instance. Incoming connections can be accepted via the [`Listener::try_accept`] method.
17//!
18//! Drop the [`BoundPort`], [`Connection`], or [`Listener`] instance will shut down the underlying
19//! connection (if any) and release the resources.

34// Reference: <https://elixir.bootlin.com/linux/v6.16.8/source/net/vmw_vsock/af_vsock.c#L136>
35const DEFAULT_CONNECT_TIMEOUT: Duration = Duration::from_secs(2);
36// Reference: <https://elixir.bootlin.com/linux/v6.16.8/source/net/vmw_vsock/virtio_transport_common.c#L24>
37const DEFAULT_CLOSE_TIMEOUT: Duration = Duration::from_secs(8);

39const DEFAULT_RX_BUF_SIZE: usize = 256 * 1024;
40// Reference: <https://elixir.bootlin.com/linux/v6.16.8/source/net/vmw_vsock/af_vsock.c#L138>
41const DEFAULT_TX_BUF_SIZE: usize = 256 * 1024;
42// Reference: <https://elixir.bootlin.com/linux/v6.16.8/source/include/linux/socket.h#L298>
43const MAX_BACKLOG: usize = 4096;
44
45const CREDIT_UPDATE_THRESHOLD: u32 = (DEFAULT_RX_BUF_SIZE / 4) as u32;
46
47fn process_rx_callback() {
48    if let Ok(vsock_space) = space::vsock_space() {
49        vsock_space.process_rx();
50    }
51}
52
53fn process_event_callback() {
54    if let Ok(vsock_space) = space::vsock_space() {
55        vsock_space.process_transport_event();
56    }
57}
58
59/// Initializes the virtio-vsock transport when the default device is present.
60pub(super) fn init() {
61    use aster_virtio::device::socket::DEVICE_NAME;
62
63    let Some(device) = aster_virtio::device::socket::get_device(DEVICE_NAME) else {
64        return;
65    };
66
67    device.init_rx_callback(process_rx_callback);
68    device.init_event_callback(process_event_callback);
69    space::init(device);
70
71    timer::init();
72}
