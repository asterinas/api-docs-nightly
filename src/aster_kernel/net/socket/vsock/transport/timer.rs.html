3use alloc::{sync::Arc, vec::Vec};
4use core::sync::atomic::{AtomicU64, Ordering};
5
6use aster_softirq::{BottomHalfDisabled, Taskless};
7use ostd::sync::SpinLock;
8use spin::Once;
9
10use crate::net::socket::vsock::transport::{conn_id::ConnId, space::vsock_space};
11
12static NEXT_GENERATION: AtomicU64 = AtomicU64::new(0);
13
14/// Returns a unique identifier for a timer event.
15///
16/// This value wraps around on `u64` overflow, so uniqueness is only guaranteed until the counter
17/// overflows. In practice, that should take long enough to be acceptable here.
18pub(super) fn next_timer_generation() -> u64 {
19    NEXT_GENERATION.fetch_add(1, Ordering::Relaxed)
20}
21
22pub(super) struct TimerEvent {
23    pub(super) conn_id: ConnId,
24    pub(super) generation: u64,
25}
26
27static PENDING_EVENTS: SpinLock<Vec<TimerEvent>, BottomHalfDisabled> = SpinLock::new(Vec::new());
28
29static TASKLESS: Once<Arc<Taskless>> = Once::new();
30
31pub(super) fn push_timer_event(conn_id: ConnId, generation: u64) {
32    let event = TimerEvent {
33        conn_id,
34        generation,
35    };
36    PENDING_EVENTS.lock().push(event);
37
38    TASKLESS.get().unwrap().schedule();
39}
40
41fn process_pending_timer_events() {
42    let events = {
43        let mut pending = PENDING_EVENTS.lock();
44        core::mem::take(&mut *pending)
45    };
46
47    let vsock_space = vsock_space().unwrap();
48    vsock_space.process_timer_events(events);
49}
50
51pub(super) fn init() {
52    TASKLESS.call_once(|| Taskless::new(process_pending_timer_events));
53}
