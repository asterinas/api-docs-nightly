30    /// Common methods for sockets, but private to the network module.
31    ///
32    /// These are implementation details of sockets, so shouldn't be accessed outside the network
33    /// module. Therefore, the whole trait is sealed.
34    pub trait SocketPrivate: Pollable {
35        /// Returns whether the socket is in non-blocking mode.
36        fn is_nonblocking(&self) -> bool;
37
38        /// Sets whether the socket is in non-blocking mode.
39        fn set_nonblocking(&self, nonblocking: bool);
40
41        /// Blocks until some events occur to complete I/O operations.
42        ///
43        /// If the socket is in non-blocking mode and the I/O operations cannot be completed
44        /// immediately, this method will fail with [`EAGAIN`] instead of blocking.
45        ///
46        /// [`EAGAIN`]: crate::error::Errno::EAGAIN
47        #[track_caller]
48        fn block_on<F, R>(&self, events: IoEvents, mut try_op: F) -> Result<R>
49        where
50            Self: Sized,
51            F: FnMut() -> Result<R>,
52        {
53            if self.is_nonblocking() {
54                try_op()
55            } else {
56                self.wait_events(events, None, try_op)
57            }
58        }
59    }
60}

62/// Operations defined on a socket.
63pub trait Socket: private::SocketPrivate + Send + Sync {
64    /// Assigns the specified address to the socket.
65    fn bind(&self, _socket_addr: SocketAddr) -> Result<()> {
66        return_errno_with_message!(Errno::EOPNOTSUPP, "bind() is not supported");
67    }
68
69    /// Builds a connection for the given address
70    fn connect(&self, _socket_addr: SocketAddr) -> Result<()> {
71        return_errno_with_message!(Errno::EOPNOTSUPP, "connect() is not supported");
72    }
73
74    /// Listens for connections on the socket.
75    fn listen(&self, _backlog: usize) -> Result<()> {
76        return_errno_with_message!(Errno::EOPNOTSUPP, "listen() is not supported");
77    }
78
79    /// Accepts a connection on the socket.
80    fn accept(&self) -> Result<(Arc<dyn FileLike>, SocketAddr)> {
81        return_errno_with_message!(Errno::EOPNOTSUPP, "accept() is not supported");
82    }
83
84    /// Shuts down part of a full-duplex connection.
85    fn shutdown(&self, _cmd: SockShutdownCmd) -> Result<()> {
86        return_errno_with_message!(Errno::EOPNOTSUPP, "shutdown() is not supported");
87    }
88
89    /// Gets the address of this socket.
90    fn addr(&self) -> Result<SocketAddr> {
91        return_errno_with_message!(Errno::EOPNOTSUPP, "getsockname() is not supported");
92    }
93
94    /// Gets the address of the peer socket.
95    fn peer_addr(&self) -> Result<SocketAddr> {
96        return_errno_with_message!(Errno::EOPNOTSUPP, "getpeername() is not supported");
97    }
98
99    /// Gets options on the socket.
100    ///
101    /// If the method succeeds, the result will be stored in the `option` parameter.
102    fn get_option(&self, _option: &mut dyn SocketOption) -> Result<()> {
103        return_errno_with_message!(Errno::EOPNOTSUPP, "getsockopt() is not supported");
104    }
105
106    /// Sets options on the socket.
107    fn set_option(&self, _option: &dyn SocketOption) -> Result<()> {
108        return_errno_with_message!(Errno::EOPNOTSUPP, "setsockopt() is not supported");
109    }
110
111    /// Sends a message on the socket.
112    fn sendmsg(
113        &self,
114        reader: &mut dyn MultiRead,
115        message_header: MessageHeader,
116        flags: SendRecvFlags,
117    ) -> Result<usize>;
118
119    /// Receives a message from the socket.
120    ///
121    /// If successful, the `io_vecs` buffer will be filled with the received content.
122    /// This method returns the length of the received message,
123    /// and the message header.
124    fn recvmsg(
125        &self,
126        writer: &mut dyn MultiWrite,
127        flags: SendRecvFlags,
128    ) -> Result<(usize, MessageHeader)>;
129
130    /// Returns a reference to the pseudo path associated with this socket.
131    fn pseudo_path(&self) -> &Path;
132}

134impl<T: Socket + 'static> FileLike for T {
135    fn read(&self, writer: &mut VmWriter) -> Result<usize> {
136        if !writer.has_avail() {
137            // Linux always returns `Ok(0)` in this case, so we follow it.
138            return Ok(0);
139        }
140
141        // TODO: Set correct flags
142        self.recvmsg(writer, SendRecvFlags::empty())
143            .map(|(len, _)| len)
144    }
145
146    fn write(&self, reader: &mut VmReader) -> Result<usize> {
147        // TODO: Set correct flags
148        self.sendmsg(
149            reader,
150            MessageHeader::new(None, Vec::new()),
151            SendRecvFlags::empty(),
152        )
153    }
154
155    fn status_flags(&self) -> StatusFlags {
156        // TODO: Support other flags (e.g., `O_ASYNC`)
157        if self.is_nonblocking() {
158            StatusFlags::O_NONBLOCK
159        } else {
160            StatusFlags::empty()
161        }
162    }
163
164    fn set_status_flags(&self, new_flags: StatusFlags) -> Result<()> {
165        // TODO: Support other flags (e.g., `O_ASYNC`)
166        if new_flags.contains(StatusFlags::O_NONBLOCK) {
167            self.set_nonblocking(true);
168        } else {
169            self.set_nonblocking(false);
170        }
171        Ok(())
172    }
173
174    fn as_socket(&self) -> Option<&dyn Socket> {
175        Some(self)
176    }
177
178    fn path(&self) -> &Path {
179        self.pseudo_path()
180    }
181
182    fn dump_proc_fdinfo(self: Arc<Self>, fd_flags: FdFlags) -> Box<dyn Display> {
183        struct FdInfo {
184            flags: u32,
185            ino: u64,
186        }
187
188        impl Display for FdInfo {
189            fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
190                writeln!(f, "pos:\t{}", 0)?;
191                writeln!(f, "flags:\t0{:o}", self.flags)?;
192                writeln!(f, "mnt_id:\t{}", SockFs::mount_node().id())?;
193                writeln!(f, "ino:\t{}", self.ino)
194            }
195        }
196
197        let mut flags = self.status_flags().bits() | self.access_mode() as u32;
198        if fd_flags.contains(FdFlags::CLOEXEC) {
199            flags |= CreationFlags::O_CLOEXEC.bits();
200        }
201
202        Box::new(FdInfo {
203            flags,
204            ino: self.pseudo_path().inode().ino(),
205        })
206    }
