3use crate::prelude::*;
4
5bitflags! {
6    /// Flags used for send/recv.
7    /// The definition is from https://elixir.bootlin.com/linux/v6.0.9/source/include/linux/socket.h
8    #[repr(C)]
9    #[derive(Pod)]
10    pub struct SendRecvFlags: i32 {
11        const MSG_OOB = 1;
12        const MSG_PEEK	= 2;
13        const MSG_DONTROUTE	= 4;
14        // const MSG_TRYHARD  =   4;       /* Synonym for MSG_DONTROUTE for DECnet */
15        const MSG_CTRUNC	= 8;
16        const MSG_PROBE	= 0x10;	/* Do not send. Only probe path f.e. for MTU */
17        const MSG_TRUNC	= 0x20;
18        const MSG_DONTWAIT	= 0x40;	/* Nonblocking io		 */
19        const MSG_EOR       = 0x80;	/* End of record */
20        const MSG_WAITALL	= 0x100;	/* Wait for a full request */
21        const MSG_FIN        = 0x200;
22        const MSG_SYN		= 0x400;
23        const MSG_CONFIRM	= 0x800;	/* Confirm path validity */
24        const MSG_RST		= 0x1000;
25        const MSG_ERRQUEUE	= 0x2000;	/* Fetch message from error queue */
26        const MSG_NOSIGNAL	= 0x4000;	/* Do not generate SIGPIPE */
27        const MSG_MORE	= 0x8000;	/* Sender will send more */
28        const MSG_WAITFORONE	= 0x10000;	/* recvmmsg(): block until 1+ packets avail */
29        const MSG_SENDPAGE_NOPOLICY = 0x10000; /* sendpage() internal : do no apply policy */
30        const MSG_SENDPAGE_NOTLAST = 0x20000; /* sendpage() internal : not the last page */
31        const MSG_BATCH	= 0x40000; /* sendmmsg(): more messages coming */
32        // const MSG_EOF         MSG_FIN
33        const MSG_NO_SHARED_FRAGS = 0x80000; /* sendpage() internal : page frags are not shared */
34        const MSG_SENDPAGE_DECRYPTED	= 0x100000; /* sendpage() internal : page may carry plain text and require encryption */
35    }
36}
37
38impl SendRecvFlags {
39    fn supported_flags() -> Self {
40        SendRecvFlags::empty()
41    }
42
43    pub fn is_all_supported(&self) -> bool {
44        let supported_flags = Self::supported_flags();
45        supported_flags.contains(*self)
46    }
47}
