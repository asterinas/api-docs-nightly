16pub trait SegmentBody: Sized + Clone + Copy {
17    // The actual message body should be `Self::CType`,
18    // but older versions of Linux use a legacy type (usually `CRtGenMsg` here).
19    // Additionally, some software, like iproute2, also uses this legacy type.
20    // Therefore, we need to handle both cases.
21    // Reference: <https://elixir.bootlin.com/linux/v6.13/source/net/core/rtnetlink.c#L2393>.
22    // FIXME: Verify whether the legacy type includes any types other than `CRtGenMsg`.
23    type CLegacyType: Pod = Self::CType;
24    type CType: Pod + From<Self::CLegacyType> + TryInto<Self> + From<Self>;
25
26    /// Reads the segment body from the `reader`.
27    ///
28    /// This method returns the body and the remaining length to be read from the `reader`.
29    fn read_from(
30        header: &CMsgSegHdr,
31        reader: &mut dyn MultiRead,
32    ) -> Result<ContinueRead<(Self, usize)>>
33    where
34        Error: From<<Self::CType as TryInto<Self>>::Error>,
35    {
36        let mut remaining_len = header.calc_payload_len_with_padding(reader)?;
37
38        // Read the body.
39        let (c_type, padding_len) = if remaining_len >= size_of::<Self::CType>() {
40            let c_type = reader.read_val_opt::<Self::CType>()?.unwrap();
41            remaining_len -= size_of_val(&c_type);
42
43            (c_type, Self::padding_len())
44        } else if remaining_len >= size_of::<Self::CLegacyType>() {
45            let legacy = reader.read_val_opt::<Self::CLegacyType>()?.unwrap();
46            remaining_len -= size_of_val(&legacy);
47
48            (Self::CType::from(legacy), Self::lecacy_padding_len())
49        } else {
50            reader.skip_some(remaining_len);
51            return Ok(ContinueRead::skipped_with_error(
52                Errno::EINVAL,
53                "the message length is too small",
54            ));
55        };
56
57        // Skip the padding bytes.
58        let padding_len = padding_len.min(remaining_len);
59        reader.skip_some(padding_len);
60        remaining_len -= padding_len;
61
62        match c_type.try_into() {
63            Ok(body) => Ok(ContinueRead::Parsed((body, remaining_len))),
64            Err(err) => {
65                reader.skip_some(remaining_len);
66                Ok(ContinueRead::SkippedErr(err.into()))
67            }
68        }
69    }
70
71    /// Writes the segment body to the `writer`.
72    fn write_to(&self, writer: &mut dyn MultiWrite) -> Result<()> {
73        // Write the body.
74        let c_body = Self::CType::from(*self);
75        writer.write_val_trunc(&c_body)?;
76
77        // Skip the padding bytes.
78        let padding_len = Self::padding_len();
79        writer.skip_some(padding_len);
80
81        Ok(())
82    }
83
84    fn padding_len() -> usize {
85        let payload_len = size_of::<Self::CType>();
86        payload_len.align_up(NLMSG_ALIGN) - payload_len
87    }
88
89    fn lecacy_padding_len() -> usize {
90        let payload_len = size_of::<Self::CLegacyType>();
91        payload_len.align_up(NLMSG_ALIGN) - payload_len
92    }
93}

95#[repr(u16)]
96#[derive(Debug, Clone, Copy, TryFromInt, PartialEq, Eq, PartialOrd, Ord)]
97#[expect(clippy::upper_case_acronyms)]
98pub enum CSegmentType {
99    // Standard netlink message types
100    NOOP = 1,
101    ERROR = 2,
102    DONE = 3,
103    OVERRUN = 4,
104
105    // protocol-level types
106    NEWLINK = 16,
107    DELLINK = 17,
108    GETLINK = 18,
109    SETLINK = 19,
110
111    NEWADDR = 20,
112    DELADDR = 21,
113    GETADDR = 22,
114
115    NEWROUTE = 24,
116    DELROUTE = 25,
117    GETROUTE = 26,
118    // TODO: The list is not exhaustive.
119}
