3use super::{SegmentBody, header::CMsgSegHdr};
4use crate::{
5    net::socket::netlink::message::{ContinueRead, attr::Attribute},
6    prelude::*,
7    util::{MultiRead, MultiWrite},
8};
9
10#[derive(Debug)]
11pub struct SegmentCommon<Body, Attr> {
12    header: CMsgSegHdr,
13    body: Body,
14    attrs: Vec<Attr>,
15}
16
17impl<Body, Attr> SegmentCommon<Body, Attr> {
18    pub const HEADER_LEN: usize = size_of::<CMsgSegHdr>();
19
20    pub fn header(&self) -> &CMsgSegHdr {
21        &self.header
22    }
23
24    pub fn header_mut(&mut self) -> &mut CMsgSegHdr {
25        &mut self.header
26    }
27
28    pub fn body(&self) -> &Body {
29        &self.body
30    }
31
32    pub fn attrs(&self) -> &Vec<Attr> {
33        &self.attrs
34    }
35}
36
37impl<Body: SegmentBody, Attr: Attribute> SegmentCommon<Body, Attr> {
38    pub const BODY_LEN: usize = size_of::<Body::CType>();
39
40    pub fn new(header: CMsgSegHdr, body: Body, attrs: Vec<Attr>) -> Self {
41        let mut res = Self {
42            header,
43            body,
44            attrs,
45        };
46        res.header.len = res.total_len() as u32;
47        res
48    }
49
50    pub fn read_from(header: &CMsgSegHdr, reader: &mut dyn MultiRead) -> Result<ContinueRead<Self>>
51    where
52        Error: From<<Body::CType as TryInto<Body>>::Error>,
53    {
54        let (body, remain_len) = match Body::read_from(header, reader)? {
55            ContinueRead::Parsed(parsed) => parsed,
56            ContinueRead::Skipped => return Ok(ContinueRead::Skipped),
57            ContinueRead::SkippedErr(err) => return Ok(ContinueRead::SkippedErr(err)),
58        };
59
60        let attrs = match Attr::read_all_from(reader, remain_len)? {
61            ContinueRead::Parsed(attrs) => attrs,
62            ContinueRead::Skipped => Vec::new(),
63            ContinueRead::SkippedErr(err) => return Ok(ContinueRead::SkippedErr(err)),
64        };
65
66        Ok(ContinueRead::Parsed(Self {
67            header: *header,
68            body,
69            attrs,
70        }))
71    }
72
73    pub fn write_to(&self, writer: &mut dyn MultiWrite) -> Result<()> {
74        writer.write_val_trunc(&self.header)?;
75
76        self.body.write_to(writer)?;
77        for attr in self.attrs.iter() {
78            attr.write_to(writer)?;
79        }
80
81        Ok(())
82    }
83
84    pub fn total_len(&self) -> usize {
85        Self::HEADER_LEN + Self::BODY_LEN + self.attrs_len()
86    }
87}
88
89impl<Body, Attr: Attribute> SegmentCommon<Body, Attr> {
90    pub fn attrs_len(&self) -> usize {
91        self.attrs
92            .iter()
93            .map(|attr| attr.total_len_with_padding())
94            .sum()
95    }
96}
