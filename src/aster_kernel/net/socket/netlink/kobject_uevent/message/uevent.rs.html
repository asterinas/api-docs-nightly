12/// `SysObj` action type.
13///
14/// Reference: <https://elixir.bootlin.com/linux/v6.14/source/include/linux/kobject.h#L53>.
15#[derive(Debug, Clone, Copy, PartialEq, Eq, TryFromInt)]
16#[repr(u8)]
17pub(super) enum SysObjAction {
18    /// Indicates the addition of a new `SysObj` to the system.
19    ///
20    /// Triggered when a device is discovered or registered.
21    Add = 0,
22
23    /// Signals the removal of a `SysObj` from the system.
24    ///
25    /// Typically occurs during device disconnection or deregistration.
26    Remove = 1,
27
28    /// Denotes a modification to the `SysObj`'s properties or state.
29    ///
30    /// Used for attribute changes that don't involve structural modifications.
31    Change = 2,
32
33    /// Represents hierarchical relocation of a `SysObj`.
34    ///
35    /// Occurs when a device moves within the device tree topology.
36    Move = 3,
37
38    /// Marks a device returning to operational status after being offlined.
39    ///
40    /// Common in hot-pluggable device scenarios.
41    Online = 4,
42
43    /// Indicates a device entering non-operational status.
44    ///
45    /// Typically precedes safe removal of hot-pluggable hardware.
46    Offline = 5,
47
48    /// Signifies successful driver-device binding.
49    ///
50    /// Occurs after successful driver probe sequence.
51    Bind = 6,
52
53    /// Indicates driver-device binding termination.
54    ///
55    /// Precedes driver unload or device removal.
56    Unbind = 7,
57}
58
59const SYSOBJ_ACTION_STRS: [&str; SysObjAction::Unbind as usize + 1] = [
60    "add", "remove", "change", "move", "online", "offline", "bind", "unbind",
61];
62
63impl FromStr for SysObjAction {
64    type Err = Error;
65
66    fn from_str(s: &str) -> Result<Self> {
67        let Some(index) = SYSOBJ_ACTION_STRS
68            .iter()
69            .position(|action_str| s == *action_str)
70        else {
71            return_errno_with_message!(Errno::EINVAL, "the string is not a valid `SysObj` action");
72        };
73
74        Ok(SysObjAction::try_from(index as u8).unwrap())
75    }
76}
77
78impl SysObjAction {
79    fn as_str(&self) -> &'static str {
80        SYSOBJ_ACTION_STRS[*self as usize]
81    }
82}

84/// Userspace event.
85pub(super) struct Uevent {
86    /// The `SysObj` action.
87    action: SysObjAction,
88    /// The absolute `SysObj` path under sysfs.
89    devpath: String,
90    /// The subsystem the event originates from
91    subsystem: String,
92    /// Other key-value arguments
93    envs: Vec<(String, String)>,
94    /// Sequence number.
95    seq_num: u64,
96}
97
98impl Uevent {
99    /// Creates a new uevent.
100    fn new(
101        action: SysObjAction,
102        devpath: String,
103        subsystem: String,
104        envs: Vec<(String, String)>,
105    ) -> Self {
106        debug_assert!(devpath.starts_with('/'));
107
108        let seq_num = SEQ_NUM_ALLOCATOR.fetch_add(1, Ordering::Relaxed);
109
110        Self {
111            action,
112            devpath,
113            subsystem,
114            envs,
115            seq_num,
116        }
117    }
118
119    /// Creates a new uevent from synthetic uevent.
120    pub(super) fn new_from_syn(
121        synth_uevent: SyntheticUevent,
122        devpath: String,
123        subsystem: String,
124        mut other_envs: Vec<(String, String)>,
125    ) -> Self {
126        let SyntheticUevent {
127            action,
128            uuid,
129            mut envs,
130        } = synth_uevent;
131
132        let uuid_key = "SYNTH_UUID".to_string();
133        if let Some(Uuid(uuid)) = uuid {
134            envs.push((uuid_key, uuid));
135        } else {
136            envs.push((uuid_key, "0".to_string()));
137        };
138
139        envs.append(&mut other_envs);
140
141        Self::new(action, devpath, subsystem, envs)
142    }
143}
144
145impl Display for Uevent {
146    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
147        let mut env_string = {
148            let len = self
149                .envs
150                .iter()
151                .map(|(key, value)| key.len() + value.len() + 2)
152                .sum();
153            String::with_capacity(len)
154        };
155
156        for (key, value) in self.envs.iter() {
157            env_string.push_str(key);
158            env_string.push('=');
159            env_string.push_str(value);
160            env_string.push('\0');
161        }
162
163        write!(
164            f,
165            "{}@{}\0ACTION={}\0DEVPATH={}\0SUBSYSTEM={}\0{}SEQNUM={}\0",
166            self.action.as_str(),
167            self.devpath,
168            self.action.as_str(),
169            self.devpath,
170            self.subsystem,
171            env_string,
172            self.seq_num
173        )
174    }
175}
176
177static SEQ_NUM_ALLOCATOR: AtomicU64 = AtomicU64::new(1);
