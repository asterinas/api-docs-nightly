3//! Handle address-related requests.
4
5use core::num::NonZeroU32;

7use super::util::finish_response;
8use crate::{
9    net::{
10        iface::{Iface, iter_all_ifaces},
11        socket::netlink::{
12            message::{CMsgSegHdr, CSegmentType, GetRequestFlags, SegHdrCommonFlags},
13            route::message::{
14                AddrAttr, AddrMessageFlags, AddrSegment, AddrSegmentBody, RtScope, RtnlSegment,
15            },
16        },
17    },
18    prelude::*,
19    util::net::CSocketAddrFamily,
20};

22pub(super) fn do_get_addr(request_segment: &AddrSegment) -> Result<Vec<RtnlSegment>> {
23    let dump_all = {
24        let flags = GetRequestFlags::from_bits_truncate(request_segment.header().flags);
25        flags.contains(GetRequestFlags::DUMP)
26    };
27    if !dump_all {
28        return_errno_with_message!(Errno::EOPNOTSUPP, "GETADDR only supports dump requests");
29    }
30
31    let mut response_segments: Vec<RtnlSegment> = iter_all_ifaces()
32        // GETADDR only supports dump mode, so we're going to report all addresses.
33        .filter_map(|iface| iface_to_new_addr(request_segment.header(), iface))
34        .map(RtnlSegment::NewAddr)
35        .collect();
36
37    finish_response(request_segment.header(), dump_all, &mut response_segments);
38
39    Ok(response_segments)
40}
41
42fn iface_to_new_addr(request_header: &CMsgSegHdr, iface: &Arc<Iface>) -> Option<AddrSegment> {
43    let ipv4_addr = iface.ipv4_addr()?;
44
45    let header = CMsgSegHdr {
46        len: 0,
47        type_: CSegmentType::NEWADDR as _,
48        flags: SegHdrCommonFlags::empty().bits(),
49        seq: request_header.seq,
50        pid: request_header.pid,
51    };
52
53    let addr_message = AddrSegmentBody {
54        family: CSocketAddrFamily::AF_INET as _,
55        prefix_len: iface.prefix_len().unwrap(),
56        flags: AddrMessageFlags::PERMANENT,
57        scope: RtScope::HOST,
58        index: NonZeroU32::new(iface.index()),
59    };
60
61    let attrs = vec![
62        AddrAttr::Address(ipv4_addr.octets()),
63        AddrAttr::Label(CString::new(iface.name()).unwrap()),
64        AddrAttr::Local(ipv4_addr.octets()),
65    ];
66
67    Some(AddrSegment::new(header, addr_message, attrs))
68}
