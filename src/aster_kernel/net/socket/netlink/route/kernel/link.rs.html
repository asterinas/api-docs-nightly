22pub(super) fn do_get_link(request_segment: &LinkSegment) -> Result<Vec<RtnlSegment>> {
23    let filter_by = FilterBy::from_request(request_segment)?;
24
25    let mut response_segments: Vec<RtnlSegment> = iter_all_ifaces()
26        // Filter to include only requested links.
27        .filter(|iface| match &filter_by {
28            FilterBy::Index(index) => *index == iface.index(),
29            FilterBy::Name(name) => *name == iface.name(),
30            FilterBy::Dump => true,
31        })
32        .map(|iface| iface_to_new_link(request_segment.header(), iface))
33        .map(RtnlSegment::NewLink)
34        .collect();
35
36    let dump_all = matches!(filter_by, FilterBy::Dump);
37
38    if !dump_all && response_segments.is_empty() {
39        return_errno_with_message!(Errno::ENODEV, "no link found");
40    }
41
42    finish_response(request_segment.header(), dump_all, &mut response_segments);
43
44    Ok(response_segments)
45}
46
47enum FilterBy<'a> {
48    Index(u32),
49    Name(&'a str),
50    Dump,
51}
52
53impl<'a> FilterBy<'a> {
54    fn from_request(request_segment: &'a LinkSegment) -> Result<Self> {
55        let dump_all = {
56            let flags = GetRequestFlags::from_bits_truncate(request_segment.header().flags);
57            flags.contains(GetRequestFlags::DUMP)
58        };
59        if dump_all {
60            validate_dumplink_request(request_segment.body())?;
61            return Ok(Self::Dump);
62        }
63
64        validate_getlink_request(request_segment.body())?;
65
66        // `index` takes precedence over `required_name`.
67
68        if let Some(required_index) = request_segment.body().index {
69            return Ok(Self::Index(required_index.get()));
70        }
71
72        let required_name = request_segment.attrs().iter().find_map(|attr| {
73            if let LinkAttr::Name(name) = attr {
74                Some(name.to_str().unwrap())
75            } else {
76                None
77            }
78        });
79        if let Some(required_name) = required_name {
80            return Ok(Self::Name(required_name));
81        }
82
83        return_errno_with_message!(
84            Errno::EINVAL,
85            "either interface name or index should be specified for non-dump mode"
86        );
87    }
88}

90// The below functions starting with `validate_` should only be enabled in strict mode.
91// Reference: <https://docs.kernel.org/userspace-api/netlink/intro.html#strict-checking>.
92
93fn validate_getlink_request(body: &LinkSegmentBody) -> Result<()> {
94    // FIXME: The Linux implementation also checks the `padding` and `change` fields,
95    // but these fields are lost during the conversion of a `CIfInfoMsg` to `LinkSegmentBody`.
96    // We should consider including the `change` field in `LinkSegmentBody`.
97    // Reference: <https://elixir.bootlin.com/linux/v6.13/source/net/core/rtnetlink.c#L4043>.
98    if !body.flags.is_empty() || body.type_ != InterfaceType::NETROM {
99        return_errno_with_message!(Errno::EINVAL, "the flags or the type is not valid");
100    }
101
102    Ok(())
103}
104
105fn validate_dumplink_request(body: &LinkSegmentBody) -> Result<()> {
106    // FIXME: The Linux implementation also checks the `padding` and `change` fields.
107    // Reference: <https://elixir.bootlin.com/linux/v6.13/source/net/core/rtnetlink.c#L2378>.
108    if !body.flags.is_empty() || body.type_ != InterfaceType::NETROM {
109        return_errno_with_message!(Errno::EINVAL, "the flags or the type is not valid");
110    }
111
112    // The check is from <https://elixir.bootlin.com/linux/v6.13/source/net/core/rtnetlink.c#L2383>.
113    if body.index.is_some() {
114        return_errno_with_message!(
115            Errno::EINVAL,
116            "filtering by interface index is not valid for link dumps"
117        );
118    }
119
120    Ok(())
121}

123fn iface_to_new_link(request_header: &CMsgSegHdr, iface: &Arc<Iface>) -> LinkSegment {
124    let header = CMsgSegHdr {
125        len: 0,
126        type_: CSegmentType::NEWLINK as _,
127        flags: SegHdrCommonFlags::empty().bits(),
128        seq: request_header.seq,
129        pid: request_header.pid,
130    };
131
132    let link_message = LinkSegmentBody {
133        family: CSocketAddrFamily::AF_UNSPEC,
134        type_: iface.type_(),
135        index: NonZero::new(iface.index()),
136        flags: iface.flags(),
137    };
138
139    let attrs = vec![
140        LinkAttr::Name(CString::new(iface.name()).unwrap()),
141        LinkAttr::Mtu(iface.mtu() as u32),
142    ];
143
144    LinkSegment::new(header, link_message, attrs)
145}
