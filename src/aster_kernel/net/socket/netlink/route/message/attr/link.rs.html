85#[derive(Debug)]
86pub enum LinkAttr {
87    Name(CString),
88    Mtu(u32),
89    TxqLen(u32),
90    LinkMode(u8),
91    ExtMask(RtExtFilter),
92}
93
94impl LinkAttr {
95    fn class(&self) -> LinkAttrClass {
96        match self {
97            LinkAttr::Name(_) => LinkAttrClass::IFNAME,
98            LinkAttr::Mtu(_) => LinkAttrClass::MTU,
99            LinkAttr::TxqLen(_) => LinkAttrClass::TXQLEN,
100            LinkAttr::LinkMode(_) => LinkAttrClass::LINKMODE,
101            LinkAttr::ExtMask(_) => LinkAttrClass::EXT_MASK,
102        }
103    }
104}
105
106impl Attribute for LinkAttr {
107    fn type_(&self) -> u16 {
108        self.class() as u16
109    }
110
111    fn payload_as_bytes(&self) -> &[u8] {
112        match self {
113            LinkAttr::Name(name) => name.as_bytes_with_nul(),
114            LinkAttr::Mtu(mtu) => mtu.as_bytes(),
115            LinkAttr::TxqLen(txq_len) => txq_len.as_bytes(),
116            LinkAttr::LinkMode(link_mode) => link_mode.as_bytes(),
117            LinkAttr::ExtMask(ext_filter) => ext_filter.as_bytes(),
118        }
119    }
120
121    fn read_from(header: &CAttrHeader, reader: &mut dyn MultiRead) -> Result<ContinueRead<Self>>
122    where
123        Self: Sized,
124    {
125        let payload_len = header.payload_len();
126
127        // TODO: Currently, `IS_NET_BYTEORDER_MASK` and `IS_NESTED_MASK` are ignored.
128        let Ok(class) = LinkAttrClass::try_from(header.type_()) else {
129            // Unknown attributes should be ignored.
130            // Reference: <https://docs.kernel.org/userspace-api/netlink/intro.html#unknown-attributes>.
131            reader.skip_some(payload_len);
132            return Ok(ContinueRead::Skipped);
133        };
134
135        let res = match (class, payload_len) {
136            (LinkAttrClass::IFNAME, 1..=IFNAME_SIZE) => {
137                let (name, namelen) =
138                    reader.read_cstring_until_end(IFNAME_SIZE.min(payload_len))?;
139                if namelen != payload_len {
140                    reader.skip_some(payload_len - namelen);
141                }
142                if name.as_bytes().len() == IFNAME_SIZE {
143                    return Ok(ContinueRead::skipped_with_error(
144                        Errno::ERANGE,
145                        "the link attribute is invalid",
146                    ));
147                }
148                Self::Name(name)
149            }
150            (LinkAttrClass::MTU, 4) => Self::Mtu(reader.read_val_opt::<u32>()?.unwrap()),
151            (LinkAttrClass::TXQLEN, 4) => Self::TxqLen(reader.read_val_opt::<u32>()?.unwrap()),
152            (LinkAttrClass::LINKMODE, 1) => Self::LinkMode(reader.read_val_opt::<u8>()?.unwrap()),
153            (LinkAttrClass::EXT_MASK, 4) => {
154                const { assert!(size_of::<RtExtFilter>() == 4) };
155                Self::ExtMask(reader.read_val_opt::<RtExtFilter>()?.unwrap())
156            }
157
158            (
159                LinkAttrClass::IFNAME
160                | LinkAttrClass::MTU
161                | LinkAttrClass::TXQLEN
162                | LinkAttrClass::LINKMODE
163                | LinkAttrClass::EXT_MASK,
164                _,
165            ) => {
166                warn!("link attribute `{:?}` contains invalid payload", class);
167                reader.skip_some(payload_len);
168                return Ok(ContinueRead::skipped_with_error(
169                    if class == LinkAttrClass::IFNAME {
170                        Errno::ERANGE
171                    } else {
172                        Errno::EINVAL
173                    },
174                    "the link attribute is invalid",
175                ));
176            }
177
178            (_, _) => {
179                warn!("link attribute `{:?}` is not supported", class);
180                reader.skip_some(payload_len);
181                return Ok(ContinueRead::Skipped);
182            }
183        };
184
185        Ok(ContinueRead::Parsed(res))
186    }
187}

189bitflags! {
190    /// New extended info filters for [`NlLinkAttr::ExtMask`].
191    ///
192    /// Reference: <https://elixir.bootlin.com/linux/v6.13/source/include/uapi/linux/rtnetlink.h#L819>.
193    #[repr(C)]
194    #[derive(Pod)]
195    pub struct RtExtFilter: u32 {
196        const VF = 1 << 0;
197        const BRVLAN = 1 << 1;
198        const BRVLAN_COMPRESSED = 1 << 2;
199        const SKIP_STATS = 1 << 3;
200        const MRP = 1 << 4;
201        const CFM_CONFIG = 1 << 5;
202        const CFM_STATUS = 1 << 6;
203        const MST = 1 << 7;
204    }
