21/// `ifaddrmsg` in Linux.
22///
23/// Reference: <https://elixir.bootlin.com/linux/v6.13/source/include/uapi/linux/if_addr.h#L8>.
24#[repr(C)]
25#[derive(Debug, Clone, Copy, Pod)]
26pub struct CIfaddrMsg {
27    pub family: u8,
28    /// The prefix length
29    pub prefix_len: u8,
30    /// Flags
31    pub flags: u8,
32    /// Address scope
33    pub scope: u8,
34    /// Link index
35    pub index: u32,
36}
37
38#[derive(Debug, Clone, Copy)]
39pub struct AddrSegmentBody {
40    pub family: i32,
41    pub prefix_len: u8,
42    pub flags: AddrMessageFlags,
43    pub scope: RtScope,
44    pub index: Option<NonZeroU32>,
45}
46
47impl TryFrom<CIfaddrMsg> for AddrSegmentBody {
48    type Error = Error;
49
50    fn try_from(value: CIfaddrMsg) -> Result<Self> {
51        // TODO: If the attribute IFA_FLAGS exists, the flags in header should be ignored.
52        let flags = AddrMessageFlags::from_bits_truncate(value.flags as u32);
53        let scope = RtScope::try_from(value.scope)?;
54        let index = NonZeroU32::new(value.index);
55
56        Ok(Self {
57            family: value.family as i32,
58            prefix_len: value.prefix_len,
59            flags,
60            scope,
61            index,
62        })
63    }
64}
65
66impl From<AddrSegmentBody> for CIfaddrMsg {
67    fn from(value: AddrSegmentBody) -> Self {
68        let index = if let Some(index) = value.index {
69            index.get()
70        } else {
71            0
72        };
73        CIfaddrMsg {
74            family: value.family as u8,
75            prefix_len: value.prefix_len,
76            flags: value.flags.bits() as u8,
77            scope: value.scope as _,
78            index,
79        }
80    }
81}

92        const HOMEADDRESS    = 0x10;

94        const TENTATIVE		 = 0x40;
95        const PERMANENT		 = 0x80;
96        const MANAGETEMPADDR = 0x100;
97        const NOPREFIXROUTE	 = 0x200;

99        const STABLE_PRIVACY = 0x800;
100    }
