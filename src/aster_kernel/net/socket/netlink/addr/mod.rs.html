9/// The socket address of a netlink socket.
10///
11/// The address contains the port number for unicast
12/// and the group IDs for multicast.
13#[derive(Debug, Clone, Copy, PartialEq, Eq)]
14pub struct NetlinkSocketAddr {
15    port: PortNum,
16    groups: GroupIdSet,
17}
18
19impl NetlinkSocketAddr {
20    /// Creates a new netlink address.
21    pub const fn new(port: PortNum, groups: GroupIdSet) -> Self {
22        Self { port, groups }
23    }
24
25    /// Creates a new unspecified address.
26    ///
27    /// Both the port ID and group numbers are left unspecified.
28    ///
29    /// Note that an unspecified address can also represent the kernel socket address.
30    pub const fn new_unspecified() -> Self {
31        Self {
32            port: UNSPECIFIED_PORT,
33            groups: GroupIdSet::new_empty(),
34        }
35    }
36
37    /// Returns the port number.
38    pub const fn port(&self) -> PortNum {
39        self.port
40    }
41
42    /// Returns the group ID set.
43    pub const fn groups(&self) -> GroupIdSet {
44        self.groups
45    }
46
47    /// Adds some new groups to the address.
48    pub fn add_groups(&mut self, groups: GroupIdSet) {
49        self.groups.add_groups(groups);
50    }
51}
52
53impl TryFrom<SocketAddr> for NetlinkSocketAddr {
54    type Error = Error;
55
56    fn try_from(value: SocketAddr) -> Result<Self> {
57        match value {
58            SocketAddr::Netlink(addr) => Ok(addr),
59            _ => return_errno_with_message!(
60                Errno::EAFNOSUPPORT,
61                "the address is in an unsupported address family"
62            ),
63        }
64    }
65}
66
67impl From<NetlinkSocketAddr> for SocketAddr {
68    fn from(value: NetlinkSocketAddr) -> Self {
69        SocketAddr::Netlink(value)
70    }
71}
