12/// The UTS namespace.
13pub struct UtsNamespace {
14    uts_name: RwMutex<UtsName>,
15    owner: Arc<UserNamespace>,
16}
17
18impl UtsNamespace {
19    /// Returns a reference to the singleton initial UTS namespace.
20    pub fn get_init_singleton() -> &'static Arc<UtsNamespace> {
21        static INIT: Once<Arc<UtsNamespace>> = Once::new();
22
23        INIT.call_once(|| {
24            let uts_name = UtsName {
25                sysname: padded(UtsName::SYSNAME.as_bytes()),
26                // Reference: <https://elixir.bootlin.com/linux/v6.16/source/init/Kconfig#L408>.
27                nodename: padded(b"(none)"),
28                release: padded(UtsName::RELEASE.as_bytes()),
29                version: padded(UtsName::VERSION.as_bytes()),
30                machine: padded(UtsName::MACHINE.as_bytes()),
31                // Reference: <https://elixir.bootlin.com/linux/v6.16/source/include/linux/uts.h#L17>.
32                domainname: padded(b"(none)"),
33            };
34
35            let owner = UserNamespace::get_init_singleton().clone();
36
37            Arc::new(Self {
38                uts_name: RwMutex::new(uts_name),
39                owner,
40            })
41        })
42    }
43
44    /// Clones a new UTS namespace from `self`.
45    pub fn new_clone(
46        &self,
47        owner: Arc<UserNamespace>,
48        posix_thread: &PosixThread,
49    ) -> Result<Arc<Self>> {
50        owner.check_cap(CapSet::SYS_ADMIN, posix_thread)?;
51        Ok(Arc::new(Self {
52            uts_name: RwMutex::new(*self.uts_name.read()),
53            owner,
54        }))
55    }
56
57    /// Returns the owner user namespace of the namespace.
58    pub fn owner_ns(&self) -> &Arc<UserNamespace> {
59        &self.owner
60    }
61
62    /// Returns a read-only lock guard for accessing the UTS name.
63    pub fn uts_name(&self) -> RwMutexReadGuard<'_, UtsName> {
64        self.uts_name.read()
65    }
66
67    /// Sets a new hostname for the UTS namespace.
68    ///
69    /// This method will fail with `EPERM` if the caller does not have the SYS_ADMIN capability
70    /// in the owner user namespace.
71    pub fn set_hostname(&self, addr: Vaddr, len: usize, ctx: &Context) -> Result<()> {
72        self.owner.check_cap(CapSet::SYS_ADMIN, ctx.posix_thread)?;
73
74        let new_host_name = copy_uts_field_from_user(addr, len as _, ctx)?;
75        debug!(
76            "set host name: {:?}",
77            CStr::from_bytes_until_nul(new_host_name.as_bytes()).unwrap()
78        );
79        self.uts_name.write().nodename = new_host_name;
80        Ok(())
81    }
82
83    /// Sets a new domain name for the UTS namespace.
84    ///
85    /// This method will fail with `EPERM` if the caller does not have the SYS_ADMIN capability
86    /// in the owner user namespace.
87    pub fn set_domainname(&self, addr: Vaddr, len: usize, ctx: &Context) -> Result<()> {
88        self.owner.check_cap(CapSet::SYS_ADMIN, ctx.posix_thread)?;
89
90        let new_domain_name = copy_uts_field_from_user(addr, len as _, ctx)?;
91        debug!(
92            "set domain name: {:?}",
93            CStr::from_bytes_until_nul(new_domain_name.as_bytes()).unwrap()
94        );
95        self.uts_name.write().domainname = new_domain_name;
96        Ok(())
97    }
98}
99
100fn copy_uts_field_from_user(addr: Vaddr, len: u32, ctx: &Context) -> Result<[u8; UTS_FIELD_LEN]> {
101    if len.cast_signed() < 0 {
102        return_errno_with_message!(Errno::EINVAL, "the buffer length cannot be negative");
103    }
104
105    let user_space = ctx.user_space();
106    let mut reader = user_space.reader(addr, len as usize)?;
107
108    // UTS fields represent C strings, which must be nul-terminated.
109    // Therefore, the user-provided buffer length cannot exceed `UTS_FIELD_LEN - 1`
110    // to ensure space for the terminating nul byte.
111    if reader.remain() > UTS_FIELD_LEN - 1 {
112        return_errno_with_message!(Errno::EINVAL, "the UTS name is too long");
113    }
114
115    let mut buffer = [0u8; UTS_FIELD_LEN];
116
117    // Partial reads are acceptable,
118    // but an error is returned if no bytes can be read successfully.
119    if let Err((err, 0)) = reader.read_fallible(&mut VmWriter::from(buffer.as_mut_slice())) {
120        return Err(err.into());
121    }
122
123    Ok(buffer)
124}
125
126const UTS_FIELD_LEN: usize = 65;
127
128#[derive(Debug, Clone, Copy, Pod)]
129#[repr(C)]
130pub struct UtsName {
131    sysname: [u8; UTS_FIELD_LEN],
132    nodename: [u8; UTS_FIELD_LEN],
133    release: [u8; UTS_FIELD_LEN],
134    version: [u8; UTS_FIELD_LEN],
135    machine: [u8; UTS_FIELD_LEN],
136    domainname: [u8; UTS_FIELD_LEN],
137}
138
139impl UtsName {
140    // Note that the following names remain constant across all UTS namespaces. They are immutable,
141    // meaning the user space cannot change them. They are stored in the UTS namespace for ease of
142    // implementing related system calls.
143    //
144    // In addition, we intentionally report Linux-like UTS values instead of Asterinas' real name
145    // and version. These spoofed values satisfy glibc, which inspects uname fields (sysname,
146    // release, version, etc.) and expects Linux-compatible data.
147
148    /// The system name.
149    pub const SYSNAME: &str = "Linux";
150
151    /// The release name.
152    pub const RELEASE: &str = "5.13.0";
153
154    /// The version name.
155    pub const VERSION: &str = {
156        const BUILD_TIMESTAMP: &str = if let Some(timestamp) = option_env!("ASTER_BUILD_TIMESTAMP")
157        {
158            timestamp
159        } else {
160            const UNIX_EPOCH: &str = "Thu Jan  1 00:00:00 UTC 1970";
161            UNIX_EPOCH
162        };
163
164        // The definition of Linux's UTS_VERSION can be found at:
165        // <https://elixir.bootlin.com/linux/v6.18/source/init/Makefile#L37>.
166        // Linux specifies that the total length of this string must not exceed 64 bytes.
167
168        // In Linux, the BUILD_VERSION represents the compilation count, which
169        // increments each time the kernel is built within the same source tree.
170        // We use a fixed value of '1' here to ensure build determinism.
171        // Reference: <https://elixir.bootlin.com/linux/v6.18/source/scripts/build-version>.
172        const BUILD_VERSION: usize = 1;
173        const SMP_FLAGS: &str = "SMP ";
174        const PREEMPT_FLAGS: &str = "";
175        const VERSION: &str =
176            const_format::formatcp!("#{BUILD_VERSION} {SMP_FLAGS}{PREEMPT_FLAGS}{BUILD_TIMESTAMP}");
177        const_assert!(VERSION.len() <= 64);
178        VERSION
179    };
180
181    /// The machine name.
182    pub const MACHINE: &str = {
183        cfg_if::cfg_if! {
184            if #[cfg(target_arch = "x86_64")] {
185                "x86_64"
186            } else if #[cfg(target_arch = "riscv64")] {
187                "riscv64"
188            } else if #[cfg(target_arch = "loongarch64")] {
189                "loongarch64"
190            } else if #[cfg(target_arch = "aarch64")] {
191                "aarch64"
192            } else {
193                compile_error!("unsupported target")
194            }
195        }
196    };
197}
