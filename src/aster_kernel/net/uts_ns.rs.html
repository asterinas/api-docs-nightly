13/// The UTS namespace.
14pub struct UtsNamespace {
15    uts_name: RwMutex<UtsName>,
16    owner: Arc<UserNamespace>,
17    stashed_dentry: StashedDentry,
18}
19
20impl UtsNamespace {
21    /// Returns a reference to the singleton initial UTS namespace.
22    pub fn get_init_singleton() -> &'static Arc<UtsNamespace> {
23        static INIT: Once<Arc<UtsNamespace>> = Once::new();
24
25        INIT.call_once(|| {
26            let uts_name = UtsName {
27                sysname: UtsField::from_bytes_until_nul(UtsName::SYSNAME.as_bytes()),
28                // Reference: <https://elixir.bootlin.com/linux/v6.16/source/init/Kconfig#L408>.
29                nodename: UtsField::from_bytes_until_nul(b"(none)"),
30                release: UtsField::from_bytes_until_nul(UtsName::RELEASE.as_bytes()),
31                version: UtsField::from_bytes_until_nul(UtsName::VERSION.as_bytes()),
32                machine: UtsField::from_bytes_until_nul(UtsName::MACHINE.as_bytes()),
33                // Reference: <https://elixir.bootlin.com/linux/v6.16/source/include/linux/uts.h#L17>.
34                domainname: UtsField::from_bytes_until_nul(b"(none)"),
35            };
36
37            let owner = UserNamespace::get_init_singleton().clone();
38            Self::new(uts_name, owner)
39        })
40    }
41
42    fn new(uts_name: UtsName, owner: Arc<UserNamespace>) -> Arc<Self> {
43        let stashed_dentry = StashedDentry::new();
44        Arc::new(Self {
45            uts_name: RwMutex::new(uts_name),
46            owner,
47            stashed_dentry,
48        })
49    }
50
51    /// Clones a new UTS namespace from `self`.
52    pub fn new_clone(
53        &self,
54        owner: Arc<UserNamespace>,
55        posix_thread: &PosixThread,
56    ) -> Result<Arc<Self>> {
57        lsm_hooks::on_capable(lsm_hooks::CapableContext::new(
58            owner.as_ref(),
59            posix_thread,
60            CapSet::SYS_ADMIN,
61        ))?;
62        Ok(Self::new(*self.uts_name.read(), owner))
63    }
64
65    /// Returns a read-only lock guard for accessing the UTS name.
66    pub fn uts_name(&self) -> RwMutexReadGuard<'_, UtsName> {
67        self.uts_name.read()
68    }
69
70    /// Sets a new hostname for the UTS namespace.
71    ///
72    /// This method will fail with `EPERM` if the POSIX thread does not have the
73    /// SYS_ADMIN capability in the owner user namespace.
74    pub fn set_hostname(&self, new_host_name: UtsField, posix_thread: &PosixThread) -> Result<()> {
75        self.check_set_permission(posix_thread)?;
76        self.set_hostname_field(new_host_name);
77        Ok(())
78    }
79
80    /// Sets a new domain name for the UTS namespace.
81    ///
82    /// This method will fail with `EPERM` if the POSIX thread does not have the
83    /// SYS_ADMIN capability in the owner user namespace.
84    pub fn set_domainname(
85        &self,
86        new_domain_name: UtsField,
87        posix_thread: &PosixThread,
88    ) -> Result<()> {
89        self.check_set_permission(posix_thread)?;
90        self.set_domainname_field(new_domain_name);
91        Ok(())
92    }
93
94    fn check_set_permission(&self, posix_thread: &PosixThread) -> Result<()> {
95        lsm_hooks::on_capable(lsm_hooks::CapableContext::new(
96            self.owner.as_ref(),
97            posix_thread,
98            CapSet::SYS_ADMIN,
99        ))
100    }
101
102    fn set_hostname_field(&self, new_host_name: UtsField) {
103        debug!("set host name: {:?}", new_host_name.as_cstr());
104        self.uts_name.write().nodename = new_host_name;
105    }
106
107    fn set_domainname_field(&self, new_domain_name: UtsField) {
108        debug!("set domain name: {:?}", new_domain_name.as_cstr());
109        self.uts_name.write().domainname = new_domain_name;
110    }
111}

113#[repr(C)]
114#[derive(Clone, Copy, Debug, Pod)]
115pub struct UtsName {
116    sysname: UtsField,
117    nodename: UtsField,
118    release: UtsField,
119    version: UtsField,
120    machine: UtsField,
121    domainname: UtsField,
122}
123
124impl UtsName {
125    // Note that the following names remain constant across all UTS namespaces. They are immutable,
126    // meaning the user space cannot change them. They are stored in the UTS namespace for ease of
127    // implementing related system calls.
128    //
129    // In addition, we intentionally report Linux-like UTS values instead of Asterinas' real name
130    // and version. These spoofed values satisfy glibc, which inspects uname fields (sysname,
131    // release, version, etc.) and expects Linux-compatible data.
132
133    /// The system name.
134    pub const SYSNAME: &str = "Linux";
135
136    /// The release name.
137    pub const RELEASE: &str = "5.13.0";
138
139    /// The version name.
140    pub const VERSION: &str = {
141        const BUILD_TIMESTAMP: &str = if let Some(timestamp) = option_env!("ASTER_BUILD_TIMESTAMP")
142        {
143            timestamp
144        } else {
145            const UNIX_EPOCH: &str = "Thu Jan  1 00:00:00 UTC 1970";
146            UNIX_EPOCH
147        };
148
149        // The definition of Linux's UTS_VERSION can be found at:
150        // <https://elixir.bootlin.com/linux/v6.18/source/init/Makefile#L37>.
151        // Linux specifies that the total length of this string must not exceed 64 bytes.
152
153        // In Linux, the BUILD_VERSION represents the compilation count, which
154        // increments each time the kernel is built within the same source tree.
155        // We use a fixed value of '1' here to ensure build determinism.
156        // Reference: <https://elixir.bootlin.com/linux/v6.18/source/scripts/build-version>.
157        const BUILD_VERSION: usize = 1;
158        const SMP_FLAGS: &str = "SMP ";
159        const PREEMPT_FLAGS: &str = "";
160        const VERSION: &str =
161            const_format::formatcp!("#{BUILD_VERSION} {SMP_FLAGS}{PREEMPT_FLAGS}{BUILD_TIMESTAMP}");
162        assert!(VERSION.len() <= UtsField::MAX_BYTES);
163        VERSION
164    };
165
166    /// The machine name.
167    pub const MACHINE: &str = {
168        cfg_if::cfg_if! {
169            if #[cfg(target_arch = "x86_64")] {
170                "x86_64"
171            } else if #[cfg(target_arch = "riscv64")] {
172                "riscv64"
173            } else if #[cfg(target_arch = "loongarch64")] {
174                "loongarch64"
175            } else if #[cfg(target_arch = "aarch64")] {
176                "aarch64"
177            } else {
178                compile_error!("unsupported target")
179            }
180        }
181    };
182
183    /// Returns the hostname.
184    pub fn nodename(&self) -> &UtsField {
185        &self.nodename
186    }
187
188    /// Returns the NIS domain name.
189    pub fn domainname(&self) -> &UtsField {
190        &self.domainname
191    }
192}

194/// A nul-terminated UTS field.
195///
196/// Although this is a POD type, it has a type invariant: a nul byte must be
197/// present, and all bytes after the first nul byte must also be nul bytes.
198/// Users outside this module must not arbitrarily mutate the content.
199#[repr(transparent)]
200#[derive(Clone, Copy, Debug, Pod)]
201pub struct UtsField([u8; UtsField::MAX_BYTES_WITH_NUL]);
202
203impl UtsField {
204    /// The maximum byte length of a UTS field, excluding the trailing nul.
205    pub const MAX_BYTES: usize = 64;
206
207    /// The storage byte length of a UTS field, including the trailing nul.
208    pub const MAX_BYTES_WITH_NUL: usize = Self::MAX_BYTES + 1;
209
210    /// Creates a UTS field from bytes, stopping at the first nul byte.
211    ///
212    /// If there is no nul byte within the first [`Self::MAX_BYTES`] bytes,
213    /// the input is truncated to [`Self::MAX_BYTES`] bytes. The returned field
214    /// is always nul-terminated, and all bytes after the first nul byte are
215    /// zeroed.
216    pub fn from_bytes_until_nul(bytes: &[u8]) -> Self {
217        let mut field = [0u8; Self::MAX_BYTES_WITH_NUL];
218        let len = bytes
219            .iter()
220            .position(|byte| *byte == 0)
221            .unwrap_or(bytes.len())
222            .min(Self::MAX_BYTES);
223        field[..len].copy_from_slice(&bytes[..len]);
224        Self(field)
225    }
226
227    /// Reads a UTS field from user space.
228    pub fn read_from(addr: Vaddr, len: usize, ctx: &Context) -> Result<Self> {
229        // UTS fields represent C strings, which must be nul-terminated.
230        // Therefore, the user-provided buffer length cannot exceed `Self::MAX_BYTES`
231        // to ensure space for the terminating nul byte.
232        if len > Self::MAX_BYTES {
233            return_errno_with_message!(Errno::EINVAL, "the UTS name is too long");
234        }
235
236        let user_space = ctx.user_space();
237        let mut reader = user_space.reader(addr, len)?;
238        let mut field = [0u8; Self::MAX_BYTES_WITH_NUL];
239
240        // Partial reads are acceptable,
241        // but an error is returned if no bytes can be read successfully.
242        if let Err((err, 0)) = reader.read_fallible(&mut VmWriter::from(field.as_mut_slice())) {
243            return Err(err.into());
244        }
245
246        Ok(Self::from_bytes_until_nul(&field))
247    }
248
249    /// Returns the UTS field as a C string.
250    pub fn as_cstr(&self) -> &CStr {
251        CStr::from_bytes_until_nul(self.0.as_bytes()).unwrap()
252    }
253
254    /// Returns the underlying byte array.
255    pub fn as_array(&self) -> &[u8; Self::MAX_BYTES_WITH_NUL] {
256        &self.0
257    }
258}
