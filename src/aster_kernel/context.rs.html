23/// The context that can be accessed from the current POSIX thread.
24#[derive(Clone)]
25pub struct Context<'a> {
26    pub process: Arc<Process>,
27    pub thread_local: &'a ThreadLocal,
28    pub posix_thread: &'a PosixThread,
29    pub thread: &'a Thread,
30    pub task: &'a Task,
31}
32
33impl Context<'_> {
34    /// Gets the userspace of the current task.
35    pub fn user_space(&self) -> CurrentUserSpace<'_> {
36        CurrentUserSpace(self.thread_local.vmar().borrow())
37    }
38}

40/// The user's memory space of the current task.
41///
42/// It provides methods to read from or write to the user space efficiently.
43//
44// FIXME: With `impl VmIo for &CurrentUserSpace<'_>`, the Rust compiler seems to think that
45// `CurrentUserSpace` is a publicly exposed type, despite the fact that it is contained in a
46// private module and is never actually exposed. Consequently, it incorrectly suppresses many dead
47// code lints (for *lots of* types that are recursively reached via `CurrentUserSpace`'s APIs). As
48// a workaround, we mark the type as `pub(crate)`. We can restore it to `pub` once the compiler bug
49// is resolved.
50pub(crate) struct CurrentUserSpace<'a>(Ref<'a, Option<Arc<Vmar>>>);
51
52/// Gets the [`CurrentUserSpace`] from the current task.
53///
54/// This is slower than [`Context::user_space`]. Don't use this getter
55/// If you get the access to the [`Context`].
56#[macro_export]
57macro_rules! current_userspace {
58    () => {
59        $crate::context::CurrentUserSpace::new(
60            $crate::process::posix_thread::AsThreadLocal::as_thread_local(
61                &ostd::task::Task::current().unwrap(),
62            )
63            .unwrap(),
64        )
65    };
66}

68impl<'a> CurrentUserSpace<'a> {
69    /// Creates a new `CurrentUserSpace` from the current task.
70    ///
71    /// If you have access to a [`Context`], it is preferable to call [`Context::user_space`].
72    ///
73    /// Otherwise, you can use the `current_userspace` macro
74    /// to obtain an instance of `CurrentUserSpace` if it will only be used once.
75    pub fn new(thread_local: &'a ThreadLocal) -> Self {
76        let vmar_ref = thread_local.vmar().borrow();
77        Self(vmar_ref)
78    }
79
80    /// Returns the `Vmar` of the current userspace.
81    ///
82    /// # Panics
83    ///
84    /// This method will panic if the current process has cleared its `Vmar`.
85    pub fn vmar(&self) -> &Vmar {
86        self.0.as_ref().unwrap()
87    }
88
89    /// Returns whether the VMAR is shared with other processes or threads.
90    pub fn is_vmar_shared(&self) -> bool {
91        // If the VMAR is not shared, its reference count should be exactly 2:
92        // one reference is held by `ThreadLocal` and the other by `ProcessVm` in `Process`.
93        Arc::strong_count(self.0.as_ref().unwrap()) > 2
94    }
95
96    /// Creates a reader to read data from the user space of the current task.
97    ///
98    /// Returns `Err` if `vaddr` and `len` do not represent a user space memory range.
99    pub fn reader(&self, vaddr: Vaddr, len: usize) -> Result<VmReader<'_, Fallible>> {
100        // Do NOT attempt to call `check_vaddr_lowerbound` here.
101        //
102        // Linux has a **delayed buffer validation** behavior:
103        // The Linux kernel assumes that a given user-space pointer is valid until it attempts to access it.
104        // For example, the following invocation of the `read` system call with a `NULL` pointer as the buffer
105        //
106        // ```c
107        // read(fd, NULL, 1);
108        // ```
109        //
110        // will return 0 (rather than an error) if the file referred to by `fd` has zero length.
111        //
112        // Asterinas's system call entry points follow a pattern of converting user-space pointers to
113        // a reader/writer first and using the reader/writer later.
114        // So adding any pointer check here would break Asterinas's delayed buffer validation behavior.
115        Ok(self.vmar().vm_space().reader(vaddr, len)?)
116    }
117
118    /// Creates a writer to write data into the user space of the current task.
119    ///
120    /// Returns `Err` if `vaddr` and `len` do not represent a user space memory range.
121    pub fn writer(&self, vaddr: Vaddr, len: usize) -> Result<VmWriter<'_, Fallible>> {
122        // Do NOT attempt to call `check_vaddr_lowerbound` here.
123        // See the comments in the `reader` method.
124        Ok(self.vmar().vm_space().writer(vaddr, len)?)
125    }
126
127    /// Creates a reader/writer pair to read data from or write data into the user space
128    /// of the current task.
129    ///
130    /// Returns `Err` if `vaddr` and `len` do not represent a user space memory range.
131    ///
132    /// This method is semantically equivalent to calling [`Self::reader`] and [`Self::writer`]
133    /// separately, but it avoids double checking the validity of the memory region.
134    pub fn reader_writer(
135        &self,
136        vaddr: Vaddr,
137        len: usize,
138    ) -> Result<(VmReader<'_, Fallible>, VmWriter<'_, Fallible>)> {
139        // Do NOT attempt to call `check_vaddr_lowerbound` here.
140        // See the comments in the `reader` method.
141        Ok(self.vmar().vm_space().reader_writer(vaddr, len)?)
142    }
143
144    /// Atomically loads a `PodAtomic` value with [`Ordering::Relaxed`] semantics.
145    ///
146    /// # Panics
147    ///
148    /// This method will panic if `vaddr` is not aligned on an `align_of::<T>()`-byte boundary.
149    ///
150    /// [`Ordering::Relaxed`]: core::sync::atomic::Ordering::Relaxed
151    pub fn atomic_load<T: PodAtomic>(&self, vaddr: Vaddr) -> Result<T> {
152        if size_of::<T>() > 0 {
153            check_vaddr_lowerbound(vaddr)?;
154        }
155
156        let user_reader = self.reader(vaddr, size_of::<T>())?;
157        Ok(user_reader.atomic_load()?)
158    }
159
160    /// Atomically updates a `PodAtomic` value with [`Ordering::Relaxed`] semantics.
161    ///
162    /// This method internally fetches the old value via [`atomic_load`], applies `op` to compute a
163    /// new value, and updates the value via [`atomic_compare_exchange`]. If the value changes
164    /// concurrently, this method will retry so the operation may be performed multiple times.
165    ///
166    /// If the update is completely successful, returns `Ok` with the old value (i.e., the value
167    /// _before_ applying `op`). Otherwise, it returns `Err`.
168    ///
169    /// # Panics
170    ///
171    /// This method will panic if `vaddr` is not aligned on an `align_of::<T>()`-byte boundary.
172    ///
173    /// [`Ordering::Relaxed`]: core::sync::atomic::Ordering::Relaxed
174    /// [`atomic_load`]: VmReader::atomic_load
175    /// [`atomic_compare_exchange`]: VmWriter::atomic_compare_exchange
176    pub fn atomic_fetch_update<T>(&self, vaddr: Vaddr, op: impl Fn(T) -> T) -> Result<T>
177    where
178        T: PodAtomic + Eq,
179    {
180        if size_of::<T>() > 0 {
181            check_vaddr_lowerbound(vaddr)?;
182        }
183
184        let (reader, writer) = self.reader_writer(vaddr, size_of::<T>())?;
185
186        let mut old_val = reader.atomic_load()?;
187        loop {
188            match writer.atomic_compare_exchange(&reader, old_val, op(old_val))? {
189                (_, true) => return Ok(old_val),
190                (cur_val, false) => old_val = cur_val,
191            }
192        }
193    }
194
195    /// Reads a C string from the user space of the current process.
196    ///
197    /// The length of the string should not exceed `max_len`, including the final nul byte.
198    /// Otherwise, this method will fail with [`Errno::ENAMETOOLONG`].
199    ///
200    /// This method is commonly used to read a file name or path. In that case, when the nul byte
201    /// cannot be found within `max_len` bytes, the correct error code is [`Errno::ENAMETOOLONG`].
202    /// However, in other cases, the caller may want to fix the error code manually.
203    pub fn read_cstring(&self, vaddr: Vaddr, max_len: usize) -> Result<CString> {
204        if max_len > 0 {
205            check_vaddr_lowerbound(vaddr)?;
206        }
207
208        // Adjust `max_len` to ensure `vaddr + max_len` does not exceed `VMAR_CAP_ADDR`.
209        // If `vaddr` is outside user address space, `userspace_max_len` will be set to zero and
210        // further call to `self.reader` will return `EFAULT`.
211        let userspace_max_len = VMAR_CAP_ADDR.saturating_sub(vaddr).min(max_len);
212
213        let mut user_reader = self.reader(vaddr, userspace_max_len)?;
214        user_reader.read_cstring_until_nul(userspace_max_len)?
215            .ok_or_else(|| if userspace_max_len == max_len {
216                // There may be more bytes in the userspace, but the length limit has been reached.
217                Error::with_message(
218                    Errno::ENAMETOOLONG,
219                    "the C string does not end before reaching the maximum length"
220                )
221            } else {
222                // There cannot be any bytes in the userspace, but the C string still does not end.
223                // This is the Linux behavior in its `do_strncpy_from_user` implementation.
224                Error::with_message(
225                    Errno::EFAULT,
226                    "the C string does not end before reaching the maximum userspace virtual address"
227                )
228            })
229    }
230}

232impl VmIo for CurrentUserSpace<'_> {
233    fn read(&self, offset: usize, writer: &mut VmWriter) -> ostd::Result<()> {
234        let copy_len = writer.avail();
235
236        if copy_len > 0 {
237            check_vaddr_lowerbound(offset)?;
238        }
239
240        let mut user_reader = self.vmar().vm_space().reader(offset, copy_len)?;
241        user_reader.read_fallible(writer).map_err(|err| err.0)?;
242        Ok(())
243    }
244
245    fn write(&self, offset: usize, reader: &mut VmReader) -> ostd::Result<()> {
246        let copy_len = reader.remain();
247
248        if copy_len > 0 {
249            check_vaddr_lowerbound(offset)?;
250        }
251
252        let mut user_writer = self.vmar().vm_space().writer(offset, copy_len)?;
253        user_writer.write_fallible(reader).map_err(|err| err.0)?;
254        Ok(())
255    }
256}

258#[inherit_methods(from = "(**self)")]
259impl VmIo for &CurrentUserSpace<'_> {
260    fn read(&self, offset: usize, writer: &mut VmWriter) -> ostd::Result<()>;
261    fn write(&self, offset: usize, reader: &mut VmReader) -> ostd::Result<()>;
262}

264/// Checks if the user space pointer is below the lowest userspace address.
265///
266/// If a pointer is below the lowest userspace address, it is likely to be a
267/// NULL pointer. Reading from or writing to a NULL pointer should trigger a
268/// segmentation fault.
269///
270/// If it is not checked here, a kernel page fault will happen and we would
271/// deny the access in the page fault handler anyway. It may save a page fault
272/// in some occasions. More importantly, double page faults may not be handled
273/// quite well on some platforms.
274fn check_vaddr_lowerbound(va: Vaddr) -> ostd::Result<()> {
275    if va < VMAR_LOWEST_ADDR {
276        return Err(ostd::Error::PageFault);
277    }
278    Ok(())
