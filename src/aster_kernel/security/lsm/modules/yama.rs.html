22impl LsmAlienAccessHook for YamaLsm {
23    fn on_alien_access(&self, context: &AlienAccessContext) -> Result<()> {
24        if context.mode().kind() != AlienAccessKind::Attach {
25            return Ok(());
26        }
27
28        let is_denied = match get_scope() {
29            YamaScope::Disabled => false,
30            YamaScope::Relational => {
31                !context.accessor_has_cap_sys_ptrace()
32                    && !is_ancestor_of(
33                        context.accessor().weak_process(),
34                        context.target().process(),
35                    )
36            }
37            YamaScope::Capability => !context.accessor_has_cap_sys_ptrace(),
38            YamaScope::NoAttach => true,
39        };
40
41        if is_denied {
42            return_errno_with_message!(Errno::EPERM, "alien access is denied due to Yama scope");
43        }
44
45        Ok(())
46    }
47}

59/// Returns the current Yama scope for alien access.
60pub fn get_scope() -> YamaScope {
61    YAMA_SCOPE.load(Ordering::Relaxed)
62}
63
64/// Sets the Yama scope for alien access.
65pub fn set_scope(new_scope: YamaScope) -> Result<()> {
66    UserNamespace::get_init_singleton().check_cap(
67        CapSet::SYS_PTRACE,
68        current_thread!().as_posix_thread().unwrap(),
69    )?;
70
71    YAMA_SCOPE
72        .fetch_update(Ordering::Relaxed, Ordering::Relaxed, |current_scope| {
73            let is_downgrading_from_no_attach =
74                current_scope == YamaScope::NoAttach && new_scope != YamaScope::NoAttach;
75            (!is_downgrading_from_no_attach).then_some(new_scope)
76        })
77        .map_err(|_| {
78            Error::with_message(
79                Errno::EINVAL,
80                "`YamaScope::NoAttach` cannot be changed once set",
81            )
82        })?;
83
84    Ok(())
85}

87/// The Yama scope levels.
88#[repr(i32)]
89#[derive(Clone, Copy, Debug, Eq, PartialEq, TryFromInt)]
90pub enum YamaScope {
91    /// No additional restrictions on alien attach.
92    Disabled = 0,
93    /// Only allow alien attach by ancestor processes, or processes with `CapSet::SYS_PTRACE`.
94    Relational = 1,
95    /// Only allow alien attach by processes with `CapSet::SYS_PTRACE`.
96    Capability = 2,
97    /// Disallow any alien attach.
98    NoAttach = 3,
99}
100
101impl From<YamaScope> for i32 {
102    fn from(scope: YamaScope) -> Self {
103        scope as i32
104    }
105}
106
107define_atomic_version_of_integer_like_type!(YamaScope, try_from = true, {
108    struct AtomicYamaScope(AtomicI32);
109});

111static YAMA_SCOPE: AtomicYamaScope = AtomicYamaScope(AtomicI32::new(YamaScope::Relational as i32));

113fn is_ancestor_of(ancestor: &Weak<Process>, descendant: Arc<Process>) -> bool {
114    let mut current = descendant;
115    loop {
116        let parent_guard = current.parent().lock();
117        let parent = parent_guard.process();
118        if parent.ptr_eq(ancestor) {
119            return true;
120        }
121        let Some(parent) = parent.upgrade() else {
122            return false;
123        };
124        drop(parent_guard);
125        current = parent;
126    }
127}
