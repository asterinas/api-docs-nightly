3use super::super::modules;
4use crate::{
5    prelude::*,
6    process::{UserNamespace, credentials::capabilities::CapSet, posix_thread::PosixThread},
7};
8
9/// Runs capability hooks in module order.
10pub fn on_capable(context: CapableContext) -> Result<()> {
11    for module in modules::active_modules() {
12        module.on_capable(&context)?;
13    }
14
15    Ok(())
16}
17
18/// The inputs for checking whether a thread has a capability.
19pub struct CapableContext<'a> {
20    target_user_ns: &'a UserNamespace,
21    posix_thread: &'a PosixThread,
22    required_cap: CapSet,
23}
24
25impl<'a> CapableContext<'a> {
26    /// Creates a capability check context.
27    pub const fn new(
28        target_user_ns: &'a UserNamespace,
29        posix_thread: &'a PosixThread,
30        required_cap: CapSet,
31    ) -> Self {
32        Self {
33            target_user_ns,
34            posix_thread,
35            required_cap,
36        }
37    }
38
39    /// Returns the user namespace against which the capability is checked.
40    #[expect(
41        dead_code,
42        reason = "Asterinas currently has only the initial user namespace"
43    )]
44    pub const fn target_user_ns(&self) -> &UserNamespace {
45        self.target_user_ns
46    }
47
48    /// Returns the thread whose credentials are checked.
49    pub const fn posix_thread(&self) -> &PosixThread {
50        self.posix_thread
51    }
52
53    /// Returns the required capability.
54    pub const fn required_cap(&self) -> CapSet {
55        self.required_cap
56    }
57}
