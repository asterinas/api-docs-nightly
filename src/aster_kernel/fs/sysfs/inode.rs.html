3use alloc::sync::{Arc, Weak};
4
5use ostd::sync::RwLock;
6
7use super::fs::SysFs;
8use crate::{
9    fs::utils::{
10        Extension, FileSystem, Inode, InodeMode, InodeType, Metadata,
11        systree_inode::{SysTreeInodeTy, SysTreeNodeKind},
12    },
13    prelude::*,
14};
15
16/// An inode abstraction used in the sysfs file system.
17pub(super) struct SysFsInode {
18    /// The corresponding node in the SysTree.
19    node_kind: SysTreeNodeKind,
20    /// The metadata of this inode.
21    ///
22    /// Most of the metadata (e.g., file size, timestamps)
23    /// can be determined upon the creation of an inode,
24    /// and are thus kept intact inside the immutable `metadata` field.
25    /// Currently, the only mutable metadata is `mode`,
26    /// which allows user space to `chmod` an inode on sysfs.
27    metadata: Metadata,
28    /// The extension of this inode.
29    extension: Extension,
30    /// The file mode (permissions) of this inode, protected by a lock.
31    mode: RwLock<InodeMode>,
32    /// Weak reference to the parent inode.
33    parent: Weak<SysFsInode>,
34    /// Weak self-reference for cyclic data structures.
35    this: Weak<SysFsInode>,
36}
37
38impl SysTreeInodeTy for SysFsInode {
39    fn new_arc(
40        node_kind: SysTreeNodeKind,
41        metadata: Metadata,
42        mode: InodeMode,
43        parent: Weak<Self>,
44    ) -> Arc<Self>
45    where
46        Self: Sized,
47    {
48        Arc::new_cyclic(|this| Self {
49            node_kind,
50            metadata,
51            extension: Extension::new(),
52            mode: RwLock::new(mode),
53            parent,
54            this: this.clone(),
55        })
56    }
57
58    fn node_kind(&self) -> &SysTreeNodeKind {
59        &self.node_kind
60    }
61
62    fn metadata(&self) -> &Metadata {
63        &self.metadata
64    }
65
66    fn mode(&self) -> Result<InodeMode> {
67        Ok(*self.mode.read())
68    }
69
70    fn set_mode(&self, mode: InodeMode) -> Result<()> {
71        *self.mode.write() = mode;
72        Ok(())
73    }
74
75    fn parent(&self) -> &Weak<Self> {
76        &self.parent
77    }
78
79    fn this(&self) -> Arc<Self> {
80        self.this
81            .upgrade()
82            .expect("invalid weak reference to `self`")
83    }
84
85    fn extension(&self) -> &Extension {
86        &self.extension
87    }
88}
89
90impl Inode for SysFsInode {
91    fn fs(&self) -> Arc<dyn FileSystem> {
92        SysFs::singleton().clone()
93    }
94
95    fn create(&self, _name: &str, _type_: InodeType, _mode: InodeMode) -> Result<Arc<dyn Inode>> {
96        Err(Error::with_message(
97            Errno::EPERM,
98            "file creation under sysfs is not allowed",
99        ))
100    }
101}
