13/// Registers a new kernel `SysNode`.
14pub(super) fn register(config_obj: Arc<dyn SysNode>) -> crate::prelude::Result<()> {
15    KERNEL_SYS_NODE_ROOT.get().unwrap().add_child(config_obj)?;
16    Ok(())
17}
18
19/// Unregisters a kernel `SysNode`.
20pub(super) fn unregister(name: &str) -> crate::prelude::Result<()> {
21    let _ = KERNEL_SYS_NODE_ROOT.get().unwrap().remove_child(name)?;
22    Ok(())
23}
24
25pub(super) fn init() {
26    KERNEL_SYS_NODE_ROOT.call_once(|| {
27        let singleton = KernelSysNodeRoot::new();
28        super::systree_singleton()
29            .root()
30            .add_child(singleton.clone())
31            .unwrap();
32
33        singleton
34    });
35}

39/// A systree node representing the `/sys/kernel` directory.
40///
41/// This node serves as the root for all kernel-related sysfs entries,
42/// including kernel parameters, debugging interfaces, and various
43/// kernel subsystem information. It corresponds to the `/kernel`
44/// directory in the sysfs filesystem.
45#[derive(Debug)]
46pub struct KernelSysNodeRoot {
47    fields: BranchNodeFields<dyn SysNode, Self>,
48}
49
50#[inherit_methods(from = "self.fields")]
51impl KernelSysNodeRoot {
52    /// Creates a new `KernelSysNodeRoot` instance.
53    fn new() -> Arc<Self> {
54        let name = SysStr::from("kernel");
55        let builder = SysAttrSetBuilder::new();
56        // TODO: Add more kernel-specific attributes.
57        let attrs = builder
58            .build()
59            .expect("Failed to build kernel attribute set");
60        Arc::new_cyclic(|weak_self| {
61            let fields = BranchNodeFields::new(name, attrs, weak_self.clone());
62            KernelSysNodeRoot { fields }
63        })
64    }
65
66    /// Adds a kernel `SysNode` to this node.
67    fn add_child(&self, new_child: Arc<dyn SysNode>) -> Result<()>;
68}
69
70inherit_sys_branch_node!(KernelSysNodeRoot, fields, {
71    fn read_attr(&self, _name: &str, _writer: &mut VmWriter) -> Result<usize> {
72        // TODO: Add support for reading attributes.
73        Err(Error::AttributeError)
74    }
75
76    fn write_attr(&self, _name: &str, _reader: &mut VmReader) -> Result<usize> {
77        // TODO: Add support for writing attributes.
78        Err(Error::AttributeError)
79    }
80
81    fn perms(&self) -> SysPerms {
82        SysPerms::DEFAULT_RW_PERMS
83    }
84});
