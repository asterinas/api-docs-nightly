20/// An inode abstraction used in the cgroup file system.
21pub(super) struct CgroupInode {
22    /// The corresponding node in the SysTree.
23    node_kind: SysTreeNodeKind,
24    /// The metadata of this inode.
25    metadata: Metadata,
26    /// The extension of this inode.
27    extension: Extension,
28    /// The file mode (permissions) of this inode, protected by a lock.
29    mode: RwLock<InodeMode>,
30    /// Weak reference to the parent inode.
31    parent: Weak<CgroupInode>,
32    /// Weak self-reference for cyclic data structures.
33    this: Weak<CgroupInode>,
34}
35
36impl SysTreeInodeTy for CgroupInode {
37    fn new_arc(
38        node_kind: SysTreeNodeKind,
39        metadata: Metadata,
40        mode: InodeMode,
41        parent: Weak<Self>,
42    ) -> Arc<Self>
43    where
44        Self: Sized,
45    {
46        Arc::new_cyclic(|this| Self {
47            node_kind,
48            metadata,
49            extension: Extension::new(),
50            mode: RwLock::new(mode),
51            parent,
52            this: this.clone(),
53        })
54    }
55
56    fn node_kind(&self) -> &SysTreeNodeKind {
57        &self.node_kind
58    }
59
60    fn metadata(&self) -> &Metadata {
61        &self.metadata
62    }
63
64    fn extension(&self) -> &Extension {
65        &self.extension
66    }
67
68    fn mode(&self) -> Result<InodeMode> {
69        Ok(*self.mode.read())
70    }
71
72    fn set_mode(&self, mode: InodeMode) -> Result<()> {
73        *self.mode.write() = mode;
74        Ok(())
75    }
76
77    fn parent(&self) -> &Weak<Self> {
78        &self.parent
79    }
80
81    fn this(&self) -> Arc<Self> {
82        self.this
83            .upgrade()
84            .expect("invalid weak reference to `self`")
85    }
86}
87
88impl Inode for CgroupInode {
89    fn fs(&self) -> Arc<dyn FileSystem> {
90        CgroupFs::singleton().clone()
91    }
92
93    fn rmdir(&self, name: &str) -> Result<()> {
94        if is_dot(name) {
95            return_errno_with_message!(Errno::EINVAL, "rmdir on .");
96        }
97        if is_dotdot(name) {
98            return_errno_with_message!(Errno::ENOTEMPTY, "rmdir on ..");
99        }
100
101        let SysTreeNodeKind::Branch(branch_node) = self.node_kind() else {
102            return_errno_with_message!(Errno::ENOTDIR, "the current node is not a branch node");
103        };
104
105        let Some(child) = branch_node.child(name) else {
106            return_errno_with_message!(Errno::ENOENT, "the child node does not exist");
107        };
108
109        let target_node = child.as_any().downcast_ref::<CgroupNode>().unwrap();
110
111        // This will succeed only if the child is empty and has not been removed.
112        target_node.mark_as_dead()?;
113
114        // This is guaranteed to remove `child` because the dentry lock prevents
115        // concurrent modification to the children, and there are no races because
116        // `mark_as_dead` can succeed at most once.
117        branch_node.remove_child(name).unwrap();
118
119        Ok(())
120    }
121
122    fn is_dentry_cacheable(&self) -> bool {
123        // Attribute nodes should not be cached because they may be dynamically
124        // created or removed based on the state of the cgroup controller.
125        // Caching them could result in stale or incorrect entries.
126        !matches!(self.node_kind, SysTreeNodeKind::Attr(..))
127    }
128}
