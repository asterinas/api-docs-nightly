40/// A file-like object that provides inotify functionality.
41///
42/// `InotifyFile` accepts events from multiple inotify subscribers (watches) on different inodes.
43/// Users should read events from this file to receive notifications about filesystem changes.
44pub struct InotifyFile {
45    // The next watch descriptor to allocate.
46    next_wd: AtomicU32,
47    // A map from watch descriptors to subscriber entries.
48    watch_map: SpinLock<HashMap<u32, SubscriberEntry>>,
49    // A mutex to synchronize `read()` operations.
50    read_mutex: Mutex<()>,
51    // Whether the file is opened in non-blocking mode.
52    is_nonblocking: AtomicBool,
53    // A bounded queue of inotify events.
54    event_queue: SpinLock<VecDeque<InotifyEvent>>,
55    // The maximum capacity of the event queue.
56    queue_capacity: usize,
57    // A pollable object for this inotify file.
58    pollee: Pollee,
59    // A weak reference to this inotify file.
60    this: Weak<InotifyFile>,
61    /// The pseudo path associated with this inotify file.
62    pseudo_path: Path,
63}
64
65impl Drop for InotifyFile {
66    /// Cleans up all subscribers when the inotify file is dropped.
67    ///
68    /// This will remove all subscribers from their inodes.
69    fn drop(&mut self) {
70        let watch_map = self.watch_map.get_mut();
71
72        for (_, entry) in watch_map.drain() {
73            let (Some(inode), Some(subscriber)) =
74                (entry.inode.upgrade(), entry.subscriber.upgrade())
75            else {
76                continue;
77            };
78
79            if inode
80                .fs_event_publisher()
81                .unwrap()
82                .remove_subscriber(&(subscriber as _))
83            {
84                inode.fs().fs_event_subscriber_stats().remove_subscriber();
85            }
86        }
87    }
88}

90/// The default maximum capacity of the event queue.
91///
92/// Reference: <https://elixir.bootlin.com/linux/v6.14/source/fs/notify/inotify/inotify_user.c#L853>
93const DEFAULT_MAX_QUEUED_EVENTS: usize = 16384;

95impl InotifyFile {
96    /// Creates a new inotify file.
97    pub fn new(is_nonblocking: bool) -> Result<Arc<Self>> {
98        let pseudo_path = AnonInodeFs::new_path(|_| "anon_inode:inotify".to_string());
99
100        Ok(Arc::new_cyclic(|weak_self| Self {
101            // Allocate watch descriptors from 1.
102            // Reference: <https://elixir.bootlin.com/linux/v6.17/source/fs/notify/inotify/inotify_user.c#L402>
103            next_wd: AtomicU32::new(1),
104            watch_map: SpinLock::new(HashMap::new()),
105            read_mutex: Mutex::new(()),
106            is_nonblocking: AtomicBool::new(is_nonblocking),
107            event_queue: SpinLock::new(VecDeque::new()),
108            queue_capacity: DEFAULT_MAX_QUEUED_EVENTS,
109            pollee: Pollee::new(),
110            this: weak_self.clone(),
111            pseudo_path,
112        }))
113    }
114
115    /// Allocates a new watch descriptor.
116    fn alloc_wd(&self) -> Result<u32> {
117        const MAX_VALID_WD: u32 = i32::MAX as u32;
118
119        let new_wd = self.next_wd.fetch_add(1, Ordering::Relaxed);
120        if new_wd > MAX_VALID_WD {
121            // Roll back the allocation if we exceed the limit.
122            self.next_wd.fetch_sub(1, Ordering::Relaxed);
123            return_errno_with_message!(Errno::ENOSPC, "the inotify watch limit is reached");
124        }
125        Ok(new_wd)
126    }
127
128    /// Adds or updates a watch on a path.
129    ///
130    /// If a watch on the path is not found, creates a new watch.
131    /// If a watch on the path is found, updates it.
132    pub fn add_watch(
133        &self,
134        path: &Path,
135        interesting: InotifyEvents,
136        options: InotifyControls,
137    ) -> Result<u32> {
138        let mut watch_map = self.watch_map.lock();
139
140        // Try to find and update the existing subscriber first.
141        let inode_weak = Arc::downgrade(path.inode());
142        let mut watch_iter = watch_map.iter();
143        while let Some((wd, entry)) = watch_iter.next() {
144            if !Weak::ptr_eq(&entry.inode, &inode_weak) {
145                continue;
146            }
147
148            // The subscriber is dead because it's a one-shot subscriber or the inode is dead.
149            // The watch is considered removed.
150            let Some(subscriber) = entry.subscriber.upgrade() else {
151                let wd = *wd;
152                watch_map.remove(&wd);
153                watch_iter = watch_map.iter();
154                continue;
155            };
156
157            subscriber.update(interesting, options)?;
158            path.inode()
159                .fs_event_publisher()
160                .unwrap()
161                .update_subscriber_events();
162
163            return Ok(*wd);
164        }
165
166        // Create a new subscriber and register it.
167
168        let inotify_subscriber = InotifySubscriber::new(self.this(), interesting, options)?;
169        let subscriber = inotify_subscriber.clone() as Arc<dyn FsEventSubscriber>;
170
171        let inode = path.inode();
172        if !inode
173            .fs_event_publisher_or_init()
174            .add_subscriber(subscriber)
175        {
176            // FIXME: This can be triggered by `unlink()` race conditions or
177            // `inotify_add_watch("/proc/self/fd/[fd]")`, where `[fd]` is a file descriptor
178            // pointing to a deleted file. Although Linux allows this, we don't support it, so an
179            // error is returned here.
180            return_errno_with_message!(
181                Errno::ENOENT,
182                "adding an inotify watch to a deleted inode is not supported yet"
183            );
184        }
185        inode.fs().fs_event_subscriber_stats().add_subscriber();
186
187        let wd = inotify_subscriber.wd();
188        let entry = SubscriberEntry {
189            inode: inode_weak,
190            subscriber: Arc::downgrade(&inotify_subscriber),
191        };
192        watch_map.insert(wd, entry);
193
194        Ok(wd)
195    }
196
197    /// Removes a watch by watch descriptor.
198    pub fn remove_watch(&self, wd: u32) -> Result<()> {
199        let mut watch_map = self.watch_map.lock();
200
201        let Some(entry) = watch_map.remove(&wd) else {
202            return_errno_with_message!(Errno::EINVAL, "the inotify watch does not exist");
203        };
204
205        let (inode, subscriber) = match (entry.inode.upgrade(), entry.subscriber.upgrade()) {
206            (Some(inode), Some(subscriber)) => (inode, subscriber),
207            // The subscriber is dead because it's a one-shot subscriber or the inode is dead.
208            // The watch is considered removed.
209            _ => return_errno_with_message!(Errno::EINVAL, "the inotify watch does not exist"),
210        };
211
212        if inode
213            .fs_event_publisher()
214            .unwrap()
215            .remove_subscriber(&(subscriber as _))
216        {
217            inode.fs().fs_event_subscriber_stats().remove_subscriber();
218        }
219
220        Ok(())
221    }

223    /// Sends an inotify event to the inotify file.
224    ///
225    /// The event will be queued and can be read by users.
226    /// If the event can be merged with the last event in the queue, it will be merged.
227    /// The event is only queued if it matches one of the subscriber's interesting events.
228    fn receive_event(&self, wd: u32, event: FsEvents, name: Option<String>) {
229        let new_event = InotifyEvent::new(wd, event, 0, name);
230
231        'notify: {
232            let mut event_queue = self.event_queue.lock();
233            if let Some(last_event) = event_queue.back()
234                && can_merge_events(last_event, &new_event)
235            {
236                event_queue.pop_back();
237                event_queue.push_back(new_event);
238                break 'notify;
239            }
240
241            // If the queue is full, drop the event.
242            // Try to queue an overflow event in advance to alert the user.
243            if event_queue.len() == self.queue_capacity - 1 {
244                let overflow_event = InotifyEvent::new(u32::MAX, FsEvents::Q_OVERFLOW, 0, None);
245                event_queue.push_back(overflow_event);
246                return;
247            } else if event_queue.len() >= self.queue_capacity {
248                return;
249            }
250
251            event_queue.push_back(new_event);
252        }
253        // The new event or the merged event makes the file readable.
254        self.pollee.notify(IoEvents::IN);
255    }
256
257    /// Pops an event from the notification queue.
258    fn pop_event(&self) -> Option<InotifyEvent> {
259        let mut event_queue = self.event_queue.lock();
260
261        let event = event_queue.pop_front();
262        // Invalidate when the queue is empty.
263        if event_queue.is_empty() {
264            self.pollee.invalidate();
265        }
266
267        event
268    }
269
270    /// Gets the total size of all events in the notification queue.
271    fn get_all_event_size(&self) -> usize {
272        let event_queue = self.event_queue.lock();
273
274        event_queue.iter().map(|event| event.total_size()).sum()
275    }
276
277    /// Tries to read events from the notification queue.
278    fn try_read(&self, writer: &mut VmWriter) -> Result<usize> {
279        // This ensures that we report continuous events even when the user program attempts to
280        // call `read()` concurrently.
281        let _guard = self.read_mutex.lock();
282
283        let mut size = 0;
284        let mut consumed_events = 0;
285
286        while let Some(event) = self.pop_event() {
287            match event.copy_to_user(writer) {
288                Ok(event_size) => {
289                    size += event_size;
290                    consumed_events += 1;
291                }
292                Err(err) => {
293                    // This won't reorder events due to `_guard`.
294                    self.event_queue.lock().push_front(event);
295                    if consumed_events == 0 {
296                        return Err(err);
297                    }
298                    return Ok(size);
299                }
300            }
301        }
302
303        if consumed_events == 0 {
304            return_errno_with_message!(Errno::EAGAIN, "no inotify events are available");
305        }
306
307        Ok(size)
308    }
309
310    fn check_io_events(&self) -> IoEvents {
311        if self.event_queue.lock().is_empty() {
312            IoEvents::empty()
313        } else {
314            IoEvents::IN
315        }
316    }
317
318    fn this(&self) -> Arc<InotifyFile> {
319        self.this.upgrade().unwrap()
320    }
321}

377    fn dump_proc_fdinfo(self: Arc<Self>, fd_flags: FdFlags) -> Box<dyn Display> {
378        struct FdInfo {
379            inner: Arc<InotifyFile>,
380            fd_flags: FdFlags,
381        }
382
383        impl Display for FdInfo {
384            fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
385                let mut flags = self.inner.status_flags().bits() | self.inner.access_mode() as u32;
386                if self.fd_flags.contains(FdFlags::CLOEXEC) {
387                    flags |= CreationFlags::O_CLOEXEC.bits();
388                }
389
390                writeln!(f, "pos:\t{}", 0)?;
391                writeln!(f, "flags:\t0{:o}", flags)?;
392                writeln!(f, "mnt_id:\t{}", AnonInodeFs::mount_node().id())?;
393                writeln!(f, "ino:\t{}", AnonInodeFs::shared_inode().ino())?;
394
395                for (wd, entry) in self.inner.watch_map.lock().iter() {
396                    let Some(inode) = entry.inode.upgrade() else {
397                        continue;
398                    };
399                    let Some(subscriber) = entry.subscriber.upgrade() else {
400                        continue;
401                    };
402                    let mask = subscriber.interesting_events().bits();
403                    let sdev = inode.fs().sb().fsid;
404                    writeln!(
405                        f,
406                        "inotify wd:{} ino:{:x} sdev:{:x} mask:{:x} ignored_mask:0",
407                        wd,
408                        inode.ino(),
409                        sdev,
410                        mask
411                    )?;
412                }
413
414                Ok(())
415            }
416        }
417
418        Box::new(FdInfo {
419            inner: self,
420            fd_flags,
421        })
422    }

425/// Checks if the event type is mergeable.
426fn is_mergeable_event_type(event: u32) -> bool {
427    event & (FsEvents::MODIFY | FsEvents::ATTRIB | FsEvents::ACCESS).bits() != 0
428}
429
430/// Checks if two inotify events can be merged.
431fn can_merge_events(existing: &InotifyEvent, new_event: &InotifyEvent) -> bool {
432    existing.header.wd == new_event.header.wd
433        && existing.name == new_event.name
434        && existing.header.event == new_event.header.event
435        && is_mergeable_event_type(new_event.header.event)
436}

438/// Represents a watch on a file or directory in the inotify system.
439///
440/// In the inotify implementation, a watch is equivalent to a subscriber. The subscriber
441/// specifies which events it wants to monitor using `InotifyEvents`, and control options
442/// using `InotifyControls`. Both the event mask and control options are stored in a single
443/// `AtomicU64` for atomic updates: the high 32 bits store options, and the low 32 bits
444/// store the event mask.
445pub struct InotifySubscriber {
446    // Interesting events and control options.
447    //
448    // This field is packed into a `u64`: the high 32 bits store options,
449    // and the low 32 bits store interesting events.
450    interesting_and_controls: AtomicU64,
451    // Whether the subscriber is one-shot and has been dead.
452    is_dead: AtomicBool,
453    // Watch descriptor.
454    wd: u32,
455    // Reference to the owning inotify file.
456    inotify_file: Arc<InotifyFile>,
457}
458
459impl InotifySubscriber {
460    /// Creates a new `InotifySubscriber` with initial interesting events and options.
461    pub fn new(
462        inotify_file: Arc<InotifyFile>,
463        interesting: InotifyEvents,
464        options: InotifyControls,
465    ) -> Result<Arc<Self>> {
466        let wd = inotify_file.alloc_wd()?;
467        let this = Arc::new(Self {
468            interesting_and_controls: AtomicU64::new(0),
469            is_dead: AtomicBool::new(false),
470            wd,
471            inotify_file,
472        });
473        // Initialize the `interesting_and_controls` field.
474        this.update_interesting_and_controls(interesting.bits(), options.bits());
475        Ok(this)
476    }
477
478    pub fn wd(&self) -> u32 {
479        self.wd
480    }
481
482    /// Loads the interesting events and options atomically.
483    fn interesting_and_controls(&self) -> (InotifyEvents, InotifyControls) {
484        let flags = self.interesting_and_controls.load(Ordering::Relaxed);
485        (
486            InotifyEvents::from_bits_truncate((flags & 0xFFFFFFFF) as u32),
487            InotifyControls::from_bits_truncate((flags >> 32) as u32),
488        )
489    }
490
491    pub fn inotify_file(&self) -> &Arc<InotifyFile> {
492        &self.inotify_file
493    }
494
495    /// Updates the interesting events and options atomically.
496    fn update(&self, interesting: InotifyEvents, options: InotifyControls) -> Result<()> {
497        if options.contains(InotifyControls::MASK_CREATE) {
498            return_errno_with_message!(Errno::EEXIST, "the inotify watch already exists");
499        }
500
501        let mut merged_interesting = interesting;
502        let mut merged_options = options;
503
504        if options.contains(InotifyControls::MASK_ADD) {
505            // There are no races because `update()` is only called under the `watch_map` lock.
506            let (old_interesting, old_options) = self.interesting_and_controls();
507            merged_interesting |= old_interesting;
508            merged_options |= old_options;
509        }
510        merged_options.remove(InotifyControls::MASK_ADD);
511
512        self.update_interesting_and_controls(merged_interesting.bits(), merged_options.bits());
513
514        Ok(())
515    }
516
517    /// Updates the interesting events and options atomically with raw bits.
518    fn update_interesting_and_controls(&self, new_interesting: u32, new_options: u32) {
519        let new_flags = ((new_options as u64) << 32) | (new_interesting as u64);
520        self.interesting_and_controls
521            .store(new_flags, Ordering::Relaxed);
522    }
523}

525impl FsEventSubscriber for InotifySubscriber {
526    /// Sends FS events to the inotify file.
527    fn deliver_event(&self, event: FsEvents, name: Option<String>) -> bool {
528        let (interesting, options) = self.interesting_and_controls();
529
530        if !event.contains(FsEvents::IN_IGNORED) && !is_interesting(interesting, event) {
531            return false;
532        }
533
534        let (is_dead, is_oneshot) = if options.contains(InotifyControls::ONESHOT) {
535            (self.is_dead.swap(true, Ordering::Relaxed), true)
536        } else {
537            (self.is_dead.load(Ordering::Relaxed), false)
538        };
539        if is_dead {
540            return false;
541        }
542
543        let inotify_file = self.inotify_file();
544        let wd = self.wd();
545
546        inotify_file.receive_event(wd, event, name);
547
548        if !event.contains(FsEvents::IN_IGNORED) && is_oneshot {
549            inotify_file.receive_event(wd, FsEvents::IN_IGNORED, None);
550        }
551
552        is_oneshot
553    }
554
555    /// Returns the events that this subscriber is interested in.
556    fn interesting_events(&self) -> FsEvents {
557        let inotify_events = self.interesting_and_controls().0;
558        FsEvents::from_bits_truncate(inotify_events.bits())
559    }
560
561    /// Returns whether the subscriber is one-shot and dead.
562    fn is_oneshot_and_dead(&self) -> bool {
563        self.is_dead.load(Ordering::Relaxed)
564    }
565}

567/// Checks if the event matches the subscriber's interesting events.
568fn is_interesting(interesting: InotifyEvents, event: FsEvents) -> bool {
569    interesting.bits() & event.bits() != 0
570}

572/// Represents an inotify event that can be read by users.
573struct InotifyEvent {
574    header: InotifyEventHeader,
575    name: Option<String>,
576}
577
578/// The header of an inotify event.
579///
580/// Reference: <https://elixir.bootlin.com/linux/v6.17.8/source/include/uapi/linux/inotify.h#L21>
581#[repr(C)]
582#[derive(Clone, Copy, Pod)]
583struct InotifyEventHeader {
584    wd: u32,
585    event: u32,
586    cookie: u32,
587    name_len: u32,
588}
589
590impl InotifyEvent {
591    fn new(wd: u32, event: FsEvents, cookie: u32, name: Option<String>) -> Self {
592        // Calculate the actual name length including the null terminator.
593        let actual_name_len = name.as_ref().map_or(0, |name| name.len() + 1);
594        // Calculate the padded name length aligned to `size_of::<InotifyEventHeader>`.
595        let pad_name_len = Self::round_event_name_len(actual_name_len);
596
597        let header = InotifyEventHeader {
598            wd,
599            event: event.bits(),
600            cookie,
601            name_len: pad_name_len as u32,
602        };
603        Self { header, name }
604    }
605
606    /// Rounds up the name length to align with `size_of::<InotifyEventHeader>()`.
607    ///
608    /// Reference: <https://elixir.bootlin.com/linux/v6.17.8/source/fs/notify/inotify/inotify_user.c#L160>
609    fn round_event_name_len(name_len: usize) -> usize {
610        const HEADER_SIZE: usize = size_of::<InotifyEventHeader>(); // 16 bytes
611        const { assert!(HEADER_SIZE.is_power_of_two()) };
612        name_len.align_up(HEADER_SIZE)
613    }
614
615    fn total_size(&self) -> usize {
616        const HEADER_SIZE: usize = size_of::<InotifyEventHeader>(); // 16 bytes
617        HEADER_SIZE + (self.header.name_len as usize)
618    }
619
620    fn copy_to_user(&self, writer: &mut VmWriter) -> Result<usize> {
621        let total_size = self.total_size();
622        if total_size > writer.avail() {
623            return_errno_with_message!(Errno::EINVAL, "the buffer is too small");
624        }
625
626        // Write the header.
627        writer.write_val(&self.header)?;
628
629        let Some(name) = self.name.as_ref() else {
630            debug_assert_eq!(self.header.name_len, 0);
631            return Ok(total_size);
632        };
633        // Write the actual name bytes.
634        writer.write_fallible(&mut VmReader::from(name.as_bytes()).to_fallible())?;
635        // Fill remaining bytes with zeros for alignment.
636        // Note that this also includes the null terminator.
637        writer.fill_zeros((self.header.name_len as usize) - name.len())?;
638
639        Ok(total_size)
640    }
641}

643bitflags! {
644    /// Represents the set of events that a subscriber wants to monitor.
645    ///
646    /// These events are used to filter notifications sent to the subscriber.
647    pub struct InotifyEvents: u32 {
648        const ACCESS        = 1 << 0;  // File was accessed
649        const MODIFY        = 1 << 1;  // File was modified
650        const ATTRIB        = 1 << 2;  // Metadata changed
651        const CLOSE_WRITE   = 1 << 3;  // Writable file was closed
652        const CLOSE_NOWRITE = 1 << 4;  // Unwritable file closed
653        const OPEN          = 1 << 5;  // File was opened
654        const MOVED_FROM    = 1 << 6;  // File was moved from X
655        const MOVED_TO      = 1 << 7;  // File was moved to Y
656        const CREATE        = 1 << 8;  // Subfile was created
657        const DELETE        = 1 << 9;  // Subfile was deleted
658        const DELETE_SELF   = 1 << 10; // Self was deleted
659        const MOVE_SELF     = 1 << 11; // Self was moved
660        const UNMOUNT       = 1 << 13; // Backing fs was unmounted
661        const Q_OVERFLOW    = 1 << 14; // Event queue overflowed
662        const IGNORED       = 1 << 15; // File was ignored
663        const CLOSE         = Self::CLOSE_WRITE.bits() | Self::CLOSE_NOWRITE.bits(); // Close events
664        const MOVE          = Self::MOVED_FROM.bits() | Self::MOVED_TO.bits();       // Move events
665        const ALL_EVENTS    = Self::ACCESS.bits() | Self::MODIFY.bits() | Self::ATTRIB.bits() |
666                             Self::CLOSE_WRITE.bits() | Self::CLOSE_NOWRITE.bits() | Self::OPEN.bits() |
667                             Self::MOVED_FROM.bits() | Self::MOVED_TO.bits() | Self::DELETE.bits() |
668                             Self::CREATE.bits() | Self::DELETE_SELF.bits() | Self::MOVE_SELF.bits();
669    }
