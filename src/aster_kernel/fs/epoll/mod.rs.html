11/// An epoll control command.
12#[derive(Copy, Clone, Debug, PartialEq)]
13pub enum EpollCtl {
14    Add(FileDesc, EpollEvent, EpollFlags),
15    Del(FileDesc),
16    Mod(FileDesc, EpollEvent, EpollFlags),
17}
18
19bitflags! {
20    /// Linux's epoll flags.
21    pub struct EpollFlags: u32 {
22        const EXCLUSIVE      = (1 << 28);
23        const WAKE_UP        = (1 << 29);
24        const ONE_SHOT       = (1 << 30);
25        const EDGE_TRIGGER   = (1 << 31);
26    }
27}
28
29/// An epoll event.
30///
31/// This could be used as either an input of epoll ctl or an output of epoll wait.
32/// The memory layout is compatible with that of C's struct epoll_event.
33#[derive(Copy, Clone, Debug, PartialEq)]
34pub struct EpollEvent {
35    /// I/O events.
36    ///
37    /// When `EpollEvent` is used as inputs, this is treated as a mask of events.
38    /// When `EpollEvent` is used as outputs, this is the active events.
39    pub events: IoEvents,
40    /// A 64-bit, user-given data.
41    pub user_data: u64,
42}
43
44impl EpollEvent {
45    /// Create a new epoll event.
46    pub fn new(events: IoEvents, user_data: u64) -> Self {
47        Self { events, user_data }
48    }
49}
