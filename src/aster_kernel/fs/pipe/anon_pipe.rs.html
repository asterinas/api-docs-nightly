21/// Creates a pair of connected pipe file handles with the default capacity.
22pub fn new_file_pair() -> Result<(Arc<InodeHandle>, Arc<InodeHandle>)> {
23    let pipe_inode = Arc::new(AnonPipeInode::new());
24    let path = PipeFs::new_path(pipe_inode);
25
26    let reader = InodeHandle::new_unchecked_access(
27        path.clone(),
28        AccessMode::O_RDONLY,
29        StatusFlags::empty(),
30    )?;
31    let writer =
32        InodeHandle::new_unchecked_access(path, AccessMode::O_WRONLY, StatusFlags::empty())?;
33
34    Ok((Arc::new(reader), Arc::new(writer)))
35}

37/// An anonymous pipe inode.
38pub struct AnonPipeInode {
39    /// The underlying pipe backend.
40    pipe: Pipe,
41    pseudo_inode: PseudoInode,
42}
43
44impl AnonPipeInode {
45    fn new() -> Self {
46        let pipe = Pipe::new();
47
48        let pseudo_inode = PipeFs::singleton().alloc_inode(
49            PseudoInodeType::Pipe,
50            mkmod!(u+rw),
51            Uid::new_root(),
52            Gid::new_root(),
53        );
54
55        Self { pipe, pseudo_inode }
56    }
57}
58
59#[inherit_methods(from = "self.pseudo_inode")]
60impl InodeIo for AnonPipeInode {
61    fn read_at(
62        &self,
63        _offset: usize,
64        _writer: &mut VmWriter,
65        _status: StatusFlags,
66    ) -> Result<usize>;
67    fn write_at(
68        &self,
69        _offset: usize,
70        _reader: &mut VmReader,
71        _status: StatusFlags,
72    ) -> Result<usize>;
73}
74
75#[inherit_methods(from = "self.pseudo_inode")]
76impl Inode for AnonPipeInode {
77    fn size(&self) -> usize;
78    fn resize(&self, _new_size: usize) -> Result<()>;
79    fn metadata(&self) -> Metadata;
80    fn extension(&self) -> &Extension;
81    fn ino(&self) -> u64;
82    fn type_(&self) -> InodeType;
83    fn mode(&self) -> Result<InodeMode>;
84    fn set_mode(&self, mode: InodeMode) -> Result<()>;
85    fn owner(&self) -> Result<Uid>;
86    fn set_owner(&self, uid: Uid) -> Result<()>;

89    fn atime(&self) -> Duration;

91    fn mtime(&self) -> Duration;
92    fn set_mtime(&self, time: Duration);
93    fn ctime(&self) -> Duration;
94    fn set_ctime(&self, time: Duration);
95    fn fs(&self) -> Arc<dyn FileSystem>;
96
97    fn open(
98        &self,
99        access_mode: AccessMode,
100        status_flags: StatusFlags,
101    ) -> Option<Result<Box<dyn FileIo>>> {
102        Some(self.pipe.open_anon(access_mode, status_flags))
103    }
