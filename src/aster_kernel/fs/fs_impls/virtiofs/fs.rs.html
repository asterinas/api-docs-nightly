27/// Filesystem magic reported for virtio-fs in `statfs`.
28const VIRTIOFS_MAGIC: u64 = 0x6573_5546;
29
30/// Block size reported to `statfs` for virtio-fs.
31const BLOCK_SIZE: usize = 4096;
32
33/// The `virtiofs` filesystem type.
34pub(super) struct VirtioFsType;
35
36impl FsType for VirtioFsType {
37    fn name(&self) -> &'static str {
38        "virtiofs"
39    }
40
41    fn properties(&self) -> FsProperties {
42        FsProperties::empty()
43    }
44
45    fn create(&self, fs_creation_ctx: &FsCreationCtx) -> Result<Arc<dyn FileSystem>> {
46        let tag = fs_creation_ctx
47            .source()
48            .ok_or_else(|| Error::with_message(Errno::EINVAL, "virtiofs source(tag) is required"))?
49            .to_string();
50
51        let device = device::find_device_by_tag(&tag)
52            .ok_or_else(|| Error::with_message(Errno::ENODEV, "virtiofs device is not found"))?;
53
54        Ok(VirtioFs::new(device, tag)? as Arc<dyn FileSystem>)
55    }
56
57    fn sysnode(&self) -> Option<Arc<dyn aster_systree::SysNode>> {
58        None
59    }
60}

62/// A mounted virtio-fs filesystem.
63pub(super) struct VirtioFs {
64    sb: SuperBlock,
65    root: Arc<VirtioFsInode>,
66    tag: String,
67    session: Arc<FuseSession>,
68    inode_cache: InodeCache,
69    fs_event_subscriber_stats: FsEventSubscriberStats,
70}
71
72impl VirtioFs {
73    fn new(device: Arc<FileSystemDevice>, tag: String) -> Result<Arc<Self>> {
74        let session = FuseSession::new(device)
75            .map_err(|_| Error::with_message(Errno::EIO, "virtiofs FUSE_INIT failed"))?;
76
77        let anon_device_id =
78            AnonDeviceId::acquire().expect("no device ID is available for virtiofs");
79        let container_dev_id = anon_device_id.id();
80        let statfs = session.do_fuse_op(FUSE_ROOT_ID, StatfsOperation)?.st();
81
82        // TODO: Update the super block fields based on `statfs` reply.
83        // For now, we set only the fields required by VFS when mounting the filesystem.
84        // No update is made for these fields later.
85        let sb = SuperBlock::from((container_dev_id, statfs));
86
87        let root_entry = session.do_fuse_op(FUSE_ROOT_ID, LookupOperation::new("."))?;
88
89        Ok(Arc::new_cyclic(|weak_fs| {
90            let root = VirtioFsInode::new_root(
91                root_entry,
92                weak_fs.clone(),
93                container_dev_id,
94                session.bump_attr_version(),
95            );
96            let inode_cache = InodeCache::new(&root);
97
98            Self {
99                sb,
100                root,
101                tag,
102                session,
103                inode_cache,
104                fs_event_subscriber_stats: FsEventSubscriberStats::new(),
105            }
106        }))
107    }
108
109    pub(super) fn session(&self) -> &Arc<FuseSession> {
110        &self.session
111    }
112
113    /// Returns the device ID of this virtio-fs mount.
114    pub(super) fn container_device_id(&self) -> DeviceId {
115        self.sb.container_dev_id
116    }
117
118    /// Reads an inode from a FUSE entry reply via the inode cache.
119    pub(super) fn lookup_inode_from_cache(
120        self: &Arc<Self>,
121        lookup_reply: EntryReply,
122        request_attr_version: AttrVersion,
123    ) -> Result<Arc<VirtioFsInode>> {
124        self.inode_cache
125            .lookup_inode(lookup_reply, request_attr_version, self)
126    }
127
128    /// Inserts a newly created inode into the inode cache.
129    pub(super) fn insert_inode_to_cache(&self, inode: &Arc<VirtioFsInode>) {
130        self.inode_cache.insert_inode(inode);
131    }
132
133    /// Removes an inode from the cache if it is still the cached entry.
134    pub(super) fn remove_inode_from_cache(
135        &self,
136        inode: &VirtioFsInode,
137    ) -> Option<Weak<VirtioFsInode>> {
138        self.inode_cache.remove_inode(inode)
139    }
140}
141
142impl From<(DeviceId, Kstatfs)> for SuperBlock {
143    fn from((container_dev_id, statfs): (DeviceId, Kstatfs)) -> Self {
144        let mut sb = SuperBlock::new(VIRTIOFS_MAGIC, BLOCK_SIZE, NAME_MAX, container_dev_id);
145
146        sb.blocks = statfs.blocks() as usize;
147        sb.bfree = statfs.bfree() as usize;
148        sb.bavail = statfs.bavail() as usize;
149        sb.files = statfs.files() as usize;
150        sb.ffree = statfs.ffree() as usize;
151        sb.bsize = statfs.bsize() as usize;
152        sb.namelen = statfs.namelen() as usize;
153        sb.frsize = statfs.frsize() as usize;
154        sb
155    }
156}
