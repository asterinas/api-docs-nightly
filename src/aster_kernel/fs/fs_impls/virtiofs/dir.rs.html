19/// A per-open directory object backed by a FUSE open handle.
20///
21/// Readdir and release requests carry this handle.
22pub(super) struct VirtioFsDir {
23    inode: Arc<VirtioFsInode>,
24    open_handle: Arc<VirtioFsOpenHandle>,
25}
26
27impl VirtioFsDir {
28    pub(super) fn new(inode: Arc<VirtioFsInode>, open_handle: Arc<VirtioFsOpenHandle>) -> Self {
29        Self { inode, open_handle }
30    }
31}
32
33impl Pollable for VirtioFsDir {
34    fn poll(&self, mask: IoEvents, _poller: Option<&mut PollHandle>) -> IoEvents {
35        let events = IoEvents::IN | IoEvents::OUT;
36        events & mask
37    }
38}
39
40impl FileOps for VirtioFsDir {
41    fn read_at(
42        &self,
43        _offset: usize,
44        _writer: &mut VmWriter,
45        _status_flags: StatusFlags,
46    ) -> Result<usize> {
47        return_errno_with_message!(Errno::EISDIR, "the inode is a directory");
48    }
49
50    fn write_at(
51        &self,
52        _offset: usize,
53        _reader: &mut VmReader,
54        _status_flags: StatusFlags,
55    ) -> Result<usize> {
56        return_errno_with_message!(Errno::EISDIR, "the inode is a directory");
57    }
58
59    fn readdir_at(&self, offset: usize, visitor: &mut dyn DirentVisitor) -> Result<usize> {
60        // FIXME: `readdir_at` should pass current status flags to the
61        // server, while `FileOps` interface currently does not expose them,
62        // so `FUSE_READDIR` uses the flags captured at open time.
63        self.inode.readdir(
64            self.open_handle.fh(),
65            offset,
66            self.open_handle.file_flags(),
67            visitor,
68        )
69    }
70}
71
72impl PerOpenFileOps for VirtioFsDir {
73    fn check_seekable(&self) -> Result<()> {
74        if self
75            .open_handle
76            .open_flags()
77            .intersects(FuseOpenFlags::FOPEN_STREAM | FuseOpenFlags::FOPEN_NONSEEKABLE)
78        {
79            return_errno_with_message!(Errno::ESPIPE, "the directory is not seekable");
80        }
81
82        Ok(())
83    }
84
85    fn is_offset_aware(&self) -> bool {
86        true
87    }
88}
