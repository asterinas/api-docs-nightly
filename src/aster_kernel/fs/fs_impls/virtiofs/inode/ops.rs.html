45    /// Reads file data through the page cache.
46    pub(in crate::fs::fs_impls::virtiofs) fn cached_read_at(
47        &self,
48        offset: usize,
49        writer: &mut VmWriter,
50        fh: FuseFileHandle,
51        flags: u32,
52    ) -> Result<usize> {
53        if self.inner.read().page_cache().is_none() {
54            return self.direct_read_at(offset, writer, fh, flags);
55        }
56
57        // FIXME: The virtio-fs session currently always requests `AUTO_INVAL_DATA`,
58        // so cached reads refresh attributes before checking the cached file size.
59        // If this flag becomes optional, reads that stay below EOF should be allowed
60        // to use still-valid cached attributes.
61        self.revalidate_attr(fh)?;
62
63        let inner = self.inner.read();
64        let file_size = self.size();
65        let start = file_size.min(offset);
66        let end = file_size.min(offset.saturating_add(writer.avail()));
67        let read_len = end - start;
68        if read_len == 0 {
69            return Ok(0);
70        }
71
72        let mut limited_writer = writer.clone_exclusive();
73        limited_writer.limit(read_len);
74        let page_cache = inner.page_cache().unwrap();
75        page_cache.read(start, &mut limited_writer)?;
76        writer.skip(read_len);
77
78        Ok(read_len)
79    }

81    /// Reads file data directly from the server.
82    pub(in crate::fs::fs_impls::virtiofs) fn direct_read_at(
83        &self,
84        offset: usize,
85        writer: &mut VmWriter,
86        fh: FuseFileHandle,
87        flags: u32,
88    ) -> Result<usize> {
89        // FIXME: Direct reads do not support short reads yet, so `FUSE_READ`
90        // must request the actual bytes to read instead of the caller's maximum
91        // buffer size. Refresh attributes before using the cached file size.
92        self.revalidate_attr(fh)?;
93
94        let _inner = self.inner.read();
95        let file_size = self.size();
96        let start = file_size.min(offset);
97        let end = file_size.min(offset.saturating_add(writer.avail()));
98        let read_len = end - start;
99
100        if read_len == 0 {
101            return Ok(0);
102        }
103
104        // FIXME: A direct read must observe writes already staged in the page cache.
105        // With the current write-through policy, this usually has no dirty pages to
106        // submit, but it is still required for future write-back semantics.
107
108        let fs = self.fs_ref();
109        let data_buf = fs.session().alloc_read_buf(read_len)?;
110
111        let copied = fs.session().read(
112            self.nodeid(),
113            ReadReq::new(fh, start as u64, read_len as u32, flags),
114            data_buf.clone(),
115        )?;
116
117        let mut segment_reader = data_buf.reader()?;
118        segment_reader.limit(copied);
119        segment_reader
120            .read_fallible(writer)
121            .map_err(|(err, _)| Error::from(err))?;
122
123        Ok(copied)
124    }

126    /// Writes file data through the page cache.
127    ///
128    /// Cached writes use write-through semantics: they copy user bytes into the
129    /// page cache, flush the dirtied range to the server, and commit metadata
130    /// only after writeback succeeds.
131    pub(in crate::fs::fs_impls::virtiofs) fn cached_write_at(
132        &self,
133        write_offset: WriteOffset,
134        reader: &mut VmReader,
135        fh: FuseFileHandle,
136        flags: u32,
137    ) -> Result<usize> {
138        if self.inner.read().page_cache().is_none() {
139            return self.direct_write_at(write_offset, reader, fh, flags);
140        }
141        let write_len = reader.remain();
142        if write_len == 0 {
143            return Ok(0);
144        }
145
146        let mut inner = self.inner.write();
147        let offset = self.resolve_write_offset(write_offset);
148        let requested_end = offset
149            .checked_add(write_len)
150            .ok_or_else(|| Error::with_message(Errno::EOVERFLOW, "virtiofs write size overflow"))?;
151
152        let old_size = self.size();
153        let page_cache = inner.page_cache().unwrap();
154
155        if requested_end > old_size {
156            // Extend the visible EOF before growing the page cache.
157            self.set_size(requested_end);
158            page_cache
159                .resize(requested_end, old_size)
160                .expect("expanding the page cache should not fail");
161        }
162
163        let mut write_through_page_cache = || -> Result<()> {
164            page_cache.write(offset, reader).map_err(Error::from)?;
165            page_cache.flush_range(offset..requested_end)
166        };
167
168        if let Err(err) = write_through_page_cache() {
169            if requested_end > old_size {
170                // Roll back in truncate order: shrink the page cache before
171                // restoring the old visible EOF.
172                page_cache.resize(old_size, requested_end)?;
173                self.set_size(old_size);
174            }
175            return Err(err);
176        }
177
178        let new_size = self.size().max(requested_end);
179        let attr_version = self.fs_ref().session().bump_attr_version();
180
181        inner.commit_local_write(new_size, attr_version);
182
183        Ok(write_len)
184    }

186    /// Writes file data directly to the server.
187    pub(in crate::fs::fs_impls::virtiofs) fn direct_write_at(
188        &self,
189        write_offset: WriteOffset,
190        reader: &mut VmReader,
191        fh: FuseFileHandle,
192        flags: u32,
193    ) -> Result<usize> {
194        let write_len = reader.remain();
195
196        let mut inner = self.inner.write();
197        let offset = self.resolve_write_offset(write_offset);
198        let write_end = offset
199            .checked_add(write_len)
200            .ok_or_else(|| Error::with_message(Errno::EOVERFLOW, "virtiofs write size overflow"))?;
201
202        if let Some(page_cache) = inner.page_cache() {
203            page_cache.invalidate_range(offset..write_end)?;
204        };
205
206        let written = self.do_direct_write(offset, reader, fh, flags)?;
207
208        // TODO: Do `evict_range` again after write to prevent stale data from
209        // being loaded into the page cache during asynchronous read-ahead.
210
211        let new_size = offset
212            .checked_add(written)
213            .ok_or_else(|| Error::with_message(Errno::EOVERFLOW, "virtiofs write size overflow"))?;
214
215        let new_size = self.size().max(new_size);
216        let attr_version = self.fs_ref().session().bump_attr_version();
217        inner.commit_local_write(new_size, attr_version);
218        self.set_size(new_size);
219
220        Ok(written)
221    }

223    fn resolve_write_offset(&self, write_offset: WriteOffset) -> usize {
224        match write_offset {
225            WriteOffset::Absolute(offset) => offset,
226            WriteOffset::Append => self.size(),
227        }
228    }

250    fn do_direct_write(
251        &self,
252        offset: usize,
253        reader: &mut VmReader,
254        fh: FuseFileHandle,
255        flags: u32,
256    ) -> Result<usize> {
257        let fs = self.fs_ref();
258        let max_write = fs.session().max_write() as usize;
259        let mut total_written = 0usize;
260
261        // `FUSE_WRITE` replies carry the accepted byte count. Submit and wait
262        // for one chunk at a time. Once any bytes are accepted, a later error
263        // is reported as a successful partial write.
264        while reader.has_remain() {
265            let write_size = reader.remain().min(max_write);
266            let data_buf = match fs.session().alloc_write_buf(write_size) {
267                Ok(data_buf) => data_buf,
268                Err(_) if total_written > 0 => return Ok(total_written),
269                Err(err) => return Err(err.into()),
270            };
271
272            let mut segment_writer = data_buf.writer().unwrap();
273            let mut request_reader = reader.clone();
274            request_reader.limit(write_size);
275            if let Err((err, _)) = segment_writer.write_fallible(&mut request_reader) {
276                if total_written > 0 {
277                    return Ok(total_written);
278                }
279
280                return Err(err.into());
281            }
282
283            let request_offset = match offset.checked_add(total_written) {
284                Some(request_offset) => request_offset,
285                None if total_written > 0 => return Ok(total_written),
286                None => {
287                    return Err(Error::with_message(
288                        Errno::EOVERFLOW,
289                        "virtiofs write offset overflow",
290                    ));
291                }
292            };
293            let ret = fs
294                .session()
295                .write(
296                    self.nodeid(),
297                    WriteReq::new(
298                        fh,
299                        request_offset as u64,
300                        write_size as u32,
301                        flags,
302                        WriteFlags::empty(),
303                    ),
304                    data_buf,
305                )
306                .map_err(Error::from);
307
308            let written = match ret {
309                Ok(written) => written,
310                Err(_) if total_written > 0 => return Ok(total_written),
311                Err(err) => return Err(err),
312            };
313
314            if written > write_size {
315                if total_written > 0 {
316                    return Ok(total_written);
317                }
318
319                return_errno_with_message!(Errno::EIO, "virtiofs write response is too large");
320            }
321            if written == 0 {
322                break;
323            }
324
325            let Some(new_total_written) = total_written.checked_add(written) else {
326                return Ok(total_written);
327            };
328            reader.skip(written);
329            total_written = new_total_written;
330            if written < write_size {
331                break;
332            }
333        }
334
335        Ok(total_written)
336    }

338    pub(super) fn open_file(
339        &self,
340        access_mode: AccessMode,
341        status_flags: StatusFlags,
342    ) -> Result<Box<dyn PerOpenFileOps>> {
343        let inode = self
344            .weak_self
345            .upgrade()
346            .ok_or_else(|| Error::with_message(Errno::EIO, "virtiofs inode is unavailable"))?;
347        let fs = self.fs_ref();
348
349        let open_out = fs.session().do_fuse_op(
350            self.nodeid(),
351            OpenOperation::new(OpenReq::new((access_mode as u32) | status_flags.bits())),
352        )?;
353        let cache_policy = if self.inner.read().page_cache().is_some()
354            && !open_out
355                .open_flags()
356                .contains(FuseOpenFlags::FOPEN_DIRECT_IO)
357        {
358            CachePolicy::Cached
359        } else {
360            CachePolicy::Direct
361        };
362        let open_handle = VirtioFsOpenHandle::new(
363            open_out.fh(),
364            self.nodeid(),
365            access_mode,
366            status_flags,
367            open_out.open_flags(),
368            self.fs.clone(),
369            ReleaseOptions::new(ReleaseKind::File, ReleaseFlags::RELEASE_FLUSH),
370        );
371        if !open_out
372            .open_flags()
373            .contains(FuseOpenFlags::FOPEN_KEEP_CACHE)
374            && let Err(err) = self.invalidate_whole_page_cache()
375        {
376            return Err(err);
377        }
378
379        // Cache the open handle for later page cache I/O.
380        if cache_policy == CachePolicy::Cached {
381            self.open_handles.insert(&open_handle);
382        }
383
384        Ok(Box::new(VirtioFsFile::new(
385            inode,
386            open_handle,
387            cache_policy,
388        )))
389    }

418    /// Commits an `EntryReply` reply for this cached inode.
419    ///
420    /// `EntryReply` replies carry both attributes and one new lookup reference.
421    /// Updating the two together keeps the client-side `nlookup` mirror in sync
422    /// with the server-side count.
423    pub(super) fn commit_entry_reply(
424        &self,
425        entry_reply: &EntryReply,
426        request_attr_version: AttrVersion,
427        stale_action: StaleAttrAction,
428    ) -> Result<()> {
429        debug_assert_eq!(entry_reply.nodeid(), self.nodeid());
430        self.lookup_count.acquire();
431
432        self.commit_attr_reply(
433            FuseAttrReply::from(entry_reply),
434            request_attr_version,
435            stale_action,
436        )
437    }

439    /// Reads directory entries and expires this directory's attribute cache.
440    ///
441    /// `READDIR` does not return a complete attribute reply for the directory,
442    /// but successful directory reads may observe server-side changes. Expiring
443    /// attributes forces later metadata-sensitive operations to revalidate.
444    pub(in crate::fs::fs_impls::virtiofs) fn readdir(
445        &self,
446        fh: FuseFileHandle,
447        offset: usize,
448        flags: u32,
449        visitor: &mut dyn DirentVisitor,
450    ) -> Result<usize> {
451        let fs = self.fs_ref();
452        let data_buf = fs
453            .session()
454            .alloc_read_buf(FUSE_READDIR_BUF_SIZE as usize)?;
455        let entries: Vec<FuseDirEntry> = fs.session().readdir(
456            self.nodeid(),
457            ReadReq::new(fh, offset as u64, FUSE_READDIR_BUF_SIZE, flags),
458            data_buf,
459        )?;
460
461        let offset_read = {
462            let try_readdir_fn = |offset: &mut usize,
463                                  visitor: &mut dyn DirentVisitor|
464             -> Result<()> {
465                for entry in &entries {
466                    let next_offset = entry.offset().get() as usize;
467                    visitor.visit(entry.name(), entry.ino(), entry.type_().into(), next_offset)?;
468                    *offset = next_offset;
469                }
470
471                Ok(())
472            };
473
474            let mut iterate_offset = offset;
475            match try_readdir_fn(&mut iterate_offset, visitor) {
476                Err(e) if iterate_offset == offset => Err(e),
477                // FIXME: FUSE directory offsets are opaque cookies, but this
478                // path currently treats them as linear values that can be
479                // subtracted to produce a cursor delta.
480                _ => Ok(iterate_offset - offset),
481            }?
482        };
483
484        self.expire_attr_cache();
485
486        Ok(offset_read)
487    }

489    /// Sets one inode timestamp through `SETATTR`.
490    ///
491    /// Calls this from VFS timestamp setters whose trait signature cannot
492    /// return an error. Failures are logged and the cached metadata is left to be
493    /// repaired by a later revalidation.
494    pub(super) fn set_time(&self, field: TimeField, time: Duration) {
495        let setattr_req = match field {
496            TimeField::Access => SetattrReq::new(SetattrValid::empty())
497                .set_atime(time.as_secs(), time.subsec_nanos()),
498            TimeField::Modify => SetattrReq::new(SetattrValid::empty())
499                .set_mtime(time.as_secs(), time.subsec_nanos()),
500            TimeField::Change => SetattrReq::new(SetattrValid::empty())
501                .set_ctime(time.as_secs(), time.subsec_nanos()),
502        };
503        if let Err(err) = self.setattr(setattr_req) {
504            warn!(
505                "virtiofs set_time failed for inode {}: {:?}",
506                self.nodeid().as_u64(),
507                err
508            );
509        }
510    }

512    /// Applies a FUSE `SETATTR` request and commits its returned attributes.
513    ///
514    /// If the reply loses the attr-version race, only fields selected by the
515    /// request's `SetattrValid` mask remain safe to merge into cached metadata.
516    pub(super) fn setattr(&self, setattr_req: SetattrReq) -> Result<()> {
517        let fs = self.fs_ref();
518        let request_attr_version = fs.session().snapshot_attr_version();
519        let valid = setattr_req.valid();
520        let attr_reply = fs
521            .session()
522            .do_fuse_op(self.nodeid(), SetattrOperation::new(setattr_req))?;
523
524        self.commit_attr_reply(
525            attr_reply,
526            request_attr_version,
527            StaleAttrAction::MergeSetattr(valid),
528        )?;
529
530        Ok(())
531    }

533    fn forget_async(&self, nlookup: u64) {
534        let nodeid = self.nodeid();
535
536        if let Some(fs) = self.fs.upgrade() {
537            work_queue::submit_work_func(
538                move || {
539                    if let Err(err) = fs.session().forget(nodeid, nlookup) {
540                        warn!(
541                            "virtiofs forget failed for inode {} with nlookup {}: {:?}",
542                            nodeid.as_u64(),
543                            nlookup,
544                            err
545                        );
546                    }
547                },
548                WorkPriority::Normal,
549            );
550        }
551    }

553    /// Revalidates a cached directory entry with `LOOKUP`.
554    ///
555    /// The entry TTL and attribute TTL are independent caches. A successful
556    /// revalidation refreshes the dentry deadline and commits the returned
557    /// attributes as an observation, so a stale attribute reply is discarded.
558    pub(super) fn revalidate_lookup(
559        &self,
560        parent_nodeid: aster_fuse::FuseNodeId,
561        name: &str,
562    ) -> Result<()> {
563        let now = MonotonicCoarseClock::get().read_time();
564        if now < *self.entry_valid_until.lock() {
565            return Ok(());
566        }
567
568        let old_nodeid = self.nodeid();
569        let fs = self.fs_ref();
570        let request_attr_version = fs.session().snapshot_attr_version();
571        let lookup_reply = fs
572            .session()
573            .do_fuse_op(parent_nodeid, LookupOperation::new(name))?;
574
575        if lookup_reply.nodeid() != old_nodeid || lookup_reply.generation() != self.generation() {
576            if let Err(err) = fs.session().forget(lookup_reply.nodeid(), 1) {
577                warn!(
578                    "virtiofs forget failed for stale lookup inode {}: {:?}",
579                    lookup_reply.nodeid().as_u64(),
580                    err
581                );
582            }
583            return_errno_with_message!(Errno::ESTALE, "virtiofs stale dentry after revalidate");
584        }
585
586        self.commit_entry_reply(
587            &lookup_reply,
588            request_attr_version,
589            StaleAttrAction::Discard,
590        )?;
591
592        *self.entry_valid_until.lock() =
593            valid_until(lookup_reply.entry_valid(), lookup_reply.entry_valid_nsec());
594
595        Ok(())
596    }

598    /// Refreshes cached attributes when their server TTL has expired.
599    ///
600    /// Calls this before operations that depend on a current size or on
601    /// page-cache coherency, such as reads, `O_APPEND` writes, and `SEEK_END`.
602    /// It is a cheap no-op while the cached attributes are still valid. When a
603    /// refresh is needed, the supplied FUSE file handle is passed to `GETATTR`
604    /// so the server may return handle-specific attributes.
605    pub(in crate::fs::fs_impls::virtiofs) fn revalidate_attr(
606        &self,
607        fh: FuseFileHandle,
608    ) -> Result<()> {
609        let now = MonotonicCoarseClock::get().read_time();
610        if self.inner.read().is_attr_valid(now) {
611            return Ok(());
612        }
613
614        let fs = self.fs_ref();
615        let request_attr_version = fs.session().snapshot_attr_version();
616        let attr_reply = fs.session().do_fuse_op(
617            self.nodeid(),
618            GetattrOperation::new(GetattrReq::new(GetattrFlags::GETATTR_FH, fh)),
619        )?;
620
621        self.commit_attr_reply(attr_reply, request_attr_version, StaleAttrAction::Discard)?;
622
623        Ok(())
624    }

627impl Drop for VirtioFsInode {
628    // FUSE forgets must run outside Drop: the session may sleep and
629    // may need VFS locks; defer to the work queue.
630    fn drop(&mut self) {
631        if let Some(fs) = self.fs.upgrade() {
632            fs.remove_inode_from_cache(self);
633        }
634
635        let nlookup = self.lookup_count.drain();
636        if nlookup > 0 {
637            self.forget_async(nlookup);
638        }
639    }
