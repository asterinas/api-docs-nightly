3use device_id::{DeviceId, MajorId, MinorId};
4use id_alloc::IdAlloc;
5use ostd::sync::Mutex;
6use spin::Once;
7
8/// An anonymous device ID that automatically recycles itself on drop.
9#[derive(Debug)]
10pub struct AnonDeviceId(DeviceId);
11
12impl AnonDeviceId {
13    /// Acquires an anonymous device ID for a pseudo filesystem.
14    pub fn acquire() -> Option<Self> {
15        DeviceIdAllocator::singleton().allocate().map(Self::new)
16    }
17
18    /// Returns the underlying `DeviceId`.
19    pub fn id(&self) -> DeviceId {
20        self.0
21    }
22
23    fn new(id: DeviceId) -> Self {
24        Self(id)
25    }
26}
27
28impl Drop for AnonDeviceId {
29    fn drop(&mut self) {
30        DeviceIdAllocator::singleton().release(self.0);
31    }
32}
33
34/// An allocator for pseudo filesystems (no backing block device) device ID.
35///
36/// This follows the Linux convention where pseudo filesystems use major=0
37/// and dynamically allocate minor numbers (starting from 1) to distinguish different
38/// pseudo filesystem instances.
39///
40/// Reference: <https://elixir.bootlin.com/linux/v6.18/source/fs/super.c#L1242-L1271>
41struct DeviceIdAllocator {
42    minor_allocator: Mutex<IdAlloc>,
43}
44
45impl DeviceIdAllocator {
46    fn singleton() -> &'static Self {
47        static SINGLETON: Once<DeviceIdAllocator> = Once::new();
48
49        SINGLETON.call_once(Self::new)
50    }
51
52    fn new() -> Self {
53        let mut minor_allocator = IdAlloc::with_capacity(MinorId::MAX.get() as usize + 1);
54        // Mark 0 as allocated to ensure minor numbers start from 1.
55        let _ = minor_allocator.alloc_specific(0).unwrap();
56
57        Self {
58            minor_allocator: Mutex::new(minor_allocator),
59        }
60    }
61
62    fn allocate(&self) -> Option<DeviceId> {
63        let major = MajorId::new(0);
64        let minor = self.minor_allocator.lock().alloc()?;
65
66        Some(DeviceId::new(major, MinorId::new(minor as u32)))
67    }
68
69    /// Frees a dynamically allocated pseudo filesystem device ID.
70    ///
71    /// # Panics
72    ///
73    /// Panics if the device ID is not anonymous.
74    fn release(&self, id: DeviceId) {
75        assert!(id.is_anonymous());
76
77        self.minor_allocator.lock().free(id.minor().get() as usize);
78    }
79}
