3use spin::Once;
4
5use crate::{
6    fs::{
7        file::mkmod,
8        pseudofs::{NaivePseudoFs, PseudoInodeType},
9        vfs::{
10            inode::Inode,
11            path::{Mount, Path},
12        },
13    },
14    prelude::*,
15    process::{Gid, Uid},
16};
17
18pub struct AnonInodeFs {
19    _private: (),
20}
21
22impl AnonInodeFs {
23    /// Returns the singleton instance of the anonymous inode file system.
24    fn singleton() -> &'static Arc<NaivePseudoFs> {
25        static ANON_INODEFS: Once<Arc<NaivePseudoFs>> = Once::new();
26
27        NaivePseudoFs::singleton(&ANON_INODEFS, "anon_inodefs", ANON_INODEFS_MAGIC)
28    }
29
30    /// Creates a pseudo `Path` for the shared inode.
31    pub fn new_path(name_fn: fn(&dyn Inode) -> String) -> Path {
32        Path::new_pseudo(
33            Self::mount_node().clone(),
34            Self::shared_inode().clone(),
35            name_fn,
36        )
37    }
38
39    /// Returns the pseudo mount node of the anonymous inode file system.
40    pub fn mount_node() -> &'static Arc<Mount> {
41        static ANON_INODEFS_MOUNT: Once<Arc<Mount>> = Once::new();
42
43        ANON_INODEFS_MOUNT.call_once(|| Mount::new_pseudo(Self::singleton().clone()))
44    }
45
46    /// Returns the shared inode of the anonymous inode file system singleton.
47    //
48    // Some members of anon_inodefs (such as epollfd, eventfd, timerfd, etc.) share
49    // the same inode. The sharing is not only within the same category (e.g., two
50    // epollfds share the same inode) but also across different categories (e.g.,
51    // an epollfd and a timerfd share the same inode). Even across namespaces, this
52    // inode is still shared. Although this Linux behavior is a bit odd, we keep it
53    // for compatibility.
54    //
55    // A small subset of members in anon_inodefs (i.e., userfaultfd, io_uring, and
56    // kvm_guest_memfd) have their own dedicated inodes. We need to support creating
57    // independent inodes within anon_inodefs for them in the future.
58    //
59    // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/anon_inodes.c#L153-L164>
60    pub fn shared_inode() -> &'static Arc<dyn Inode> {
61        static SHARED_INODE: Once<Arc<dyn Inode>> = Once::new();
62
63        SHARED_INODE.call_once(|| {
64            let shared_inode = Self::singleton().alloc_inode(
65                PseudoInodeType::AnonInode,
66                mkmod!(u+rw),
67                Uid::new_root(),
68                Gid::new_root(),
69            );
70
71            Arc::new(shared_inode)
72        })
73    }
74}
75
76// Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/include/uapi/linux/magic.h#L93>
77const ANON_INODEFS_MAGIC: u64 = 0x09041934;
