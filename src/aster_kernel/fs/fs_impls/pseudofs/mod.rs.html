3//! The infrastructure of pseudo FSes along with some naive implementations of pseudo FSes.
4//!
5//! Pseudo FSes are those that have no physical storage.
6//! Some pseudo FSes are trivial,
7//! e.g., [`PipeFs`], [`SockFs`], [`PidfdFs`], and [`AnonInodeFs`].
8//! Such trivial pseudo FSes are directly implemented in this module
9//! using a generic pseudo FS implementation called [`NaivePseudoFs`].
10//!
11//! On the other hand, some pseudo FSes are quite complex,
12//! including `RamFs`, `ProcFs`, and `SysFs`. Just to name a few.
13//! They are implemented in their own modules.
14//!
15//! Pseudo FSes are not backed by physical storage devices,
16//! but they still need to be assigned container device IDs
17//! (see [`Metadata::container_dev_id`]).
18//! [Linux's behavior](https://elixir.bootlin.com/linux/v6.19.3/source/fs/super.c#L1254)
19//! is to assign _anonymous device IDs_ (whose major IDs are 0).
20//! As such, this module provides [`AnonDeviceId`] to acquire and recycle
21//! anonymous device IDs.
22

60/// A pseudo file system that manages pseudo inodes, such as pipe inodes and socket inodes.
61pub struct NaivePseudoFs {
62    name: &'static str,
63    _anon_device_id: AnonDeviceId,
64    sb: SuperBlock,
65    root: Arc<dyn Inode>,
66    inode_allocator: AtomicU64,
67    fs_event_subscriber_stats: FsEventSubscriberStats,
68}
69
70impl FileSystem for NaivePseudoFs {
71    fn name(&self) -> &'static str {
72        self.name
73    }
74
75    fn sync(&self) -> Result<()> {
76        // do nothing
77        Ok(())
78    }
79
80    fn root_inode(&self) -> Arc<dyn Inode> {
81        self.root.clone()
82    }
83
84    fn sb(&self) -> SuperBlock {
85        self.sb.clone()
86    }
87
88    fn fs_event_subscriber_stats(&self) -> &FsEventSubscriberStats {
89        &self.fs_event_subscriber_stats
90    }
91}
92
93impl NaivePseudoFs {
94    /// Returns a reference to the singleton pseudo file system.
95    fn singleton(
96        fs: &'static Once<Arc<Self>>,
97        name: &'static str,
98        magic: u64,
99    ) -> &'static Arc<Self> {
100        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/libfs.c#L659-L689>
101        fs.call_once(|| {
102            let anon_device_id =
103                AnonDeviceId::acquire().expect("no device ID is available for pseudofs");
104            let container_dev_id = anon_device_id.id();
105            Arc::new_cyclic(move |weak_fs: &Weak<Self>| Self {
106                name,
107                _anon_device_id: anon_device_id,
108                sb: SuperBlock::new(magic, aster_block::BLOCK_SIZE, NAME_MAX, container_dev_id),
109                root: Arc::new(PseudoInode::new(
110                    ROOT_INO,
111                    PseudoInodeType::Root,
112                    mkmod!(u+rw),
113                    Uid::new_root(),
114                    Gid::new_root(),
115                    weak_fs.clone(),
116                    container_dev_id,
117                )),
118                inode_allocator: AtomicU64::new(ROOT_INO + 1),
119                fs_event_subscriber_stats: FsEventSubscriberStats::new(),
120            })
121        })
122    }
123
124    pub fn alloc_inode(
125        self: &Arc<Self>,
126        type_: PseudoInodeType,
127        mode: InodeMode,
128        uid: Uid,
129        gid: Gid,
130    ) -> PseudoInode {
131        PseudoInode::new(
132            self.alloc_id(),
133            type_,
134            mode,
135            uid,
136            gid,
137            Arc::downgrade(self),
138            self.sb.container_dev_id,
139        )
140    }
141
142    fn alloc_id(&self) -> u64 {
143        self.inode_allocator.fetch_add(1, Ordering::Relaxed)
144    }
145
146    pub(super) fn container_dev_id(&self) -> DeviceId {
147        self.sb.container_dev_id
148    }
149}

158/// Root Inode ID.
159const ROOT_INO: u64 = 1;

161#[derive(Clone, Copy, Debug, Eq, PartialEq)]
162pub enum PseudoInodeType {
163    Root,
164    Pipe,
165    Socket,
166    AnonInode,
167    Pidfd,
168    Ns,
169}
170
171impl From<PseudoInodeType> for InodeType {
172    fn from(pseudo_type: PseudoInodeType) -> Self {
173        match pseudo_type {
174            PseudoInodeType::Root => InodeType::Dir,
175            PseudoInodeType::Pipe => InodeType::NamedPipe,
176            PseudoInodeType::Socket => InodeType::Socket,
177            PseudoInodeType::AnonInode => InodeType::Unknown,
178            PseudoInodeType::Pidfd => InodeType::Unknown,
179            PseudoInodeType::Ns => InodeType::File,
180        }
181    }
182}

184/// A pseudo inode that does not correspond to any real path in the file system.
185pub struct PseudoInode {
186    metadata: SpinLock<Metadata>,
187    extension: Extension,
188    fs: Weak<NaivePseudoFs>,
189    is_anon: bool,
190}
191
192impl PseudoInode {
193    fn new(
194        ino: u64,
195        type_: PseudoInodeType,
196        mode: InodeMode,
197        uid: Uid,
198        gid: Gid,
199        fs: Weak<NaivePseudoFs>,
200        dev_id: DeviceId,
201    ) -> Self {
202        let now = now();
203        let type_ = InodeType::from(type_);
204
205        let metadata = Metadata {
206            ino,
207            size: 0,
208            optimal_block_size: aster_block::BLOCK_SIZE,
209            nr_sectors_allocated: 0,
210            last_access_at: now,
211            last_modify_at: now,
212            last_meta_change_at: now,
213            type_,
214            mode,
215            nr_hard_links: 1,
216            uid,
217            gid,
218            container_dev_id: dev_id,
219            self_dev_id: None,
220        };
221
222        PseudoInode {
223            metadata: SpinLock::new(metadata),
224            extension: Extension::new(),
225            fs,
226            is_anon: type_ == InodeType::Unknown,
227        }
228    }
229}

231impl InodeIo for PseudoInode {
232    fn read_at(
233        &self,
234        _offset: usize,
235        _writer: &mut VmWriter,
236        _status: StatusFlags,
237    ) -> Result<usize> {
238        return_errno_with_message!(
239            Errno::ESPIPE,
240            "pseudo inodes cannot be read at a specific offset"
241        );
242    }
243
244    fn write_at(
245        &self,
246        _offset: usize,
247        _reader: &mut VmReader,
248        _status: StatusFlags,
249    ) -> Result<usize> {
250        return_errno_with_message!(
251            Errno::ESPIPE,
252            "pseudo inodes cannot be written at a specific offset"
253        );
254    }
255}

257impl Inode for PseudoInode {
258    fn size(&self) -> usize {
259        self.metadata.lock().size
260    }
261
262    fn resize(&self, _new_size: usize) -> Result<()> {
263        return_errno_with_message!(Errno::EINVAL, "pseudo inodes can not be resized");
264    }
265
266    fn metadata(&self) -> Metadata {
267        *self.metadata.lock()
268    }
269
270    fn extension(&self) -> &Extension {
271        &self.extension
272    }
273
274    fn ino(&self) -> u64 {
275        self.metadata.lock().ino
276    }
277
278    fn type_(&self) -> InodeType {
279        self.metadata.lock().type_
280    }
281
282    fn mode(&self) -> Result<InodeMode> {
283        Ok(self.metadata.lock().mode)
284    }
285
286    fn set_mode(&self, mode: InodeMode) -> Result<()> {
287        if self.is_anon {
288            return_errno_with_message!(
289                Errno::EOPNOTSUPP,
290                "the mode of anonymous inodes cannot be changed"
291            );
292        }
293
294        let mut meta = self.metadata.lock();
295        meta.mode = mode;
296        meta.last_meta_change_at = now();
297        Ok(())
298    }
299
300    fn owner(&self) -> Result<Uid> {
301        Ok(self.metadata.lock().uid)
302    }
303
304    fn set_owner(&self, uid: Uid) -> Result<()> {
305        let mut meta = self.metadata.lock();
306        meta.uid = uid;
307        meta.last_meta_change_at = now();
308        Ok(())
309    }
310
311    fn group(&self) -> Result<Gid> {
312        Ok(self.metadata.lock().gid)
313    }
314
315    fn set_group(&self, gid: Gid) -> Result<()> {
316        let mut meta = self.metadata.lock();
317        meta.gid = gid;
318        meta.last_meta_change_at = now();
319        Ok(())
320    }
321
322    fn atime(&self) -> Duration {
323        self.metadata.lock().last_access_at
324    }
325
326    fn set_atime(&self, time: Duration) {
327        self.metadata.lock().last_access_at = time;
328    }
329
330    fn mtime(&self) -> Duration {
331        self.metadata.lock().last_modify_at
332    }
333
334    fn set_mtime(&self, time: Duration) {
335        self.metadata.lock().last_modify_at = time;
336    }
337
338    fn ctime(&self) -> Duration {
339        self.metadata.lock().last_meta_change_at
340    }
341
342    fn set_ctime(&self, time: Duration) {
343        self.metadata.lock().last_meta_change_at = time;
344    }
345
346    fn open(
347        &self,
348        _access_mode: AccessMode,
349        _status_flags: StatusFlags,
350    ) -> Option<Result<Box<dyn FileIo>>> {
351        Some(Err(Error::with_message(
352            Errno::ENXIO,
353            "the pseudo inode is not re-openable",
354        )))
355    }
356
357    fn fs(&self) -> Arc<dyn FileSystem> {
358        self.fs.upgrade().unwrap()
359    }
360}
361
362fn now() -> Duration {
363    RealTimeCoarseClock::get().read_time()
