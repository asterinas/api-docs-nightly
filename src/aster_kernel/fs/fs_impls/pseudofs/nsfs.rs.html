33/// A pseudo filesystem for namespace files.
34struct NsFs {
35    _private: (),
36}
37
38impl NsFs {
39    /// Returns the singleton instance of the ns file system.
40    pub(self) fn singleton() -> &'static Arc<NaivePseudoFs> {
41        static NSFS: Once<Arc<NaivePseudoFs>> = Once::new();
42        NaivePseudoFs::singleton(&NSFS, "nsfs", NSFS_MAGIC)
43    }
44
45    /// Creates a pseudo `Path` for a namespace file.
46    pub(self) fn new_path<T: NsCommonOps>(ino: u64, ns: Arc<T>) -> Path {
47        let ns_inode = {
48            let fs = Arc::downgrade(Self::singleton());
49            Arc::new(NsInode::new(ino, Uid::new_root(), Gid::new_root(), ns, fs))
50        };
51
52        Path::new_pseudo(Self::mount_node().clone(), ns_inode, |inode| {
53            let inode = inode.downcast_ref::<NsInode<T>>().unwrap();
54            inode.name().to_string()
55        })
56    }
57
58    /// Returns the pseudo mount node of the ns file system.
59    pub(self) fn mount_node() -> &'static Arc<Mount> {
60        static NSFS_MOUNT: Once<Arc<Mount>> = Once::new();
61        NSFS_MOUNT.call_once(|| Mount::new_pseudo(Self::singleton().clone()))
62    }
63}

65/// An inode representing a namespace entry in [`NsFs`].
66pub(in crate::fs) struct NsInode<T: NsCommonOps> {
67    common: PseudoInode,
68    ns: Arc<T>,
69    name: String,
70}
71
72impl<T: NsCommonOps> NsInode<T> {
73    fn new(ino: u64, uid: Uid, gid: Gid, ns: Arc<T>, fs: Weak<NaivePseudoFs>) -> Self {
74        let mode = mkmod!(a+r);
75        let fs = fs.upgrade().unwrap();
76        let common = PseudoInode::new(
77            ino,
78            PseudoInodeType::Ns,
79            mode,
80            uid,
81            gid,
82            Arc::downgrade(&fs),
83            fs.container_dev_id(),
84        );
85        let name = format!("{}:[{}]", T::NAME, ino);
86
87        Self { common, ns, name }
88    }
89
90    fn name(&self) -> &str {
91        &self.name
92    }
93
94    pub(in crate::fs) fn ns(&self) -> &Arc<T> {
95        &self.ns
96    }
97}

107    fn mode(&self) -> Result<InodeMode>;
108    fn owner(&self) -> Result<Uid>;
109    fn set_owner(&self, uid: Uid) -> Result<()>;
110    fn group(&self) -> Result<Gid>;
111    fn set_group(&self, gid: Gid) -> Result<()>;
112    fn atime(&self) -> Duration;
113    fn set_atime(&self, time: Duration);
114    fn mtime(&self) -> Duration;
115    fn set_mtime(&self, time: Duration);
116    fn ctime(&self) -> Duration;
117    fn set_ctime(&self, time: Duration);

119
120    fn open(
121        &self,
122        access_mode: AccessMode,
123        _status_flags: StatusFlags,
124    ) -> Option<Result<Box<dyn FileIo>>> {
125        // FIXME: This may not be the most appropriate place to check the access mode,
126        // but the check must not be bypassed even if the current process has the
127        // CAP_DAC_OVERRIDE capability. It is hard to find a better place for it,
128        // and an extra check here does no harm.
129        if access_mode.is_writable() {
130            return Some(Err(Error::with_message(
131                Errno::EPERM,
132                "ns files cannot be opened as writable",
133            )));
134        }
135
136        let ns_file = NsFile {
137            ns: self.ns.clone(),
138        };
139        Some(Ok(Box::new(ns_file) as Box<dyn FileIo>))
140    }
141
142    fn set_mode(&self, _mode: InodeMode) -> Result<()> {
143        return_errno_with_message!(Errno::EPERM, "the mode of ns inodes cannot be changed");
144    }

163/// A file handle referencing a live namespace.
164pub struct NsFile<T: NsCommonOps> {
165    ns: Arc<T>,
166}
167
168impl<T: NsCommonOps> NsFile<T> {
169    /// Returns a reference to the underlying namespace.
170    pub fn ns(&self) -> &Arc<T> {
171        &self.ns
172    }
173}
174
175impl<T: NsCommonOps> FileIo for NsFile<T> {
176    fn check_seekable(&self) -> Result<()> {
177        return_errno_with_message!(Errno::ESPIPE, "ns files are not seekable");
178    }
179
180    fn is_offset_aware(&self) -> bool {
181        false
182    }
183
184    fn check_positional_io(&self) -> Result<()> {
185        // Namespace files support `pread`/`pwrite` but not `lseek`.
186        Ok(())
187    }
188
189    fn ioctl(&self, raw_ioctl: RawIoctl) -> Result<i32> {
190        use ioctl_defs::*;
191
192        dispatch_ioctl!(match raw_ioctl {
193            _cmd @ GetUserNs => {
194                let user_ns = self.ns.owner_user_ns().ok_or_else(|| {
195                    Error::with_message(
196                        Errno::EPERM,
197                        "the namespace does not have an owner user namespace",
198                    )
199                })?;
200
201                let current = current!();
202                let current_user_ns = current.user_ns().lock();
203                if !current_user_ns.is_same_or_ancestor_of(user_ns) {
204                    return_errno_with_message!(
205                        Errno::EPERM,
206                        "the current namespace is not an ancestor of the owner user namespace"
207                    );
208                }
209
210                open_ns_as_file(user_ns).map(|fd| fd.into())
211            }
212            _cmd @ GetParent => {
213                let parent = self.ns.parent()?;
214                open_ns_as_file(parent).map(|fd| fd.into())
215            }
216            _cmd @ GetType => {
217                let clone_flags = CloneFlags::from(T::TYPE);
218                Ok(clone_flags.bits().cast_signed())
219            }
220            cmd @ GetOwnerUid => {
221                let ns = self.ns.as_ref() as &dyn Any;
222                let user_ns = ns.downcast_ref::<UserNamespace>().ok_or_else(|| {
223                    Error::with_message(
224                        Errno::EINVAL,
225                        "the ns file does not correspond to a user namespace",
226                    )
227                })?;
228                let uid = user_ns.owner_uid()?;
229                cmd.write(&uid.into())?;
230                Ok(0)
231            }
232            // TODO: Support additional ioctl commands
233            _ => return_errno_with_message!(Errno::ENOTTY, "the ioctl command is unknown"),
234        })
235    }
236}

270/// Opens a namespace as a file and returns the file descriptor.
271fn open_ns_as_file<T: NsCommonOps>(ns: &Arc<T>) -> Result<FileDesc> {
272    let path = ns.get_path();
273    let inode_handle = InodeHandle::new(path.clone(), AccessMode::O_RDONLY, StatusFlags::empty())?;
274
275    let current_task = Task::current().unwrap();
276    let thread_local = current_task.as_thread_local().unwrap();
277    let mut file_table_ref = thread_local.borrow_file_table_mut();
278    let mut file_table = file_table_ref.unwrap().write();
279    let fd = file_table.insert(Arc::new(inode_handle), FdFlags::CLOEXEC);
280
281    Ok(fd)
282}

284/// Common operations shared by all namespace types.
285///
286/// Implementors represent a specific kind of namespace (e.g., UTS, mount, user)
287/// and must provide the associated metadata and traversal methods required by
288/// [`NsFs`] and [`NsFile`].
289pub trait NsCommonOps: Any + Send + Sync + 'static + Sized {
290    /// The human-readable name of this namespace kind (derived from [`Self::TYPE`]).
291    const NAME: &str = Self::TYPE.as_str();
292
293    /// The [`NsType`] discriminant for this namespace kind.
294    const TYPE: NsType;
295
296    /// Returns the owner user namespace.
297    ///
298    /// Every namespace is owned by the user namespace in which it was created.
299    /// The only exception is the initial user namespace, which has no owner;
300    /// in that case this method returns `None`.
301    fn owner_user_ns(&self) -> Option<&Arc<UserNamespace>>;
302
303    /// Returns the parent namespace, if one exists.
304    ///
305    /// This is meaningful only for hierarchical namespace types (user and PID
306    /// namespaces). Non-hierarchical namespaces should return `EINVAL`.
307    fn parent(&self) -> Result<&Arc<Self>>;
308
309    /// Returns the dentry cache used to maintain the namespace's [`NsFs`] entry.
310    ///
311    /// Each namespace instance embeds a [`StashedDentry`] so that repeated
312    /// lookups of its pseudo-filesystem path can reuse the same [`Dentry`]
313    /// instead of allocating a new one every time.
314    fn stashed_dentry(&self) -> &StashedDentry;
315
316    /// Returns the pseudo filesystem [`Path`] associated with this namespace.
317    ///
318    /// The path is lazily created on first access and cached via
319    /// [`StashedDentry`] for subsequent calls.
320    fn get_path(self: &Arc<Self>) -> Path {
321        self.stashed_dentry().get_path(self)
322    }
323}
324
325/// A lazily-populated cache for a namespace's [`Dentry`] in [`NsFs`].
326///
327/// When a namespace is first asked for its pseudo-filesystem path, a new
328/// [`Dentry`] is allocated and stored here as a [`Weak`] reference.
329/// Subsequent requests reuse the cached dentry if it is still alive;
330/// otherwise a fresh one is created and the cache is updated.
331pub struct StashedDentry {
332    ino: u64,
333    dentry: Mutex<Weak<Dentry>>,
334}
335
336impl PartialEq for StashedDentry {
337    fn eq(&self, other: &Self) -> bool {
338        self.ino == other.ino
339    }
340}
341
342impl Eq for StashedDentry {}
343
344impl PartialOrd for StashedDentry {
345    fn partial_cmp(&self, other: &Self) -> Option<core::cmp::Ordering> {
346        Some(self.cmp(other))
347    }
348}
349
350impl Ord for StashedDentry {
351    fn cmp(&self, other: &Self) -> core::cmp::Ordering {
352        self.ino.cmp(&other.ino)
353    }
354}
355
356impl StashedDentry {
357    /// Creates a new, empty cache (no dentry allocated yet).
358    pub fn new() -> Self {
359        let ino = NsFs::singleton().alloc_id();
360        Self {
361            ino,
362            dentry: Mutex::new(Weak::new()),
363        }
364    }
365
366    /// Returns the [`Path`] for the given namespace, reusing a cached
367    /// [`Dentry`] when possible or creating a new one if the cache is stale.
368    pub(self) fn get_path<T: NsCommonOps>(&self, ns: &Arc<T>) -> Path {
369        let mut dentry = self.dentry.lock();
370        if let Some(dentry) = dentry.upgrade() {
371            return Path::new(NsFs::mount_node().clone(), dentry);
372        }
373
374        let path = NsFs::new_path(self.ino, ns.clone());
375        *dentry = Arc::downgrade(path.dentry());
376
377        path
378    }
379}

381#[derive(Clone, Copy, Debug, PartialEq)]
382pub enum NsType {
383    Cgroup,
384    Ipc,
385    Mnt,
386    #[expect(unused)]
387    Net,
388    #[expect(unused)]
389    Pid,
390    #[expect(unused)]
391    Time,
392    User,
393    Uts,
394}
395
396impl NsType {
397    const fn as_str(&self) -> &'static str {
398        match self {
399            NsType::Cgroup => "cgroup",
400            NsType::Ipc => "ipc",
401            NsType::Mnt => "mnt",
402            NsType::Net => "net",
403            NsType::Pid => "pid",
404            NsType::Time => "time",
405            NsType::User => "user",
406            NsType::Uts => "uts",
407        }
408    }
409}
410
411impl From<NsType> for CloneFlags {
412    fn from(value: NsType) -> Self {
413        match value {
414            NsType::Cgroup => CloneFlags::CLONE_NEWCGROUP,
415            NsType::Ipc => CloneFlags::CLONE_NEWIPC,
416            NsType::Mnt => CloneFlags::CLONE_NEWNS,
417            NsType::Net => CloneFlags::CLONE_NEWNET,
418            NsType::Pid => CloneFlags::CLONE_NEWPID,
419            NsType::Time => CloneFlags::CLONE_NEWTIME,
420            NsType::User => CloneFlags::CLONE_NEWUSER,
421            NsType::Uts => CloneFlags::CLONE_NEWUTS,
422        }
423    }
424}
