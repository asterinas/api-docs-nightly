32/// Represents the inode at `/proc/sys/kernel`.
33pub struct KernelDirOps;
34
35impl KernelDirOps {
36    pub fn new_inode(parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
37        // Reference:
38        // <https://elixir.bootlin.com/linux/v6.16.5/source/kernel/sysctl.c#L1765>
39        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/proc_sysctl.c#L978>
40        ProcDir::new(Self, parent, mkmod!(a+rx))
41    }
42
43    const STATIC_ENTRIES: &'static [StaticEntry] = &[
44        (
45            "cap_last_cap",
46            InodeType::File,
47            CapLastCapFileOps::new_inode,
48        ),
49        ("domainname", InodeType::File, DomainnameFileOps::new_inode),
50        ("hostname", InodeType::File, HostnameFileOps::new_inode),
51        ("osrelease", InodeType::File, OsReleaseFileOps::new_inode),
52        ("pid_max", InodeType::File, PidMaxFileOps::new_inode),
53        ("tainted", InodeType::File, TaintedFileOps::new_inode),
54        ("version", InodeType::File, VersionFileOps::new_inode),
55    ];
56}
57
58impl ProcDirOps for KernelDirOps {
59    fn lookup_child(&self, this_dir: &ProcDir<Self>, name: &str) -> Result<Arc<dyn Inode>> {
60        if let Some(child) = lookup_child_from_table(name, Self::STATIC_ENTRIES, |f| {
61            (f)(this_dir.this_weak().clone())
62        }) {
63            return Ok(child);
64        }
65
66        if name == "yama" && is_yama_enabled() {
67            return Ok(YamaDirOps::new_inode(this_dir.this_weak().clone()));
68        }
69
70        return_errno_with_message!(Errno::ENOENT, "the file does not exist");
71    }
72
73    fn visit_entries_from_offset<'a, F>(&'a self, offset: usize, visit_fn: F) -> Result<()>
74    where
75        F: FnMut(ReaddirEntry<'a>) -> Result<()>,
76    {
77        let yama_entry = is_yama_enabled().then(|| ListedEntry::new("yama", InodeType::Dir));
78
79        visit_listed_entries(
80            offset,
81            listed_entries_from_table(Self::STATIC_ENTRIES).chain(yama_entry),
82            visit_fn,
83        )
84    }
85}
