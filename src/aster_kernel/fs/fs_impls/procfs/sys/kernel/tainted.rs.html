18// Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/include/linux/panic.h#L76>
19const TAINT_FLAGS_COUNT: u64 = 20;
20const TAINT_FLAGS_MASK: u64 = (1 << TAINT_FLAGS_COUNT) - 1;
21
22static TAINTED_MASK: AtomicU64 = AtomicU64::new(0);
23
24/// Represents the inode at `/proc/sys/kernel/tainted`.
25pub struct TaintedFileOps;
26
27impl TaintedFileOps {
28    pub fn new_inode(parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
29        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/kernel/sysctl.c#L1588>
30        ProcFile::new(Self, parent, mkmod!(a+r, u+w))
31    }
32}
33
34impl ProcFileOps for TaintedFileOps {
35    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
36        let mut printer = VmPrinter::new_skip(writer, offset);
37
38        writeln!(printer, "{}", TAINTED_MASK.load(Ordering::Relaxed))?;
39
40        Ok(printer.bytes_written())
41    }
42
43    fn write_at(&self, _offset: usize, reader: &mut VmReader) -> Result<usize> {
44        lsm_hooks::on_capable(lsm_hooks::CapableContext::new(
45            UserNamespace::get_init_singleton().as_ref(),
46            current_thread!().as_posix_thread().unwrap(),
47            CapSet::SYS_ADMIN,
48        ))?;
49
50        let (mask, read_bytes) = read_u64_from(reader)?;
51
52        TAINTED_MASK.fetch_or(mask & TAINT_FLAGS_MASK, Ordering::Relaxed);
53
54        Ok(read_bytes)
55    }
56}
