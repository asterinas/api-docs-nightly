20/// Represents the inode at `/proc/[pid]`.
21pub struct PidDirOps(
22    // The `/proc/<pid>` directory is a superset of the `/proc/<pid>/task/<tid>` directory.
23    // So we embed `TidDirOps` here so that `PidDirOps` can "inherit" entries and methods
24    // from `TidDirOps`.
25    TidDirOps,
26);
27
28impl PidDirOps {
29    pub fn new_inode(pid_entry: Arc<PidEntry>, parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
30        let this = Self(TidDirOps::new(pid_entry));
31        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3493>
32        ProcDirBuilder::new(this, mkmod!(a+rx))
33            .parent(parent)
34            .build()
35            .unwrap()
36    }
37
38    pub(super) fn pid_entry(&self) -> &Arc<PidEntry> {
39        self.0.pid_entry()
40    }
41
42    pub(super) fn tid_dir_ops(&self) -> &TidDirOps {
43        &self.0
44    }
45
46    #[expect(clippy::type_complexity)]
47    const STATIC_ENTRIES: &[StaticDirEntry<fn(&PidDirOps, Weak<dyn Inode>) -> Arc<dyn Inode>>] = &[
48        ("task", InodeType::Dir, TaskDirOps::new_inode),
49        (
50            "stat",
51            InodeType::File,
52            task::stat::StatFileOps::new_process_inode,
53        ),
54    ];
55}
56
57impl DirOps for PidDirOps {
58    fn lookup_child(&self, this_dir: &ProcDir<Self>, name: &str) -> Result<Arc<dyn Inode>> {
59        if self.0.pid_entry().type_().is_none() {
60            return_errno_with_message!(Errno::ESRCH, "the process does not exist");
61        }
62
63        // Look up entries that either exist under `/proc/<pid>`
64        // but not under `/proc/<pid>/task/<tid>`,
65        // or entries whose contents differ between `/proc/<pid>` and `/proc/<pid>/task/<tid>`.
66        if let Some(child) = lookup_child_from_table(name, Self::STATIC_ENTRIES, |f| {
67            (f)(self, this_dir.this_weak().clone())
68        }) {
69            return Ok(child);
70        }
71
72        // For all other children, the content is the same under both `/proc/<pid>` and `/proc/<pid>/task/<tid>`.
73        self.0
74            .lookup_static_child(this_dir.this_weak().clone(), name)
75    }
76
77    fn revalidation_policy(&self) -> RevalidationPolicy {
78        RevalidationPolicy::REVALIDATE_EXISTS
79    }
80
81    fn revalidate_exists(&self, _name: &str, _child: &dyn Inode) -> bool {
82        matches!(self.pid_entry().type_(), Some(PidEntryType::Process))
83    }
84
85    fn visit_entries_from_offset<'a, F>(&'a self, offset: usize, visit_fn: F) -> Result<()>
86    where
87        F: FnMut(ReaddirEntry<'a>) -> Result<()>,
88    {
89        if self.0.pid_entry().type_().is_none() {
90            return_errno_with_message!(Errno::ENOENT, "the process does not exist");
91        }
92
93        const PROCESS_OVERRIDE_NAMES: &[&str] = &["stat"];
94
95        // Add the children that are inherited from `/proc/<pid>/task/<tid>` but not overridden
96        // by process-specific entries under `/proc/<pid>`.
97        let listed_entries = listed_entries_from_table(Self::STATIC_ENTRIES).chain(
98            self.0
99                .static_listed_entries()
100                .filter(|entry| !PROCESS_OVERRIDE_NAMES.contains(&entry.name())),
101        );
102
103        visit_listed_entries(offset, listed_entries, visit_fn)
104    }
105}
