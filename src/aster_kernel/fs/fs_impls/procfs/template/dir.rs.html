26/// Wraps directory-specific procfs operations as a VFS inode.
27///
28/// `ProcDir` owns the directory implementation object,
29/// tracks parent linkage for `.` and `..`,
30/// and forwards inode methods that are common to all procfs directories.
31pub struct ProcDir<D: DirOps> {
32    inner: D,
33    this: Weak<ProcDir<D>>,
34    parent: Option<Weak<dyn Inode>>,
35    common: Common,
36}
37
38impl<D: DirOps> ProcDir<D> {
39    /// Creates the root procfs directory inode.
40    pub fn new_root(
41        dir: D,
42        fs: Weak<dyn FileSystem>,
43        ino: u64,
44        sb: &SuperBlock,
45        mode: InodeMode,
46    ) -> Arc<Self> {
47        let metadata = Metadata::new_dir(ino, mode, BLOCK_SIZE, sb.container_dev_id);
48        let common = Common::new(metadata, fs);
49
50        Arc::new_cyclic(|weak_self| Self {
51            inner: dir,
52            this: weak_self.clone(),
53            parent: None,
54            common,
55        })
56    }
57
58    /// Creates a non-root procfs directory inode under `parent`.
59    pub(super) fn new(
60        dir: D,
61        fs: Weak<dyn FileSystem>,
62        parent: Option<Weak<dyn Inode>>,
63        mode: InodeMode,
64    ) -> Arc<Self> {
65        let common = {
66            let arc_fs = fs.upgrade().unwrap();
67            let procfs = arc_fs.downcast_ref::<ProcFs>().unwrap();
68            let ino = procfs.alloc_id();
69            let metadata = Metadata::new_dir(ino, mode, BLOCK_SIZE, procfs.sb().container_dev_id);
70            Common::new(metadata, fs)
71        };
72        Arc::new_cyclic(|weak_self| Self {
73            inner: dir,
74            this: weak_self.clone(),
75            parent,
76            common,
77        })
78    }
79
80    fn this(&self) -> Arc<ProcDir<D>> {
81        self.this.upgrade().unwrap()
82    }
83
84    pub fn this_weak(&self) -> &Weak<ProcDir<D>> {
85        &self.this
86    }
87
88    fn parent(&self) -> Option<Arc<dyn Inode>> {
89        self.parent.as_ref().and_then(|p| p.upgrade())
90    }
91
92    /// Returns the directory-specific procfs operations.
93    pub fn inner(&self) -> &D {
94        &self.inner
95    }
96}

153
154    fn readdir_at(&self, offset: usize, visitor: &mut dyn DirentVisitor) -> Result<usize> {
155        /// Returns the always-present `.` and `..` entries for a procfs directory.
156        fn special_entries<D: DirOps + 'static>(
157            dir: &ProcDir<D>,
158        ) -> impl Iterator<Item = (&'static str, Arc<dyn Inode>, usize)> {
159            let this_inode: Arc<dyn Inode> = dir.this();
160            let parent_inode = dir.parent().unwrap_or_else(|| this_inode.clone());
161            [(".", this_inode, 1), ("..", parent_inode, 2)].into_iter()
162        }
163
164        let try_readdir = |offset: &mut usize, visitor: &mut dyn DirentVisitor| -> Result<()> {
165            for (name, inode, next_offset) in special_entries(self) {
166                if next_offset <= *offset {
167                    continue;
168                }
169
170                visitor.visit(name, inode.ino(), inode.type_(), next_offset)?;
171                *offset = next_offset;
172            }
173
174            self.inner.visit_entries_from_offset(*offset, |child| {
175                visitor.visit(
176                    child.name.as_ref(),
177                    child.ino,
178                    child.type_,
179                    child.next_offset,
180                )?;
181                *offset = child.next_offset;
182                Ok(())
183            })?;
184            Ok(())
185        };
186
187        let mut iterate_offset = offset;
188        match try_readdir(&mut iterate_offset, visitor) {
189            Err(e) if iterate_offset == offset => Err(e),
190            _ => Ok(iterate_offset - offset),
191        }
192    }
193
194    fn link(&self, _old: &Arc<dyn Inode>, _name: &str) -> Result<()> {
195        Err(Error::new(Errno::EPERM))
196    }
197
198    fn unlink(&self, _name: &str) -> Result<()> {
199        Err(Error::new(Errno::EPERM))
200    }
201
202    fn rmdir(&self, _name: &str) -> Result<()> {
203        Err(Error::new(Errno::EPERM))
204    }
205
206    fn lookup(&self, name: &str) -> Result<Arc<dyn Inode>> {
207        if is_dot(name) {
208            return Ok(self.this());
209        }
210        if is_dotdot(name) {
211            return Ok(self.parent().unwrap_or(self.this()));
212        }
213
214        self.inner.lookup_child(self, name)
215    }
216
217    fn rename(&self, _old_name: &str, _target: &Arc<dyn Inode>, _new_name: &str) -> Result<()> {
218        Err(Error::new(Errno::EPERM))
219    }

222        self.inner.revalidation_policy()
223    }
224

226        self.inner.revalidate_exists(name, child)
227    }
228
229    fn revalidate_absent(&self, name: &str) -> bool {
230        self.inner.revalidate_absent(name)
231    }
232
233    fn seek_end(&self) -> Option<usize> {
234        // Seeking directories under `/proc` with `SEEK_END` will start from zero.
235        Some(0)
236    }

239pub trait DirOps: Sync + Send + Sized {
240    /// Looks up a child inode in `this_dir` by name.
241    fn lookup_child(&self, this_dir: &ProcDir<Self>, name: &str) -> Result<Arc<dyn Inode>>;
242
243    /// Visits readdir entries whose continuation offsets are strictly greater than `offset`.
244    ///
245    /// Build entries with helpers from the [`readdir`] submodule and forward to
246    /// [`readdir::visit_readdir_entries`]. Common patterns:
247    ///
248    /// - Static or sequential children: [`readdir::sequential_readdir_entries`]
249    ///   or the [`readdir::visit_listed_entries`] convenience wrapper.
250    /// - Children identified by a stable integer key: [`readdir::keyed_readdir_entries`].
251    /// - Mixed children: invoke [`readdir::visit_readdir_entries`] more than
252    ///   once per call, once per group. See `RootDirOps`.
253    ///
254    /// The default implementation reports no entries.
255    fn visit_entries_from_offset<'a, F>(&'a self, _offset: usize, _visit_fn: F) -> Result<()>
256    where
257        F: FnMut(ReaddirEntry<'a>) -> Result<()>,
258    {
259        Ok(())
260    }
261
262    /// Returns the dentry cache revalidation policy for this directory.
263    fn revalidation_policy(&self) -> RevalidationPolicy {
264        RevalidationPolicy::empty()
265    }
266
267    /// Checks whether a positive lookup result still exists.
268    ///
269    /// This method is only consulted when [`Self::revalidation_policy`] includes
270    /// `REVALIDATE_EXISTS`.
271    #[must_use]
272    fn revalidate_exists(&self, _name: &str, _child: &dyn Inode) -> bool {
273        true
274    }
275
276    /// Checks whether a negative lookup result is still absent.
277    ///
278    /// This method is only consulted when [`Self::revalidation_policy`] includes
279    /// `REVALIDATE_ABSENT`.
280    #[must_use]
281    fn revalidate_absent(&self, _name: &str) -> bool {
282        true
283    }
284}

286/// A statically declared procfs child entry.
287///
288/// The tuple stores the exported filename,
289/// the inode type reported to [`Inode::readdir_at`],
290/// and the constructor used by lookup paths to instantiate the inode.
291pub type StaticDirEntry<Fp> = (&'static str, InodeType, Fp);
292
293/// Looks up a statically declared child from a `StaticDirEntry` table.
294///
295/// The `constructor_adaptor` receives the stored constructor payload
296/// and can bind any per-directory context that is needed at the call site.
297pub fn lookup_child_from_table<Fp, F>(
298    name: &str,
299    table: &[StaticDirEntry<Fp>],
300    constructor_adaptor: F,
301) -> Option<Arc<dyn Inode>>
302where
303    Fp: Copy,
304    F: FnOnce(Fp) -> Arc<dyn Inode>,
305{
306    table
307        .iter()
308        .find(|(child_name, _, _)| *child_name == name)
309        .map(|(_, _, child_constructor)| (constructor_adaptor)(*child_constructor))
310}
311
312/// Converts a static procfs child table into listed entries.
313pub fn listed_entries_from_table<'a, Fp>(
314    table: &'a [StaticDirEntry<Fp>],
315) -> impl Iterator<Item = ListedEntry<'a>> + 'a
316where
317    Fp: Copy + 'a,
318{
319    table
320        .iter()
321        .map(|(name, type_, _)| ListedEntry::new(*name, *type_))
322}

324mod readdir {
325    //! `readdir` resumption types and strategies.
326    //!
327    //! When user space calls `getdents`, it passes an offset and expects the
328    //! kernel to emit entries strictly past it. Each emitted entry carries a
329    //! continuation offset that user space remembers for the next call.
330    //!
331    //! # Vocabulary
332    //!
333    //! - **Offset** - the cursor passed to `Inode::readdir_at`. Maps to
334    //!   `dirent::d_off`. Offsets `1` and `2` are reserved for `.` and `..`.
335    //!
336    //! - **Key** - an integer that uniquely identifies one dynamic child
337    //!   (PID, TID, FD). Comes from kernel state, not iteration order.
338    //!
339    //! - **Continuation offset** - the offset stored in [`ReaddirEntry`].
340    //!   Computed by [`sequential_readdir_entries`] from index or
341    //!   [`keyed_readdir_entries`] from key:
342    //!
343    //!   ```text
344    //!   continuation_offset = first_entry_offset + index + 1   // sequential_*
345    //!   continuation_offset = first_entry_offset + key   + 1   // keyed_*
346    //!   ```
347
348    use super::*;
349
350    /// A listed entry that is currently visible in a procfs directory.
351    ///
352    /// Unlike [`ReaddirEntry`], this type does not carry a continuation offset.
353    /// It is intended for higher-level directory descriptions,
354    /// while [`ReaddirEntry`] derives offsets from the iteration strategy.
355    pub struct ListedEntry<'a> {
356        pub(super) name: Cow<'a, str>,
357        pub(super) type_: InodeType,
358    }
359
360    impl<'a> ListedEntry<'a> {
361        /// Creates a listed entry with the given filename and inode type.
362        pub fn new(name: impl Into<Cow<'a, str>>, type_: InodeType) -> Self {
363            Self {
364                name: name.into(),
365                type_,
366            }
367        }
368
369        /// Returns the filename of the listed entry.
370        pub fn name(&self) -> &str {
371            self.name.as_ref()
372        }
373    }
374
375    /// A directory entry emitted by [`DirOps::visit_entries_from_offset`].
376    ///
377    /// The `next_offset` is the continuation offset that should be used as the
378    /// starting point of the next [`Inode::readdir_at`] call.
379    pub struct ReaddirEntry<'a> {
380        pub(super) name: Cow<'a, str>,
381        pub(super) ino: u64,
382        pub(super) type_: InodeType,
383        pub(super) next_offset: usize,
384    }
385
386    impl<'a> ReaddirEntry<'a> {
387        /// Creates an entry with an explicit continuation offset.
388        pub(super) fn new(
389            name: impl Into<Cow<'a, str>>,
390            ino: u64,
391            type_: InodeType,
392            next_offset: usize,
393        ) -> Self {
394            Self {
395                name: name.into(),
396                ino,
397                type_,
398                next_offset,
399            }
400        }
401    }
402
403    /// A placeholder inode number for procfs entries.
404    ///
405    /// [`Inode::readdir_at`] reports procfs entries without instantiating child inodes first,
406    /// so the inode number returned here is only a placeholder.
407    /// Lookup still creates the real child inode on demand when the path is opened.
408    //
409    // TODO: Adopt deterministic inode numbers derived from stable entity identity
410    // (for example, PID) as the long-term fix for `d_ino`/`st_ino` consistency.
411    pub(super) const PLACEHOLDER_DIRENT_INO: u64 = 2;
412
413    /// Builds sequential directory entries whose continuation offsets increase by one.
414    ///
415    /// This helper is appropriate for directories whose visible children already
416    /// have a deterministic iteration order.
417    /// It reserves offsets `1` and `2` for `.` and `..`,
418    /// so callers typically pass `2` as `first_entry_offset`.
419    pub fn sequential_readdir_entries<'a, I>(
420        offset: usize,
421        first_entry_offset: usize,
422        entries: I,
423    ) -> impl Iterator<Item = ReaddirEntry<'a>>
424    where
425        I: IntoIterator<Item = ListedEntry<'a>>,
426    {
427        entries
428            .into_iter()
429            .enumerate()
430            .filter_map(move |(idx, entry)| {
431                let next_offset = first_entry_offset.saturating_add(idx).saturating_add(1);
432                let ListedEntry { name, type_ } = entry;
433                (next_offset > offset)
434                    .then(|| ReaddirEntry::new(name, PLACEHOLDER_DIRENT_INO, type_, next_offset))
435            })
436    }
437
438    /// Builds directory entries whose offsets are derived from stable integer keys.
439    ///
440    /// This is useful for procfs directories such as `/proc`, `/proc/[pid]/task`,
441    /// and `/proc/[pid]/fd`, where Linux uses identifiers like PID, TID, or FD to
442    /// keep iteration stable across mutations.
443    ///
444    /// Names are produced only after the corresponding key passes the offset check.
445    /// The resulting continuation offset for key `k` is `first_entry_offset + k + 1`,
446    /// so callers should provide non-negative keys in increasing order.
447    /// Keys do not need to be dense,
448    /// but each key must stay stable for the duration of one directory snapshot.
449    pub fn keyed_readdir_entries<'a, I, F>(
450        offset: usize,
451        first_entry_offset: usize,
452        keys: I,
453        mut entry_fn: F,
454    ) -> impl Iterator<Item = ReaddirEntry<'a>>
455    where
456        I: IntoIterator<Item = usize>,
457        F: FnMut(usize) -> ListedEntry<'a>,
458    {
459        let start_key = offset.saturating_sub(first_entry_offset);
460        keys.into_iter().filter_map(move |key| {
461            let next_offset = first_entry_offset.saturating_add(key).saturating_add(1);
462            (key >= start_key && next_offset > offset).then(|| {
463                let ListedEntry { name, type_ } = entry_fn(key);
464                ReaddirEntry::new(name, PLACEHOLDER_DIRENT_INO, type_, next_offset)
465            })
466        })
467    }
468
469    /// Visits the given [`ReaddirEntry`] values in order, calling `visit_fn` for each one.
470    pub fn visit_readdir_entries<'a, I, F>(entries: I, mut visit_fn: F) -> Result<()>
471    where
472        I: IntoIterator<Item = ReaddirEntry<'a>>,
473        F: FnMut(ReaddirEntry<'a>) -> Result<()>,
474    {
475        for entry in entries {
476            visit_fn(entry)?;
477        }
478
479        Ok(())
480    }
481
482    /// Visits listed entries using sequential continuation offsets.
483    pub fn visit_listed_entries<'a, I, F>(offset: usize, listed: I, visit_fn: F) -> Result<()>
484    where
485        I: IntoIterator<Item = ListedEntry<'a>>,
486        F: FnMut(ReaddirEntry<'a>) -> Result<()>,
487    {
488        visit_readdir_entries(sequential_readdir_entries(offset, 2, listed), visit_fn)
489    }
490}
