32/// A trait to abstract all individual cgroup sub-controllers.
33trait SubControl {
34    fn read_attr_at(&self, name: &str, offset: usize, writer: &mut VmWriter) -> Result<usize>;
35
36    fn write_attr(&self, name: &str, reader: &mut VmReader) -> Result<usize>;
37}

39/// Defines the static properties and behaviors of a specific cgroup sub-controller.
40trait SubControlStatic: SubControl + Sized + 'static {
41    /// Creates a new instance of the sub-controller.
42    fn new(is_root: bool, is_active: bool) -> Self;
43
44    /// Returns the `SubCtrlType` enum variant corresponding to this sub-controller.
45    fn type_() -> SubCtrlType;
46
47    /// Reads and clones the `Arc` of this sub-controller in the given `Controller`.
48    fn read_from(controller: &Controller) -> Arc<SubController<Self>>;
49}

51/// The type of a sub-controller in the cgroup subsystem.
52#[derive(Clone, Copy, Debug, Eq, PartialEq)]
53pub(super) enum SubCtrlType {
54    CpuSet,
55    Cpu,
56    Memory,
57    Pids,
58}
59
60impl SubCtrlType {
61    // Keep this in the Linux-visible controller order used by `Display`.
62    const ALL: [Self; 4] = [Self::CpuSet, Self::Cpu, Self::Memory, Self::Pids];
63
64    const fn as_str(self) -> &'static str {
65        match self {
66            Self::CpuSet => "cpuset",
67            Self::Cpu => "cpu",
68            Self::Memory => "memory",
69            Self::Pids => "pids",
70        }
71    }
72}

74impl FromStr for SubCtrlType {
75    type Err = aster_systree::Error;
76
77    fn from_str(s: &str) -> Result<Self> {
78        match s {
79            "cpuset" => Ok(SubCtrlType::CpuSet),
80            "cpu" => Ok(SubCtrlType::Cpu),
81            "memory" => Ok(SubCtrlType::Memory),
82            "pids" => Ok(SubCtrlType::Pids),
83            _ => Err(Error::NotFound),
84        }
85    }
86}

88bitflags! {
89    /// A set of sub-controller types, represented as bitflags.
90    pub(super) struct SubCtrlSet: u8 {
91        const CPUSET = 1 << 0;
92        const CPU = 1 << 1;
93        const MEMORY = 1 << 2;
94        const PIDS = 1 << 3;
95    }
96}

98impl SubCtrlSet {
99    /// Checks whether a sub-control is active in the current set.
100    pub(super) fn contains_type(&self, ctrl_type: SubCtrlType) -> bool {
101        self.contains(ctrl_type.into())
102    }
103
104    /// Adds a sub-control type to the current set.
105    pub(super) fn add_type(&mut self, ctrl_type: SubCtrlType) {
106        *self |= ctrl_type.into()
107    }
108
109    /// Removes a sub-control type from the current set.
110    pub(super) fn remove_type(&mut self, ctrl_type: SubCtrlType) {
111        *self -= ctrl_type.into()
112    }
113
114    /// Returns an iterator over the sub-controller types in the current set.
115    pub(super) fn iter_types(&self) -> impl Iterator<Item = SubCtrlType> + '_ {
116        SubCtrlType::ALL
117            .into_iter()
118            .filter(|&ctrl_type| self.contains_type(ctrl_type))
119    }
120}

122impl Display for SubCtrlSet {
123    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
124        let mut iter = self.iter_types();
125        if let Some(first) = iter.next() {
126            write!(f, "{}", first.as_str())?;
127            for ctrl_type in iter {
128                write!(f, " {}", ctrl_type.as_str())?;
129            }
130        }
131
132        Ok(())
133    }
134}

165/// The sub-controller for a specific cgroup controller type.
166///
167/// If the sub-controller is inactive, the `inner` field will be `None`.
168struct SubController<T: SubControlStatic> {
169    inner: Option<T>,
170    /// The parent sub-controller in the hierarchy.
171    ///
172    /// This field is used to traverse the controller hierarchy.
173    parent: Option<Arc<SubController<T>>>,
174}
175
176impl<T: SubControlStatic> SubController<T> {
177    fn new(parent_controller: Option<&Controller>) -> Self {
178        let is_root = parent_controller.is_none();
179        let is_active = if let Some(parent) = parent_controller {
180            parent.active_set().contains_type(T::type_())
181        } else {
182            true
183        };
184
185        let inner = if is_active || T::type_() == SubCtrlType::Cpu {
186            // `cpu.stat` exists regardless of whether `+cpu` has been enabled, so the
187            // CPU sub-controller must remain instantiated even while inactive.
188            Some(T::new(is_root, is_active))
189        } else {
190            None
191        };
192
193        let parent = parent_controller.map(T::read_from);
194
195        Self { inner, parent }
196    }
197}

203impl<T: SubControlStatic> TryGetSubControl for SubController<T> {
204    fn try_get(&self) -> Option<&dyn SubControl> {
205        self.inner.as_ref().map(|sub_ctrl| sub_ctrl as _)
206    }

209/// The controller for a single cgroup.
210///
211/// This struct can manage the activation state of each sub-control, and dispatches read/write
212/// operations to the appropriate sub-controllers.
213///
214/// The following is an explanation of the activation for sub-controls and sub-controllers. When a
215/// cgroup activates a specific sub-control (e.g., memory, io), it means this control capability is
216/// being delegated to its children. Consequently, the corresponding sub-controller within the
217/// child nodes will be activated.
218///
219/// The root node serves as the origin for all these control capabilities, so the sub-controllers
220/// it possesses are always active. For any other node, only if its parent node first enables a
221/// sub-control, its corresponding sub-controller will be activated.
222pub struct Controller {
223    /// A set of types of active sub-controllers.
224    ///
225    /// Updates to this set are serialized by `CgroupMembership::write_lock()`.
226    active_set: AtomicSubCtrlSet,
227
228    cpuset: Rcu<Arc<SubController<CpuSetController>>>,
229    cpu: Rcu<Arc<SubController<CpuController>>>,
230    memory: Rcu<Arc<SubController<MemoryController>>>,
231    pids: Rcu<Arc<SubController<PidsController>>>,
232}
233
234impl Controller {
235    /// Creates a new controller manager for a cgroup.
236    pub(super) fn new(parent_controller: Option<&Controller>) -> Self {
237        let cpuset_controller = Arc::new(SubController::new(parent_controller));
238        let cpu_controller = Arc::new(SubController::new(parent_controller));
239        let memory_controller = Arc::new(SubController::new(parent_controller));
240        let pids_controller = Arc::new(SubController::new(parent_controller));
241
242        Self {
243            active_set: AtomicSubCtrlSet::new(SubCtrlSet::empty()),
244            cpuset: Rcu::new(cpuset_controller),
245            cpu: Rcu::new(cpu_controller),
246            memory: Rcu::new(memory_controller),
247            pids: Rcu::new(pids_controller),
248        }
249    }
250
251    pub(super) fn init_attr_set(builder: &mut SysAttrSetBuilder, is_root: bool) {
252        CpuSetController::init_attr_set(builder, is_root);
253        CpuController::init_attr_set(builder, is_root);
254        MemoryController::init_attr_set(builder, is_root);
255        PidsController::init_attr_set(builder, is_root);
256    }
257
258    fn read_sub(&self, ctrl_type: SubCtrlType) -> Arc<dyn TryGetSubControl> {
259        match ctrl_type {
260            SubCtrlType::CpuSet => CpuSetController::read_from(self),
261            SubCtrlType::Cpu => CpuController::read_from(self),
262            SubCtrlType::Memory => MemoryController::read_from(self),
263            SubCtrlType::Pids => PidsController::read_from(self),
264        }
265    }
266
267    /// Returns whether the attribute with the given name is absent in this controller.
268    pub(super) fn is_attr_absent(&self, name: &str) -> bool {
269        let Some((subsys, _)) = name.split_once('.') else {
270            return false;
271        };
272        let Ok(ctrl_type) = SubCtrlType::from_str(subsys) else {
273            return false;
274        };
275
276        let sub_controller = self.read_sub(ctrl_type);
277        if sub_controller.try_get().is_none() {
278            // If the sub-controller is not active, all its attributes are considered absent.
279            true
280        } else {
281            false
282        }
283    }
284
285    pub(super) fn read_attr_at(
286        &self,
287        name: &str,
288        offset: usize,
289        writer: &mut VmWriter,
290    ) -> Result<usize> {
291        let Some((subsys, _)) = name.split_once('.') else {
292            return Err(Error::NotFound);
293        };
294        let ctrl_type = SubCtrlType::from_str(subsys)?;
295
296        let sub_controller = self.read_sub(ctrl_type);
297        let Some(controller) = sub_controller.try_get() else {
298            return Err(Error::IsDead);
299        };
300
301        controller.read_attr_at(name, offset, writer)
302    }
303
304    pub(super) fn write_attr(&self, name: &str, reader: &mut VmReader) -> Result<usize> {
305        let Some((subsys, _)) = name.split_once('.') else {
306            return Err(Error::NotFound);
307        };
308        let ctrl_type = SubCtrlType::from_str(subsys)?;
309
310        let sub_controller = self.read_sub(ctrl_type);
311        let Some(controller) = sub_controller.try_get() else {
312            return Err(Error::IsDead);
313        };
314
315        controller.write_attr(name, reader)
316    }
317
318    /// Activates a sub-control of the specified type.
319    pub(super) fn activate(
320        &self,
321        ctrl_type: SubCtrlType,
322        current_node: &dyn CgroupSysNode,
323        parent_controller: Option<&Controller>,
324        cgroup_membership: &mut CgroupMembership,
325    ) -> Result<()> {
326        let mut active_set = self.active_set();
327        if active_set.contains_type(ctrl_type) {
328            return Ok(());
329        }
330
331        // A cgroup can activate the sub-control only if this
332        // sub-control has been activated in its parent cgroup.
333        if parent_controller
334            .is_some_and(|controller| !controller.active_set().contains_type(ctrl_type))
335        {
336            return Err(Error::NotFound);
337        }
338
339        active_set.add_type(ctrl_type);
340        self.active_set.store(active_set, Ordering::Relaxed);
341        self.update_sub_controllers_for_descents(ctrl_type, current_node, cgroup_membership);
342
343        Ok(())
344    }
345
346    /// Deactivates a sub-control of the specified type.
347    pub(super) fn deactivate(
348        &self,
349        ctrl_type: SubCtrlType,
350        current_node: &dyn CgroupSysNode,
351        cgroup_membership: &mut CgroupMembership,
352    ) -> Result<()> {
353        let mut active_set = self.active_set();
354        if !active_set.contains_type(ctrl_type) {
355            return Ok(());
356        }
357
358        // If any child node has activated this sub-control,
359        // the deactivation operation will be rejected.
360        for child in current_node.children() {
361            let cgroup_child = Arc::downcast::<CgroupNode>(child).unwrap();
362            // This is race-free because if a child wants to activate a sub-controller, it must
363            // acquire the write lock of `CgroupMembership` which has already been held here.
364            if cgroup_child
365                .controller()
366                .active_set()
367                .contains_type(ctrl_type)
368            {
369                return Err(Error::InvalidOperation);
370            }
371        }
372
373        active_set.remove_type(ctrl_type);
374        self.active_set.store(active_set, Ordering::Relaxed);
375        self.update_sub_controllers_for_descents(ctrl_type, current_node, cgroup_membership);
376
377        Ok(())
378    }
379
380    fn update_sub_controllers_for_descents(
381        &self,
382        ctrl_type: SubCtrlType,
383        current_node: &dyn CgroupSysNode,
384        cgroup_membership: &mut CgroupMembership,
385    ) {
386        fn update_sub_controller_for_one_child(
387            child: &Arc<dyn SysObj>,
388            ctrl_type: SubCtrlType,
389            parent_controller: &Controller,
390            cgroup_membership: &mut CgroupMembership,
391        ) {
392            let child_node = child.as_any().downcast_ref::<CgroupNode>().unwrap();
393            match ctrl_type {
394                SubCtrlType::CpuSet => {
395                    let new_controller = Arc::new(SubController::new(Some(parent_controller)));
396                    child_node.controller().cpuset.update(new_controller);
397                }
398                SubCtrlType::Cpu => {
399                    // Preserve the accumulated CPU accounting while toggling `+cpu` on the
400                    // parent. The base usage fields of `cpu.stat` are always tracked, and only
401                    // the extra throttling-related fields depend on whether the controller is active.
402                    let is_enabled = parent_controller
403                        .active_set()
404                        .contains_type(SubCtrlType::Cpu);
405                    let guard = child_node.controller().cpu.read();
406                    if is_enabled {
407                        guard.get().enable();
408                    } else {
409                        guard.get().disable();
410                    }
411                }
412                SubCtrlType::Memory => {
413                    let new_controller = Arc::new(SubController::new(Some(parent_controller)));
414                    child_node.controller().memory.update(new_controller);
415                }
416                SubCtrlType::Pids => {
417                    let mut new_controller: SubController<PidsController> =
418                        SubController::new(Some(parent_controller));
419                    if let Some(inner) = new_controller.inner.as_mut() {
420                        // When the pids sub-controller is being activated, initialize
421                        // `pids.current` with the number of processes already present
422                        // in this cgroup's subtree. The parent's counter is already
423                        // correct because charges propagated through inactive levels.
424                        let count = cgroup_membership.count_subtree_processes(child_node);
425                        if count > 0 {
426                            inner.init_count(count);
427                        }
428                    }
429                    child_node
430                        .controller()
431                        .pids
432                        .update(Arc::new(new_controller));
433                }
434            }
435        }
436
437        let mut descents = VecDeque::new();
438
439        // Subtree-control writes hold `CgroupMembership::write_lock()`, so operations
440        // protected by `CgroupMembership` like controller activation and deactivation
441        // and child node creation are serialized while the subtree updates are applied.
442        //
443        // Concurrent child removal is still fine: If a child is removed, we may update
444        // a sub-controller that's about to be destroyed, which is harmless.
445
446        // Update the direct children first.
447        current_node.visit_children_with(0, &mut |child_node| {
448            descents.push_back(child_node.clone());
449            update_sub_controller_for_one_child(child_node, ctrl_type, self, cgroup_membership);
450
451            Some(())
452        });
453
454        // Then update all the other descendent nodes.
455        while let Some(node) = descents.pop_front() {
456            let current_node = Arc::downcast::<CgroupNode>(node).unwrap();
457            let current_controller = current_node.controller();
458            current_node.visit_children_with(0, &mut |child_node| {
459                descents.push_back(child_node.clone());
460                update_sub_controller_for_one_child(
461                    child_node,
462                    ctrl_type,
463                    current_controller,
464                    cgroup_membership,
465                );
466
467                Some(())
468            });
469        }
470    }
471
472    pub(super) fn active_set(&self) -> SubCtrlSet {
473        self.active_set.load(Ordering::Relaxed)
474    }
475}

477// For pids sub-controller
478impl Controller {
479    /// Charges a process in the pids sub-controller hierarchy.
480    ///
481    /// This operation is used for explicit migration and will not enforce
482    /// the `pids.max` limit.
483    ///
484    /// Reference: <https://docs.kernel.org/admin-guide/cgroup-v2.html#pid>
485    pub(super) fn charge_pids(&self) {
486        let guard = self.pids.read();
487        let sub = guard.get();
488        sub.charge_hierarchy();
489    }
490
491    /// Pre-charges a process in the pids sub-controller hierarchy,
492    /// enforcing `pids.max` at each level.
493    ///
494    /// This is used at fork time. Returns a guard that will roll back the
495    /// charge unless it is explicitly applied.
496    ///
497    /// Note that concurrent operations of moving processes to cgroups and
498    /// setting `pids.max` may cause the actual charge to exceed the peak at
499    /// charge time.
500    pub fn pre_charge_pids<'a>(
501        &'a self,
502        _cgroup_membership: &'a CgroupMembership,
503    ) -> core::result::Result<PidsPreCharge<'a>, TryChargeError> {
504        let guard = self.pids.read();
505        let sub = guard.get();
506        sub.try_charge_hierarchy()?;
507
508        // This will not outlive `_cgroup_membership`, so we will roll back
509        // the charge for the right pids sub-controller.
510        Ok(PidsPreCharge { controller: self })
511    }
512
513    /// Uncharges a process in the pids sub-controller hierarchy.
514    pub(super) fn uncharge_pids(&self) {
515        let guard = self.pids.read();
516        let sub = guard.get();
517        sub.uncharge_hierarchy();
518    }
519}

521/// Represents a pre-charged pids slot that rolls back on drop.
522pub struct PidsPreCharge<'a> {
523    controller: &'a Controller,
524}
525
526impl PidsPreCharge<'_> {
527    /// Applies the pre-charge and prevents rollback on drop.
528    pub(super) fn apply(self) {
529        core::mem::forget(self);
530    }
531}
532
533impl Drop for PidsPreCharge<'_> {
534    fn drop(&mut self) {
535        self.controller.uncharge_pids();
536    }
537}
