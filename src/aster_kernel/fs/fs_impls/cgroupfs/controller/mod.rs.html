32/// A trait to abstract all individual cgroup sub-controllers.
33trait SubControl {
34    fn is_attr_absent(&self, _name: &str) -> bool {
35        false
36    }
37
38    fn read_attr_at(&self, name: &str, offset: usize, writer: &mut VmWriter) -> Result<usize>;
39
40    fn write_attr(&self, name: &str, reader: &mut VmReader) -> Result<usize>;
41}
42
43/// Defines the static properties and behaviors of a specific cgroup sub-controller.
44trait SubControlStatic: SubControl + Sized + 'static {
45    /// Creates a new instance of the sub-controller.
46    fn new(is_root: bool, is_active: bool) -> Self;
47
48    /// Returns the `SubCtrlType` enum variant corresponding to this sub-controller.
49    fn type_() -> SubCtrlType;
50
51    /// Reads and clones the `Arc` of this sub-controller in the given `Controller`.
52    fn read_from(controller: &Controller) -> Arc<SubController<Self>>;
53}

55/// The type of a sub-controller in the cgroup subsystem.
56#[derive(Clone, Copy, Debug, Eq, PartialEq)]
57pub(super) enum SubCtrlType {
58    CpuSet,
59    Cpu,
60    Memory,
61    Pids,
62}
63
64impl SubCtrlType {
65    // Keep this in the Linux-visible controller order used by `Display`.
66    const ALL: [Self; 4] = [Self::CpuSet, Self::Cpu, Self::Memory, Self::Pids];
67
68    const fn as_str(self) -> &'static str {
69        match self {
70            Self::CpuSet => "cpuset",
71            Self::Cpu => "cpu",
72            Self::Memory => "memory",
73            Self::Pids => "pids",
74        }
75    }
76}

78impl FromStr for SubCtrlType {
79    type Err = aster_systree::Error;
80
81    fn from_str(s: &str) -> Result<Self> {
82        match s {
83            "cpuset" => Ok(SubCtrlType::CpuSet),
84            "cpu" => Ok(SubCtrlType::Cpu),
85            "memory" => Ok(SubCtrlType::Memory),
86            "pids" => Ok(SubCtrlType::Pids),
87            _ => Err(Error::NotFound),
88        }
89    }
90}

92bitflags! {
93    /// A set of sub-controller types, represented as bitflags.
94    pub(super) struct SubCtrlSet: u8 {
95        const CPUSET = 1 << 0;
96        const CPU = 1 << 1;
97        const MEMORY = 1 << 2;
98        const PIDS = 1 << 3;
99    }
100}

102impl SubCtrlSet {
103    /// Checks whether a sub-control is active in the current set.
104    pub(super) fn contains_type(&self, ctrl_type: SubCtrlType) -> bool {
105        self.contains(ctrl_type.into())
106    }
107
108    /// Adds a sub-control type to the current set.
109    pub(super) fn add_type(&mut self, ctrl_type: SubCtrlType) {
110        *self |= ctrl_type.into()
111    }
112
113    /// Removes a sub-control type from the current set.
114    pub(super) fn remove_type(&mut self, ctrl_type: SubCtrlType) {
115        *self -= ctrl_type.into()
116    }
117
118    /// Returns an iterator over the sub-controller types in the current set.
119    pub(super) fn iter_types(&self) -> impl Iterator<Item = SubCtrlType> + '_ {
120        SubCtrlType::ALL
121            .into_iter()
122            .filter(|&ctrl_type| self.contains_type(ctrl_type))
123    }
124}
125
126impl Display for SubCtrlSet {
127    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
128        let mut iter = self.iter_types();
129        if let Some(first) = iter.next() {
130            write!(f, "{}", first.as_str())?;
131            for ctrl_type in iter {
132                write!(f, " {}", ctrl_type.as_str())?;
133            }
134        }
135
136        Ok(())
137    }
138}

169/// The sub-controller for a specific cgroup controller type.
170///
171/// If the sub-controller is inactive, the `inner` field will be `None`.
172struct SubController<T: SubControlStatic> {
173    inner: Option<T>,
174    /// The parent sub-controller in the hierarchy.
175    ///
176    /// This field is used to traverse the controller hierarchy.
177    parent: Option<Arc<SubController<T>>>,
178}
179
180impl<T: SubControlStatic> SubController<T> {
181    fn new(parent_controller: Option<&Controller>) -> Self {
182        let is_root = parent_controller.is_none();
183        let is_active = if let Some(parent) = parent_controller {
184            parent.active_set().contains_type(T::type_())
185        } else {
186            true
187        };
188
189        let inner = if is_active || T::type_() == SubCtrlType::Cpu {
190            // `cpu.stat` exists regardless of whether `+cpu` has been enabled, so the
191            // CPU sub-controller must remain instantiated even while inactive.
192            Some(T::new(is_root, is_active))
193        } else {
194            None
195        };
196
197        let parent = parent_controller.map(T::read_from);
198
199        Self { inner, parent }
200    }
201}

203trait TryGetSubControl {
204    fn try_get(&self) -> Option<&dyn SubControl>;
205}

207impl<T: SubControlStatic> TryGetSubControl for SubController<T> {
208    fn try_get(&self) -> Option<&dyn SubControl> {
209        self.inner.as_ref().map(|sub_ctrl| sub_ctrl as _)
210    }

213/// The controller for a single cgroup.
214///
215/// This struct can manage the activation state of each sub-control, and dispatches read/write
216/// operations to the appropriate sub-controllers.
217///
218/// The following is an explanation of the activation for sub-controls and sub-controllers. When a
219/// cgroup activates a specific sub-control (e.g., memory, io), it means this control capability is
220/// being delegated to its children. Consequently, the corresponding sub-controller within the
221/// child nodes will be activated.
222///
223/// The root node serves as the origin for all these control capabilities, so the sub-controllers
224/// it possesses are always active. For any other node, only if its parent node first enables a
225/// sub-control, its corresponding sub-controller will be activated.
226pub struct Controller {
227    /// A set of types of active sub-controllers.
228    ///
229    /// Updates to this set are serialized by `CgroupMembership::write_lock()`.
230    active_set: AtomicSubCtrlSet,
231
232    cpuset: Rcu<Arc<SubController<CpuSetController>>>,
233    cpu: Rcu<Arc<SubController<CpuController>>>,
234    memory: Rcu<Arc<SubController<MemoryController>>>,
235    pids: Rcu<Arc<SubController<PidsController>>>,
236}
237
238impl Controller {
239    /// Creates a new controller manager for a cgroup.
240    pub(super) fn new(parent_controller: Option<&Controller>) -> Self {
241        let cpuset_controller = Arc::new(SubController::new(parent_controller));
242        let cpu_controller = Arc::new(SubController::new(parent_controller));
243        let memory_controller = Arc::new(SubController::new(parent_controller));
244        let pids_controller = Arc::new(SubController::new(parent_controller));
245
246        Self {
247            active_set: AtomicSubCtrlSet::new(SubCtrlSet::empty()),
248            cpuset: Rcu::new(cpuset_controller),
249            cpu: Rcu::new(cpu_controller),
250            memory: Rcu::new(memory_controller),
251            pids: Rcu::new(pids_controller),
252        }
253    }
254
255    pub(super) fn init_attr_set(builder: &mut SysAttrSetBuilder, is_root: bool) {
256        CpuSetController::init_attr_set(builder, is_root);
257        CpuController::init_attr_set(builder, is_root);
258        MemoryController::init_attr_set(builder, is_root);
259        PidsController::init_attr_set(builder, is_root);
260    }
261
262    fn read_sub(&self, ctrl_type: SubCtrlType) -> Arc<dyn TryGetSubControl> {
263        match ctrl_type {
264            SubCtrlType::CpuSet => CpuSetController::read_from(self),
265            SubCtrlType::Cpu => CpuController::read_from(self),
266            SubCtrlType::Memory => MemoryController::read_from(self),
267            SubCtrlType::Pids => PidsController::read_from(self),
268        }
269    }
270
271    /// Returns whether the attribute with the given name is absent in this controller.
272    pub(super) fn is_attr_absent(&self, name: &str) -> bool {
273        let Some((subsys, _)) = name.split_once('.') else {
274            return false;
275        };
276        let Ok(ctrl_type) = SubCtrlType::from_str(subsys) else {
277            return false;
278        };
279
280        let sub_controller = self.read_sub(ctrl_type);
281        let Some(controller) = sub_controller.try_get() else {
282            // If the sub-controller is not active, all its attributes are considered absent.
283            return true;
284        };
285
286        controller.is_attr_absent(name)
287    }
288
289    pub(super) fn read_attr_at(
290        &self,
291        name: &str,
292        offset: usize,
293        writer: &mut VmWriter,
294    ) -> Result<usize> {
295        let Some((subsys, _)) = name.split_once('.') else {
296            return Err(Error::NotFound);
297        };
298        let ctrl_type = SubCtrlType::from_str(subsys)?;
299
300        let sub_controller = self.read_sub(ctrl_type);
301        let Some(controller) = sub_controller.try_get() else {
302            return Err(Error::IsDead);
303        };
304
305        controller.read_attr_at(name, offset, writer)
306    }
307
308    pub(super) fn write_attr(&self, name: &str, reader: &mut VmReader) -> Result<usize> {
309        let Some((subsys, _)) = name.split_once('.') else {
310            return Err(Error::NotFound);
311        };
312        let ctrl_type = SubCtrlType::from_str(subsys)?;
313
314        let sub_controller = self.read_sub(ctrl_type);
315        let Some(controller) = sub_controller.try_get() else {
316            return Err(Error::IsDead);
317        };
318
319        controller.write_attr(name, reader)
320    }
321
322    /// Activates a sub-control of the specified type.
323    pub(super) fn activate(
324        &self,
325        ctrl_type: SubCtrlType,
326        current_node: &dyn CgroupSysNode,
327        parent_controller: Option<&Controller>,
328        cgroup_membership: &mut CgroupMembership,
329    ) -> Result<()> {
330        let mut active_set = self.active_set();
331        if active_set.contains_type(ctrl_type) {
332            return Ok(());
333        }
334
335        // A cgroup can activate the sub-control only if this
336        // sub-control has been activated in its parent cgroup.
337        if parent_controller
338            .is_some_and(|controller| !controller.active_set().contains_type(ctrl_type))
339        {
340            return Err(Error::NotFound);
341        }
342
343        active_set.add_type(ctrl_type);
344        self.active_set.store(active_set, Ordering::Relaxed);
345        self.update_sub_controllers_for_descents(ctrl_type, current_node, cgroup_membership);
346
347        Ok(())
348    }
349
350    /// Deactivates a sub-control of the specified type.
351    pub(super) fn deactivate(
352        &self,
353        ctrl_type: SubCtrlType,
354        current_node: &dyn CgroupSysNode,
355        cgroup_membership: &mut CgroupMembership,
356    ) -> Result<()> {
357        let mut active_set = self.active_set();
358        if !active_set.contains_type(ctrl_type) {
359            return Ok(());
360        }
361
362        // If any child node has activated this sub-control,
363        // the deactivation operation will be rejected.
364        for child in current_node.children() {
365            let cgroup_child = Arc::downcast::<CgroupNode>(child).unwrap();
366            // This is race-free because if a child wants to activate a sub-controller, it must
367            // acquire the write lock of `CgroupMembership` which has already been held here.
368            if cgroup_child
369                .controller()
370                .active_set()
371                .contains_type(ctrl_type)
372            {
373                return Err(Error::InvalidOperation);
374            }
375        }
376
377        active_set.remove_type(ctrl_type);
378        self.active_set.store(active_set, Ordering::Relaxed);
379        self.update_sub_controllers_for_descents(ctrl_type, current_node, cgroup_membership);
380
381        Ok(())
382    }
383
384    fn update_sub_controllers_for_descents(
385        &self,
386        ctrl_type: SubCtrlType,
387        current_node: &dyn CgroupSysNode,
388        cgroup_membership: &mut CgroupMembership,
389    ) {
390        fn update_sub_controller_for_one_child(
391            child: &Arc<dyn SysObj>,
392            ctrl_type: SubCtrlType,
393            parent_controller: &Controller,
394            cgroup_membership: &mut CgroupMembership,
395        ) {
396            let child_node = child.as_any().downcast_ref::<CgroupNode>().unwrap();
397            match ctrl_type {
398                SubCtrlType::CpuSet => {
399                    let new_controller = Arc::new(SubController::new(Some(parent_controller)));
400                    child_node.controller().cpuset.update(new_controller);
401                }
402                SubCtrlType::Cpu => {
403                    let mut new_controller: SubController<CpuController> =
404                        SubController::new(Some(parent_controller));
405                    {
406                        let guard = child_node.controller().cpu.read();
407                        let previous_controller = guard.get();
408                        new_controller
409                            .inner
410                            .as_mut()
411                            .unwrap()
412                            .init_stats(previous_controller.inner.as_ref().unwrap());
413                    }
414                    child_node.controller().cpu.update(Arc::new(new_controller));
415                }
416                SubCtrlType::Memory => {
417                    let new_controller = Arc::new(SubController::new(Some(parent_controller)));
418                    child_node.controller().memory.update(new_controller);
419                }
420                SubCtrlType::Pids => {
421                    let mut new_controller: SubController<PidsController> =
422                        SubController::new(Some(parent_controller));
423                    if let Some(inner) = new_controller.inner.as_mut() {
424                        // When the pids sub-controller is being activated, initialize
425                        // `pids.current` with the number of processes already present
426                        // in this cgroup's subtree. The parent's counter is already
427                        // correct because charges propagated through inactive levels.
428                        let count = cgroup_membership.count_subtree_processes(child_node);
429                        if count > 0 {
430                            inner.init_count(count);
431                        }
432                    }
433                    child_node
434                        .controller()
435                        .pids
436                        .update(Arc::new(new_controller));
437                }
438            }
439        }
440
441        let mut descents = VecDeque::new();
442
443        // Subtree-control writes hold `CgroupMembership::write_lock()`, so operations
444        // protected by `CgroupMembership` like controller activation and deactivation
445        // and child node creation are serialized while the subtree updates are applied.
446        //
447        // Concurrent child removal is still fine: If a child is removed, we may update
448        // a sub-controller that's about to be destroyed, which is harmless.
449
450        // Update the direct children first.
451        current_node.visit_children_with(0, &mut |child_node| {
452            descents.push_back(child_node.clone());
453            update_sub_controller_for_one_child(child_node, ctrl_type, self, cgroup_membership);
454
455            Some(())
456        });
457
458        // Then update all the other descendent nodes.
459        while let Some(node) = descents.pop_front() {
460            let current_node = Arc::downcast::<CgroupNode>(node).unwrap();
461            let current_controller = current_node.controller();
462            current_node.visit_children_with(0, &mut |child_node| {
463                descents.push_back(child_node.clone());
464                update_sub_controller_for_one_child(
465                    child_node,
466                    ctrl_type,
467                    current_controller,
468                    cgroup_membership,
469                );
470
471                Some(())
472            });
473        }
474    }
475
476    pub(super) fn active_set(&self) -> SubCtrlSet {
477        self.active_set.load(Ordering::Relaxed)
478    }
479}

481// For pids sub-controller
482impl Controller {
483    /// Charges a process in the pids sub-controller hierarchy.
484    ///
485    /// This operation is used for explicit migration and will not enforce
486    /// the `pids.max` limit.
487    ///
488    /// Reference: <https://docs.kernel.org/admin-guide/cgroup-v2.html#pid>
489    pub(super) fn charge_pids(&self) {
490        let guard = self.pids.read();
491        let sub = guard.get();
492        sub.charge_hierarchy();
493    }
494
495    /// Pre-charges a process in the pids sub-controller hierarchy,
496    /// enforcing `pids.max` at each level.
497    ///
498    /// This is used at fork time. Returns a guard that will roll back the
499    /// charge unless it is explicitly applied.
500    ///
501    /// Note that concurrent operations of moving processes to cgroups and
502    /// setting `pids.max` may cause the actual charge to exceed the peak at
503    /// charge time.
504    pub fn pre_charge_pids<'a>(
505        &'a self,
506        _cgroup_membership: &'a CgroupMembership,
507    ) -> core::result::Result<PidsPreCharge<'a>, TryChargeError> {
508        let guard = self.pids.read();
509        let sub = guard.get();
510        sub.try_charge_hierarchy()?;
511
512        // This will not outlive `_cgroup_membership`, so we will roll back
513        // the charge for the right pids sub-controller.
514        Ok(PidsPreCharge { controller: self })
515    }
516
517    /// Uncharges a process in the pids sub-controller hierarchy.
518    pub(super) fn uncharge_pids(&self) {
519        let guard = self.pids.read();
520        let sub = guard.get();
521        sub.uncharge_hierarchy();
522    }
523}

525/// Represents a pre-charged pids slot that rolls back on drop.
526pub struct PidsPreCharge<'a> {
527    controller: &'a Controller,
528}
529
530impl PidsPreCharge<'_> {
531    /// Applies the pre-charge and prevents rollback on drop.
532    pub(super) fn apply(self) {
533        core::mem::forget(self);
534    }
535}
536
537impl Drop for PidsPreCharge<'_> {
538    fn drop(&mut self) {
539        self.controller.uncharge_pids();
540    }
541}
