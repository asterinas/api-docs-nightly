23/// A sub-controller responsible for CPU resource management in the cgroup subsystem.
24///
25/// This controller always exists so that `cpu.stat` remains readable even before `+cpu`
26/// is enabled. When inactive, only the base usage fields are exposed and the non-root
27/// CPU control files are hidden.
28pub struct CpuController {
29    /// Persistent CPU usage accounting for this cgroup.
30    stats: CpuStats,
31    /// Optional CPU resource control attributes exposed only when `+cpu` is active.
32    control: Option<CpuControl>,
33}
34
35/// CPU usage accounting that survives CPU controller toggles.
36struct CpuStats {
37    /// A counter accumulates CPU time spent executing user-space code.
38    ///
39    /// The counter is in units of one [`Jiffies`] per increment.
40    user: PerCpuCounter,
41    /// A counter accumulates CPU time spent executing kernel code on behalf of tasks.
42    ///
43    /// The counter is in units of one [`Jiffies`] per increment.
44    system: PerCpuCounter,
45}
46
47/// CPU resource controls that are reset whenever `+cpu` is re-enabled.
48struct CpuControl {
49    /// A value stores the configured relative CPU share for scheduler integration.
50    weight: AtomicU32,
51    /// A value stores the configured CPU bandwidth limit for scheduler integration.
52    max: SpinLock<CpuMax>,
53}
54
55/// A CPU bandwidth limit.
56#[derive(Clone, Copy)]
57struct CpuMax {
58    /// The configured CPU runtime budget for bandwidth enforcement.
59    quota_usec: u64,
60    /// The window over which the CPU runtime budget is intended to apply.
61    period_usec: u64,
62}
63
64/// Specifies the cgroup CPU sub-controller receives one [`Jiffies`] of charge.
65#[derive(Clone, Copy)]
66pub enum CpuStatKind {
67    /// Charges one [`Jiffies`] to `user_usec`.
68    User,
69    /// Charges one [`Jiffies`] to `system_usec`.
70    System,
71}
72
73impl CpuController {
74    pub(super) fn init_attr_set(builder: &mut SysAttrSetBuilder, is_root: bool) {
75        builder.add(SysStr::from("cpu.stat"), SysPerms::DEFAULT_RO_ATTR_PERMS);
76
77        if !is_root {
78            builder.add(SysStr::from("cpu.weight"), SysPerms::DEFAULT_RW_ATTR_PERMS);
79            builder.add(SysStr::from("cpu.max"), SysPerms::DEFAULT_RW_ATTR_PERMS);
80        }
81    }
82
83    fn new(is_active: bool) -> Self {
84        Self {
85            stats: CpuStats::new(),
86            control: is_active.then(CpuControl::new),
87        }
88    }
89
90    /// Initializes CPU usage statistics from the previous sub-controller.
91    ///
92    /// Toggling `+cpu` must reset some controller attributes such as `cpu.weight`
93    /// and `cpu.max`, but `cpu.stat` remains visible and continues to report the
94    /// accumulated CPU usage.
95    pub(super) fn init_stats(&mut self, previous: &Self) {
96        // No one cares which CPU the count is on. Therefore, choose the BSP for simplicity.
97        self.stats.user.add_on_cpu(
98            CpuId::bsp(),
99            isize::try_from(previous.stats.user.sum_all_cpus()).unwrap_or(isize::MAX),
100        );
101        self.stats.system.add_on_cpu(
102            CpuId::bsp(),
103            isize::try_from(previous.stats.system.sum_all_cpus()).unwrap_or(isize::MAX),
104        );
105    }
106
107    fn control(&self) -> Result<&CpuControl> {
108        self.control.as_ref().ok_or(Error::AttributeError)
109    }
110
111    fn read_cpu_stat(&self, printer: &mut VmPrinter) -> Result<usize> {
112        /// Converts a per-CPU counter expressed in jiffies to microseconds.
113        fn counter_usec(counter: &PerCpuCounter) -> u64 {
114            let jiffies = u64::try_from(counter.sum_all_cpus()).unwrap_or(u64::MAX);
115            u64::try_from(Jiffies::new(jiffies).as_duration().as_micros()).unwrap_or(u64::MAX)
116        }
117
118        let user_usec = counter_usec(&self.stats.user);
119        let system_usec = counter_usec(&self.stats.system);
120        writeln!(
121            printer,
122            "usage_usec {}",
123            user_usec.saturating_add(system_usec)
124        )?;
125        writeln!(printer, "user_usec {}", user_usec)?;
126        writeln!(printer, "system_usec {}", system_usec)?;
127
128        if self.control.is_some() {
129            // TODO: Support CPU bandwidth control statistics. These fields are reported as `0`
130            // for now because the cgroup CPU sub-controller does not yet implement throttling or
131            // burst accounting.
132            writeln!(printer, "nr_periods 0")?;
133            writeln!(printer, "nr_throttled 0")?;
134            writeln!(printer, "throttled_usec 0")?;
135            writeln!(printer, "nr_bursts 0")?;
136            writeln!(printer, "burst_usec 0")?;
137        }
138
139        Ok(printer.bytes_written())
140    }
141
142    /// Accounts one [`Jiffies`] of CPU time on `cpu`.
143    fn account(&self, cpu: CpuId, stat_kind: CpuStatKind) {
144        match stat_kind {
145            CpuStatKind::User => self.stats.user.add_on_cpu(cpu, 1),
146            CpuStatKind::System => self.stats.system.add_on_cpu(cpu, 1),
147        }
148    }
149}
150
151impl CpuControl {
152    fn new() -> Self {
153        // The default CPU weight is 100 and the default CPU bandwidth limit is
154        // unlimited quota with a 100ms period.
155        //
156        // Reference: <https://docs.kernel.org/admin-guide/cgroup-v2.html#cpu-interface-files>
157        const DEFAULT_WEIGHT: u32 = 100;
158        const DEFAULT_QUOTA_USEC: u64 = u64::MAX;
159        const DEFAULT_PERIOD_USEC: u64 = 100_000;
160
161        Self {
162            weight: AtomicU32::new(DEFAULT_WEIGHT),
163            max: SpinLock::new(CpuMax {
164                quota_usec: DEFAULT_QUOTA_USEC,
165                period_usec: DEFAULT_PERIOD_USEC,
166            }),
167        }
168    }
169}
170
171impl CpuStats {
172    fn new() -> Self {
173        Self {
174            user: PerCpuCounter::new(),
175            system: PerCpuCounter::new(),
176        }
177    }
178}

180impl super::SubControl for CpuController {
181    fn is_attr_absent(&self, name: &str) -> bool {
182        matches!(name, "cpu.weight" | "cpu.max") && self.control.is_none()
183    }
184
185    fn read_attr_at(&self, name: &str, offset: usize, writer: &mut VmWriter) -> Result<usize> {
186        let mut printer = VmPrinter::new_skip(writer, offset);
187
188        match name {
189            "cpu.stat" => return self.read_cpu_stat(&mut printer),
190            "cpu.weight" => {
191                let weight = self.control()?.weight.load(Ordering::Relaxed);
192                writeln!(printer, "{}", weight)?;
193            }
194            "cpu.max" => {
195                let max = *self.control()?.max.lock();
196                if max.quota_usec == u64::MAX {
197                    writeln!(printer, "max {}", max.period_usec)?;
198                } else {
199                    writeln!(printer, "{} {}", max.quota_usec, max.period_usec)?;
200                }
201            }
202            _ => return Err(Error::AttributeError),
203        }
204
205        Ok(printer.bytes_written())
206    }
207
208    fn write_attr(&self, name: &str, reader: &mut VmReader) -> Result<usize> {
209        match name {
210            "cpu.weight" => {
211                let control = self.control()?;
212
213                // The cgroup v2 CPU weight is a scheduling weight in the range 1..=10000.
214                //
215                // Reference: <https://docs.kernel.org/admin-guide/cgroup-v2.html#cpu-interface-files>
216                const MIN_WEIGHT: u32 = 1;
217                const MAX_WEIGHT: u32 = 10_000;
218
219                let (content, len) = reader
220                    .read_cstring_until_end(MAX_ATTR_SIZE)
221                    .map_err(|_| Error::PageFault)?;
222                let weight = content
223                    .to_str()
224                    .map_err(|_| Error::InvalidOperation)?
225                    .trim()
226                    .parse::<u32>()
227                    .map_err(|_| Error::InvalidOperation)?;
228                if !(MIN_WEIGHT..=MAX_WEIGHT).contains(&weight) {
229                    return Err(Error::InvalidOperation);
230                }
231
232                // TODO: Enforce cgroup CPU weight for scheduling and remove this warning.
233                warn!(
234                    "`cpu.weight` is accepted but not enforced yet; weight = {}",
235                    weight
236                );
237                control.weight.store(weight, Ordering::Relaxed);
238
239                Ok(len)
240            }
241            "cpu.max" => {
242                let control = self.control()?;
243
244                // The cgroup v2 CPU bandwidth limit is `$MAX $PERIOD`, where `$MAX` is
245                // either `max` or at least 1ms and `$PERIOD` is in the range 1ms..=1s.
246                //
247                // Reference: <https://docs.kernel.org/admin-guide/cgroup-v2.html#cpu-interface-files>
248                const MIN_QUOTA_USEC: u64 = 1_000;
249                const MIN_PERIOD_USEC: u64 = 1_000;
250                const MAX_PERIOD_USEC: u64 = 1_000_000;
251
252                let (content, len) = reader
253                    .read_cstring_until_end(MAX_ATTR_SIZE)
254                    .map_err(|_| Error::PageFault)?;
255                let content = content
256                    .to_str()
257                    .map_err(|_| Error::InvalidOperation)?
258                    .trim();
259                let mut tokens = content.split_whitespace();
260
261                let Some(quota_token) = tokens.next() else {
262                    return Err(Error::InvalidOperation);
263                };
264                let period_token = tokens.next();
265
266                // Check whether `quota_usec` is valid.
267                let quota_usec = if quota_token == "max" {
268                    u64::MAX
269                } else if let Ok(quota_usec) = quota_token.parse::<u64>() {
270                    if quota_usec < MIN_QUOTA_USEC {
271                        return Err(Error::InvalidOperation);
272                    }
273                    quota_usec
274                } else {
275                    return Err(Error::InvalidOperation);
276                };
277
278                // Check whether `period_usec` is valid.
279                let period_usec = if let Some(period_usec) =
280                    period_token.and_then(|period_token| period_token.parse::<u64>().ok())
281                {
282                    if !(MIN_PERIOD_USEC..=MAX_PERIOD_USEC).contains(&period_usec) {
283                        return Err(Error::InvalidOperation);
284                    }
285                    Some(period_usec)
286                } else {
287                    None
288                };
289
290                // TODO: Enforce CPU bandwidth throttling and remove this warning.
291                warn!(
292                    "`cpu.max` is accepted but not enforced yet; quota = {}, period = {:?}",
293                    quota_usec, period_usec
294                );
295                let mut max = control.max.lock();
296                max.quota_usec = quota_usec;
297                if let Some(period_usec) = period_usec {
298                    max.period_usec = period_usec;
299                }
300
301                Ok(len)
302            }
303            _ => Err(Error::AttributeError),
304        }
305    }
306}

322impl super::SubController<CpuController> {
323    /// Accounts one [`Jiffies`] of CPU time for this cgroup and all of its ancestors.
324    fn account_hierarchy(&self, stat_kind: CpuStatKind) {
325        // This is race-free because `charge_cpu_time` holds `cgroup_guard` when
326        // invoking this method.
327        let cpu = CpuId::current_racy();
328
329        let mut current = Some(self);
330        while let Some(node) = current {
331            node.inner.as_ref().unwrap().account(cpu, stat_kind);
332            current = node.parent.as_deref();
333        }
334    }
335}
336
337impl super::Controller {
338    /// Charges one [`Jiffies`] in the CPU sub-controller hierarchy.
339    fn charge_cpu_time<G: AsAtomicModeGuard + ?Sized>(&self, guard: &G, stat_kind: CpuStatKind) {
340        self.cpu.read_with(guard).account_hierarchy(stat_kind);
341    }
342}

344/// Charges one [`Jiffies`] of CPU time to `process`'s cgroup hierarchy.
345///
346/// If `process` is not attached to a non-root cgroup, the charge is applied to the root cgroup.
347pub fn charge_cpu_time(process: &Process, stat_kind: CpuStatKind) {
348    let cgroup_guard = process.cgroup();
349
350    if let Some(cgroup) = cgroup_guard.get() {
351        cgroup
352            .controller()
353            .charge_cpu_time(&cgroup_guard, stat_kind);
354    } else {
355        CgroupSystem::singleton()
356            .controller()
357            .charge_cpu_time(&cgroup_guard, stat_kind);
358    }
