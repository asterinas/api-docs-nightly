17/// The cgroup namespace for the unified cgroup hierarchy.
18pub struct CgroupNamespace {
19    root: Arc<dyn CgroupSysNode>,
20    owner: Arc<UserNamespace>,
21    stashed_dentry: StashedDentry,
22}
23
24impl CgroupNamespace {
25    /// Returns a reference to the singleton initial cgroup namespace.
26    pub fn get_init_singleton() -> &'static Arc<Self> {
27        static INIT: Once<Arc<CgroupNamespace>> = Once::new();
28
29        INIT.call_once(|| {
30            Arc::new(Self {
31                root: CgroupSystem::singleton().clone(),
32                owner: UserNamespace::get_init_singleton().clone(),
33                stashed_dentry: StashedDentry::new(),
34            })
35        })
36    }
37
38    /// Creates a new cgroup namespace rooted at the given cgroup.
39    pub fn new_clone(
40        current_cgroup: Option<Arc<CgroupNode>>,
41        owner: Arc<UserNamespace>,
42        posix_thread: &PosixThread,
43    ) -> Result<Arc<Self>> {
44        owner.check_cap(CapSet::SYS_ADMIN, posix_thread)?;
45
46        let root: Arc<dyn CgroupSysNode> = match current_cgroup {
47            Some(current_cgroup) => current_cgroup,
48            None => CgroupSystem::singleton().clone(),
49        };
50
51        Ok(Arc::new(Self {
52            root,
53            owner,
54            stashed_dentry: StashedDentry::new(),
55        }))
56    }
57
58    /// Returns the cgroup subtree root exposed by this namespace.
59    pub fn root_node(&self) -> Arc<dyn SysBranchNode> {
60        self.root.clone()
61    }
62
63    /// Renders the cgroup path visible from this namespace.
64    ///
65    /// Linux reports `/proc/[pid]/cgroup` relative to the namespace root.
66    /// When the target lies outside that subtree, the path climbs up with
67    /// `..` components instead of failing.
68    pub fn virtualize_path(&self, cgroup: Arc<dyn CgroupSysNode>) -> String {
69        let root_path = self.root.path();
70        let cgroup_path = cgroup.path();
71        virtualize_path_from(root_path.as_ref(), cgroup_path.as_ref())
72    }
73}
74
75fn virtualize_path_from(root_path: &str, target_path: &str) -> String {
76    fn path_components(path: &str) -> impl Iterator<Item = &str> + Clone + '_ {
77        path.split('/').filter(|component| !component.is_empty())
78    }
79
80    let root_components = path_components(root_path);
81    let target_components = path_components(target_path);
82    let shared_len = root_components
83        .clone()
84        .zip(target_components.clone())
85        .take_while(|(root_component, target_component)| root_component == target_component)
86        .count();
87    let root_suffix = root_components.clone().skip(shared_len);
88    let target_suffix = target_components.clone().skip(shared_len);
89
90    // `SysObj::path()` already returns canonical absolute cgroup paths, so
91    // virtualization reduces to computing a relative path between them.
92    root_suffix
93        .map(|_| "..")
94        .chain(target_suffix)
95        .fold(String::from("/"), |mut path, component| {
96            if path.len() > 1 {
97                path.push('/');
98            }
99            path.push_str(component);
100            path
101        })
102}
103
104impl NsCommonOps for CgroupNamespace {
105    const TYPE: NsType = NsType::Cgroup;
106
107    fn owner_user_ns(&self) -> Option<&Arc<UserNamespace>> {
108        Some(&self.owner)
109    }
110
111    fn parent(&self) -> Result<&Arc<Self>> {
112        return_errno_with_message!(
113            Errno::EINVAL,
114            "a cgroup namespace does not have a parent namespace"
115        );
116    }
117
118    fn stashed_dentry(&self) -> &StashedDentry {
119        &self.stashed_dentry
120    }
121}

123#[cfg(ktest)]
124mod tests {
125    use ostd::prelude::ktest;
126
127    use super::virtualize_path_from;
128
129    #[ktest]
130    fn virtualize_cgroup_path_for_same_node() {
131        assert_eq!(virtualize_path_from("/", "/"), "/");
132        assert_eq!(virtualize_path_from("/base", "/base"), "/");
133    }
134
135    #[ktest]
136    fn virtualize_cgroup_path_for_descendant() {
137        assert_eq!(virtualize_path_from("/base", "/base/nested"), "/nested");
138    }
139
140    #[ktest]
141    fn virtualize_cgroup_path_for_sibling() {
142        assert_eq!(virtualize_path_from("/base", "/peer"), "/../peer");
143    }
144
145    #[ktest]
146    fn virtualize_cgroup_path_for_ancestor() {
147        assert_eq!(virtualize_path_from("/base", "/"), "/..");
148    }
