3//! Classification of logical block ranges as mapped runs or sparse holes.
4
5use super::block_manager::BlockPtrTree;
6use crate::fs::ext2::prelude::*;
7
8/// Direct-I/O block-range classification for the current logical interval.
9#[derive(Debug, Eq, PartialEq)]
10pub(super) enum IoRange {
11    /// A contiguous mapped device-block range.
12    Mapped(Range<Ext2Bid>),
13    /// A hole in the file expressed as a logical-block interval.
14    Hole(Range<Iblock>),
15}
16
17/// Iterator over mapped runs and sparse holes in a logical block range.
18///
19/// Each item preserves the range classification needed by direct I/O:
20/// contiguous device-block mappings remain grouped,
21/// and sparse logical ranges remain explicit holes.
22pub(super) struct IoRangeIter<'a> {
23    range: Range<Iblock>,
24    block_ptr_tree: RwMutexReadGuard<'a, BlockPtrTree>,
25}
26
27impl<'a> IoRangeIter<'a> {
28    /// Creates an iterator over the logical block `range` using `block_ptr_tree` for lookups.
29    pub(super) fn new(
30        range: Range<Iblock>,
31        block_ptr_tree: RwMutexReadGuard<'a, BlockPtrTree>,
32    ) -> Self {
33        Self {
34            range,
35            block_ptr_tree,
36        }
37    }
38
39    /// Returns the next logical run for direct I/O planning.
40    ///
41    /// `IoRange::Mapped` returns one run whose logical blocks are backed by
42    /// contiguous physical device blocks. `IoRange::Hole` returns a known hole
43    /// run that may stop early at direct/indirect region boundaries.
44    pub(super) fn next(&mut self) -> Result<Option<IoRange>> {
45        if self.range.is_empty() {
46            return Ok(None);
47        }
48
49        let start_iblock = self.range.start;
50        let max_blocks = self.range.len() as u32;
51        let device_block_range = self
52            .block_ptr_tree
53            .lookup_block_range(start_iblock, max_blocks)?;
54
55        if device_block_range.is_empty() {
56            // Linux's ext2 documents the slow case where it iterates unmapped
57            // space block by block, as shown in the reference link below. We
58            // keep the same sparse-file semantics, but optimize by asking the
59            // block-pointer tree for a conservative hole run instead of
60            // re-walking from the root for every logical block.
61            //
62            // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/ext2/inode.c#L905>
63            let hole_len = self
64                .block_ptr_tree
65                .approx_hole_blocks(start_iblock, max_blocks)?;
66            debug_assert!(hole_len > 0);
67            self.range.start += hole_len;
68            return Ok(Some(IoRange::Hole(start_iblock..start_iblock + hole_len)));
69        }
70
71        self.range.start += device_block_range.len() as u32;
72        Ok(Some(IoRange::Mapped(device_block_range)))
73    }
74}

76#[cfg(ktest)]
77mod test {
78    use ostd::prelude::ktest;
79
80    use super::*;
81    use crate::{
82        fs::fs_impls::ext2::{
83            inode::{RAW_BLOCK_PTRS_LEN, block_manager::RawBlockPtrs},
84            test_utils::Ext2FixtureBuilder,
85        },
86        prelude::*,
87        time::clocks,
88    };
89
90    fn make_block_ptr_tree(
91        block_ptrs: [u32; RAW_BLOCK_PTRS_LEN],
92        sector_count: u32,
93        fs: &Arc<crate::fs::fs_impls::ext2::fs::Ext2>,
94    ) -> BlockPtrTree {
95        BlockPtrTree::new(
96            RawBlockPtrs::new(sector_count, block_ptrs),
97            Arc::downgrade(fs),
98        )
99    }
100
101    #[ktest]
102    fn io_range_iter_yields_mapped_and_holes() {
103        clocks::init_for_ktest();
104        let f = Ext2FixtureBuilder::new(2, 256).build().unwrap();
105
106        // Set up direct pointers: blocks 0-2 mapped, 3-6 holes, 7-8 mapped.
107        let mut block_ptrs = [0u32; RAW_BLOCK_PTRS_LEN];
108        block_ptrs[0] = 50;
109        block_ptrs[1] = 51;
110        block_ptrs[2] = 52;
111        // 3..7 are holes (zero)
112        block_ptrs[7] = 60;
113        block_ptrs[8] = 61;
114
115        let tree = make_block_ptr_tree(block_ptrs, 0, &f.ext2);
116        let lock = RwMutex::new(tree);
117        let guard = lock.read();
118
119        let mut iter = IoRangeIter::new(0..9, guard);
120
121        // First: mapped blocks 0-2 -> device bids 50..53
122        let item = iter.next().unwrap().unwrap();
123        assert_eq!(item, IoRange::Mapped(50..53));
124
125        // Second: hole covering blocks 3..7
126        let item = iter.next().unwrap().unwrap();
127        assert_eq!(item, IoRange::Hole(3..7));
128
129        // Third: mapped blocks 7-8 -> device bids 60..62
130        let item = iter.next().unwrap().unwrap();
131        assert_eq!(item, IoRange::Mapped(60..62));
132
133        // Done
134        let item = iter.next().unwrap();
135        assert!(item.is_none());
136    }
