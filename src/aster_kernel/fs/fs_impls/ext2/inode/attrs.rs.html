3//! Inode metadata operations: mode, ownership, timestamps, and xattr.
4//!
5//! Ext2 inode metadata includes the file mode, user and group ownership,
6//! timestamps, and extended attributes. Attribute mutations must update the
7//! relevant change timestamps and mark the inode descriptor dirty so the
8//! on-disk inode can be refreshed during writeback.
9
10use device_id::DeviceId;

17            inode::Metadata,

19        },
20    },
21    process::{Gid, Uid},
22};

24impl Inode {
25    /// Returns the encoded device ID for special files.
26    pub(in crate::fs::fs_impls::ext2) fn device_id(&self) -> u64 {
27        debug_assert!(self.type_ == InodeType::CharDevice || self.type_ == InodeType::BlockDevice);
28
29        let inner = self.inner.read();
30        match &inner.payload {
31            super::InodePayload::Device { device_id } => *device_id,
32            _ => 0,
33        }
34    }
35
36    /// Sets the encoded device ID for special files and persists it.
37    pub(in crate::fs::fs_impls::ext2) fn set_device_id(&self, device_id: u64) -> Result<()> {
38        if self.type_ != InodeType::CharDevice && self.type_ != InodeType::BlockDevice {
39            return_errno!(Errno::EINVAL);
40        }
41        let mut inner = self.inner.write();
42        inner.payload = super::InodePayload::Device { device_id };
43        inner.set_ctime(utils::now());
44        Ok(())
45    }
46
47    /// Returns the inode metadata snapshot for stat-like queries.
48    pub(in crate::fs::fs_impls::ext2) fn metadata(&self) -> Metadata {
49        let inner = self.inner.read();
50        let block_meta = inner.raw_block_ptrs();
51
52        let container_dev_id = match self.fs.upgrade() {
53            Some(fs) => fs.block_device().id(),
54            None => DeviceId::null(),
55        };
56        let self_dev_id =
57            if self.type_ == InodeType::CharDevice || self.type_ == InodeType::BlockDevice {
58                DeviceId::from_encoded_u64(block_meta.read_device_id())
59            } else {
60                None
61            };
62        Metadata {
63            ino: self.ino as u64,
64            size: inner.file_size(),
65            optimal_block_size: BLOCK_SIZE,
66            nr_sectors_allocated: block_meta.sector_count as usize,
67            last_access_at: inner.atime(),
68            last_modify_at: inner.mtime(),
69            last_meta_change_at: inner.ctime(),
70            type_: self.type_,
71            mode: inner.mode(),
72            nr_hard_links: inner.link_count() as usize,
73            uid: Uid::new(inner.uid()),
74            gid: Gid::new(inner.gid()),
75            container_dev_id,
76            self_dev_id,
77            birth_at: None,
78        }
79    }

156    /// Reads one extended-attribute value and writes it to `value_writer`.
157    pub(in crate::fs::fs_impls::ext2) fn get_xattr(
158        &self,
159        name: XattrName,
160        value_writer: &mut VmWriter,
161    ) -> Result<usize> {
162        let xattr = self.xattr.as_ref().ok_or(Error::with_message(
163            Errno::ENODATA,
164            "xattr not supported on this inode type",
165        ))?;
166        xattr.get_xattr(name, value_writer)
167    }
168
169    /// Lists extended attribute names in one namespace and writes them to `list_writer`.
170    pub(in crate::fs::fs_impls::ext2) fn list_xattr(
171        &self,
172        namespace: XattrNamespace,
173        list_writer: &mut VmWriter,
174    ) -> Result<usize> {
175        let Some(xattr) = self.xattr.as_ref() else {
176            return Ok(0);
177        };
178        xattr.list_xattr(namespace, list_writer)
179    }
180
181    /// Creates or replaces one extended attribute.
182    pub(in crate::fs::fs_impls::ext2) fn set_xattr(
183        &self,
184        name: XattrName,
185        value_reader: &mut VmReader,
186        flags: XattrSetFlags,
187    ) -> Result<()> {
188        let xattr = self.xattr.as_ref().ok_or(Error::with_message(
189            Errno::EPERM,
190            "xattr not supported on this inode type",
191        ))?;
192        xattr.set_xattr(name, value_reader, flags)?;
193        let new_bid = xattr.bid();
194
195        let mut inner = self.inner.write();
196        inner.set_file_acl(new_bid);
197        inner.set_ctime(utils::now());
198        Ok(())
199    }
200
201    /// Removes one extended attribute.
202    pub(in crate::fs::fs_impls::ext2) fn remove_xattr(&self, name: XattrName) -> Result<()> {
203        let xattr = self.xattr.as_ref().ok_or(Error::with_message(
204            Errno::EPERM,
205            "xattr not supported on this inode type",
206        ))?;
207        xattr.remove_xattr(name)?;
208        let new_bid = xattr.bid();
209
210        let mut inner = self.inner.write();
211        inner.set_file_acl(new_bid);
212        inner.set_ctime(utils::now());
213        Ok(())
214    }
