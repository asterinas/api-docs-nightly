3//! Symlink read and write for ext2 inodes.
4//!
5//! Ext2 distinguishes two storage strategies for symbolic link targets:
6//!
7//! - **Fast symlink** — targets up to 59 bytes are stored inline in the
8//!   60-byte raw block-pointer area of the inode (`i_block[0..15]`) without
9//!   allocating any data blocks. One byte is reserved for the
10//!   Linux-compatible trailing NUL.
11//! - **Slow symlink** — longer targets are written to an allocated data block
12//!   through the normal page-cache path.
13
14use super::{
15    super::Ext2, Inode, InodeInner, InodePayload, MAX_FAST_SYMLINK_LEN, RAW_BLOCK_PTRS_LEN,
16    block_manager::RawBlockPtrs,
17};
18use crate::fs::ext2::{prelude::*, utils};
19
20/// Inline fast-symlink target stored in the raw `i_block` byte area.
21#[derive(Debug)]
22pub(super) struct FastSymlinkTarget {
23    block_ptrs: [u32; RAW_BLOCK_PTRS_LEN],
24}
25
26impl FastSymlinkTarget {
27    pub(super) fn new(block_ptrs: [u32; RAW_BLOCK_PTRS_LEN]) -> Self {
28        Self { block_ptrs }
29    }
30
31    fn new_zeros() -> Self {
32        Self {
33            block_ptrs: [0; RAW_BLOCK_PTRS_LEN],
34        }
35    }
36
37    fn write(&mut self, target: &[u8]) -> Result<()> {
38        debug_assert!(target.len() <= MAX_FAST_SYMLINK_LEN);
39        self.block_ptrs.as_mut_bytes()[..target.len()].copy_from_slice(target);
40        Ok(())
41    }
42
43    fn read(&self, len: usize) -> Vec<u8> {
44        self.block_ptrs.as_bytes()[..len].to_vec()
45    }
46
47    pub(super) fn block_ptrs(&self) -> [u32; RAW_BLOCK_PTRS_LEN] {
48        self.block_ptrs
49    }
50}
51
52impl Inode {
53    /// Reads symbolic link target bytes and decodes them as UTF-8.
54    pub(in crate::fs::fs_impls::ext2) fn read_link(&self) -> Result<String> {
55        if self.type_ != InodeType::SymLink {
56            return_errno!(Errno::EINVAL);
57        }
58
59        let inner = self.inner.read();
60        inner.read_link()
61    }
62
63    /// Writes symbolic link target bytes into either fast-inline or slow-page-cache storage.
64    pub(in crate::fs::fs_impls::ext2) fn write_link(&self, target: &str) -> Result<()> {
65        if self.type_ != InodeType::SymLink {
66            return_errno!(Errno::EINVAL);
67        }
68
69        let target_len = target.len();
70        if target_len >= BLOCK_SIZE {
71            return_errno!(Errno::ENAMETOOLONG);
72        }
73        let fs = self.fs()?;
74        let mut inner = self.inner.write();
75        inner.write_link(&fs, target)?;
76        inner.set_mtime_ctime(utils::now());
77        Ok(())
78    }
79}
80
81impl InodeInner {
82    /// Returns whether this symlink uses fast (inline) storage.
83    pub(super) fn is_fast_symlink(&self) -> bool {
84        matches!(self.payload, InodePayload::FastSymlink { .. })
85    }
86
87    fn write_link(&mut self, fs: &Arc<Ext2>, target: &str) -> Result<()> {
88        let target_len = target.len();
89
90        // Linux reserves one byte in `i_block` for a trailing NUL.
91        if target.len() < MAX_FAST_SYMLINK_LEN {
92            // Fast path: store target inline in the block pointer area.
93            if let InodePayload::DataBacked { block_manager, .. } = &self.payload {
94                block_manager.truncate_to_byte_len(0);
95            }
96            let mut fast_target = FastSymlinkTarget::new_zeros();
97            fast_target.write(target.as_bytes())?;
98            self.payload = InodePayload::FastSymlink {
99                target: fast_target,
100            };
101        } else {
102            // Slow path: write through the page cache.
103            if !matches!(self.payload, InodePayload::DataBacked { .. }) {
104                let raw_block_ptrs = RawBlockPtrs::new(
105                    InodePayload::xattr_sectors(self.desc.file_acl),
106                    [0; RAW_BLOCK_PTRS_LEN],
107                );
108                self.payload = InodePayload::new_data_backed(
109                    self.file_size(),
110                    raw_block_ptrs,
111                    Arc::downgrade(fs),
112                );
113            }
114            self.prepare_write(fs.as_ref(), 0, target_len)?;
115            self.page_cache().write_bytes(0, target.as_bytes())?;
116        }
117
118        self.set_file_size(target_len);
119        Ok(())
120    }
121
122    fn read_link(&self) -> Result<String> {
123        let link_size = self.file_size();
124
125        if let InodePayload::FastSymlink { target } = &self.payload {
126            // Exclude the Linux-compatible trailing NUL byte from the target.
127            let read_len = link_size.min(MAX_FAST_SYMLINK_LEN - 1);
128            let target_bytes = target.read(read_len);
129            return String::from_utf8(target_bytes)
130                .map_err(|_| Error::with_message(Errno::EIO, "symlink target is not valid UTF-8"));
131        }
132
133        let mut target = vec![0u8; link_size];
134        self.page_cache().read_bytes(0, &mut target).map_err(|_| {
135            Error::with_message(Errno::EIO, "failed to read symlink target from page cache")
136        })?;
137
138        String::from_utf8(target)
139            .map_err(|_| Error::with_message(Errno::EIO, "symlink target is not valid UTF-8"))
140    }
141}
