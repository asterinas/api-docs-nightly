12const PTRS_PER_BLOCK: usize = BLOCK_SIZE / size_of::<u32>();
13const SECTORS_PER_BLOCK: u32 = (BLOCK_SIZE / SECTOR_SIZE) as u32;

16/// An ext2 inode's block-pointer tree.
17///
18/// Each tree is rooted in the 15 raw block pointer slots stored in the inode.
19/// The first 12 slots address data blocks directly,
20/// while the remaining slots address progressively deeper indirect levels.
21///
22/// | Slot index | Region           | Addressable blocks (4 KiB blocks)     |
23/// |------------|------------------|---------------------------------------|
24/// | 0 – 11     | Direct           | 12 blocks                             |
25/// | 12         | Single-indirect  | 1 024 blocks                          |
26/// | 13         | Double-indirect  | 1 024² blocks                         |
27/// | 14         | Triple-indirect  | 1 024³ blocks                         |
28///
29/// A zero pointer at any level represents an unallocated sparse range.
30/// Non-zero pointers must refer either to data blocks at the leaf level
31/// or to indirect metadata blocks at the level selected by the slot.
32#[derive(Debug)]
33pub(in crate::fs::fs_impls::ext2::inode) struct BlockPtrTree {
34    raw_block_ptrs: Dirty<RawBlockPtrs>,
35    indirect_blocks_manager: Mutex<IndirectBlockManager>,
36}

97    /// Resolves a logical block to a contiguous physical block range,
98    /// allocating new blocks if the mapping does not yet exist.
99    ///
100    /// The returned range starts at the physical block backing `iblock` and
101    /// extends for at most `max_blocks` contiguous physical blocks, but may
102    /// be shorter when the contiguous run ends or a leaf-pointer boundary is
103    /// reached.
104    ///
105    /// - `Existing`: `iblock` was already allocated; the range covers the
106    ///   contiguous prefix of physical blocks starting from that position.
107    /// - `NewlyAllocated`: `iblock` was a hole; new data blocks (and any
108    ///   required indirect metadata blocks) were allocated and linked into
109    ///   the tree. The range covers the freshly allocated data blocks.
110    //
111    // TODO: When a leaf block range is partially allocated, we walk the indirect tree
112    // twice: once to discover the existing prefix, then again to find the hole
113    // and allocate. A cursor over the leaf block could eliminate the second walk.
114    pub(in crate::fs::fs_impls::ext2::inode) fn resolve_block_range(
115        &mut self,
116        fs: &Ext2,
117        iblock: Iblock,
118        max_blocks: u32,
119    ) -> Result<ResolvedBlockRange> {
120        if max_blocks == 0 {
121            return_errno_with_message!(Errno::EINVAL, "zero block allocation requested");
122        }
123
124        let walk = self.walk_at(iblock)?;
125        if walk.is_complete() {
126            // The starting block already exists, so report its contiguous range.
127            let existing_range = self.existing_contiguous_data_blocks(&walk, max_blocks)?;
128            return Ok(ResolvedBlockRange::Existing(existing_range));
129        }
130
131        // Allocate the missing indirect metadata blocks, plus as many contiguous
132        // data blocks as we can place before the current leaf boundary.
133        let (indirect_blks, data_blks) = self.planned_allocation(&walk, max_blocks)?;
134        let mut guard = self.allocate_blocks(fs, indirect_blks, data_blks, &walk)?;
135        self.link_allocated_blocks(&guard, &walk)?;
136        guard.commit();
137        Ok(ResolvedBlockRange::NewlyAllocated(
138            guard.data_blocks.clone(),
139        ))
140    }

142    /// Truncates blocks to the new byte length (best-effort).
143    ///
144    /// This is a best-effort operation. Errors are logged but not propagated.
145    /// Leaked blocks from partial failures are recoverable by e2fsck. Linux
146    /// also follows this practice (see
147    /// <https://elixir.bootlin.com/linux/v7.0/source/fs/ext2/inode.c#L1172>).
148    pub(in crate::fs::fs_impls::ext2::inode) fn truncate_to_byte_len(
149        &mut self,
150        fs: &Ext2,
151        new_size: usize,
152    ) {
153        // First logical block to free = ceil(new_size / block_size).
154        let iblock = match Iblock::try_from(new_size.div_ceil(BLOCK_SIZE)) {
155            Ok(ib) => ib,
156            Err(_) => {
157                error!("truncate: size exceeds ext2 limits, new_size={}", new_size);
158                return;
159            }
160        };
161
162        let walk = match self.walk_at(iblock) {
163            Ok(w) => w,
164            Err(err) => {
165                error!("truncate: failed to compute block walk, err: {:?}", err);
166                return;
167            }
168        };
169
170        if walk.is_direct_data_block() {
171            if let Err(err) = self.truncate_direct_slots(fs, walk.root_slot() as usize) {
172                error!("truncate: truncate_direct_slots failed, err: {:?}", err);
173            }
174        } else if let Err(err) = self.truncate_indirect_path(fs, &walk) {
175            error!("truncate: truncate_indirect_path failed, err: {:?}", err);
176        }
177
178        self.free_indirect_roots_after(fs, walk.root_slot() as usize);
179    }

181    /// Returns a conservative hole run starting from `iblock`, capped at
182    /// `max_blocks`.
183    ///
184    /// The caller must have already verified that `iblock` is unallocated.
185    /// Returns `0` if the block is actually allocated. The result may stop
186    /// early at a direct/indirect region boundary, but it never overestimates:
187    /// every block in the returned interval is known to be unmapped.
188    pub(in crate::fs::fs_impls::ext2::inode) fn approx_hole_blocks(
189        &self,
190        iblock: Iblock,
191        max_blocks: u32,
192    ) -> Result<u32> {
193        if max_blocks == 0 {
194            return Ok(0);
195        }
196        let walk = self.walk_at(iblock)?;
197
198        if walk.is_complete() {
199            return Ok(0);
200        }
201
202        let leaf_level = walk.leaf_level();
203        let hole_level = walk.hole_level();
204        let subtree_indirect_levels = leaf_level - hole_level;
205
206        let mut hole_len = 0u32;
207        let ptrs = PTRS_PER_BLOCK as u32;
208
209        // `hole_level` is where the first zero pointer was found. A leaf-level
210        // hole needs a short slot scan; a higher-level zero pointer roots a
211        // fully sparse subtree that can be skipped without probing each block.
212        //
213        // The computed length is intentionally conservative: it may stop before
214        // the full hole ends, but it never crosses a possibly allocated pointer.
215        if subtree_indirect_levels == 0 {
216            let slot = walk.slot_at(hole_level);
217            if hole_level == 0 {
218                let scan_len = max_blocks.min(12 - slot);
219                let scan_end = slot + scan_len;
220                for &bid in &self.raw_block_ptrs.block_ptrs[slot as usize..scan_end as usize] {
221                    if bid != 0 {
222                        break;
223                    }
224                    hole_len += 1;
225                }
226            } else {
227                let parent_bid = walk.parent_bid_at(hole_level);
228                let mut indirect_block_manager = self.indirect_blocks_manager.lock();
229                let indirect_block = indirect_block_manager.find(parent_bid)?;
230                let scan_len = max_blocks.min(ptrs - slot);
231                let scan_end = slot + scan_len;
232                for idx in slot..scan_end {
233                    let bid = indirect_block.read_bid(idx as usize)?;
234                    if bid != 0 {
235                        break;
236                    }
237                    hole_len += 1;
238                }
239            }
240        } else {
241            let subtree = ptrs.pow(subtree_indirect_levels as u32);
242            let mut slot_offset = 0u32;
243            for level in (hole_level + 1)..=leaf_level {
244                slot_offset = slot_offset * ptrs + walk.slot_at(level);
245            }
246            hole_len = subtree - slot_offset;
247        }
248
249        Ok(hole_len.min(max_blocks))
250    }

252    fn truncate_direct_slots(&mut self, fs: &Ext2, start_idx: usize) -> Result<()> {
253        let start = start_idx.min(12);
254        for idx in start..12 {
255            let bid = self.raw_block_ptrs.block_ptrs[idx];
256            if bid == 0 {
257                continue;
258            }
259            fs.free_blocks(bid, 1)?;
260            self.raw_block_ptrs.block_ptrs[idx] = 0;
261            self.raw_block_ptrs.sector_count = self
262                .raw_block_ptrs
263                .sector_count
264                .saturating_sub(SECTORS_PER_BLOCK);
265        }
266        Ok(())
267    }

269    fn truncate_indirect_path(&mut self, fs: &Ext2, walk: &BlockPointerWalk) -> Result<()> {
270        // If the truncation point is exactly at the start of an indirect
271        // block (innermost slot = 0), trim trailing zero slots and re-walk the
272        // shorter path. This raises the shared level so the whole block can be
273        // detached at the parent without reading the lower subtree.
274        let trimmed = self.walk_path(walk.trim_trailing_zero_slots())?;
275
276        // `detach_level` is the highest level at which we detach and free a whole
277        // subtree. Slots to the right of the cut below `detach_level` are cleared
278        // by `free_indirect_right_side`.
279        let detach_level = self.find_detach_level(walk, &trimmed)?;
280
281        if let Some(detached_bid) = self.detach_subtree_root(walk, &trimmed, detach_level)? {
282            self.free_block_subtree(fs, detached_bid, walk.subtree_indirect_levels(detach_level));
283        }
284
285        self.free_indirect_right_side(fs, walk, &trimmed, detach_level)?;
286
287        Ok(())
288    }

290    fn find_detach_level(
291        &self,
292        walk: &BlockPointerWalk,
293        trimmed: &BlockPointerWalk,
294    ) -> Result<usize> {
295        // Start from the deepest level if the walk was complete, or from the
296        // level where the walk stopped on a zero pointer.
297        let mut detach_level = if trimmed.is_complete() {
298            trimmed.leaf_level()
299        } else {
300            trimmed.hole_level()
301        };
302
303        // Walk upward: if all entries to the left of the truncation slot in
304        // an indirect block are zero, the entire block can be detached at the
305        // parent level. Stop as soon as a non-zero kept entry is found.
306        while detach_level > 0 {
307            let current_bid = trimmed.parent_bid_at(detach_level);
308
309            let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
310            let block = indirect_blocks_manager.find(current_bid)?;
311            let keep_entries = walk.slot_at(detach_level) as usize;
312            let mut all_zero = true;
313            for idx in 0..keep_entries {
314                if block.read_bid(idx)? != 0 {
315                    all_zero = false;
316                    break;
317                }
318            }
319            if !all_zero {
320                break;
321            }
322
323            // Left side is all zeros; move up one level.
324            detach_level -= 1;
325        }
326
327        Ok(detach_level)
328    }

330    fn detach_subtree_root(
331        &mut self,
332        walk: &BlockPointerWalk,
333        trimmed: &BlockPointerWalk,
334        detach_level: usize,
335    ) -> Result<Option<Ext2Bid>> {
336        let detached_bid = if detach_level == 0 {
337            // Subtree root is referenced directly from `inode.block_ptrs[]`.
338            let slot = walk.root_slot() as usize;
339            let bid = self.raw_block_ptrs.block_ptrs[slot];
340            self.raw_block_ptrs.block_ptrs[slot] = 0;
341            bid
342        } else {
343            // Subtree root is a slot inside a parent indirect block.
344            let parent_bid = trimmed.parent_bid_at(detach_level);
345            let slot = walk.slot_at(detach_level) as usize;
346            let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
347            let parent_block = indirect_blocks_manager.find_mut(parent_bid)?;
348            let bid = parent_block.read_bid(slot)?;
349            parent_block.write_bid(slot, 0)?;
350            bid
351        };
352
353        if detached_bid != 0 {
354            Ok(Some(detached_bid))
355        } else {
356            Ok(None)
357        }
358    }

360    fn free_indirect_right_side(
361        &mut self,
362        fs: &Ext2,
363        walk: &BlockPointerWalk,
364        trimmed: &BlockPointerWalk,
365        detach_level: usize,
366    ) -> Result<()> {
367        let ptrs_per_block = PTRS_PER_BLOCK;
368        // From `detach_level` down to level 1, clear all slots to the right of the
369        // truncation slot and free their subtrees.
370        for level in (1..=detach_level).rev() {
371            let current_bid = trimmed.parent_bid_at(level);
372
373            let start_idx = (walk.slot_at(level) as usize) + 1;
374            let child_indirect_levels = walk.subtree_indirect_levels(level);
375            // Collect block numbers before releasing the lock; `free_block_subtree`
376            // may re-acquire the indirect block manager recursively.
377            let child_blocks = {
378                let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
379                let block = indirect_blocks_manager.find_mut(current_bid)?;
380                let mut child_blocks = Vec::new();
381                for idx in start_idx..ptrs_per_block {
382                    let bid = block.read_bid(idx)?;
383                    if bid == 0 {
384                        continue;
385                    }
386                    block.write_bid(idx, 0)?;
387                    child_blocks.push(bid);
388                }
389                child_blocks
390            };
391            for bid in child_blocks {
392                self.free_block_subtree(fs, bid, child_indirect_levels);
393            }
394        }
395        Ok(())
396    }

398    fn free_indirect_roots_after(&mut self, fs: &Ext2, root_slot: usize) {
399        // The conditions are cumulative: truncating before the single-indirect
400        // root frees slots 12, 13, and 14; truncating at or before the
401        // single-indirect root frees slots 13 and 14; truncating at or before
402        // the double-indirect root frees slot 14.
403        if root_slot < 12 {
404            let bid = self.raw_block_ptrs.block_ptrs[12];
405            if bid != 0 {
406                self.raw_block_ptrs.block_ptrs[12] = 0;
407                self.free_block_subtree(fs, bid, 1);
408            }
409        }
410        if root_slot <= 12 {
411            let bid = self.raw_block_ptrs.block_ptrs[13];
412            if bid != 0 {
413                self.raw_block_ptrs.block_ptrs[13] = 0;
414                self.free_block_subtree(fs, bid, 2);
415            }
416        }
417        if root_slot <= 13 {
418            let bid = self.raw_block_ptrs.block_ptrs[14];
419            if bid != 0 {
420                self.raw_block_ptrs.block_ptrs[14] = 0;
421                self.free_block_subtree(fs, bid, 3);
422            }
423        }
424    }

437    /// Computes a `BlockPointerWalk` by walking the on-disk pointer chain.
438    fn walk_path(&self, path: BlockPointerPath) -> Result<BlockPointerWalk> {
439        let top_bid = self.raw_block_ptrs.block_ptrs[path.slots[0] as usize];
440        let num_levels = path.num_levels;
441        let mut visited_entries = SmallVec::new();
442        if top_bid == 0 {
443            // Zero pointer means the path is broken at level 0.
444            return Ok(BlockPointerWalk {
445                path,
446                is_complete: false,
447                visited_entries,
448            });
449        }
450        visited_entries.push(top_bid);
451
452        let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
453        let mut parent_bid = top_bid;
454        for level in 1..num_levels {
455            let next_bid = indirect_blocks_manager
456                .find(parent_bid)?
457                .read_bid(path.slots[level] as usize)?;
458            if next_bid == 0 {
459                return Ok(BlockPointerWalk {
460                    path,
461                    is_complete: false,
462                    visited_entries,
463                });
464            }
465            visited_entries.push(next_bid);
466            parent_bid = next_bid;
467        }
468
469        Ok(BlockPointerWalk {
470            path,
471            is_complete: true,
472            visited_entries,
473        })
474    }

476    fn existing_contiguous_data_blocks(
477        &self,
478        walk: &BlockPointerWalk,
479        max_blocks: u32,
480    ) -> Result<Range<Ext2Bid>> {
481        if max_blocks == 0 {
482            return_errno_with_message!(Errno::EINVAL, "zero block range requested");
483        }
484        // A partial walk means the logical block lands in a hole, so there is
485        // no existing physical range to report.
486        if !walk.is_complete() {
487            return Ok(0..0);
488        }
489        // The last chain entry is the first existing data block for `iblock`.
490        let first_bid = walk.bid_at(walk.leaf_level());
491        if first_bid == 0 {
492            return Ok(0..0);
493        }
494
495        let max_count = walk.max_blocks_in_leaf(max_blocks);
496        let mut count = 1u32;
497        let start_slot = walk.leaf_slot() as usize;
498
499        if walk.is_direct_data_block() {
500            // Direct blocks are stored inline in the inode, so scan forward in
501            // `i_block[]` while the next physical block stays contiguous.
502            while count < max_count {
503                let slot = start_slot + count as usize;
504                let Some(&next_bid) = self.raw_block_ptrs.block_ptrs.get(slot) else {
505                    break;
506                };
507                if next_bid == 0 || next_bid != first_bid.saturating_add(count) {
508                    break;
509                }
510                count += 1;
511            }
512        } else {
513            // Indirect cases share the same contiguity rule, but the leaf slots
514            // live in the final indirect block reached by the walk.
515            let leaf_bid = walk.parent_bid_at(walk.leaf_level());
516            let mut indirect_block_manager = self.indirect_blocks_manager.lock();
517            let leaf_block = indirect_block_manager.find(leaf_bid)?;
518            while count < max_count {
519                let slot = start_slot + count as usize;
520                let next_bid = leaf_block.read_bid(slot)?;
521                if next_bid == 0 || next_bid != first_bid.saturating_add(count) {
522                    break;
523                }
524                count += 1;
525            }
526        }
527
528        Ok(first_bid..first_bid.saturating_add(count))
529    }

531    /// Returns how many indirect metadata blocks and data blocks are needed.
532    ///
533    /// The returned tuple is `(indirect_blks, data_blks)`. `indirect_blks`
534    /// covers the missing metadata levels below the first hole in `walk`,
535    /// while `data_blks` is the contiguous data range that can be placed in the
536    /// current leaf without crossing its block-pointer boundary.
537    fn planned_allocation(&self, walk: &BlockPointerWalk, max_blocks: u32) -> Result<(u32, u32)> {
538        if max_blocks == 0 {
539            return_errno_with_message!(Errno::EINVAL, "zero block allocation requested");
540        }
541
542        // Missing child levels correspond to indirect metadata blocks that
543        // must be allocated before any data block can be linked into the tree.
544        // Never allocate past the current leaf boundary even if the caller asks
545        // for a larger contiguous range.
546        let indirect_blks = walk.subtree_indirect_levels(walk.hole_level());
547        let max_data_blks = walk.max_blocks_in_leaf(max_blocks);
548        if indirect_blks > 0 {
549            // If the path is incomplete, the fresh leaf starts empty, so we
550            // can request the whole bounded data range immediately.
551            return Ok((indirect_blks, max_data_blks));
552        }
553
554        // The indirect path already exists, so only count how many consecutive
555        // empty slots remain in the current leaf.
556        let start_slot = walk.leaf_slot() as usize;
557        let mut count = 1u32;
558        if walk.is_direct_data_block() {
559            // Direct blocks live inline in the inode's `i_block[]`.
560            while count < max_data_blks {
561                let slot = start_slot + count as usize;
562                let Some(&next_bid) = self.raw_block_ptrs.block_ptrs.get(slot) else {
563                    break;
564                };
565                if next_bid != 0 {
566                    break;
567                }
568                count += 1;
569            }
570        } else {
571            // Indirect cases use the final indirect block as the allocation leaf.
572            let leaf_bid = walk.parent_bid_at(walk.leaf_level());
573            let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
574            let leaf_block = indirect_blocks_manager.find(leaf_bid)?;
575            while count < max_data_blks {
576                let slot = start_slot + count as usize;
577                if leaf_block.read_bid(slot)? != 0 {
578                    break;
579                }
580                count += 1;
581            }
582        }
583
584        Ok((0, count))
585    }

587    /// Frees all blocks in a pointer subtree.
588    fn free_block_subtree(&mut self, fs: &Ext2, block_bid: Ext2Bid, indirect_levels: u32) {
589        if block_bid == 0 {
590            return;
591        }
592
593        if indirect_levels == 0 {
594            if let Err(err) = fs.free_blocks(block_bid, 1) {
595                // Best-effort free path logs errors and proceeds.
596                error!(
597                    "free_block_subtree: failed to free data block {}: {:?}",
598                    block_bid, err
599                );
600                return;
601            }
602            self.raw_block_ptrs.sector_count = self
603                .raw_block_ptrs
604                .sector_count
605                .saturating_sub(SECTORS_PER_BLOCK);
606            return;
607        }
608
609        let child_blocks = {
610            let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
611            match indirect_blocks_manager.read_child_bids(block_bid) {
612                Ok(children) => {
613                    indirect_blocks_manager.remove(block_bid);
614                    children
615                }
616                Err(_) => {
617                    // Skip the damaged subtree after logging the read failure so
618                    // cleanup can continue for the remaining subtree.
619                    error!(
620                        "free_block_subtree: failed to read indirect block {} (indirect_levels {})",
621                        block_bid, indirect_levels
622                    );
623                    return;
624                }
625            }
626        };
627
628        for bid in child_blocks {
629            self.free_block_subtree(fs, bid, indirect_levels - 1);
630        }
631
632        if let Err(err) = fs.free_blocks(block_bid, 1) {
633            error!(
634                "free_block_subtree: failed to free indirect block {}: {:?}",
635                block_bid, err
636            );
637            return;
638        }
639        self.raw_block_ptrs.sector_count = self
640            .raw_block_ptrs
641            .sector_count
642            .saturating_sub(SECTORS_PER_BLOCK);
643    }
644
645    /// Allocates indirect metadata blocks and data blocks for a missing path.
646    ///
647    /// Indirect blocks are allocated first so the eventual data range can be
648    /// placed near the newly allocated metadata when possible. The returned
649    /// guard frees all allocated blocks on drop unless `commit` is called after
650    /// the blocks have been spliced into the tree.
651    fn allocate_blocks<'a>(
652        &self,
653        fs: &'a Ext2,
654        indirect_blks: u32,
655        data_blks: u32,
656        walk: &BlockPointerWalk,
657    ) -> Result<BlockAllocGuard<'a>> {
658        let mut alloc_goal = walk
659            .visited_entries
660            .last()
661            .copied()
662            .unwrap_or(self.raw_block_ptrs.block_ptrs[0]);
663
664        // Allocate the missing indirect metadata first, then place data blocks
665        // immediately after the metadata run when possible.
666        let mut guard = BlockAllocGuard::new(fs, indirect_blks);
667
668        let mut remaining_indirect_blks = indirect_blks;
669        while remaining_indirect_blks > 0 {
670            let allocated = fs.alloc_blocks(remaining_indirect_blks, alloc_goal)?;
671            debug_assert!(allocated.end >= allocated.start);
672            let allocated_count = allocated.end - allocated.start;
673            debug_assert!(allocated_count > 0 && allocated_count <= remaining_indirect_blks);
674
675            alloc_goal = allocated.end;
676            remaining_indirect_blks -= allocated_count;
677            guard.extend_indirect_blocks(allocated);
678        }
679
680        let data_blocks_range = fs.alloc_blocks(data_blks, alloc_goal)?;
681        debug_assert!(data_blocks_range.end >= data_blocks_range.start);
682        let allocated_count = data_blocks_range.end - data_blocks_range.start;
683        guard.track_data_blocks(data_blocks_range.clone());
684        debug_assert!(allocated_count > 0 && allocated_count <= data_blks);
685
686        // We must wait until the blocks are initialized before we can proceed.
687        // Otherwise, concurrent page faults may see uninitialized blocks.
688        //
689        // TODO: In the write path, if the entire block is going to be
690        // overwritten, we should write the real payload instead of zeroing it.
691        Self::zero_new_blocks(fs, &data_blocks_range)?;
692
693        Ok(guard)
694    }
695
696    /// Zeroes newly allocated data blocks before exposing them via mapped reads.
697    fn zero_new_blocks(fs: &Ext2, block_range: &Range<Ext2Bid>) -> Result<()> {
698        let mut io_batch = IoBatch::with_capacity(1);
699
700        let bio_segment = BioSegment::alloc(block_range.len(), BioDirection::ToDevice);
701        let mut segment_writer = bio_segment.writer().unwrap();
702        segment_writer.fill_zeros(block_range.len() * BLOCK_SIZE);
703        fs.write_blocks_async(block_range.start, bio_segment, None, &mut io_batch)?;
704
705        io_batch.wait_all()?;
706        Ok(())
707    }

709    /// Splices allocated blocks into the block-pointer tree.
710    ///
711    /// If no indirect metadata blocks were allocated, this only fills data bids
712    /// into existing leaf slots. Otherwise it builds the missing indirect chain,
713    /// links the chain root into the first hole, and updates the inode sector
714    /// count for both metadata and data blocks.
715    fn link_allocated_blocks(
716        &mut self,
717        guard: &BlockAllocGuard,
718        walk: &BlockPointerWalk,
719    ) -> Result<()> {
720        let indirect_blocks = &guard.indirect_blocks;
721        let data_blocks = &guard.data_blocks;
722
723        let total_blks = indirect_blocks.len() as u32 + data_blocks.len() as u32;
724        let added_sectors = total_blks * SECTORS_PER_BLOCK;
725        let new_block_count = self
726            .raw_block_ptrs
727            .sector_count
728            .checked_add(added_sectors)
729            .ok_or_else(|| Error::with_message(Errno::EIO, "inode block count overflow"))?;
730
731        if indirect_blocks.is_empty() {
732            // The path already exists; only fill data pointers into existing slots.
733            self.fill_existing_leaf_slots(walk, data_blocks)?;
734        } else {
735            self.link_new_indirect_chain(walk, indirect_blocks, data_blocks)?;
736        }
737
738        self.raw_block_ptrs.sector_count = new_block_count;
739        Ok(())
740    }
741
742    /// Fills data pointers into existing slots when no new indirect blocks are needed.
743    fn fill_existing_leaf_slots(
744        &mut self,
745        walk: &BlockPointerWalk,
746        data_blocks: &Range<Ext2Bid>,
747    ) -> Result<()> {
748        let hole_level = walk.hole_level();
749
750        if hole_level == 0 {
751            let slot = walk.root_slot() as usize;
752            self.write_data_range_to_direct_slots(slot, data_blocks)?;
753        } else {
754            let parent_bid = walk.parent_bid_at(hole_level);
755            let slot = walk.slot_at(hole_level) as usize;
756            let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
757            let parent_block = indirect_blocks_manager.find_mut(parent_bid)?;
758            Self::write_data_range_to_indirect_block(parent_block, slot, data_blocks)?;
759        }
760        Ok(())
761    }
762
763    /// Builds a new indirect-block chain and links it into the tree.
764    fn link_new_indirect_chain(
765        &mut self,
766        walk: &BlockPointerWalk,
767        indirect_blocks: &[Ext2Bid],
768        data_blocks: &Range<Ext2Bid>,
769    ) -> Result<()> {
770        let hole_level = walk.hole_level();
771        let chain_root_bid = indirect_blocks[0];
772        let mut indirect_blocks_manager = self.indirect_blocks_manager.lock();
773
774        let result = (|| -> Result<()> {
775            // Build the indirect blocks chain: each block points to the next;
776            // the leaf block points to the allocated data range.
777            debug_assert_eq!(
778                indirect_blocks.len() as u32,
779                walk.subtree_indirect_levels(hole_level)
780            );
781            let mut child_levels = walk.child_levels(hole_level);
782            for (indirect_blk_idx, new_bid) in indirect_blocks.iter().copied().enumerate() {
783                let level = child_levels
784                    .next()
785                    .expect("one child level per allocated indirect block");
786
787                let mut block = IndirectBlock::alloc_new(new_bid)?;
788                let start_slot = walk.slot_at(level) as usize;
789                if indirect_blk_idx + 1 == indirect_blocks.len() {
790                    Self::write_data_range_to_indirect_block(&mut block, start_slot, data_blocks)?;
791                } else {
792                    block.write_bid(start_slot, indirect_blocks[indirect_blk_idx + 1])?;
793                }
794                indirect_blocks_manager.insert(block)?;
795            }
796
797            // Splice the chain root into the parent pointer.
798            if hole_level == 0 {
799                let slot = walk.root_slot() as usize;
800                if self.raw_block_ptrs.block_ptrs[slot] != 0 {
801                    return_errno_with_message!(
802                        Errno::EIO,
803                        "block pointer changed during allocation"
804                    );
805                }
806                self.raw_block_ptrs.block_ptrs[slot] = chain_root_bid;
807            } else {
808                let parent_bid = walk.parent_bid_at(hole_level);
809                let parent_block = indirect_blocks_manager.find_mut(parent_bid)?;
810                let slot = walk.slot_at(hole_level) as usize;
811                if parent_block.read_bid(slot)? != 0 {
812                    return_errno_with_message!(
813                        Errno::EIO,
814                        "block pointer changed during allocation"
815                    );
816                }
817                parent_block.write_bid(slot, chain_root_bid)?;
818            }
819
820            Ok(())
821        })();
822
823        if let Err(err) = result {
824            // Evict any indirect blocks already inserted into the cache.
825            for &bid in indirect_blocks.iter() {
826                indirect_blocks_manager.remove(bid);
827            }
828            return Err(err);
829        }
830
831        Ok(())
832    }
833
834    /// Writes a contiguous range of allocated data block IDs into direct
835    /// `i_block[]` slots.
836    ///
837    /// Callers guarantee that the target slots are zero because `walk_path`
838    /// confirmed the hole and the inode lock is held throughout.
839    fn write_data_range_to_direct_slots(
840        &mut self,
841        start_slot: usize,
842        data_range: &Range<Ext2Bid>,
843    ) -> Result<()> {
844        let end_slot = start_slot + data_range.len();
845        let slots = &mut self.raw_block_ptrs.block_ptrs[start_slot..end_slot];
846
847        for (entry, bid) in slots.iter_mut().zip(data_range.clone()) {
848            debug_assert_eq!(*entry, 0);
849            *entry = bid;
850        }
851        Ok(())
852    }
853
854    /// Writes a contiguous range of allocated data block IDs into an indirect
855    /// block.
856    ///
857    /// The caller guarantees that every target slot is zero; the debug
858    /// assertion keeps that invariant explicit during development.
859    fn write_data_range_to_indirect_block(
860        block: &mut IndirectBlock,
861        start_slot: usize,
862        data_range: &Range<Ext2Bid>,
863    ) -> Result<()> {
864        let end_slot = start_slot + data_range.len();
865
866        for (slot, bid) in (start_slot..end_slot).zip(data_range.clone()) {
867            debug_assert_eq!(block.read_bid(slot).unwrap_or(0), 0);
868            block.write_bid(slot, bid)?;
869        }
870        Ok(())
871    }

874/// On-disk block pointer state mirrored from `RawInode`.
875///
876/// The mirrored fields are `i_blocks` (sector count) and the 15-entry
877/// `i_block[]` array.
878///
879/// This copy exists to break a locking cycle between `Inode` and
880/// `PageCacheBackend`. Both need to look up or allocate blocks, but `Inode`
881/// also drives `PageCacheBackend` operations. Giving both sides access to this
882/// struct instead of routing everything through the `InodeInner` lock avoids a
883/// deadlock. The struct is the authoritative in-memory state and is written
884/// back to the inode on sync.
885#[derive(Clone, Copy, Debug)]
886pub(in crate::fs::fs_impls::ext2::inode) struct RawBlockPtrs {
887    pub sector_count: u32,
888    pub block_ptrs: [u32; RAW_BLOCK_PTRS_LEN],
889}
890
891impl RawBlockPtrs {
892    /// Creates a `RawBlockPtrs` from the given sector count and pointer array.
893    pub(in crate::fs::fs_impls::ext2::inode) fn new(
894        sector_count: u32,
895        block_ptrs: [u32; RAW_BLOCK_PTRS_LEN],
896    ) -> Self {
897        Self {
898            sector_count,
899            block_ptrs,
900        }
901    }
902
903    /// Reads the ext2 special-file device encoding stored in `i_block`.
904    pub(in crate::fs::fs_impls::ext2::inode) fn read_device_id(&self) -> u64 {
905        let (major, minor) = if self.block_ptrs[0] != 0 {
906            let old_encoded_device = self.block_ptrs[0];
907            // Old_decode_dev: (major << 8) | minor with 8-bit major/minor.
908            (
909                ((old_encoded_device >> 8) & 0xFF),
910                (old_encoded_device & 0xFF),
911            )
912        } else {
913            let encoded_device = self.block_ptrs[1];
914            // Decode the extended major/minor bit layout.
915            (
916                ((encoded_device & 0xFFF00) >> 8),
917                ((encoded_device & 0xFF) | ((encoded_device >> 12) & 0xFFF00)),
918            )
919        };
920
921        encode_device_numbers(major, minor)
922    }
923
924    /// Writes a device ID into the ext2 special-file `i_block` layout.
925    pub(in crate::fs::fs_impls::ext2::inode) fn write_device_id(&mut self, device_id: u64) {
926        let (major, minor) = decode_device_numbers(device_id);
927
928        // Old_valid_dev: MAJOR/MINOR must both fit in 8 bits.
929        if major < 256 && minor < 256 {
930            self.block_ptrs[0] = (major << 8) | minor;
931            self.block_ptrs[1] = 0;
932        } else {
933            self.block_ptrs[0] = 0;
934            self.block_ptrs[1] = (minor & 0xFF) | (major << 8) | ((minor & !0xFF) << 12);
935            self.block_ptrs[2] = 0;
936        }
937    }
938}

949/// Full block-pointer path from `i_block[]` to a logical data block.
950///
951/// `slots[0]` is always the slot in the inode's `i_block[]` array.
952/// `slots[1..num_levels]` are the successive slots inside indirect
953/// blocks, from outermost to innermost.
954#[derive(Clone, Copy, Debug)]
955struct BlockPointerPath {
956    slots: [u32; MAX_BLOCK_POINTER_LEVELS],
957    num_levels: usize,
958}
959
960impl BlockPointerPath {
961    /// Creates a block-pointer path for a logical block index.
962    fn new(iblock: Iblock) -> Result<Self> {
963        let ptrs = PTRS_PER_BLOCK as u32;
964        let ptr_shift = ptrs.trailing_zeros();
965        let ptr_mask = ptrs - 1;
966        let double_blocks = 1u32 << (ptr_shift * 2);
967
968        if iblock < 12 {
969            return Ok(Self::from_slots([iblock, 0, 0, 0], 1));
970        }
971        let block = iblock - 12;
972
973        if block < ptrs {
974            return Ok(Self::from_slots([12, block, 0, 0], 2));
975        }
976        let block = block - ptrs;
977
978        if block < double_blocks {
979            return Ok(Self::from_slots(
980                [13, block >> ptr_shift, block & ptr_mask, 0],
981                3,
982            ));
983        }
984        let block = block - double_blocks;
985
986        if (block >> (ptr_shift * 2)) < ptrs {
987            return Ok(Self::from_slots(
988                [
989                    14,
990                    block >> (ptr_shift * 2),
991                    (block >> ptr_shift) & ptr_mask,
992                    block & ptr_mask,
993                ],
994                4,
995            ));
996        }
997
998        return_errno_with_message!(Errno::EINVAL, "block number exceeds ext2 pointer tree");
999    }
1000
1001    fn from_slots(slots: [u32; MAX_BLOCK_POINTER_LEVELS], num_levels: usize) -> Self {
1002        debug_assert!((1..=MAX_BLOCK_POINTER_LEVELS).contains(&num_levels));
1003        Self { slots, num_levels }
1004    }
1005}

1007/// Traversal state from an inode's `i_block[]` array to a logical data block.
1008///
1009/// The state combines the slot path selected by the logical block index with
1010/// the block IDs observed in the on-disk pointer chain.
1011///
1012/// For a 4 KiB block size (1024 pointers per indirect block):
1013///
1014/// ```text
1015///   Logical block 0..11 (direct):
1016///     levels=1:  i_block[0..12]  →  data block
1017///
1018///   Logical block 12..1035 (single indirect):
1019///     levels=2:  i_block[12]  →  indirect[slot]  →  data block
1020///
1021///   Logical block 1036..1049611 (double indirect):
1022///     levels=3:  i_block[13]  →  L1[a]  →  L2[b]  →  data block
1023///
1024///   Logical block 1049612..1074791435 (triple indirect):
1025///     levels=4:  i_block[14]  →  L1[a]  →  L2[b]  →  L3[c]  →  data block
1026/// ```
1027///
1028/// `level` is a 0-based index into this chain: level 0 is always the inode
1029/// `i_block[]` slot, and levels 1.. are successive indirect-block slots.
1030#[derive(Debug)]
1031struct BlockPointerWalk {
1032    /// Slot path from inode `i_block[]` to the target logical block.
1033    ///
1034    /// This is pure addressing metadata; physical block numbers read while
1035    /// following the path are recorded in `visited_entries`.
1036    path: BlockPointerPath,
1037    /// Whether every pointer in the path was non-zero.
1038    is_complete: bool,
1039    /// Non-zero entries visited so far, one per completed level.
1040    ///
1041    /// If the walk is incomplete, the first hole is at
1042    /// `visited_entries.len()`.
1043    visited_entries: SmallVec<[Ext2Bid; MAX_BLOCK_POINTER_LEVELS]>,
1044}
1045
1046impl BlockPointerWalk {
1047    /// Returns the root slot in inode `i_block[]`.
1048    fn root_slot(&self) -> u32 {
1049        self.path.slots[0]
1050    }
1051
1052    /// Returns whether this walk names an inline direct data-block slot.
1053    fn is_direct_data_block(&self) -> bool {
1054        self.root_slot() < 12
1055    }
1056
1057    /// Returns the total number of levels in the pointer chain.
1058    #[cfg(ktest)]
1059    fn num_levels(&self) -> usize {
1060        self.path.num_levels
1061    }
1062
1063    /// Returns the slot at the given level of the block pointer tree.
1064    fn slot_at(&self, level: usize) -> u32 {
1065        debug_assert!(level < self.path.num_levels);
1066        self.path.slots[level]
1067    }
1068
1069    /// Returns the slot at the deepest (leaf) level.
1070    fn leaf_slot(&self) -> u32 {
1071        self.slot_at(self.leaf_level())
1072    }
1073
1074    /// Returns the index (0-based) of the deepest (leaf) level in the pointer chain.
1075    fn leaf_level(&self) -> usize {
1076        self.path.num_levels - 1
1077    }
1078
1079    /// Trims trailing zero slots from the innermost levels.
1080    ///
1081    /// Used by `truncate_to_byte_len` to raise the shared path level when the
1082    /// truncation point sits at the start of an indirect block.
1083    fn trim_trailing_zero_slots(&self) -> BlockPointerPath {
1084        let mut num_levels = self.path.num_levels;
1085        while num_levels > 1 && self.path.slots[num_levels - 1] == 0 {
1086            num_levels -= 1;
1087        }
1088
1089        BlockPointerPath::from_slots(self.path.slots, num_levels)
1090    }
1091
1092    /// Returns the indirect-block levels in the subtree rooted at `root_level`.
1093    ///
1094    /// Equals `leaf_level() - root_level`: the number of indirect blocks that
1095    /// must still be traversed from the node at `root_level` down to data
1096    /// blocks.
1097    /// Pass this directly as the `indirect_levels` argument to `free_block_subtree`.
1098    fn subtree_indirect_levels(&self, root_level: usize) -> u32 {
1099        debug_assert!(root_level <= self.leaf_level());
1100        (self.leaf_level() - root_level) as u32
1101    }
1102
1103    /// Returns the child levels below `level`, from outermost to innermost.
1104    fn child_levels(&self, level: usize) -> impl Iterator<Item = usize> {
1105        debug_assert!(level <= self.leaf_level());
1106        (level + 1)..=self.leaf_level()
1107    }
1108
1109    /// Returns the maximum block count before crossing this walk's leaf boundary.
1110    fn max_blocks_in_leaf(&self, max_blocks: u32) -> u32 {
1111        let boundary = if self.is_direct_data_block() {
1112            11 - self.leaf_slot()
1113        } else {
1114            PTRS_PER_BLOCK as u32 - 1 - self.leaf_slot()
1115        };
1116        max_blocks.min(boundary + 1)
1117    }
1118
1119    fn is_complete(&self) -> bool {
1120        self.is_complete
1121    }
1122
1123    fn bid_at(&self, level: usize) -> Ext2Bid {
1124        debug_assert!(level < self.visited_entries.len());
1125        self.visited_entries[level]
1126    }
1127
1128    fn parent_bid_at(&self, level: usize) -> Ext2Bid {
1129        debug_assert!(level > 0);
1130        self.bid_at(level - 1)
1131    }
1132
1133    fn hole_level(&self) -> usize {
1134        debug_assert!(!self.is_complete());
1135        self.visited_entries.len()
1136    }
1137}

1139/// Rollback guard for metadata and data blocks allocated through the block-pointer tree.
1140#[derive(Debug)]
1141struct BlockAllocGuard<'a> {
1142    fs: &'a Ext2,
1143    indirect_blocks: Vec<Ext2Bid>,
1144    data_blocks: Range<Ext2Bid>,
1145    committed: bool,
1146}
1147
1148impl<'a> BlockAllocGuard<'a> {
1149    fn new(fs: &'a Ext2, indirect_blocks: u32) -> Self {
1150        Self {
1151            fs,
1152            indirect_blocks: Vec::with_capacity(indirect_blocks as usize),
1153            data_blocks: Range { start: 0, end: 0 },
1154            committed: false,
1155        }
1156    }
1157
1158    fn extend_indirect_blocks(&mut self, indirect_blocks: Range<Ext2Bid>) {
1159        self.indirect_blocks.extend(indirect_blocks);
1160    }
1161
1162    fn track_data_blocks(&mut self, data_blocks: Range<Ext2Bid>) {
1163        debug_assert!(self.data_blocks.is_empty());
1164        self.data_blocks = data_blocks;
1165    }
1166
1167    fn commit(&mut self) {
1168        self.committed = true;
1169    }
1170}
1171
1172impl Drop for BlockAllocGuard<'_> {
1173    fn drop(&mut self) {
1174        if self.committed {
1175            return;
1176        }
1177
1178        for &bid in self.indirect_blocks.iter() {
1179            if let Err(err) = self.fs.free_blocks(bid, 1) {
1180                error!(
1181                    "failed to free indirect block {} in rollback: {:?}",
1182                    bid, err
1183                );
1184            }
1185        }
1186
1187        if self.data_blocks.is_empty() {
1188            return;
1189        }
1190        let free_result = self
1191            .fs
1192            .free_blocks(self.data_blocks.start, self.data_blocks.len() as u32);
1193        if let Err(err) = free_result {
1194            error!("failed to free data blocks in rollback: {:?}", err);
1195        }
1196    }
1197}
