17/// Bridges the inode's logical file view and the physical block device.
18///
19/// `InodeBlockManager` maintains the relationship between logical file
20/// blocks, ext2 physical block addresses, and the page-cache view of file
21/// contents. Sparse logical ranges are represented by absent block mappings,
22/// while allocated ranges must remain consistent with the inode's
23/// block-pointer tree.
24#[derive(Debug)]
25pub(super) struct InodeBlockManager {
26    /// Translates logical file block indices to physical device block addresses and
27    /// manages block allocation and truncation.
28    block_ptr_tree: RwMutex<BlockPtrTree>,
29    /// Cached `npages` bound for `PageCache`.
30    npages: AtomicUsize,
31    /// File system handle for indirect I/O and BIO submission.
32    fs: Weak<Ext2>,
33}
34
35impl InodeBlockManager {
36    /// Creates a new block manager wrapping the given block-pointer tree.
37    pub(super) fn new(block_ptr_tree: BlockPtrTree, fs: Weak<Ext2>, npages: usize) -> Self {
38        Self {
39            block_ptr_tree: RwMutex::new(block_ptr_tree),
40            npages: AtomicUsize::new(npages),
41            fs,
42        }
43    }
44
45    /// Returns a strong reference to the owning filesystem.
46    pub(super) fn fs(&self) -> Result<Arc<Ext2>> {
47        self.fs
48            .upgrade()
49            .ok_or_else(|| Error::with_message(Errno::EIO, "filesystem already dropped"))
50    }
51
52    /// Looks up a single logical block -> physical block.
53    pub(super) fn lookup_block(&self, iblock: Iblock) -> Result<Option<Ext2Bid>> {
54        let tree = self.block_ptr_tree.read();
55        tree.lookup_block(iblock)
56    }
57
58    /// Returns a snapshot of the raw block pointer state.
59    pub(super) fn raw_block_ptrs(&self) -> RawBlockPtrs {
60        *self.block_ptr_tree.read().raw_block_ptrs()
61    }
62
63    /// Returns whether the block-pointer tree has uncommitted changes.
64    pub(super) fn is_dirty(&self) -> bool {
65        self.block_ptr_tree.read().is_dirty()
66    }
67
68    /// Clears the block-pointer dirty flag after writeback.
69    pub(super) fn clear_dirty(&self) {
70        self.block_ptr_tree.write().clear_dirty();
71    }
72
73    /// Creates an iterator over existing and hole block ranges.
74    ///
75    /// The returned iterator holds a read lock on the block-pointer tree for
76    /// its entire lifetime. Callers should consume it promptly to avoid
77    /// blocking concurrent allocations or truncations on this inode.
78    pub(super) fn iter_io_ranges(&self, block_range: Range<Iblock>) -> IoRangeIter<'_> {
79        let tree = self.block_ptr_tree.read();
80        IoRangeIter::new(block_range, tree)
81    }
82
83    /// Truncates blocks to the new_size (best-effort).
84    pub(super) fn truncate_to_byte_len(&self, new_size: usize) {
85        let fs = match self.fs() {
86            Ok(fs) => fs,
87            Err(err) => {
88                error!("truncate: failed to get fs reference, err: {:?}", err);
89                return;
90            }
91        };
92        let mut tree = self.block_ptr_tree.write();
93        tree.truncate_to_byte_len(&fs, new_size)
94    }
95
96    /// Flushes all dirty cached indirect blocks to the device.
97    pub(super) fn sync_indirect_blocks(&self) -> Result<()> {
98        self.block_ptr_tree.write().sync_indirect_blocks()
99    }
100
101    /// Allocates missing data blocks that cover the requested logical block range.
102    pub(super) fn allocate_range_blocks(&self, start_block: usize, end_block: usize) -> Result<()> {
103        let fs = self.fs()?;
104        let mut tree = self.block_ptr_tree.write();
105        let mut current_block = start_block;
106        while current_block < end_block {
107            let iblock = Iblock::try_from(current_block)
108                .map_err(|_| Error::with_message(Errno::EINVAL, "logical block number overflow"))?;
109            let remaining = u32::try_from(end_block - current_block)
110                .map_err(|_| Error::with_message(Errno::EINVAL, "block range length overflow"))?;
111
112            let block_range = tree.resolve_block_range(&fs, iblock, remaining)?;
113            match block_range {
114                ResolvedBlockRange::Existing(range) => {
115                    debug_assert!(!range.is_empty());
116                    current_block += range.len();
117                }
118                ResolvedBlockRange::NewlyAllocated(range) => {
119                    debug_assert!(!range.is_empty());
120                    current_block += range.len();
121                }
122            }
123        }
124        Ok(())
125    }
126
127    /// Updates the cached page-cache capacity bound.
128    pub(super) fn set_npages(&self, npages: usize) {
129        self.npages.store(npages, Ordering::Release);
130    }
131}
132
133impl BlockAsPageCacheBackend for InodeBlockManager {
134    fn submit_read_bio(
135        &self,
136        idx: usize,
137        bio_segment: BioSegment,
138        complete_fn: BioCompleteFn,
139        io_batch: &mut IoBatch,
140    ) -> Result<()> {
141        if idx >= self.npages.load(Ordering::Acquire) {
142            return_errno_with_message!(Errno::EINVAL, "invalid read size");
143        }
144        let iblock = Iblock::try_from(idx)
145            .map_err(|_| Error::with_message(Errno::EINVAL, "logical block number overflow"))?;
146        match self.lookup_block(iblock)? {
147            Some(bid) => {
148                let fs = self.fs()?;
149                fs.read_blocks_async(bid, bio_segment, Some(complete_fn), io_batch)
150            }
151            None => {
152                // Encountered a hole, zero fill the page.
153                complete_fn(BioStatus::Zeros);
154                Ok(())
155            }
156        }
157    }
158
159    fn submit_write_bio(
160        &self,
161        idx: usize,
162        bio_segment: BioSegment,
163        complete_fn: BioCompleteFn,
164        io_batch: &mut IoBatch,
165    ) -> Result<()> {
166        if idx >= self.npages.load(Ordering::Acquire) {
167            return_errno_with_message!(Errno::EINVAL, "invalid write size");
168        }
169        let iblock = Iblock::try_from(idx)
170            .map_err(|_| Error::with_message(Errno::EINVAL, "logical block number overflow"))?;
171        let fs = self.fs()?;
172
173        // TODO: Refactor `lookup_block` and `resolve_block_range`. Currently
174        // `bio_segment.nblocks()` is always 1, so the point lookup works correctly, but the
175        // semantics are misleading because `bio_segment` can represent a contiguous block range.
176        // The block is already allocated; write it directly.
177        if let Some(bid) = self.lookup_block(iblock)? {
178            return fs.write_blocks_async(bid, bio_segment, Some(complete_fn), io_batch);
179        }
180
181        // Encounter a hole; allocate a block. Since we dropped the read lock
182        // above, another thread may have filled the hole; the `Existing` arm
183        // below handles that race.
184        let mut tree = self.block_ptr_tree.write();
185        let step = tree.resolve_block_range(&fs, iblock, bio_segment.nblocks() as u32)?;
186        let bid = match step {
187            ResolvedBlockRange::NewlyAllocated(r) => r.start,
188            ResolvedBlockRange::Existing(r) => r.start,
189        };
190
191        fs.write_blocks_async(bid, bio_segment, Some(complete_fn), io_batch)
192    }
