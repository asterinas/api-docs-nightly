5use hashbrown::HashMap;

7use crate::fs::fs_impls::ext2::{Ext2, prelude::*};

9/// Inode-local cache for ext2 indirect metadata blocks.
10///
11/// Resident indirect blocks belong to exactly one inode and are owned by the
12/// corresponding `BlockPtrTree`. Dirty indirect blocks remain associated with
13/// that inode until they are synced or removed.
14///
15/// # Invariants
16///
17/// - Cached blocks are never shared across inodes.
18/// - Dirty blocks must remain reachable until they are either written back or
19///   discarded as part of freeing their owning subtree.
20#[derive(Debug)]
21pub(super) struct IndirectBlockManager {
22    cache: HashMap<Ext2Bid, IndirectBlock>,
23    fs: Weak<Ext2>,
24}
25
26impl IndirectBlockManager {
27    /// Creates a new indirect block manager.
28    pub(super) fn new(fs: Weak<Ext2>) -> Self {
29        Self {
30            cache: HashMap::new(),
31            fs,
32        }
33    }
34
35    /// Returns a shared reference to the indirect block at `bid`, loading it if needed.
36    pub(super) fn find(&mut self, bid: Ext2Bid) -> Result<&IndirectBlock> {
37        self.ensure_cached(bid)?;
38        Ok(self
39            .cache
40            .get(&bid)
41            .expect("indirect block must be cached after ensure_cached"))
42    }
43
44    /// Returns a mutable reference to the indirect block at `bid`, loading it if needed.
45    pub(super) fn find_mut(&mut self, bid: Ext2Bid) -> Result<&mut IndirectBlock> {
46        self.ensure_cached(bid)?;
47        Ok(self
48            .cache
49            .get_mut(&bid)
50            .expect("indirect block must be cached after ensure_cached"))
51    }
52
53    /// Reads all non-zero child block pointers from an indirect block.
54    pub(super) fn read_child_bids(&mut self, bid: Ext2Bid) -> Result<Vec<Ext2Bid>> {
55        let ptrs_per_block = BLOCK_SIZE / size_of::<u32>();
56        let block = self.find(bid)?;
57        let mut child_bids = Vec::new();
58        for idx in 0..ptrs_per_block {
59            match block.read_bid(idx) {
60                Ok(0) => continue,
61                Ok(child_bid) => child_bids.push(child_bid),
62                Err(_) => break,
63            }
64        }
65        Ok(child_bids)
66    }
67
68    /// Inserts an indirect block into the cache.
69    pub(super) fn insert(&mut self, block: IndirectBlock) -> Result<()> {
70        self.cache.insert(block.bid(), block);
71        Ok(())
72    }
73
74    /// Removes an indirect block from the cache without writing it back.
75    pub(super) fn remove(&mut self, bid: Ext2Bid) -> Option<IndirectBlock> {
76        self.cache.remove(&bid)
77    }
78
79    /// Writes all dirty cached indirect blocks back to the device.
80    pub(super) fn sync(&mut self) -> Result<()> {
81        let fs = self.fs()?;
82        let dirty_bids: Vec<Ext2Bid> = self
83            .cache
84            .iter()
85            .filter_map(|(bid, block)| block.is_dirty().then_some(*bid))
86            .collect();
87
88        // TODO: do this flush job in best effort (skip the failed flush).
89        for bid in dirty_bids {
90            let block = self
91                .cache
92                .get_mut(&bid)
93                .expect("dirty indirect block key must still be cached");
94            Self::write_block(&fs, block)?;
95        }
96
97        Ok(())
98    }
99
100    fn ensure_cached(&mut self, bid: Ext2Bid) -> Result<()> {
101        if !self.cache.contains_key(&bid) {
102            let block = self.load_block(bid)?;
103            self.cache.insert(bid, block);
104        }
105        Ok(())
106    }
107
108    fn load_block(&self, bid: Ext2Bid) -> Result<IndirectBlock> {
109        let fs = self.fs()?;
110        let mut block = IndirectBlock::alloc_uninit()?;
111        block.set_bid(bid);
112        let bio_segment = BioSegment::new_from_segment(
113            Segment::<()>::from(block.frame().clone()).into(),
114            BioDirection::FromDevice,
115        );
116        fs.read_blocks(bid, bio_segment)
117            .map_err(|_| Error::with_message(Errno::EIO, "failed to submit indirect block read"))?;
118        block.mark_clean();
119        Ok(block)
120    }
121
122    fn write_block(fs: &Arc<Ext2>, block: &mut IndirectBlock) -> Result<()> {
123        if !block.is_dirty() {
124            return Ok(());
125        }
126
127        let bio_segment = BioSegment::new_from_segment(
128            Segment::<()>::from(block.frame().clone()).into(),
129            BioDirection::ToDevice,
130        );
131        fs.write_blocks(block.bid(), bio_segment).map_err(|_| {
132            Error::with_message(Errno::EIO, "failed to submit indirect block writeback")
133        })?;
134
135        block.mark_clean();
136        Ok(())
137    }
138
139    fn fs(&self) -> Result<Arc<Ext2>> {
140        self.fs
141            .upgrade()
142            .ok_or_else(|| Error::with_message(Errno::EIO, "filesystem already dropped"))
143    }
144}

146/// One resident indirect metadata block.
147///
148/// Each `IndirectBlock` is a single page-frame that holds one ext2 indirect
149/// metadata block (i.e. an array of `u32` block-number pointers). It tracks
150/// its own device block address (`bid`) and a dirty flag so that write-back
151/// can be skipped for clean entries.
152#[derive(Debug)]
153pub(super) struct IndirectBlock {
154    block: Frame<()>,
155    bid: Ext2Bid,
156    dirty: bool,
157}
158
159impl IndirectBlock {
160    /// Allocates an uninitialized indirect block frame.
161    pub(super) fn alloc_uninit() -> Result<Self> {
162        Ok(Self {
163            block: FrameAllocOptions::new().zeroed(false).alloc_frame()?,
164            bid: 0,
165            dirty: false,
166        })
167    }
168
169    /// Allocates a zeroed indirect block and marks it dirty.
170    pub(super) fn alloc_new(bid: Ext2Bid) -> Result<Self> {
171        Ok(Self {
172            block: FrameAllocOptions::new().alloc_frame()?,
173            bid,
174            dirty: true,
175        })
176    }
177
178    /// Reads the block pointer at slot `idx`.
179    pub(super) fn read_bid(&self, idx: usize) -> Result<Ext2Bid> {
180        let slot_offset = self.slot_offset(idx);
181        self.block
182            .read_val(slot_offset)
183            .map_err(|_| Error::with_message(Errno::EIO, "failed to read indirect pointer"))
184    }
185
186    /// Writes a block pointer at slot `idx` and marks the block dirty.
187    pub(super) fn write_bid(&mut self, idx: usize, bid: Ext2Bid) -> Result<()> {
188        let slot_offset = self.slot_offset(idx);
189        self.block
190            .write_val(slot_offset, &bid)
191            .map_err(|_| Error::with_message(Errno::EIO, "failed to write indirect pointer"))?;
192        self.dirty = true;
193        Ok(())
194    }
195
196    /// Returns whether the block has been modified since the last write-back.
197    pub(super) fn is_dirty(&self) -> bool {
198        self.dirty
199    }
200
201    /// Returns the device block address of this indirect block.
202    pub(super) fn bid(&self) -> Ext2Bid {
203        self.bid
204    }
205
206    fn frame(&self) -> &Frame<()> {
207        &self.block
208    }
209
210    fn mark_clean(&mut self) {
211        self.dirty = false;
212    }
213
214    fn set_bid(&mut self, bid: Ext2Bid) {
215        self.bid = bid;
216    }
217
218    fn slot_offset(&self, idx: usize) -> usize {
219        debug_assert!(idx < BLOCK_SIZE / size_of::<Ext2Bid>());
220        idx * size_of::<Ext2Bid>()
221    }
222}
