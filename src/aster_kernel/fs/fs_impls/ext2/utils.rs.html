3//! Small helpers shared across the ext2 module.
4//!
5//! - `Dirty` — a wrapper that tracks whether its inner value has been
6//!   mutated, for writeback scheduling.
7//! - `IsPowerOf` — a trait for testing whether a number is a power of
8//!   another (used by sparse-superblock backup-group selection).
9//! - `now` — a convenience function that reads the real-time coarse clock.
10//! - `duration_to_ext2_secs` — converts kernel durations to clamped ext2
11//!   timestamp seconds.
12
13use core::ops::MulAssign;
14
15use super::prelude::*;
16use crate::prelude::warn;

18pub(super) trait IsPowerOf: Copy + Sized + MulAssign + PartialOrd {
19    /// Returns whether `self` equals `x^k` for some `k > 0`.
20    ///
21    /// `x` must be greater than 1.
22    fn is_power_of(&self, x: Self) -> bool {
23        let mut power = x;
24        while power < *self {
25            power *= x;
26        }
27
28        power == *self
29    }
30}
31
32macro_rules! impl_ipo_for {
33    ($($ipo_ty:ty),*) => {
34        $(impl IsPowerOf for $ipo_ty {})*
35    };
36}
37
38impl_ipo_for!(
39    u8, u16, u32, u64, u128, i8, i16, i32, i64, i128, isize, usize
40);

42/// A value with dirty tracking.
43pub(super) struct Dirty<T: Debug> {
44    value: T,
45    dirty: bool,
46}
47
48impl<T: Debug> Dirty<T> {
49    /// Creates a new `Dirty` value without setting the dirty flag.
50    pub(super) fn new(val: T) -> Dirty<T> {
51        Dirty {
52            value: val,
53            dirty: false,
54        }
55    }
56
57    /// Creates a new `Dirty` value with the dirty flag set.
58    pub(super) fn _new_dirty(val: T) -> Dirty<T> {
59        Dirty {
60            value: val,
61            dirty: true,
62        }
63    }
64
65    /// Returns whether the value is dirty.
66    pub(super) fn is_dirty(&self) -> bool {
67        self.dirty
68    }
69
70    /// Clears the dirty flag.
71    pub(super) fn clear_dirty(&mut self) {
72        self.dirty = false;
73    }
74}
75
76impl<T: Debug> Deref for Dirty<T> {
77    type Target = T;
78
79    fn deref(&self) -> &T {
80        &self.value
81    }
82}
83
84impl<T: Debug> DerefMut for Dirty<T> {
85    fn deref_mut(&mut self) -> &mut T {
86        self.dirty = true;
87        &mut self.value
88    }
89}
90
91impl<T: Debug> Drop for Dirty<T> {
92    fn drop(&mut self) {
93        if self.is_dirty() {
94            warn!("dropped while dirty: {:?}", self.value);
95        }
96    }
97}
98
99impl<T: Debug> Debug for Dirty<T> {
100    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
101        let tag = if self.dirty { "Dirty" } else { "Clean" };
102        write!(f, "[{}] {:?}", tag, self.value)
103    }
104}
105
106/// Returns the current time.
107pub(super) fn now() -> Duration {
108    crate::time::clocks::RealTimeCoarseClock::get().read_time()
109}
110
111/// Converts a `Duration` to a 32-bit ext2 timestamp, clamping to `u32::MAX`.
112pub(super) fn duration_to_ext2_secs(d: Duration) -> u32 {
113    u32::try_from(d.as_secs()).unwrap_or(u32::MAX)
114}
