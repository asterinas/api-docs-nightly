39/// An `OverlayFs` is a union pseudo file system employed to merge
40/// upper and lower directories that potentially comes from different
41/// file systems, into a single, unified view at a designated mount point.
42pub struct OverlayFs {
43    /// The writable upper layer.
44    upper: OverlayUpper,
45    /// The read-only lower layer.
46    lower: OverlayLower,
47    /// The work directory.
48    work: OverlayWork,
49    /// Configuration settings.
50    config: OverlayConfig,
51    /// Super block.
52    sb: OverlaySB,
53    /// Unique inode number generator.
54    next_ino: AtomicU64,
55    /// FS event subscriber stats for this file system.
56    fs_event_subscriber_stats: FsEventSubscriberStats,
57    /// Weak self reference.
58    self_: Weak<OverlayFs>,
59}
60
61/// The mutable upper layer of an `OverlayFs`.
62struct OverlayUpper {
63    path: Path,
64}
65
66/// The immutable lower layer of an `OverlayFs`.
67/// A lower layer may contain multiple `Path`s with different mount points.
68struct OverlayLower {
69    /// Layered dentries from top to bottom.
70    paths: Vec<Path>,
71    // TODO: Support data-only lower layers.
72}
73
74/// The work directory. Must reside in
75/// the same file system as the upper layer.
76struct OverlayWork {
77    path: Path,
78    // TODO: Align the work directory's behavior with Linux.
79}
80
81/// Provides an unified inode abstraction for its user, internal it
82/// manages the layered regular inodes.
83struct OverlayInode {
84    /// The unique inode number.
85    ino: u64,
86    /// The inode type.
87    type_: InodeType,
88    /// The name parameter in `Inode::create` issued by the parent.
89    /// This field is used to build hierarchical upper inodes.
90    /// The lock is intended to implement `rename`.
91    name_upon_creation: SpinLock<String>,
92    /// The extension of this inode.
93    extension: Extension,
94    /// The parent inode. `None` for root inode.
95    parent: Option<Arc<OverlayInode>>,
96    /// The mutable upper regular inode.
97    upper: Mutex<Option<Arc<dyn Inode>>>,
98    /// Whether the upper inode is an opaque directory.
99    upper_is_opaque: bool,
100    /// The immutable lower layered regular inodes.
101    lowers: Vec<Arc<dyn Inode>>,
102    /// Weak fs reference.
103    fs: Weak<OverlayFs>,
104    /// Weak self reference.
105    self_: Weak<OverlayInode>,
106}

108impl OverlayFs {
109    /// Creates a new OverlayFs instance.
110    ///
111    /// # Arguments
112    /// * `upper` - The upper directory (writable layer)
113    /// * `lower` - Vector of lower directories (read-only layers, in priority order)
114    /// * `work` - The work directory (must be empty and on same filesystem as upper)
115    ///
116    /// # Returns
117    /// An `Arc<OverlayFs>` on success, or an error if validation fails.
118    ///
119    /// # Errors
120    /// * `EINVAL` - If work and upper are on different filesystems
121    /// * `EINVAL` - If work is not empty
122    pub fn new(upper: Path, lower: Vec<Path>, work: Path) -> Result<Arc<Self>> {
123        Self::validate_work_and_upper(&work, &upper)?;
124        Self::validate_work_empty(&work)?;
125
126        Ok(Arc::new_cyclic(|weak| Self {
127            upper: OverlayUpper { path: upper },
128            lower: OverlayLower { paths: lower },
129            work: OverlayWork { path: work },
130            config: OverlayConfig::default(),
131            sb: OverlaySB,
132            next_ino: AtomicU64::new(0),
133            fs_event_subscriber_stats: FsEventSubscriberStats::new(),
134            self_: weak.clone(),
135        }))
136    }
137
138    /// Validates that work is on the same filesystem as upper.
139    fn validate_work_and_upper(work: &Path, upper: &Path) -> Result<()> {
140        if !Arc::ptr_eq(upper.mount_node(), work.mount_node()) {
141            return_errno_with_message!(
142                Errno::EINVAL,
143                "workdir and upperdir must reside under the same mount"
144            );
145        }
146        Ok(())
147    }
148
149    /// Validates that work is empty.
150    fn validate_work_empty(work: &Path) -> Result<()> {
151        let mut counter = DirentCounter::new();
152        let _ = work.inode().readdir_at(0, &mut counter);
153        if counter.count() > 0 {
154            return_errno_with_message!(Errno::EINVAL, "workdir must be empty");
155        }
156        Ok(())
157    }
158}

218    /// Lookups the target child `OverlayInode`. If the child is not present in cache,
219    /// it will be built from the layered lookups within the lower layers.
220    pub fn lookup(&self, name: &str) -> Result<Arc<dyn Inode>> {
221        match self.lookup_inner(name) {
222            Ok(Some(inode)) => Ok(inode),
223            Ok(None) => Err(Error::new(Errno::ENOENT)),
224            Err(e) => Err(e),
225        }
226    }
227
228    /// Creates a new non-exist child `OverlayInode` in the upper layer.
229    /// If the parent directories do not exist, they will be created recursively in the upper layer.
230    pub fn create(&self, name: &str, type_: InodeType, mode: InodeMode) -> Result<Arc<dyn Inode>> {
231        if self.type_ != InodeType::Dir {
232            return_errno!(Errno::ENOTDIR);
233        }
234
235        // TODO: Hold the upper lock from here to avoid race condition
236        let is_whiteout = match self.lookup_inner(name) {
237            Ok(Some(_)) => return_errno!(Errno::EEXIST),
238            Ok(None) => true,
239            Err(e) => {
240                if e.error() != Errno::ENOENT {
241                    return Err(e);
242                } else {
243                    false
244                }
245            }
246        };
247
248        if !is_whiteout {
249            self.build_upper_recursively_if_needed()?;
250        }
251
252        // Protect the whole create operation
253        let upper_guard = self.upper.lock();
254        let upper = upper_guard.as_ref().unwrap();
255
256        let mut upper_is_opaque = false;
257        if is_whiteout {
258            // Delete the whiteout file first then create the new file
259            // or the new opaque directory.
260            upper.unlink(&whiteout_name(name))?;
261
262            if type_ == InodeType::Dir {
263                upper_is_opaque = true;
264            }
265        }
266
267        let new_upper = upper.create(name, type_, mode)?;
268        if upper_is_opaque {
269            new_upper.set_xattr(
270                XattrName::try_from_full_name(OPAQUE_DIR_XATTR_NAME).unwrap(),
271                &mut VmReader::from(WHITEOUT_AND_OPAQUE_XATTR_VALUE.as_slice()).to_fallible(),
272                XattrSetFlags::CREATE_ONLY,
273            )?;
274        }
275
276        let new_child = Arc::new_cyclic(|weak| OverlayInode {
277            ino: new_upper.ino(),
278            type_,
279            name_upon_creation: SpinLock::new(String::from(name)),
280            extension: Extension::new(),
281            parent: Some(self.self_.upgrade().unwrap()),
282            upper: Mutex::new(Some(new_upper)),
283            upper_is_opaque,
284            lowers: Vec::new(),
285            fs: self.fs.clone(),
286            self_: weak.clone(),
287        });
288        Ok(new_child)
289    }

291    /// Writes data to the target inode, if it resides in the lower layer,

305    }

307    pub fn read_at(
308        &self,
309        offset: usize,
310        writer: &mut VmWriter,
311        status_flags: StatusFlags,
312    ) -> Result<usize> {
313        if self.type_ == InodeType::Dir {

320    /// Returns the children objects in a unified view.
321    /// The object from the upper layer with the same name will mask the lower ones.
322    pub fn readdir_at(&self, offset: usize, visitor: &mut dyn DirentVisitor) -> Result<usize> {
323        if self.type_ != InodeType::Dir {
324            return_errno!(Errno::ENOTDIR);
325        }
326
327        let overlay_dir_visitor = self.readdir_inner(offset)?;
328
329        for (offset, (name, ino, type_)) in overlay_dir_visitor.as_merged_view() {
330            visitor.visit(name, *ino, *type_, *offset)?;
331        }
332
333        Ok(overlay_dir_visitor.cur_offset())
334    }

336    /// Deletes the target file by creating a "whiteout" file from the upper layer.
337    /// The corresponding parent directories will be created also if they do not exist.
338    pub fn unlink(&self, name: &str) -> Result<()> {
339        // TODO: Hold the upper lock from here to avoid race condition
340        let inode = self.lookup(name)?;
341        let target = inode.downcast_ref::<OverlayInode>().unwrap();
342        if target.type_() == InodeType::Dir {
343            return_errno!(Errno::EISDIR);
344        }
345
346        let mut upper_guard = self.upper.lock();
347        if upper_guard.is_none() {
348            drop(upper_guard);
349            self.build_upper_recursively_if_needed()?;
350            upper_guard = self.upper.lock();
351        }
352
353        let upper = upper_guard.as_ref().unwrap();
354        let target_has_valid_lower = target.has_valid_lower();
355        if target.has_valid_upper() {
356            upper.unlink(name)?;
357        } else {
358            assert!(target_has_valid_lower);
359        }
360
361        if target_has_valid_lower {
362            let whiteout = upper.create(&whiteout_name(name), InodeType::File, mkmod!(a+r, u+w))?;
363            // FIXME: Align the whiteout xattr behavior with Linux
364            whiteout.set_xattr(
365                XattrName::try_from_full_name(WHITEOUT_XATTR_NAME).unwrap(),
366                &mut VmReader::from(WHITEOUT_AND_OPAQUE_XATTR_VALUE.as_slice()).to_fallible(),
367                XattrSetFlags::CREATE_ONLY,
368            )?;
369        }
370
371        Ok(())
372    }

374    /// Deletes the target directory by creating an "opaque" directory from the upper layer.
375    /// The corresponding parent directories will be created also if they do not exist.
376    pub fn rmdir(&self, name: &str) -> Result<()> {
377        // TODO: Hold the upper lock from here to avoid race condition
378        let inode = self.lookup(name)?;
379        let target = inode.downcast_ref::<OverlayInode>().unwrap();
380        if target.type_() != InodeType::Dir {
381            return_errno!(Errno::ENOTDIR);
382        }
383
384        self.build_upper_recursively_if_needed()?;
385        let upper_guard = self.upper.lock();
386        let upper = upper_guard.as_ref().unwrap();
387
388        let visitor = target.readdir_inner(0)?;
389        if visitor.visited_files() > 0 {
390            return_errno!(Errno::ENOTEMPTY);
391        }
392
393        // Delete all the whiteout files if necessary
394        if visitor.contains_whiteout() {
395            let target_upper = target.upper().unwrap();
396
397            let mut target_visitor = Vec::<String>::new();
398            target_upper.readdir_at(0, &mut target_visitor)?;
399
400            for whiteout in target_visitor.iter().skip(2) {
401                assert!(whiteout.starts_with(WHITEOUT_PREFIX));
402                target_upper.unlink(whiteout)?;
403            }
404        }
405
406        upper.rmdir(name)?;
407
408        let whiteout = upper.create(&whiteout_name(name), InodeType::File, mkmod!(a+r, u+w))?;
409        // FIXME: Align the whiteout xattr behavior with Linux
410        whiteout.set_xattr(
411            XattrName::try_from_full_name(WHITEOUT_XATTR_NAME).unwrap(),
412            &mut VmReader::from(WHITEOUT_AND_OPAQUE_XATTR_VALUE.as_slice()).to_fallible(),
413            XattrSetFlags::CREATE_ONLY,
414        )?;
415
416        Ok(())
417    }

423    pub fn resize(&self, new_size: usize) -> Result<()> {
424        if self.type_ == InodeType::Dir {
425            return_errno_with_message!(Errno::EISDIR, "the inode is a directory");
426        }
427        if self.type_ != InodeType::File {
428            return_errno_with_message!(Errno::EINVAL, "the inode is not a regular file");
429        }
430
431        if self.get_top_valid_inode().size() == new_size {
432            return Ok(());
433        }
434
435        let upper = self.build_upper_recursively_if_needed()?;
436        upper.resize(new_size)
437    }

551impl OverlayInode {
552    // Returns the top valid inode who must exist.
553    fn get_top_valid_inode(&self) -> Arc<dyn Inode> {
554        if let Some(upper) = self.upper() {
555            return upper;
556        }
557
558        self.get_top_valid_lower_inode().cloned().unwrap()
559    }
560
561    /// Returns the top valid lower inode.
562    fn get_top_valid_lower_inode(&self) -> Option<&Arc<dyn Inode>> {
563        if !self.has_valid_lower() {
564            return None;
565        }
566
567        // Note that the whiteout or opaque check is performed in `lookup` and `create`,
568        // the only two places where an `OverlayInode` can be created.
569        // So a lower inode can never be a whiteout file or opaque directory.
570        Some(&self.lowers[0])
571    }
572
573    fn has_valid_lower(&self) -> bool {
574        !self.lowers.is_empty()
575    }
576
577    fn has_valid_upper(&self) -> bool {
578        self.upper.lock().is_some()
579    }
580
581    /// Returns the upper inode if it exists.
582    fn upper(&self) -> Option<Arc<dyn Inode>> {
583        // Note that the whiteout or opaque check is performed in `lookup` and `create`,
584        // the only two places where an `OverlayInode` can be created.
585        // So the upper inode can never be a whiteout file or opaque directory.
586        self.upper.lock().clone()
587    }
588
589    fn num_lowers(&self) -> usize {
590        self.lowers.len()
591    }
592
593    fn is_opaque_dir(&self) -> bool {
594        self.type_ == InodeType::Dir && self.upper_is_opaque
595    }
596
597    fn name_upon_creation(&self) -> String {
598        self.name_upon_creation.lock().clone()
599    }
600
601    fn overlay_fs(&self) -> Arc<OverlayFs> {
602        self.fs.upgrade().unwrap()
603    }
604
605    /// Lookups the target regular inodes in a layered manner then
606    /// builds the corresponding `OverlayInode`.
607    /// The whiteout and opaque checks are performed here only.
608    fn lookup_inner(&self, name: &str) -> Result<Option<Arc<dyn Inode>>> {
609        if self.type_ != InodeType::Dir {
610            return_errno!(Errno::ENOTDIR);
611        }
612
613        let mut type_ = None;
614        let mut upper_is_opaque = false;
615        let mut upper_is_not_dir = false;
616
617        let upper_child = if let Some(upper) = self.upper.lock().as_ref() {
618            // First check whiteout then opaque
619            if upper.lookup(&whiteout_name(name)).is_ok() {
620                // Provide whiteout information for `create`
621                return Ok(None);
622            }
623
624            match upper.lookup(name) {
625                Ok(child) => {
626                    let child_type = child.type_();
627                    if child_type == InodeType::Dir {
628                        upper_is_opaque = is_opaque_dir(&child)?;
629                    } else {
630                        upper_is_not_dir = true;
631                    }
632
633                    let _ = type_.insert(child_type);
634                    Some(child)
635                }
636                Err(e) => {
637                    if e.error() == Errno::ENOENT {
638                        None
639                    } else {
640                        return Err(e);
641                    }
642                }
643            }
644        } else {
645            None
646        };
647
648        let lower_children = if upper_is_opaque || upper_is_not_dir {
649            vec![]
650        } else {
651            let mut children = Vec::new();
652            for lower in &self.lowers {
653                if lower.lookup(&whiteout_name(name)).is_ok() {
654                    break;
655                }
656
657                if let Ok(child) = lower.lookup(name) {
658                    let child_type = child.type_();
659                    let is_child_opaque = child_type == InodeType::Dir && is_opaque_dir(&child)?;
660
661                    if upper_child.is_none() && children.is_empty() {
662                        children.push(child);
663                        let _ = type_.insert(child_type);
664                    } else {
665                        let type_ = type_.unwrap();
666                        if type_ != InodeType::Dir || type_ != child_type {
667                            break;
668                        } else {
669                            children.push(child);
670                        }
671                    }
672
673                    if is_child_opaque {
674                        break;
675                    }
676                }
677            }
678            children
679        };
680
681        if upper_child.is_none() && lower_children.is_empty() {
682            return_errno!(Errno::ENOENT);
683        }
684
685        let ino = if let Some(upper) = &upper_child {
686            UniqueNoGenerator::gen_unique_ino(0 as LayerIdx, upper.ino())?
687        } else {
688            UniqueNoGenerator::gen_unique_ino(1 as LayerIdx, lower_children[0].ino())?
689        };
690        let child_ovl_inode = Arc::new_cyclic(|weak| OverlayInode {
691            ino,
692            type_: type_.unwrap(),
693            name_upon_creation: SpinLock::new(String::from(name)),
694            extension: Extension::new(),
695            parent: Some(self.self_.upgrade().unwrap()),
696            upper: Mutex::new(upper_child),
697            upper_is_opaque,
698            lowers: lower_children,
699            fs: self.fs.clone(),
700            self_: weak.clone(),
701        });
702
703        Ok(Some(child_ovl_inode))
704    }
705
706    fn readdir_inner(&self, offset: usize) -> Result<OverlayDirVisitor> {
707        let mut overlay_visitor = OverlayDirVisitor::new();
708        let (mut layer_idx, fs_offset) = UniqueNoGenerator::parse_unique_offset(offset);
709
710        // Process all the potential whiteout files before `layer_idx`
711        if layer_idx > 0 {
712            overlay_visitor.set_whiteout_only_mode();
713
714            let upper = self.upper();
715            for idx in 0..=layer_idx {
716                let cur_inode = if idx == 0 {
717                    upper.as_ref()
718                } else {
719                    self.lowers.get(idx as usize - 1)
720                };
721
722                if let Some(cur_inode) = cur_inode {
723                    cur_inode.readdir_at(0, &mut overlay_visitor)?;
724                }
725            }
726
727            overlay_visitor.unset_whiteout_only_mode();
728        }
729
730        // Process all files from `layer_idx` and `fs_offset`
731        if layer_idx == 0 {
732            if let Some(upper) = self.upper() {
733                debug_assert!(upper.type_() == InodeType::Dir);
734                upper.readdir_at(fs_offset, &mut overlay_visitor)?;
735            }
736
737            layer_idx += 1;
738            overlay_visitor.set_cur_layer(layer_idx);
739        }
740
741        if !self.is_opaque_dir() && layer_idx > 0 && layer_idx as usize <= self.lowers.len() {
742            // TODO: Figure out how to check the opaque directories within lower layers.
743            let first_lower = &self.lowers[layer_idx as usize - 1];
744            first_lower.readdir_at(fs_offset, &mut overlay_visitor)?;
745
746            layer_idx += 1;
747            overlay_visitor.set_cur_layer(layer_idx);
748
749            for lower in self.lowers.iter().skip(layer_idx as usize - 1) {
750                lower.readdir_at(0, &mut overlay_visitor)?;
751
752                layer_idx += 1;
753                overlay_visitor.set_cur_layer(layer_idx);
754            }
755        }
756
757        Ok(overlay_visitor)
758    }
759
760    fn build_upper_recursively_if_needed(&self) -> Result<Arc<dyn Inode>> {
761        let mut upper_guard = self.upper.lock();
762        if let Some(upper) = upper_guard.as_ref() {
763            return Ok(upper.clone());
764        }
765
766        debug_assert!(self.parent.is_some());
767        // FIXME: Should we hold every upper locks from lower to upper
768        // for such a long period?
769        let parent_upper = self
770            .parent
771            .as_ref()
772            .unwrap()
773            .build_upper_recursively_if_needed()?;
774
775        let mode = self.get_top_valid_lower_inode().unwrap().mode()?;
776        let new_upper = parent_upper.create(&self.name_upon_creation(), self.type_, mode)?;
777
778        // There must exist a valid lower inode if the upper is missing
779        assert!(!self.lowers.is_empty());
780        self.do_copy_up(&new_upper)?;
781
782        let _ = upper_guard.insert(new_upper.clone());
783        Ok(new_upper)
784    }
785
786    /// Do the "copy-up" operation for the given upper inode.
787    fn do_copy_up(&self, upper_inode: &Arc<dyn Inode>) -> Result<()> {
788        if self.lowers.is_empty() {
789            return Ok(());
790        }
791        let Some(lower_inode) = self.get_top_valid_lower_inode() else {
792            return Ok(());
793        };
794
795        let upper_type = upper_inode.type_();
796        if upper_type != lower_inode.type_() {
797            return Ok(());
798        }
799
800        // First copy the metadata, then the data, finally the xattr
801        Self::copy_up_metadata(lower_inode, upper_inode)?;
802
803        if upper_type == InodeType::File {
804            Self::copy_up_data(lower_inode, upper_inode)?;
805        }
806
807        Self::copy_up_xattr(lower_inode, upper_inode)?;
808        Ok(())
809    }
810
811    fn copy_up_metadata(lower: &Arc<dyn Inode>, upper: &Arc<dyn Inode>) -> Result<()> {
812        // TODO: We lack an efficient whole metadata copy API.
813
814        // The mode is copied up upon creation.
815        upper.set_owner(lower.owner()?)?;
816        upper.set_group(lower.group()?)?;
817        upper.set_atime(lower.atime());
818        upper.set_mtime(lower.mtime());
819        upper.set_ctime(lower.ctime());
820        Ok(())
821    }
822
823    fn copy_up_data(lower: &Arc<dyn Inode>, upper: &Arc<dyn Inode>) -> Result<()> {
824        debug_assert!(lower.type_() == InodeType::File && upper.type_() == InodeType::File);
825        if lower.size() == 0 {
826            return Ok(());
827        }
828
829        // TODO: Find a way to cut this copy, like just copy chunks of data from two page caches directly.
830        let lower_size = lower.size();
831        let data_buf = FrameAllocOptions::new()
832            .zeroed(false)
833            .alloc_segment(lower_size.align_up(BLOCK_SIZE) / BLOCK_SIZE)?;
834
835        let mut writer = data_buf.writer().to_fallible();
836        let read_len = lower.read_at(0, &mut writer, StatusFlags::empty())?;
837
838        let mut reader = data_buf.reader().to_fallible();
839        let _ = upper.write_at(0, reader.limit(read_len), StatusFlags::empty())?;
840        Ok(())
841    }
842
843    fn copy_up_xattr(lower: &Arc<dyn Inode>, upper: &Arc<dyn Inode>) -> Result<()> {
844        debug_assert!(lower.type_() == upper.type_());
845
846        let list_len = lower.list_xattr(
847            XattrNamespace::Trusted,
848            &mut VmWriter::from([].as_mut_slice()).to_fallible(),
849        )?;
850        if list_len == 0 {
851            return Ok(());
852        }
853        let mut list = vec![0u8; list_len];
854        lower.list_xattr(
855            XattrNamespace::Trusted,
856            &mut VmWriter::from(list.as_mut_slice()).to_fallible(),
857        )?;
858
859        let value_buf = FrameAllocOptions::new()
860            .zeroed(false)
861            .alloc_segment(XATTR_VALUE_MAX_LEN / PAGE_SIZE)?;
862        for name in list
863            .split(|&byte| byte == 0)
864            .map(|slice| String::from_utf8_lossy(slice))
865        {
866            if name.is_empty() {
867                break;
868            }
869            let value_len = lower.get_xattr(
870                XattrName::try_from_full_name(name.as_ref()).unwrap(),
871                &mut value_buf.writer().to_fallible(),
872            )?;
873            let mut value_reader = value_buf.reader().to_fallible();
874            upper.set_xattr(
875                XattrName::try_from_full_name(name.as_ref()).unwrap(),
876                value_reader.limit(value_len),
877                XattrSetFlags::CREATE_ONLY,
878            )?;
879        }
880        Ok(())
881    }
882}

895fn is_opaque_dir(inode: &Arc<dyn Inode>) -> Result<bool> {
896    assert_eq!(inode.type_(), InodeType::Dir);
897
898    let name = XattrName::try_from_full_name(OPAQUE_DIR_XATTR_NAME).unwrap();
899    let mut value = [0u8];
900    if let Err(e) = inode.get_xattr(
901        name,
902        &mut VmWriter::from(value.as_mut_slice()).to_fallible(),
903    ) {
904        match e.error() {
905            Errno::E2BIG | Errno::ENODATA | Errno::EOPNOTSUPP | Errno::ERANGE => {
906                return Ok(false);
907            }
908            _ => return Err(e),
909        }
910    };
911    Ok(value == WHITEOUT_AND_OPAQUE_XATTR_VALUE)
912}

981/// The index of the layer of an `OverlayFs`.
982type LayerIdx = u8; // Currently only support 256 layers.
983
984/// A visitor used by `OverlayFs` that merges the objects
985/// from the upper layer and the lower layer.
986struct OverlayDirVisitor {
987    dir_map: BTreeMap<usize, (String, u64, InodeType)>,
988    dir_set: HashSet<String>,
989    whiteout_set: HashSet<String>,
990    visited_files: usize,
991    cur_layer: LayerIdx,
992    whiteout_only_mode: bool,
993}
994
995impl DirentVisitor for OverlayDirVisitor {
996    fn visit(&mut self, name: &str, fs_ino: u64, type_: InodeType, fs_offset: usize) -> Result<()> {
997        if self.whiteout_only_mode && name.starts_with(WHITEOUT_PREFIX) {
998            self.dir_set
999                .insert(name[WHITEOUT_PREFIX_SIZE..].to_string());
1000            return Ok(());
1001        }
1002
1003        let unique_offset = UniqueNoGenerator::gen_unique_offset(self.cur_layer, fs_offset)?;
1004        let unique_ino = UniqueNoGenerator::gen_unique_ino(self.cur_layer, fs_ino)?;
1005
1006        if self.dir_set.contains(name) || self.whiteout_set.contains(name) {
1007            return Ok(());
1008        }
1009        if name.starts_with(WHITEOUT_PREFIX) {
1010            self.whiteout_set
1011                .insert(name[WHITEOUT_PREFIX_SIZE..].to_string());
1012            return Ok(());
1013        }
1014
1015        let name = name.to_string();
1016        self.dir_set.insert(name.clone());
1017
1018        if name != "." && name != ".." {
1019            self.visited_files += 1;
1020        }
1021
1022        debug_assert!(!self.dir_map.contains_key(&unique_offset));
1023        let _ = self
1024            .dir_map
1025            .insert(unique_offset, (name, unique_ino, type_));
1026        Ok(())
1027    }
1028}
1029

1031    pub fn new() -> Self {
1032        Self {
1033            dir_map: BTreeMap::new(),
1034            dir_set: HashSet::new(),

1042    /// Returns the merged view of the directory.

1045    }

1049    }

1053    }

1060    }

1064    }

1072    }

1075struct UniqueNoGenerator;
1076
1077// Unique offset and ino layout: `| LayerIdx (8 bits) | Real fs offset or ino (56 bits) |`
1078impl UniqueNoGenerator {
1079    const NUM_HIGHER_BITS: usize = 8;
1080    const NUM_LOWER_BITS: usize = 56;
1081    const HIGHER_MASK: usize = 0xFF00_0000_0000_0000;
1082    const LOWER_MASK: usize = 0x00FF_FFFF_FFFF_FFFF;
1083
1084    pub fn gen_unique_offset(layer_idx: LayerIdx, fs_offset: usize) -> Result<usize> {
1085        if fs_offset & Self::HIGHER_MASK != 0 {
1086            return_errno_with_message!(Errno::EOVERFLOW, "fs offset overflow");
1087        }
1088        Ok(((layer_idx as usize) << Self::NUM_LOWER_BITS) | fs_offset)
1089    }
1090
1091    // XXX: Linux uses embedded fsid for unique ino. Should we follow the technique?
1092    pub fn gen_unique_ino(layer_idx: LayerIdx, fs_ino: u64) -> Result<u64> {
1093        if (fs_ino as usize) & Self::HIGHER_MASK != 0 {
1094            return_errno_with_message!(Errno::EOVERFLOW, "fs ino overflow");
1095        }
1096        Ok(((layer_idx as u64) << Self::NUM_LOWER_BITS) | fs_ino)
1097    }
1098
1099    pub fn parse_unique_offset(offset: usize) -> (LayerIdx, usize) {
1100        let layer = (offset >> Self::NUM_LOWER_BITS) as LayerIdx;
1101        let offset = offset & Self::LOWER_MASK;
1102        (layer, offset)
1103    }
1104}

1174                }

1176            }

1187            .map(|&lower| path_resolver.lookup(&FsPath::try_from(lower)?))
1188            .collect::<Result<Vec<_>>>()?;
1189        let work = path_resolver.lookup(&FsPath::try_from(work)?)?;
1190

1192    }
