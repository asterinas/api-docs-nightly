6#[repr(C, packed)]
7#[derive(Clone, Copy, Debug, Default)]
8// The in-memory superblock info
9pub struct ExfatSuperBlock {
10    /// num of sectors in volume
11    pub num_sectors: u64,
12    /// num of clusters in volume
13    pub num_clusters: u32,
14    /// sector size in bytes
15    pub sector_size: u32,
16    /// cluster size in bytes
17    pub cluster_size: u32,
18    pub cluster_size_bits: u32,
19    /// cluster size in sectors
20    pub sect_per_cluster: u32,
21    pub sect_per_cluster_bits: u32,
22    /// FAT1 start sector
23    pub fat1_start_sector: u64,
24    /// FAT2 start sector
25    pub fat2_start_sector: u64,
26    /// data area start sector
27    pub data_start_sector: u64,
28    /// number of FAT sectors
29    pub num_fat_sectors: u32,
30    /// root dir cluster
31    pub root_dir: u32,
32    /// number of dentries per cluster
33    pub dentries_per_clu: u32,
34    /// volume flags
35    pub vol_flags: u32,
36    /// volume flags to retain
37    pub vol_flags_persistent: u32,
38    /// cluster search pointer
39    pub cluster_search_ptr: u32,
40    /// number of used clusters
41    pub used_clusters: u32,
42}

44const DENTRY_SIZE_BITS: u32 = 5;
45
46impl TryFrom<ExfatBootSector> for ExfatSuperBlock {
47    type Error = crate::error::Error;
48    fn try_from(sector: ExfatBootSector) -> Result<ExfatSuperBlock> {
49        const EXFAT_CLUSTERS_UNTRACKED: u32 = !0;
50        let mut block = ExfatSuperBlock {
51            sect_per_cluster_bits: sector.sector_per_cluster_bits as u32,
52            sect_per_cluster: 1 << sector.sector_per_cluster_bits as u32,
53
54            cluster_size_bits: (sector.sector_per_cluster_bits + sector.sector_size_bits) as u32,
55            cluster_size: 1 << (sector.sector_per_cluster_bits + sector.sector_size_bits) as u32,
56
57            sector_size: 1 << sector.sector_size_bits,
58            num_fat_sectors: sector.fat_length,
59            fat1_start_sector: sector.fat_offset as u64,
60            fat2_start_sector: sector.fat_offset as u64,
61
62            data_start_sector: sector.cluster_offset as u64,
63            num_sectors: sector.vol_length,
64            num_clusters: sector.cluster_count + EXFAT_RESERVED_CLUSTERS,
65
66            root_dir: sector.root_cluster,
67
68            vol_flags: sector.vol_flags as u32,
69            vol_flags_persistent: (sector.vol_flags & (VOLUME_DIRTY | MEDIA_FAILURE)) as u32,
70
71            cluster_search_ptr: EXFAT_FIRST_CLUSTER,
72
73            used_clusters: EXFAT_CLUSTERS_UNTRACKED,
74
75            dentries_per_clu: 1
76                << ((sector.sector_per_cluster_bits + sector.sector_size_bits) as u32
77                    - DENTRY_SIZE_BITS),
78        };
79
80        if block.num_fat_sectors == 2 {
81            block.fat2_start_sector += block.num_fat_sectors as u64;
82        }
83
84        Ok(block)
85    }
86}
