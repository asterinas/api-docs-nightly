24#[derive(Debug, Default)]
25pub(super) struct ExfatBitmap {
26    // Start cluster of allocation bitmap.
27    chain: ExfatChain,
28    bitvec: BitVec<BitStore>,
29    dirty_bytes: VecDeque<Range<usize>>,
30
31    // Used to track the number of free clusters.
32    num_free_cluster: u32,

34}

36impl ExfatBitmap {
37    pub(super) fn load(
38        fs_weak: Weak<ExfatFs>,
39        root_page_cache: &Vmo,
40        root_chain: ExfatChain,
41    ) -> Result<Self> {
42        let dentry_iterator = ExfatDentryIterator::new(root_page_cache, 0, None)?;
43
44        for dentry_result in dentry_iterator {
45            let dentry = dentry_result?;
46            if let ExfatDentry::Bitmap(bitmap_dentry) = dentry {
47                // If the last bit of bitmap is 0, it is a valid bitmap.
48                if (bitmap_dentry.flags & 0x1) == 0 {
49                    return Self::load_bitmap_from_dentry(fs_weak.clone(), &bitmap_dentry);
50                }
51            }
52        }
53
54        return_errno_with_message!(Errno::EINVAL, "bitmap not found")
55    }
56
57    fn load_bitmap_from_dentry(fs_weak: Weak<ExfatFs>, dentry: &ExfatBitmapDentry) -> Result<Self> {
58        let fs = fs_weak.upgrade().unwrap();
59        let num_clusters = (dentry.size as usize).align_up(fs.cluster_size()) / fs.cluster_size();
60
61        let chain = ExfatChain::new(
62            fs_weak.clone(),
63            dentry.start_cluster,
64            Some(num_clusters as u32),
65            FatChainFlags::ALLOC_POSSIBLE,
66        )?;
67        let mut buf = vec![0; dentry.size as usize];
68
69        fs.read_meta_at(chain.physical_cluster_start_offset(), &mut buf)?;
70        let mut free_cluster_num = 0;
71        for idx in 0..fs.super_block().num_clusters - EXFAT_RESERVED_CLUSTERS {
72            if (buf[idx as usize / BITS_PER_BYTE] & (1 << (idx % BITS_PER_BYTE as u32))) == 0 {
73                free_cluster_num += 1;
74            }
75        }
76        Ok(ExfatBitmap {
77            chain,
78            bitvec: BitVec::from_slice(&buf),
79            dirty_bytes: VecDeque::new(),
80            num_free_cluster: free_cluster_num,
81            fs: fs_weak,
82        })
83    }
84
85    fn fs(&self) -> Arc<ExfatFs> {
86        self.fs.upgrade().unwrap()
87    }

117    pub(super) fn is_cluster_range_unused(&self, clusters: Range<ClusterID>) -> Result<bool> {
118        if !self.fs().is_cluster_range_valid(clusters.clone()) {
119            return_errno_with_message!(Errno::EINVAL, "invalid cluster ranges.")
120        }
121
122        for id in clusters {
123            if self.bitvec[(id - EXFAT_RESERVED_CLUSTERS) as usize] {
124                return Ok(false);
125            }
126        }
127        Ok(true)
128    }

130    /// Return the first unused cluster.
131    pub(super) fn find_next_unused_cluster(&self, cluster: ClusterID) -> Result<ClusterID> {
132        let clusters = self.find_next_unused_cluster_range_by_bits(cluster, 1)?;
133        Ok(clusters.start)
134    }

136    /// Return the first unused cluster range, set num_clusters=1 to find a single cluster.
137    fn find_next_unused_cluster_range_by_bits(
138        &self,
139        search_start_cluster: ClusterID,
140        num_clusters: u32,
141    ) -> Result<Range<ClusterID>> {
142        if !self
143            .fs()
144            .is_cluster_range_valid(search_start_cluster..search_start_cluster + num_clusters)
145        {
146            return_errno_with_message!(Errno::ENOSPC, "free contiguous clusters not available.")
147        }
148
149        let mut cur_index = search_start_cluster - EXFAT_RESERVED_CLUSTERS;
150        let end_index = self.fs().super_block().num_clusters - EXFAT_RESERVED_CLUSTERS;
151        let search_end_index = end_index - num_clusters + 1;
152        let mut range_start_index: ClusterID;
153
154        while cur_index < search_end_index {
155            if !self.is_used(cur_index as usize) {
156                range_start_index = cur_index;
157                let mut cnt = 0;
158                while cnt < num_clusters
159                    && cur_index < end_index
160                    && !self.is_used(cur_index as usize)
161                {
162                    cnt += 1;
163                    cur_index += 1;
164                }
165                if cnt >= num_clusters {
166                    return Ok(range_start_index + EXFAT_RESERVED_CLUSTERS
167                        ..range_start_index + EXFAT_RESERVED_CLUSTERS + num_clusters);
168                }
169            }
170            cur_index += 1;
171        }
172        return_errno!(Errno::ENOSPC)
173    }

175    /// Make sure the bit at the range start position is 0.
176    fn adjust_head_pos(
177        &self,
178        bytes: &[BitStore],
179        mut cur_unit_index: u32,
180        mut cur_unit_offset: u32,
181        total_cluster_num: u32,
182    ) -> (u32, u32) {
183        let unit_size: u32 = (BITS_PER_BYTE * size_of::<BitStore>()) as u32;
184        while cur_unit_index < total_cluster_num {
185            let leading_zeros = bytes[cur_unit_index as usize].leading_zeros();
186            let head_cluster_num = unit_size - cur_unit_offset;
187            if leading_zeros == 0 {
188                // Fall over to the next unit, we need to continue checking.
189                cur_unit_index += 1;
190                cur_unit_offset = 0;
191            } else {
192                // Stop at current unit, we may need to adjust the cur_offset
193                cur_unit_offset = cur_unit_offset.max(unit_size - leading_zeros);
194                break;
195            }
196        }
197        (cur_unit_index, cur_unit_offset)
198    }

200    /// Check if the next mid_unit_num units are zero.
201    /// If not, return the index of the first not zero unit.
202    fn check_mid_units(&self, bytes: &[BitStore], cur_unit_index: u32, mid_unit_num: u32) -> u32 {
203        for i in 1..mid_unit_num + 1 {
204            if bytes[(cur_unit_index + i) as usize] != 0 {
205                return cur_unit_index + 1;
206            }
207        }
208        cur_unit_index
209    }
210
211    /// Check if the tail unit is valid.
212    /// Currently not used.
213    fn check_tail_bits(
214        &self,
215        bytes: &[BitStore],
216        tail_idx: u32,
217        tail_cluster_num: u32,
218        complete_unit_num: u32,
219        rest_cluster_num: u32,
220    ) -> bool {
221        let valid_bytes_num = if rest_cluster_num > 0 {
222            complete_unit_num + 1
223        } else {
224            complete_unit_num
225        };
226        let mut tail_byte: u8 = 0;
227        if tail_idx == complete_unit_num {
228            tail_byte |= 0xFF_u8 - ((1_u8 << rest_cluster_num) - 1);
229        }
230        if tail_idx < valid_bytes_num {
231            tail_byte |= bytes[tail_idx as usize];
232        }
233        let tailing_zeros = tail_byte.trailing_zeros();
234        tailing_zeros >= tail_cluster_num
235    }

237    fn make_range(
238        &self,
239        cur_unit_index: u32,
240        cur_unit_offset: u32,
241        num_clusters: u32,
242    ) -> Range<ClusterID> {
243        let unit_size: u32 = (BITS_PER_BYTE * size_of::<BitStore>()) as u32;
244        let result_bit_index = cur_unit_index * unit_size + cur_unit_offset;
245        result_bit_index + EXFAT_RESERVED_CLUSTERS
246            ..result_bit_index + EXFAT_RESERVED_CLUSTERS + num_clusters
247    }

249    /// Return the next contiguous unused clusters, set cluster_num=1 to find a single cluster
250    pub(super) fn find_next_unused_cluster_range(
251        &self,
252        search_start_cluster: ClusterID,
253        num_clusters: u32,
254    ) -> Result<Range<ClusterID>> {
255        if !self
256            .fs()
257            .is_cluster_range_valid(search_start_cluster..search_start_cluster + num_clusters)
258        {
259            return_errno!(Errno::ENOSPC)
260        }
261
262        let bytes: &[BitStore] = self.bitvec.as_raw_slice();
263        let unit_size: u32 = (BITS_PER_BYTE * size_of::<BitStore>()) as u32;
264        let start_cluster_index = search_start_cluster - EXFAT_RESERVED_CLUSTERS;
265        let mut cur_unit_index = start_cluster_index / unit_size;
266        let mut cur_unit_offset = start_cluster_index % unit_size;
267        let total_cluster_num = self.fs().super_block().num_clusters - EXFAT_RESERVED_CLUSTERS;
268        let complete_unit_num = total_cluster_num / unit_size;
269        let rest_cluster_num = total_cluster_num % unit_size;
270        let valid_bytes_num = if rest_cluster_num > 0 {
271            complete_unit_num + 1
272        } else {
273            complete_unit_num
274        };
275        if num_clusters <= unit_size {
276            // If this case, back to the simple function
277            return self.find_next_unused_cluster_range_by_bits(search_start_cluster, num_clusters);
278        }
279        // Treat a continuous bit chunk as lead_bits+mid_units+tail_bits (mid_units are unit aligned)
280        // For example: 11110000 00000000 00000000 00111111
281        //                  **** -------- -------- ..
282        //                  ^(start bit)
283        // (*): head_bits;  (-): mid_units;  (.): tail_bits
284        // The start bit can be identified with a pair (cur_unit_index, cur_unit_offset)
285        while cur_unit_index < complete_unit_num {
286            // First, adjust the cur_idx to a proper head.
287            (cur_unit_index, cur_unit_offset) =
288                self.adjust_head_pos(bytes, cur_unit_index, cur_unit_offset, total_cluster_num);
289            let head_cluster_num = unit_size - cur_unit_offset;
290            let mid_unit_num = (num_clusters - head_cluster_num) / unit_size;
291            let tail_cluster_num = (num_clusters - head_cluster_num) % unit_size;
292            // If the last complete unit to be check is out of range, stop searching
293            if cur_unit_index + mid_unit_num >= complete_unit_num {
294                break;
295            }
296            // Then check for the mid units, these units should be all zero
297            // Due to previous check, there will be no array out of bounds situation
298            let ret = self.check_mid_units(bytes, cur_unit_index, mid_unit_num);
299            if ret != cur_unit_index {
300                // Mid_checks failed, should go back to the first step.
301                cur_unit_index = ret;
302                cur_unit_offset = 0;
303                continue;
304            }
305            // At last, check for the tail bits
306            if tail_cluster_num == 0 {
307                return Ok(self.make_range(cur_unit_index, cur_unit_offset, num_clusters));
308            }
309            let mut tail_byte: u8 = 0;
310            let tail_idx = cur_unit_index + mid_unit_num + 1;
311            if tail_idx == complete_unit_num {
312                tail_byte |= 0xFF_u8 - ((1_u8 << rest_cluster_num) - 1);
313            }
314            if tail_idx < valid_bytes_num {
315                tail_byte |= bytes[tail_idx as usize];
316            }
317            let tailing_zeros = tail_byte.trailing_zeros();
318            if tail_cluster_num > tailing_zeros {
319                cur_unit_index = tail_idx;
320                cur_unit_offset = tailing_zeros + 1;
321                continue;
322            }
323            // If we reach here, it means we have found a result
324            return Ok(self.make_range(cur_unit_index, cur_unit_offset, num_clusters));
325        }
326        return_errno!(Errno::ENOSPC)
327    }

329    pub(super) fn num_free_clusters(&self) -> u32 {

333    fn set_range(&mut self, clusters: Range<ClusterID>, bit: bool, sync: bool) -> Result<()> {
334        if !self.fs().is_cluster_range_valid(clusters.clone()) {
335            return_errno_with_message!(Errno::EINVAL, "invalid cluster ranges.")
336        }
337
338        for cluster_id in clusters.clone() {
339            let index = (cluster_id - EXFAT_RESERVED_CLUSTERS) as usize;
340            let old_bit = self.is_used(index);
341            self.bitvec.set(index, bit);
342
343            if !old_bit && bit {
344                self.num_free_cluster -= 1;
345            } else if old_bit && !bit {
346                self.num_free_cluster += 1;
347            }
348        }
349
350        self.write_to_disk(clusters.clone(), sync)?;
351
352        Ok(())
353    }
354
355    fn write_to_disk(&mut self, clusters: Range<ClusterID>, sync: bool) -> Result<()> {
356        let unit_size = size_of::<BitStore>() * BITS_PER_BYTE;
357        let start_byte_off: usize = (clusters.start - EXFAT_RESERVED_CLUSTERS) as usize / unit_size;
358        let end_byte_off: usize =
359            ((clusters.end - EXFAT_RESERVED_CLUSTERS) as usize).align_up(unit_size) / unit_size;
360
361        let bytes: &[BitStore] = self.bitvec.as_raw_slice();
362        let byte_chunk = &bytes[start_byte_off..end_byte_off];
363
364        let pos = self.chain.walk_to_cluster_at_offset(start_byte_off)?;
365
366        let phys_offset = pos.0.physical_cluster_start_offset() + pos.1;
367        self.fs().write_meta_at(phys_offset, byte_chunk)?;
368
369        let byte_range = phys_offset..phys_offset + byte_chunk.len();
370
371        if sync {
372            self.fs().sync_meta_at(byte_range.clone())?;
373        } else {
374            self.dirty_bytes.push_back(byte_range.clone());
375        }
376
377        Ok(())
378    }
379
380    pub(super) fn sync(&mut self) -> Result<()> {
381        while let Some(range) = self.dirty_bytes.pop_front() {
382            self.fs().sync_meta_at(range)?;
383        }
384        Ok(())
385    }
