45    pub struct FatAttr : u16{
46        /// This inode is read only.
47        const READONLY  = 0x0001;
48        /// This inode is hidden. This attribute is not supported in our implementation.
49        const HIDDEN    = 0x0002;
50        /// This inode belongs to the OS. This attribute is not supported in our implementation.
51        const SYSTEM    = 0x0004;
52        /// This inode represents a volume. This attribute is not supported in our implementation.
53        const VOLUME    = 0x0008;
54        /// This inode represents a directory.
55        const DIRECTORY = 0x0010;
56        /// This file has been touched since the last DOS backup was performed on it. This attribute is not supported in our implementation.
57        const ARCHIVE   = 0x0020;
58    }

61impl FatAttr {
62    /// Convert attribute bits and a mask to the UNIX mode.
63    fn make_mode(&self, mount_option: ExfatMountOptions, mode: InodeMode) -> InodeMode {
64        let mut ret = mode;
65        if self.contains(FatAttr::READONLY) && !self.contains(FatAttr::DIRECTORY) {
66            ret.remove(mkmod!(a+w));
67        }
68        if self.contains(FatAttr::DIRECTORY) {
69            ret.remove(InodeMode::from_bits_truncate(mount_option.fs_dmask));
70        } else {
71            ret.remove(InodeMode::from_bits_truncate(mount_option.fs_fmask));
72        }
73        ret
74    }

243    fn count_num_sub_inode_and_dir(&self, fs_guard: &MutexGuard<()>) -> Result<(usize, usize)> {
244        if !self.start_chain.is_current_cluster_valid() {

249        let mut sub_inodes = 0;

251        for dentry_result in iterator {

253            if let ExfatDentry::File(file) = dentry {
254                sub_inodes += 1;
255                if FatAttr::from_bits_truncate(file.attribute).contains(FatAttr::DIRECTORY) {
256                    sub_dirs += 1;

259        }

263    /// Resize current inode to new_size.
264    /// The `size_allocated` field in inode can be enlarged, while the `size` field will not.
265    fn resize(&mut self, new_size: usize, fs_guard: &MutexGuard<()>) -> Result<()> {
266        let fs = self.fs();
267        let cluster_size = fs.cluster_size();
268
269        let num_clusters = self.num_clusters();
270        let new_num_clusters = (new_size.align_up(cluster_size) / cluster_size) as u32;
271
272        let sync = self.is_sync();
273
274        match new_num_clusters.cmp(&num_clusters) {
275            Ordering::Greater => {
276                // New clusters should be allocated.
277                self.start_chain
278                    .extend_clusters(new_num_clusters - num_clusters, sync)?;
279            }
280            Ordering::Less => {
281                // Some exist clusters should be truncated.
282                self.start_chain
283                    .remove_clusters_from_tail(num_clusters - new_num_clusters, sync)?;
284                if new_size < self.size {
285                    // Valid data is truncated.
286                    self.size = new_size;
287                }
288            }
289            _ => {}
290        };
291        self.size_allocated = new_size;
292
293        Ok(())
294    }

296    /// Update inode information back to the disk to sync this inode.
297    /// Should lock the file system before calling this function.
298    fn write_inode(&self, sync: bool, fs_guard: &MutexGuard<()>) -> Result<()> {
299        // Root dir should not be updated.
300        if self.ino == EXFAT_ROOT_INO {
301            return Ok(());
302        }
303
304        // If the inode or its parent is already unlinked, there is no need for updating it.
305        if self.is_deleted || !self.dentry_set_position.0.is_current_cluster_valid() {
306            return Ok(());
307        }
308
309        let parent = self.get_parent_inode().unwrap_or_else(|| unimplemented!());
310        let page_cache = parent.page_cache().unwrap();
311
312        // Need to read the latest dentry set from parent inode.
313
314        let mut dentry_set =
315            ExfatDentrySet::read_from(&page_cache, self.dentry_entry as usize * DENTRY_SIZE)?;
316
317        let mut file_dentry = dentry_set.get_file_dentry();
318        let mut stream_dentry = dentry_set.get_stream_dentry();
319
320        file_dentry.attribute = self.attr.bits();
321
322        file_dentry.create_utc_offset = self.ctime.utc_offset;
323        file_dentry.create_date = self.ctime.date;
324        file_dentry.create_time = self.ctime.time;
325        file_dentry.create_time_cs = self.ctime.increment_10ms;
326
327        file_dentry.modify_utc_offset = self.mtime.utc_offset;
328        file_dentry.modify_date = self.mtime.date;
329        file_dentry.modify_time = self.mtime.time;
330        file_dentry.modify_time_cs = self.mtime.increment_10ms;
331
332        file_dentry.access_utc_offset = self.atime.utc_offset;
333        file_dentry.access_date = self.atime.date;
334        file_dentry.access_time = self.atime.time;
335
336        stream_dentry.valid_size = self.size as u64;
337        stream_dentry.size = self.size_allocated as u64;
338        stream_dentry.start_cluster = self.start_chain.cluster_id();
339        stream_dentry.flags = self.start_chain.flags().bits();
340
341        dentry_set.set_file_dentry(&file_dentry);
342        dentry_set.set_stream_dentry(&stream_dentry);
343        dentry_set.update_checksum();
344
345        //Update the page cache of parent inode.
346        let start_off = self.dentry_entry as usize * DENTRY_SIZE;
347        let bytes = dentry_set.to_le_bytes();
348
349        page_cache.write_bytes(start_off, &bytes)?;
350        if sync {
351            page_cache.decommit(start_off..start_off + bytes.len())?;
352        }
353
354        Ok(())
355    }

357    /// Read all sub-inodes from the given position(offset) in this directory.
358    /// The number of inodes to read is given by dir_cnt.
359    /// Return (the new offset after read, the number of sub-inodes read).
360    fn visit_sub_inodes(
361        &self,
362        offset: usize,
363        dir_cnt: usize,
364        visitor: &mut dyn DirentVisitor,
365        fs_guard: &MutexGuard<()>,
366    ) -> Result<(usize, usize)> {
367        if !self.inode_type.is_directory() {
368            return_errno!(Errno::ENOTDIR)
369        }
370        if dir_cnt == 0 {
371            return Ok((offset, 0));
372        }
373
374        let fs = self.fs();
375        let cluster_size = fs.cluster_size();
376
377        let mut iter = ExfatDentryIterator::new(self.page_cache.pages(), offset, None)?;
378
379        let mut dir_read = 0;
380        let mut current_off = offset;
381
382        // We need to skip empty or deleted dentry.
383        while dir_read < dir_cnt {
384            let dentry_result = iter.next();
385
386            if dentry_result.is_none() {
387                return_errno_with_message!(Errno::ENOENT, "inode data not available")
388            }
389
390            let dentry = dentry_result.unwrap()?;
391
392            if let ExfatDentry::File(file) = dentry {
393                if let Ok(dentry_set_size) =
394                    self.visit_sub_inode(&file, &mut iter, current_off, visitor, fs_guard)
395                {
396                    current_off += dentry_set_size;
397                    dir_read += 1;
398                } else {
399                    return Ok((current_off, dir_read));
400                }
401            } else {
402                current_off += DENTRY_SIZE;
403            }
404        }
405
406        Ok((current_off, dir_read))
407    }

409    /// Visit a sub-inode at offset. Return the dentry-set size of the sub-inode.
410    /// Dirent visitor will extract information from the inode.
411    fn visit_sub_inode(
412        &self,
413        file_dentry: &ExfatFileDentry,
414        iter: &mut ExfatDentryIterator,
415        offset: usize,
416        visitor: &mut dyn DirentVisitor,
417        fs_guard: &MutexGuard<()>,
418    ) -> Result<usize> {
419        if !self.inode_type.is_directory() {
420            return_errno!(Errno::ENOTDIR)
421        }
422        let fs = self.fs();
423        let cluster_size = fs.cluster_size();
424
425        let dentry_position = self.start_chain.walk_to_cluster_at_offset(offset)?;
426
427        if let Some(child_inode) = fs.find_opened_inode(make_hash_index(
428            dentry_position.0.cluster_id(),
429            dentry_position.1 as u32,
430        )) {
431            // Inode already exists.
432            let child_inner = child_inode.inner.read();
433
434            for i in 0..(child_inner.dentry_set_size / DENTRY_SIZE - 1) {
435                let dentry_result = iter.next();
436                if dentry_result.is_none() {
437                    return_errno_with_message!(Errno::ENOENT, "inode data not available")
438                }
439                dentry_result.unwrap()?;
440            }
441            visitor.visit(
442                &child_inner.name.to_string(),
443                child_inner.ino,
444                child_inner.inode_type,
445                offset,
446            )?;
447
448            Ok(child_inner.dentry_set_size)
449        } else {
450            // Otherwise, create a new node and insert it to hash map.
451            let ino = fs.alloc_inode_number();
452            let child_inode = ExfatInode::read_from_iterator(
453                fs.clone(),
454                offset / DENTRY_SIZE,
455                dentry_position,
456                file_dentry,
457                iter,
458                self.hash_index(),
459                fs_guard,
460            )?;
461            let _ = fs.insert_inode(child_inode.clone());
462            let child_inner = child_inode.inner.read();
463
464            visitor.visit(
465                &child_inner.name.to_string(),
466                ino,
467                child_inner.inode_type,
468                offset,
469            )?;
470            Ok(child_inner.dentry_set_size)
471        }
472    }

474    /// Look up a target with "name", cur inode represent a dir.
475    /// Return (target inode, dentries start offset, dentry set size).
476    /// No inode should hold the write lock.
477    fn lookup_by_name(
478        &self,
479        target_name: &str,
480        case_sensitive: bool,
481        fs_guard: &MutexGuard<()>,
482    ) -> Result<Arc<ExfatInode>> {
483        let fs = self.fs();
484
485        let target_upcase = if !case_sensitive {
486            fs.upcase_table().lock().str_to_upcase(target_name)?
487        } else {
488            target_name.to_string()
489        };
490
491        // FIXME: This isn't expected by the compiler.
492        #[expect(non_local_definitions)]
493        impl DirentVisitor for Vec<(String, usize)> {
494            fn visit(
495                &mut self,
496                name: &str,
497                ino: u64,
498                type_: InodeType,
499                offset: usize,
500            ) -> Result<()> {
501                self.push((name.into(), offset));
502                Ok(())
503            }
504        }
505
506        let mut name_and_offsets: Vec<(String, usize)> = vec![];
507        self.visit_sub_inodes(
508            0,
509            self.num_sub_inodes as usize,
510            &mut name_and_offsets,
511            fs_guard,
512        )?;
513
514        for (name, offset) in name_and_offsets {
515            let name_upcase = if !case_sensitive {
516                fs.upcase_table().lock().str_to_upcase(&name)?
517            } else {
518                name
519            };
520
521            if name_upcase.eq(&target_upcase) {
522                let chain_off = self.start_chain.walk_to_cluster_at_offset(offset)?;
523                let hash = make_hash_index(chain_off.0.cluster_id(), chain_off.1 as u32);
524                let inode = fs.find_opened_inode(hash).unwrap();
525
526                return Ok(inode.clone());
527            }
528        }
529        return_errno!(Errno::ENOENT)
530    }

532    fn delete_associated_secondary_clusters(
533        &mut self,
534        dentry: &ExfatDentry,
535        fs_guard: &MutexGuard<()>,
536    ) -> Result<()> {
537        let fs = self.fs();
538        let cluster_size = fs.cluster_size();
539        match dentry {
540            ExfatDentry::VendorAlloc(inner) => {
541                if !fs.is_valid_cluster(inner.start_cluster) {
542                    return Ok(());
543                }
544                let num_to_free = (inner.size as usize / cluster_size) as u32;
545                ExfatChain::new(
546                    self.fs.clone(),
547                    inner.start_cluster,
548                    Some(num_to_free),
549                    FatChainFlags::ALLOC_POSSIBLE,
550                )?
551                .remove_clusters_from_tail(num_to_free, self.is_sync())?;
552            }
553            ExfatDentry::GenericSecondary(inner) => {
554                if !fs.is_valid_cluster(inner.start_cluster) {
555                    return Ok(());
556                }
557                let num_to_free = (inner.size as usize / cluster_size) as u32;
558                ExfatChain::new(
559                    self.fs.clone(),
560                    inner.start_cluster,
561                    Some(num_to_free),
562                    FatChainFlags::ALLOC_POSSIBLE,
563                )?
564                .remove_clusters_from_tail(num_to_free, self.is_sync())?;
565            }
566            _ => {}
567        };
568        Ok(())
569    }

616impl ExfatInode {
617    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
618        let inner = self.inner.upread();
619        if inner.inode_type.is_directory() {
620            return_errno!(Errno::EISDIR)
621        }
622        let (read_off, read_len) = {
623            let file_size = inner.size;
624            let start = file_size.min(offset);
625            let end = file_size.min(offset + writer.avail());
626            (start, end - start)
627        };
628        inner.page_cache.pages().read(read_off, writer)?;
629
630        inner.upgrade().update_atime()?;
631        Ok(read_len)
632    }
633
634    // The offset and the length of buffer must be multiples of the block size.
635    fn read_direct_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
636        let inner = self.inner.upread();
637        if inner.inode_type.is_directory() {
638            return_errno!(Errno::EISDIR)
639        }
640        if !is_block_aligned(offset) || !is_block_aligned(writer.avail()) {
641            return_errno_with_message!(Errno::EINVAL, "not block-aligned");
642        }
643
644        let sector_size = inner.fs().sector_size();
645
646        let (read_off, read_len) = {
647            let file_size = inner.size;
648            let start = file_size.min(offset).align_down(sector_size);
649            let end = file_size
650                .min(offset + writer.avail())
651                .align_down(sector_size);
652            (start, end - start)
653        };
654
655        inner
656            .page_cache
657            .discard_range(read_off..read_off + read_len);
658
659        let bio_segment = BioSegment::alloc(1, BioDirection::FromDevice);
660
661        let start_pos = inner.start_chain.walk_to_cluster_at_offset(read_off)?;
662        let cluster_size = inner.fs().cluster_size();
663        let mut cur_cluster = start_pos.0.clone();
664        let mut cur_offset = start_pos.1;
665        for _ in Bid::from_offset(read_off)..Bid::from_offset(read_off + read_len) {
666            let physical_bid =
667                Bid::from_offset(cur_cluster.cluster_id() as usize * cluster_size + cur_offset);
668            inner
669                .fs()
670                .block_device()
671                .read_blocks(physical_bid, bio_segment.clone())?;
672            bio_segment.reader().unwrap().read_fallible(writer)?;
673
674            cur_offset += BLOCK_SIZE;
675            if cur_offset >= cluster_size {
676                cur_cluster = cur_cluster.walk(1)?;
677                cur_offset %= BLOCK_SIZE;
678            }
679        }
680
681        inner.upgrade().update_atime()?;
682        Ok(read_len)
683    }

685    fn write_at(&self, offset: usize, reader: &mut VmReader) -> Result<usize> {
686        let write_len = reader.remain();
687        // We need to obtain the fs lock to resize the file.
688        let new_size = {
689            let mut inner = self.inner.write();
690            if inner.inode_type.is_directory() {
691                return_errno!(Errno::EISDIR)
692            }
693
694            let file_size = inner.size;
695            let file_allocated_size = inner.size_allocated;
696            let new_size = offset + write_len;
697            let fs = inner.fs();
698            let fs_guard = fs.lock();
699            if new_size > file_size {
700                if new_size > file_allocated_size {
701                    inner.resize(new_size, &fs_guard)?;
702                }
703                inner.page_cache.resize(new_size)?;
704            }
705            new_size.max(file_size)
706        };
707
708        // Locks released here, so that file write can be parallelized.
709        let inner = self.inner.upread();
710        inner.page_cache.pages().write(offset, reader)?;
711
712        // Update timestamps and size.
713        {
714            let mut inner = inner.upgrade();
715
716            inner.update_atime_and_mtime()?;
717            inner.size = new_size;
718        }
719
720        let inner = self.inner.read();
721
722        // Write data back.
723        if inner.is_sync() {
724            let fs = inner.fs();
725            let fs_guard = fs.lock();
726            inner.sync_all(&fs_guard)?;
727        }
728
729        Ok(write_len)
730    }

732    fn write_direct_at(&self, offset: usize, reader: &mut VmReader) -> Result<usize> {
733        let write_len = reader.remain();
734        let inner = self.inner.upread();
735        if inner.inode_type.is_directory() {
736            return_errno!(Errno::EISDIR)
737        }
738        if !is_block_aligned(offset) || !is_block_aligned(write_len) {
739            return_errno_with_message!(Errno::EINVAL, "not block-aligned");
740        }
741
742        let file_size = inner.size;
743        let file_allocated_size = inner.size_allocated;
744        let end_offset = offset + write_len;
745
746        let start = offset.min(file_size);
747        let end = end_offset.min(file_size);
748        inner.page_cache.discard_range(start..end);
749
750        let new_size = {
751            let mut inner = inner.upgrade();
752            if end_offset > file_size {
753                let fs = inner.fs();
754                let fs_guard = fs.lock();
755                if end_offset > file_allocated_size {
756                    inner.resize(end_offset, &fs_guard)?;
757                }
758                inner.page_cache.resize(end_offset)?;
759            }
760            file_size.max(end_offset)
761        };
762
763        let inner = self.inner.upread();
764
765        let bio_segment = BioSegment::alloc(1, BioDirection::ToDevice);
766        let start_pos = inner.start_chain.walk_to_cluster_at_offset(offset)?;
767        let cluster_size = inner.fs().cluster_size();
768        let mut cur_cluster = start_pos.0.clone();
769        let mut cur_offset = start_pos.1;
770        for _ in Bid::from_offset(offset)..Bid::from_offset(end_offset) {
771            bio_segment.writer().unwrap().write_fallible(reader)?;
772            let physical_bid =
773                Bid::from_offset(cur_cluster.cluster_id() as usize * cluster_size + cur_offset);
774            let fs = inner.fs();
775            fs.block_device()
776                .write_blocks(physical_bid, bio_segment.clone())?;
777
778            cur_offset += BLOCK_SIZE;
779            if cur_offset >= cluster_size {
780                cur_cluster = cur_cluster.walk(1)?;
781                cur_offset %= BLOCK_SIZE;
782            }
783        }
784
785        {
786            let mut inner = inner.upgrade();
787            inner.update_atime_and_mtime()?;
788            inner.size = new_size;
789        }
790
791        let inner = self.inner.read();
792        // Sync this inode since size has changed.
793        if inner.is_sync() {
794            let fs = inner.fs();
795            let fs_guard = fs.lock();
796            inner.sync_metadata(&fs_guard)?;
797        }
798
799        Ok(write_len)
800    }

879    fn build_from_dentry_set(
880        fs: Arc<ExfatFs>,
881        dentry_set: &ExfatDentrySet,
882        dentry_set_position: ExfatChainPosition,
883        dentry_entry: u32,
884        parent_hash: usize,
885        fs_guard: &MutexGuard<()>,
886    ) -> Result<Arc<ExfatInode>> {
887        const EXFAT_MINIMUM_DENTRY: usize = 3;
888
889        let ino = fs.alloc_inode_number();
890
891        if dentry_set.len() < EXFAT_MINIMUM_DENTRY {
892            return_errno_with_message!(Errno::EINVAL, "invalid dentry length")
893        }
894
895        let dentry_set_size = dentry_set.len() * DENTRY_SIZE;
896
897        let fs_weak = Arc::downgrade(&fs);
898
899        let file = dentry_set.get_file_dentry();
900        let attr = FatAttr::from_bits_truncate(file.attribute);
901
902        let inode_type = if attr.contains(FatAttr::DIRECTORY) {
903            InodeType::Dir
904        } else {
905            InodeType::File
906        };
907
908        let ctime = DosTimestamp::new(
909            file.create_time,
910            file.create_date,
911            file.create_time_cs,
912            file.create_utc_offset,
913        )?;
914        let mtime = DosTimestamp::new(
915            file.modify_time,
916            file.modify_date,
917            file.modify_time_cs,
918            file.modify_utc_offset,
919        )?;
920        let atime = DosTimestamp::new(
921            file.access_time,
922            file.access_date,
923            0,
924            file.access_utc_offset,
925        )?;
926
927        let stream = dentry_set.get_stream_dentry();
928        let size = stream.valid_size as usize;
929        let size_allocated = stream.size as usize;
930
931        if attr.contains(FatAttr::DIRECTORY) && size != size_allocated {
932            return_errno_with_message!(
933                Errno::EINVAL,
934                "allocated_size and valid_size can only be different for files!"
935            )
936        }
937
938        let chain_flag = FatChainFlags::from_bits_truncate(stream.flags);
939        let start_cluster = stream.start_cluster;
940        let num_clusters = size_allocated.align_up(fs.cluster_size()) / fs.cluster_size();
941
942        let start_chain = ExfatChain::new(
943            fs_weak.clone(),
944            start_cluster,
945            Some(num_clusters as u32),
946            chain_flag,
947        )?;
948
949        let name = dentry_set.get_name(fs.upcase_table())?;
950        let inode = Arc::new_cyclic(|weak_self| ExfatInode {
951            inner: RwMutex::new(ExfatInodeInner {
952                ino,
953                dentry_set_position,
954                dentry_set_size,
955                dentry_entry,
956                inode_type,
957                attr,
958                start_chain,
959                size,
960                size_allocated,
961                atime,
962                mtime,
963                ctime,
964                num_sub_inodes: 0,
965                num_sub_dirs: 0,
966                name,
967                is_deleted: false,
968                parent_hash,
969                fs: fs_weak,
970                page_cache: PageCache::with_capacity(size, weak_self.clone() as _).unwrap(),
971            }),
972            extension: Extension::new(),
973        });
974
975        if matches!(inode_type, InodeType::Dir) {
976            let inner = inode.inner.upread();
977            let num_sub_inode_dir: (usize, usize) = inner.count_num_sub_inode_and_dir(fs_guard)?;
978
979            let mut inode_inner = inner.upgrade();
980
981            inode_inner.num_sub_inodes = num_sub_inode_dir.0 as u32;
982            inode_inner.num_sub_dirs = num_sub_inode_dir.1 as u32;
983        }
984
985        Ok(inode)
986    }

1009    /// Find empty dentry. If not found, expand the cluster chain.
1010    fn find_empty_dentries(&self, num_dentries: usize, fs_guard: &MutexGuard<()>) -> Result<usize> {
1011        let inner = self.inner.upread();
1012
1013        let dentry_iterator =
1014            ExfatDentryIterator::new(inner.page_cache.pages(), 0, Some(inner.size))?;
1015
1016        let mut contiguous_unused = 0;
1017        let mut entry_id = 0;
1018
1019        for dentry_result in dentry_iterator {
1020            let dentry = dentry_result?;
1021            match dentry {
1022                ExfatDentry::UnUsed | ExfatDentry::Deleted(_) => {
1023                    contiguous_unused += 1;
1024                }
1025                _ => {
1026                    contiguous_unused = 0;
1027                }
1028            }
1029            if contiguous_unused >= num_dentries {
1030                return Ok(entry_id - (num_dentries - 1));
1031            }
1032            entry_id += 1;
1033        }
1034
1035        // Empty entries not found, allocate new cluster.
1036
1037        if inner.size >= EXFAT_MAX_DENTRIES as usize * DENTRY_SIZE {
1038            return_errno!(Errno::ENOSPC)
1039        }
1040        let fs = inner.fs();
1041        let cluster_size = fs.cluster_size();
1042        let cluster_to_be_allocated =
1043            (num_dentries * DENTRY_SIZE).align_up(cluster_size) / cluster_size;
1044
1045        let is_sync = inner.is_sync();
1046        let old_size_allocated = inner.size_allocated;
1047        let new_size_allocated = old_size_allocated + cluster_size * cluster_to_be_allocated;
1048        {
1049            let mut inner = inner.upgrade();
1050            inner
1051                .start_chain
1052                .extend_clusters(cluster_to_be_allocated as u32, is_sync)?;
1053
1054            inner.size_allocated = new_size_allocated;
1055            inner.size = new_size_allocated;
1056
1057            inner.page_cache.resize(new_size_allocated)?;
1058        }
1059        let inner = self.inner.read();
1060
1061        // We need to write unused dentries (i.e. 0) to page cache.
1062        inner
1063            .page_cache
1064            .pages()
1065            .clear(old_size_allocated..new_size_allocated)?;
1066
1067        Ok(entry_id)
1068    }

1071    fn add_entry(
1072        &self,
1073        name: &str,
1074        inode_type: InodeType,
1075        mode: InodeMode,
1076        fs_guard: &MutexGuard<()>,
1077    ) -> Result<Arc<ExfatInode>> {
1078        if name.len() > MAX_NAME_LENGTH {
1079            return_errno!(Errno::ENAMETOOLONG)
1080        }
1081        let fs = self.inner.read().fs();
1082
1083        // TODO: remove trailing periods of pathname.
1084        // Do not allow creation of files with names ending with period(s).
1085
1086        let name_dentries = name.len().div_ceil(EXFAT_FILE_NAME_LEN);
1087        let num_dentries = name_dentries + 2; // FILE Entry + Stream Entry + Name Entry
1088
1089        // We update the size of inode before writing page_cache, but it is fine since we've cleaned the page_cache.
1090        let entry = self.find_empty_dentries(num_dentries, fs_guard)? as u32;
1091
1092        let dentry_set = ExfatDentrySet::from(fs.clone(), name, inode_type, mode)?;
1093
1094        let start_off = entry as usize * DENTRY_SIZE;
1095        let end_off = (entry as usize + num_dentries) * DENTRY_SIZE;
1096
1097        let inner = self.inner.upread();
1098        inner
1099            .page_cache
1100            .pages()
1101            .write_bytes(start_off, &dentry_set.to_le_bytes())?;
1102
1103        let mut inner = inner.upgrade();
1104
1105        inner.num_sub_inodes += 1;
1106        if inode_type.is_directory() {
1107            inner.num_sub_dirs += 1;
1108        }
1109
1110        let pos = inner.start_chain.walk_to_cluster_at_offset(start_off)?;
1111
1112        let new_inode = ExfatInode::build_from_dentry_set(
1113            fs.clone(),
1114            &dentry_set,
1115            pos,
1116            entry,
1117            inner.hash_index(),
1118            fs_guard,
1119        )?;
1120
1121        if inode_type.is_directory() && !fs.mount_option().zero_size_dir {
1122            // TODO: We need to resize the directory so that it contains at least 1 cluster if zero_size_dir is not enabled.
1123            // new_inode.resize(new_size)
1124        }
1125        Ok(new_inode)
1126    }

1128    // Delete dentry set for current directory.
1129    fn delete_dentry_set(
1130        &self,
1131        offset: usize,
1132        len: usize,
1133        fs_guard: &MutexGuard<()>,
1134    ) -> Result<()> {
1135        let fs = self.inner.read().fs();
1136        let mut buf = vec![0; len];
1137
1138        self.inner
1139            .read()
1140            .page_cache
1141            .pages()
1142            .read_bytes(offset, &mut buf)?;
1143
1144        let num_dentry = len / DENTRY_SIZE;
1145
1146        let cluster_size = fs.cluster_size();
1147        for i in 0..num_dentry {
1148            let buf_offset = DENTRY_SIZE * i;
1149            // Delete cluster chain if needed.
1150            let dentry = ExfatDentry::try_from(RawExfatDentry::from_bytes(
1151                &buf[buf_offset..buf_offset + DENTRY_SIZE],
1152            ))?;
1153            self.inner
1154                .write()
1155                .delete_associated_secondary_clusters(&dentry, fs_guard)?;
1156            // Mark this dentry as deleted.
1157            buf[buf_offset] &= 0x7F;
1158        }
1159
1160        self.inner
1161            .read()
1162            .page_cache
1163            .pages()
1164            .write_bytes(offset, &buf)?;
1165
1166        // FIXME: We must make sure that there are no spare tailing clusters in a directory.
1167        Ok(())
1168    }

1188    fn update_subdir_parent_hash(&self, fs_guard: &MutexGuard<()>) -> Result<()> {
1189        let inner = self.inner.read();
1190        if !inner.inode_type.is_directory() {
1191            return Ok(());
1192        }
1193        let new_parent_hash = self.hash_index();
1194        let sub_dir = inner.num_sub_inodes;
1195        let mut child_offsets: Vec<usize> = vec![];
1196        // FIXME: This isn't expected by the compiler.
1197        #[expect(non_local_definitions)]
1198        impl DirentVisitor for Vec<usize> {
1199            fn visit(
1200                &mut self,
1201                name: &str,
1202                ino: u64,
1203                type_: InodeType,
1204                offset: usize,
1205            ) -> Result<()> {
1206                self.push(offset);
1207                Ok(())
1208            }
1209        }
1210        inner.visit_sub_inodes(0, sub_dir as usize, &mut child_offsets, fs_guard)?;
1211

1213        for offset in child_offsets {
1214            let child_dentry_pos = start_chain.walk_to_cluster_at_offset(offset)?;
1215            let child_hash =
1216                make_hash_index(child_dentry_pos.0.cluster_id(), child_dentry_pos.1 as u32);
1217            let child_inode = inner.fs().find_opened_inode(child_hash).unwrap();
1218            child_inode.inner.write().parent_hash = new_parent_hash;
1219        }
1220        Ok(())
1221    }

1223    /// Unlink a file or remove a directory.
1224    /// Need to delete dentry set and inode.
1225    /// Delete the file contents if delete_content is set.
1226    fn delete_inode(
1227        &self,
1228        inode: Arc<ExfatInode>,
1229        delete_contents: bool,
1230        fs_guard: &MutexGuard<()>,
1231    ) -> Result<()> {
1232        // Delete directory contents directly.
1233        let is_dir = inode.inner.read().inode_type.is_directory();
1234        if delete_contents {
1235            if is_dir {
1236                inode.inner.write().resize(0, fs_guard)?;
1237                inode.inner.read().page_cache.resize(0)?;
1238            }
1239            // Set the delete flag.
1240            inode.inner.write().is_deleted = true;
1241        }
1242        // Remove the inode.
1243        self.inner.read().fs().remove_inode(inode.hash_index());
1244        // Remove dentry set.
1245        let dentry_set_offset = inode.inner.read().dentry_entry as usize * DENTRY_SIZE;
1246        let dentry_set_len = inode.inner.read().dentry_set_size;
1247        self.delete_dentry_set(dentry_set_offset, dentry_set_len, fs_guard)?;
1248        self.inner.write().update_metadata_for_delete(is_dir);
1249        Ok(())
1250    }

1317    fn resize(&self, new_size: usize) -> Result<()> {
1318        let inner = self.inner.upread();
1319
1320        if inner.inode_type.is_directory() {
1321            return_errno!(Errno::EISDIR)
1322        }
1323        if !inner.inode_type.is_regular_file() {
1324            return_errno!(Errno::EINVAL);
1325        }
1326
1327        let file_size = inner.size;
1328        let fs = inner.fs();
1329        let fs_guard = fs.lock();
1330
1331        inner.upgrade().resize(new_size, &fs_guard)?;
1332
1333        // Update the size of page cache.
1334        let inner = self.inner.read();
1335
1336        // We will delay updating the page_cache size when enlarging an inode until the real write.
1337        if new_size < file_size {
1338            self.inner.read().page_cache.resize(new_size)?;
1339        }
1340
1341        // Sync this inode since size has changed.
1342        if inner.is_sync() {
1343            inner.sync_metadata(&fs_guard)?;
1344        }
1345
1346        Ok(())
1347    }

1481    fn readdir_at(&self, dir_cnt: usize, visitor: &mut dyn DirentVisitor) -> Result<usize> {
1482        let inner = self.inner.upread();
1483
1484        if dir_cnt >= (inner.num_sub_inodes + 2) as usize {
1485            return Ok(0);
1486        }
1487
1488        let mut empty_visitor = EmptyVisitor;
1489
1490        let dir_read = {
1491            let fs = inner.fs();
1492            let fs_guard = fs.lock();
1493
1494            let mut dir_read = 0usize;
1495
1496            if dir_cnt == 0
1497                && visitor
1498                    .visit(".", inner.ino, inner.inode_type, 0xFFFFFFFFFFFFFFFEusize)
1499                    .is_ok()
1500            {
1501                dir_read += 1;
1502            }
1503
1504            if dir_cnt <= 1 {
1505                let parent_inode = inner.get_parent_inode().unwrap();
1506                let parent_inner = parent_inode.inner.read();
1507                let ino = parent_inner.ino;
1508                let type_ = parent_inner.inode_type;
1509                if visitor
1510                    .visit("..", ino, type_, 0xFFFFFFFFFFFFFFFFusize)
1511                    .is_ok()
1512                {
1513                    dir_read += 1;
1514                }
1515            }
1516
1517            // Skip . and ..
1518            let dir_to_skip = dir_cnt.saturating_sub(2);
1519
1520            // Skip previous directories.
1521            let (off, _) = inner.visit_sub_inodes(0, dir_to_skip, &mut empty_visitor, &fs_guard)?;
1522            let (_, read) = inner.visit_sub_inodes(
1523                off,
1524                inner.num_sub_inodes as usize - dir_to_skip,
1525                visitor,
1526                &fs_guard,
1527            )?;
1528            dir_read += read;
1529            dir_read
1530        };
1531
1532        inner.upgrade().update_atime()?;
1533
1534        Ok(dir_read)
1535    }

1631    fn rename(&self, old_name: &str, target: &Arc<dyn Inode>, new_name: &str) -> Result<()> {
1632        if is_dot_or_dotdot(old_name) || is_dot_or_dotdot(new_name) {
1633            return_errno!(Errno::EISDIR);
1634        }
1635        if old_name.len() > MAX_NAME_LENGTH || new_name.len() > MAX_NAME_LENGTH {
1636            return_errno!(Errno::ENAMETOOLONG)
1637        }
1638        let Some(target_) = target.downcast_ref::<ExfatInode>() else {
1639            return_errno_with_message!(Errno::EINVAL, "not an exfat inode")
1640        };
1641        if !self.inner.read().inode_type.is_directory()
1642            || !target_.inner.read().inode_type.is_directory()
1643        {
1644            return_errno!(Errno::ENOTDIR)
1645        }
1646
1647        let fs = self.inner.read().fs();
1648        let fs_guard = fs.lock();
1649        // Rename something to itself, return success directly.
1650        let up_old_name = fs.upcase_table().lock().str_to_upcase(old_name)?;
1651        let up_new_name = fs.upcase_table().lock().str_to_upcase(new_name)?;
1652        if self.inner.read().ino == target_.inner.read().ino && up_old_name.eq(&up_new_name) {
1653            return Ok(());
1654        }
1655
1656        // Read 'old_name' file or dir and its dentries.
1657        let old_inode = self
1658            .inner
1659            .read()
1660            .lookup_by_name(old_name, true, &fs_guard)?;
1661        // FIXME: Users may be confused, since inode with the same upper case name will be removed.
1662        let lookup_exist_result = target_
1663            .inner
1664            .read()
1665            .lookup_by_name(new_name, false, &fs_guard);
1666        // Check for the corner cases.
1667        if let Ok(ref exist_inode) = lookup_exist_result {
1668            check_corner_cases_for_rename(&old_inode, exist_inode)?;
1669        }
1670
1671        // All checks are done here. This is a valid rename and it needs to modify the metadata.
1672        self.delete_inode(old_inode.clone(), false, &fs_guard)?;
1673        // Create the new dentries.
1674        let new_inode =
1675            target_.add_entry(new_name, old_inode.type_(), old_inode.mode()?, &fs_guard)?;
1676        // Update metadata.
1677        old_inode.copy_metadata_from(new_inode);
1678        // Update its children's parent_hash.
1679        old_inode.update_subdir_parent_hash(&fs_guard)?;
1680        // Insert back.
1681        let _ = fs.insert_inode(old_inode.clone());
1682        // Remove the exist 'new_name' file.
1683        if let Ok(exist_inode) = lookup_exist_result {
1684            target_.delete_inode(exist_inode, true, &fs_guard)?;
1685        }
1686        // Update the times.
1687        self.inner.write().update_atime_and_mtime()?;
1688        target_.inner.write().update_atime_and_mtime()?;
1689        // Sync
1690        if self.inner.read().is_sync() || target_.inner.read().is_sync() {
1691            // TODO: what if fs crashed between syncing?
1692            old_inode.inner.read().sync_all(&fs_guard)?;
1693            target_.inner.read().sync_all(&fs_guard)?;
1694            self.inner.read().sync_all(&fs_guard)?;
1695        }
1696        Ok(())
1697    }
