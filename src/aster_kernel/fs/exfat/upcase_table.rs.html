6use align_ext::AlignExt;
7
8use super::{
9    constants::UNICODE_SIZE,
10    dentry::{ExfatDentry, ExfatDentryIterator, ExfatUpcaseDentry, UTF16Char},
11    fat::ExfatChain,
12    fs::ExfatFs,
13    utils::calc_checksum_32,
14};
15use crate::{fs::exfat::fat::FatChainFlags, prelude::*, vm::vmo::Vmo};
16
17const UPCASE_MANDATORY_SIZE: usize = 128;
18
19#[derive(Debug)]
20pub(super) struct ExfatUpcaseTable {
21    upcase_table: [u16; UPCASE_MANDATORY_SIZE],
22    fs: Weak<ExfatFs>,
23}
24
25impl ExfatUpcaseTable {
26    pub(super) fn empty() -> Self {
27        Self {
28            upcase_table: [0; UPCASE_MANDATORY_SIZE],
29            fs: Weak::default(),
30        }
31    }
32
33    pub(super) fn load(
34        fs_weak: Weak<ExfatFs>,
35        root_page_cache: &Vmo,
36        root_chain: ExfatChain,
37    ) -> Result<Self> {
38        let dentry_iterator = ExfatDentryIterator::new(root_page_cache, 0, None)?;
39
40        for dentry_result in dentry_iterator {
41            let dentry = dentry_result?;
42            if let ExfatDentry::Upcase(upcase_dentry) = dentry {
43                return Self::load_table_from_dentry(fs_weak, &upcase_dentry);
44            }
45        }
46
47        return_errno_with_message!(Errno::EINVAL, "Upcase table not found")
48    }
49
50    fn load_table_from_dentry(fs_weak: Weak<ExfatFs>, dentry: &ExfatUpcaseDentry) -> Result<Self> {
51        if (dentry.size as usize) < UPCASE_MANDATORY_SIZE * UNICODE_SIZE {
52            return_errno_with_message!(Errno::EINVAL, "Upcase table too small")
53        }
54
55        let fs = fs_weak.upgrade().unwrap();
56        let num_clusters = (dentry.size as usize).align_up(fs.cluster_size()) / fs.cluster_size();
57        let chain = ExfatChain::new(
58            fs_weak.clone(),
59            dentry.start_cluster,
60            Some(num_clusters as u32),
61            FatChainFlags::ALLOC_POSSIBLE,
62        )?;
63
64        let mut buf = vec![0; dentry.size as usize];
65        fs.read_meta_at(chain.physical_cluster_start_offset(), &mut buf)?;
66
67        if dentry.checksum != calc_checksum_32(&buf) {
68            return_errno_with_message!(Errno::EINVAL, "invalid checksum")
69        }
70
71        let mut res = ExfatUpcaseTable {
72            upcase_table: [0; UPCASE_MANDATORY_SIZE],
73            fs: fs_weak,
74        };
75
76        for i in 0..UPCASE_MANDATORY_SIZE {
77            res.upcase_table[i] = (buf[2 * i] as u16) | ((buf[2 * i + 1] as u16) << 8);
78        }
79
80        Ok(res)
81    }
82
83    pub(super) fn str_to_upcase(&self, value: &str) -> Result<String> {
84        // TODO: use upcase table
85        Ok(value.to_uppercase())
86    }
87
88    pub(super) fn slice_to_upcase(&self, buf: &mut [UTF16Char]) -> Result<()> {
89        for value in buf {
90            *value = self.char_to_upcase(*value)?;
91        }
92        Ok(())
93    }
94
95    pub(super) fn char_to_upcase(&self, value: UTF16Char) -> Result<UTF16Char> {
96        if (value as usize) < UPCASE_MANDATORY_SIZE {
97            Ok(self.upcase_table[value as usize])
98        } else {
99            Ok(value)
100        }
101    }
102}
