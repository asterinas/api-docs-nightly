3use alloc::boxed::ThinBox;
4
5use super::{FlockList, Inode, RangeLockList};
6use crate::fs::notify::FsEventPublisher;

8/// Context for FS locks.
9pub struct FsLockContext {
10    range_lock_list: RangeLockList,
11    flock_list: FlockList,
12}
13
14impl FsLockContext {
15    pub(self) fn new() -> Self {
16        Self {
17            range_lock_list: RangeLockList::new(),
18            flock_list: FlockList::new(),
19        }
20    }
21
22    /// Returns a reference to the range lock list.
23    pub fn range_lock_list(&self) -> &RangeLockList {
24        &self.range_lock_list
25    }
26
27    /// Returns a reference to the flock list.
28    pub fn flock_list(&self) -> &FlockList {
29        &self.flock_list
30    }
31}
32
33/// A trait that instantiates kernel types for the inode [`Extension`].
34///
35/// [`Extension`]: super::Extension
36pub trait InodeExt {
37    /// Gets or initializes the FS event publisher.
38    ///
39    /// If the publisher does not exist for this inode, it will be created.
40    fn fs_event_publisher_or_init(&self) -> &FsEventPublisher;
41
42    /// Returns a reference to the FS event publisher.
43    ///
44    /// If the publisher does not exist for this inode, a [`None`] will be returned.
45    fn fs_event_publisher(&self) -> Option<&FsEventPublisher>;
46
47    /// Gets or initializes the FS lock context.
48    ///
49    /// If the context does not exist for this inode, it will be created.
50    fn fs_lock_context_or_init(&self) -> &FsLockContext;
51
52    /// Returns a reference to the FS lock context.
53    ///
54    /// If the context does not exist for this inode, a [`None`] will be returned.
55    fn fs_lock_context(&self) -> Option<&FsLockContext>;
56}
57
58impl InodeExt for dyn Inode {
59    fn fs_event_publisher_or_init(&self) -> &FsEventPublisher {
60        self.extension()
61            .group1()
62            .call_once(|| ThinBox::new_unsize(FsEventPublisher::new()))
63            .downcast_ref()
64            .unwrap()
65    }
66
67    fn fs_event_publisher(&self) -> Option<&FsEventPublisher> {
68        Some(self.extension().group1().get()?.downcast_ref().unwrap())
69    }
70
71    fn fs_lock_context_or_init(&self) -> &FsLockContext {
72        self.extension()
73            .group2()
74            .call_once(|| ThinBox::new_unsize(FsLockContext::new()))
75            .downcast_ref()
76            .unwrap()
77    }
78
79    fn fs_lock_context(&self) -> Option<&FsLockContext> {
80        Some(self.extension().group2().get()?.downcast_ref().unwrap())
81    }
82}
