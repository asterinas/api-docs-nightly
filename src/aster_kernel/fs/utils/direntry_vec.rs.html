3use aster_util::slot_vec::SlotVec;
4
5use super::Inode;
6use crate::prelude::*;
7
8pub trait DirEntryVecExt {
9    /// Finds the entry by the `name`.
10    fn find_entry_by_name(&self, name: &str) -> Option<&Arc<dyn Inode>>;
11
12    /// Puts the entry given by `f` into the vector if it is not found by the `name`.
13    fn put_entry_if_not_found(
14        &mut self,
15        name: &str,
16        f: impl FnOnce() -> Arc<dyn Inode>,
17    ) -> &Arc<dyn Inode>;
18
19    /// Removes the entry by the `name`.
20    fn remove_entry_by_name(&mut self, name: &str) -> Option<(String, Arc<dyn Inode>)>;
21}
22
23impl DirEntryVecExt for SlotVec<(String, Arc<dyn Inode>)> {
24    fn find_entry_by_name(&self, name: &str) -> Option<&Arc<dyn Inode>> {
25        if let Some((_, inode)) = self.iter().find(|(child_name, _)| child_name == name) {
26            Some(inode)
27        } else {
28            None
29        }
30    }
31
32    fn put_entry_if_not_found(
33        &mut self,
34        name: &str,
35        f: impl FnOnce() -> Arc<dyn Inode>,
36    ) -> &Arc<dyn Inode> {
37        let idx = self
38            .idxes_and_items()
39            .find(|(_, (child_name, _))| child_name == name)
40            .map(|(idx, _)| idx);
41        let idx = if let Some(idx) = idx {
42            idx
43        } else {
44            let inode = f();
45            self.put((String::from(name), inode))
46        };
47        &self.get(idx).unwrap().1
48    }
49
50    fn remove_entry_by_name(&mut self, name: &str) -> Option<(String, Arc<dyn Inode>)> {
51        let idx = self
52            .idxes_and_items()
53            .find(|(_, (child_name, _))| child_name == name)
54            .map(|(idx, _)| idx);
55        if let Some(idx) = idx {
56            self.remove(idx)
57        } else {
58            None
59        }
60    }
61}
