8/// A visitor for dir entries.
9pub trait DirentVisitor {
10    /// Visit a dir entry.
11    ///
12    /// If the visitor succeeds in visiting the given inode, an `Ok(())` is returned;
13    /// Otherwise, an error is returned. Different implementations for `DirentVisitor`
14    /// may choose to report errors for different reasons. Regardless of the exact
15    /// errors and reasons, `readdir`-family methods shall stop feeding the visitor
16    /// with the next inode as long as an error is returned by the visitor.
17    ///
18    /// # Example
19    ///
20    /// `Vec<String>` is implemented as `DirentVisitor` so that the file names
21    /// under a dir can be easily collected, which is convenient for testing purposes.
22    ///
23    /// ```no_run
24    /// let mut all_dirents = Vec::new();
25    /// let dir_inode = todo!("create an inode");
26    /// dir_inode.readdir_at(0, &mut all_dirents).unwrap();
27    /// ```
28    fn visit(&mut self, name: &str, ino: u64, type_: InodeType, offset: usize) -> Result<()>;
29}
30
31impl DirentVisitor for Vec<String> {
32    fn visit(&mut self, name: &str, ino: u64, type_: InodeType, offset: usize) -> Result<()> {
33        self.push(name.into());
34        Ok(())
35    }
36}

38/// Utility to count directory entries, excluding "." and ".."
39#[derive(Default)]
40pub struct DirentCounter(usize);
41
42impl DirentCounter {
43    pub fn new() -> Self {
44        Self(0)
45    }
46
47    pub fn count(&self) -> usize {
48        self.0
49    }
50}
51
52impl DirentVisitor for DirentCounter {
53    fn visit(&mut self, name: &str, _ino: u64, _type: InodeType, _offset: usize) -> Result<()> {
54        if !is_dot_or_dotdot(name) {
55            self.0 += 1;
56        }
57        Ok(())
58    }
59}
