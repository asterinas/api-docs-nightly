9/// Represents different namespaces with different capabilities
10/// for extended attributes (xattrs).
11#[derive(Debug, PartialEq, Eq, Clone, Copy, TryFromInt, Hash)]
12#[repr(u8)]
13pub enum XattrNamespace {
14    User = 1,
15    Trusted = 2,
16    System = 3,
17    Security = 4,
18    // More namespaces can be added here.
19}
20
21/// Represents the name of an xattr. It includes both a valid namespace
22/// and a full name string slice, which contains the namespace prefix.
23///
24/// For example, "user.foo" is a valid xattr name, and its namespace
25/// is `XattrNamespace::User`.
26#[derive(Debug, Hash)]
27pub struct XattrName<'a> {
28    namespace: XattrNamespace,
29    full_name: &'a str,
30}
31
32impl XattrNamespace {
33    pub fn try_from_full_name(full_name: &str) -> Option<XattrNamespace> {
34        const USER_PREFIX: &str = "user.";
35        const TRUSTED_PREFIX: &str = "trusted.";
36        const SYSTEM_PREFIX: &str = "system.";
37        const SECURITY_PREFIX: &str = "security.";
38
39        if full_name.starts_with(USER_PREFIX) {
40            Some(XattrNamespace::User)
41        } else if full_name.starts_with(TRUSTED_PREFIX) {
42            Some(XattrNamespace::Trusted)
43        } else if full_name.starts_with(SYSTEM_PREFIX) {
44            Some(XattrNamespace::System)
45        } else if full_name.starts_with(SECURITY_PREFIX) {
46            Some(XattrNamespace::Security)
47        } else {
48            None
49        }
50    }
51
52    pub fn is_user(&self) -> bool {
53        matches!(self, XattrNamespace::User)
54    }
55}
56
57impl<'a> XattrName<'a> {
58    pub fn try_from_full_name(full_name: &'a str) -> Option<Self> {
59        let namespace = XattrNamespace::try_from_full_name(full_name)?;
60        Some(Self {
61            namespace,
62            full_name,
63        })
64    }
65
66    pub fn namespace(&self) -> XattrNamespace {
67        self.namespace
68    }
69
70    pub const fn full_name(&self) -> &'a str {
71        self.full_name
72    }
73
74    pub const fn full_name_len(&self) -> usize {
75        self.full_name.len()
76    }
77}
