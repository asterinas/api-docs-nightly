3use core::sync::atomic::AtomicU32;
4
5use atomic_integer_wrapper::define_atomic_version_of_integer_like_type;
6use bitflags::bitflags;
7
8bitflags! {
9    pub struct StatusFlags: u32 {
10        /// append on each write
11        const O_APPEND = 1 << 10;
12        /// non block
13        const O_NONBLOCK = 1 << 11;
14        /// synchronized I/O, data
15        const O_DSYNC = 1 << 12;
16        /// signal-driven I/O
17        const O_ASYNC = 1 << 13;
18        /// direct I/O
19        const O_DIRECT = 1 << 14;
20        /// on x86_64, O_LARGEFILE is 0
21        /// not update st_atime
22        const O_NOATIME = 1 << 18;
23        /// synchronized I/O, data and metadata
24        const O_SYNC = 1 << 20;
25        /// equivalent of POSIX.1's O_EXEC
26        const O_PATH = 1 << 21;
27    }
28}
29
30impl From<u32> for StatusFlags {
31    fn from(value: u32) -> Self {
32        Self::from_bits_truncate(value)
33    }
34}
35
36impl From<StatusFlags> for u32 {
37    fn from(value: StatusFlags) -> Self {
38        value.bits()
39    }
40}
41
42define_atomic_version_of_integer_like_type!(StatusFlags, {
43    /// An atomic version of `StatusFlags`.
44    #[derive(Debug)]
45    pub struct AtomicStatusFlags(AtomicU32);
46});
