32/// A trait that abstracts an inode type backed by a `SysTree` node,
33/// e.g., a `SysFs` inode and a `CgroupFs` inode.
34///
35/// The struct implementing this trait will have a default implementation for
36/// the [`Inode`] trait. Users only need to additionally implement the
37/// [`Inode::fs`] method.
38#[expect(dead_code)]
39pub(in crate::fs) trait SysTreeInodeTy: Send + Sync + 'static {
40    fn new_arc(
41        node_kind: SysTreeNodeKind,
42        metadata: Metadata,
43        mode: InodeMode,
44        parent: Weak<Self>,
45    ) -> Arc<Self>
46    where
47        Self: Sized + 'static;
48
49    fn node_kind(&self) -> &SysTreeNodeKind;
50
51    fn metadata(&self) -> &Metadata;
52
53    fn mode(&self) -> Result<InodeMode>;
54
55    fn set_mode(&self, mode: InodeMode) -> Result<()>;
56
57    fn extension(&self) -> &Extension;
58
59    fn parent(&self) -> &Weak<Self>;
60
61    fn this(&self) -> Arc<Self>
62    where
63        Self: Sized + 'static;
64
65    fn new_root(root_node: Arc<dyn SysBranchNode>) -> Arc<Self>
66    where
67        Self: Sized + 'static,
68    {
69        let node_kind = SysTreeNodeKind::Branch(root_node);
70        let parent = Weak::new();
71        Self::new_branch_dir(node_kind, None, parent)
72    }
73
74    fn new_metadata(ino: u64, type_: InodeType) -> Metadata {
75        let now = RealTimeCoarseClock::get().read_time();
76        Metadata {
77            dev: 0,
78            ino,
79            size: 0,
80            blk_size: 1024,
81            blocks: 0,
82            atime: now,
83            mtime: now,
84            ctime: now,
85            type_,
86            // The mode field in metadata will not be used
87            mode: mkmod!(a=),
88            nlinks: 1,
89            uid: Uid::new_root(),
90            gid: Gid::new_root(),
91            rdev: 0,
92        }
93    }
94
95    fn new_attr(attr: SysAttr, node: Arc<dyn SysNode>, parent: Weak<Self>) -> Arc<Self>
96    where
97        Self: Sized + 'static,
98    {
99        let node_kind = SysTreeNodeKind::Attr(attr.clone(), node);
100        let ino = ino::from_node_kind(&node_kind);
101        let metadata = Self::new_metadata(ino, InodeType::File);
102        let mode = attr.perms().into();
103        Self::new_arc(node_kind, metadata, mode, parent)
104    }
105
106    fn new_symlink(symlink: Arc<dyn SysSymlink>, parent: Weak<Self>) -> Arc<Self>
107    where
108        Self: Sized + 'static,
109    {
110        let node_kind = SysTreeNodeKind::Symlink(symlink);
111        let ino = ino::from_node_kind(&node_kind);
112        let metadata = Self::new_metadata(ino, InodeType::SymLink);
113        let mode = mkmod!(a+rwx);
114        Self::new_arc(node_kind, metadata, mode, parent)
115    }
116
117    fn new_branch_dir(
118        node_kind: SysTreeNodeKind, // Must be SysTreeNodeKind::Branch
119        mode: Option<InodeMode>,
120        parent: Weak<Self>,
121    ) -> Arc<Self>
122    where
123        Self: Sized + 'static,
124    {
125        let ino = ino::from_node_kind(&node_kind);
126        let metadata = Self::new_metadata(ino, InodeType::Dir);
127        let SysTreeNodeKind::Branch(branch_node) = &node_kind else {
128            panic!("new_branch_dir called with non-branch SysTreeNodeKind");
129        };
130
131        let mode = mode.unwrap_or_else(|| branch_node.perms().into());
132        Self::new_arc(node_kind, metadata, mode, parent)
133    }
134
135    fn new_leaf_dir(
136        node_kind: SysTreeNodeKind, // Must be SysTreeNodeKind::Leaf
137        mode: Option<InodeMode>,
138        parent: Weak<Self>,
139    ) -> Arc<Self>
140    where
141        Self: Sized + 'static,
142    {
143        let ino = ino::from_node_kind(&node_kind);
144        let metadata = Self::new_metadata(ino, InodeType::Dir); // Leaf nodes are represented as Dirs
145
146        let SysTreeNodeKind::Leaf(leaf_node) = &node_kind else {
147            panic!("new_leaf_dir called with non-leaf SysTreeNodeKind");
148        };
149
150        let mode = mode.unwrap_or_else(|| leaf_node.perms().into());
151        Self::new_arc(node_kind, metadata, mode, parent)
152    }
153
154    fn lookup_node_or_attr(&self, name: &str, sysnode: &dyn SysBranchNode) -> Result<Arc<dyn Inode>>
155    where
156        Self: Sized + 'static,
157    {
158        // Try finding a child node (Branch, Leaf, Symlink) first
159        if let Some(child_sysnode) = sysnode.child(name) {
160            let child_type = child_sysnode.type_();
161            match child_type {
162                SysNodeType::Branch => {
163                    let child_branch = child_sysnode
164                        .cast_to_branch()
165                        .ok_or(Error::new(Errno::EIO))?;
166                    let inode = Self::new_branch_dir(
167                        SysTreeNodeKind::Branch(child_branch),
168                        None,
169                        Arc::downgrade(&self.this()),
170                    );
171                    Ok(inode)
172                }
173                SysNodeType::Leaf => {
174                    let child_leaf_node =
175                        child_sysnode.cast_to_node().ok_or(Error::new(Errno::EIO))?;
176                    let inode = Self::new_leaf_dir(
177                        SysTreeNodeKind::Leaf(child_leaf_node),
178                        None,
179                        Arc::downgrade(&self.this()),
180                    );
181                    Ok(inode)
182                }
183                SysNodeType::Symlink => {
184                    let child_symlink = child_sysnode
185                        .cast_to_symlink()
186                        .ok_or(Error::new(Errno::EIO))?;
187                    let inode = Self::new_symlink(child_symlink, Arc::downgrade(&self.this()));
188                    Ok(inode)
189                }
190            }
191        } else {
192            // If no child node found, try finding an attribute of the current branch node
193            if sysnode.is_attr_absent(name) {
194                return_errno_with_message!(Errno::ENOENT, "attribute is not present");
195            }
196            let Some(attr) = sysnode.node_attrs().get(name) else {
197                return_errno_with_message!(Errno::ENOENT, "child node or attribute not found");
198            };
199
200            let parent_node_arc: Arc<dyn SysNode> = match &self.node_kind() {
201                SysTreeNodeKind::Branch(branch_arc) => branch_arc.clone(),
202                // This case shouldn't happen if lookup_node_or_attr is called correctly
203                _ => {
204                    return_errno_with_message!(
205                        Errno::EIO,
206                        "lookup_node_or_attr called on non-branch inode"
207                    );
208                }
209            };
210
211            let inode = Self::new_attr(attr.clone(), parent_node_arc, Arc::downgrade(&self.this()));
212            Ok(inode)
213        }
214    }
215
216    fn lookup_attr(&self, name: &str, sysnode: &dyn SysNode) -> Result<Arc<dyn Inode>>
217    where
218        Self: Sized + 'static,
219    {
220        // This function is called when the current inode is a Leaf directory
221        if sysnode.is_attr_absent(name) {
222            return_errno_with_message!(Errno::ENOENT, "attribute is not present");
223        }
224        let Some(attr) = sysnode.node_attrs().get(name) else {
225            return_errno_with_message!(Errno::ENOENT, "child node or attribute not found");
226        };
227
228        let leaf_node_arc: Arc<dyn SysNode> = match &self.node_kind() {
229            SysTreeNodeKind::Leaf(leaf_arc) => leaf_arc.clone(),
230            // This case shouldn't happen if lookup_attr is called correctly
231            _ => {
232                return_errno_with_message!(Errno::EIO, "lookup_attr called on non-leaf inode");
233            }
234        };
235
236        let inode = Self::new_attr(attr.clone(), leaf_node_arc, Arc::downgrade(&self.this()));
237        Ok(inode)
238    }
239
240    fn new_dentry_iter(&self, min_ino: Ino) -> impl Iterator<Item = Dentry> + '_
241    where
242        Self: Sized + 'static,
243    {
244        match &self.node_kind() {
245            SysTreeNodeKind::Branch(branch_node) => {
246                let attrs = branch_node.node_attrs();
247                let attr_iter = AttrDentryIter::new(
248                    Some((attrs.iter(), branch_node.as_ref())),
249                    self.metadata().ino,
250                    min_ino,
251                );
252                let child_objs = branch_node.children();
253                let node_iter = NodeDentryIter::new(child_objs, min_ino);
254                let special_iter = ThisAndParentDentryIter::new(self, min_ino);
255                attr_iter.chain(node_iter).chain(special_iter)
256            }
257            SysTreeNodeKind::Leaf(leaf_node) => {
258                let attrs = leaf_node.node_attrs();
259                let attr_iter = AttrDentryIter::new(
260                    Some((attrs.iter(), leaf_node.as_ref())),
261                    self.metadata().ino,
262                    min_ino,
263                );
264                let node_iter = NodeDentryIter::new(Vec::new(), min_ino);
265                let special_iter = ThisAndParentDentryIter::new(self, min_ino);
266                attr_iter.chain(node_iter).chain(special_iter)
267            }
268            SysTreeNodeKind::Attr(_, _) | SysTreeNodeKind::Symlink(_) => {
269                let attr_iter = AttrDentryIter::new(None, self.metadata().ino, min_ino);
270                let node_iter = NodeDentryIter::new(Vec::new(), min_ino);
271                let special_iter = ThisAndParentDentryIter::new(self, min_ino);
272                attr_iter.chain(node_iter).chain(special_iter)
273            }
274        }
275    }
276
277    fn is_dir(&self) -> bool {
278        matches!(
279            self.node_kind(),
280            SysTreeNodeKind::Branch(_) | SysTreeNodeKind::Leaf(_)
281        )
282    }
283}

285/// An enum that represent one of the variants of `SysTree` nodes.
286#[derive(Debug)]
287pub(in crate::fs) enum SysTreeNodeKind {
288    Branch(Arc<dyn SysBranchNode>),
289    Leaf(Arc<dyn SysNode>),
290    Attr(SysAttr, Arc<dyn SysNode>),
291    Symlink(Arc<dyn SysSymlink>),
292}

294impl<KInode: SysTreeInodeTy + Send + Sync + 'static> InodeIo for KInode {
295    default fn read_at(
296        &self,
297        offset: usize,
298        buf: &mut VmWriter,
299        _status_flags: StatusFlags,
300    ) -> Result<usize> {
301        let SysTreeNodeKind::Attr(attr, leaf) = &self.node_kind() else {
302            return Err(Error::new(Errno::EINVAL));
303        };
304
305        let len = leaf.read_attr_at(attr.name(), offset, buf)?;
306
307        Ok(len)
308    }
309
310    default fn write_at(
311        &self,
312        offset: usize,
313        buf: &mut VmReader,
314        _status_flags: StatusFlags,
315    ) -> Result<usize> {
316        let SysTreeNodeKind::Attr(attr, leaf) = &self.node_kind() else {
317            return Err(Error::new(Errno::EINVAL));
318        };
319
320        let len = if offset == 0 {
321            leaf.write_attr(attr.name(), buf)?
322        } else {
323            leaf.write_attr_at(attr.name(), offset, buf)?
324        };
325
326        Ok(len)
327    }
328}

468    default fn lookup(&self, name: &str) -> Result<Arc<dyn Inode>> {
469        if self.type_() != InodeType::Dir {
470            return_errno_with_message!(Errno::ENOTDIR, "not a directory inode");
471        }
472
473        if name == "." {
474            return Ok(self.this());
475        } else if name == ".." {
476            return Ok(self.parent().upgrade().unwrap_or_else(|| self.this()));
477        }
478
479        // Dispatch based on the concrete type of the directory inode
480        match &self.node_kind() {
481            SysTreeNodeKind::Branch(branch_node) => {
482                // Current inode is a directory corresponding to a SysBranchNode
483                self.lookup_node_or_attr(name, branch_node.as_ref())
484            }
485            SysTreeNodeKind::Leaf(leaf_node) => {
486                // Current inode is a directory corresponding to a SysNode (Leaf)
487                // Leaf directories only contain attributes, not other nodes.
488                self.lookup_attr(name, leaf_node.as_ref())
489            }
490            // Attr and Symlink nodes are not directories
491            SysTreeNodeKind::Attr(_, _) | SysTreeNodeKind::Symlink(_) => {
492                Err(Error::new(Errno::ENOTDIR))
493            }
494        }
495    }

497    default fn readdir_at(&self, offset: usize, visitor: &mut dyn DirentVisitor) -> Result<usize> {
498        // Why interpreting the `offset` argument as an inode number?
499        //
500        // It may take multiple `getdents` system calls
501        // -- and thus multiple calls to this method --
502        // to list a large directory when the syscall is provided a small buffer.
503        // Between these calls,
504        // the directory may have new entries added or existing ones removed
505        // by some concurrent users that are working on the directory.
506        // In such situations,
507        // missing some of the concurrently-added entries is inevitable,
508        // but reporting the same entry multiple times would be
509        // very confusing to the user.
510        //
511        // To address this issue,
512        // the `readdir_at` method reports entries starting from a user-given `offset`
513        // and returns an increment that the next call should be put on the `offset` argument
514        // to avoid getting duplicated entries.
515        //
516        // Different file systems may interpret the meaning of
517        // the `offset` argument differently:
518        // one may take it as a _byte_ offset,
519        // while the other may treat it as an _index_.
520        // This freedom is guaranteed by Linux as documented in
521        // [the man page of getdents](https://man7.org/linux/man-pages/man2/getdents.2.html).
522        //
523        // Our implementation of sysfs interprets the `offset`
524        // as an _inode number_.
525        // By inode numbers, directory entries will have a _stable_ order
526        // across different calls to `readdir_at`.
527        // The `new_dentry_iter` is responsible for filtering out entries
528        // with inode numbers less than `start_ino`.
529        let start_ino = offset as Ino;
530        let mut count = 0;
531        let mut last_ino = start_ino;
532
533        let dentries = {
534            let mut dentries: Vec<_> = self.new_dentry_iter(start_ino).collect();
535            dentries.sort_by_key(|d| d.ino);
536            dentries
537        };
538
539        for dentry in dentries {
540            let res = visitor.visit(&dentry.name, dentry.ino, dentry.type_, dentry.ino as usize);
541
542            if res.is_err() {
543                if count == 0 {
544                    return_errno_with_message!(Errno::EINVAL, "dirent visitor error");
545                } else {
546                    break;
547                }
548            }
549            count += 1;
550            last_ino = dentry.ino;
551        }
552
553        if count == 0 {
554            return Ok(0);
555        }
556
557        let next_ino = last_ino + 1;
558        Ok((next_ino - start_ino) as usize)
559    }

601// Update AttrDentryIter to filter by min_ino
602struct AttrDentryIter<'a, I: Iterator<Item = &'a SysAttr>> {
603    attrs_and_node: Option<(I, &'a dyn SysNode)>,
604    dir_ino: Ino,
605    min_ino: Ino,
606}
607
608impl<'a, I: Iterator<Item = &'a SysAttr>> AttrDentryIter<'a, I> {
609    fn new(attrs_and_node: Option<(I, &'a dyn SysNode)>, dir_ino: Ino, min_ino: Ino) -> Self {
610        Self {
611            attrs_and_node,
612            dir_ino,
613            min_ino,
614        }
615    }
616}
617
618impl<'a, I: Iterator<Item = &'a SysAttr>> Iterator for AttrDentryIter<'a, I> {
619    type Item = Dentry;
620
621    fn next(&mut self) -> Option<Dentry> {
622        let (iter, sys_node) = self.attrs_and_node.as_mut()?;
623
624        for attr in iter {
625            if sys_node.is_attr_absent(attr.name()) {
626                continue; // Skip hidden attributes
627            }
628
629            let attr_ino = ino::from_dir_ino_and_attr_id(self.dir_ino, attr.id());
630
631            if attr_ino >= self.min_ino {
632                // Filter by min_ino
633                return Some(Dentry {
634                    ino: attr_ino,
635                    name: attr.name().clone(),
636                    type_: InodeType::File,
637                });
638            }
639        }
640        None
641    }
642}

644struct NodeDentryIter {
645    nodes: Vec<Arc<dyn SysObj>>,
646    min_ino: Ino,
647    index: usize,
648}
649
650impl NodeDentryIter {
651    fn new(nodes: Vec<Arc<dyn SysObj>>, min_ino: Ino) -> Self {
652        Self {
653            nodes,
654            min_ino,
655            index: 0,
656        }
657    }
658}
659
660impl Iterator for NodeDentryIter {
661    type Item = Dentry;
662
663    fn next(&mut self) -> Option<Dentry> {
664        while self.index < self.nodes.len() {
665            let obj = &self.nodes[self.index];
666            self.index += 1;
667            let obj_ino = ino::from_sysnode_id(obj.id());
668
669            if obj_ino >= self.min_ino {
670                // Filter by min_ino here
671                let type_ = match obj.type_() {
672                    SysNodeType::Branch => InodeType::Dir,
673                    // Leaf nodes are presented as directories containing their attributes
674                    SysNodeType::Leaf => InodeType::Dir,
675                    SysNodeType::Symlink => InodeType::SymLink,
676                };
677                return Some(Dentry {
678                    ino: obj_ino,
679                    name: obj.name().clone(),
680                    type_,
681                });
682            }
683        }
684        None
685    }
686}

688struct ThisAndParentDentryIter<'a, KInode: SysTreeInodeTy> {
689    inode: &'a KInode,
690    min_ino: Ino,
691    state: u8, // 0 = self, 1 = parent, 2 = done
692}
693
694impl<'a, KInode: SysTreeInodeTy> ThisAndParentDentryIter<'a, KInode> {
695    fn new(inode: &'a KInode, min_ino: Ino) -> Self {
696        Self {
697            inode,
698            min_ino,
699            state: 0,
700        }
701    }
702}
703
704impl<KInode: SysTreeInodeTy> Iterator for ThisAndParentDentryIter<'_, KInode> {
705    type Item = Dentry;
706
707    fn next(&mut self) -> Option<Dentry> {
708        match self.state {
709            0 => {
710                self.state = 1;
711                if self.inode.metadata().ino >= self.min_ino {
712                    Some(Dentry {
713                        ino: self.inode.metadata().ino,
714                        name: Cow::from("."),
715                        type_: InodeType::Dir,
716                    })
717                } else {
718                    self.next()
719                }
720            }
721            1 => {
722                self.state = 2;
723                let parent_ino = self
724                    .inode
725                    .parent()
726                    .upgrade()
727                    .map_or(self.inode.metadata().ino, |p| p.metadata().ino);
728                if parent_ino >= self.min_ino {
729                    Some(Dentry {
730                        ino: parent_ino,
731                        name: Cow::from(".."),
732                        type_: InodeType::Dir,
733                    })
734                } else {
735                    None
736                }
737            }
738            _ => None,
739        }
740    }
741}

743/// A directory entry of sysfs.
744struct Dentry {
745    pub ino: Ino,
746    pub name: SysStr,
747    pub type_: InodeType,
748}
749
750mod ino {
751    use super::{Ino, SysNodeId, SysTreeNodeKind};
752
753    const ATTR_ID_BITS: u8 = 8;
754
755    pub fn from_sysnode_id(node_id: &SysNodeId) -> Ino {
756        node_id.as_u64() << ATTR_ID_BITS
757    }
758
759    pub fn from_dir_ino_and_attr_id(dir_ino: Ino, attr_id: u8) -> Ino {
760        dir_ino + (attr_id as Ino)
761    }
762
763    pub fn from_node_kind(inner: &SysTreeNodeKind) -> Ino {
764        match inner {
765            SysTreeNodeKind::Branch(branch_node) => from_sysnode_id(branch_node.id()),
766            SysTreeNodeKind::Leaf(leaf_node) => from_sysnode_id(leaf_node.id()),
767            SysTreeNodeKind::Symlink(symlink_node) => from_sysnode_id(symlink_node.id()),
768            SysTreeNodeKind::Attr(attr, node) => {
769                // node here is the parent (Branch or Leaf)
770                let dir_ino = from_sysnode_id(node.id()); // Get parent dir ino
771                from_dir_ino_and_attr_id(dir_ino, attr.id())
772            }
773        }
774    }
