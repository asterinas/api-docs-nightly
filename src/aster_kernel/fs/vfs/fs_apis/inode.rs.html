31pub struct Metadata {
32    /// The inode number, which uniquely identifies the file within the filesystem.
33    ///
34    /// Corresponds to `st_ino`.
35    pub ino: u64,
36
37    /// The size of the inode.
38    ///
39    /// The interpretation depends on `inode_type`:
40    /// - **Regular File**: The total length of the file content.
41    /// - **Directory**: The size of the directory's internal table (usually a multiple of block size).
42    /// - **Symbolic Link**: The length of the target pathname.
43    /// - **Device/Socket/FIFO**: Usually zero.
44    ///
45    /// Corresponds to `st_size`.
46    pub size: usize,
47
48    /// The optimal block size for filesystem I/O operations.
49    ///
50    /// Corresponds to `st_blksize`.
51    pub optimal_block_size: usize,
52
53    /// The number of 512-byte sectors allocated for the inode on disk.
54    ///
55    /// This represents physical usage.
56    /// For sparse files (those having holes), `size` is greater than this field.
57    /// For files with preallocated blocks (`FALLOC_FL_KEEP_SIZE`),
58    /// `size` is smaller than this field.
59    ///
60    /// Corresponds to `st_blocks`.
61    pub nr_sectors_allocated: usize,
62
63    /// The timestamp of the last access to the inode's data.
64    ///
65    /// Corresponds to `st_atime`.
66    pub last_access_at: Duration,
67
68    /// The timestamp of the last modification to the inode's content.
69    ///
70    /// Corresponds to `st_mtime`.
71    pub last_modify_at: Duration,
72
73    /// The timestamp of the last change to the inode's metadata.
74    ///
75    /// This is updated when permissions, ownership, or link count change,
76    /// not just when the inode content is modified.
77    ///
78    /// Corresponds to `st_ctime`.
79    pub last_meta_change_at: Duration,
80
81    /// The type of the inode (e.g., regular file, directory, symlink).
82    ///
83    /// Derived from the file type bits of `st_mode` (using the `S_IFMT` mask).
84    pub type_: InodeType,
85
86    /// The inode mode, representing access permissions.
87    ///
88    /// Derived from the permission bits of `st_mode`.
89    pub mode: InodeMode,

91    /// The number of hard links pointing to this inode.
92    ///
93    /// Corresponds to `st_nlink`.
94    pub nr_hard_links: usize,
95
96    /// The User ID (UID) of the inode's owner.
97    ///
98    /// Corresponds to `st_uid`.
99    pub uid: Uid,
100
101    /// The Group ID (GID) of the inode's owner.
102    ///
103    /// Corresponds to `st_gid`.
104    pub gid: Gid,
105
106    /// The ID of the device containing the inode.
107    ///
108    /// For persisted files, this device could be an on-disk partition
109    /// or a logical volume (with RAID).
110    /// For pseudo files (e.g., those on sockfs), this device is also "pseudo".
111    ///
112    /// Corresponds to `st_dev`.
113    pub container_dev_id: DeviceId,
114
115    /// The device ID of the inode itself, if this inode represents a
116    /// special device file (character or block).
117    ///
118    /// Corresponds to `st_rdev`.

236bitflags! {
237    /// Policy that controls when the VFS revalidates cached inode entries.
238    ///
239    /// # Why revalidation is needed
240    ///
241    /// Normally, inode creation and deletion go through VFS operations such as
242    /// `create`, `unlink`, `mkdir`, and `rmdir`. This lets the VFS keep its
243    /// dentry cache consistent with the underlying filesystem.
244    ///
245    /// However, some filesystems create or delete inodes without going through
246    /// the VFS. For example, procfs entries under `/proc` appear and disappear
247    /// as processes are forked and reaped. Networked filesystems may also be
248    /// modified by remote peers. For such filesystems, the VFS must revalidate
249    /// cached entries by asking the parent directory whether a cached child is
250    /// still valid.
251    ///
252    /// # Revalidation protocol
253    ///
254    /// A directory inode declares its policy through
255    /// [`Inode::revalidation_policy`].
256    ///
257    /// - [`REVALIDATE_EXISTS`](Self::REVALIDATE_EXISTS) makes the VFS call
258    ///   [`Inode::revalidate_exists`] on positive cache hits. If it returns
259    ///   `false`, the cached entry is dropped and the lookup is retried.
260    /// - [`REVALIDATE_ABSENT`](Self::REVALIDATE_ABSENT) makes the VFS call
261    ///   [`Inode::revalidate_absent`] on negative cache hits. If it returns
262    ///   `false`, the cached entry is dropped and the lookup is retried.
263    ///
264    /// If neither flag is set, cached entries are trusted unconditionally. The
265    /// policy must be fixed for each inode. Non-directory inodes should return
266    /// an empty policy.
267    ///
268    /// # Performance
269    ///
270    /// The revalidation callbacks run on every matching cache hit, so
271    /// filesystems should keep them cheap.
272    pub struct RevalidationPolicy: u8 {
273        /// Revalidate positive cache entries (cached child inodes).
274        ///
275        /// Set this when the directory may spontaneously
276        /// _lose_ children without going through VFS `unlink`/`rmdir`.
277        /// For example, procfs sets this on `/proc`
278        /// because PID directories disappear when processes exit.
279        ///
280        /// When set, every positive cache hit calls
281        /// [`Inode::revalidate_exists`] on the parent directory.
282        const REVALIDATE_EXISTS = 1 << 0;
283
284        /// Revalidate negative cache entries (names known to be absent).
285        ///
286        /// Set this when the directory may spontaneously _gain_ children
287        /// without going through VFS `create`/`mkdir`/`link`.
288        /// For example, procfs sets this on `/proc`
289        /// because PID directories appear when processes are forked.
290        ///
291        /// When set, every negative cache hit calls
292        /// [`Inode::revalidate_absent`] on the parent directory.
293        const REVALIDATE_ABSENT = 1 << 1;
294    }
295}

506    /// Used to check for read/write/execute permissions on a file.
507    ///
508    /// Similar to Linux, using "fsuid" here allows setting filesystem permissions
509    /// without changing the "normal" uids for other tasks.
510    fn check_permission(&self, mut perm: Permission) -> Result<()> {
511        let creds = match Task::current() {
512            Some(task) => match task.as_posix_thread() {
513                Some(thread) => thread.credentials(),
514                None => return Ok(()),
515            },
516            None => return Ok(()),
517        };
518
519        // With DAC_OVERRIDE capability, the user can bypass some permission checks.
520        if creds.effective_capset().contains(CapSet::DAC_OVERRIDE) {
521            // Read/write DACs are always overridable.
522            perm -= Permission::MAY_READ | Permission::MAY_WRITE;
523
524            // Executable DACs are overridable when there is at least one exec bit set.
525            if perm.may_exec() {
526                let metadata = self.metadata();
527                let mode = metadata.mode;
528
529                if mode.is_owner_executable()
530                    || mode.is_group_executable()
531                    || mode.is_other_executable()
532                {
533                    perm -= Permission::MAY_EXEC;
534                } else {
535                    return_errno_with_message!(
536                        Errno::EACCES,
537                        "root execute permission denied: no execute bits set"
538                    );
539                }
540            }
541        }
542
543        perm =
544            perm.intersection(Permission::MAY_READ | Permission::MAY_WRITE | Permission::MAY_EXEC);
545        let metadata = self.metadata();
546        let mode = metadata.mode;
547
548        if metadata.uid == creds.fsuid() {
549            if (perm.may_read() && !mode.is_owner_readable())
550                || (perm.may_write() && !mode.is_owner_writable())
551                || (perm.may_exec() && !mode.is_owner_executable())
552            {
553                return_errno_with_message!(Errno::EACCES, "owner permission check failed");
554            }
555        } else if metadata.gid == creds.fsgid() {
556            if (perm.may_read() && !mode.is_group_readable())
557                || (perm.may_write() && !mode.is_group_writable())
558                || (perm.may_exec() && !mode.is_group_executable())
559            {
560                return_errno_with_message!(Errno::EACCES, "group permission check failed");
561            }
562        } else if (perm.may_read() && !mode.is_other_readable())
563            || (perm.may_write() && !mode.is_other_writable())
564            || (perm.may_exec() && !mode.is_other_executable())
565        {
566            return_errno_with_message!(Errno::EACCES, "other permission check failed");
567        }
568
569        Ok(())
570    }

597pub struct InodeWriter<'a> {
598    inner: &'a dyn Inode,
599    offset: usize,
600}
601
602impl Write for InodeWriter<'_> {
603    #[inline]
604    fn write(&mut self, buf: &[u8]) -> IoResult<usize> {
605        let mut reader = VmReader::from(buf).to_fallible();
606        let write_len = self
607            .inner
608            .write_at(self.offset, &mut reader, StatusFlags::empty())
609            .map_err(|_| IoError::new(IoErrorKind::WriteZero, "failed to write buffer"))?;
610        self.offset += write_len;
611        Ok(write_len)
612    }
613
614    #[inline]
615    fn flush(&mut self) -> IoResult<()> {
616        Ok(())
617    }
618}

675/// Represents the various operation modes for fallocate.
676///
677/// Each mode determines whether the target disk space within a file
678/// will be allocated, deallocated, or zeroed, among other operations.
679#[derive(Clone, Copy, Debug, Eq, PartialEq)]
680pub enum FallocMode {
681    /// Allocates disk space within the range specified.
682    Allocate,
683    /// Like `Allocate`, but does not change the file size.
684    AllocateKeepSize,
685    /// Makes shared file data extents private to guarantee subsequent writes.
686    AllocateUnshareRange,
687    /// Deallocates space (creates a hole) while keeping the file size unchanged.
688    PunchHoleKeepSize,
689    /// Converts a file range to zeros, expanding the file if necessary.
690    ZeroRange,
691    /// Like `ZeroRange`, but does not change the file size.
692    ZeroRangeKeepSize,
