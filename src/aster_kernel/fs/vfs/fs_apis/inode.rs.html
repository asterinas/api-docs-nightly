31pub struct Metadata {
32    /// The inode number, which uniquely identifies the file within the filesystem.
33    ///
34    /// Corresponds to `st_ino`.
35    pub ino: u64,
36
37    /// The size of the inode.
38    ///
39    /// The interpretation depends on `inode_type`:
40    /// - **Regular File**: The total length of the file content.
41    /// - **Directory**: The size of the directory's internal table (usually a multiple of block size).
42    /// - **Symbolic Link**: The length of the target pathname.
43    /// - **Device/Socket/FIFO**: Usually zero.
44    ///
45    /// Corresponds to `st_size`.
46    pub size: usize,
47
48    /// The optimal block size for filesystem I/O operations.
49    ///
50    /// Corresponds to `st_blksize`.
51    pub optimal_block_size: usize,
52
53    /// The number of 512-byte sectors allocated for the inode on disk.
54    ///
55    /// This represents physical usage.
56    /// For sparse files (those having holes), `size` is greater than this field.
57    /// For files with preallocated blocks (`FALLOC_FL_KEEP_SIZE`),
58    /// `size` is smaller than this field.
59    ///
60    /// Corresponds to `st_blocks`.
61    pub nr_sectors_allocated: usize,
62
63    /// The timestamp of the last access to the inode's data.
64    ///
65    /// Corresponds to `st_atime`.
66    pub last_access_at: Duration,
67
68    /// The timestamp of the last modification to the inode's content.
69    ///
70    /// Corresponds to `st_mtime`.
71    pub last_modify_at: Duration,
72
73    /// The timestamp of the last change to the inode's metadata.
74    ///
75    /// This is updated when permissions, ownership, or link count change,
76    /// not just when the inode content is modified.
77    ///
78    /// Corresponds to `st_ctime`.
79    pub last_meta_change_at: Duration,
80
81    /// The type of the inode (e.g., regular file, directory, symlink).
82    ///
83    /// Derived from the file type bits of `st_mode` (using the `S_IFMT` mask).
84    pub type_: InodeType,
85
86    /// The inode mode, representing access permissions.
87    ///
88    /// Derived from the permission bits of `st_mode`.
89    pub mode: InodeMode,
90
91    /// The number of hard links pointing to this inode.
92    ///
93    /// Corresponds to `st_nlink`.
94    pub nr_hard_links: usize,
95
96    /// The User ID (UID) of the inode's owner.
97    ///
98    /// Corresponds to `st_uid`.
99    pub uid: Uid,
100
101    /// The Group ID (GID) of the inode's owner.
102    ///
103    /// Corresponds to `st_gid`.
104    pub gid: Gid,
105
106    /// The ID of the device containing the inode.
107    ///
108    /// For persisted files, this device could be an on-disk partition
109    /// or a logical volume (with RAID).
110    /// For pseudo files (e.g., those on sockfs), this device is also "pseudo".
111    ///
112    /// Corresponds to `st_dev`.
113    pub container_dev_id: DeviceId,
114
115    /// The device ID of the inode itself, if this inode represents a
116    /// special device file (character or block).
117    ///
118    /// Corresponds to `st_rdev`.
119    pub self_dev_id: Option<DeviceId>,
120}

236/// I/O operations in an [`Inode`].
237///
238/// This abstracts the common I/O operations used by both [`Inode`] (for regular files) and
239/// [`FileIo`] (for special files).

244        offset: usize,
245        writer: &mut VmWriter,
246        status_flags: StatusFlags,
247    ) -> Result<usize>;
248
249    /// Writes data from the given `VmReader` into the file.
250    fn write_at(
251        &self,
252        offset: usize,
253        reader: &mut VmReader,
254        status_flags: StatusFlags,
255    ) -> Result<usize>;
256}

361    /// Returns whether a VFS dentry for this inode should be put into the dentry cache.
362    ///
363    /// The dentry cache in the VFS layer can accelerate the lookup of inodes. So usually,
364    /// it is preferable to use the dentry cache. And thus, the default return value of this method
365    /// is `true`.
366    ///
367    /// But this caching can raise consistency issues in certain use cases. Specifically, the dentry
368    /// cache works on the assumption that all FS operations go through the dentry layer first.
369    /// This is why the dentry cache can reflect the up-to-date FS state. Yet, this assumption
370    /// may be broken. If the inodes of a file system may "disappear" without unlinking through the
371    /// VFS layer, then their dentries should not be cached. For example, an inode in procfs
372    /// (say, `/proc/1/fd/2`) can "disappear" without notice from the perspective of the dentry cache.
373    /// So for such inodes, they are incompatible with the dentry cache. And this method returns `false`.
374    ///
375    /// Note that if any ancestor directory of an inode has this method returns `false`, then
376    /// this inode would not be cached by the dentry cache, even when the method of this
377    /// inode returns `true`.
378    fn is_dentry_cacheable(&self) -> bool {
379        true
380    }

422    /// Used to check for read/write/execute permissions on a file.
423    ///
424    /// Similar to Linux, using "fsuid" here allows setting filesystem permissions
425    /// without changing the "normal" uids for other tasks.
426    fn check_permission(&self, mut perm: Permission) -> Result<()> {
427        let creds = match Task::current() {
428            Some(task) => match task.as_posix_thread() {
429                Some(thread) => thread.credentials(),
430                None => return Ok(()),
431            },
432            None => return Ok(()),
433        };
434
435        // With DAC_OVERRIDE capability, the user can bypass some permission checks.
436        if creds.effective_capset().contains(CapSet::DAC_OVERRIDE) {
437            // Read/write DACs are always overridable.
438            perm -= Permission::MAY_READ | Permission::MAY_WRITE;
439
440            // Executable DACs are overridable when there is at least one exec bit set.
441            if perm.may_exec() {
442                let metadata = self.metadata();
443                let mode = metadata.mode;
444
445                if mode.is_owner_executable()
446                    || mode.is_group_executable()
447                    || mode.is_other_executable()
448                {
449                    perm -= Permission::MAY_EXEC;
450                } else {
451                    return_errno_with_message!(
452                        Errno::EACCES,
453                        "root execute permission denied: no execute bits set"
454                    );
455                }
456            }
457        }
458
459        perm =
460            perm.intersection(Permission::MAY_READ | Permission::MAY_WRITE | Permission::MAY_EXEC);
461        let metadata = self.metadata();
462        let mode = metadata.mode;
463
464        if metadata.uid == creds.fsuid() {
465            if (perm.may_read() && !mode.is_owner_readable())
466                || (perm.may_write() && !mode.is_owner_writable())
467                || (perm.may_exec() && !mode.is_owner_executable())
468            {
469                return_errno_with_message!(Errno::EACCES, "owner permission check failed");
470            }
471        } else if metadata.gid == creds.fsgid() {
472            if (perm.may_read() && !mode.is_group_readable())
473                || (perm.may_write() && !mode.is_group_writable())
474                || (perm.may_exec() && !mode.is_group_executable())
475            {
476                return_errno_with_message!(Errno::EACCES, "group permission check failed");
477            }
478        } else if (perm.may_read() && !mode.is_other_readable())
479            || (perm.may_write() && !mode.is_other_writable())
480            || (perm.may_exec() && !mode.is_other_executable())
481        {
482            return_errno_with_message!(Errno::EACCES, "other permission check failed");
483        }
484
485        Ok(())
486    }
487}

513pub struct InodeWriter<'a> {
514    inner: &'a dyn Inode,
515    offset: usize,
516}
517
518impl Write for InodeWriter<'_> {
519    #[inline]
520    fn write(&mut self, buf: &[u8]) -> IoResult<usize> {
521        let mut reader = VmReader::from(buf).to_fallible();
522        let write_len = self
523            .inner
524            .write_at(self.offset, &mut reader, StatusFlags::empty())
525            .map_err(|_| IoError::new(IoErrorKind::WriteZero, "failed to write buffer"))?;
526        self.offset += write_len;
527        Ok(write_len)
528    }
529
530    #[inline]
531    fn flush(&mut self) -> IoResult<()> {
532        Ok(())
533    }
534}

545/// An extension is a set of object groups that is attached to an inode.
546///
547/// In this structure, we do not specify the exact type, but instead use [`Any`], which makes the
548/// FS types (e.g., [`Inode`]) independent of the kernel types. This allows the file system
549/// implementation to exist outside the kernel.
550#[derive(Debug)]
551pub struct Extension {
552    group1: Once<ThinBox<dyn Any + Send + Sync>>,
553    group2: Once<ThinBox<dyn Any + Send + Sync>>,
554}
555

557    /// Creates a new, empty extension.
558    pub fn new() -> Self {

562        }
563    }
564
565    /// Gets the first extension group.
566    pub fn group1(&self) -> &Once<ThinBox<dyn Any + Send + Sync>> {
567        &self.group1
568    }
569
570    /// Gets the second extension group.
571    pub fn group2(&self) -> &Once<ThinBox<dyn Any + Send + Sync>> {
572        &self.group2
573    }

591/// Represents the various operation modes for fallocate.
592///
593/// Each mode determines whether the target disk space within a file
594/// will be allocated, deallocated, or zeroed, among other operations.
595#[derive(Clone, Copy, Debug, Eq, PartialEq)]
596pub enum FallocMode {
597    /// Allocates disk space within the range specified.
598    Allocate,
599    /// Like `Allocate`, but does not change the file size.
600    AllocateKeepSize,
601    /// Makes shared file data extents private to guarantee subsequent writes.
602    AllocateUnshareRange,
603    /// Deallocates space (creates a hole) while keeping the file size unchanged.
604    PunchHoleKeepSize,
605    /// Converts a file range to zeros, expanding the file if necessary.
606    ZeroRange,
607    /// Like `ZeroRange`, but does not change the file size.
608    ZeroRangeKeepSize,
609    /// Removes a range of bytes without leaving a hole.
610    CollapseRange,
611    /// Inserts space within a file without overwriting existing data.
612    InsertRange,
613}
