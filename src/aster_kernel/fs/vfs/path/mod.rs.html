41/// A `Path` is used to represent an exact location in the VFS tree.
42///
43/// Each `Path` corresponds to a node in the VFS tree, and a single node
44/// may have multiple `Path` instances referencing it due to mount operations.
45#[derive(Clone, Debug)]
46pub struct Path {
47    mount: Arc<Mount>,
48    dentry: Arc<Dentry>,
49}
50
51impl PartialEq for Path {
52    fn eq(&self, other: &Self) -> bool {
53        self.mount.id() == other.mount.id() && Arc::ptr_eq(&self.dentry, &other.dentry)
54    }
55}

66    /// Creates a new `Path` to represent the child directory of a file system.
67    pub fn new_fs_child(&self, name: &str, type_: InodeType, mode: InodeMode) -> Result<Self> {
68        let dir_dentry = self.dentry.as_dir_dentry_or_err()?;

74    /// Creates a new `Path` to represent an unnamed temporary file.
75    ///
76    /// The returned inode has no directory entry and is invisible to `readdir`.
77    /// If created hard-linkable, it may later be given a name via a link
78    /// operation; otherwise it can never be linked.
79    pub fn create_tmpfile(
80        &self,
81        mode: InodeMode,
82        hard_linkability: HardLinkability,
83    ) -> Result<Self> {
84        let dir_dentry = self.dentry.as_dir_dentry_or_err()?;
85        self.check_dir_entry_mutation()?;
86        let tmp_inode = self.inode().create_tmpfile(mode, hard_linkability)?;
87        let tmp_dentry = Dentry::new_anonymous(tmp_inode, &dir_dentry);
88        Ok(Self::new(self.mount.clone(), tmp_dentry))
89    }

120    /// Opens the `Path` with the given `OpenArgs`.
121    ///
122    /// Returns an `InodeHandle` on success.
123    pub fn open(&self, open_args: OpenArgs) -> Result<InodeHandle> {
124        let inode = self.inode().as_ref();
125        let inode_type = inode.type_();
126        let creation_flags = &open_args.creation_flags;
127        let status_flags = &open_args.status_flags;
128
129        if creation_flags.contains(CreationFlags::O_CREAT)
130            && creation_flags.contains(CreationFlags::O_EXCL)
131        {
132            return_errno_with_message!(Errno::EEXIST, "the file already exists");
133        }
134        if creation_flags.contains(CreationFlags::O_CREAT) && inode_type == InodeType::Dir {
135            return_errno_with_message!(
136                Errno::EISDIR,
137                "O_CREAT is specified but the file is a directory"
138            );
139        }
140
141        if inode_type == InodeType::SymLink
142            && creation_flags.contains(CreationFlags::O_NOFOLLOW)
143            && !status_flags.contains(StatusFlags::O_PATH)
144        {
145            return_errno_with_message!(Errno::ELOOP, "the file is a symlink");
146        }
147        if creation_flags.contains(CreationFlags::O_DIRECTORY) && inode_type != InodeType::Dir {
148            return_errno_with_message!(
149                Errno::ENOTDIR,
150                "O_DIRECTORY is specified but the file is not a directory"
151            );
152        }
153
154        if inode_type.is_regular_file()
155            && creation_flags.contains(CreationFlags::O_TRUNC)
156            && !status_flags.contains(StatusFlags::O_PATH)
157        {
158            self.resize(0)?;
159        }
160
161        InodeHandle::new(self.clone(), open_args.access_mode, *status_flags)
162    }

164    /// Gets the parent `Path` within the same mount.
165    ///
166    /// This method returns the parent path within the same filesystem/mount.
167    /// It does NOT cross mount boundaries. If the current path is the root of a mount,
168    /// it will return `None`.
169    ///
170    /// For cross-filesystem parent lookup, use `effective_parent()` instead.
171    pub(super) fn parent_within_mount(&self) -> Option<Self> {
172        let parent = self.dentry.parent()?;
173        Some(Self::new(self.mount.clone(), parent))
174    }

176    /// Gets the top `Path` of the current.
177    ///
178    /// Used when different file systems are mounted on the same mount point.
179    ///
180    /// For example, first `mount /dev/sda1 /mnt` and then `mount /dev/sda2 /mnt`.
181    /// After the second mount is completed, the content of the first mount will be overridden.
182    /// We need to recursively obtain the top `Path`.
183    pub fn get_top_path(mut self) -> Self {
184        while self.dentry.is_mountpoint() {
185            if let Some(child_mount) = self.mount.get(&self.dentry) {
186                let inner = child_mount.root_dentry().clone();
187                self = Self::new(child_mount, inner);
188            } else {
189                break;
190            }
191        }
192
193        self
194    }

211    /// Checks if this path is reachable from the given `root` path.
212    ///
213    /// A path is considered reachable if it is the same as or a descendant
214    /// of the `root` path. The check traverses upwards from the current path,
215    /// crossing mount point boundaries as necessary, until it either finds
216    /// the `root` path or reaches the global root.
217    fn is_reachable_from(&self, root: &Path) -> bool {
218        let mut owned;
219        let mut current = self;
220
221        loop {
222            if current.mount.id() == root.mount.id() {
223                return current.dentry.is_equal_or_descendant_of(&root.dentry);
224            }
225
226            let Some(parent_mount) = current.mount.parent().and_then(|mount| mount.upgrade())
227            else {
228                // `upgrade()` may fail if `current` is in a mount namespace that has been
229                // dropped. In this situation, `current` is not reachable from `root` because
230                // `root` holds strong reference counts for all its descendant mounts.
231                return false;
232            };
233
234            owned = Path::new(
235                parent_mount,
236                current
237                    .mount
238                    .mountpoint()
239                    .expect("Mounts with parents must have a mount point")
240                    .clone(),
241            );
242            current = &owned;
243        }
244    }

255    /// Checks whether the path is on a writable mount and filesystem.
256    fn check_mount_writable(&self) -> Result<()> {
257        if self.mount.flags().contains(PerMountFlags::RDONLY)
258            || self.fs().flags().contains(FsFlags::RDONLY)
259        {
260            return_errno_with_message!(Errno::EROFS, "the mount or filesystem is read-only");
261        }
262
263        Ok(())
264    }
265
266    /// Checks whether a directory entry may be modified in this directory.
267    fn check_dir_entry_mutation(&self) -> Result<()> {
268        self.check_mount_writable()?;
269        self.inode()
270            .check_permission(Permission::MAY_WRITE | Permission::MAY_EXEC)
271    }

273    /// Checks whether an inode may be used as the source of a hard link.
274    fn check_hardlink_source(&self) -> Result<()> {
275        let current_thread = current_thread!();
276        let Some(posix_thread) = current_thread.as_posix_thread() else {
277            return Ok(());
278        };
279
280        let metadata = self.metadata();
281        let fsuid = posix_thread.credentials().fsuid();
282
283        // The source inode owner can hardlink all they like.
284        if fsuid == metadata.uid {
285            return Ok(());
286        }
287
288        // Hardlinking to unreadable or unwritable sources is dangerous.
289        if self
290            .inode()
291            .check_permission(Permission::MAY_READ | Permission::MAY_WRITE)
292            .is_err()
293        {
294            return_errno_with_message!(
295                Errno::EPERM,
296                "the source is not permitted to be hard-linked"
297            );
298        }
299
300        // Non-owners may only hard-link regular files.
301        if self.type_() != InodeType::File {
302            return_errno_with_message!(
303                Errno::EPERM,
304                "the source is not permitted to be hard-linked"
305            );
306        }
307
308        // Hardlinking to setuid or executable setgid files requires CAP_FOWNER.
309        if (metadata.mode.has_set_uid()
310            || (metadata.mode.has_set_gid() && metadata.mode.is_group_executable()))
311            && lsm_hooks::on_capable(lsm_hooks::CapableContext::new(
312                UserNamespace::get_init_singleton().as_ref(),
313                posix_thread,
314                CapSet::FOWNER,
315            ))
316            .is_err()
317        {
318            return_errno_with_message!(
319                Errno::EPERM,
320                "the source is not permitted to be hard-linked"
321            );
322        }
323
324        Ok(())
325    }

335impl Path {
336    /// Mounts a filesystem at the current path.
337    ///
338    /// This method attaches a given filesystem to the VFS tree at the location
339    /// represented by `self`. The current path becomes the mountpoint for the new
340    /// filesystem.
341    ///
342    /// Returns the newly created child mount on success.
343    ///
344    /// # Errors
345    ///
346    /// Returns `ENOTDIR` if the path is not a directory.
347    /// Returns `EINVAL` if attempting to mount on root or if the path is not
348    /// in the current mount namespace.
349    pub fn mount(
350        &self,
351        fs: Arc<dyn FileSystem>,
352        flags: PerMountFlags,
353        source: Option<String>,
354        ctx: &Context,
355    ) -> Result<Arc<Mount>> {
356        if self.type_() != InodeType::Dir {
357            return_errno_with_message!(Errno::ENOTDIR, "the path is not a directory");
358        }
359
360        let fs_ref = ctx.thread_local.borrow_fs();
361        let path_resolver = fs_ref.resolver().read();
362
363        if path_resolver.root() == self {
364            return_errno_with_message!(Errno::EINVAL, "the root cannot be mounted on");
365        }
366
367        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
368        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
369        if !current_mnt_ns.owns(&self.mount) {
370            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
371        }
372
373        let child_mount = self.mount.do_mount(fs, flags, &self.dentry, source)?;
374
375        Ok(child_mount)
376    }
377
378    /// Unmounts the filesystem mounted at the current path.
379    ///
380    /// Returns the unmounted child mount on success.
381    ///
382    /// # Errors
383    ///
384    /// Returns `EINVAL` in the following cases:
385    /// - The current path is not a mount root.
386    /// - The mount of the current path is the root mount.
387    /// - The current path is not in the current mount namespace.
388    pub fn unmount(&self, ctx: &Context) -> Result<Arc<Mount>> {
389        if !self.is_mount_root() {
390            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
391        }
392
393        let Some(mountpoint) = self.mount.mountpoint() else {
394            return_errno_with_message!(Errno::EINVAL, "the root mount cannot be unmounted");
395        };
396
397        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
398        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
399        if !current_mnt_ns.owns(&self.mount) {
400            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
401        }
402
403        self.mount.sync()?;
404
405        let parent_mount = self.mount.parent().unwrap().upgrade().unwrap();
406        let child_mount = parent_mount.do_unmount(&mountpoint)?;
407
408        Ok(child_mount)
409    }

412    ///
413    /// If `fs_flags` is provided, it will update the flags of the mounted filesystem,
414    /// otherwise, only the flags of the current mount will be updated.

416    /// # Errors
417    ///
418    /// Returns `EINVAL` in the following cases:
419    /// - The current path is not a mount root.
420    /// - The current path is not in the current mount namespace.
421    pub fn remount(
422        &self,
423        mount_flags: PerMountFlags,
424        fs_flags: Option<FsFlags>,
425        data: Option<CString>,
426        ctx: &Context,
427    ) -> Result<()> {
428        if !self.is_mount_root() {
429            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
430        };
431
432        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
433        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
434        if !current_mnt_ns.owns(&self.mount) {
435            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
436        }
437
438        self.mount.remount(mount_flags, fs_flags, data, ctx)
439    }

441    /// Creates a bind mount from the current path to the destination path.
442    ///
443    /// Creates a new mount tree that mirrors either the root mount (non-recursive)
444    /// or the entire mount subtree (recursive), and attaches it to the destination path.
445    ///
446    /// # Errors
447    ///
448    /// Returns `ENOTDIR` if one of the source and destination is a directory
449    /// and the other is not.
450    ///
451    /// Returns `EINVAL` if any of the following holds:
452    /// - The destination path is not in the current mount namespace.
453    /// - The source path is a mount namespace file that would create a namespace loop.
454    pub fn bind_mount_to(&self, dst_path: &Self, recursive: bool, ctx: &Context) -> Result<()> {
455        let can_bind = {
456            let src_is_dir = self.type_() == InodeType::Dir;
457            let dst_is_dir = dst_path.type_() == InodeType::Dir;
458            (src_is_dir && dst_is_dir) || (!src_is_dir && !dst_is_dir)
459        };
460        if !can_bind {
461            return_errno_with_message!(
462                Errno::ENOTDIR,
463                "the source and destination must both be directories or both be non-directories"
464            );
465        }
466
467        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
468        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
469
470        // Linux does not require the source path to belong to the current
471        // mount namespace, but still rejects source mount namespace files that
472        // would form loops.
473        if !current_mnt_ns.owns(&dst_path.mount) {
474            return_errno_with_message!(
475                Errno::EINVAL,
476                "the destination path is not in this mount namespace"
477            );
478        }
479        if current_mnt_ns.would_form_mnt_ns_loop(&self.dentry) {
480            return_errno_with_message!(
481                Errno::EINVAL,
482                "the source is a mount namespace file that would create a namespace loop"
483            );
484        }
485
486        let current_mnt_ns_weak = Arc::downgrade(current_mnt_ns);
487        let new_mount = self.mount.clone_mount_tree(
488            &self.dentry,
489            &current_mnt_ns_weak,
490            recursive,
491            MountNsFileCopying::Copy,
492        )?;
493        new_mount.graft_mount_tree(dst_path);
494        Ok(())
495    }
496
497    /// Moves a mount tree from the current path to the destination path.
498    ///
499    /// # Errors
500    ///
501    /// Returns `ENOTDIR` if the `dst_path` is not a directory.
502    ///
503    /// Returns `EINVAL` in the following cases:
504    /// - The current path is not a mount root.
505    /// - The mount of the current path is the root mount.
506    /// - Either source or destination path is not in the current mount namespace.
507    ///
508    /// Returns `ELOOP` in the following cases:
509    /// - The destination path is inside the subtree being moved.
510    /// - The mount tree contains a mount namespace file that would create a namespace loop.
511    pub fn move_mount_to(&self, dst_path: &Self, ctx: &Context) -> Result<()> {
512        if !self.is_mount_root() {
513            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
514        };
515        if self.mount_node().parent().is_none() {
516            return_errno_with_message!(Errno::EINVAL, "the root mount can not be moved");
517        }
518
519        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
520        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
521        if !current_mnt_ns.owns(&self.mount) {
522            return_errno_with_message!(
523                Errno::EINVAL,
524                "the source path is not in this mount namespace"
525            );
526        }
527        if !current_mnt_ns.owns(&dst_path.mount) {
528            return_errno_with_message!(
529                Errno::EINVAL,
530                "the destination path is not in this mount namespace"
531            );
532        }
533        current_mnt_ns.check_no_mnt_ns_loop_in_tree(self.mount_node())?;
534        if dst_path
535            .mount_node()
536            .is_equal_or_descendant_of(self.mount_node())
537        {
538            // Reject moves that would place a mount beneath itself, because the mount tree
539            // must remain acyclic.
540            return_errno_with_message!(
541                Errno::ELOOP,
542                "the destination path is inside the mount subtree being moved"
543            );
544        }
545
546        self.mount.graft_mount_tree(dst_path);
547
548        Ok(())
549    }
550
551    /// Sets the propagation type of the mount of this `Path`.
552    pub fn set_mount_propagation(
553        &self,
554        prop: MountPropType,
555        recursive: bool,
556        ctx: &Context,
557    ) -> Result<()> {
558        if !self.is_mount_root() {
559            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
560        };
561
562        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
563        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
564        if !current_mnt_ns.owns(&self.mount) {
565            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
566        }
567
568        self.mount.set_propagation(prop, recursive);
569
570        Ok(())
571    }

581    /// Creates a `Path` by making an inode of the `type_` with the `mode`.
582    pub fn mknod(&self, name: &str, mode: InodeMode, type_: MknodType) -> Result<Self> {
583        let dir_dentry = self.dentry.as_dir_dentry_or_err()?;

585        let inner = dir_dentry.mknod(name, mode, type_)?;

589    /// Links a new name for the `Path`.
590    pub fn link(&self, old: &Self, name: &str) -> Result<()> {
591        if !Arc::ptr_eq(&old.mount, &self.mount) {
592            return_errno_with_message!(Errno::EXDEV, "the operation cannot cross mounts");
593        }
594
595        let dir_dentry = self.dentry.as_dir_dentry_or_err()?;
596        old.check_hardlink_source()?;
597        self.check_dir_entry_mutation()?;
598        dir_dentry.link(old.inode(), name)
599    }
600
601    /// Unlinks a name from the `Path`.
602    pub fn unlink(&self, name: &str) -> Result<()> {
603        let dir_dentry = self.dentry.as_dir_dentry_or_err()?;
604        self.check_dir_entry_mutation()?;
605        dir_dentry.unlink(name)
606    }
607
608    /// Removes a directory by `rmdir()` the inner inode.
609    pub fn rmdir(&self, name: &str) -> Result<()> {
610        let dir_dentry = self.dentry.as_dir_dentry_or_err()?;
611        self.check_dir_entry_mutation()?;
612        dir_dentry.rmdir(name)
613    }
614
615    /// Renames a `Path` to the new `Path` by `rename()` the inner inode.
616    pub fn rename(&self, old_name: &str, new_dir: &Self, new_name: &str) -> Result<()> {
617        if !Arc::ptr_eq(&self.mount, &new_dir.mount) {
618            return_errno_with_message!(Errno::EXDEV, "the operation cannot cross mounts");
619        }
620
621        let old_dir_dentry = self.dentry.as_dir_dentry_or_err()?;
622        let new_dir_dentry = new_dir.dentry.as_dir_dentry_or_err()?;
623
624        self.check_dir_entry_mutation()?;
625        if !Arc::ptr_eq(&self.dentry, &new_dir.dentry) {
626            new_dir.check_dir_entry_mutation()?;
627        }
628
629        old_dir_dentry.rename(old_name, &new_dir_dentry, new_name)
630    }

675/// Checks if the file name is ".", indicating it's the current directory.
676pub const fn is_dot(filename: &str) -> bool {
677    let name_bytes = filename.as_bytes();
678    name_bytes.len() == 1 && name_bytes[0] == DOT_BYTE
679}
680
681/// Checks if the file name is "..", indicating it's the parent directory.
682pub const fn is_dotdot(filename: &str) -> bool {
683    let name_bytes = filename.as_bytes();
684    name_bytes.len() == 2 && name_bytes[0] == DOT_BYTE && name_bytes[1] == DOT_BYTE
685}
686
687/// Checks if the file name is "." or "..", indicating it's the current or parent directory.
688pub const fn is_dot_or_dotdot(filename: &str) -> bool {
689    let name_bytes = filename.as_bytes();
690    let name_len = name_bytes.len();
691    if name_len == 1 {
692        name_bytes[0] == DOT_BYTE
693    } else if name_len == 2 {
694        name_bytes[0] == DOT_BYTE && name_bytes[1] == DOT_BYTE
695    } else {
696        false
697    }
698}
