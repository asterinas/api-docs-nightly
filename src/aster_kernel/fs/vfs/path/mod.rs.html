38/// A `Path` is used to represent an exact location in the VFS tree.
39///
40/// Each `Path` corresponds to a node in the VFS tree, and a single node
41/// may have multiple `Path` instances referencing it due to mount operations.
42#[derive(Clone, Debug)]
43pub struct Path {
44    mount: Arc<Mount>,
45    dentry: Arc<Dentry>,
46}
47
48impl PartialEq for Path {
49    fn eq(&self, other: &Self) -> bool {
50        self.mount.id() == other.mount.id() && Arc::ptr_eq(&self.dentry, &other.dentry)
51    }
52}
53
54impl Eq for Path {}

56impl Path {
57    /// Creates a new `Path` to represent the root directory of a file system.
58    pub fn new_fs_root(mount: Arc<Mount>) -> Self {
59        let inner = mount.root_dentry().clone();
60        Self::new(mount, inner)
61    }
62
63    /// Creates a new `Path` to represent the child directory of a file system.
64    pub fn new_fs_child(&self, name: &str, type_: InodeType, mode: InodeMode) -> Result<Self> {
65        if self
66            .inode()
67            .check_permission(Permission::MAY_WRITE)
68            .is_err()
69        {
70            return_errno!(Errno::EACCES);
71        }
72        let new_child_dentry = self
73            .dentry
74            .as_dir_dentry_or_err()?
75            .create(name, type_, mode)?;
76        Ok(Self::new(self.mount.clone(), new_child_dentry))
77    }

83        name_fn: fn(&dyn Inode) -> String,

85        let dentry = Dentry::new_pseudo(inode, name_fn);

108    /// Opens the `Path` with the given `OpenArgs`.
109    ///
110    /// Returns an `InodeHandle` on success.
111    pub fn open(&self, open_args: OpenArgs) -> Result<InodeHandle> {
112        let inode = self.inode().as_ref();
113        let inode_type = inode.type_();
114        let creation_flags = &open_args.creation_flags;
115        let status_flags = &open_args.status_flags;
116
117        if creation_flags.contains(CreationFlags::O_CREAT)
118            && creation_flags.contains(CreationFlags::O_EXCL)
119        {
120            return_errno_with_message!(Errno::EEXIST, "the file already exists");
121        }
122        if creation_flags.contains(CreationFlags::O_CREAT) && inode_type == InodeType::Dir {
123            return_errno_with_message!(
124                Errno::EISDIR,
125                "O_CREAT is specified but the file is a directory"
126            );
127        }
128
129        if inode_type == InodeType::SymLink
130            && creation_flags.contains(CreationFlags::O_NOFOLLOW)
131            && !status_flags.contains(StatusFlags::O_PATH)
132        {
133            return_errno_with_message!(Errno::ELOOP, "the file is a symlink");
134        }
135        if creation_flags.contains(CreationFlags::O_DIRECTORY) && inode_type != InodeType::Dir {
136            return_errno_with_message!(
137                Errno::ENOTDIR,
138                "O_DIRECTORY is specified but the file is not a directory"
139            );
140        }
141
142        if inode_type.is_regular_file()
143            && creation_flags.contains(CreationFlags::O_TRUNC)
144            && !status_flags.contains(StatusFlags::O_PATH)
145        {
146            self.resize(0)?;
147        }
148
149        InodeHandle::new(self.clone(), open_args.access_mode, *status_flags)
150    }

152    /// Gets the real name of the `Path` from its dentry.
153    pub fn name(&self) -> String {
154        self.dentry.name()
155    }

157    /// Gets the parent `Path` within the same mount.
158    ///
159    /// This method returns the parent path within the same filesystem/mount.
160    /// It does NOT cross mount boundaries. If the current path is the root of a mount,
161    /// it will return `None`.
162    ///
163    /// For cross-filesystem parent lookup, use `effective_parent()` instead.
164    pub(super) fn parent_within_mount(&self) -> Option<Self> {
165        let parent = self.dentry.parent()?;
166        Some(Self::new(self.mount.clone(), parent))
167    }

169    /// Gets the top `Path` of the current.
170    ///
171    /// Used when different file systems are mounted on the same mount point.
172    ///
173    /// For example, first `mount /dev/sda1 /mnt` and then `mount /dev/sda2 /mnt`.
174    /// After the second mount is completed, the content of the first mount will be overridden.
175    /// We need to recursively obtain the top `Path`.
176    pub fn get_top_path(mut self) -> Self {
177        while self.dentry.is_mountpoint() {
178            if let Some(child_mount) = self.mount.get(&self.dentry) {
179                let inner = child_mount.root_dentry().clone();
180                self = Self::new(child_mount, inner);
181            } else {
182                break;
183            }
184        }
185
186        self
187    }

189    /// Returns whether this path is the same as or a descendant of `ancestor`
190    /// within the same mount.
191    pub fn is_equal_or_descendant_of(&self, ancestor: &Path) -> bool {
192        self.mount.id() == ancestor.mount.id()
193            && self.dentry.is_equal_or_descendant_of(&ancestor.dentry)
194    }

196    /// Finds the corresponding `Path` in the given mount namespace.
197    fn find_corresponding_mount(&self, mnt_ns: &Arc<MountNamespace>) -> Option<Self> {
198        let corresponding_mount = self.mount.find_corresponding_mount(mnt_ns)?;
199        let corresponding_path = Self::new(corresponding_mount, self.dentry.clone());
200
201        Some(corresponding_path)
202    }

204    /// Checks if this path is reachable from the given `root` path.
205    ///
206    /// A path is considered reachable if it is the same as or a descendant
207    /// of the `root` path. The check traverses upwards from the current path,
208    /// crossing mount point boundaries as necessary, until it either finds
209    /// the `root` path or reaches the global root.
210    fn is_reachable_from(&self, root: &Path) -> bool {
211        let mut owned;
212        let mut current = self;
213
214        loop {
215            if current.mount.id() == root.mount.id() {
216                return current.dentry.is_equal_or_descendant_of(&root.dentry);
217            }
218
219            let Some(parent_mount) = current.mount.parent().and_then(|mount| mount.upgrade())
220            else {
221                // `upgrade()` may fail if `current` is in a mount namespace that has been
222                // dropped. In this situation, `current` is not reachable from `root` because
223                // `root` holds strong reference counts for all its descendant mounts.
224                return false;
225            };
226
227            owned = Path::new(
228                parent_mount,
229                current
230                    .mount
231                    .mountpoint()
232                    .expect("Mounts with parents must have a mount point")
233                    .clone(),
234            );
235            current = &owned;
236        }
237    }

239    /// Returns true if the `Path` represents a pseudo file.
240    fn is_pseudo(&self) -> bool {
241        self.dentry.is_pseudo()
242    }

244    fn this(&self) -> Self {
245        self.clone()
246    }

256impl Path {
257    /// Mounts a filesystem at the current path.
258    ///
259    /// This method attaches a given filesystem to the VFS tree at the location
260    /// represented by `self`. The current path becomes the mountpoint for the new
261    /// filesystem.
262    ///
263    /// Returns the newly created child mount on success.
264    ///
265    /// # Errors
266    ///
267    /// Returns `ENOTDIR` if the path is not a directory.
268    /// Returns `EINVAL` if attempting to mount on root or if the path is not
269    /// in the current mount namespace.
270    pub fn mount(
271        &self,
272        fs: Arc<dyn FileSystem>,
273        flags: PerMountFlags,
274        source: Option<String>,
275        ctx: &Context,
276    ) -> Result<Arc<Mount>> {
277        if self.type_() != InodeType::Dir {
278            return_errno_with_message!(Errno::ENOTDIR, "the path is not a directory");
279        }
280
281        let fs_ref = ctx.thread_local.borrow_fs();
282        let path_resolver = fs_ref.resolver().read();
283
284        if path_resolver.root() == self {
285            return_errno_with_message!(Errno::EINVAL, "the root cannot be mounted on");
286        }
287
288        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
289        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
290        if !current_mnt_ns.owns(&self.mount) {
291            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
292        }
293
294        let child_mount = self.mount.do_mount(fs, flags, &self.dentry, source)?;
295
296        Ok(child_mount)
297    }
298
299    /// Unmounts the filesystem mounted at the current path.
300    ///
301    /// Returns the unmounted child mount on success.
302    ///
303    /// # Errors
304    ///
305    /// Returns `EINVAL` in the following cases:
306    /// - The current path is not a mount root.
307    /// - The mount of the current path is the root mount.
308    /// - The current path is not in the current mount namespace.
309    pub fn unmount(&self, ctx: &Context) -> Result<Arc<Mount>> {
310        if !self.is_mount_root() {
311            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
312        }
313
314        let Some(mountpoint) = self.mount.mountpoint() else {
315            return_errno_with_message!(Errno::EINVAL, "the root mount cannot be unmounted");
316        };
317
318        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
319        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
320        if !current_mnt_ns.owns(&self.mount) {
321            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
322        }
323
324        let parent_mount = self.mount.parent().unwrap().upgrade().unwrap();
325        let child_mount = parent_mount.do_unmount(&mountpoint)?;
326
327        Ok(child_mount)
328    }
329
330    /// Remounts the filesystem with new `PerMountFlags` and optionally new `FsFlags`.
331    ///
332    /// If `fs_flags` is provided, it will update the flags of the mounted filesystem,
333    /// otherwise, only the flags of the current mount will be updated.
334    ///
335    /// # Errors
336    ///
337    /// Returns `EINVAL` in the following cases:
338    /// - The current path is not a mount root.
339    /// - The current path is not in the current mount namespace.
340    pub fn remount(
341        &self,
342        mount_flags: PerMountFlags,
343        fs_flags: Option<FsFlags>,
344        data: Option<CString>,
345        ctx: &Context,
346    ) -> Result<()> {
347        if !self.is_mount_root() {
348            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
349        };
350
351        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
352        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
353        if !current_mnt_ns.owns(&self.mount) {
354            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
355        }
356
357        self.mount.remount(mount_flags, fs_flags, data, ctx)
358    }

360    /// Creates a bind mount from the current path to the destination path.
361    ///
362    /// Creates a new mount tree that mirrors either the root mount (non-recursive)
363    /// or the entire mount subtree (recursive), and attaches it to the destination path.
364    ///
365    /// # Errors
366    ///
367    /// Returns `ENOTDIR` if one of the source and destination is a directory
368    /// and the other is not.
369    ///
370    /// Returns `EINVAL` if any of the following holds:
371    /// - The destination path is not in the current mount namespace.
372    /// - The source path is a mount namespace file that would create a namespace loop.
373    pub fn bind_mount_to(&self, dst_path: &Self, recursive: bool, ctx: &Context) -> Result<()> {
374        let can_bind = {
375            let src_is_dir = self.type_() == InodeType::Dir;
376            let dst_is_dir = dst_path.type_() == InodeType::Dir;
377            (src_is_dir && dst_is_dir) || (!src_is_dir && !dst_is_dir)
378        };
379        if !can_bind {
380            return_errno_with_message!(
381                Errno::ENOTDIR,
382                "the source and destination must both be directories or both be non-directories"
383            );
384        }
385
386        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
387        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
388
389        // Linux does not require the source path to belong to the current
390        // mount namespace, but still rejects source mount namespace files that
391        // would form loops.
392        if !current_mnt_ns.owns(&dst_path.mount) {
393            return_errno_with_message!(
394                Errno::EINVAL,
395                "the destination path is not in this mount namespace"
396            );
397        }
398        if current_mnt_ns.would_form_mnt_ns_loop(&self.dentry) {
399            return_errno_with_message!(
400                Errno::EINVAL,
401                "the source is a mount namespace file that would create a namespace loop"
402            );
403        }
404
405        let current_mnt_ns_weak = Arc::downgrade(current_mnt_ns);
406        let new_mount = self.mount.clone_mount_tree(
407            &self.dentry,
408            &current_mnt_ns_weak,
409            recursive,
410            MountNsFileCopying::Copy,
411        );
412        new_mount.graft_mount_tree(dst_path);
413        Ok(())
414    }
415
416    /// Moves a mount tree from the current path to the destination path.
417    ///
418    /// # Errors
419    ///
420    /// Returns `ENOTDIR` if the `dst_path` is not a directory.
421    ///
422    /// Returns `EINVAL` in the following cases:
423    /// - The current path is not a mount root.
424    /// - The mount of the current path is the root mount.
425    /// - Either source or destination path is not in the current mount namespace.
426    ///
427    /// Returns `ELOOP` in the following cases:
428    /// - The destination path is inside the subtree being moved.
429    /// - The mount tree contains a mount namespace file that would create a namespace loop.
430    pub fn move_mount_to(&self, dst_path: &Self, ctx: &Context) -> Result<()> {
431        if !self.is_mount_root() {
432            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
433        };
434        if self.mount_node().parent().is_none() {
435            return_errno_with_message!(Errno::EINVAL, "the root mount can not be moved");
436        }
437
438        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
439        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
440        if !current_mnt_ns.owns(&self.mount) {
441            return_errno_with_message!(
442                Errno::EINVAL,
443                "the source path is not in this mount namespace"
444            );
445        }
446        if !current_mnt_ns.owns(&dst_path.mount) {
447            return_errno_with_message!(
448                Errno::EINVAL,
449                "the destination path is not in this mount namespace"
450            );
451        }
452        current_mnt_ns.check_no_mnt_ns_loop_in_tree(self.mount_node())?;
453        if dst_path
454            .mount_node()
455            .is_equal_or_descendant_of(self.mount_node())
456        {
457            // Reject moves that would place a mount beneath itself, because the mount tree
458            // must remain acyclic.
459            return_errno_with_message!(
460                Errno::ELOOP,
461                "the destination path is inside the mount subtree being moved"
462            );
463        }
464
465        self.mount.graft_mount_tree(dst_path);
466
467        Ok(())
468    }
469
470    /// Sets the propagation type of the mount of this `Path`.
471    pub fn set_mount_propagation(
472        &self,
473        prop: MountPropType,
474        recursive: bool,
475        ctx: &Context,
476    ) -> Result<()> {
477        if !self.is_mount_root() {
478            return_errno_with_message!(Errno::EINVAL, "the path is not a mount root");
479        };
480
481        let current_ns_proxy = ctx.thread_local.borrow_ns_proxy();
482        let current_mnt_ns = current_ns_proxy.unwrap().mnt_ns();
483        if !current_mnt_ns.owns(&self.mount) {
484            return_errno_with_message!(Errno::EINVAL, "the path is not in this mount namespace");
485        }
486
487        self.mount.set_propagation(prop, recursive);
488
489        Ok(())
490    }

493// Methods inherited from `Dentry`.
494#[inherit_methods(from = "self.dentry")]
495impl Path {
496    pub fn inode(&self) -> &Arc<dyn Inode>;
497    pub fn type_(&self) -> InodeType;
498
499    /// Creates a `Path` by making an inode of the `type_` with the `mode`.
500    pub fn mknod(&self, name: &str, mode: InodeMode, type_: MknodType) -> Result<Self> {
501        let inner = self
502            .dentry
503            .as_dir_dentry_or_err()?
504            .mknod(name, mode, type_)?;
505        Ok(Self::new(self.mount.clone(), inner))
506    }
507
508    /// Links a new name for the `Path`.
509    pub fn link(&self, old: &Self, name: &str) -> Result<()> {
510        if !Arc::ptr_eq(&old.mount, &self.mount) {
511            return_errno_with_message!(Errno::EXDEV, "the operation cannot cross mounts");
512        }
513
514        self.dentry.as_dir_dentry_or_err()?.link(old.inode(), name)
515    }
516
517    /// Unlinks a name from the `Path`.
518    pub fn unlink(&self, name: &str) -> Result<()> {
519        self.dentry.as_dir_dentry_or_err()?.unlink(name)
520    }
521
522    /// Removes a directory by `rmdir()` the inner inode.
523    pub fn rmdir(&self, name: &str) -> Result<()> {
524        self.dentry.as_dir_dentry_or_err()?.rmdir(name)
525    }
526
527    /// Renames a `Path` to the new `Path` by `rename()` the inner inode.
528    pub fn rename(&self, old_name: &str, new_dir: &Self, new_name: &str) -> Result<()> {
529        if !Arc::ptr_eq(&self.mount, &new_dir.mount) {
530            return_errno_with_message!(Errno::EXDEV, "the operation cannot cross mounts");
531        }
532
533        DirDentry::rename(&self.dentry, old_name, &new_dir.dentry, new_name)
534    }
535}

573/// Checks if the file name is ".", indicating it's the current directory.
574pub const fn is_dot(filename: &str) -> bool {
575    let name_bytes = filename.as_bytes();
576    name_bytes.len() == 1 && name_bytes[0] == DOT_BYTE
577}
578
579/// Checks if the file name is "..", indicating it's the parent directory.
580pub const fn is_dotdot(filename: &str) -> bool {
581    let name_bytes = filename.as_bytes();
582    name_bytes.len() == 2 && name_bytes[0] == DOT_BYTE && name_bytes[1] == DOT_BYTE
583}
584
585/// Checks if the file name is "." or "..", indicating it's the current or parent directory.
586pub const fn is_dot_or_dotdot(filename: &str) -> bool {
587    let name_bytes = filename.as_bytes();
588    let name_len = name_bytes.len();
589    if name_len == 1 {
590        name_bytes[0] == DOT_BYTE
591    } else if name_len == 2 {
592        name_bytes[0] == DOT_BYTE && name_bytes[1] == DOT_BYTE
593    } else {
594        false
595    }
596}
597
598const DOT_BYTE: u8 = b'.';
