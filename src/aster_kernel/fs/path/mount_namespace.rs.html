14/// Represents a mount namespace, which encapsulates a mount tree and provides
15/// isolation for filesystem views between different threads.
16///
17/// A `MountNamespace` only allows operations on [`Mount`]s that belong to that `MountNamespace`.
18/// If the operation target includes [`Mount`]s from other `MountNamespace`s, it will be directly
19/// rejected and return an `Err`.
20pub struct MountNamespace {
21    /// The root mount of this namespace.
22    root: Arc<Mount>,
23    /// The user namespace that owns this mount namespace.
24    owner: Arc<UserNamespace>,
25}
26
27impl MountNamespace {
28    /// Returns a reference to the singleton initial mount namespace.
29    #[doc(hidden)]
30    pub fn get_init_singleton() -> &'static Arc<MountNamespace> {
31        static INIT: Once<Arc<MountNamespace>> = Once::new();
32
33        INIT.call_once(|| {
34            let owner = UserNamespace::get_init_singleton().clone();
35            let rootfs = RamFs::new();
36
37            Arc::new_cyclic(|weak_self| {
38                let root = Mount::new_root(rootfs, weak_self.clone());
39                MountNamespace { root, owner }
40            })
41        })
42    }
43
44    /// Gets the root mount of this namespace.
45    pub fn root(&self) -> &Arc<Mount> {
46        &self.root
47    }
48
49    /// Creates a new filesystem resolver for this namespace.
50    ///
51    /// The resolver is initialized with the root and current working directory
52    /// both set to the **effective root** of this mount namespace.
53    ///
54    /// The "effective root" refers to the currently visible root directory, which
55    /// may differ from the original root filesystem if overlay mounts exist.
56    pub fn new_path_resolver(&self) -> PathResolver {
57        let root = Path::new_fs_root(self.root.clone()).get_top_path();
58        let cwd = Path::new_fs_root(self.root.clone()).get_top_path();
59        PathResolver::new(root, cwd)
60    }
61
62    /// Creates a deep copy of this mount namespace, including the entire mount tree.
63    ///
64    /// This is typically used when creating a new namespace for a process or thread.
65    pub fn new_clone(
66        &self,
67        owner: Arc<UserNamespace>,
68        posix_thread: &PosixThread,
69    ) -> Result<Arc<MountNamespace>> {
70        owner.check_cap(CapSet::SYS_ADMIN, posix_thread)?;
71
72        let root_mount = &self.root;
73        let new_mnt_ns = Arc::new_cyclic(|weak_self| {
74            let new_root =
75                root_mount.clone_mount_tree(root_mount.root_dentry(), Some(weak_self), true);
76
77            MountNamespace {
78                root: new_root,
79                owner,
80            }
81        });
82
83        Ok(new_mnt_ns)
84    }
85
86    /// Flushes all pending filesystem metadata and cached file data to the device
87    /// for all mounted filesystems in this mount namespace.
88    pub fn sync(&self) -> Result<()> {
89        let mut mount_queue = VecDeque::new();
90        let mut visited_filesystems = hashbrown::HashSet::new();
91        mount_queue.push_back(self.root.clone());
92
93        while let Some(current_mount) = mount_queue.pop_front() {
94            let fs_ptr = Arc::as_ptr(current_mount.fs());
95            // Only sync each filesystem once.
96            if visited_filesystems.insert(fs_ptr) {
97                current_mount.sync()?;
98            }
99
100            let children = current_mount.children.read();
101            for child_mount in children.values() {
102                mount_queue.push_back(child_mount.clone());
103            }
104        }
105
106        Ok(())
107    }
108
109    /// Returns the owner user namespace of the namespace.
110    pub fn owner(&self) -> &Arc<UserNamespace> {
111        &self.owner
112    }
113
114    /// Checks whether a given mount belongs to this mount namespace.
115    pub fn owns(self: &Arc<Self>, mount: &Mount) -> bool {
116        mount.mnt_ns().as_ptr() == Arc::as_ptr(self)
117    }
118}

120// When a mount namespace is dropped, it means that the corresponding mount
121// tree is no longer valid. Therefore, all mounts in its mount tree should be
122// detached from their parents and cleared of their mountpoints.
123impl Drop for MountNamespace {
124    fn drop(&mut self) {
125        let mut worklist = VecDeque::new();
126        worklist.push_back(self.root.clone());
127        while let Some(current_mount) = worklist.pop_front() {
128            let mut children = current_mount.children.write();
129            for (_, child) in children.drain() {
130                child.set_parent(None);
131                child.clear_mountpoint();
132                worklist.push_back(child);
133            }
134        }
135    }
136}
