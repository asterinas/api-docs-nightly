24/// The basic operations defined on a file
25pub trait FileLike: Pollable + Send + Sync + Any {
26    fn read(&self, writer: &mut VmWriter) -> Result<usize> {
27        return_errno_with_message!(Errno::EBADF, "the file is not valid for reading");
28    }
29
30    fn write(&self, reader: &mut VmReader) -> Result<usize> {
31        return_errno_with_message!(Errno::EBADF, "the file is not valid for writing");
32    }
33
34    /// Read at the given file offset.
35    ///
36    /// The file must be seekable to support `read_at`.
37    /// Unlike [`read`], `read_at` will not change the file offset.
38    ///
39    /// [`read`]: FileLike::read
40    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
41        return_errno_with_message!(Errno::ESPIPE, "read_at is not supported");
42    }
43
44    /// Write at the given file offset.
45    ///
46    /// The file must be seekable to support `write_at`.
47    /// Unlike [`write`], `write_at` will not change the file offset.
48    /// If the file is append-only, the `offset` will be ignored.
49    ///
50    /// [`write`]: FileLike::write
51    fn write_at(&self, offset: usize, reader: &mut VmReader) -> Result<usize> {
52        return_errno_with_message!(Errno::ESPIPE, "write_at is not supported");
53    }
54
55    fn ioctl(&self, raw_ioctl: RawIoctl) -> Result<i32> {
56        // `ENOTTY` means that "The specified operation does not apply to the kind of object that
57        // the file descriptor references".
58        // Reference: <https://man7.org/linux/man-pages/man2/ioctl.2.html>.
59        return_errno_with_message!(Errno::ENOTTY, "ioctl is not supported");
60    }
61
62    /// Obtains the mappable object to map this file into the user address space.
63    ///
64    /// If this file has a corresponding mappable object of [`Mappable`],
65    /// then it can be either an inode or an MMIO region.
66    fn mappable(&self) -> Result<Mappable> {
67        // `ENODEV` means that "The underlying filesystem of the specified file does not support
68        // memory mapping".
69        // Reference: <https://man7.org/linux/man-pages/man2/mmap.2.html>.
70        return_errno_with_message!(Errno::ENODEV, "the file is not mappable");
71    }
72
73    fn resize(&self, new_size: usize) -> Result<()> {
74        return_errno_with_message!(Errno::EINVAL, "resize is not supported");
75    }
76
77    fn status_flags(&self) -> StatusFlags {
78        StatusFlags::empty()
79    }
80
81    fn set_status_flags(&self, _new_flags: StatusFlags) -> Result<()> {
82        return_errno_with_message!(Errno::EINVAL, "set_status_flags is not supported");
83    }
84
85    fn access_mode(&self) -> AccessMode {
86        AccessMode::O_RDWR
87    }
88
89    fn seek(&self, seek_from: SeekFrom) -> Result<usize> {
90        return_errno_with_message!(Errno::ESPIPE, "seek is not supported");
91    }
92
93    fn fallocate(&self, mode: FallocMode, offset: usize, len: usize) -> Result<()> {
94        return_errno_with_message!(Errno::EOPNOTSUPP, "fallocate is not supported");
95    }
96
97    fn as_socket(&self) -> Option<&dyn Socket> {
98        None
99    }
100
101    fn path(&self) -> &Path;
102
103    /// Dumps information to appear in the `fdinfo` file under procfs.
104    ///
105    /// This method must not break atomic mode because it will be called with the file table's spin
106    /// lock held. There are two strategies for implementing this method:
107    ///  - If the necessary information can be obtained without breaking atomic mode, the method
108    ///    can collect and return the information directly. `Arc<Self>` should be dropped and
109    ///    should not appear in the returned `Box<dyn Display>`.
110    ///  - Otherwise, if the file can be dropped asynchronously in another process, the method can
111    ///    return a `Box<dyn Display>` containing the `Arc<Self>`, so that the information can be
112    ///    collected later in its `Display::display()` method, after dropping the file table's spin
113    ///    lock.
114    fn dump_proc_fdinfo(self: Arc<Self>, fd_flags: FdFlags) -> Box<dyn Display>;
115}
116
117impl dyn FileLike {
118    pub fn downcast_ref<T: FileLike>(&self) -> Option<&T> {
119        (self as &dyn Any).downcast_ref::<T>()
120    }
121
122    pub fn read_bytes(&self, buf: &mut [u8]) -> Result<usize> {
123        let mut writer = VmWriter::from(buf).to_fallible();
124        self.read(&mut writer)
125    }
126
127    pub fn write_bytes(&self, buf: &[u8]) -> Result<usize> {
128        let mut reader = VmReader::from(buf).to_fallible();
129        self.write(&mut reader)
130    }
131
132    pub fn read_bytes_at(&self, offset: usize, buf: &mut [u8]) -> Result<usize> {
133        let mut writer = VmWriter::from(buf).to_fallible();
134        self.read_at(offset, &mut writer)
135    }
136
137    #[expect(dead_code)]
138    pub fn write_bytes_at(&self, offset: usize, buf: &[u8]) -> Result<usize> {
139        let mut reader = VmReader::from(buf).to_fallible();
140        self.write_at(offset, &mut reader)
141    }
142
143    pub fn as_socket_or_err(&self) -> Result<&dyn Socket> {
144        self.as_socket()
145            .ok_or_else(|| Error::with_message(Errno::ENOTSOCK, "the file is not a socket"))
146    }
147
148    pub fn as_inode_handle_or_err(&self) -> Result<&InodeHandle> {
149        self.downcast_ref().ok_or_else(|| {
150            Error::with_message(Errno::EINVAL, "the file is not related to an inode")
151        })
152    }
153}

155/// An object that may be memory mapped into the user address space.
156#[derive(Debug, Clone)]
157pub enum Mappable {
158    /// A VMO (i.e., page cache).
159    Vmo(Arc<Vmo>),
160    /// An MMIO region.
161    IoMem(IoMem),
162}
