14/// Represents the inode at /proc/filesystems.
15pub struct FileSystemsFileOps;
16
17impl FileSystemsFileOps {
18    pub fn new_inode(parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
19        // Reference:
20        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/filesystems.c#L259>
21        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/generic.c#L549-L550>
22        ProcFileBuilder::new(Self, mkmod!(a+r))
23            .parent(parent)
24            .build()
25            .unwrap()
26    }
27}
28
29impl FileOps for FileSystemsFileOps {
30    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
31        let mut printer = VmPrinter::new_skip(writer, offset);
32
33        crate::fs::registry::with_iter(|iter| -> core::result::Result<(), VmPrinterError> {
34            for (fs_name, fs_type) in iter {
35                if fs_type.properties().contains(FsProperties::NEED_DISK) {
36                    writeln!(printer, "\t{}", fs_name)?;
37                } else {
38                    writeln!(printer, "nodev\t{}", fs_name)?;
39                }
40            }
41
42            Ok(())
43        })?;
44
45        Ok(printer.bytes_written())
46    }
47}
