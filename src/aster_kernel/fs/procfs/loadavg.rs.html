3//! This module offers `/proc/loadavg` file support, which tells the user space
4//! about the cpu load average for the last 1, 5, and 15 minutes.
5//!
6//! Reference: <https://www.man7.org/linux/man-pages/man5/proc_loadavg.5.html>
7
8use aster_util::printer::VmPrinter;
9
10use crate::{
11    fs::{
12        procfs::template::{FileOps, ProcFileBuilder},
13        utils::{Inode, mkmod},
14    },
15    prelude::*,
16    process::posix_thread,
17    sched::{self, loadavg::get_loadavg},
18};
19
20/// Represents the inode at `/proc/loadavg`.
21pub struct LoadAvgFileOps;
22
23impl LoadAvgFileOps {
24    pub fn new_inode(parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
25        // Reference:
26        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/loadavg.c#L33>
27        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/generic.c#L549-L550>
28        ProcFileBuilder::new(Self, mkmod!(a+r))
29            .parent(parent)
30            .build()
31            .unwrap()
32    }
33}
34
35impl FileOps for LoadAvgFileOps {
36    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
37        let mut printer = VmPrinter::new_skip(writer, offset);
38
39        let avg = get_loadavg();
40        let (nr_queued, nr_running) = sched::nr_queued_and_running();
41        writeln!(
42            printer,
43            "{:.2} {:.2} {:.2} {}/{} {}",
44            avg[0],
45            avg[1],
46            avg[2],
47            nr_running,
48            nr_queued,
49            posix_thread::last_tid(),
50        )?;
51
52        Ok(printer.bytes_written())
53    }
54}
