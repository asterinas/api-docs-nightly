46/// Represents the inode at `/proc/[pid]/task`.
47pub struct TaskDirOps(Arc<Process>);
48
49impl TaskDirOps {
50    pub fn new_inode(dir: &PidDirOps, parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
51        let process_ref = dir.0.process_ref.clone();
52        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3316>
53        let task_dir_inode = ProcDirBuilder::new(Self(process_ref), mkmod!(a+rx))
54            .parent(parent)
55            .build()
56            .unwrap();
57
58        let weak_ptr = Arc::downgrade(&task_dir_inode);
59        dir.0.process_ref.tasks().lock().register_observer(weak_ptr);
60
61        task_dir_inode
62    }
63}

65/// Represents the inode at `/proc/[pid]/task/[tid]`.
66#[derive(Clone)]
67pub(super) struct TidDirOps {
68    pub(super) process_ref: Arc<Process>,
69    /// If `thread_ref` is `None`, this corresponds to a process-level `/proc/[pid]/*` file.
70    /// Otherwise, this corresponds to a thread-level `/proc/[pid]/task/[tid]/*` file.
71    pub(super) thread_ref: Option<Arc<Thread>>,
72}
73
74impl TidDirOps {
75    pub fn new_inode(
76        process_ref: Arc<Process>,
77        thread_ref: Arc<Thread>,
78        parent: Weak<dyn Inode>,
79    ) -> Arc<dyn Inode> {
80        ProcDirBuilder::new(
81            Self {
82                process_ref,
83                thread_ref: Some(thread_ref),
84            },
85            // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3796>
86            mkmod!(a+rx),
87        )
88        .parent(parent)
89        .build()
90        .unwrap()
91    }
92
93    pub fn thread(&self) -> Arc<Thread> {
94        self.thread_ref
95            .clone()
96            .unwrap_or_else(|| self.process_ref.main_thread())
97    }
98
99    #[expect(clippy::type_complexity)]
100    const STATIC_ENTRIES: &'static [(
101        &'static str,
102        fn(&TidDirOps, Weak<dyn Inode>) -> Arc<dyn Inode>,
103    )] = &[
104        ("cgroup", CgroupFileOps::new_inode),
105        ("cmdline", CmdlineFileOps::new_inode),
106        ("comm", CommFileOps::new_inode),
107        ("environ", EnvironFileOps::new_inode),
108        ("exe", ExeSymOps::new_inode),
109        ("fd", FdDirOps::<fd::FileSymOps>::new_inode),
110        ("fdinfo", FdDirOps::<fd::FileInfoOps>::new_inode),
111        ("gid_map", GidMapFileOps::new_inode),
112        ("mem", MemFileOps::new_inode),
113        ("mountinfo", MountInfoFileOps::new_inode),
114        ("oom_score_adj", OomScoreAdjFileOps::new_inode),
115        ("stat", StatFileOps::new_inode),
116        ("status", StatusFileOps::new_inode),
117        ("uid_map", UidMapFileOps::new_inode),
118        ("maps", MapsFileOps::new_inode),
119        ("mounts", MountsFileOps::new_inode),
120    ];
121}
122
123impl DirOps for TidDirOps {
124    fn lookup_child(&self, dir: &ProcDir<Self>, name: &str) -> Result<Arc<dyn Inode>> {
125        let mut cached_children = dir.cached_children().write();
126        self.lookup_child_locked(&mut cached_children, dir.this_weak().clone(), name)
127    }
128
129    fn populate_children<'a>(
130        &self,
131        dir: &'a ProcDir<Self>,
132    ) -> RwMutexUpgradeableGuard<'a, SlotVec<(String, Arc<dyn Inode>)>> {
133        let mut cached_children = dir.cached_children().write();
134        self.populate_children_locked(&mut cached_children, dir.this_weak().clone());
135        cached_children.downgrade()
136    }
137}
138
139impl TidDirOps {
140    pub(super) fn lookup_child_locked(
141        &self,
142        cached_children: &mut SlotVec<(String, Arc<dyn Inode>)>,
143        this_ptr: Weak<dyn Inode>,
144        name: &str,
145    ) -> Result<Arc<dyn Inode>> {
146        if let Some(child) =
147            lookup_child_from_table(name, cached_children, Self::STATIC_ENTRIES, |f| {
148                (f)(self, this_ptr)
149            })
150        {
151            return Ok(child);
152        }
153
154        return_errno_with_message!(Errno::ENOENT, "the file does not exist");
155    }
156
157    pub(super) fn populate_children_locked(
158        &self,
159        cached_children: &mut SlotVec<(String, Arc<dyn Inode>)>,
160        this_ptr: Weak<dyn Inode>,
161    ) {
162        populate_children_from_table(cached_children, Self::STATIC_ENTRIES, |f| {
163            (f)(self, this_ptr.clone())
164        });
165    }
166}

168impl Observer<TidEvent> for ProcDir<TaskDirOps> {
169    fn on_events(&self, events: &TidEvent) {
170        let TidEvent::Exit(tid) = events;
171
172        let mut cached_children = self.cached_children().write();
173        cached_children.remove_entry_by_name(&tid.to_string());
174    }
175}
176
177impl DirOps for TaskDirOps {
178    // Lock order: task set -> cached entries
179    //
180    // Note that inverting the lock order is non-trivial because `Observer::on_events` will be
181    // called with the task set locked.
182
183    fn lookup_child(&self, dir: &ProcDir<Self>, name: &str) -> Result<Arc<dyn Inode>> {
184        let Ok(tid) = name.parse::<Tid>() else {
185            return_errno_with_message!(Errno::ENOENT, "the name is not a valid TID");
186        };
187
188        for task in self.0.tasks().lock().as_slice() {
189            let thread_ref = task.as_thread().unwrap();
190            if thread_ref.as_posix_thread().unwrap().tid() != tid {
191                continue;
192            }
193
194            let mut cached_children = dir.cached_children().write();
195            return Ok(cached_children
196                .put_entry_if_not_found(name, || {
197                    TidDirOps::new_inode(
198                        self.0.clone(),
199                        thread_ref.clone(),
200                        dir.this_weak().clone(),
201                    )
202                })
203                .clone());
204        }
205
206        return_errno_with_message!(Errno::ENOENT, "the thread does not exist")
207    }
208
209    fn populate_children<'a>(
210        &self,
211        dir: &'a ProcDir<Self>,
212    ) -> RwMutexUpgradeableGuard<'a, SlotVec<(String, Arc<dyn Inode>)>> {
213        let tasks = self.0.tasks().lock();
214        let mut cached_dentries = dir.cached_children().write();
215
216        for task in tasks.as_slice() {
217            let thread_ref = task.as_thread().unwrap();
218            cached_dentries.put_entry_if_not_found(
219                &task.as_posix_thread().unwrap().tid().to_string(),
220                || {
221                    TidDirOps::new_inode(
222                        self.0.clone(),
223                        thread_ref.clone(),
224                        dir.this_weak().clone(),
225                    )
226                },
227            );
228        }
229
230        cached_dentries.downgrade()
231    }
