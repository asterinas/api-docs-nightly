16/// Represents the inode at `/proc/[pid]/task/[tid]/status` (and also `/proc/[pid]/status`).
17/// See <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/array.c#L148>.
18/// FIXME: Some fields are not implemented yet.
19///
20/// Fields:
21/// - Name:   The name of the process.
22/// - State:  The current state of the process (e.g., R for running, S for sleeping).
23/// - Tgid:   The Thread Group ID, which is the same as the process ID for the main thread.
24/// - Pid:    The process ID.
25/// - PPid:   The parent process ID.
26/// - TracerPid: The PID of the process tracing this process, or 0 if not being traced.
27/// - Uid:    Real, effective, saved set, and filesystem UIDs.
28/// - Gid:    Real, effective, saved set, and filesystem GIDs.
29/// - FDSize: The number of file descriptor slots currently allocated.
30/// - Groups: Supplementary group IDs.
31/// - VmPeak: Peak virtual memory size.
32/// - VmSize: Current virtual memory size.
33/// - VmLck:  Locked memory size.
34/// - VmPin:  Pinned memory size.
35/// - VmHWM:  Peak resident set size ("high water mark").
36/// - VmRSS:  Resident set size.
37/// - VmData: Size of data segment.
38/// - VmStk:  Size of stack segment.
39/// - VmExe:  Size of text segment.
40/// - VmLib:  Shared library code size.
41/// - VmPTE:  Page table entries size.
42/// - VmSwap: Swapped-out virtual memory size by anonymous private pages.
43/// - Threads: Number of threads in this process.
44/// - SigQ:   Current signal queue size and limit.
45/// - SigPnd: Threads pending signals.
46/// - ShdPnd: Shared pending signals.
47/// - SigBlk: Blocked signals.
48/// - SigIgn: Ignored signals.
49/// - SigCgt: Caught signals.
50/// - CapInh: Inheritable capabilities.
51/// - CapPrm: Permitted capabilities.
52/// - CapEff: Effective capabilities.
53/// - CapBnd: Bounding set.
54/// - CapAmb: Ambient capabilities.
55/// - Seccomp: Seccomp mode.
56/// - Cpus_allowed: CPUs allowed for this process.
57/// - Cpus_allowed_list: List of CPUs allowed for this process.
58/// - Mems_allowed: Memory nodes allowed for this process.
59/// - Mems_allowed_list: List of memory nodes allowed for this process.
60/// - voluntary_ctxt_switches: Number of voluntary context switches.
61/// - nonvoluntary_ctxt_switches: Number of nonvoluntary context switches.
62pub struct StatusFileOps(TidDirOps);
63
64impl StatusFileOps {
65    pub fn new_inode(dir: &TidDirOps, parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
66        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3326>
67        ProcFileBuilder::new(Self(dir.clone()), mkmod!(a+r))
68            .parent(parent)
69            .build()
70            .unwrap()
71    }
72}
73
74impl FileOps for StatusFileOps {
75    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
76        let mut printer = VmPrinter::new_skip(writer, offset);
77
78        let process = self.0.process_ref.as_ref();
79        let thread = self.0.thread();
80        let posix_thread = thread.as_posix_thread().unwrap();
81
82        // According to the Linux implementation, a process's `/proc/<pid>/status`
83        // is exactly the same as its main thread's `/proc/<pid>/task/<pid>/status`.
84        //
85        // Reference:
86        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3326>
87        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3675>
88
89        writeln!(
90            printer,
91            "Name:\t{}",
92            posix_thread.thread_name().lock().name().to_string_lossy()
93        )?;
94
95        let state = if thread.is_exited() {
96            "Z (zombie)"
97        } else {
98            match posix_thread.sleeping_state() {
99                SleepingState::Running => "R (running)",
100                SleepingState::Interruptible => "S (sleeping)",
101                SleepingState::Uninterruptible => "D (disk sleep)",
102                SleepingState::StopBySignal => "T (stopped)",
103                SleepingState::StopByPtrace => "t (tracing stop)",
104            }
105        };
106        writeln!(printer, "State:\t{}", state)?;
107
108        writeln!(printer, "Tgid:\t{}", process.pid())?;
109        writeln!(printer, "Pid:\t{}", posix_thread.tid())?;
110        writeln!(printer, "PPid:\t{}", process.parent().pid())?;
111        writeln!(printer, "TracerPid:\t{}", 0)?;
112
113        let credentials = posix_thread.credentials();
114        writeln!(
115            printer,
116            "Uid:\t{}\t{}\t{}\t{}",
117            u32::from(credentials.ruid()),
118            u32::from(credentials.euid()),
119            u32::from(credentials.suid()),
120            u32::from(credentials.fsuid()),
121        )?;
122        writeln!(
123            printer,
124            "Gid:\t{}\t{}\t{}\t{}",
125            u32::from(credentials.rgid()),
126            u32::from(credentials.egid()),
127            u32::from(credentials.sgid()),
128            u32::from(credentials.fsgid()),
129        )?;
130
131        writeln!(
132            printer,
133            "FDSize:\t{}",
134            posix_thread
135                .file_table()
136                .lock()
137                .as_ref()
138                .map(|file_table| file_table.read().len())
139                .unwrap_or(0)
140        )?;
141
142        if let Some(vmar_ref) = process.lock_vmar().as_ref() {
143            let vsize = vmar_ref.get_mappings_total_size();
144            let anon = vmar_ref.get_rss_counter(RssType::RSS_ANONPAGES) * (PAGE_SIZE / 1024);
145            let file = vmar_ref.get_rss_counter(RssType::RSS_FILEPAGES) * (PAGE_SIZE / 1024);
146            let rss = anon + file;
147            writeln!(
148                printer,
149                "VmSize:\t{} kB\nVmRSS:\t{} kB\nRssAnon:\t{} kB\nRssFile:\t{} kB",
150                vsize, rss, anon, file
151            )?;
152        }
153
154        if process.pid() == posix_thread.tid() {
155            writeln!(
156                printer,
157                "Threads:\t{}",
158                process.tasks().lock().as_slice().len()
159            )?;
160        }
161
162        Ok(printer.bytes_written())
163    }
164}
