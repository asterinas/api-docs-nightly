19/// A single entry in the mounts file.
20struct MountEntry<'a> {
21    /// Filesystem-specific information or "none".
22    source: &'a str,
23    /// Mount point relative to the process's root directory.
24    mount_point: &'a str,
25    /// The type of the filesystem in the form "type[.subtype]".
26    fs_type: &'a str,
27    /// Per-mount flags.
28    mount_flags: PerMountFlags,
29    /// The dump field is used by the dump(8) program to determine which
30    /// filesystems need to be dumped.
31    dump: u32,
32    /// The fsck(8) program uses this field.
33    pass: u32,
34}
35
36impl core::fmt::Display for MountEntry<'_> {
37    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
38        write!(
39            f,
40            "{} {} {} {} {} {}",
41            &self.source,
42            &self.mount_point,
43            &self.fs_type,
44            &self.mount_flags,
45            &self.dump,
46            &self.pass,
47        )
48    }
49}

51/// Represents the inode at `/proc/[pid]/task/[tid]/mounts` (and also `/proc/[pid]/mounts`).
52pub struct MountsFileOps(TidDirOps);
53
54impl MountsFileOps {
55    pub fn new_inode(dir: &TidDirOps, parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
56        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3351>
57        ProcFileBuilder::new(Self(dir.clone()), mkmod!(a+r))
58            .parent(parent)
59            .build()
60            .unwrap()
61    }
62
63    /// Reads mount information for `/proc/[pid]/mounts` and `/proc/mounts`.
64    ///
65    /// Provides a simplified view of mounted filesystems in the traditional
66    /// `/etc/fstab` format.
67    fn read_mounts(
68        &self,
69        path_resolver: &PathResolver,
70        offset: usize,
71        writer: &mut VmWriter,
72    ) -> Result<usize> {
73        let mut printer = VmPrinter::new_skip(writer, offset);
74
75        for mount in path_resolver.collect_visible_mounts() {
76            let parent = mount.parent().and_then(|parent| parent.upgrade());
77            let is_resolver_root_mount = Arc::ptr_eq(&mount, path_resolver.root().mount_node());
78            let mount_point = make_mount_point_path(
79                is_resolver_root_mount,
80                parent.as_ref(),
81                mount.as_ref(),
82                path_resolver,
83            );
84            let mount_flags = mount.flags();
85            let fs_type = mount.fs().name();
86            let source = mount.source().unwrap_or("none");
87
88            // The dump and pass fields are hardcoded to 0, because the kernel considers them
89            // userspace policy (managed by /etc/fstab) and does not store them in the VFS layer.
90            // This behavior is consistent with Linux.
91            //
92            // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc_namespace.c#L130>.
93            let dump = 0;
94            let pass = 0;
95
96            let entry = MountEntry {
97                source,
98                mount_point: &mount_point,
99                fs_type,
100                mount_flags,
101                dump,
102                pass,
103            };
104
105            writeln!(printer, "{}", entry)?;
106        }
107
108        Ok(printer.bytes_written())
109    }
110}
111
112impl FileOps for MountsFileOps {
113    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
114        let thread = self.0.thread();
115        let posix_thread = thread.as_posix_thread().unwrap();
116
117        let fs = posix_thread.read_fs();
118        let path_resolver = fs.resolver().read();
119        self.read_mounts(&path_resolver, offset, writer)
120    }
121}
