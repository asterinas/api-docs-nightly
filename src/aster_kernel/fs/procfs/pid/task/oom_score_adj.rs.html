17/// Represents the inode at `/proc/[pid]/task/[tid]/oom_score_adj` (and also `/proc/[pid]/oom_score_adj`).
18pub struct OomScoreAdjFileOps(Arc<Process>);
19
20impl OomScoreAdjFileOps {
21    pub fn new_inode(dir: &TidDirOps, parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
22        let process_ref = dir.process_ref.clone();
23        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3386>
24        ProcFileBuilder::new(Self(process_ref), mkmod!(a+r, u+w))
25            .parent(parent)
26            .build()
27            .unwrap()
28    }
29}
30
31impl FileOps for OomScoreAdjFileOps {
32    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
33        let mut printer = VmPrinter::new_skip(writer, offset);
34
35        let oom_score_adj = self.0.oom_score_adj().load(Ordering::Relaxed);
36        writeln!(printer, "{}", oom_score_adj)?;
37
38        Ok(printer.bytes_written())
39    }
40
41    fn write_at(&self, _offset: usize, reader: &mut VmReader) -> Result<usize> {
42        let (cstr, read_bytes) = reader.read_cstring_until_end(BUF_SIZE_I32 - 1)?;
43        let val = cstr
44            .to_str()
45            .ok()
46            .map(|str| str.trim())
47            .and_then(|str| str.parse::<i32>().ok())
48            .ok_or_else(|| {
49                Error::with_message(Errno::EINVAL, "the value is not a valid integer")
50            })?;
51        if !(OOM_SCORE_ADJ_MIN..=OOM_SCORE_ADJ_MAX).contains(&val) {
52            return_errno_with_message!(Errno::EINVAL, "the OOM score adjustment is out of range");
53        }
54
55        // TODO: If the new adjustment value is smaller than the smallest
56        // adjustment value that the process has ever reached and the writer
57        // does not have the `SYS_RESOURCE` capability, we should fail with
58        // `EACCES`. See
59        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L1152>.
60        self.0.oom_score_adj().store(val as i16, Ordering::Relaxed);
61
62        Ok(read_bytes)
63    }
64}
65
66/// Worst case buffer size needed for holding an integer.
67// Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/internal.h#L163>.
68const BUF_SIZE_I32: usize = 13;
69
70// FIXME: Support OOM killer and move these constants to a more appropriate place.
71const OOM_SCORE_ADJ_MIN: i32 = -1000;
72const OOM_SCORE_ADJ_MAX: i32 = 1000;
