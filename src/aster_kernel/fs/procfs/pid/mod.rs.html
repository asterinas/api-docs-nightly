20/// Represents the inode at `/proc/[pid]`.
21pub struct PidDirOps(
22    // The `/proc/<pid>` directory is a superset of the `/proc/<pid>/task/<tid>` directory.
23    // So we embed `TidDirOps` here so that `PidDirOps` can "inherit" entries and methods
24    // from `TidDirOps`.
25    TidDirOps,
26);
27
28impl PidDirOps {
29    pub fn new_inode(process_ref: Arc<Process>, parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
30        let tid_dir_ops = TidDirOps {
31            process_ref,
32            thread_ref: None,
33        };
34        // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/base.c#L3493>
35        ProcDirBuilder::new(Self(tid_dir_ops.clone()), mkmod!(a+rx))
36            .parent(parent)
37            // The PID directories must be volatile, because it is just associated with one process.
38            .volatile()
39            .build()
40            .unwrap()
41    }
42
43    #[expect(clippy::type_complexity)]
44    const STATIC_ENTRIES: &'static [(
45        &'static str,
46        fn(&PidDirOps, Weak<dyn Inode>) -> Arc<dyn Inode>,
47    )] = &[("task", TaskDirOps::new_inode)];
48}
49
50impl DirOps for PidDirOps {
51    fn lookup_child(&self, dir: &ProcDir<Self>, name: &str) -> Result<Arc<dyn Inode>> {
52        let mut cached_children = dir.cached_children().write();
53
54        // Look up entries that either exist under `/proc/<pid>`
55        // but not under `/proc/<pid>/task/<tid>`,
56        // or entries whose contents differ between `/proc/<pid>` and `/proc/<pid>/task/<tid>`.
57        if let Some(child) =
58            lookup_child_from_table(name, &mut cached_children, Self::STATIC_ENTRIES, |f| {
59                (f)(self, dir.this_weak().clone())
60            })
61        {
62            return Ok(child);
63        }
64
65        // For all other children, the content is the same under both `/proc/<pid>` and `/proc/<pid>/task/<tid>`.
66        self.0
67            .lookup_child_locked(&mut cached_children, dir.this_weak().clone(), name)
68    }
69
70    fn populate_children<'a>(
71        &self,
72        dir: &'a ProcDir<Self>,
73    ) -> RwMutexUpgradeableGuard<'a, SlotVec<(String, Arc<dyn Inode>)>> {
74        let mut cached_children = dir.cached_children().write();
75
76        // Populate entries that either exist under `/proc/<pid>`
77        // but not under `/proc/<pid>/task/<tid>`,
78        // or whose contents differ between the two paths.
79        populate_children_from_table(&mut cached_children, Self::STATIC_ENTRIES, |f| {
80            (f)(self, dir.this_weak().clone())
81        });
82
83        // Populate the remaining children that are identical
84        // under both `/proc/<pid>` and `/proc/<pid>/task/<tid>`.
85        self.0
86            .populate_children_locked(&mut cached_children, dir.this_weak().clone());
87
88        cached_children.downgrade()
89    }
90}
