3//! This module offers `/proc/cmdline` file support, which provides
4//! information about arguments passed to the kernel at boot time.
5//!
6//! Reference: <https://man7.org/linux/man-pages/man5/proc_cmdline.5.html>
7
8use aster_util::printer::VmPrinter;
9use ostd::boot::boot_info;
10
11use crate::{
12    fs::{
13        procfs::template::{FileOps, ProcFileBuilder},
14        utils::{Inode, mkmod},
15    },
16    prelude::*,
17};
18
19/// Represents the inode at `/proc/cmdline`.
20pub struct CmdLineFileOps;
21
22impl CmdLineFileOps {
23    pub fn new_inode(parent: Weak<dyn Inode>) -> Arc<dyn Inode> {
24        // Reference:
25        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/cmdline.c#L19>
26        // <https://elixir.bootlin.com/linux/v6.16.5/source/fs/proc/generic.c#L549-L550>
27        ProcFileBuilder::new(Self, mkmod!(a+r))
28            .parent(parent)
29            .build()
30            .unwrap()
31    }
32}
33
34impl FileOps for CmdLineFileOps {
35    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
36        let mut printer = VmPrinter::new_skip(writer, offset);
37
38        // TODO: Parse additional kernel command line information with `bootconfig`.
39        // See <https://docs.kernel.org/admin-guide/bootconfig.html> for details.
40        writeln!(printer, "{}", boot_info().kernel_cmdline)?;
41
42        Ok(printer.bytes_written())
43    }
44}
