63pub fn init_in_first_process(ctx: &Context) {
64    let fs = ctx.thread_local.borrow_fs();
65    let path_resolver = fs.resolver().read();
66
67    // Initialize the file table for the first process.
68    let tty_path = FsPath::try_from("/dev/console").unwrap();
69    let stdin = {
70        let open_args = OpenArgs::from_modes(AccessMode::O_RDONLY, mkmod!(u+r));
71        path_resolver
72            .lookup(&tty_path)
73            .unwrap()
74            .open(open_args)
75            .unwrap()
76    };
77    let stdout = {
78        let open_args = OpenArgs::from_modes(AccessMode::O_WRONLY, mkmod!(u+w));
79        path_resolver
80            .lookup(&tty_path)
81            .unwrap()
82            .open(open_args)
83            .unwrap()
84    };
85    let stderr = {
86        let open_args = OpenArgs::from_modes(AccessMode::O_WRONLY, mkmod!(u+w));
87        path_resolver
88            .lookup(&tty_path)
89            .unwrap()
90            .open(open_args)
91            .unwrap()
92    };
93
94    let mut file_table_ref = ctx.thread_local.borrow_file_table_mut();
95    let mut file_table = file_table_ref.unwrap().write();
96
97    file_table.insert(Arc::new(stdin), FdFlags::empty());
98    file_table.insert(Arc::new(stdout), FdFlags::empty());
99    file_table.insert(Arc::new(stderr), FdFlags::empty());
100}
