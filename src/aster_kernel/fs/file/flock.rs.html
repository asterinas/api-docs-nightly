10/// Represents a file lock (FLOCK) with an owner and type.
11#[derive(Clone, Debug)]
12struct Flock {
13    /// Owner of the lock, which is an opened file descriptor.
14    owner: Weak<dyn FileLike>,
15    /// Type of the lock, either shared or exclusive.
16    type_: FlockType,
17}
18
19/// Represents a Flock item that can be held in a list of file locks.
20/// Each FlockItem contains a lock and a wait queue for threads that are blocked by the lock.
21pub struct FlockItem {
22    lock: Flock,
23    /// A wait queue for any threads that are blocked by this lock.
24    waitqueue: Arc<WaitQueue>,
25}
26
27impl FlockItem {
28    /// Creates a new FlockItem with the specified owner and lock type.
29    pub fn new(owner: &Arc<dyn FileLike>, type_: FlockType) -> Self {
30        Self {
31            lock: Flock {
32                owner: Arc::downgrade(owner),
33                type_,
34            },
35            waitqueue: Arc::new(WaitQueue::new()),
36        }
37    }
38
39    /// Returns the owner of the lock if it exists.
40    pub fn owner(&self) -> Option<Arc<dyn FileLike>> {
41        Weak::upgrade(&self.lock.owner)
42    }
43
44    /// Checks if this lock has the same owner as another lock.
45    pub fn same_owner_with(&self, other: &Self) -> bool {
46        self.lock.owner.ptr_eq(&other.lock.owner)
47    }
48
49    /// Returns true if this lock conflicts with another lock.
50    /// Two locks conflict if they have different owners and at least one of them is an exclusive lock.
51    pub fn conflict_with(&self, other: &Self) -> bool {
52        if self.same_owner_with(other) {
53            return false;
54        }
55        if self.lock.type_ == FlockType::ExclusiveLock
56            || other.lock.type_ == FlockType::ExclusiveLock
57        {
58            return true;
59        }
60        false
61    }
62
63    /// Wakes all threads that are waiting for this lock.
64    pub fn wake_all(&self) {
65        self.waitqueue.wake_all();
66    }
67}
68
69impl Clone for FlockItem {
70    fn clone(&self) -> Self {
71        Self {
72            lock: self.lock.clone(),
73            waitqueue: self.waitqueue.clone(),
74        }
75    }
76}
77
78/// When a FlockItem is dropped, it wakes all threads that are waiting for the lock.
79impl Drop for FlockItem {
80    fn drop(&mut self) {
81        self.waitqueue.wake_all();
82    }
83}
84
85impl Debug for FlockItem {
86    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
87        f.debug_struct("Flock")
88            .field("owner", &self.lock.owner.as_ptr())
89            .field("type_", &self.lock.type_)
90            .finish()
91    }
92}

94/// Represents a list of non-POSIX file advisory locks (FLOCK).
95/// The list is used to manage file locks and resolve conflicts between them.
96pub struct FlockList {
97    inner: Mutex<Vec<FlockItem>>,
98}
99
100impl FlockList {
101    /// Creates a new FlockList.
102    pub fn new() -> Self {
103        Self {
104            inner: Mutex::new(Vec::new()),
105        }
106    }
107
108    /// Attempts to set a lock on the file.
109    ///
110    /// If no conflicting locks exist, the lock is set and the function returns `Ok(())`.
111    /// If a conflicting lock exists:
112    /// - If waker is not `None`, it is added to the conflicting lock's waitqueue, and the function returns `EAGAIN`.
113    /// - If waker is `None`, the function returns `EAGAIN`.
114    fn try_set_lock(&self, req_lock: &FlockItem, waker: Option<&Arc<Waker>>) -> Result<()> {
115        let mut list = self.inner.lock();
116        if let Some(conflict_lock) = list.iter().find(|l| req_lock.conflict_with(l)) {
117            if let Some(waker) = waker {
118                conflict_lock.waitqueue.enqueue(waker.clone());
119            }
120            return_errno_with_message!(Errno::EAGAIN, "the file is locked");
121        } else {
122            match list.iter().position(|l| req_lock.same_owner_with(l)) {
123                Some(idx) => {
124                    list[idx] = req_lock.clone();
125                }
126                None => {
127                    list.push(req_lock.clone());
128                }
129            }
130            Ok(())
131        }
132    }
133
134    /// Sets a lock on the file.
135    ///
136    /// If no conflicting locks exist, the lock is set and the function returns `Ok(())`.
137    /// If is_nonblocking is true and a conflicting lock exists, the function returns `EAGAIN`.
138    /// Otherwise, the function waits until the lock can be acquired or until it is interrupted by a signal.
139    pub fn set_lock(&self, req_lock: FlockItem, is_nonblocking: bool) -> Result<()> {
140        debug!(
141            "set_lock with Flock: {:?}, is_nonblocking: {}",
142            req_lock, is_nonblocking
143        );
144        if is_nonblocking {
145            self.try_set_lock(&req_lock, None)
146        } else {
147            let (waiter, waker) = Waiter::new_pair();
148            waiter.pause_until(|| {
149                let result = self.try_set_lock(&req_lock, Some(&waker));
150                if result.is_err_and(|err| err.error() == Errno::EAGAIN) {
151                    None
152                } else {
153                    Some(result)
154                }
155            })?
156        }
157    }
158
159    /// Unlocks the specified owner, waking any waiting threads.
160    /// If the owner is no longer valid, the lock is removed from the list.
161    /// If the owner is valid, the lock is removed from the list and all threads waiting for the lock are woken.
162    /// The function does nothing if the owner is not found in the list.
163    /// The function is called when the file is closed or the lock is released.
164    pub fn unlock(&self, req_owner: &InodeHandle) {
165        debug!("unlock with owner: {:?}", req_owner as *const InodeHandle);
166        let mut list = self.inner.lock();
167        list.retain(|lock| {
168            if let Some(owner) = lock.owner() {
169                if ptr::eq(
170                    Arc::as_ptr(&owner) as *const InodeHandle,
171                    req_owner as *const InodeHandle,
172                ) {
173                    lock.wake_all(); // Wake all threads waiting for this lock.
174                    false // Remove lock from the list.
175                } else {
176                    true // Keep lock in the list.
177                }
178            } else {
179                false // Remove lock if the owner is no longer valid.
180            }
181        });
182    }
183}
184
185impl Default for FlockList {
186    fn default() -> Self {
187        Self::new()
188    }
189}

191/// Represents the type of a Flock - either shared or exclusive.
192#[repr(u16)]
193#[derive(Clone, Copy, Debug, PartialEq)]
194pub enum FlockType {
195    /// Represents a shared lock.
196    SharedLock = 0,
197    /// Represents an exclusive lock.
198    ExclusiveLock = 1,
199}
