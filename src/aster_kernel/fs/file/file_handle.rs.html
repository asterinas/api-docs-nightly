21/// The basic operations defined on a file
22pub trait FileLike: Pollable + Send + Sync + Any {
23    /// Reads data from this file.
24    ///
25    /// By default, this method returns `EBADF`
26    /// if the file is not readable,
27    /// or `EINVAL` if the file type does not support `read`.
28    /// These are distinct conditions:
29    /// [`access_mode`] may say that a file is readable,
30    /// while the file type still does not support `read`,
31    /// as with epoll files and pid files.
32    ///
33    /// Implementors should override this method if the file type supports reads.
34    /// An overriding implementation must check whether [`access_mode`] is readable
35    /// before performing the operation.
36    ///
37    /// [`access_mode`]: FileLike::access_mode
38    fn read(&self, writer: &mut VmWriter) -> Result<usize> {
39        if !self.access_mode().is_readable() {
40            return_errno_with_message!(Errno::EBADF, "the file is not opened for reading");
41        }
42        return_errno_with_message!(Errno::EINVAL, "read is not supported for this file type");
43    }
44
45    /// Writes data to this file.
46    ///
47    /// By default, this method returns `EBADF`
48    /// if the file is not writable,
49    /// or `EINVAL` if the file type does not support `write`.
50    /// These are distinct conditions:
51    /// [`access_mode`] may say that a file is writable,
52    /// while the file type still does not support `write`,
53    /// as with epoll files and pid files.
54    ///
55    /// Implementors should override this method if the file type supports writes.
56    /// An overriding implementation must check whether [`access_mode`] is writable
57    /// before performing the operation.
58    ///
59    /// [`access_mode`]: FileLike::access_mode
60    fn write(&self, reader: &mut VmReader) -> Result<usize> {
61        if !self.access_mode().is_writable() {
62            return_errno_with_message!(Errno::EBADF, "the file is not opened for writing");
63        }
64        return_errno_with_message!(Errno::EINVAL, "write is not supported for this file type");
65    }
66
67    /// Read at the given file offset.
68    ///
69    /// The file must be seekable to support `read_at`.
70    /// Unlike [`read`], `read_at` will not change the file offset.
71    ///
72    /// [`read`]: FileLike::read
73    fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
74        return_errno_with_message!(Errno::ESPIPE, "read_at is not supported");
75    }
76
77    /// Write at the given file offset.
78    ///
79    /// The file must be seekable to support `write_at`.
80    /// Unlike [`write`], `write_at` will not change the file offset.
81    /// If the file is append-only, the `offset` will be ignored.
82    ///
83    /// [`write`]: FileLike::write
84    fn write_at(&self, offset: usize, reader: &mut VmReader) -> Result<usize> {
85        return_errno_with_message!(Errno::ESPIPE, "write_at is not supported");
86    }
87
88    fn ioctl(&self, raw_ioctl: RawIoctl) -> Result<i32> {
89        // `ENOTTY` means that "The specified operation does not apply to the kind of object that
90        // the file descriptor references".
91        // Reference: <https://man7.org/linux/man-pages/man2/ioctl.2.html>.
92        return_errno_with_message!(Errno::ENOTTY, "ioctl is not supported");
93    }
94
95    /// Obtains the mappable object to map this file into the user address space.
96    ///
97    /// If this file has a corresponding mappable object of [`Mappable`],
98    /// then it can be either an inode or an MMIO region.
99    fn mappable(&self) -> Result<Mappable> {
100        // `ENODEV` means that "The underlying filesystem of the specified file does not support
101        // memory mapping".
102        // Reference: <https://man7.org/linux/man-pages/man2/mmap.2.html>.
103        return_errno_with_message!(Errno::ENODEV, "the file is not mappable");
104    }
105
106    fn resize(&self, new_size: usize) -> Result<()> {
107        return_errno_with_message!(Errno::EINVAL, "resize is not supported");
108    }
109
110    fn status_flags(&self) -> StatusFlags {
111        StatusFlags::empty()
112    }
113
114    fn set_status_flags(&self, _new_flags: StatusFlags) -> Result<()> {
115        return_errno_with_message!(Errno::EINVAL, "set_status_flags is not supported");
116    }
117
118    /// Returns the access mode of this file.
119    ///
120    /// The access mode indicates whether the file descriptor is readable or writable.
121    /// Readability is required for operations
122    /// such as [`read`], [`read_at`], and read-only memory mappings.
123    /// Writability is required for operations
124    /// such as [`write`], [`write_at`], [`resize`], [`fallocate`],
125    /// and writable memory mappings.
126    ///
127    /// For inode-backed files,
128    /// their access modes are determined dynamically for each opened file.
129    /// For special files such as epoll files, eventfd files, and pid files,
130    /// they are determined statically by the file types.
131    ///
132    /// [`read`]: FileLike::read
133    /// [`read_at`]: FileLike::read_at
134    /// [`write`]: FileLike::write
135    /// [`write_at`]: FileLike::write_at
136    /// [`resize`]: FileLike::resize
137    /// [`fallocate`]: FileLike::fallocate
138    fn access_mode(&self) -> AccessMode;
139
140    fn seek(&self, seek_from: SeekFrom) -> Result<usize> {
141        return_errno_with_message!(Errno::ESPIPE, "seek is not supported");
142    }
143
144    fn fallocate(&self, mode: FallocMode, offset: usize, len: usize) -> Result<()> {
145        return_errno_with_message!(Errno::EOPNOTSUPP, "fallocate is not supported");
146    }
147
148    fn as_socket(&self) -> Option<&dyn Socket> {
149        None
150    }
151
152    fn path(&self) -> &Path;
153
154    /// Dumps information to appear in the `fdinfo` file under procfs.
155    ///
156    /// This method must not break atomic mode because it will be called with the file table's spin
157    /// lock held. There are two strategies for implementing this method:
158    ///  - If the necessary information can be obtained without breaking atomic mode, the method
159    ///    can collect and return the information directly. `Arc<Self>` should be dropped and
160    ///    should not appear in the returned `Box<dyn Display>`.
161    ///  - Otherwise, if the file can be dropped asynchronously in another process, the method can
162    ///    return a `Box<dyn Display>` containing the `Arc<Self>`, so that the information can be
163    ///    collected later in its `Display::display()` method, after dropping the file table's spin
164    ///    lock.
165    fn dump_proc_fdinfo(self: Arc<Self>, fd_flags: FdFlags) -> Box<dyn Display>;
166}
167
168impl dyn FileLike {
169    pub fn downcast_ref<T: FileLike>(&self) -> Option<&T> {
170        (self as &dyn Any).downcast_ref::<T>()
171    }
172
173    pub fn read_bytes(&self, buf: &mut [u8]) -> Result<usize> {
174        let mut writer = VmWriter::from(buf).to_fallible();
175        self.read(&mut writer)
176    }
177
178    pub fn write_bytes(&self, buf: &[u8]) -> Result<usize> {
179        let mut reader = VmReader::from(buf).to_fallible();
180        self.write(&mut reader)
181    }
182
183    pub fn read_bytes_at(&self, offset: usize, buf: &mut [u8]) -> Result<usize> {
184        let mut writer = VmWriter::from(buf).to_fallible();
185        self.read_at(offset, &mut writer)
186    }
187
188    #[expect(dead_code)]
189    pub fn write_bytes_at(&self, offset: usize, buf: &[u8]) -> Result<usize> {
190        let mut reader = VmReader::from(buf).to_fallible();
191        self.write_at(offset, &mut reader)
192    }
193
194    pub fn as_socket_or_err(&self) -> Result<&dyn Socket> {
195        self.as_socket()
196            .ok_or_else(|| Error::with_message(Errno::ENOTSOCK, "the file is not a socket"))
197    }
198
199    pub fn as_inode_handle_or_err(&self) -> Result<&InodeHandle> {
200        self.downcast_ref().ok_or_else(|| {
201            Error::with_message(Errno::EINVAL, "the file is not related to an inode")
202        })
203    }
204}

206/// An object that may be memory mapped into the user address space.
207#[derive(Clone, Debug)]
208pub enum Mappable {
209    /// A VMO (i.e., page cache).
210    Vmo(Arc<Vmo>),
211    /// An MMIO region.
212    IoMem(IoMem),
