7/// The data structure in a directory's data block. It is stored in a linked list.
8///
9/// Each entry contains the name of the entry, the inode number, the file type,
10/// and the distance within the directory file to the next entry.
11#[derive(Clone, Debug)]
12pub struct DirEntry {
13    /// The header part.
14    header: DirEntryHeader,
15    /// Name of the entry, up to 255 bytes (excluding the null terminator).
16    name: CStr256,
17}
18
19impl DirEntry {
20    const ALIGN: usize = 4;
21    const HEADER_LEN: usize = size_of::<DirEntryHeader>();
22    const PARENT_OFFSET: usize = Self::HEADER_LEN + Self::ALIGN;
23
24    /// Constructs a new `DirEntry` object with the specified inode (`ino`),
25    /// name (`name`), and file type (`inode_type`).
26    pub(super) fn new(ino: u32, name: &str, inode_type: InodeType) -> Self {
27        let header = DirEntryHeader::new(ino, inode_type, name.len());
28        Self {
29            header,
30            name: CStr256::from(name),
31        }
32    }
33
34    /// Constructs a `DirEntry` with the name "." and `self_ino` as its inode.
35    pub(super) fn self_entry(self_ino: u32) -> Self {
36        Self::new(self_ino, ".", InodeType::Dir)
37    }
38
39    /// Constructs a `DirEntry` with the name ".." and `parent_ino` as its inode.
40    pub(super) fn parent_entry(parent_ino: u32) -> Self {
41        Self::new(parent_ino, "..", InodeType::Dir)
42    }
43
44    /// Returns a reference to the header.
45    fn header(&self) -> &DirEntryHeader {
46        &self.header
47    }
48
49    /// Returns the inode number.
50    pub fn ino(&self) -> u32 {
51        self.header.ino
52    }
53
54    /// Returns the name.
55    pub fn name(&self) -> &str {
56        self.name.as_str().unwrap()
57    }
58
59    /// Returns the inode type of the entry.
60    pub fn type_(&self) -> InodeType {
61        InodeType::from(DirEntryFileType::from(self.header.inode_type))
62    }
63
64    /// Returns the distance to the next entry.
65    pub fn record_len(&self) -> usize {
66        self.header.record_len as _
67    }
68
69    /// Modifies the distance to the next entry.
70    pub(super) fn set_record_len(&mut self, record_len: usize) {
71        debug_assert!(record_len >= self.actual_len());
72        self.header.record_len = record_len as _;
73    }
74
75    /// Returns the actual length of the current entry.
76    pub(super) fn actual_len(&self) -> usize {
77        (Self::HEADER_LEN + self.header.name_len as usize).align_up(Self::ALIGN)
78    }
79}

81/// The header of `DirEntry`.
82#[repr(C)]
83#[derive(Clone, Copy, Debug, Pod)]
84pub(super) struct DirEntryHeader {
85    /// Inode number
86    ino: u32,
87    /// Directory entry length
88    record_len: u16,
89    /// Name Length
90    name_len: u8,
91    /// Type indicator
92    inode_type: u8,
93}
94
95impl DirEntryHeader {
96    pub(super) fn new(ino: u32, inode_type: InodeType, name_len: usize) -> Self {
97        debug_assert!(name_len <= MAX_FNAME_LEN);
98
99        let record_len = (DirEntry::HEADER_LEN + name_len).align_up(DirEntry::ALIGN) as u16;
100        Self {
101            ino,
102            record_len,
103            name_len: name_len as u8,
104            inode_type: DirEntryFileType::from(inode_type) as _,
105        }
106    }
107}
108
109/// The type indicator in the `DirEntry`.
110#[repr(u8)]
111#[derive(Copy, Clone, Debug, Eq, PartialEq)]
112enum DirEntryFileType {
113    Unknown = 0,
114    File = 1,
115    Dir = 2,
116    Char = 3,
117    Block = 4,
118    Fifo = 5,
119    Socket = 6,
120    Symlink = 7,
121}
122
123impl From<u8> for DirEntryFileType {
124    fn from(value: u8) -> Self {
125        // If the file type is an undefined value, set it to `Unknown`.
126        // See <https://elixir.bootlin.com/linux/v6.13/source/fs/fs_types.c#L37>.
127        match value {
128            1 => Self::File,
129            2 => Self::Dir,
130            3 => Self::Char,
131            4 => Self::Block,
132            5 => Self::Fifo,
133            6 => Self::Socket,
134            7 => Self::Symlink,
135            _ => Self::Unknown,
136        }
137    }
138}
139
140impl From<InodeType> for DirEntryFileType {
141    fn from(inode_type: InodeType) -> Self {
142        match inode_type {
143            InodeType::Unknown => DirEntryFileType::Unknown,
144            InodeType::NamedPipe => Self::Fifo,
145            InodeType::CharDevice => Self::Char,
146            InodeType::Dir => Self::Dir,
147            InodeType::BlockDevice => Self::Block,
148            InodeType::File => Self::File,
149            InodeType::SymLink => Self::Symlink,
150            InodeType::Socket => Self::Socket,
151        }
152    }
153}
154
155impl From<DirEntryFileType> for InodeType {
156    fn from(file_type: DirEntryFileType) -> Self {
157        match file_type {
158            DirEntryFileType::Fifo => Self::NamedPipe,
159            DirEntryFileType::Char => Self::CharDevice,
160            DirEntryFileType::Dir => Self::Dir,
161            DirEntryFileType::Block => Self::BlockDevice,
162            DirEntryFileType::File => Self::File,
163            DirEntryFileType::Symlink => Self::SymLink,
164            DirEntryFileType::Socket => Self::Socket,
165            DirEntryFileType::Unknown => Self::Unknown,
166        }
167    }
168}

170/// A reader for reading `DirEntry` from the page cache.
171pub struct DirEntryReader<'a> {
172    page_cache: &'a PageCache,
173    from_offset: usize,
174    name_buf: Option<[u8; MAX_FNAME_LEN]>,
175}
176
177/// An iterator for iterating `DirEntryItem` from the
178/// page cache given a start offset.
179pub(super) struct DirEntryIter<'a> {
180    page_cache: &'a PageCache,
181    offset: usize,
182}
183
184impl<'a> DirEntryReader<'a> {
185    /// Constructs a reader with the given page cache and offset.
186    pub(super) fn new(page_cache: &'a PageCache, from_offset: usize) -> Self {
187        Self {
188            page_cache,
189            from_offset,
190            name_buf: None,
191        }
192    }
193
194    /// Returns an iterator for iterating `DirEntryItem`s.
195    pub fn iter(&self) -> DirEntryIter<'a> {
196        DirEntryIter {
197            page_cache: self.page_cache,
198            offset: self.from_offset,
199        }
200    }
201
202    /// Returns an iterator for iterating `DirEntry`s.
203    pub fn iter_entries(&'a mut self) -> impl Iterator<Item = DirEntry> + 'a {
204        let iter = self.iter();
205        iter.filter_map(|entry_item| match self.read_name(&entry_item) {
206            Ok(name_buf) => Some(DirEntry {
207                header: entry_item.header,
208                name: CStr256::from(name_buf),
209            }),
210            Err(_) => None,
211        })
212    }
213
214    /// Whether the directory contains an entry with the given name.
215    pub fn contains_entry(&mut self, name: &str) -> bool {
216        let mut iter = self.iter();
217        iter.any(|entry_item| {
218            if entry_item.name_len() != name.len() {
219                return false;
220            }
221            match self.read_name(&entry_item) {
222                Ok(name_buf) => name_buf == name.as_bytes(),
223                Err(_) => false,
224            }
225        })
226    }
227
228    /// Returns the target entry with the given name.
229    pub fn find_entry_item(&mut self, name: &str) -> Option<DirEntryItem> {
230        let mut iter = self.iter();
231        let name_len = name.len();
232        let name_bytes = name.as_bytes();
233        iter.find(|entry_item| {
234            if entry_item.name_len() != name_len {
235                return false;
236            }
237
238            match self.read_name(entry_item) {
239                Ok(name_buf) => name_buf == name_bytes,
240                Err(_) => false,
241            }
242        })
243    }
244
245    /// Returns the number of entries in the directory.
246    pub fn entry_count(&self) -> usize {
247        self.iter().count()
248    }
249
250    /// Reads the name of the entry from the page cache to the inner buffer.
251    fn read_name(&mut self, entry_item: &DirEntryItem) -> Result<&[u8]> {
252        if self.name_buf.is_none() {
253            self.name_buf = Some([0; MAX_FNAME_LEN]);
254        }
255
256        let name_len = entry_item.name_len();
257        let name_buf = &mut self.name_buf.as_mut().unwrap()[..name_len];
258
259        let offset = entry_item.offset + DirEntry::HEADER_LEN;
260        self.page_cache.pages().read_bytes(offset, name_buf)?;
261        Ok(name_buf)
262    }
263}
264
265impl DirEntryIter<'_> {
266    /// Reads a `DirEntryItem` at the current offset.
267    fn read_next_dir_entry(&mut self) -> Option<DirEntryItem> {
268        if self.offset >= self.page_cache.pages().size() {
269            return None;
270        };
271
272        let header = self
273            .page_cache
274            .pages()
275            .read_val::<DirEntryHeader>(self.offset)
276            .ok()?;
277
278        if header.record_len == 0 {
279            return None;
280        }
281
282        let next_dir_entry = DirEntryItem {
283            header,
284            offset: self.offset,
285        };
286
287        self.offset += next_dir_entry.header.record_len as usize;
288        Some(next_dir_entry)
289    }
290}
291
292impl Iterator for DirEntryIter<'_> {
293    type Item = DirEntryItem;
294
295    fn next(&mut self) -> Option<Self::Item> {
296        loop {
297            let next_item = self.read_next_dir_entry()?;
298
299            // Skip an entry if it is not used (ino == 0, see [1]) or its file type is unknown.
300            //
301            // [1]: <https://www.nongnu.org/ext2-doc/ext2.html#linked-directory-entry-structure>.
302            if next_item.header.ino == 0
303                || DirEntryFileType::from(next_item.header.inode_type) == DirEntryFileType::Unknown
304            {
305                continue;
306            }
307
308            return Some(next_item);
309        }
310    }
311}

313/// A directory entry item describes the basic information of a `DirEntry`,
314/// including the entry header and the entry's offset. The entry name is not
315/// present and will be retrieved from the page cache when needed.
316#[derive(Clone, Copy, Debug)]
317pub(super) struct DirEntryItem {
318    header: DirEntryHeader,
319    offset: usize,
320}
321
322impl DirEntryItem {
323    /// Returns a reference to the header.
324    pub fn header(&self) -> &DirEntryHeader {
325        &self.header
326    }
327
328    /// Returns the offset of the entry.
329    pub fn offset(&self) -> usize {
330        self.offset
331    }
332
333    /// Returns the inode number.
334    pub fn ino(&self) -> u32 {
335        self.header.ino as _
336    }
337
338    /// Modifies the inode number.
339    pub fn set_ino(&mut self, ino: u32) {
340        self.header.ino = ino as _;
341    }
342
343    /// Returns the length of the name.
344    pub fn name_len(&self) -> usize {
345        self.header.name_len as _
346    }
347
348    /// Returns the inode type of the entry.
349    pub fn type_(&self) -> InodeType {
350        InodeType::from(DirEntryFileType::from(self.header.inode_type))
351    }
352
353    /// Returns the distance to the next entry.
354    pub fn record_len(&self) -> usize {
355        self.header.record_len as _
356    }
357
358    /// Modifies the distance to the next entry.
359    pub fn set_record_len(&mut self, record_len: usize) {
360        debug_assert!(record_len >= self.actual_len());
361        self.header.record_len = record_len as _;
362    }
363
364    /// Returns the actual length of the current entry.
365    pub fn actual_len(&self) -> usize {
366        (DirEntry::HEADER_LEN + self.name_len()).align_up(DirEntry::ALIGN)
367    }
368
369    /// Returns the length of the gap between the current entry and the next entry.
370    pub fn gap_len(&self) -> usize {
371        self.record_len() - self.actual_len()
372    }
373}

375/// A writer for modifying `DirEntry` of the page cache.
376pub struct DirEntryWriter<'a> {
377    page_cache: &'a PageCache,
378    offset: usize,
379    name_buf: Option<[u8; MAX_FNAME_LEN]>,
380}
381
382impl<'a> DirEntryWriter<'a> {
383    /// Constructs a writer with the given page cache and offset.
384    pub(super) fn new(page_cache: &'a PageCache, from_offset: usize) -> Self {
385        Self {
386            page_cache,
387            offset: from_offset,
388            name_buf: None,
389        }
390    }
391
392    /// Writes a `DirEntry` at the current offset. The name is written after the header.
393    pub fn write_entry(&mut self, header: &DirEntryHeader, name: &str) -> Result<()> {
394        self.page_cache.pages().write_val(self.offset, header)?;
395        self.page_cache
396            .pages()
397            .write_bytes(self.offset + DirEntry::HEADER_LEN, name.as_bytes())?;
398
399        self.offset += header.record_len as usize;
400        Ok(())
401    }
402
403    /// Writes the header of a `DirEntry` at the current offset.
404    pub fn write_header_only(&mut self, header: &DirEntryHeader) -> Result<()> {
405        self.page_cache.pages().write_val(self.offset, header)?;
406        self.offset += header.record_len as usize;
407        Ok(())
408    }
409
410    /// Initializes two special `DirEntry`s ("." and "..") with the given inode numbers.
411    pub fn init_dir(&mut self, self_ino: u32, parent_ino: u32) -> Result<()> {
412        debug_assert!(self.page_cache.pages().size() == 0 && self.offset == 0);
413
414        self.page_cache.pages().resize(BLOCK_SIZE)?;
415
416        let self_header = DirEntryHeader::new(self_ino, InodeType::Dir, 1);
417        self.write_entry(&self_header, ".")?;
418        debug_assert_eq!(self.offset, DirEntry::PARENT_OFFSET);
419
420        let mut parent_header = DirEntryHeader::new(parent_ino, InodeType::Dir, 2);
421        parent_header.record_len = (BLOCK_SIZE - self.offset) as _;
422        self.write_entry(&parent_header, "..")
423    }
424
425    /// Appends a new `DirEntry` starting from the current offset.
426    ///
427    /// If there is a gap between existing entries, inserts the new entry into the gap;
428    /// If there is no available space, expands the size and appends the new entry at the end.
429    /// `check_existence` is determined to check if the new entry already exists.
430    pub fn append_new_entry(
431        &mut self,
432        header: DirEntryHeader,
433        name: &str,
434        check_existence: bool,
435    ) -> Result<()> {
436        let name_len = name.len();
437        debug_assert_eq!(header.name_len as usize, name_len);
438        let name_bytes = name.as_bytes();
439        let mut entry_item_with_enough_gap = None;
440        for entry_item in DirEntryReader::new(self.page_cache, self.offset).iter() {
441            if entry_item_with_enough_gap.is_none()
442                && entry_item.gap_len() >= header.record_len as usize
443            {
444                entry_item_with_enough_gap = Some(entry_item);
445                if !check_existence {
446                    break;
447                }
448            }
449
450            if check_existence
451                && entry_item.name_len() == name_len
452                && self.read_name(&entry_item)? == name_bytes
453            {
454                return_errno!(Errno::EEXIST);
455            }
456        }
457
458        if let Some(entry_item) = entry_item_with_enough_gap {
459            self.append_entry_in_the_gap(entry_item, header, name)?;
460        } else {
461            self.append_entry_in_the_end(header, name)?;
462        }
463        Ok(())
464    }
465
466    fn append_entry_in_the_gap(
467        &mut self,
468        mut entry_with_enough_gap: DirEntryItem,
469        mut header: DirEntryHeader,
470        name: &str,
471    ) -> Result<()> {
472        // Write in the gap between existing entries.
473        header.record_len = entry_with_enough_gap.gap_len() as u16;
474        entry_with_enough_gap.set_record_len(entry_with_enough_gap.actual_len());
475
476        self.offset = entry_with_enough_gap.offset;
477        self.write_header_only(&entry_with_enough_gap.header)?;
478        self.write_entry(&header, name)
479    }
480
481    fn append_entry_in_the_end(&mut self, mut header: DirEntryHeader, name: &str) -> Result<()> {
482        // Resize and append it at the new block.
483        let old_size = self.page_cache.pages().size();
484        let new_size = old_size + BLOCK_SIZE;
485        self.page_cache.resize(new_size)?;
486        header.record_len = BLOCK_SIZE as _;
487
488        self.offset = old_size;
489        self.write_entry(&header, name)
490    }
491
492    /// Removes and returns an existing `DirEntry` indicated by `name`.
493    pub fn remove_entry(&mut self, name: &str) -> Result<DirEntryItem> {
494        let mut pre_entry_item = None;
495        let name_len = name.len();
496        let name_bytes = name.as_bytes();
497        let mut iter = DirEntryReader::new(self.page_cache, DirEntry::PARENT_OFFSET).iter();
498        let Some(target_entry_item) = iter.find(|entry| {
499            if entry.offset < self.offset {
500                pre_entry_item = Some(*entry);
501            }
502
503            entry.offset == self.offset
504                && entry.name_len() == name_len
505                && self.read_name(entry).unwrap() == name_bytes
506        }) else {
507            return_errno!(Errno::ENOENT);
508        };
509        let is_last_entry = iter.next().is_none();
510
511        let mut pre_entry_item = pre_entry_item.unwrap();
512        let pre_offset = pre_entry_item.offset;
513        if is_last_entry {
514            // Shrink the size.
515            let new_size = pre_offset.align_up(BLOCK_SIZE);
516            self.page_cache.resize(new_size)?;
517            pre_entry_item.set_record_len(new_size - pre_offset);
518            self.offset = pre_offset;
519            self.write_header_only(&pre_entry_item.header)?;
520        } else {
521            // Update the previous entry.
522            pre_entry_item
523                .set_record_len(pre_entry_item.record_len() + target_entry_item.record_len());
524            self.offset = pre_offset;
525            self.write_header_only(&pre_entry_item.header)?;
526        }
527
528        Ok(target_entry_item)
529    }
530
531    /// Renames the `DirEntry` from `old_name` to the `new_name` from the current offset.
532    ///
533    /// It will moves the `DirEntry` to another position,
534    /// if the record length is not big enough.
535    pub fn rename_entry(&mut self, old_name: &str, new_name: &str) -> Result<()> {
536        let entry_item = DirEntryReader::new(self.page_cache, self.offset)
537            .find_entry_item(old_name)
538            .ok_or(Error::new(Errno::ENOENT))?;
539        let old_record_len = entry_item.record_len();
540
541        let mut new_entry_header =
542            DirEntryHeader::new(entry_item.ino(), entry_item.type_(), new_name.len());
543        if new_entry_header.record_len as usize <= old_record_len {
544            // Just rename the entry.
545            new_entry_header.record_len = old_record_len as _;
546            self.offset = entry_item.offset;
547            self.write_entry(&new_entry_header, new_name)?;
548        } else {
549            // Move to another position.
550            self.offset = entry_item.offset;
551            self.remove_entry(old_name)?;
552            self.offset = DirEntry::PARENT_OFFSET;
553            self.append_new_entry(new_entry_header, new_name, false)?;
554        }
555
556        Ok(())
557    }
558
559    /// Reads the name of the entry from the page cache to the inner buffer.
560    fn read_name(&mut self, item: &DirEntryItem) -> Result<&[u8]> {
561        if self.name_buf.is_none() {
562            self.name_buf = Some([0; MAX_FNAME_LEN]);
563        }
564
565        let name_len = item.name_len();
566        let name_buf = &mut self.name_buf.as_mut().unwrap()[..name_len];
567
568        let offset = item.offset + DirEntry::HEADER_LEN;
569        self.page_cache.pages().read_bytes(offset, name_buf)?;
570
571        Ok(name_buf)
572    }
