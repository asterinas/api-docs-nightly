40pub struct Inode {

43    // This corresponds to the `i_pipe` field in `struct inode` in Linux.

54    pub(super) fn new(
55        ino: u32,
56        block_group_idx: usize,
57        desc: Dirty<InodeDesc>,
58        fs: Weak<Ext2>,
59    ) -> Arc<Self> {
60        Arc::new_cyclic(|weak_self| Self {
61            ino,
62            type_: desc.type_,
63            named_pipe: if desc.type_ == InodeType::NamedPipe {
64                Some(Box::new(Pipe::new()))
65            } else {
66                None
67            },
68            block_group_idx,
69            xattr: match desc.type_ {
70                InodeType::Dir | InodeType::File => {
71                    Some(Xattr::new(desc.acl, weak_self.clone(), fs.clone()))
72                }
73                _ => None,
74            },
75            inner: RwMutex::new(InodeInner::new(desc, weak_self.clone(), fs.clone())),
76            fs,
77            extension: Extension::new(),
78        })
79    }

126    pub fn resize(&self, new_size: usize) -> Result<()> {
127        if self.type_ == InodeType::Dir {
128            return_errno!(Errno::EISDIR);
129        }
130        if self.type_ != InodeType::File {
131            return_errno!(Errno::EINVAL);
132        }
133
134        let inner = self.inner.upread();
135        if new_size == inner.file_size() {
136            return Ok(());
137        }
138
139        let mut inner = inner.upgrade();
140        inner.resize(new_size)?;
141
142        let now = now();
143        inner.set_mtime(now);
144        inner.set_ctime(now);
145
146        Ok(())
147    }

149    pub fn create(
150        &self,
151        name: &str,
152        inode_type: InodeType,
153        file_perm: FilePerm,
154    ) -> Result<Arc<Self>> {
155        if name.len() > MAX_FNAME_LEN {
156            return_errno!(Errno::ENAMETOOLONG);
157        }
158        if self.type_ != InodeType::Dir {
159            return_errno!(Errno::ENOTDIR);
160        }
161
162        let inner = self.inner.upread();
163        if inner.hard_links() == 0 {
164            return_errno_with_message!(Errno::ENOENT, "dir removed");
165        }
166
167        let inode = self
168            .fs()
169            .create_inode(self.block_group_idx, inode_type, file_perm)?;
170        let is_dir = inode_type == InodeType::Dir;
171        if let Err(e) = inode.init(self.ino) {
172            self.fs().free_inode(inode.ino, is_dir).unwrap();
173            return Err(e);
174        }
175
176        let mut inner = inner.upgrade();
177        if let Err(e) = inner.append_new_entry(inode.ino, inode_type, name, true) {
178            self.fs().free_inode(inode.ino, is_dir).unwrap();
179            return Err(e);
180        }
181
182        let now = now();
183        inner.set_mtime(now);
184        inner.set_ctime(now);
185
186        Ok(inode)
187    }

201    pub fn lookup(&self, name: &str) -> Result<Arc<Self>> {
202        if name.len() > MAX_FNAME_LEN {
203            return_errno!(Errno::ENAMETOOLONG);
204        }
205        if self.type_ != InodeType::Dir {
206            return_errno!(Errno::ENOTDIR);
207        }
208
209        let inner = self.inner.read();
210        if inner.hard_links() == 0 {
211            return_errno_with_message!(Errno::ENOENT, "dir removed");
212        }
213
214        let ino = inner
215            .find_entry_item(name)
216            .map(|entry| entry.ino())
217            .ok_or(Error::new(Errno::ENOENT))?;
218        drop(inner);
219        self.fs().lookup_inode(ino)
220    }

222    pub fn link(&self, inode: &Inode, name: &str) -> Result<()> {
223        if name.len() > MAX_FNAME_LEN {
224            return_errno!(Errno::ENAMETOOLONG);
225        }
226        if self.type_ != InodeType::Dir {
227            return_errno!(Errno::ENOTDIR);
228        }
229
230        let inner = self.inner.upread();
231        if inner.hard_links() == 0 {
232            return_errno_with_message!(Errno::ENOENT, "dir removed");
233        }
234
235        let inode_type = inode.inode_type();
236        if inode_type == InodeType::Dir {
237            return_errno!(Errno::EPERM);
238        }
239
240        let mut inner = inner.upgrade();
241        inner.append_new_entry(inode.ino, inode_type, name, true)?;
242        let now = now();
243        inner.set_mtime(now);
244        inner.set_ctime(now);
245        drop(inner);
246
247        let mut inode_inner = inode.inner.write();
248        inode_inner.inc_hard_links();
249        inode_inner.set_ctime(now);
250
251        Ok(())
252    }

254    pub fn unlink(&self, name: &str) -> Result<()> {
255        if is_dot_or_dotdot(name) {
256            return_errno!(Errno::EISDIR);
257        }
258
259        let file = self.lookup(name)?;
260        if file.inode_type() == InodeType::Dir {
261            return_errno!(Errno::EISDIR);
262        }
263
264        let (mut self_inner, mut file_inner) = write_lock_two_inodes(self, &file);
265        // When we got the lock, the dir may have been modified by another thread
266        if self_inner.hard_links() == 0 {
267            return_errno_with_message!(Errno::ENOENT, "dir removed");
268        }
269        let (offset, new_ino) = self_inner
270            .find_entry_item(name)
271            .map(|entry| (entry.offset(), entry.ino()))
272            .ok_or(Error::new(Errno::ENOENT))?;
273        if file.ino != new_ino {
274            return_errno!(Errno::ENOENT);
275        }
276        let potential_new_file = self.fs().lookup_inode(file.ino)?;
277        if !Arc::ptr_eq(&file, &potential_new_file) {
278            return_errno!(Errno::ENOENT);
279        }
280
281        self_inner.remove_entry_at(name, offset)?;
282        file_inner.dec_hard_links();
283        let now = now();
284        self_inner.set_mtime(now);
285        self_inner.set_ctime(now);
286        file_inner.set_ctime(now);
287
288        Ok(())
289    }

291    pub fn rmdir(&self, name: &str) -> Result<()> {
292        if is_dot(name) {
293            return_errno_with_message!(Errno::EINVAL, "rmdir on .");
294        }
295        if is_dotdot(name) {
296            return_errno_with_message!(Errno::ENOTEMPTY, "rmdir on ..");
297        }
298
299        let dir_inode = self.lookup(name)?;
300        let dir_inner = dir_inode.inner.read();
301        if dir_inner.inode_type() != InodeType::Dir {
302            return_errno!(Errno::ENOTDIR);
303        }
304        if dir_inner.entry_count() > 2 {
305            return_errno!(Errno::ENOTEMPTY);
306        }
307        drop(dir_inner);
308
309        let (mut self_inner, mut dir_inner) = write_lock_two_inodes(self, &dir_inode);
310        // When we got the lock, the dir may have been modified by another thread
311        if self_inner.hard_links() == 0 {
312            return_errno_with_message!(Errno::ENOENT, "dir removed");
313        }
314        let (offset, new_ino) = self_inner
315            .find_entry_item(name)
316            .map(|entry| (entry.offset(), entry.ino()))
317            .ok_or(Error::new(Errno::ENOENT))?;
318        if dir_inode.ino != new_ino {
319            return_errno!(Errno::ENOENT);
320        }
321        let potential_new_dir = self.fs().lookup_inode(dir_inode.ino)?;
322        if !Arc::ptr_eq(&dir_inode, &potential_new_dir) {
323            return_errno!(Errno::ENOENT);
324        }
325        if dir_inner.inode_type() != InodeType::Dir {
326            return_errno!(Errno::ENOTDIR);
327        }
328        if dir_inner.entry_count() > 2 {
329            return_errno!(Errno::ENOTEMPTY);
330        }
331
332        self_inner.remove_entry_at(name, offset)?;
333        let now = now();
334        self_inner.set_mtime(now);
335        self_inner.set_ctime(now);
336        dir_inner.dec_hard_links();
337        dir_inner.dec_hard_links(); // For "."
338
339        Ok(())
340    }

342    /// Rename within its own directory.
343    fn rename_within(&self, old_name: &str, new_name: &str) -> Result<()> {
344        if self.type_ != InodeType::Dir {
345            return_errno!(Errno::ENOTDIR);
346        }
347
348        let self_inner = self.inner.upread();
349        if self_inner.hard_links() == 0 {
350            return_errno_with_message!(Errno::ENOENT, "dir removed");
351        }
352
353        let fs = self.fs();
354        let (src_offset, src_inode, src_inode_typ) = {
355            let entry = self_inner
356                .find_entry_item(old_name)
357                .ok_or(Error::new(Errno::ENOENT))?;
358            (entry.offset(), fs.lookup_inode(entry.ino())?, entry.type_())
359        };
360
361        let Some(dst_ino) = self_inner
362            .find_entry_item(new_name)
363            .map(|entry| entry.ino())
364        else {
365            let mut self_inner = self_inner.upgrade();
366            self_inner.rename_entry_at(old_name, new_name, src_offset)?;
367            let now = now();
368            self_inner.set_mtime(now);
369            self_inner.set_ctime(now);
370            drop(self_inner);
371
372            src_inode.set_ctime(now);
373            return Ok(());
374        };
375        if src_inode.ino == dst_ino {
376            // Same inode, do nothing
377            return Ok(());
378        }
379        let dst_inode = fs.lookup_inode(dst_ino)?;
380        drop(self_inner);
381
382        let (mut self_inner, mut dst_inner) = write_lock_two_inodes(self, &dst_inode);
383        // When we got the lock, the dir may have been modified by another thread
384        if self_inner.hard_links() == 0 {
385            return_errno_with_message!(Errno::ENOENT, "dir removed");
386        }
387
388        let (src_offset, new_src_ino) = self_inner
389            .find_entry_item(old_name)
390            .map(|entry| (entry.offset(), entry.ino()))
391            .ok_or(Error::new(Errno::ENOENT))?;
392        if src_inode.ino != new_src_ino {
393            return_errno!(Errno::ENOENT);
394        }
395        let potential_new_src = fs.lookup_inode(src_inode.ino)?;
396        if !Arc::ptr_eq(&src_inode, &potential_new_src) {
397            return_errno!(Errno::ENOENT);
398        }
399
400        let new_dst_entry = self_inner
401            .find_entry_item(new_name)
402            .ok_or(Error::new(Errno::ENOENT))?;
403        let dst_offset = new_dst_entry.offset();
404        if dst_inode.ino != new_dst_entry.ino() {
405            return_errno!(Errno::ENOENT);
406        }
407        let potential_new_dst = fs.lookup_inode(dst_inode.ino)?;
408        if !Arc::ptr_eq(&dst_inode, &potential_new_dst) {
409            return_errno!(Errno::ENOENT);
410        }
411
412        let dst_inode_typ = new_dst_entry.type_();
413        match (src_inode_typ, dst_inode_typ) {
414            (InodeType::Dir, InodeType::Dir) => {
415                if dst_inner.entry_count() > 2 {
416                    return_errno!(Errno::ENOTEMPTY);
417                }
418            }
419            (InodeType::Dir, _) => {
420                return_errno!(Errno::ENOTDIR);
421            }
422            (_, InodeType::Dir) => {
423                return_errno!(Errno::EISDIR);
424            }
425            _ => {}
426        }
427
428        self_inner.remove_entry_at(new_name, dst_offset)?;
429        self_inner.rename_entry_at(old_name, new_name, src_offset)?;
430        let now = now();
431        self_inner.set_mtime(now);
432        self_inner.set_ctime(now);
433
434        dst_inner.dec_hard_links();
435        if dst_inode_typ == InodeType::Dir {
436            dst_inner.dec_hard_links(); // For "."
437        }
438        dst_inner.set_ctime(now);
439        drop(self_inner);
440        drop(dst_inner);
441
442        src_inode.set_ctime(now);
443
444        Ok(())
445    }

447    pub fn rename(&self, old_name: &str, target: &Inode, new_name: &str) -> Result<()> {
448        if is_dot_or_dotdot(old_name) || is_dot_or_dotdot(new_name) {
449            return_errno!(Errno::EISDIR);
450        }
451        if old_name.len() > MAX_FNAME_LEN || new_name.len() > MAX_FNAME_LEN {
452            return_errno!(Errno::ENAMETOOLONG);
453        }
454
455        // Rename inside the inode
456        if self.ino == target.ino {
457            return self.rename_within(old_name, new_name);
458        }
459
460        let (self_inner, target_inner) = read_lock_two_inodes(self, target);
461        if self_inner.inode_type() != InodeType::Dir || target_inner.inode_type() != InodeType::Dir
462        {
463            return_errno!(Errno::ENOTDIR);
464        }
465        if self_inner.hard_links() == 0 || target_inner.hard_links() == 0 {
466            return_errno_with_message!(Errno::ENOENT, "dir removed");
467        }
468
469        let fs = self.fs();
470        let (src_offset, src_inode, src_inode_typ) = {
471            let entry = self_inner
472                .find_entry_item(old_name)
473                .ok_or(Error::new(Errno::ENOENT))?;
474            (entry.offset(), fs.lookup_inode(entry.ino())?, entry.type_())
475        };
476        // Avoid renaming a directory to a subdirectory of itself
477        if src_inode.ino == target.ino {
478            return_errno!(Errno::EINVAL);
479        }
480        let is_dir = src_inode_typ == InodeType::Dir;
481
482        let Some(dst_ino) = target_inner
483            .find_entry_item(new_name)
484            .map(|entry| entry.ino())
485        else {
486            drop(self_inner);
487            drop(target_inner);
488
489            let mut write_guards = if is_dir {
490                write_lock_multiple_inodes(vec![&src_inode, target, self])
491            } else {
492                write_lock_multiple_inodes(vec![target, self])
493            };
494
495            // When we got the lock, the dir may have been modified by another thread
496            let mut self_inner = write_guards.pop().unwrap();
497            let mut target_inner = write_guards.pop().unwrap();
498            if self_inner.hard_links() == 0 || target_inner.hard_links() == 0 {
499                return_errno_with_message!(Errno::ENOENT, "dir removed");
500            }
501            let (src_offset, new_src_ino) = self_inner
502                .find_entry_item(old_name)
503                .map(|entry| (entry.offset(), entry.ino()))
504                .ok_or(Error::new(Errno::ENOENT))?;
505            if src_inode.ino != new_src_ino {
506                return_errno!(Errno::ENOENT);
507            }
508            let potential_new_src = fs.lookup_inode(src_inode.ino)?;
509            if !Arc::ptr_eq(&src_inode, &potential_new_src) {
510                return_errno!(Errno::ENOENT);
511            }
512
513            self_inner.remove_entry_at(old_name, src_offset)?;
514            target_inner.append_new_entry(src_inode.ino, src_inode_typ, new_name, false)?;
515            let now = now();
516            self_inner.set_mtime(now);
517            self_inner.set_ctime(now);
518            target_inner.set_mtime(now);
519            target_inner.set_ctime(now);
520
521            if is_dir {
522                let mut src_inner = write_guards.pop().unwrap();
523                src_inner.set_parent_ino(target.ino)?;
524                src_inner.set_ctime(now);
525            } else {
526                drop(self_inner);
527                drop(target_inner);
528                src_inode.set_ctime(now);
529            }
530
531            return Ok(());
532        };
533        if src_inode.ino == dst_ino {
534            // Same inode, do nothing
535            return Ok(());
536        }
537        // Avoid renaming a subdirectory to a directory.
538        if self.ino == dst_ino {
539            return_errno!(Errno::ENOTEMPTY);
540        }
541        let dst_inode = fs.lookup_inode(dst_ino)?;
542        drop(self_inner);
543        drop(target_inner);
544
545        let mut write_guards = if is_dir {
546            write_lock_multiple_inodes(vec![&src_inode, &dst_inode, target, self])
547        } else {
548            write_lock_multiple_inodes(vec![&dst_inode, target, self])
549        };
550
551        // When we got the lock, the dir may have been modified by another thread
552        let mut self_inner = write_guards.pop().unwrap();
553        let mut target_inner = write_guards.pop().unwrap();
554        if self_inner.hard_links() == 0 || target_inner.hard_links() == 0 {
555            return_errno_with_message!(Errno::ENOENT, "dir removed");
556        }
557
558        let (src_offset, new_src_ino) = self_inner
559            .find_entry_item(old_name)
560            .map(|entry| (entry.offset(), entry.ino()))
561            .ok_or(Error::new(Errno::ENOENT))?;
562        if src_inode.ino != new_src_ino {
563            return_errno!(Errno::ENOENT);
564        }
565        let potential_new_src = fs.lookup_inode(src_inode.ino)?;
566        if !Arc::ptr_eq(&src_inode, &potential_new_src) {
567            return_errno!(Errno::ENOENT);
568        }
569
570        let new_dst_entry = target_inner
571            .find_entry_item(new_name)
572            .ok_or(Error::new(Errno::ENOENT))?;
573        let dst_offset = new_dst_entry.offset();
574        if dst_inode.ino != new_dst_entry.ino() {
575            return_errno!(Errno::ENOENT);
576        }
577        let potential_new_dst = fs.lookup_inode(dst_inode.ino)?;
578        if !Arc::ptr_eq(&dst_inode, &potential_new_dst) {
579            return_errno!(Errno::ENOENT);
580        }
581
582        let mut dst_inner = write_guards.pop().unwrap();
583        let dst_inode_typ = new_dst_entry.type_();
584        match (src_inode_typ, dst_inode_typ) {
585            (InodeType::Dir, InodeType::Dir) => {
586                if dst_inner.entry_count() > 2 {
587                    return_errno!(Errno::ENOTEMPTY);
588                }
589            }
590            (InodeType::Dir, _) => {
591                return_errno!(Errno::ENOTDIR);
592            }
593            (_, InodeType::Dir) => {
594                return_errno!(Errno::EISDIR);
595            }
596            _ => {}
597        }
598
599        self_inner.remove_entry_at(old_name, src_offset)?;
600        target_inner.remove_entry_at(new_name, dst_offset)?;
601        target_inner.append_new_entry(src_inode.ino, src_inode_typ, new_name, false)?;
602        dst_inner.dec_hard_links();
603        let now = now();
604        self_inner.set_mtime(now);
605        self_inner.set_ctime(now);
606        target_inner.set_mtime(now);
607        target_inner.set_ctime(now);
608        dst_inner.set_ctime(now);
609
610        if is_dir {
611            dst_inner.dec_hard_links(); // For "."
612            let mut src_inner = write_guards.pop().unwrap();
613            src_inner.set_parent_ino(target.ino)?;
614            src_inner.set_ctime(now);
615        } else {
616            drop(self_inner);
617            drop(target_inner);
618            drop(dst_inner);
619            src_inode.set_ctime(now);
620        }
621
622        Ok(())
623    }

625    pub fn readdir_at(&self, offset: usize, visitor: &mut dyn DirentVisitor) -> Result<usize> {
626        if self.type_ != InodeType::Dir {
627            return_errno!(Errno::ENOTDIR);
628        }
629
630        let offset_read = {
631            let inner = self.inner.read();
632            if inner.hard_links() == 0 {
633                return_errno_with_message!(Errno::ENOENT, "dir removed");
634            }
635
636            let try_readdir = |offset: &mut usize, visitor: &mut dyn DirentVisitor| -> Result<()> {
637                let mut dir_entry_reader = DirEntryReader::new(&inner.page_cache, *offset);
638                for dir_entry in dir_entry_reader.iter_entries() {
639                    visitor.visit(
640                        dir_entry.name(),
641                        dir_entry.ino() as u64,
642                        dir_entry.type_(),
643                        dir_entry.record_len(),
644                    )?;
645                    *offset += dir_entry.record_len();
646                }
647
648                Ok(())
649            };
650
651            let mut iterate_offset = offset;
652            match try_readdir(&mut iterate_offset, visitor) {
653                Err(e) if iterate_offset == offset => Err(e),
654                _ => Ok(iterate_offset - offset),
655            }?
656        };
657
658        self.set_atime(now());
659
660        Ok(offset_read)
661    }

728    pub fn write_at(&self, offset: usize, reader: &mut VmReader) -> Result<usize> {
729        if self.type_ != InodeType::File {
730            return_errno!(Errno::EISDIR);
731        }
732
733        let inner = self.inner.upread();
734        let file_size = inner.file_size();
735        let new_size = offset + reader.remain();
736
737        let (bytes_written, mut upgraded_inner) = if new_size > file_size {
738            let mut inner = inner.upgrade();
739            let len = inner.extend_write_at(offset, reader)?;
740            (len, inner)
741        } else {
742            let len = inner.write_at(offset, reader)?;
743            (len, inner.upgrade())
744        };
745
746        let now = now();
747        upgraded_inner.set_mtime(now);
748        upgraded_inner.set_ctime(now);
749
750        Ok(bytes_written)
751    }

804    pub fn fallocate(&self, mode: FallocMode, offset: usize, len: usize) -> Result<()> {
805        match mode {
806            FallocMode::PunchHoleKeepSize => {
807                // Make the whole operation atomic
808                let inner = self.inner.write();
809
810                let file_size = inner.file_size();
811                if offset >= file_size {
812                    return Ok(());
813                }
814                let end_offset = file_size.min(offset + len);
815
816                // TODO: Think of a more light-weight approach
817                inner.page_cache.fill_zeros(offset..end_offset)?;
818                Ok(())
819            }
820            // We extend the compatibility here since Ext2 in Linux
821            // does not natively support `Allocate` and `AllocateKeepSize`.
822            FallocMode::Allocate => {
823                let new_size = offset + len;
824                if new_size > self.file_size() {
825                    self.resize(new_size)?;
826                }
827                Ok(())
828            }
829            FallocMode::AllocateKeepSize => Ok(()),
830            _ => {
831                return_errno_with_message!(
832                    Errno::EOPNOTSUPP,
833                    "fallocate with the specified flags is not supported"
834                );
835            }
836        }
837    }

916fn read_lock_two_inodes<'a>(
917    this: &'a Inode,
918    other: &'a Inode,
919) -> (
920    RwMutexReadGuard<'a, InodeInner>,
921    RwMutexReadGuard<'a, InodeInner>,
922) {
923    if this.ino < other.ino {
924        let this = this.inner.read();
925        let other = other.inner.read();
926        (this, other)
927    } else {
928        let other = other.inner.read();
929        let this = this.inner.read();
930        (this, other)
931    }

952fn write_lock_multiple_inodes(inodes: Vec<&Inode>) -> Vec<RwMutexWriteGuard<'_, InodeInner>> {
953    // Record the index information of the input
954    let mut ordered_inodes: Vec<(usize, &Inode)> = inodes.into_iter().enumerate().collect();
955    // Sort in ascending order of ino
956    ordered_inodes.sort_unstable_by_key(|&(_, inode)| inode.ino);
957    // Acquire the guards in order, and record by the input index.
958    // This ensures that the output order is consistent with the input.
959    let mut guards = vec![None; ordered_inodes.len()];
960    for (original_idx, inode) in ordered_inodes {
961        guards[original_idx] = Some(Rc::new(inode.inner.write()));
962    }
963    guards
964        .into_iter()
965        .map(|guard| Rc::into_inner(guard.unwrap()).unwrap())
966        .collect()
967}

994    pub fn read_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
995        let (offset, read_len) = {
996            let file_size = self.inode_impl.file_size();
997            let start = file_size.min(offset);
998            let end = file_size.min(offset + writer.avail());
999            (start, end - start)
1000        };
1001
1002        self.page_cache.pages().read(offset, writer)?;
1003        Ok(read_len)
1004    }

1006    pub fn read_direct_at(&self, offset: usize, writer: &mut VmWriter) -> Result<usize> {
1007        debug_assert!(is_block_aligned(offset) && is_block_aligned(writer.avail()));
1008        let (offset, read_len) = {
1009            let file_size = self.inode_impl.file_size();
1010            let start = file_size.min(offset).align_down(BLOCK_SIZE);
1011            let end = file_size
1012                .min(offset + writer.avail())
1013                .align_down(BLOCK_SIZE);
1014            (start, end - start)
1015        };
1016        if read_len == 0 {
1017            return Ok(read_len);
1018        }
1019        self.page_cache.discard_range(offset..offset + read_len);
1020
1021        let start_bid = Bid::from_offset(offset).to_raw() as Ext2Bid;
1022        let buf_nblocks = read_len / BLOCK_SIZE;
1023        self.inode_impl
1024            .read_blocks(start_bid, buf_nblocks, writer)?;
1025
1026        Ok(read_len)
1027    }

1029    pub fn write_at(&self, offset: usize, reader: &mut VmReader) -> Result<usize> {
1030        let write_len = reader.remain();
1031        self.page_cache.pages().write(offset, reader)?;
1032        Ok(write_len)
1033    }

1035    pub fn extend_write_at(&mut self, offset: usize, reader: &mut VmReader) -> Result<usize> {
1036        let write_len = reader.remain();
1037        let new_size = offset + write_len;
1038        self.page_cache.resize(new_size.align_up(BLOCK_SIZE))?;
1039        self.page_cache.pages().write(offset, reader)?;
1040        self.inode_impl.resize(new_size)?;
1041        Ok(write_len)
1042    }

1044    pub fn write_direct_at(&mut self, offset: usize, reader: &mut VmReader) -> Result<usize> {
1045        debug_assert!(is_block_aligned(offset) && is_block_aligned(reader.remain()));
1046        let file_size = self.inode_impl.file_size();
1047        let write_len = reader.remain();
1048        let end_offset = offset + write_len;
1049
1050        let start = offset.min(file_size);
1051        let end = end_offset.min(file_size);
1052        self.page_cache.discard_range(start..end);
1053
1054        if end_offset > file_size {
1055            self.inode_impl.resize(end_offset)?;
1056        }
1057
1058        let start_bid = Bid::from_offset(offset).to_raw() as Ext2Bid;
1059        let buf_nblocks = write_len / BLOCK_SIZE;
1060        self.inode_impl
1061            .write_blocks(start_bid, buf_nblocks, reader)?;
1062
1063        Ok(write_len)
1064    }

1078    }

1394    /// Expands inode size.
1395    ///
1396    /// After a successful expansion, the size will be enlarged to `new_size`,
1397    /// which may result in an increased block count.
1398    fn expand(&mut self, new_size: usize) -> Result<()> {
1399        let new_blocks = self.desc.size_to_blocks(new_size);
1400        let old_blocks = self.desc.blocks_count();
1401
1402        // Expands block count if necessary
1403        if new_blocks > old_blocks {
1404            if new_blocks - old_blocks > self.fs().super_block().free_blocks_count() {
1405                return_errno_with_message!(Errno::ENOSPC, "not enough free blocks");
1406            }
1407            self.expand_blocks(old_blocks..new_blocks)?;
1408        }
1409
1410        // Expands the size
1411        self.update_size(new_size);
1412        Ok(())
1413    }
1414
1415    /// Expands inode blocks.
1416    ///
1417    /// After a successful expansion, the block count will be enlarged to `range.end`.
1418    fn expand_blocks(&mut self, range: Range<Ext2Bid>) -> Result<()> {
1419        let mut current_range = range.clone();
1420        while !current_range.is_empty() {
1421            let Ok(expand_cnt) = self.try_expand_blocks(current_range.clone()) else {
1422                self.shrink_blocks(range.start..current_range.start);
1423                return_errno_with_message!(Errno::ENOSPC, "can not allocate blocks");
1424            };
1425            current_range.start += expand_cnt;
1426        }
1427
1428        Ok(())
1429    }

1431    /// Attempts to expand a range of blocks and returns the number of consecutive
1432    /// blocks successfully allocated.
1433    ///
1434    /// Note that the returned number may be less than the requested range if there
1435    /// isn't enough consecutive space available or if there is a necessity to allocate
1436    /// indirect blocks.
1437    fn try_expand_blocks(&mut self, range: Range<Ext2Bid>) -> Result<Ext2Bid> {
1438        // Calculates the maximum number of consecutive blocks that can be allocated in
1439        // this round, as well as the number of additional indirect blocks required for
1440        // the allocation.
1441        let (max_cnt, indirect_cnt) = {
1442            let bid_path = BidPath::from(range.start);
1443            let max_cnt = (range.len() as Ext2Bid).min(bid_path.cnt_to_next_indirect());
1444            let indirect_cnt = match bid_path {
1445                BidPath::Direct(_) => 0,
1446                BidPath::Indirect(0) => 1,
1447                BidPath::Indirect(_) => 0,
1448                BidPath::DbIndirect(0, 0) => 2,
1449                BidPath::DbIndirect(_, 0) => 1,
1450                BidPath::DbIndirect(_, _) => 0,
1451                BidPath::TbIndirect(0, 0, 0) => 3,
1452                BidPath::TbIndirect(_, 0, 0) => 2,
1453                BidPath::TbIndirect(_, _, 0) => 1,
1454                BidPath::TbIndirect(_, _, _) => 0,
1455            };
1456            (max_cnt, indirect_cnt)
1457        };
1458
1459        // Calculates the block_group_idx to advise the filesystem on which group
1460        // to prioritize for allocation.
1461        let block_group_idx = self
1462            .last_alloc_device_bid
1463            .map_or(self.inode().block_group_idx, |id| {
1464                ((id + 1) / self.fs().blocks_per_group()) as usize
1465            });
1466
1467        // Allocates the blocks only, no indirect blocks are required.
1468        if indirect_cnt == 0 {
1469            let device_range = self
1470                .fs()
1471                .alloc_blocks(block_group_idx, max_cnt)
1472                .ok_or_else(|| Error::new(Errno::ENOSPC))?;
1473            if let Err(e) = self.set_device_range(range.start, device_range.clone()) {
1474                self.fs().free_blocks(device_range).unwrap();
1475                return Err(e);
1476            }
1477
1478            self.last_alloc_device_bid = Some(device_range.end - 1);
1479            let nr_blocks = device_range.len() as Ext2Bid;
1480            self.desc.set_data_blocks(range.start + nr_blocks);
1481            return Ok(nr_blocks);
1482        }
1483
1484        // Allocates the required additional indirect blocks and at least one block.
1485        let (indirect_bids, device_range) = {
1486            let mut indirect_bids: Vec<Ext2Bid> = Vec::with_capacity(indirect_cnt as usize);
1487            let mut total_cnt = max_cnt + indirect_cnt;
1488            let mut device_range: Option<Range<Ext2Bid>> = None;
1489            while device_range.is_none() {
1490                let Some(mut range) = self.fs().alloc_blocks(block_group_idx, total_cnt) else {
1491                    for indirect_bid in indirect_bids.iter() {
1492                        self.fs()
1493                            .free_blocks(*indirect_bid..*indirect_bid + 1)
1494                            .unwrap();
1495                    }
1496                    return_errno!(Errno::ENOSPC);
1497                };
1498                total_cnt -= range.len() as Ext2Bid;
1499
1500                // Stores the bids for indirect blocks.
1501                while (indirect_bids.len() as Ext2Bid) < indirect_cnt && !range.is_empty() {
1502                    indirect_bids.push(range.start);
1503                    range.start += 1;
1504                }
1505
1506                if !range.is_empty() {
1507                    device_range = Some(range);
1508                }
1509            }
1510
1511            (indirect_bids, device_range.unwrap())
1512        };
1513
1514        if let Err(e) = self.set_indirect_bids(range.start, &indirect_bids) {
1515            self.free_indirect_blocks_required_by(range.start).unwrap();
1516            return Err(e);
1517        }
1518
1519        if let Err(e) = self.set_device_range(range.start, device_range.clone()) {
1520            self.fs().free_blocks(device_range).unwrap();
1521            self.free_indirect_blocks_required_by(range.start).unwrap();
1522            return Err(e);
1523        }
1524
1525        self.last_alloc_device_bid = Some(device_range.end - 1);
1526        let nr_blocks = device_range.len() as Ext2Bid;
1527        self.desc.set_data_blocks(range.start + nr_blocks);
1528        Ok(nr_blocks)
1529    }

1531    /// Sets the device block IDs for a specified range.
1532    ///
1533    /// It updates the mapping between the file's block IDs and the device's block IDs
1534    /// starting from `start_bid`. It maps each block ID in the file to the corresponding
1535    /// block ID on the device based on the provided `device_range`.
1536    fn set_device_range(&mut self, start_bid: Ext2Bid, device_range: Range<Ext2Bid>) -> Result<()> {
1537        match BidPath::from(start_bid) {
1538            BidPath::Direct(idx) => {
1539                let mut block_ptrs = self.block_manager.block_ptrs.write();
1540                for (i, bid) in device_range.enumerate() {
1541                    self.desc.block_ptrs.set_direct(idx as usize + i, bid);
1542                    block_ptrs.set_direct(idx as usize + i, bid);
1543                }
1544            }
1545            BidPath::Indirect(idx) => {
1546                let indirect_bid = self.desc.block_ptrs.indirect();
1547                assert!(indirect_bid != 0);
1548                let mut indirect_blocks = self.block_manager.indirect_blocks.write();
1549                let indirect_block = indirect_blocks.find_mut(indirect_bid)?;
1550                for (i, bid) in device_range.enumerate() {
1551                    indirect_block.write_bid(idx as usize + i, &bid)?;
1552                }
1553            }
1554            BidPath::DbIndirect(lvl1_idx, lvl2_idx) => {
1555                let mut indirect_blocks = self.block_manager.indirect_blocks.write();
1556                let lvl1_indirect_bid = {
1557                    let db_indirect_bid = self.desc.block_ptrs.db_indirect();
1558                    assert!(db_indirect_bid != 0);
1559                    let db_indirect_block = indirect_blocks.find(db_indirect_bid)?;
1560                    db_indirect_block.read_bid(lvl1_idx as usize)?
1561                };
1562                assert!(lvl1_indirect_bid != 0);
1563
1564                let lvl1_indirect_block = indirect_blocks.find_mut(lvl1_indirect_bid)?;
1565                for (i, bid) in device_range.enumerate() {
1566                    lvl1_indirect_block.write_bid(lvl2_idx as usize + i, &bid)?;
1567                }
1568            }
1569            BidPath::TbIndirect(lvl1_idx, lvl2_idx, lvl3_idx) => {
1570                let mut indirect_blocks = self.block_manager.indirect_blocks.write();
1571                let lvl2_indirect_bid = {
1572                    let lvl1_indirect_bid = {
1573                        let tb_indirect_bid = self.desc.block_ptrs.tb_indirect();
1574                        assert!(tb_indirect_bid != 0);
1575                        let tb_indirect_block = indirect_blocks.find(tb_indirect_bid)?;
1576                        tb_indirect_block.read_bid(lvl1_idx as usize)?
1577                    };
1578                    assert!(lvl1_indirect_bid != 0);
1579                    let lvl1_indirect_block = indirect_blocks.find(lvl1_indirect_bid)?;
1580                    lvl1_indirect_block.read_bid(lvl2_idx as usize)?
1581                };
1582                assert!(lvl2_indirect_bid != 0);
1583
1584                let lvl2_indirect_block = indirect_blocks.find_mut(lvl2_indirect_bid)?;
1585                for (i, bid) in device_range.enumerate() {
1586                    lvl2_indirect_block.write_bid(lvl3_idx as usize + i, &bid)?;
1587                }
1588            }
1589        }
1590        Ok(())
1591    }

1593    /// Sets the device block IDs for indirect blocks required by a specific block ID.
1594    ///
1595    /// It assigns a sequence of block IDs (`indirect_bids`) on the device to be used
1596    /// as indirect blocks for a given file block ID (`bid`).
1597    fn set_indirect_bids(&mut self, bid: Ext2Bid, indirect_bids: &[Ext2Bid]) -> Result<()> {
1598        assert!((1..=3).contains(&indirect_bids.len()));
1599
1600        let bid_path = BidPath::from(bid);
1601        let mut block_ptrs = self.block_manager.block_ptrs.write();
1602        let mut indirect_blocks = self.block_manager.indirect_blocks.write();
1603        for indirect_bid in indirect_bids.iter() {
1604            let indirect_block = IndirectBlock::alloc()?;
1605            indirect_blocks.insert(*indirect_bid, indirect_block)?;
1606
1607            match bid_path {
1608                BidPath::Indirect(idx) => {
1609                    assert_eq!(idx, 0);
1610                    self.desc.block_ptrs.set_indirect(*indirect_bid);
1611                    block_ptrs.set_indirect(*indirect_bid);
1612                }
1613                BidPath::DbIndirect(lvl1_idx, lvl2_idx) => {
1614                    assert_eq!(lvl2_idx, 0);
1615                    if self.desc.block_ptrs.db_indirect() == 0 {
1616                        self.desc.block_ptrs.set_db_indirect(*indirect_bid);
1617                        block_ptrs.set_db_indirect(*indirect_bid);
1618                    } else {
1619                        let db_indirect_block =
1620                            indirect_blocks.find_mut(self.desc.block_ptrs.db_indirect())?;
1621                        db_indirect_block.write_bid(lvl1_idx as usize, indirect_bid)?;
1622                    }
1623                }
1624                BidPath::TbIndirect(lvl1_idx, lvl2_idx, lvl3_idx) => {
1625                    assert_eq!(lvl3_idx, 0);
1626                    if self.desc.block_ptrs.tb_indirect() == 0 {
1627                        self.desc.block_ptrs.set_tb_indirect(*indirect_bid);
1628                        block_ptrs.set_tb_indirect(*indirect_bid);
1629                    } else {
1630                        let lvl1_indirect_bid = {
1631                            let tb_indirect_block =
1632                                indirect_blocks.find(self.desc.block_ptrs.tb_indirect())?;
1633                            tb_indirect_block.read_bid(lvl1_idx as usize)?
1634                        };
1635
1636                        if lvl1_indirect_bid == 0 {
1637                            let tb_indirect_block =
1638                                indirect_blocks.find_mut(self.desc.block_ptrs.tb_indirect())?;
1639                            tb_indirect_block.write_bid(lvl1_idx as usize, indirect_bid)?;
1640                        } else {
1641                            let lvl1_indirect_block =
1642                                indirect_blocks.find_mut(lvl1_indirect_bid)?;
1643                            lvl1_indirect_block.write_bid(lvl2_idx as usize, indirect_bid)?;
1644                        }
1645                    }
1646                }
1647                BidPath::Direct(_) => panic!(),
1648            }
1649        }

1657    /// which may result in an decreased block count.

1663        if new_blocks < old_blocks {

1678    /// Shrinks inode blocks.
1679    ///
1680    /// After the reduction, the block count will be decreased to `range.start`.
1681    fn shrink_blocks(&mut self, range: Range<Ext2Bid>) {
1682        let mut current_range = range.clone();
1683        while !current_range.is_empty() {
1684            let free_cnt = self.try_shrink_blocks(current_range.clone());
1685            current_range.end -= free_cnt;
1686        }
1687
1688        self.desc.set_data_blocks(range.start);
1689        self.last_alloc_device_bid = if range.start == 0 {
1690            None
1691        } else {
1692            Some(
1693                DeviceRangeReader::new(&self.block_manager, (range.start - 1)..range.start)
1694                    .unwrap()
1695                    .read()
1696                    .unwrap()
1697                    .start,
1698            )
1699        };

1702    /// Attempts to shrink a range of blocks and returns the number of blocks
1703    /// successfully freed.
1704    ///
1705    /// Note that the returned number may be less than the requested range if needs
1706    /// to free the indirect blocks that are no longer required.
1707    fn try_shrink_blocks(&mut self, range: Range<Ext2Bid>) -> Ext2Bid {
1708        // Calculates the maximum range of blocks that can be freed in this round.
1709        let range = {
1710            let max_cnt = (range.len() as Ext2Bid)
1711                .min(BidPath::from(range.end - 1).last_lvl_idx() as Ext2Bid + 1);
1712            (range.end - max_cnt)..range.end
1713        };
1714
1715        let fs = self.fs();
1716        let device_range_reader =
1717            DeviceRangeReader::new(&self.block_manager, range.clone()).unwrap();
1718        for device_range in device_range_reader {
1719            fs.free_blocks(device_range.clone()).unwrap();
1720        }
1721
1722        self.free_indirect_blocks_required_by(range.start).unwrap();
1723        range.len() as Ext2Bid
1724    }

1726    /// Frees the indirect blocks required by the specified block ID.
1727    ///
1728    /// It ensures that the indirect blocks that are required by the block ID
1729    /// are properly released.
1730    fn free_indirect_blocks_required_by(&mut self, bid: Ext2Bid) -> Result<()> {
1731        let bid_path = BidPath::from(bid);
1732        if bid_path.last_lvl_idx() != 0 {
1733            return Ok(());
1734        }
1735        if bid == 0 {
1736            return Ok(());
1737        }
1738
1739        let mut block_ptrs = self.block_manager.block_ptrs.write();
1740        let mut indirect_blocks = self.block_manager.indirect_blocks.write();
1741        match bid_path {
1742            BidPath::Indirect(_) => {
1743                let indirect_bid = self.desc.block_ptrs.indirect();
1744                if indirect_bid == 0 {
1745                    return Ok(());
1746                }
1747
1748                self.desc.block_ptrs.set_indirect(0);
1749                block_ptrs.set_indirect(0);
1750                indirect_blocks.remove(indirect_bid);
1751                self.fs()
1752                    .free_blocks(indirect_bid..indirect_bid + 1)
1753                    .unwrap();
1754            }
1755            BidPath::DbIndirect(lvl1_idx, _) => {
1756                let db_indirect_bid = self.desc.block_ptrs.db_indirect();
1757                if db_indirect_bid == 0 {
1758                    return Ok(());
1759                }
1760
1761                let fs = self.fs();
1762                let lvl1_indirect_bid = {
1763                    let db_indirect_block = indirect_blocks.find(db_indirect_bid)?;
1764                    db_indirect_block.read_bid(lvl1_idx as usize)?
1765                };
1766                if lvl1_indirect_bid != 0 {
1767                    indirect_blocks.remove(lvl1_indirect_bid);
1768                    fs.free_blocks(lvl1_indirect_bid..lvl1_indirect_bid + 1)
1769                        .unwrap();
1770                }
1771                if lvl1_idx == 0 {
1772                    self.desc.block_ptrs.set_db_indirect(0);
1773                    block_ptrs.set_db_indirect(0);
1774                    indirect_blocks.remove(db_indirect_bid);
1775                    fs.free_blocks(db_indirect_bid..db_indirect_bid + 1)
1776                        .unwrap();
1777                }
1778            }
1779            BidPath::TbIndirect(lvl1_idx, lvl2_idx, _) => {
1780                let tb_indirect_bid = self.desc.block_ptrs.tb_indirect();
1781                if tb_indirect_bid == 0 {
1782                    return Ok(());
1783                }
1784
1785                let fs = self.fs();
1786                let lvl1_indirect_bid = {
1787                    let tb_indirect_block = indirect_blocks.find(tb_indirect_bid)?;
1788                    tb_indirect_block.read_bid(lvl1_idx as usize)?
1789                };
1790                if lvl1_indirect_bid != 0 {
1791                    let lvl2_indirect_bid = {
1792                        let lvl1_indirect_block = indirect_blocks.find(lvl1_indirect_bid)?;
1793                        lvl1_indirect_block.read_bid(lvl2_idx as usize)?
1794                    };
1795                    if lvl2_indirect_bid != 0 {
1796                        indirect_blocks.remove(lvl2_indirect_bid);
1797                        fs.free_blocks(lvl2_indirect_bid..lvl2_indirect_bid + 1)
1798                            .unwrap();
1799                    }
1800                    if lvl2_idx == 0 {
1801                        indirect_blocks.remove(lvl1_indirect_bid);
1802                        fs.free_blocks(lvl1_indirect_bid..lvl1_indirect_bid + 1)
1803                            .unwrap();
1804                    }
1805                }
1806
1807                if lvl2_idx == 0 && lvl1_idx == 0 {
1808                    self.desc.block_ptrs.set_tb_indirect(0);
1809                    block_ptrs.set_tb_indirect(0);
1810                    indirect_blocks.remove(tb_indirect_bid);
1811                    fs.free_blocks(tb_indirect_bid..tb_indirect_bid + 1)
1812                        .unwrap();
1813                }
1814            }
1815            BidPath::Direct(_) => panic!(),
1816        }
1817
1818        Ok(())
1819    }

1836/// Manages the inode blocks and block I/O operations.
1837struct InodeBlockManager {
1838    nblocks: AtomicUsize,
1839    /// Maintains a second copy of block pointers for page cache use, distinct from
1840    /// the one in `InodeDesc`.
1841    ///
1842    /// Updates occur infrequently and are performed on both copies, whereas

1977/// A reader to get the corresponding device block IDs for a specified range.
1978///
1979/// It calculates and returns the range of block IDs on the device that would map to
1980/// the file's block range. This is useful for translating file-level block addresses
1981/// to their locations on the physical storage device.
1982struct DeviceRangeReader<'a> {
1983    block_ptrs: RwMutexReadGuard<'a, BlockPtrs>,
1984    indirect_blocks: RwMutexWriteGuard<'a, IndirectBlockCache>,
1985    range: Range<Ext2Bid>,
1986    indirect_block: Option<IndirectBlock>,
1987}
1988
1989impl<'a> DeviceRangeReader<'a> {
1990    /// Creates a new reader.
1991    ///
1992    /// # Panics
1993    ///
1994    /// If the 'range' is empty, this method will panic.
1995    pub fn new(block_manager: &'a InodeBlockManager, range: Range<Ext2Bid>) -> Result<Self> {
1996        assert!(!range.is_empty());
1997
1998        let mut reader = Self {
1999            block_ptrs: block_manager.block_ptrs.read(),
2000            indirect_blocks: block_manager.indirect_blocks.write(),
2001            range,
2002            indirect_block: None,
2003        };
2004        reader.update_indirect_block()?;
2005        Ok(reader)
2006    }
2007
2008    /// Reads the corresponding device block IDs for a specified range.
2009    ///
2010    /// Note that the returned device range size may be smaller than the requested range
2011    /// due to possible inconsecutive block allocation.
2012    pub fn read(&mut self) -> Result<Range<Ext2Bid>> {
2013        let bid_path = BidPath::from(self.range.start);
2014        let max_cnt = self
2015            .range
2016            .len()
2017            .min(bid_path.cnt_to_next_indirect() as usize);
2018        let start_idx = bid_path.last_lvl_idx();
2019
2020        // Reads the device block ID range
2021        let mut device_range: Option<Range<Ext2Bid>> = None;
2022        for i in start_idx..start_idx + max_cnt {
2023            let device_bid = match &self.indirect_block {
2024                None => self.block_ptrs.direct(i),
2025                Some(indirect_block) => indirect_block.read_bid(i)?,
2026            };
2027            match device_range {
2028                Some(ref mut range) => {
2029                    if device_bid == range.end {
2030                        range.end += 1;
2031                    } else {
2032                        break;
2033                    }
2034                }
2035                None => {
2036                    device_range = Some(device_bid..device_bid + 1);
2037                }
2038            }
2039        }
2040        let device_range = device_range.unwrap();
2041
2042        // Updates the range
2043        self.range.start += device_range.len() as Ext2Bid;
2044        if device_range.len() == max_cnt && !self.range.is_empty() {
2045            // Updates the indirect block
2046            self.update_indirect_block()?;
2047        }
2048
2049        Ok(device_range)
2050    }
2051
2052    fn update_indirect_block(&mut self) -> Result<()> {
2053        let bid_path = BidPath::from(self.range.start);
2054        match bid_path {
2055            BidPath::Direct(_) => {
2056                self.indirect_block = None;
2057            }
2058            BidPath::Indirect(_) => {
2059                let indirect_bid = self.block_ptrs.indirect();
2060                let indirect_block = self.indirect_blocks.find(indirect_bid)?;
2061                self.indirect_block = Some(indirect_block.clone());
2062            }
2063            BidPath::DbIndirect(lvl1_idx, _) => {
2064                let lvl1_indirect_bid = {
2065                    let db_indirect_block =
2066                        self.indirect_blocks.find(self.block_ptrs.db_indirect())?;
2067                    db_indirect_block.read_bid(lvl1_idx as usize)?
2068                };
2069                let lvl1_indirect_block = self.indirect_blocks.find(lvl1_indirect_bid)?;
2070                self.indirect_block = Some(lvl1_indirect_block.clone())
2071            }
2072            BidPath::TbIndirect(lvl1_idx, lvl2_idx, _) => {
2073                let lvl2_indirect_bid = {
2074                    let lvl1_indirect_bid = {
2075                        let tb_indirect_block =
2076                            self.indirect_blocks.find(self.block_ptrs.tb_indirect())?;
2077                        tb_indirect_block.read_bid(lvl1_idx as usize)?
2078                    };
2079                    let lvl1_indirect_block = self.indirect_blocks.find(lvl1_indirect_bid)?;
2080                    lvl1_indirect_block.read_bid(lvl2_idx as usize)?
2081                };
2082                let lvl2_indirect_block = self.indirect_blocks.find(lvl2_indirect_bid)?;
2083                self.indirect_block = Some(lvl2_indirect_block.clone())
2084            }
2085        }
2086
2087        Ok(())
2088    }
2089}
