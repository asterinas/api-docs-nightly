17pub(super) const SCHED_NORMAL: u32 = 0;
18pub(super) const SCHED_FIFO: u32 = 1;
19pub(super) const SCHED_RR: u32 = 2;
20// pub(super) const SCHED_BATCH: u32 = 3; // Not supported.
21// SCHED_ISO: Reserved but not implemented yet on Linux.
22pub(super) const SCHED_IDLE: u32 = 5;
23// pub(super) const SCHED_DEADLINE: u32 = 6; // Not supported.
24// pub(super) const SCHED_EXT: u32 = 7; // Not supported.
25
26#[derive(Default, Debug, Pod, Clone, Copy)]
27#[repr(C)]
28pub(super) struct LinuxSchedAttr {
29    // Size of this structure
30    pub(super) size: u32,
31
32    // Policy (SCHED_*)
33    pub(super) sched_policy: u32,
34    // Flags
35    pub(super) sched_flags: u64,
36
37    // Nice value (SCHED_NORMAL, SCHED_BATCH)
38    pub(super) sched_nice: i32,
39
40    // Static priority (SCHED_FIFO, SCHED_RR)
41    pub(super) sched_priority: u32,
42
43    // For SCHED_DEADLINE
44    pub(super) sched_runtime: u64,
45    pub(super) sched_deadline: u64,
46    pub(super) sched_period: u64,
47
48    // Utilization hints
49    pub(super) sched_util_min: u32,
50    pub(super) sched_util_max: u32,
51}
52
53// Reference: <https://elixir.bootlin.com/linux/v6.17.7/source/include/uapi/linux/sched/types.h#L7>
54const SCHED_ATTR_SIZE_VER0: u32 = 48;
55// Reference: <https://elixir.bootlin.com/linux/v6.17.7/source/include/uapi/linux/sched/types.h#L8>
56const SCHED_ATTR_SIZE_VER1: u32 = 56;
57
58const_assert!(size_of::<LinuxSchedAttr>() == SCHED_ATTR_SIZE_VER1 as usize);

60impl TryFrom<SchedPolicy> for LinuxSchedAttr {
61    type Error = Error;
62
63    fn try_from(value: SchedPolicy) -> Result<Self> {
64        Ok(match value {
65            SchedPolicy::Stop => LinuxSchedAttr {
66                sched_policy: SCHED_FIFO,
67                sched_priority: 99, // Linux uses 99 as the default priority for STOP tasks.
68                ..Default::default()
69            },
70
71            SchedPolicy::RealTime { rt_prio, rt_policy } => LinuxSchedAttr {
72                sched_policy: match rt_policy {
73                    RealTimePolicy::Fifo => SCHED_FIFO,
74                    RealTimePolicy::RoundRobin { .. } => SCHED_RR,
75                },
76                sched_priority: rt_to_static(rt_prio),
77                ..Default::default()
78            },
79
80            // The SCHED_IDLE policy is mapped to the highest nice value of
81            // `SchedPolicy::Fair` instead of `SchedPolicy::Idle`. Tasks of the
82            // latter policy are invisible to the user API.
83            SchedPolicy::Fair(Nice::MAX) => LinuxSchedAttr {
84                sched_policy: SCHED_IDLE,
85                ..Default::default()
86            },
87
88            SchedPolicy::Fair(nice) => LinuxSchedAttr {
89                sched_policy: SCHED_NORMAL,
90                sched_nice: nice.value().get().into(),
91                ..Default::default()
92            },
93
94            SchedPolicy::Idle => return_errno_with_message!(
95                Errno::EACCES,
96                "scheduling attributes for idle tasks are not accessible"
97            ),
98        })
99    }
100}

102impl TryFrom<LinuxSchedAttr> for SchedPolicy {
103    type Error = Error;
104
105    fn try_from(value: LinuxSchedAttr) -> Result<Self> {
106        Ok(match value.sched_policy {
107            SCHED_FIFO | SCHED_RR => SchedPolicy::RealTime {
108                rt_prio: static_to_rt(value.sched_priority)?,
109                rt_policy: match value.sched_policy {
110                    SCHED_FIFO => RealTimePolicy::Fifo,
111                    SCHED_RR => RealTimePolicy::RoundRobin {
112                        base_slice_factor: None,
113                    },
114                    _ => unreachable!(),
115                },
116            },
117
118            _ if value.sched_priority != 0 => {
119                return_errno_with_message!(Errno::EINVAL, "invalid scheduling priority")
120            }
121
122            SCHED_NORMAL => SchedPolicy::Fair(Nice::new(
123                i8::try_from(value.sched_nice)
124                    .ok()
125                    .and_then(|n| n.try_into().ok())
126                    .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid nice number"))?,
127            )),
128
129            // The SCHED_IDLE policy is mapped to the highest nice value of
130            // `SchedPolicy::Fair` instead of `SchedPolicy::Idle`. Tasks of the
131            // latter policy are invisible to the user API.
132            SCHED_IDLE => SchedPolicy::Fair(Nice::MAX),
133
134            _ => return_errno_with_message!(Errno::EINVAL, "invalid scheduling policy"),
135        })
136    }
137}

139pub(super) fn read_linux_sched_attr_from_user(
140    addr: Vaddr,
141    ctx: &Context,
142) -> Result<LinuxSchedAttr> {
143    // The code below is written according to the Linux implementation.
144    // Reference: <https://elixir.bootlin.com/linux/v6.17.7/source/kernel/sched/syscalls.c#L889>
145
146    let user_space = ctx.user_space();
147
148    let raw_size = user_space.read_val::<u32>(addr)?;
149    let user_size = if raw_size == 0 {
150        SCHED_ATTR_SIZE_VER0
151    } else {
152        raw_size
153    };
154    if user_size < SCHED_ATTR_SIZE_VER0 || user_size > PAGE_SIZE as u32 {
155        let _ = user_space.write_val(addr, &(size_of::<LinuxSchedAttr>() as u32));
156        return_errno_with_message!(Errno::E2BIG, "invalid scheduling attribute size");
157    }
158
159    let mut attr = user_space
160        .read_val_compat::<LinuxSchedAttr>(addr, user_size as usize)
161        .inspect_err(|err| {
162            if err.error() == Errno::E2BIG {
163                let _ = user_space.write_val(addr, &(size_of::<LinuxSchedAttr>() as u32));
164            }
165        })?;
166    // If `attr.size` is modified concurrently, we should use the original size.
167    attr.size = user_size;
168
169    // TODO: Check whether `sched_flags` is valid.
170
171    Ok(attr)
172}

174pub(super) fn write_linux_sched_attr_to_user(
175    mut attr: LinuxSchedAttr,
176    addr: Vaddr,
177    user_size: u32,
178    ctx: &Context,
179) -> Result<()> {
180    if user_size < SCHED_ATTR_SIZE_VER0 || user_size > PAGE_SIZE as u32 {
181        return_errno_with_message!(Errno::EINVAL, "invalid scheduling attribute size");
182    }
183
184    attr.size = (size_of::<LinuxSchedAttr>() as u32).min(user_size);
185
186    let range = &SCHED_PRIORITY_RANGE[attr.sched_policy as usize];
187    attr.sched_util_min = *range.start();
188    attr.sched_util_max = *range.end();
189
190    ctx.user_space()
191        .write_val_compat(addr, user_size as usize, &attr)?
192        .ignore_trailing();
193
194    Ok(())
195}

197pub(super) fn access_sched_attr_with<T>(
198    tid: Tid,
199    ctx: &Context,
200    f: impl FnOnce(&SchedAttr) -> Result<T>,
201) -> Result<T> {
202    if tid.cast_signed() < 0 {
203        return_errno_with_message!(Errno::EINVAL, "all negative TIDs are not valid");
204    }
205
206    if tid == 0 {
207        return f(ctx.thread.sched_attr());
208    }
209
210    let Some(thread) = thread_table::get_thread(tid) else {
211        return_errno_with_message!(Errno::ESRCH, "the target thread does not exist");
212    };
213    f(thread.sched_attr())
214}

216pub fn sys_sched_getattr(
217    tid: Tid,
218    addr: Vaddr,
219    user_size: u32,
220    flags: u32,
221    ctx: &Context,
222) -> Result<SyscallReturn> {
223    if addr == 0 {
224        return_errno_with_message!(Errno::EINVAL, "invalid user space address");
225    }
226    if flags != 0 {
227        // Linux also has no support for any flags yet.
228        return_errno_with_message!(Errno::EINVAL, "invalid flags");
229    }
230
231    let policy = access_sched_attr_with(tid, ctx, |attr| Ok(attr.policy()))?;
232    let attr: LinuxSchedAttr = policy
233        .try_into()
234        .expect("all user-visible scheduling attributes should be valid");
235    write_linux_sched_attr_to_user(attr, addr, user_size, ctx)?;
236
237    Ok(SyscallReturn::Return(0))
