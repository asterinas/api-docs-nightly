3use super::SyscallReturn;
4use crate::{
5    fs,
6    fs::{
7        file_table::{FileDesc, get_file_fast},
8        path::{AT_FDCWD, FsPath},
9        utils::PATH_MAX,
10    },
11    prelude::*,
12    process::ResourceType,
13};
14
15pub fn sys_ftruncate(fd: FileDesc, len: isize, ctx: &Context) -> Result<SyscallReturn> {
16    debug!("fd = {}, length = {}", fd, len);
17
18    check_length(len, ctx)?;
19
20    let mut file_table = ctx.thread_local.borrow_file_table_mut();
21    let file = get_file_fast!(&mut file_table, fd);
22    file.resize(len as usize)?;
23    fs::notify::on_change(file.path());
24    Ok(SyscallReturn::Return(0))
25}
26
27pub fn sys_truncate(path_ptr: Vaddr, len: isize, ctx: &Context) -> Result<SyscallReturn> {
28    let path_name = ctx.user_space().read_cstring(path_ptr, PATH_MAX)?;
29    debug!("path = {:?}, length = {}", path_name, len);
30
31    check_length(len, ctx)?;
32
33    let dir_path = {
34        let path_name = path_name.to_string_lossy();
35        let fs_path = FsPath::from_fd_and_path(AT_FDCWD, &path_name)?;
36        ctx.thread_local
37            .borrow_fs()
38            .resolver()
39            .read()
40            .lookup(&fs_path)?
41    };
42    dir_path.resize(len as usize)?;
43    fs::notify::on_change(&dir_path);
44    Ok(SyscallReturn::Return(0))
45}
46
47#[inline]
48fn check_length(len: isize, ctx: &Context) -> Result<()> {
49    if len < 0 {
50        return_errno_with_message!(Errno::EINVAL, "length is negative");
51    }
52
53    let max_file_size = {
54        let resource_limits = ctx.process.resource_limits();
55        resource_limits
56            .get_rlimit(ResourceType::RLIMIT_FSIZE)
57            .get_cur() as usize
58    };
59    if len as usize > max_file_size {
60        return_errno_with_message!(Errno::EFBIG, "length is larger than the maximum file size");
61    }
62    Ok(())
63}
