17pub fn sys_removexattr(path_ptr: Vaddr, name_ptr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
18    let user_space = ctx.user_space();
19    let path = user_space.read_cstring(path_ptr, MAX_FILENAME_LEN)?;
20
21    removexattr(XattrFileCtx::Path(path), name_ptr, &user_space, ctx)?;
22
23    Ok(SyscallReturn::Return(0))
24}

26pub fn sys_lremovexattr(path_ptr: Vaddr, name_ptr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
27    let user_space = ctx.user_space();
28    let path = user_space.read_cstring(path_ptr, MAX_FILENAME_LEN)?;
29

31
32    Ok(SyscallReturn::Return(0))
33}

35pub fn sys_fremovexattr(fd: FileDesc, name_ptr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
36    let mut file_table = ctx.thread_local.borrow_file_table_mut();
37    let file = get_file_fast!(&mut file_table, fd);
38
39    let user_space = ctx.user_space();
40    removexattr(XattrFileCtx::FileHandle(file), name_ptr, &user_space, ctx)?;
41
42    Ok(SyscallReturn::Return(0))
43}
44
45fn removexattr(
46    file_ctx: XattrFileCtx,
47    name_ptr: Vaddr,
48    user_space: &CurrentUserSpace,
49    ctx: &Context,
50) -> Result<()> {
51    let name_cstr = read_xattr_name_cstr_from_user(name_ptr, user_space)?;
52    let name_str = name_cstr.to_string_lossy();
53    let xattr_name = parse_xattr_name(name_str.as_ref())?;
54    check_xattr_namespace(xattr_name.namespace(), ctx)?;
55
56    match lookup_path_for_xattr(&file_ctx, ctx) {
57        Ok(path) => {
58            path.remove_xattr(xattr_name)?;
59            fs::notify::on_attr_change(&path);
60            Ok(())
61        }
62        Err(e) => Err(e),
63    }
64}
