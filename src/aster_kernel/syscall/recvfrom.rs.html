3use super::SyscallReturn;
4use crate::{
5    fs::file_table::{FileDesc, get_file_fast},
6    net::socket::util::SendRecvFlags,
7    prelude::*,
8    util::net::write_socket_addr_to_user,
9};
10
11pub fn sys_recvfrom(
12    sockfd: FileDesc,
13    buf: Vaddr,
14    len: usize,
15    flags: i32,
16    src_addr: Vaddr,
17    addrlen_ptr: Vaddr,
18    ctx: &Context,
19) -> Result<SyscallReturn> {
20    let flags = SendRecvFlags::from_bits_truncate(flags);
21    debug!(
22        "sockfd = {sockfd}, buf = 0x{buf:x}, len = {len}, flags = {flags:?}, src_addr = 0x{src_addr:x}, addrlen_ptr = 0x{addrlen_ptr:x}"
23    );
24
25    let mut file_table = ctx.thread_local.borrow_file_table_mut();
26    let file = get_file_fast!(&mut file_table, sockfd);
27    let socket = file.as_socket_or_err()?;
28
29    let user_space = ctx.user_space();
30    let mut writers = user_space.writer(buf, len)?;
31
32    let (recv_size, message_header) =
33        socket
34            .recvmsg(&mut writers, flags)
35            .map_err(|err| match err.error() {
36                // FIXME: `recvfrom` should not be restarted if a timeout has been set on the socket using `setsockopt`.
37                Errno::EINTR => Error::new(Errno::ERESTARTSYS),
38                _ => err,
39            })?;
40
41    if let Some(socket_addr) = message_header.addr()
42        && src_addr != 0
43    {
44        write_socket_addr_to_user(socket_addr, src_addr, addrlen_ptr)?;
45    }
46
47    Ok(SyscallReturn::Return(recv_size as _))
48}
