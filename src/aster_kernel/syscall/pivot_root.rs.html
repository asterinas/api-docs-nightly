3use super::SyscallReturn;
4use crate::{
5    fs::path::FsPath, prelude::*, process::posix_thread::thread_table::with_global_threads,
6    syscall::constants::MAX_FILENAME_LEN,
7};
8
9pub fn sys_pivot_root(
10    new_root_ptr: Vaddr,
11    put_old_ptr: Vaddr,
12    ctx: &Context,
13) -> Result<SyscallReturn> {
14    let new_root_name = ctx
15        .user_space()
16        .read_cstring(new_root_ptr, MAX_FILENAME_LEN)?;
17    let put_old_name = ctx
18        .user_space()
19        .read_cstring(put_old_ptr, MAX_FILENAME_LEN)?;
20
21    debug!(
22        "pivot_root: new_root = {:?}, put_old = {:?}",
23        new_root_name, put_old_name
24    );
25
26    let new_root_name = new_root_name.to_string_lossy();
27    let new_root_path = FsPath::try_from(new_root_name.as_ref())?;
28    let put_old_name = put_old_name.to_string_lossy();
29    let put_old_path = FsPath::try_from(put_old_name.as_ref())?;
30
31    // TODO: Locking the global thread table here is a workaround. We need to use a more
32    // suitable lock (i.e. the global mount lock or the namespace lock) to avoid deadlock.
33    with_global_threads(move |table| {
34        let fs_ref = ctx.thread_local.borrow_fs();
35        let mut fs_resolver = fs_ref.resolver().write();
36
37        fs_resolver.pivot_root(new_root_path, put_old_path, table, ctx)
38    })?;
39
40    Ok(SyscallReturn::Return(0))
41}
