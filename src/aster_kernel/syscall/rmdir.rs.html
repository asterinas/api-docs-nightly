3use super::SyscallReturn;
4use crate::{
5    fs::{
6        file_table::FileDesc,
7        path::{AT_FDCWD, FsPath, SplitPath},
8    },
9    prelude::*,
10    syscall::constants::MAX_FILENAME_LEN,
11};
12
13pub fn sys_rmdir(path_addr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
14    self::sys_rmdirat(AT_FDCWD, path_addr, ctx)
15}
16
17pub(super) fn sys_rmdirat(
18    dirfd: FileDesc,
19    path_addr: Vaddr,
20    ctx: &Context,
21) -> Result<SyscallReturn> {
22    let path_name = ctx.user_space().read_cstring(path_addr, MAX_FILENAME_LEN)?;
23    debug!("dirfd = {}, path_addr = {:?}", dirfd, path_addr);
24
25    let path_name = path_name.to_string_lossy();
26    let (dir_path, name) = {
27        let (parent_path_name, target_name) = path_name.split_dirname_and_basename()?;
28        let fs_path = FsPath::from_fd_and_path(dirfd, parent_path_name)?;
29        (
30            ctx.thread_local
31                .borrow_fs()
32                .resolver()
33                .read()
34                .lookup(&fs_path)?,
35            target_name,
36        )
37    };
38
39    dir_path.rmdir(name)?;
40    Ok(SyscallReturn::Return(0))
41}
