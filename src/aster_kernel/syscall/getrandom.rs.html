3use super::SyscallReturn;
4use crate::{device, prelude::*};
5
6pub fn sys_getrandom(buf: Vaddr, count: usize, flags: u32, ctx: &Context) -> Result<SyscallReturn> {
7    let flags = GetRandomFlags::from_bits(flags)
8        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid flags"))?;
9    debug!(
10        "buf = 0x{:x}, count = 0x{:x}, flags = {:?}",
11        buf, count, flags
12    );
13
14    if flags.contains(GetRandomFlags::GRND_INSECURE | GetRandomFlags::GRND_RANDOM) {
15        return_errno_with_message!(
16            Errno::EINVAL,
17            "requesting insecure and blocking randomness makes no sense"
18        );
19    }
20
21    // Currently we don't really generate true randomness by collecting environment noise, so we
22    // will never block.
23    // TODO: Support `GRND_NONBLOCK` and `GRND_INSECURE`.
24
25    let user_space = ctx.user_space();
26    let mut writer = user_space.writer(buf, count)?;
27    let read_len = if flags.contains(GetRandomFlags::GRND_RANDOM) {
28        device::getrandom(&mut writer)?
29    } else {
30        device::geturandom(&mut writer)?
31    };
32    Ok(SyscallReturn::Return(read_len as isize))
33}
34
35bitflags::bitflags! {
36    /// Flags for `getrandom`.
37    ///
38    /// Reference: <https://elixir.bootlin.com/linux/v6.16.9/source/include/uapi/linux/random.h#L56>.
39    struct GetRandomFlags: u32 {
40        const GRND_NONBLOCK = 0x0001;
41        const GRND_RANDOM = 0x0002;
42        const GRND_INSECURE = 0x0004;
43    }
44}
