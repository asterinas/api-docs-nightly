3//! This module implements the `setns` syscall.
4//!
5//! This syscall reassociates the calling thread with a namespace specified by a
6//! file descriptor. The `flags` argument determines which type of namespace can be
7//! joined.
8//!
9//! The file descriptor `fd` can refer to:
10//! 1. A namespace file from `/proc/[pid]/ns/`.
11//! 2. A `PidFile` opened by `pidfd_open` or by opening `/proc/[pid]` directory.
12
13use crate::{
14    fs::{file_table::FileDesc, path::MountNamespace},
15    net::uts_ns::UtsNamespace,
16    prelude::*,
17    process::{
18        CloneFlags, ContextSetNsAdminApi, NsProxy, NsProxyBuilder, PidFile,
19        check_unsupported_ns_flags, credentials::capabilities::CapSet, posix_thread::AsPosixThread,
20    },
21    syscall::SyscallReturn,
22};
23
24pub fn sys_setns(fd: FileDesc, flags: u32, ctx: &Context) -> Result<SyscallReturn> {
25    let ns_type_flags = CloneFlags::from_bits(flags)
26        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid `setns` flags"))?;
27    debug!("setns flags = {:?}", ns_type_flags);
28
29    let file = {
30        let file_table = ctx.thread_local.borrow_file_table();
31        let file_table_locked = file_table.unwrap().read();
32        file_table_locked.get_file(fd)?.clone()
33    };
34
35    let new_ns_proxy = if let Some(pid_file) = file.downcast_ref::<PidFile>() {
36        build_proxy_from_pid_file(pid_file, ns_type_flags, ctx)?
37    }
38    // TODO: Support setting namespaces from `/proc/[pid]/ns`.
39    else {
40        return_errno_with_message!(
41            Errno::EINVAL,
42            "the FD does not refer to a supported namespace file"
43        );
44    };
45
46    // Install the newly created `NsProxy`.
47    ctx.set_ns_proxy(Arc::new(new_ns_proxy));
48
49    Ok(SyscallReturn::Return(0))
50}
51
52fn build_proxy_from_pid_file(
53    pid_file: &PidFile,
54    flags: CloneFlags,
55    ctx: &Context,
56) -> Result<NsProxy> {
57    if flags.is_empty() {
58        return_errno_with_message!(Errno::EINVAL, "flags must be specified with a PID file");
59    }
60
61    // Check for any flags that are not namespace-related.
62    if !(flags - CloneFlags::CLONE_NS_FLAGS).is_empty() {
63        return_errno_with_message!(Errno::EINVAL, "invalid flags are specified with a PID file");
64    }
65
66    if flags.contains(CloneFlags::CLONE_NEWUSER) {
67        return_errno_with_message!(Errno::EINVAL, "setting a user namespace is not supported");
68    }
69
70    check_unsupported_ns_flags(flags)?;
71
72    let target_thread = pid_file
73        .process_opt()
74        .ok_or_else(|| Error::with_message(Errno::ESRCH, "the target process has been reaped"))?
75        .main_thread();
76    let target_proxy = target_thread.as_posix_thread().unwrap().ns_proxy().lock();
77    let Some(target_proxy) = target_proxy.as_ref() else {
78        return_errno_with_message!(Errno::ESRCH, "the target process has exited");
79    };
80
81    let current_proxy = ctx.thread_local.borrow_ns_proxy();
82    let current_proxy = current_proxy.unwrap();
83
84    let mut builder = NsProxyBuilder::new(current_proxy);
85
86    if flags.contains(CloneFlags::CLONE_NEWUTS) {
87        let target_ns = target_proxy.uts_ns();
88        set_uts_ns(&mut builder, target_ns, ctx)?;
89    }
90
91    if flags.contains(CloneFlags::CLONE_NEWNS) {
92        if ctx.thread_local.is_fs_shared() {
93            return_errno_with_message!(
94                Errno::EINVAL,
95                "setting a mount namespace is not allowed with shared filesystem information"
96            );
97        }
98
99        let target_ns = target_proxy.mnt_ns();
100        set_mnt_ns(&mut builder, target_ns, ctx)?;
101    }
102
103    // TODO: Support setting other namespaces from the target process.
104
105    Ok(builder.build())
106}
107
108fn set_uts_ns(
109    builder: &mut NsProxyBuilder,
110    target_ns: &Arc<UtsNamespace>,
111    ctx: &Context,
112) -> Result<()> {
113    // Verify the thread has SYS_ADMIN capability in the target namespace's owner
114    // and the current user namespace.
115    target_ns
116        .owner_ns()
117        .check_cap(CapSet::SYS_ADMIN, ctx.posix_thread)?;
118    ctx.thread_local
119        .borrow_user_ns()
120        .check_cap(CapSet::SYS_ADMIN, ctx.posix_thread)?;
121
122    // TODO: Are the checks above sufficient?
123
124    builder.uts_ns(target_ns.clone());
125
126    Ok(())
127}

129fn set_mnt_ns(
130    builder: &mut NsProxyBuilder,
131    target_ns: &Arc<MountNamespace>,
132    ctx: &Context,
133) -> Result<()> {
134    // Verify the thread has SYS_ADMIN capability in the target namespace's owner
135    // and the current user namespace.
136    target_ns
137        .owner()
138        .check_cap(CapSet::SYS_ADMIN, ctx.posix_thread)?;
139    ctx.thread_local
140        .borrow_user_ns()
141        .check_cap(CapSet::SYS_ADMIN, ctx.posix_thread)?;
142
143    // TODO: Are the checks above sufficient?
144
145    builder.mnt_ns(target_ns.clone());
146
147    Ok(())
