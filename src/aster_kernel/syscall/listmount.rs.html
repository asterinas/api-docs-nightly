3use ostd::mm::VmIo;
4
5use super::SyscallReturn;
6use crate::{fs::vfs::path::MNT_UNIQUE_ID_MIN, prelude::*};
7
8/// Size in bytes of the first published `mnt_id_req` layout; the only
9/// size currently accepted.
10const MNT_ID_REQ_SIZE_VER0: u32 = 24;
11
12/// Sentinel for `mnt_id_req::mnt_id` requesting the namespace root.
13const LSMT_ROOT: u64 = u64::MAX;
14
15/// `flags` bit: return descendants in descending `unique_id` order.
16const LISTMOUNT_REVERSE: u32 = 1 << 0;
17
18/// Upper bound on `nr_mnt_ids`.
19/// Reference: <https://elixir.bootlin.com/linux/v6.17/source/fs/namespace.c#L6036>
20const LISTMOUNT_NR_LIMIT: usize = 1_000_000;
21
22/// Linux's `struct mnt_id_req`.
23#[repr(C)]
24#[derive(Clone, Copy, Debug, Default, Pod)]
25struct MntIdReq {
26    /// Caller-side `sizeof(struct mnt_id_req)`.
27    size: u32,
28    /// Reserved; must be zero.
29    spare: u32,
30    /// Parent mount's `unique_id`, or [`LSMT_ROOT`] for the namespace
31    /// root.
32    mnt_id: u64,
33    /// Pagination cursor: only descendants with `unique_id > param`
34    /// (or `<` in reverse) are returned. `0` starts from the beginning.
35    param: u64,
36}
37
38const _: () = assert!(size_of::<MntIdReq>() as u32 == MNT_ID_REQ_SIZE_VER0);
39
40pub fn sys_listmount(
41    req_ptr: Vaddr,
42    mnt_ids_ptr: Vaddr,
43    nr_mnt_ids: usize,
44    flags: u32,
45    ctx: &Context,
46) -> Result<SyscallReturn> {
47    debug!(
48        "req_ptr = 0x{:x}, mnt_ids_ptr = 0x{:x}, nr_mnt_ids = {}, flags = 0x{:x}",
49        req_ptr, mnt_ids_ptr, nr_mnt_ids, flags,
50    );
51
52    if flags & !LISTMOUNT_REVERSE != 0 {
53        return_errno_with_message!(Errno::EINVAL, "invalid listmount flags");
54    }
55    let reverse = flags & LISTMOUNT_REVERSE != 0;
56
57    if nr_mnt_ids > LISTMOUNT_NR_LIMIT {
58        return_errno_with_message!(Errno::EOVERFLOW, "nr_mnt_ids too large");
59    }
60
61    let user_space = ctx.user_space();
62
63    // Validate the output buffer up front so an invalid pointer surfaces
64    // as `EFAULT` before any namespace work is done.
65    let mut mnt_ids_writer = user_space.writer(mnt_ids_ptr, nr_mnt_ids * size_of::<u64>())?;
66
67    let req: MntIdReq = user_space.read_val(req_ptr)?;
68    if req.size != MNT_ID_REQ_SIZE_VER0 {
69        return_errno_with_message!(Errno::EINVAL, "unsupported mnt_id_req size");
70    }
71    if req.spare != 0 {
72        return_errno_with_message!(Errno::EINVAL, "non-zero spare field");
73    }
74    if req.mnt_id != LSMT_ROOT && req.mnt_id < MNT_UNIQUE_ID_MIN {
75        return_errno_with_message!(Errno::EINVAL, "mnt_id is not a valid unique mount ID");
76    }
77    if req.param != 0 && req.param < MNT_UNIQUE_ID_MIN {
78        return_errno_with_message!(Errno::EINVAL, "param is not a valid unique mount ID");
79    }
80
81    let ns_proxy = ctx.thread_local.borrow_ns_proxy();
82    let mnt_ns = ns_proxy.unwrap().mnt_ns();
83
84    let parent_mount = if req.mnt_id == LSMT_ROOT {
85        mnt_ns.root().clone()
86    } else {
87        mnt_ns
88            .lookup_by_unique_id(req.mnt_id)
89            .ok_or_else(|| Error::with_message(Errno::ENOENT, "no such mount"))?
90    };
91
92    if nr_mnt_ids == 0 {
93        return Ok(SyscallReturn::Return(0));
94    }
95
96    let mounts = mnt_ns.descendant_mounts_of(&parent_mount);
97    let ids: Vec<_> = if reverse {
98        mounts
99            .rev()
100            .map(|mount| mount.unique_id())
101            .filter(|&id| req.param == 0 || id < req.param)
102            .take(nr_mnt_ids)
103            .collect()
104    } else {
105        mounts
106            .map(|mount| mount.unique_id())
107            .filter(|&id| id > req.param)
108            .take(nr_mnt_ids)
109            .collect()
110    };
111
112    for id in &ids {
113        mnt_ids_writer.write_val(id)?;
114    }
115
116    Ok(SyscallReturn::Return(ids.len() as isize))
117}
