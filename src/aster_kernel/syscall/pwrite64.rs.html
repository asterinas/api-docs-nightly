3use super::SyscallReturn;
4use crate::{
5    fs,
6    fs::file_table::{FileDesc, get_file_fast},
7    prelude::*,
8};
9
10pub fn sys_pwrite64(
11    fd: FileDesc,
12    user_buf_ptr: Vaddr,
13    user_buf_len: usize,
14    offset: i64,
15    ctx: &Context,
16) -> Result<SyscallReturn> {
17    debug!(
18        "fd = {}, user_buf_ptr = 0x{:x}, user_buf_len = 0x{:x}, offset = 0x{:x}",
19        fd, user_buf_ptr, user_buf_len, offset
20    );
21
22    if offset < 0 {
23        return_errno_with_message!(Errno::EINVAL, "offset cannot be negative");
24    }
25
26    let mut file_table = ctx.thread_local.borrow_file_table_mut();
27    let file = get_file_fast!(&mut file_table, fd);
28
29    // TODO: Check (f.file->f_mode & FMODE_PWRITE); We don't have f_mode in our FileLike trait
30    if user_buf_len == 0 {
31        return Ok(SyscallReturn::Return(0));
32    }
33    if offset.checked_add(user_buf_len as i64).is_none() {
34        return_errno_with_message!(Errno::EINVAL, "offset + user_buf_len overflow");
35    }
36
37    let user_space = ctx.user_space();
38    let mut reader = user_space.reader(user_buf_ptr, user_buf_len)?;
39    let write_len = file.write_at(offset as _, &mut reader)?;
40
41    if write_len > 0 {
42        fs::notify::on_modify(&file);
43    }
44
45    Ok(SyscallReturn::Return(write_len as _))
46}
