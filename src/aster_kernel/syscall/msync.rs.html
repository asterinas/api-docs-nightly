5use super::SyscallReturn;
6use crate::{prelude::*, thread::kernel_thread::ThreadOptions, vm::vmar::VMAR_CAP_ADDR};

8pub fn sys_msync(addr: Vaddr, len: usize, flag: i32, ctx: &Context) -> Result<SyscallReturn> {
9    let flags = MsyncFlags::from_bits(flag)
10        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid flags"))?;
11    debug!("addr = 0x{:x}, len = {}, flags = {:?}", addr, len, flags);
12
13    if flags.contains(MsyncFlags::MS_ASYNC | MsyncFlags::MS_SYNC) {
14        return_errno_with_message!(
15            Errno::EINVAL,
16            "MS_ASYNC and MS_SYNC cannot be specified together"
17        );
18    }
19
20    if !addr.is_multiple_of(PAGE_SIZE) {
21        return_errno_with_message!(Errno::EINVAL, "the mapping address is not aligned");
22    }
23    if len == 0 {
24        return Ok(SyscallReturn::Return(0));
25    }
26    if VMAR_CAP_ADDR.checked_sub(addr).is_none_or(|gap| gap < len) {
27        // FIXME: Linux returns `ENOMEM` if `(addr + len).align_up(PAGE_SIZE)` overflows. Here, we
28        // perform a stricter validation.
29        return_errno_with_message!(Errno::ENOMEM, "the mapping range is not in userspace");
30    }
31    let addr_range = addr..(addr + len).align_up(PAGE_SIZE);
32
33    let user_space = ctx.user_space();
34    let vmar = user_space.vmar();
35    let query_guard = vmar.query(addr_range.clone());
36
37    // Check if the range is fully mapped.
38    if !query_guard.is_fully_mapped() {
39        return_errno_with_message!(
40            Errno::ENOMEM,
41            "the range contains pages that are not mapped"
42        );
43    }
44
45    // Do nothing if not file-backed, as <https://pubs.opengroup.org/onlinepubs/9699919799/> says.
46    let inodes = query_guard
47        .iter()
48        .filter_map(|m| m.inode().cloned())
49        .collect::<Vec<_>>();
50    let task_fn = move || {
51        for inode in inodes {
52            // TODO: Sync a necessary range instead of syncing the whole inode.
53            let _ = inode.sync_all();
54        }
55    };
56
57    // If neither MS_SYNC nor MS_ASYNC is specified, Linux defaults to MS_ASYNC behavior.
58    if flags.contains(MsyncFlags::MS_SYNC) {
59        task_fn();
60    } else {
61        ThreadOptions::new(task_fn).spawn();
62    }
63
64    Ok(SyscallReturn::Return(0))
65}
66
67bitflags! {
68    /// Flags for `msync`.
69    ///
70    /// Reference: <https://elixir.bootlin.com/linux/v6.15.1/source/include/uapi/asm-generic/mman-common.h#L42>.
71    struct MsyncFlags: i32 {
72        /// Performs `msync` asynchronously.
73        const MS_ASYNC      = 0x01;
74        /// Invalidates cache so that other processes mapping the same file
75        /// will immediately see the changes after this `msync` call.
76        ///
77        /// Should be a no-op since we use the same page cache for all processes.
78        const MS_INVALIDATE = 0x02;
79        /// Performs `msync` synchronously.
80        const MS_SYNC       = 0x04;
81    }
82}
