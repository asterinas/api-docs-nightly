3use ostd::power::{ExitCode, poweroff, restart};
4
5use super::SyscallReturn;
6use crate::{prelude::*, process::credentials::capabilities::CapSet};
7
8// Linux reboot magic constants.
9const LINUX_REBOOT_MAGIC1: u32 = 0xfee1dead;
10const LINUX_REBOOT_MAGIC2: u32 = 0x28121969;
11const LINUX_REBOOT_MAGIC2A: u32 = 0x05121996;
12const LINUX_REBOOT_MAGIC2B: u32 = 0x16041998;
13const LINUX_REBOOT_MAGIC2C: u32 = 0x20112000;
14
15/// Linux reboot commands.
16#[derive(Debug, Clone, Copy, PartialEq, Eq, TryFromInt)]
17#[repr(u32)]
18enum RebootCmd {
19    Restart = 0x01234567,
20    Halt = 0xcdef0123,
21    PowerOff = 0x4321fedc,
22    // TODO: Add more reboot sub-commands.
23}
24
25pub fn sys_reboot(
26    magic1: u32,
27    magic2: u32,
28    op: u32,
29    _arg: Vaddr,
30    ctx: &Context,
31) -> Result<SyscallReturn> {
32    debug!(
33        "[sys_reboot]: magic1 = {:#x}, magic2 = {:#x}, op = {:#x}",
34        magic1, magic2, op
35    );
36
37    // Verify magic numbers
38    if magic1 != LINUX_REBOOT_MAGIC1 {
39        return_errno_with_message!(Errno::EINVAL, "the reboot magic is invalid");
40    }
41    if magic2 != LINUX_REBOOT_MAGIC2
42        && magic2 != LINUX_REBOOT_MAGIC2A
43        && magic2 != LINUX_REBOOT_MAGIC2B
44        && magic2 != LINUX_REBOOT_MAGIC2C
45    {
46        return_errno_with_message!(Errno::EINVAL, "the reboot magic is invalid");
47    }
48
49    if !ctx
50        .posix_thread
51        .credentials()
52        .effective_capset()
53        .contains(CapSet::SYS_BOOT)
54    {
55        return_errno_with_message!(Errno::EPERM, "reboot without SYS_BOOT is not allowed");
56    }
57
58    let cmd = RebootCmd::try_from(op)?;
59
60    // TODO: Perform any necessary cleanup before powering off or restarting.
61    match cmd {
62        RebootCmd::Restart => restart(ExitCode::Success),
63        RebootCmd::Halt | RebootCmd::PowerOff => poweroff(ExitCode::Success),
64    }
65}
