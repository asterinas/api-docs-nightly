3use super::SyscallReturn;
4use crate::{
5    ipc::{
6        IpcFlags,
7        semaphore::system_v::{
8            PermissionMode,
9            sem_set::{SEMMNI, SEMMSL, check_sem, create_sem_set, create_sem_set_with_id},
10        },
11    },
12    prelude::*,
13};
14
15pub fn sys_semget(key: i32, nsems: i32, semflags: i32, ctx: &Context) -> Result<SyscallReturn> {
16    if nsems < 0 || nsems as usize > SEMMSL {
17        return_errno!(Errno::EINVAL);
18    }
19    if key < 0 {
20        return_errno!(Errno::EINVAL);
21    }
22
23    let flags = IpcFlags::from_bits_truncate(semflags as u32);
24    let mode: u16 = (semflags as u32 & 0x1FF) as u16;
25    let nsems = nsems as usize;
26    let credentials = ctx.posix_thread.credentials();
27
28    debug!(
29        "[sys_semget] key = {}, nsems = {}, flags = {:?}",
30        key, nsems, semflags
31    );
32
33    // Create a new semaphore set directly
34    const IPC_NEW: i32 = 0;
35    if key == IPC_NEW || (key as usize > SEMMNI && flags.contains(IpcFlags::IPC_CREAT)) {
36        if nsems == 0 {
37            return_errno!(Errno::EINVAL);
38        }
39        return Ok(SyscallReturn::Return(
40            create_sem_set(nsems, mode, credentials)? as isize,
41        ));
42    }
43
44    // Get a semaphore set, and create if necessary
45    match check_sem(
46        key,
47        Some(nsems),
48        PermissionMode::ALTER | PermissionMode::READ,
49    ) {
50        Ok(_) => {
51            if flags.contains(IpcFlags::IPC_CREAT | IpcFlags::IPC_EXCL) {
52                return_errno!(Errno::EEXIST);
53            }
54        }
55        Err(err) => {
56            let need_create = err.error() == Errno::ENOENT && flags.contains(IpcFlags::IPC_CREAT);
57            if !need_create {
58                return Err(err);
59            }
60            if nsems == 0 {
61                return_errno!(Errno::EINVAL);
62            }
63
64            create_sem_set_with_id(key, nsems, mode, credentials)?
65        }
66    };
67
68    Ok(SyscallReturn::Return(key as isize))
69}
