3use crate::{
4    fs::file_table::{FdFlags, FileDesc, get_file_fast},
5    prelude::*,
6    process::{PidFile, credentials::capabilities::CapSet, posix_thread::AsPosixThread},
7    syscall::SyscallReturn,
8};
9
10pub fn sys_pidfd_getfd(
11    pidfd: FileDesc,
12    targetfd: FileDesc,
13    flags: u32,
14    ctx: &Context,
15) -> Result<SyscallReturn> {
16    // The `flags` argument is reserved for future use. Currently, it must be specified as 0.
17    if flags != 0 {
18        return_errno_with_message!(Errno::EINVAL, "invalid flags");
19    }
20    debug!(
21        "pidfd_getfd: pidfd={}, targetfd={}, flags={}",
22        pidfd, targetfd, flags
23    );
24
25    let mut file_table = ctx.thread_local.borrow_file_table_mut();
26    let file = get_file_fast!(&mut file_table, pidfd);
27    let Some(pid_file) = file.downcast_ref::<PidFile>() else {
28        return_errno_with_message!(Errno::EBADF, "the file is not a PID file");
29    };
30
31    let process = pid_file
32        .process_opt()
33        .ok_or_else(|| Error::with_message(Errno::ESRCH, "the target process has been reaped"))?;
34
35    // The calling process should have PTRACE_MODE_ATTACH_REALCREDS permissions (see ptrace(2))
36    // over the process referred to by `pidfd`.
37    // Currently, this is implemented as requiring the calling process to have the
38    // CAP_SYS_PTRACE capability, which is stricter.
39    // TODO: Implement appropriate PTRACE_MODE_ATTACH_REALCREDS permission check.
40    if process
41        .user_ns()
42        .lock()
43        .check_cap(CapSet::SYS_PTRACE, ctx.posix_thread)
44        .is_err()
45    {
46        return_errno_with_message!(
47            Errno::EPERM,
48            "the calling process does not have the required permissions"
49        );
50    }
51
52    let main_thread = process.main_thread();
53
54    // Get the file description corresponding to to the file descriptor `targetfd` in the process
55    // referred to by the PID file.
56    let target_file_table = main_thread.as_posix_thread().unwrap().file_table();
57    let target_file = target_file_table
58        .lock()
59        .as_ref()
60        .ok_or_else(|| Error::with_message(Errno::ESRCH, "the target process has exited"))?
61        .read()
62        .get_file(targetfd)?
63        .clone();
64
65    // Duplicate the file descriptor into the caller's file descriptor table.
66    let new_fd = {
67        let mut file_table_locked = file_table.unwrap().write();
68        file_table_locked.insert(target_file, FdFlags::CLOEXEC)
69    };
70
71    Ok(SyscallReturn::Return(new_fd as _))
72}
