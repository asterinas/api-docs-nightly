17pub fn sys_openat(
18    dirfd: FileDesc,
19    path_addr: Vaddr,
20    flags: u32,
21    mode: u16,
22    ctx: &Context,
23) -> Result<SyscallReturn> {
24    let path = ctx.user_space().read_cstring(path_addr, MAX_FILENAME_LEN)?;
25    debug!(
26        "dirfd = {}, path = {:?}, flags = {}, mode = {}",
27        dirfd, path, flags, mode
28    );
29
30    let file_handle = {
31        let path = path.to_string_lossy();
32        let fs_path = FsPath::from_fd_and_path(dirfd, path.as_ref())?;
33
34        let fs_ref = ctx.thread_local.borrow_fs();
35        let mask_mode = mode & !fs_ref.umask().get();
36
37        let path_resolver = fs_ref.resolver().read();
38        do_open(
39            &path_resolver,
40            &fs_path,
41            flags,
42            InodeMode::from_bits_truncate(mask_mode),
43        )
44        .map_err(|err| match err.error() {
45            Errno::EINTR => Error::new(Errno::ERESTARTSYS),
46            _ => err,
47        })?
48    };
49
50    let fd = {
51        let file_table = ctx.thread_local.borrow_file_table();
52        let mut file_table_locked = file_table.unwrap().write();
53        let fd_flags =
54            if CreationFlags::from_bits_truncate(flags).contains(CreationFlags::O_CLOEXEC) {
55                FdFlags::CLOEXEC
56            } else {
57                FdFlags::empty()
58            };
59        file_table_locked.insert(file_handle.clone(), fd_flags)
60    };
61    let file_like: Arc<dyn FileLike> = file_handle;
62    fs::notify::on_open(&file_like);
63    Ok(SyscallReturn::Return(fd as _))
64}

66pub fn sys_open(path_addr: Vaddr, flags: u32, mode: u16, ctx: &Context) -> Result<SyscallReturn> {
67    self::sys_openat(AT_FDCWD, path_addr, flags, mode, ctx)
68}

70pub fn sys_creat(path_addr: Vaddr, mode: u16, ctx: &Context) -> Result<SyscallReturn> {
71    let flags =
72        AccessMode::O_WRONLY as u32 | CreationFlags::O_CREAT.bits() | CreationFlags::O_TRUNC.bits();
73    self::sys_openat(AT_FDCWD, path_addr, flags, mode, ctx)
74}

76fn do_open(
77    path_resolver: &PathResolver,
78    fs_path: &FsPath,
79    flags: u32,
80    mode: InodeMode,
81) -> Result<Arc<dyn FileLike>> {
82    let open_args = OpenArgs::from_flags_and_mode(flags, mode)?;
83
84    let lookup_res = if open_args.follow_tail_link() {
85        path_resolver.lookup_unresolved(fs_path)?
86    } else {
87        path_resolver.lookup_unresolved_no_follow(fs_path)?
88    };
89
90    let file_handle: Arc<dyn FileLike> = match lookup_res {
91        LookupResult::Resolved(path) => Arc::new(path.open(open_args)?),
92        LookupResult::AtParent(result) => {
93            if !open_args.creation_flags.contains(CreationFlags::O_CREAT)
94                || open_args.status_flags.contains(StatusFlags::O_PATH)
95            {
96                return_errno_with_message!(Errno::ENOENT, "the file does not exist");
97            }
98            if open_args
99                .creation_flags
100                .contains(CreationFlags::O_DIRECTORY)
101            {
102                return_errno_with_message!(
103                    Errno::EINVAL,
104                    "O_CREAT and O_DIRECTORY cannot be specified together"
105                );
106            }
107            if result.target_is_dir() {
108                return_errno_with_message!(
109                    Errno::EISDIR,
110                    "O_CREAT is specified but the file is a directory"
111                );
112            }
113
114            let (parent, tail_name) = result.into_parent_and_basename();
115            let new_path =
116                parent.new_fs_child(&tail_name, InodeType::File, open_args.inode_mode)?;
117            fs::notify::on_create(&parent, || tail_name.clone());
118
119            // Don't check access mode for newly created file.
120            Arc::new(InodeHandle::new_unchecked_access(
121                new_path,
122                open_args.access_mode,
123                open_args.status_flags,
124            )?)
125        }
126    };
127
128    Ok(file_handle)
129}
