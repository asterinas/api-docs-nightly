15pub fn sys_move_mount(
16    from_dfd: RawFileDesc,
17    from_path_addr: Vaddr,
18    to_dfd: RawFileDesc,
19    to_path_addr: Vaddr,
20    flags: u32,
21    ctx: &Context,
22) -> Result<SyscallReturn> {
23    let flags = MoveMountFlags::try_from(flags)?;
24    let supported_flags = MoveMountFlags::MOVE_MOUNT_F_EMPTY_PATH;
25    if !(flags - supported_flags).is_empty() {
26        return_errno_with_message!(Errno::EOPNOTSUPP, "unsupported move_mount flags");
27    }
28    super::fsopen::check_mount_api_capability(ctx)?;
29
30    let from_path = ctx
31        .user_space()
32        .read_cstring(from_path_addr, MAX_FILENAME_LEN)?;
33    let source = if from_path.is_empty() {
34        if !flags.contains(MoveMountFlags::MOVE_MOUNT_F_EMPTY_PATH) {
35            return_errno_with_message!(Errno::ENOENT, "the source path is empty");
36        }
37        MoveMountSource::Detached(get_detached_mount(from_dfd, ctx)?)
38    } else {
39        MoveMountSource::Path(from_path)
40    };
41
42    let to_path = ctx
43        .user_space()
44        .read_cstring(to_path_addr, MAX_FILENAME_LEN)?;
45
46    let fs_ref = ctx.thread_local.borrow_fs();
47    let path_resolver = fs_ref.resolver().read();
48    let to_path = to_path.to_string_lossy();
49    let to_fs_path = FsPath::from_fd_at(to_dfd, &to_path, EmptyPathStr::Reject)?;
50    let target_path = path_resolver.lookup(&to_fs_path)?;
51    match source {
52        MoveMountSource::Detached(detached_mount) => {
53            let detached_root = Path::new_fs_root(detached_mount);
54            detached_root.move_mount_to(&target_path, ctx)?;
55        }
56        MoveMountSource::Path(from_path) => {
57            let from_path = from_path.to_string_lossy();
58            let from_fs_path = FsPath::from_fd_at(from_dfd, &from_path, EmptyPathStr::Reject)?;
59            let source_path = path_resolver.lookup(&from_fs_path)?;
60            source_path.move_mount_to(&target_path, ctx)?;
61        }
62    };
63
64    Ok(SyscallReturn::Return(0))
65}

67enum MoveMountSource {
68    Detached(Arc<Mount>),
69    Path(CString),
70}

72fn get_detached_mount(from_dfd: RawFileDesc, ctx: &Context) -> Result<Arc<Mount>> {
73    let mut file_table = ctx.thread_local.borrow_file_table_mut();
74    let file = get_file_fast!(&mut file_table, from_dfd.try_into()?);
75    let mount_file = file
76        .downcast_ref::<DetachedMountFile>()
77        .ok_or_else(|| Error::with_message(Errno::EINVAL, "the file is not a detached mount"))?;
78
79    Ok(mount_file.mount())
80}

82bitflags! {
83    struct MoveMountFlags: u32 {
84        const MOVE_MOUNT_F_SYMLINKS   = 0x0000_0001;
85        const MOVE_MOUNT_F_AUTOMOUNTS = 0x0000_0002;
86        const MOVE_MOUNT_F_EMPTY_PATH = 0x0000_0004;
87        const MOVE_MOUNT_T_SYMLINKS   = 0x0000_0010;
88        const MOVE_MOUNT_T_AUTOMOUNTS = 0x0000_0020;
89        const MOVE_MOUNT_T_EMPTY_PATH = 0x0000_0040;
90        const MOVE_MOUNT_SET_GROUP    = 0x0000_0100;
91        const MOVE_MOUNT_BENEATH      = 0x0000_0200;
92    }
93}
94
95impl TryFrom<u32> for MoveMountFlags {
96    type Error = Error;
97
98    fn try_from(value: u32) -> Result<Self> {
99        Self::from_bits(value)
100            .ok_or_else(|| Error::with_message(Errno::EINVAL, "unknown move_mount flags"))
101    }
102}
