15pub fn sys_statfs(path_ptr: Vaddr, statfs_buf_ptr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
16    let user_space = ctx.user_space();
17    let path_name = user_space.read_cstring(path_ptr, PATH_MAX)?;
18    debug!(
19        "path = {:?}, statfs_buf_ptr = 0x{:x}",
20        path_name, statfs_buf_ptr,
21    );
22
23    let fs = {
24        let path_name = path_name.to_string_lossy();
25        let fs_path = FsPath::try_from(path_name.as_ref())?;
26        let path = ctx
27            .thread_local
28            .borrow_fs()
29            .resolver()
30            .read()
31            .lookup(&fs_path)?;
32        path.fs()
33    };
34
35    let statfs = Statfs::from(fs.sb());
36    user_space.write_val(statfs_buf_ptr, &statfs)?;
37    Ok(SyscallReturn::Return(0))
38}

40pub fn sys_fstatfs(fd: FileDesc, statfs_buf_ptr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
41    debug!("fd = {}, statfs_buf_addr = 0x{:x}", fd, statfs_buf_ptr);
42
43    let fs = {
44        let mut file_table = ctx.thread_local.borrow_file_table_mut();
45        let file = get_file_fast!(&mut file_table, fd);
46        file.path().fs()
47    };
48
49    let statfs = Statfs::from(fs.sb());
50    ctx.user_space().write_val(statfs_buf_ptr, &statfs)?;
51    Ok(SyscallReturn::Return(0))
52}

54/// FS Stat
55#[derive(Debug, Clone, Copy, Pod, Default)]
56#[repr(C)]
57struct Statfs {
58    /// Type of filesystem
59    f_type: u64,
60    /// Optimal transfer block size
61    f_bsize: usize,
62    /// Total data blocks in filesystem
63    f_blocks: usize,
64    /// Free blocks in filesystem
65    f_bfree: usize,
66    /// Free blocks available to unprivileged user
67    f_bavail: usize,
68    /// Total inodes in filesystem
69    f_files: usize,
70    /// Free inodes in filesystem
71    f_ffree: usize,
72    /// Filesystem ID
73    f_fsid: u64,
74    /// Maximum length of filenames
75    f_namelen: usize,
76    /// Fragment size
77    f_frsize: usize,
78    /// Mount flags of filesystem
79    f_flags: u64,
80    /// Padding bytes reserved for future use
81    f_spare: [u64; 4],
82}
83
84impl From<SuperBlock> for Statfs {
85    fn from(sb: SuperBlock) -> Self {
86        Self {
87            f_type: sb.magic,
88            f_bsize: sb.bsize,
89            f_blocks: sb.blocks,
90            f_bfree: sb.bfree,
91            f_bavail: sb.bavail,
92            f_files: sb.files,
93            f_ffree: sb.ffree,
94            f_fsid: sb.fsid,
95            f_namelen: sb.namelen,
96            f_frsize: sb.frsize,
97            f_flags: sb.flags,
98            f_spare: [0u64; 4],
99        }
100    }
101}
