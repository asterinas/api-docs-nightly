16pub fn sys_rt_sigprocmask(
17    how: u32,
18    set_ptr: Vaddr,
19    oldset_ptr: Vaddr,
20    sigset_size: usize,
21    ctx: &Context,
22) -> Result<SyscallReturn> {
23    let mask_op = MaskOp::try_from(how)?;
24    debug!(
25        "mask op = {:?}, set_ptr = 0x{:x}, oldset_ptr = 0x{:x}, sigset_size = {}",
26        mask_op, set_ptr, oldset_ptr, sigset_size
27    );
28    if sigset_size != 8 {
29        return_errno_with_message!(Errno::EINVAL, "sigset size is not equal to 8");
30    }
31    do_rt_sigprocmask(mask_op, set_ptr, oldset_ptr, ctx)?;
32    Ok(SyscallReturn::Return(0))
33}
34
35fn do_rt_sigprocmask(
36    mask_op: MaskOp,
37    set_ptr: Vaddr,
38    oldset_ptr: Vaddr,
39    ctx: &Context,
40) -> Result<()> {
41    let old_sig_mask_value = ctx.posix_thread.sig_mask().load(Ordering::Relaxed);
42    debug!("old sig mask value: 0x{:x}", old_sig_mask_value);
43    if oldset_ptr != 0 {
44        ctx.user_space()
45            .write_val(oldset_ptr, &old_sig_mask_value)?;
46    }
47
48    let sig_mask_ref = ctx.posix_thread.sig_mask();
49    if set_ptr != 0 {
50        let mut read_mask = ctx.user_space().read_val::<SigMask>(set_ptr)?;
51        match mask_op {
52            MaskOp::Block => {
53                // According to man pages, "it is not possible to block SIGKILL or SIGSTOP.
54                // Attempts to do so are silently ignored."
55                read_mask -= SIGKILL;
56                read_mask -= SIGSTOP;
57                sig_mask_ref.store(old_sig_mask_value + read_mask, Ordering::Relaxed);
58            }
59            MaskOp::Unblock => {
60                sig_mask_ref.store(old_sig_mask_value - read_mask, Ordering::Relaxed)
61            }
62            MaskOp::SetMask => {
63                read_mask -= SIGKILL; // Cannot block SIGKILL
64                read_mask -= SIGSTOP; // Cannot block SIGSTOP
65                sig_mask_ref.store(read_mask, Ordering::Relaxed);
66            }
67        }
68    }
69    debug!("new set = {:x?}", sig_mask_ref.load(Ordering::Relaxed));
70
71    Ok(())
72}
73
74#[derive(Debug, Clone, Copy, PartialEq, Eq, TryFromInt)]
75#[repr(u32)]
76pub enum MaskOp {
77    Block = 0,
78    Unblock = 1,
79    SetMask = 2,
80}
