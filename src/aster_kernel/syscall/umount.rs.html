3use super::SyscallReturn;
4use crate::{
5    fs::path::{AT_FDCWD, FsPath},
6    prelude::*,
7    syscall::constants::MAX_FILENAME_LEN,
8};
9
10pub fn sys_umount(path_addr: Vaddr, flags: u64, ctx: &Context) -> Result<SyscallReturn> {
11    let path_name = ctx.user_space().read_cstring(path_addr, MAX_FILENAME_LEN)?;
12    let umount_flags = UmountFlags::from_bits_truncate(flags as u32);
13    debug!("path = {:?}, flags = {:?}", path_name, umount_flags);
14
15    umount_flags.check_unsupported_flags()?;
16
17    let path_name = path_name.to_string_lossy();
18    let fs_path = FsPath::from_fd_and_path(AT_FDCWD, &path_name)?;
19
20    let target_path = if umount_flags.contains(UmountFlags::UMOUNT_NOFOLLOW) {
21        ctx.thread_local
22            .borrow_fs()
23            .resolver()
24            .read()
25            .lookup_no_follow(&fs_path)?
26    } else {
27        ctx.thread_local
28            .borrow_fs()
29            .resolver()
30            .read()
31            .lookup(&fs_path)?
32    };
33
34    target_path.unmount(ctx)?;
35
36    Ok(SyscallReturn::Return(0))
37}
38
39bitflags! {
40    struct UmountFlags: u32 {
41        const MNT_FORCE       = 0x00000001;	// Attempt to forcibily umount.
42        const MNT_DETACH      = 0x00000002;	// Just detach from the tree.
43        const MNT_EXPIRE      = 0x00000004;	// Mark for expiry.
44        const UMOUNT_NOFOLLOW = 0x00000008;	// Don't follow symlink on umount.
45    }
46}
47
48impl UmountFlags {
49    fn check_unsupported_flags(&self) -> Result<()> {
50        let supported_flags = UmountFlags::MNT_FORCE
51            | UmountFlags::MNT_DETACH
52            | UmountFlags::MNT_EXPIRE
53            | UmountFlags::UMOUNT_NOFOLLOW;
54        let unsupported_flags = *self - supported_flags;
55        if !unsupported_flags.is_empty() {
56            return_errno_with_message!(Errno::EINVAL, "unsupported flags");
57        }
58        Ok(())
59    }
60}
