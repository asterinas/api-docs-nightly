16pub fn sys_sendmsg(
17    sockfd: FileDesc,
18    user_msghdr_ptr: Vaddr,
19    flags: i32,
20    ctx: &Context,
21) -> Result<SyscallReturn> {
22    let user_space = ctx.user_space();
23    let c_user_msghdr: CUserMsgHdr = user_space.read_val(user_msghdr_ptr)?;
24    let flags = SendRecvFlags::from_bits_truncate(flags);
25
26    debug!(
27        "sockfd = {}, user_msghdr = {:x?}, flags = {:?}",
28        sockfd, c_user_msghdr, flags
29    );
30
31    let file = {
32        // Reading control messages may access the file table,
33        // so we have to clone the file and drop the file table reference here.
34        let file_table = ctx.thread_local.borrow_file_table();
35        let file_table_locked = file_table.unwrap().read();
36        file_table_locked.get_file(sockfd)?.clone()
37    };
38    let socket = file.as_socket_or_err()?;
39
40    let total_bytes = send_one_message(socket, &c_user_msghdr, &user_space, flags)?;
41
42    Ok(SyscallReturn::Return(total_bytes as _))
43}
44
45pub(super) fn send_one_message(
46    socket: &dyn Socket,
47    c_user_msghdr: &CUserMsgHdr,
48    user_space: &CurrentUserSpace,
49    flags: SendRecvFlags,
50) -> Result<usize> {
51    let message_header = {
52        let addr = c_user_msghdr.read_socket_addr_from_user()?;
53        let control_messages = c_user_msghdr.read_control_messages_from_user(user_space)?;
54        MessageHeader::new(addr, control_messages)
55    };
56    let mut io_vec_reader = c_user_msghdr.copy_reader_array_from_user(user_space)?;
57
58    socket
59        .sendmsg(&mut io_vec_reader, message_header, flags)
60        .map_err(|err| match err.error() {
61            // FIXME: `sendmsg` should not be restarted if a timeout has been set on the socket using `setsockopt`.
62            Errno::EINTR => Error::new(Errno::ERESTARTSYS),
63            _ => err,
64        })
65}
