23pub fn sys_pidfd_send_signal(
24    pidfd: FileDesc,
25    sig_num: u64,
26    info_ptr: Vaddr,
27    flags: u32,
28    ctx: &Context,
29) -> Result<SyscallReturn> {
30    let flags = PidfdSendSignalFlags::try_from(flags)?;
31    let sig_num = SigNum::try_from(sig_num as u8)?;
32    debug!(
33        "pidfd={}, info_ptr={:#x}, flags={:?}",
34        pidfd, info_ptr, flags
35    );
36
37    let siginfo = read_siginfo_from_user(info_ptr, sig_num, ctx)?;
38    let signal = RawSignal::new(siginfo);
39
40    let target = get_target_from_pidfd(pidfd, flags, ctx)?;
41
42    let is_self = match &target {
43        SignalTarget::Thread { tid, tgid: _ } => *tid == ctx.posix_thread.tid(),
44        SignalTarget::Process { pid } => *pid == ctx.posix_thread.tid(),
45        SignalTarget::ProcessGroup { pgid: _ } => false,
46    };
47
48    if !is_self && (siginfo.si_code >= 0 || siginfo.si_code == SI_TKILL) {
49        return_errno_with_message!(
50            Errno::EPERM,
51            "signals with custom code can only be sent to the current thread/process"
52        );
53    }
54
55    match target {
56        SignalTarget::Thread { tid, tgid } => {
57            tgkill(tid, tgid, Some(Box::new(signal) as Box<dyn Signal>), ctx)?;
58        }
59        SignalTarget::Process { pid } => {
60            kill(pid, Some(Box::new(signal) as Box<dyn Signal>), ctx)?;
61        }
62        SignalTarget::ProcessGroup { pgid } => {
63            kill_group(pgid, Some(signal), ctx)?;
64        }
65    }
66
67    Ok(SyscallReturn::Return(0))
68}

70fn read_siginfo_from_user(info_ptr: Vaddr, sig_num: SigNum, ctx: &Context) -> Result<siginfo_t> {
71    if info_ptr != 0 {
72        let si = ctx.user_space().read_val::<siginfo_t>(info_ptr)?;
73        if si.si_signo != sig_num.as_u8() as i32 {
74            return_errno_with_message!(
75                Errno::EINVAL,
76                "`siginfo.si_signo` does not match the specified signal number"
77            );
78        }
79        Ok(si)
80    } else {
81        // If `info_ptr` is NULL, the kernel constructs a default `siginfo_t` structure
82        // whose fields match the values that are implicitly supplied when a signal is sent using the kill(2).
83        Ok(UserSignal::new_kill(sig_num, ctx).to_info())
84    }
85}

87fn get_target_from_pidfd(
88    pidfd: FileDesc,
89    flags: PidfdSendSignalFlags,
90    ctx: &Context,
91) -> Result<SignalTarget> {
92    // Helper closures to create signal targets.
93    let thread_target = |tid: Tid, tgid: Pid| SignalTarget::Thread { tid, tgid };
94    let process_target = |pid: Pid| SignalTarget::Process { pid };
95    let group_target = |pgid: Pgid| SignalTarget::ProcessGroup { pgid };
96
97    let target = match pidfd {
98        PIDFD_SELF_THREAD => match flags {
99            PidfdSendSignalFlags::Default | PidfdSendSignalFlags::Thread => {
100                thread_target(ctx.posix_thread.tid(), ctx.process.pid())
101            }
102            PidfdSendSignalFlags::ThreadGroup => process_target(ctx.process.pid()),
103            PidfdSendSignalFlags::ProcessGroup => group_target(ctx.posix_thread.tid()),
104        },
105        PIDFD_SELF_THREAD_GROUP => match flags {
106            PidfdSendSignalFlags::Default | PidfdSendSignalFlags::ThreadGroup => {
107                process_target(ctx.process.pid())
108            }
109            PidfdSendSignalFlags::Thread => thread_target(
110                ctx.process.main_thread().as_posix_thread().unwrap().tid(),
111                ctx.process.pid(),
112            ),
113            PidfdSendSignalFlags::ProcessGroup => group_target(ctx.process.pid()),
114        },
115        _ => {
116            let mut file_table = ctx.thread_local.borrow_file_table_mut();
117            let file = get_file_fast!(&mut file_table, pidfd);
118
119            // FIXME: On Linux, a pidfd can be also obtained by opening a `/proc/pid` directory.
120            // Reference: <https://man7.org/linux/man-pages/man2/pidfd_send_signal.2.html>
121            let Some(pid_file) = file.downcast_ref::<PidFile>() else {
122                return_errno_with_message!(Errno::EBADF, "the file is not a PID file");
123            };
124
125            let Some(process) = pid_file.process_opt() else {
126                return_errno_with_message!(Errno::ESRCH, "the target process has been reaped");
127            };
128
129            match flags {
130                PidfdSendSignalFlags::Default => {
131                    // FIXME: On Linux, a pidfd can refer to either a process or a thread.
132                    // We currently only support pidfds that refer to processes.
133                    process_target(process.pid())
134                }
135                PidfdSendSignalFlags::Thread => {
136                    // FIXME: On Linux, the signal can be sent to any thread.
137                    // We currently only support the main thread.
138                    thread_target(
139                        process.main_thread().as_posix_thread().unwrap().tid(),
140                        process.pid(),
141                    )
142                }
143                PidfdSendSignalFlags::ThreadGroup => process_target(process.pid()),
144                PidfdSendSignalFlags::ProcessGroup => group_target(process.pid()),
145            }
146        }
147    };
148
149    Ok(target)
150}
151
152enum SignalTarget {
153    Thread { tid: Tid, tgid: Pid },
154    Process { pid: Pid },
155    ProcessGroup { pgid: Pgid },
156}

158// Reference: <https://elixir.bootlin.com/linux/v6.18/source/include/uapi/linux/pidfd.h#L19>.
159#[derive(Clone, Copy, PartialEq, Eq, Debug, TryFromInt)]
160#[repr(u32)]
161enum PidfdSendSignalFlags {
162    Default = 0x0,
163    Thread = 0x1,
164    ThreadGroup = 0x2,
165    ProcessGroup = 0x4,
166}
