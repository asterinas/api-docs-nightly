21pub fn sys_rt_sigaction(
22    sig_num: u8,
23    sig_action_addr: Vaddr,
24    old_sig_action_addr: Vaddr,
25    sigset_size: u64,
26    ctx: &Context,
27) -> Result<SyscallReturn> {
28    let sig_num = SigNum::try_from(sig_num)?;
29    debug!(
30        "signal = {}, sig_action_addr = 0x{:x}, old_sig_action_addr = 0x{:x}, sigset_size = {}",
31        sig_num.sig_name(),
32        sig_action_addr,
33        old_sig_action_addr,
34        sigset_size
35    );
36
37    if sigset_size != 8 {
38        return_errno_with_message!(Errno::EINVAL, "sigset size is not equal to 8");
39    }
40
41    let sig_dispositions = ctx.process.sig_dispositions().lock();
42    let mut sig_dispositions = sig_dispositions.lock();
43
44    let old_action = if sig_action_addr != 0 {
45        if sig_num == SIGKILL || sig_num == SIGSTOP {
46            return_errno_with_message!(
47                Errno::EINVAL,
48                "cannot set a new signal action for SIGKILL and SIGSTOP"
49            );
50        }
51
52        let sig_action_c = ctx.user_space().read_val::<sigaction_t>(sig_action_addr)?;
53        let sig_action = SigAction::from(sig_action_c);
54        trace!("sig action = {:?}", sig_action);
55        if sig_action.will_ignore(sig_num) {
56            discard_signals_if_ignored(ctx, sig_num);
57        }
58
59        sig_dispositions.set(sig_num, sig_action)?
60    } else {
61        sig_dispositions.get(sig_num)
62    };
63
64    if old_sig_action_addr != 0 {
65        let old_action_c = old_action.as_c_type();
66        ctx.user_space()
67            .write_val(old_sig_action_addr, &old_action_c)?;
68    }
69
70    Ok(SyscallReturn::Return(0))
71}

73/// Discard signals if the new action is to ignore the signal.
74///
75/// Ref: <https://elixir.bootlin.com/linux/v6.13/source/kernel/signal.c#L4323>
76//
77// POSIX 3.3.1.3:
78// Setting a signal action to SIG_IGN for a signal that is
79// pending shall cause the pending signal to be discarded,
80// whether or not it is blocked.
81//
82// Setting a signal action to SIG_DFL for a signal that is
83// pending and whose default action is to ignore the signal
84// (for example, SIGCHLD), shall cause the pending signal to
85// be discarded, whether or not it is blocked
86fn discard_signals_if_ignored(ctx: &Context, signum: SigNum) {
87    let mask = SigSet::new_full() - signum;
88
89    for task in ctx.process.tasks().lock().as_slice() {
90        let Some(posix_thread) = task.as_posix_thread() else {
91            continue;
92        };
93
94        while posix_thread.dequeue_signal(&mask).is_some() {}
95    }
96}
