3use super::SyscallReturn;
4use crate::{
5    fs::{
6        file_table::{FileDesc, get_file_fast},
7        utils::{FlockItem, FlockType},
8    },
9    prelude::*,
10};
11
12pub fn sys_flock(fd: FileDesc, ops: i32, ctx: &Context) -> Result<SyscallReturn> {
13    debug!("flock: fd: {}, ops: {:?}", fd, ops);
14
15    let mut file_table = ctx.thread_local.borrow_file_table_mut();
16    let file = get_file_fast!(&mut file_table, fd);
17    let inode_file = file.as_inode_handle_or_err()?;
18    let ops: FlockOps = FlockOps::from_i32(ops)?;
19    if ops.contains(FlockOps::LOCK_UN) {
20        inode_file.unlock_flock()?;
21    } else {
22        let is_nonblocking = ops.contains(FlockOps::LOCK_NB);
23        let flock = {
24            let type_ = FlockType::from(ops);
25            FlockItem::new(&file, type_)
26        };
27        inode_file
28            .set_flock(flock, is_nonblocking)
29            .map_err(|err| match err.error() {
30                Errno::EINTR => Error::new(Errno::ERESTARTSYS),
31                _ => err,
32            })?;
33    }
34    Ok(SyscallReturn::Return(0))
35}
36
37impl From<FlockOps> for FlockType {
38    fn from(ops: FlockOps) -> Self {
39        if ops.contains(FlockOps::LOCK_EX) {
40            Self::ExclusiveLock
41        } else if ops.contains(FlockOps::LOCK_SH) {
42            Self::SharedLock
43        } else {
44            panic!("invalid flockops");
45        }
46    }
47}
48
49bitflags! {
50    struct FlockOps: i32 {
51        /// Shared lock
52        const LOCK_SH = 1;
53        /// Exclusive lock
54        const LOCK_EX = 2;
55        // Or'd with one of the above to prevent blocking
56        const LOCK_NB = 4;
57        // Remove lock
58        const LOCK_UN = 8;
59    }
60}
61
62impl FlockOps {
63    fn from_i32(bits: i32) -> Result<Self> {
64        if let Some(ops) = Self::from_bits(bits) {
65            if ops.contains(Self::LOCK_SH) {
66                if ops.contains(Self::LOCK_EX) || ops.contains(Self::LOCK_UN) {
67                    return_errno_with_message!(Errno::EINVAL, "invalid operation");
68                }
69            } else if ops.contains(Self::LOCK_EX) {
70                if ops.contains(Self::LOCK_UN) {
71                    return_errno_with_message!(Errno::EINVAL, "invalid operation");
72                }
73            } else if !ops.contains(Self::LOCK_UN) {
74                return_errno_with_message!(Errno::EINVAL, "invalid operation");
75            }
76            Ok(ops)
77        } else {
78            return_errno_with_message!(Errno::EINVAL, "invalid operation");
79        }
80    }
81}
