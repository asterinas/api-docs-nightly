3use aster_time::read_monotonic_time;
4use ostd::mm::VmIo;
5
6use super::SyscallReturn;
7use crate::{prelude::*, process::process_table};
8
9#[repr(C)]
10#[padding_struct]
11#[derive(Debug, Default, Clone, Copy, Pod)]
12struct SysInfo {
13    uptime: i64,     /* Seconds since boot */
14    loads: [u64; 3], /* 1, 5, and 15 minute load averages */
15    totalram: u64,   /* Total usable main memory size */
16    freeram: u64,    /* Available memory size */
17    sharedram: u64,  /* Amount of shared memory */
18    bufferram: u64,  /* Memory used by buffers */
19    totalswap: u64,  /* Total swap space size */
20    freeswap: u64,   /* swap space still available */
21    procs: u16,      /* Number of current processes */
22    totalhigh: u64,  /* Total high memory size */
23    freehigh: u64,   /* Available high memory size */
24    mem_unit: u32,   /* Memory unit size in bytes */
25}
26
27pub fn sys_sysinfo(sysinfo_addr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
28    let info = SysInfo {
29        uptime: read_monotonic_time().as_secs() as i64,
30        totalram: crate::vm::mem_total() as u64,
31        freeram: osdk_frame_allocator::load_total_free_size() as u64,
32        procs: process_table::process_num() as u16,
33        // `mem_unit` will always be 1 byte since Asterinas only supports
34        // 64-bit CPU architectures.
35        mem_unit: 1,
36        ..Default::default() // TODO: add other system information
37    };
38    ctx.user_space().write_val(sysinfo_addr, &info)?;
39    Ok(SyscallReturn::Return(0))
40}
