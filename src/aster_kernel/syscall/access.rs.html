13pub fn sys_faccessat(
14    dirfd: FileDesc,
15    path_ptr: Vaddr,
16    mode: u16,
17    ctx: &Context,
18) -> Result<SyscallReturn> {
19    debug!(
20        "faccessat: dirfd = {}, path_ptr = {:#x}, mode = {:o}",
21        dirfd, path_ptr, mode
22    );
23
24    do_faccessat(dirfd, path_ptr, mode, 0, ctx)
25}

27pub fn sys_access(path_ptr: Vaddr, mode: u16, ctx: &Context) -> Result<SyscallReturn> {
28    debug!("access: path_ptr = {:#x}, mode = {:o}", path_ptr, mode);
29
30    do_faccessat(AT_FDCWD, path_ptr, mode, 0, ctx)
31}

48bitflags! {
49    struct FaccessatFlags: u32 {
50        const AT_EACCESS = 0x200;
51        const AT_SYMLINK_NOFOLLOW = 0x100;
52        const AT_EMPTY_PATH = 0x1000;
53    }
54}
55
56bitflags! {
57    struct AccessMode: u16 {
58        const R_OK = 0x4;
59        const W_OK = 0x2;
60        const X_OK = 0x1;
61        // We could ignore F_OK in bitflags.
62        // const F_OK = 0x0;
63    }
64}

66pub fn do_faccessat(
67    dirfd: FileDesc,
68    path_ptr: Vaddr,
69    mode: u16,
70    flags: u32,
71    ctx: &Context,
72) -> Result<SyscallReturn> {
73    let mode = AccessMode::from_bits(mode)
74        .ok_or_else(|| Error::with_message(Errno::EINVAL, "Invalid mode"))?;
75    let flags = FaccessatFlags::from_bits(flags)
76        .ok_or_else(|| Error::with_message(Errno::EINVAL, "Invalid flags"))?;
77
78    let path_name = ctx.user_space().read_cstring(path_ptr, PATH_MAX)?;
79    debug!(
80        "dirfd = {}, path_name = {:?}, mode = {:o}, flags = {:?}",
81        dirfd, path_name, mode, flags
82    );
83
84    let path = {
85        let path_name = path_name.to_string_lossy();
86        let fs_path = if flags.contains(FaccessatFlags::AT_EMPTY_PATH) && path_name.is_empty() {
87            FsPath::from_fd(dirfd)?
88        } else {
89            FsPath::from_fd_and_path(dirfd, &path_name)?
90        };
91
92        let fs_ref = ctx.thread_local.borrow_fs();
93        let path_resolver = fs_ref.resolver().read();
94        if flags.contains(FaccessatFlags::AT_SYMLINK_NOFOLLOW) {
95            path_resolver.lookup_no_follow(&fs_path)?
96        } else {
97            path_resolver.lookup(&fs_path)?
98        }
99    };
100
101    // AccessMode::empty() means F_OK and no more permission check needed.
102    if mode.is_empty() {
103        return Ok(SyscallReturn::Return(0));
104    }
105
106    let inode = path.inode();
107
108    // FIXME: The current implementation is dummy
109    if mode.contains(AccessMode::R_OK) {
110        inode.check_permission(Permission::MAY_READ)?;
111    }
112
113    if mode.contains(AccessMode::W_OK) {
114        inode.check_permission(Permission::MAY_WRITE)?;
115    }
116
117    if mode.contains(AccessMode::X_OK) {
118        inode.check_permission(Permission::MAY_EXEC)?;
119    }
120
121    Ok(SyscallReturn::Return(0))
122}
