3use ostd::mm::VmIo;
4
5use super::SyscallReturn;
6use crate::prelude::*;
7
8pub fn sys_getgroups(size: i32, group_list_addr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
9    debug!("size = {}, group_list_addr = 0x{:x}", size, group_list_addr);
10
11    if size < 0 {
12        return_errno_with_message!(Errno::EINVAL, "size cannot be negative");
13    }
14
15    let credentials = ctx.posix_thread.credentials();
16    let groups = credentials.groups();
17
18    if size == 0 {
19        return Ok(SyscallReturn::Return(groups.len() as _));
20    }
21
22    if groups.len() > size as usize {
23        return_errno_with_message!(
24            Errno::EINVAL,
25            "size is less than the number of supplementary group IDs"
26        );
27    }
28
29    let user_space = ctx.user_space();
30    for (idx, gid) in groups.iter().enumerate() {
31        let addr = group_list_addr + idx * size_of_val(gid);
32        user_space.write_val(addr, gid)?;
33    }
34
35    Ok(SyscallReturn::Return(groups.len() as _))
36}
