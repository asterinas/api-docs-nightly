17const STATX_ATTR_MOUNT_ROOT: u64 = 0x0000_2000;
18
19pub fn sys_statx(
20    dirfd: FileDesc,
21    filename_ptr: Vaddr,
22    flags: u32,
23    mask: u32,
24    statx_buf_ptr: Vaddr,
25    ctx: &Context,
26) -> Result<SyscallReturn> {
27    let user_space = ctx.user_space();
28    let filename = user_space.read_cstring(filename_ptr, MAX_FILENAME_LEN)?;
29    let flags = StatxFlags::from_bits(flags)
30        .ok_or(Error::with_message(Errno::EINVAL, "invalid statx flags"))?;
31    let mask = StatxMask::from_bits_truncate(mask);
32    debug!(
33        "dirfd = {}, filename = {:?}, flags = {:?}, mask = {:?}, statx_buf_ptr = 0x{:x}",
34        dirfd, filename, flags, mask, statx_buf_ptr,
35    );
36
37    if flags.contains(StatxFlags::AT_STATX_FORCE_SYNC)
38        && flags.contains(StatxFlags::AT_STATX_DONT_SYNC)
39    {
40        return_errno_with_message!(Errno::EINVAL, "invalid statx flags");
41    }
42
43    if mask.contains(StatxMask::STATX_RESERVED) {
44        return_errno_with_message!(
45            Errno::EINVAL,
46            "mask reserved for future struct statx expansion"
47        );
48    }
49
50    let path = {
51        let filename = filename.to_string_lossy();
52        let fs_path = if flags.contains(StatxFlags::AT_EMPTY_PATH) && filename.is_empty() {
53            // TODO: We can retrieve the metadata from `FileLike`. See `sys_fstat` as an example.
54            FsPath::from_fd(dirfd)?
55        } else {
56            FsPath::from_fd_and_path(dirfd, &filename)?
57        };
58
59        let fs_ref = ctx.thread_local.borrow_fs();
60        let path_resolver = fs_ref.resolver().read();
61        if flags.contains(StatxFlags::AT_SYMLINK_NOFOLLOW) {
62            path_resolver.lookup_no_follow(&fs_path)?
63        } else {
64            path_resolver.lookup(&fs_path)?
65        }
66    };
67
68    let statx = Statx::new(&path);
69
70    user_space.write_val(statx_buf_ptr, &statx)?;
71    Ok(SyscallReturn::Return(0))
72}

74/// Structures for the extended file attribute retrieval system call statx.
75#[derive(Debug, Clone, Copy, Pod, Default)]
76#[repr(C)]
77pub struct Statx {
78    /// Indicates which fields in the `statx` structure were successfully filled,
79    /// reflecting the state information supported by the filesystem.
80    stx_mask: u32,
81    /// Preferred general I/O size
82    stx_blksize: u32,
83    /// Flags conveying information about the file
84    stx_attributes: u64,
85    /// Number of hard links
86    stx_nlink: u32,
87    /// User ID of owner
88    stx_uid: u32,
89    /// Group ID of owner
90    stx_gid: u32,
91    /// File mode
92    stx_mode: u16,
93    /// Padding
94    __spare0: [u16; 1],
95    /// Inode number
96    stx_ino: u64,
97    /// File size
98    stx_size: u64,
99    /// Number of 512-byte blocks allocated
100    stx_blocks: u64,
101    /// Mask to show what's supported in stx_attributes
102    stx_attributes_mask: u64,
103    /// Last access time
104    stx_atime: StatxTimestamp,
105    /// File creation time
106    stx_btime: StatxTimestamp,
107    /// Last attribute change time
108    stx_ctime: StatxTimestamp,
109    /// Last data modification time
110    stx_mtime: StatxTimestamp,
111    /// Device ID of special file (if bdev/cdev)
112    stx_rdev_major: u32,
113    stx_rdev_minor: u32,
114    /// ID of device containing file
115    stx_dev_major: u32,
116    stx_dev_minor: u32,
117    /// Mount ID
118    stx_mnt_id: u64,
119    /// Memory buffer alignment for direct I/O
120    stx_dio_mem_align: u32,
121    /// File offset alignment for direct I/O
122    stx_dio_offset_align: u32,
123    /// Spare space for future expansion
124    __spare3: [u64; 12],
125}
126
127impl Statx {
128    fn new(path: &Path) -> Self {
129        let info = path.metadata();
130
131        let (stx_dev_major, stx_dev_minor) = device_id::decode_device_numbers(info.dev);
132        let (stx_rdev_major, stx_rdev_minor) = device_id::decode_device_numbers(info.rdev);
133
134        // TODO: Support more `stx_attributes` flags.
135        let stx_attributes_mask = STATX_ATTR_MOUNT_ROOT;
136
137        let mut stx_attributes = 0;
138        if path.is_mount_root() {
139            stx_attributes |= STATX_ATTR_MOUNT_ROOT;
140        }
141
142        let stx_mask = StatxMask::STATX_BASIC_STATS.bits()
143            | StatxMask::STATX_BTIME.bits()
144            | StatxMask::STATX_MNT_ID.bits();
145
146        Self {
147            // FIXME: All zero fields below are dummy implementations that need to be improved in the future.
148            stx_mask,
149            stx_blksize: info.blk_size as u32,
150            stx_attributes,
151            stx_nlink: info.nlinks as u32,
152            stx_uid: info.uid.into(),
153            stx_gid: info.gid.into(),
154            stx_mode: info.type_ as u16 | info.mode.bits(),
155            __spare0: [0; 1],
156            stx_ino: info.ino,
157            stx_size: info.size as u64,
158            stx_blocks: (info.blocks * (info.blk_size / 512)) as u64,
159            stx_attributes_mask,
160            stx_atime: StatxTimestamp::from(info.atime),
161            stx_btime: StatxTimestamp::from(info.atime),
162            stx_ctime: StatxTimestamp::from(info.ctime),
163            stx_mtime: StatxTimestamp::from(info.ctime),
164            stx_rdev_major,
165            stx_rdev_minor,
166            stx_dev_major,
167            stx_dev_minor,
168            stx_mnt_id: path.mount_node().id() as u64,
169            stx_dio_mem_align: 0,
170            stx_dio_offset_align: 0,
171            __spare3: [0; 12],
172        }
173    }
174}

176/// Statx Timestamp (seconds and nanoseconds)
177#[derive(Debug, Clone, Copy, Pod, Default)]
178#[repr(C)]
179pub struct StatxTimestamp {
180    /// Seconds
181    tv_sec: i64,
182    /// Nanoseconds
183    tv_nsec: u32,
184    __reserved: i32,
185}
186
187impl From<Duration> for StatxTimestamp {
188    fn from(duration: Duration) -> Self {
189        Self {
190            tv_sec: duration.as_secs() as i64,
191            tv_nsec: duration.subsec_nanos(),
192            __reserved: 0,
193        }
194    }
195}

197bitflags! {
198    /// Flags can be used to influence a pathname-based lookup.
199    /// Flags can also be used to control what sort of synchronization the
200    /// kernel will do when querying a file on a remote filesystem.
201    struct StatxFlags: u32 {
202        /// Allow empty relative pathname to operate on dirfd directly.
203        const AT_EMPTY_PATH         = 1 << 12;
204        /// Suppress terminal automount traversal.
205        const AT_NO_AUTOMOUNT       = 1 << 11;
206        /// Do not follow symbolic links.
207        const AT_SYMLINK_NOFOLLOW   = 1 << 8;
208        /// Do whatever stat() does.
209        const AT_STATX_SYNC_AS_STAT = 0;
210        /// Force the attributes to be sync'd with the server.
211        const AT_STATX_FORCE_SYNC   = 1 << 13;
212        /// Don't sync attributes with the server.
213        const AT_STATX_DONT_SYNC    = 1 << 14;
214    }
215}

217bitflags! {
218    /// Flags to be stx_mask.
219    /// Query request/result mask for statx() and struct statx::stx_mask.
220    /// These bits should be set in the mask argument of statx() to request
221    /// particular items when calling statx().
222    pub struct StatxMask: u32 {
223        /// Want stx_mode & S_IFMT
224        const STATX_TYPE            = 0x00000001;
225        /// Want stx_mode & ~S_IFMT
226        const STATX_MODE            = 0x00000002;
227        /// Want stx_nlink
228        const STATX_NLINK           = 0x00000004;
229        /// Want stx_uid
230        const STATX_UID             = 0x00000008;
231        /// Want stx_gid
232        const STATX_GID             = 0x00000010;
233        /// Want stx_atime
234        const STATX_ATIME           = 0x00000020;
235        /// Want stx_mtime
236        const STATX_MTIME           = 0x00000040;
237        /// Want stx_ctime
238        const STATX_CTIME           = 0x00000080;
239        /// Want stx_ino
240        const STATX_INO             = 0x00000100;
241        /// Want stx_size
242        const STATX_SIZE            = 0x00000200;
243        /// Want stx_blocks
244        const STATX_BLOCKS          = 0x00000400;
245        /// All of the above (stx_mode, stx_nlink, etc.)
246        const STATX_BASIC_STATS     = 0x000007ff;
247        /// Want stx_btime
248        const STATX_BTIME           = 0x00000800;
249        /// Want stx_mnt_id
250        const STATX_MNT_ID          = 0x00001000;
251        /// Want stx_dio_mem_align and stx_dio_offset_align
252        const STATX_DIOALIGN        = 0x00002000;
253        /// Want extended stx_mount_id
254        const STATX_MNT_ID_UNIQUE   = 0x00004000;
255        /// Want stx_subvol
256        const STATX_SUBVOL          = 0x00008000;
257        /// Want stx_write_atomic
258        const STATX_WRITE_ATOMIC    = 0x00010000;
259        /// Wand dio read alignment info
260        const STATX_DIO_READ_ALIGN  = 0x00020000;
261        /// Reserved for future struct statx expansion
262        const STATX_RESERVED		= 0x80000000;
263        /// Deprecated: The same as STATX_BASIC_STATS | STATX_BTIME
264        const STATX_ALL             = 0x00000fff;
265    }
