17pub fn sys_listxattr(
18    path_ptr: Vaddr,
19    list_ptr: Vaddr, // The given list is used to place xattr (null-terminated) names.
20    list_len: usize,
21    ctx: &Context,
22) -> Result<SyscallReturn> {
23    let user_space = ctx.user_space();
24    let path = user_space.read_cstring(path_ptr, MAX_FILENAME_LEN)?;
25
26    let len = listxattr(
27        XattrFileCtx::Path(path),
28        list_ptr,
29        list_len,
30        &user_space,
31        ctx,
32    )?;
33
34    Ok(SyscallReturn::Return(len as _))
35}

57pub fn sys_flistxattr(
58    fd: FileDesc,
59    list_ptr: Vaddr, // The given list is used to place xattr (null-terminated) names.
60    list_len: usize,
61    ctx: &Context,
62) -> Result<SyscallReturn> {
63    let mut file_table = ctx.thread_local.borrow_file_table_mut();
64    let file = get_file_fast!(&mut file_table, fd);
65
66    let user_space = ctx.user_space();
67    let len = listxattr(
68        XattrFileCtx::FileHandle(file),
69        list_ptr,
70        list_len,
71        &user_space,
72        ctx,
73    )?;
74
75    Ok(SyscallReturn::Return(len as _))
76}

78fn listxattr(
79    file_ctx: XattrFileCtx,
80    list_ptr: Vaddr,
81    list_len: usize,
82    user_space: &CurrentUserSpace,
83    ctx: &Context,
84) -> Result<usize> {
85    if list_len > XATTR_LIST_MAX_LEN {
86        return_errno_with_message!(Errno::E2BIG, "xattr list too long");
87    }
88
89    let namespace = get_current_xattr_namespace(ctx);
90    let mut list_writer = user_space.writer(list_ptr, list_len)?;
91
92    let path = lookup_path_for_xattr(&file_ctx, ctx)?;
93    path.list_xattr(namespace, &mut list_writer)
94}
95
96fn get_current_xattr_namespace(ctx: &Context) -> XattrNamespace {
97    let credentials = ctx.posix_thread.credentials();
98    let permitted_capset = credentials.permitted_capset();
99    let effective_capset = credentials.effective_capset();
100
101    if permitted_capset.contains(CapSet::SYS_ADMIN) && effective_capset.contains(CapSet::SYS_ADMIN)
102    {
103        XattrNamespace::Trusted
104    } else {
105        XattrNamespace::User
106    }
107}
