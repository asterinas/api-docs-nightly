7        vfs::path::{AT_FDCWD, EmptyPathStr, FsPath, SplitPath, SplitPathError},
8    },
9    prelude::*,
10    syscall::constants::MAX_FILENAME_LEN,
11};

13pub fn sys_unlinkat(
14    dirfd: RawFileDesc,
15    path_addr: Vaddr,
16    flags: u32,
17    ctx: &Context,
18) -> Result<SyscallReturn> {
19    let flags =
20        UnlinkFlags::from_bits(flags).ok_or(Error::with_message(Errno::EINVAL, "invalid flags"))?;
21    if flags.contains(UnlinkFlags::AT_REMOVEDIR) {
22        return super::rmdir::sys_rmdirat(dirfd, path_addr, ctx);
23    }
24
25    let path_name = ctx.user_space().read_cstring(path_addr, MAX_FILENAME_LEN)?;
26    debug!("dirfd = {}, path = {:?}", dirfd, path_name);
27
28    let path_name = path_name.to_string_lossy();
29    let (dir_path, name) = {
30        let (parent_path_name, target_name) = path_name
31            .split_dirname_and_basename()
32            .map_err(SplitPathError::reject_root_as_is_dir)?;
33
34        let fs_ref = ctx.thread_local.borrow_fs();
35        let path_resolver = fs_ref.resolver().read();
36
37        let parent_fs_path = FsPath::from_fd_at(dirfd, parent_path_name, EmptyPathStr::Reject)?;
38        let dir_path = path_resolver.lookup(&parent_fs_path)?;
39
40        // A trailing slash makes the unlink fail in one of two ways:
41        // `dir/` with `EISDIR`, `file/` with `ENOTDIR`.
42        if path_name.ends_with('/') {
43            let target = path_resolver.lookup_at_path(&dir_path, target_name)?;
44            if target.type_() == InodeType::Dir {
45                return_errno_with_message!(Errno::EISDIR, "a directory cannot be unlinked");
46            } else {
47                return_errno_with_message!(
48                    Errno::ENOTDIR,
49                    "the path ends with a slash but is not a directory"
50                );
51            }
52        }
53
54        (dir_path, target_name)
55    };
56
57    dir_path.unlink(name)?;
58    Ok(SyscallReturn::Return(0))
59}
60
61pub fn sys_unlink(path_addr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
62    sys_unlinkat(AT_FDCWD, path_addr, 0, ctx)
63}
