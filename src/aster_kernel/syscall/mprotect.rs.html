3use align_ext::AlignExt;
4
5use super::SyscallReturn;
6use crate::{
7    prelude::*,
8    vm::{perms::VmPerms, vmar::VMAR_CAP_ADDR},
9};
10
11pub fn sys_mprotect(addr: Vaddr, len: usize, perms: u64, ctx: &Context) -> Result<SyscallReturn> {
12    let vm_perms = VmPerms::from_user_bits(perms as u32)?;
13    debug!(
14        "addr = 0x{:x}, len = 0x{:x}, perms = {:?}",
15        addr, len, vm_perms
16    );
17
18    // According to Linux behavior,
19    // <https://elixir.bootlin.com/linux/v6.0.9/source/mm/mprotect.c#L681>,
20    // `addr` is checked even if `len` is 0.
21    if !addr.is_multiple_of(PAGE_SIZE) {
22        return_errno_with_message!(Errno::EINVAL, "the mapping address is not aligned");
23    }
24    if len == 0 {
25        return Ok(SyscallReturn::Return(0));
26    }
27    if VMAR_CAP_ADDR.checked_sub(addr).is_none_or(|gap| gap < len) {
28        // FIXME: Linux returns `ENOMEM` if `(addr + len).align_up(PAGE_SIZE)` overflows. Here, we
29        // perform a stricter validation.
30        return_errno_with_message!(Errno::ENOMEM, "the mapping range is not in userspace");
31    }
32    let addr_range = addr..(addr + len).align_up(PAGE_SIZE);
33
34    // On x86_64 and riscv64, `PROT_WRITE` implies `PROT_READ`.
35    // Reference:
36    // <https://man7.org/linux/man-pages/man2/mprotect.2.html>,
37    // Section 5.11.3 from <https://www.intel.com/content/dam/www/public/us/en/documents/manuals/64-ia-32-architectures-software-developer-vol-3a-part-1-manual.pdf>,
38    // <https://riscv.github.io/riscv-isa-manual/snapshot/privileged/#translation>.
39    #[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
40    let vm_perms = if !vm_perms.contains(VmPerms::READ) && vm_perms.contains(VmPerms::WRITE) {
41        vm_perms | VmPerms::READ
42    } else {
43        vm_perms
44    };
45
46    let user_space = ctx.user_space();
47    let vmar = user_space.vmar();
48    vmar.protect(vm_perms, addr_range)?;
49
50    Ok(SyscallReturn::Return(0))
51}
