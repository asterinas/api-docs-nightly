3use super::SyscallReturn;
4use crate::{
5    prelude::*,
6    process::{Pid, process_table},
7};
8
9pub fn sys_getpgid(pid: Pid, ctx: &Context) -> Result<SyscallReturn> {
10    debug!("pid = {}", pid);
11
12    // The documentation quoted below is from
13    // <https://www.man7.org/linux/man-pages/man2/getpgid.2.html>.
14
15    // "If `pid` is equal to 0, getpgid() shall return the process group ID of the calling
16    // process."
17    if pid == 0 {
18        return Ok(SyscallReturn::Return(ctx.process.pgid() as _));
19    }
20
21    let process = process_table::get_process(pid).ok_or(Error::with_message(
22        Errno::ESRCH,
23        "the process to get the PGID does not exist",
24    ))?;
25
26    // The man pages allow the implementation to return `EPERM` if `process` is in a different
27    // session than the current process. Linux does not perform this check by default, but some
28    // strict security policies (e.g. SELinux) may do so.
29
30    Ok(SyscallReturn::Return(process.pgid() as _))
31}
