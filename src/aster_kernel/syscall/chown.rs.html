14pub fn sys_fchown(fd: FileDesc, uid: i32, gid: i32, ctx: &Context) -> Result<SyscallReturn> {
15    debug!("fd = {}, uid = {}, gid = {}", fd, uid, gid);
16
17    let uid = to_optional_id(uid, Uid::new)?;
18    let gid = to_optional_id(gid, Gid::new)?;
19    if uid.is_none() && gid.is_none() {
20        return Ok(SyscallReturn::Return(0));
21    }
22
23    let mut file_table = ctx.thread_local.borrow_file_table_mut();
24    let file = get_file_fast!(&mut file_table, fd);
25    let path = file.path();
26    if let Some(uid) = uid {
27        path.set_owner(uid)?;
28    }
29    if let Some(gid) = gid {
30        path.set_group(gid)?;
31    }
32    Ok(SyscallReturn::Return(0))
33}

35pub fn sys_chown(path_ptr: Vaddr, uid: i32, gid: i32, ctx: &Context) -> Result<SyscallReturn> {
36    self::sys_fchownat(AT_FDCWD, path_ptr, uid, gid, 0, ctx)
37}

39pub fn sys_lchown(path_ptr: Vaddr, uid: i32, gid: i32, ctx: &Context) -> Result<SyscallReturn> {
40    self::sys_fchownat(
41        AT_FDCWD,
42        path_ptr,
43        uid,
44        gid,
45        ChownFlags::AT_SYMLINK_NOFOLLOW.bits(),
46        ctx,
47    )
48}

50pub fn sys_fchownat(
51    dirfd: FileDesc,
52    path_ptr: Vaddr,
53    uid: i32,
54    gid: i32,
55    flags: u32,
56    ctx: &Context,
57) -> Result<SyscallReturn> {
58    let path_name = ctx.user_space().read_cstring(path_ptr, PATH_MAX)?;
59    let flags = ChownFlags::from_bits(flags)
60        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid flags"))?;
61    debug!(
62        "dirfd = {}, path = {:?}, uid = {}, gid = {}, flags = {:?}",
63        dirfd, path_name, uid, gid, flags
64    );
65
66    if flags.contains(ChownFlags::AT_EMPTY_PATH) && path_name.is_empty() {
67        return self::sys_fchown(dirfd, uid, gid, ctx);
68    }
69
70    let uid = to_optional_id(uid, Uid::new)?;
71    let gid = to_optional_id(gid, Gid::new)?;
72    if uid.is_none() && gid.is_none() {
73        return Ok(SyscallReturn::Return(0));
74    }
75
76    let path = {
77        let path_name = path_name.to_string_lossy();
78        let fs_path = FsPath::from_fd_and_path(dirfd, &path_name)?;
79
80        let fs_ref = ctx.thread_local.borrow_fs();
81        let path_resolver = fs_ref.resolver().read();
82        if flags.contains(ChownFlags::AT_SYMLINK_NOFOLLOW) {
83            path_resolver.lookup_no_follow(&fs_path)?
84        } else {
85            path_resolver.lookup(&fs_path)?
86        }
87    };
88
89    if let Some(uid) = uid {
90        path.set_owner(uid)?;
91    }
92    if let Some(gid) = gid {
93        path.set_group(gid)?;
94    }
95    Ok(SyscallReturn::Return(0))
96}

98fn to_optional_id<T>(id: i32, f: impl Fn(u32) -> T) -> Result<Option<T>> {
99    let id = if id >= 0 {
100        Some(f(id as u32))
101    } else if id == -1 {
102        // If the owner or group is specified as -1, then that ID is not changed.
103        None
104    } else {
105        return_errno!(Errno::EINVAL);
106    };
107
108    Ok(id)
109}
110
111bitflags! {
112    struct ChownFlags: u32 {
113        const AT_SYMLINK_NOFOLLOW = 1 << 8;
114        const AT_EMPTY_PATH = 1 << 12;
115    }
