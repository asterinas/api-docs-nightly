17pub fn sys_getrlimit(resource: u32, rlim_addr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
18    let old_raw = do_prlimit64(&ctx.process, resource, None, ctx)?;
19    ctx.user_space().write_val(rlim_addr, &old_raw)?;
20    Ok(SyscallReturn::Return(0))
21}
22
23pub fn sys_setrlimit(resource: u32, new_rlim_addr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
24    let new_raw = ctx.user_space().read_val(new_rlim_addr)?;
25    do_prlimit64(&ctx.process, resource, Some(new_raw), ctx)?;
26    Ok(SyscallReturn::Return(0))
27}

29pub fn sys_prlimit64(
30    pid: Pid,
31    resource: u32,
32    new_rlim_addr: Vaddr,
33    old_rlim_addr: Vaddr,
34    ctx: &Context,
35) -> Result<SyscallReturn> {
36    let user_space = ctx.user_space();
37
38    let new_raw = if new_rlim_addr == 0 {
39        None
40    } else {
41        Some(user_space.read_val(new_rlim_addr)?)
42    };
43
44    let old_raw = if pid == 0 || pid == ctx.process.pid() {
45        do_prlimit64(&ctx.process, resource, new_raw, ctx)?
46    } else {
47        let target_process = process_table::get_process(pid).ok_or_else(|| {
48            Error::with_message(Errno::ESRCH, "the target process does not exist")
49        })?;
50        // Check permissions
51        check_rlimit_perm(&target_process, ctx)?;
52        do_prlimit64(&target_process, resource, new_raw, ctx)?
53    };
54
55    if old_rlim_addr != 0 {
56        user_space.write_val(old_rlim_addr, &old_raw)?;
57    }
58
59    Ok(SyscallReturn::Return(0))
60}
61
62fn do_prlimit64(
63    target_process: &Process,
64    resource: u32,
65    new_raw: Option<RawRLimit64>,
66    ctx: &Context,
67) -> Result<RawRLimit64> {
68    let resource = ResourceType::try_from(resource)?;
69    debug!(
70        "pid = {}, resource = {:?}, new_raw = {:?}",
71        target_process.pid(),
72        resource,
73        new_raw,
74    );
75
76    let rlimit = {
77        let resource_limits = target_process.resource_limits();
78        resource_limits.get_rlimit(resource)
79    };
80
81    let old_raw = if let Some(new_raw) = new_raw {
82        if resource == ResourceType::RLIMIT_NOFILE && new_raw.max > SYSCTL_NR_OPEN {
83            return_errno_with_message!(Errno::EPERM, "the new limit exceeds the system limit");
84        }
85        rlimit.set_raw_rlimit(new_raw, ctx)?
86    } else {
87        rlimit.get_raw_rlimit()
88    };
89
90    Ok(old_raw)
91}

93/// Checks whether the current process has permission to access
94/// the resource limits of the target process.
95// Reference: <https://man7.org/linux/man-pages/man2/prlimit.2.html>
96fn check_rlimit_perm(target_process: &Process, ctx: &Context) -> Result<()> {
97    if target_process
98        .user_ns()
99        .lock()
100        .check_cap(CapSet::SYS_RESOURCE, ctx.posix_thread)
101        .is_ok()
102    {
103        return Ok(());
104    }
105
106    let (current_ruid, current_rgid) = {
107        let current_cred = ctx.posix_thread.credentials();
108        (current_cred.ruid(), current_cred.rgid())
109    };
110
111    let target_cred = target_process
112        .main_thread()
113        .as_posix_thread()
114        .unwrap()
115        .credentials();
116
117    if current_ruid == target_cred.ruid()
118        && current_ruid == target_cred.euid()
119        && current_ruid == target_cred.suid()
120        && current_rgid == target_cred.rgid()
121        && current_rgid == target_cred.egid()
122        && current_rgid == target_cred.sgid()
123    {
124        return Ok(());
125    }
126
127    return_errno_with_message!(
128        Errno::EPERM,
129        "accessing the resource limits of the target process is not allowed"
130    )
131}
