8pub fn sys_mremap(
9    old_addr: Vaddr,
10    old_size: usize,
11    new_size: usize,
12    flags: i32,
13    new_addr: Vaddr,
14    ctx: &Context,
15) -> Result<SyscallReturn> {
16    let flags = MremapFlags::from_bits(flags)
17        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid flags"))?;
18    let new_addr = do_sys_mremap(old_addr, old_size, new_size, flags, new_addr, ctx)?;
19    Ok(SyscallReturn::Return(new_addr as _))
20}
21
22fn do_sys_mremap(
23    old_addr: Vaddr,
24    old_size: usize,
25    new_size: usize,
26    flags: MremapFlags,
27    new_addr: Vaddr,
28    ctx: &Context,
29) -> Result<Vaddr> {
30    debug!(
31        "mremap: old_addr = 0x{:x}, old_size = {}, new_size = {}, flags = {:?}, new_addr = 0x{:x}",
32        old_addr, old_size, new_size, flags, new_addr,
33    );
34
35    if !old_addr.is_multiple_of(PAGE_SIZE) {
36        return_errno_with_message!(Errno::EINVAL, "mremap: `old_addr` must be page-aligned");
37    }
38    if new_size == 0 {
39        return_errno_with_message!(Errno::EINVAL, "mremap: `new_size` cannot be zero");
40    }
41    if old_size == 0 {
42        return_errno_with_message!(
43            Errno::EINVAL,
44            "mremap: copying shareable mapping is not supported"
45        );
46    }
47
48    if old_size.checked_add(PAGE_SIZE).is_none() || new_size.checked_add(PAGE_SIZE).is_none() {
49        return_errno_with_message!(Errno::EINVAL, "mremap: the size overflows")
50    }
51    let old_size = old_size.align_up(PAGE_SIZE);
52    let new_size = new_size.align_up(PAGE_SIZE);
53
54    let user_space = ctx.user_space();
55    let vmar = user_space.vmar();
56
57    if !flags.contains(MremapFlags::MREMAP_FIXED) && new_size <= old_size {
58        // We can shrink a old range which spans multiple mappings. See
59        // <https://github.com/google/gvisor/blob/95d875276806484f974ce9e95556a561331f8e22/test/syscalls/linux/mremap.cc#L100-L117>.
60        vmar.resize_mapping(old_addr, old_size, new_size, false)?;
61        return Ok(old_addr);
62    }
63
64    if flags.contains(MremapFlags::MREMAP_MAYMOVE) {
65        if flags.contains(MremapFlags::MREMAP_FIXED) {
66            vmar.remap(old_addr, old_size, Some(new_addr), new_size)
67        } else {
68            vmar.remap(old_addr, old_size, None, new_size)
69        }
70    } else {
71        if flags.contains(MremapFlags::MREMAP_FIXED) {
72            return_errno_with_message!(
73                Errno::EINVAL,
74                "mremap: `MREMAP_FIXED` specified without also specifying `MREMAP_MAYMOVE`"
75            );
76        }
77        // We can ensure that `new_size > old_size` here. Since we are enlarging
78        // the old mapping, it is necessary to check whether the old range lies
79        // in a single mapping.
80        //
81        // FIXME: According to <https://man7.org/linux/man-pages/man2/mremap.2.html>,
82        // if the `MREMAP_MAYMOVE` flag is not set, and the mapping cannot
83        // be expanded at the current `Vaddr`, we should return an `ENOMEM`.
84        // However, `resize_mapping` returns a `EACCES` in this case.
85        vmar.resize_mapping(old_addr, old_size, new_size, true)?;
86        Ok(old_addr)
87    }
88}
89
90bitflags! {
91    struct MremapFlags: i32 {
92        const MREMAP_MAYMOVE = 1 << 0;
93        const MREMAP_FIXED = 1 << 1;
94        // TODO: Add support for this flag, which exists since Linux 5.7.
95        // const MREMAP_DONTUNMAP = 1 << 2;
96    }
97}
