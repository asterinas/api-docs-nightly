8pub fn sys_mremap(
9    old_addr: Vaddr,
10    old_size: usize,
11    new_size: usize,
12    flags: i32,
13    new_addr: Vaddr,
14    ctx: &Context,
15) -> Result<SyscallReturn> {
16    let flags = MremapFlags::from_bits(flags)
17        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid flags"))?;
18    let new_addr = do_sys_mremap(old_addr, old_size, new_size, flags, new_addr, ctx)?;
19    Ok(SyscallReturn::Return(new_addr as _))
20}
21
22fn do_sys_mremap(
23    old_addr: Vaddr,
24    old_size: usize,
25    new_size: usize,
26    flags: MremapFlags,
27    new_addr: Vaddr,
28    ctx: &Context,
29) -> Result<Vaddr> {
30    debug!(
31        "old_addr = 0x{:x}, old_size = {}, new_size = {}, flags = {:?}, new_addr = 0x{:x}",
32        old_addr, old_size, new_size, flags, new_addr,
33    );
34
35    if !old_addr.is_multiple_of(PAGE_SIZE) {
36        return_errno_with_message!(Errno::EINVAL, "old_addr must be page-aligned");
37    }
38    if new_size == 0 {
39        return_errno_with_message!(Errno::EINVAL, "new_size cannot be zero");
40    }
41    if old_size == 0 {
42        return_errno_with_message!(Errno::EINVAL, "copying shareable mapping is not supported");
43    }
44
45    if old_size.checked_add(PAGE_SIZE).is_none() || new_size.checked_add(PAGE_SIZE).is_none() {
46        return_errno_with_message!(Errno::EINVAL, "the size overflows")
47    }
48
49    // `MREMAP_DONTUNMAP` keeps the old VMA in the tree as an anonymous
50    // zero-fill-on-demand mapping; we must move pages even when the sizes
51    // are equal.  `MREMAP_MAYMOVE` is required because the mapping cannot
52    // stay at its current address (the physical pages must be relocated).
53    if flags.contains(MremapFlags::MREMAP_DONTUNMAP) {
54        if !flags.contains(MremapFlags::MREMAP_MAYMOVE) {
55            return_errno_with_message!(
56                Errno::EINVAL,
57                "MREMAP_DONTUNMAP must be combined with MREMAP_MAYMOVE"
58            );
59        }
60        if new_size != old_size {
61            return_errno_with_message!(
62                Errno::EINVAL,
63                "MREMAP_DONTUNMAP requires new_size to equal old_size"
64            );
65        }
66    }
67
68    // Align `old_size` and `new_size` after checking them.
69    let old_size = old_size.align_up(PAGE_SIZE);
70    let new_size = new_size.align_up(PAGE_SIZE);
71
72    let action = if flags.contains(MremapFlags::MREMAP_DONTUNMAP) {
73        RemapOldMappingAction::Keep
74    } else {
75        RemapOldMappingAction::Unmap
76    };
77
78    let user_space = ctx.user_space();
79    let vmar = user_space.vmar();
80
81    // When `MREMAP_DONTUNMAP` is set, we must move the mapping rather than
82    // shrinking in place, even though `new_size` == `old_size`.
83    if !flags.contains(MremapFlags::MREMAP_FIXED)
84        && new_size <= old_size
85        && action == RemapOldMappingAction::Unmap
86    {
87        // We can shrink a old range which spans multiple mappings. See
88        // <https://github.com/google/gvisor/blob/95d875276806484f974ce9e95556a561331f8e22/test/syscalls/linux/mremap.cc#L100-L117>.
89        vmar.resize_mapping(old_addr, old_size, new_size, false)?;
90        return Ok(old_addr);
91    }
92
93    if flags.contains(MremapFlags::MREMAP_MAYMOVE) {
94        if flags.contains(MremapFlags::MREMAP_FIXED) {
95            vmar.remap(old_addr, old_size, Some(new_addr), new_size, action)
96        } else {
97            vmar.remap(old_addr, old_size, None, new_size, action)
98        }
99    } else {
100        if flags.contains(MremapFlags::MREMAP_FIXED) {
101            return_errno_with_message!(
102                Errno::EINVAL,
103                "MREMAP_FIXED specified without also specifying MREMAP_MAYMOVE"
104            );
105        }
106        // We can ensure that `new_size > old_size` here. Since we are enlarging
107        // the old mapping, it is necessary to check whether the old range lies
108        // in a single mapping.
109        vmar.resize_mapping(old_addr, old_size, new_size, true)?;
110        Ok(old_addr)
111    }
112}
113
114bitflags! {
115    struct MremapFlags: i32 {
116        const MREMAP_MAYMOVE = 1 << 0;
117        const MREMAP_FIXED = 1 << 1;
118        const MREMAP_DONTUNMAP = 1 << 2;
119    }
