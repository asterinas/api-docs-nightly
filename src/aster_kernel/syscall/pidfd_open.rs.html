3use crate::{
4    fs::{file_table::FdFlags, utils::StatusFlags},
5    prelude::*,
6    process::{Pid, PidFile, process_table},
7    syscall::SyscallReturn,
8};
9
10pub fn sys_pidfd_open(pid: Pid, flags: u32, ctx: &Context) -> Result<SyscallReturn> {
11    let is_nonblocking = {
12        let flags = PidfdFlags::from_bits(flags)
13            .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid flags"))?;
14        debug!("pid = {}, flags = {:?}", pid, flags);
15        flags.contains(PidfdFlags::PIDFD_NONBLOCK)
16    };
17
18    if pid.cast_signed() < 0 {
19        return_errno_with_message!(Errno::EINVAL, "all negative PIDs are not valid");
20    }
21
22    let process = process_table::get_process(pid)
23        .ok_or_else(|| Error::with_message(Errno::ESRCH, "the process does not exist"))?;
24
25    let pid_fd = {
26        let pid_file = Arc::new(PidFile::new(process, is_nonblocking));
27        let file_table = ctx.thread_local.borrow_file_table();
28        let mut file_table_locked = file_table.unwrap().write();
29        // "the close-on-exec flag is set on the file descriptor."
30        // Reference: <https://man7.org/linux/man-pages/man2/pidfd_open.2.html>.
31        file_table_locked.insert(pid_file, FdFlags::CLOEXEC)
32    };
33
34    Ok(SyscallReturn::Return(pid_fd as _))
35}
36
37bitflags! {
38    struct PidfdFlags: u32 {
39        const PIDFD_NONBLOCK = StatusFlags::O_NONBLOCK.bits();
40    }
41}
