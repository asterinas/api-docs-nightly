3use super::SyscallReturn;
4use crate::{prelude::*, process::posix_thread::Personality};
5
6pub fn sys_personality(personality: u32, ctx: &Context) -> Result<SyscallReturn> {
7    // FIXME: Figure out how personality is inherited across `clone` or `execve` in Linux,
8    // and implement it properly.
9    let old_personality = ctx.posix_thread.personality() as isize;
10    if personality == GET_PERSONALITY {
11        return Ok(SyscallReturn::Return(old_personality));
12    }
13
14    if Personality::from_bits_truncate(personality).contains(Personality::ADDR_NO_RANDOMIZE) {
15        // FIXME: Figure out the Linux behavior when `ADDR_NO_RANDOMIZE` is set.
16        warn!("`personality(ADDR_NO_RANDOMIZE)` is accepted, but still does not disable ASLR");
17    }
18
19    // Linux accepts any value for `personality` except the query value.
20    // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/kernel/exec_domain.c#L38-L46>
21    ctx.posix_thread.set_personality(personality);
22    Ok(SyscallReturn::Return(old_personality))
23}
24
25const GET_PERSONALITY: u32 = 0xffffffff;
