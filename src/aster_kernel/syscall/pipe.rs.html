15pub fn sys_pipe2(fds: Vaddr, flags: u32, ctx: &Context) -> Result<SyscallReturn> {
16    debug!("flags: {:?}", flags);
17
18    let (pipe_reader, pipe_writer) = pipe::new_file_pair()?;
19
20    let fd_flags = if CreationFlags::from_bits_truncate(flags).contains(CreationFlags::O_CLOEXEC) {
21        FdFlags::CLOEXEC
22    } else {
23        FdFlags::empty()
24    };
25
26    let file_table = ctx.thread_local.borrow_file_table();
27    let mut file_table_locked = file_table.unwrap().write();
28
29    let pipe_fds = PipeFds {
30        reader_fd: file_table_locked.insert(pipe_reader, fd_flags),
31        writer_fd: file_table_locked.insert(pipe_writer, fd_flags),
32    };
33    debug!("pipe_fds: {:?}", pipe_fds);
34
35    if let Err(err) = ctx.user_space().write_val(fds, &pipe_fds) {
36        file_table_locked.close_file(pipe_fds.reader_fd).unwrap();
37        file_table_locked.close_file(pipe_fds.writer_fd).unwrap();
38        return Err(err.into());
39    }
40
41    Ok(SyscallReturn::Return(0))
42}
43
44pub fn sys_pipe(fds: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
45    self::sys_pipe2(fds, 0, ctx)
46}
47
48#[derive(Debug, Clone, Copy, Pod)]
49#[repr(C)]
50struct PipeFds {
51    reader_fd: FileDesc,
52    writer_fd: FileDesc,
53}
