3use ostd::{
4    arch::cpu::context::{FsBase, GsBase},
5    mm::{MAX_USERSPACE_VADDR, VmIo},
6};
7
8use super::SyscallReturn;
9use crate::prelude::*;
10
11#[expect(non_camel_case_types)]
12#[repr(u64)]
13#[derive(Debug, TryFromInt)]
14enum ArchPrctlCode {
15    ARCH_SET_GS = 0x1001,
16    ARCH_SET_FS = 0x1002,
17    ARCH_GET_FS = 0x1003,
18    ARCH_GET_GS = 0x1004,
19}
20
21pub fn sys_arch_prctl(code: u64, addr: usize, ctx: &Context) -> Result<SyscallReturn> {
22    let arch_prctl_code = ArchPrctlCode::try_from(code)?;
23    debug!(
24        "arch_prctl_code: {:?}, addr = 0x{:x}",
25        arch_prctl_code, addr
26    );
27    let res = do_arch_prctl(arch_prctl_code, addr, ctx)?;
28    Ok(SyscallReturn::Return(res as _))
29}
30
31fn do_arch_prctl(code: ArchPrctlCode, addr: usize, ctx: &Context) -> Result<u64> {
32    let supp = ctx.thread_local.supp_user_context();
33
34    match code {
35        ArchPrctlCode::ARCH_SET_GS => {
36            if addr >= MAX_USERSPACE_VADDR {
37                return_errno_with_message!(Errno::EPERM, "gsbase must be a userspace address");
38            }
39            supp.gs_base().set(GsBase::new(addr));
40            Ok(0)
41        }
42        ArchPrctlCode::ARCH_SET_FS => {
43            if addr >= MAX_USERSPACE_VADDR {
44                return_errno_with_message!(Errno::EPERM, "fsbase must be a userspace address");
45            }
46            supp.fs_base().set(FsBase::new(addr));
47            Ok(0)
48        }
49        ArchPrctlCode::ARCH_GET_FS => {
50            let base = supp.fs_base().get().addr();
51            ctx.user_space().write_val(addr, &base)?;
52            Ok(0)
53        }
54        ArchPrctlCode::ARCH_GET_GS => {
55            let base = supp.gs_base().get().addr();
56            ctx.user_space().write_val(addr, &base)?;
57            Ok(0)
58        }
59    }
60}
