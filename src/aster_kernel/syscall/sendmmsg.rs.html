7    net::socket::{Socket, util::SendRecvFlags},
8    prelude::*,
9    syscall::{SyscallReturn, sendmsg::send_one_message},
10    util::net::CUserMsgHdr,

12
13pub fn sys_sendmmsg(
14    sockfd: FileDesc,
15    mmsghdrs_addr: Vaddr,
16    count: usize,
17    flags: i32,
18    ctx: &Context,
19) -> Result<SyscallReturn> {
20    let flags = SendRecvFlags::from_bits(flags)
21        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid send recv flags"))?;
22
23    debug!(
24        "sockfd = {}, mmsghdrs = {:#x}, count = {}, flags = {:?}",
25        sockfd, mmsghdrs_addr, count, flags
26    );
27
28    if !flags.is_empty() {
29        warn!("sendmmsg flags {:?} are not supported", flags);
30    }
31
32    let file = {
33        // Reading control messages may access the file table,
34        // so we have to clone the file and drop the file table reference here.
35        let file_table = ctx.thread_local.borrow_file_table();
36        let file_table_locked = file_table.unwrap().read();
37        file_table_locked.get_file(sockfd)?.clone()
38    };
39    let socket = file.as_socket_or_err()?;
40
41    let mut sent_msgs = 0;
42    match send_mmsg_hdrs(socket, mmsghdrs_addr, count, flags, &mut sent_msgs, ctx) {
43        // Only return error if no packets are sent successfully.
44        Err(e) if sent_msgs == 0 => Err(e),
45        _ => Ok(SyscallReturn::Return(sent_msgs as _)),
46    }
47}
48
49#[repr(C)]
50#[padding_struct]
51#[derive(Debug, Clone, Copy, Pod)]
52struct CMmsgHdr {
53    msg_hdr: CUserMsgHdr,
54    msg_len: u32,
55}
56
57fn send_mmsg_hdrs(
58    socket: &dyn Socket,
59    mmsghdrs_addr: Vaddr,
60    count: usize,
61    flags: SendRecvFlags,
62    sent_msgs: &mut usize,
63    ctx: &Context,
64) -> Result<()> {
65    let user_space = ctx.user_space();
66
67    for i in 0..count {
68        let addr = mmsghdrs_addr + size_of::<CMmsgHdr>() * i;
69        let mut mmsghdr = user_space.read_val::<CMmsgHdr>(addr)?;
70
71        let sent_bytes = send_one_message(socket, &mmsghdr.msg_hdr, &user_space, flags)?;
72
73        mmsghdr.msg_len = sent_bytes as u32;
74        user_space.write_val(addr, &mmsghdr)?;
75
76        *sent_msgs += 1;
77    }
78
79    Ok(())
80}
