3use super::SyscallReturn;
4use crate::{
5    fs::{
6        file_table::FdFlags,
7        inode_handle::InodeHandle,
8        ramfs::memfd::{MAX_MEMFD_NAME_LEN, MemfdFlags, MemfdInodeHandle},
9    },
10    prelude::*,
11};
12
13pub fn sys_memfd_create(name_addr: Vaddr, flags: u32, ctx: &Context) -> Result<SyscallReturn> {
14    // FIXME: When `name` is too long, `read_cstring` returns `EFAULT`. However,
15    // according to <https://man7.org/linux/man-pages/man2/memfd_create.2.html>,
16    // we should return `EINVAL` in this case.
17    let name = ctx
18        .user_space()
19        .read_cstring(name_addr, MAX_MEMFD_NAME_LEN + 1)?;
20    debug!("sys_memfd_create: name = {:?}, flags = {}", name, flags);
21
22    let fd = {
23        let memfd_flags = MemfdFlags::from_bits(flags).ok_or(Errno::EINVAL)?;
24        let fd_flags = if memfd_flags.contains(MemfdFlags::MFD_CLOEXEC) {
25            FdFlags::CLOEXEC
26        } else {
27            FdFlags::empty()
28        };
29        let file_table = ctx.thread_local.borrow_file_table();
30        let mut file_table_locked = file_table.unwrap().write();
31
32        // FIXME: Support `MFD_HUGETLB`.
33        if memfd_flags.contains(MemfdFlags::MFD_HUGETLB) {
34            warn!("`MFD_HUGETLB` not supported");
35        }
36
37        if memfd_flags.contains(MemfdFlags::MFD_NOEXEC_SEAL | MemfdFlags::MFD_EXEC) {
38            return_errno_with_message!(
39                Errno::EINVAL,
40                "`MFD_EXEC` and `MFD_NOEXEC_SEAL` cannot be specified together"
41            );
42        }
43
44        let memfd_file = InodeHandle::new_memfd(name.to_string_lossy().into_owned(), memfd_flags)?;
45
46        file_table_locked.insert(Arc::new(memfd_file), fd_flags)
47    };
48
49    Ok(SyscallReturn::Return(fd as _))
50}
