20pub fn sys_rt_sigtimedwait(
21    set_ptr: Vaddr,
22    info_ptr: Vaddr,
23    timeout_ptr: Vaddr,
24    sigset_size: usize,
25    ctx: &Context,
26) -> Result<SyscallReturn> {
27    debug!(
28        "set_ptr = {:#?}, info_ptr = {:#?}, timeout_ptr = {:#?}, sigset_size = {}",
29        set_ptr, info_ptr, timeout_ptr, sigset_size
30    );
31
32    // Validate sigset size
33    if sigset_size != size_of::<SigMask>() {
34        return_errno_with_message!(Errno::EINVAL, "invalid sigset size");
35    }
36
37    // Read the signal set
38    let mask = {
39        let mut set: SigSet = ctx.user_space().read_val(set_ptr)?;
40
41        // Remove SIGKILL and SIGSTOP as they cannot be waited for
42        set -= SIGKILL;
43        set -= SIGSTOP;
44
45        !set
46    };
47
48    // Read timeout if provided
49    let timeout = if timeout_ptr != 0 {
50        let timespec: timespec_t = ctx.user_space().read_val(timeout_ptr)?;
51        Some(Duration::try_from(timespec)?)
52    } else {
53        None
54    };
55
56    debug!(
57        "pid = {}, sig_mask = {:?}, timeout = {:?}",
58        ctx.process.pid(),
59        mask,
60        timeout
61    );
62
63    let block_list = ctx.posix_thread.sig_mask().load(Ordering::Relaxed);
64    // Fast path: If a signal is already pending, dequeue and return it immediately.
65    if let Some(signal) = dequeue_signal_with_checking_ignore(ctx, mask, block_list) {
66        if info_ptr != 0 {
67            let siginfo = signal.to_info();
68            ctx.user_space().write_val(info_ptr, &siginfo)?;
69        }
70
71        return Ok(SyscallReturn::Return(signal.num().as_u8() as _));
72    }
73
74    with_sigmask_changed(
75        ctx,
76        |sig_mask| sig_mask & mask,
77        || {
78            // Wait for a signal to arrive or timeout.
79            let waiter = Waiter::new_pair().0;
80            let signal = waiter
81                .pause_until_or_timeout(
82                    || dequeue_signal_with_checking_ignore(ctx, mask, block_list),
83                    timeout.map(ManagedTimeout::new),
84                )
85                .map_err(|e| {
86                    if e.error() == Errno::ETIME {
87                        Error::new(Errno::EAGAIN)
88                    } else {
89                        e
90                    }
91                })?;
92
93            if info_ptr != 0 {
94                let siginfo = signal.to_info();
95                ctx.user_space().write_val(info_ptr, &siginfo)?;
96            }
97
98            Ok(SyscallReturn::Return(signal.num().as_u8() as _))
99        },
100    )
101}

103/// Dequeue a signal from the thread's pending signal queue.
104///
105/// If the signal is ignored and not blocked, it will be dropped and
106/// the next signal will be checked.
107fn dequeue_signal_with_checking_ignore(
108    ctx: &Context,
109    mask: SigMask,
110    block_list: SigMask,
111) -> Option<Box<dyn Signal>> {
112    let sig_dispositions = ctx.process.sig_dispositions().lock();
113    let sig_disposition = sig_dispositions.lock();
114    while let Some(signal) = ctx.posix_thread.dequeue_signal(&mask) {
115        // If the signal is ignored and not blocked, the signal will be directly dropped.
116        if !block_list.contains(signal.num()) && sig_disposition.will_ignore(signal.as_ref()) {
117            continue;
118        }
119
120        return Some(signal);
121    }
122
123    None
124}
