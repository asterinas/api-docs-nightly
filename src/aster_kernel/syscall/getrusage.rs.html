3use int_to_c_enum::TryFromInt;
4use ostd::mm::VmIo;

9#[derive(Debug, Copy, Clone, TryFromInt, PartialEq)]
10#[repr(i32)]
11enum RusageTarget {
12    ForSelf = 0,
13    Children = -1,
14    Thread = 1,
15}
16
17pub fn sys_getrusage(target: i32, rusage_addr: Vaddr, ctx: &Context) -> Result<SyscallReturn> {
18    let rusage_target = RusageTarget::try_from(target)?;
19
20    debug!(
21        "target = {:?}, rusage_addr = {}",
22        rusage_target, rusage_addr,
23    );
24
25    if rusage_addr != 0 {
26        let rusage = match rusage_target {
27            RusageTarget::ForSelf => {
28                let process = ctx.process.as_ref();
29                rusage_t {
30                    ru_utime: process.prof_clock().user_clock().read_time().into(),
31                    ru_stime: process.prof_clock().kernel_clock().read_time().into(),
32                    ..Default::default()
33                }
34            }
35            RusageTarget::Thread => {
36                let posix_thread = ctx.posix_thread;
37                rusage_t {
38                    ru_utime: posix_thread.prof_clock().user_clock().read_time().into(),
39                    ru_stime: posix_thread.prof_clock().kernel_clock().read_time().into(),
40                    ..Default::default()
41                }
42            }
43            RusageTarget::Children => {
44                let (user_time, kernel_time) = ctx.process.reaped_children_stats().lock().get();
45                rusage_t {
46                    ru_utime: user_time.into(),
47                    ru_stime: kernel_time.into(),
48                    ..Default::default()
49                }
50            }
51        };
52
53        ctx.user_space().write_val(rusage_addr, &rusage)?;
54    }
55
56    Ok(SyscallReturn::Return(0))
57}

59#[repr(C)]
60#[derive(Debug, Default, Copy, Clone, Pod)]
61pub struct rusage_t {
62    /// user time used
63    pub ru_utime: timeval_t,
64    /// system time used
65    pub ru_stime: timeval_t,
66    /// maximum resident set size
67    pub ru_maxrss: u64,
68    /// integral shared memory size
69    pub ru_ixrss: u64,
70    /// integral unshared data size
71    pub ru_idrss: u64,
72    /// integral unshared stack size
73    pub ru_isrss: u64,
74    /// page reclaims
75    pub ru_minflt: u64,
76    /// page faults
77    pub ru_majflt: u64,
78    /// swaps
79    pub ru_nswap: u64,
80    /// block input operations
81    pub ru_inblock: u64,
82    /// block output operations
83    pub ru_oublock: u64,
84    /// messages sent
85    pub ru_msgsnd: u64,
86    /// messages received
87    pub ru_msgrcv: u64,
88    /// signals received
89    pub ru_nsignals: u64,
90    /// voluntary ctx switches
91    pub ru_nvcsw: u64,
92    /// involuntary
93    pub ru_nivcsw: u64,
94}
