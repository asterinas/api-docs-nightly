13bitflags! {
14    struct CloseRangeFlags: u32 {
15        const UNSHARE = 1 << 1;
16        const CLOEXEC = 1 << 2;
17    }
18}
19
20pub fn sys_close(fd: FileDesc, ctx: &Context) -> Result<SyscallReturn> {
21    debug!("fd = {}", fd);
22
23    let file = {
24        let file_table = ctx.thread_local.borrow_file_table();
25        let mut file_table_locked = file_table.unwrap().write();
26        let _ = file_table_locked.get_file(fd)?;
27        file_table_locked.close_file(fd).unwrap()
28    };
29
30    fs::notify::on_close(&file);
31
32    // Cleanup work needs to be done in the `Drop` impl.
33    //
34    // We don't mind the races between closing the file descriptor and using the file descriptor,
35    // because such races are explicitly allowed in the man pages. See the "Multithreaded processes
36    // and close()" section in <https://man7.org/linux/man-pages/man2/close.2.html>.
37    drop(file);
38
39    // Linux has error codes for the close() system call for diagnostic and remedial purposes, but
40    // only for a small subset of file systems such as NFS. We currently have no support for such
41    // file systems, so it's fine to just return zero.
42    //
43    // For details, see the discussion at <https://github.com/asterinas/asterinas/issues/506> and
44    // the "Dealing with error returns from close()" section at
45    // <https://man7.org/linux/man-pages/man2/close.2.html>.
46    Ok(SyscallReturn::Return(0))
47}

49pub fn sys_close_range(
50    first: u32,
51    last: u32,
52    raw_flags: u32,
53    ctx: &Context,
54) -> Result<SyscallReturn> {
55    debug!("first = {}, last = {}, flags = {}", first, last, raw_flags);
56
57    if last < first {
58        return_errno!(Errno::EINVAL);
59    }
60
61    let flags = CloseRangeFlags::from_bits(raw_flags).ok_or_else(|| Error::new(Errno::EINVAL))?;
62
63    if flags.contains(CloseRangeFlags::UNSHARE) {
64        // FIXME: While directly invoking `unshare_files` is logically correct,
65        // it might not be the most efficient approach.
66        // `unshare_files` clones the entire file table by duplicating all its entries.
67        // However, in the context of `close_range`,
68        // cloning files that are about to be closed is unnecessary overhead.
69        ctx.unshare_files();
70    }
71
72    let file_table = ctx.thread_local.borrow_file_table();
73
74    let mut files_to_drop = Vec::new();
75
76    {
77        let mut file_table_locked = file_table.unwrap().write();
78
79        let table_len = file_table_locked.len() as u32;
80        if first >= table_len {
81            return Ok(SyscallReturn::Return(0));
82        }
83        let actual_last = last.min(table_len - 1);
84
85        for fd in first..=actual_last {
86            let fd = fd as FileDesc;
87
88            if flags.contains(CloseRangeFlags::CLOEXEC) {
89                if let Ok(entry) = file_table_locked.get_entry_mut(fd) {
90                    entry.set_flags(entry.flags() | FdFlags::CLOEXEC);
91                }
92            } else if let Some(file) = file_table_locked.close_file(fd) {
93                files_to_drop.push(file);
94            }
95        }
96    }
97
98    drop(files_to_drop);
99
100    Ok(SyscallReturn::Return(0))
101}
