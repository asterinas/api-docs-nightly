12pub fn sys_get_priority(which: i32, who: u32, ctx: &Context) -> Result<SyscallReturn> {
13    let prio_target = PriorityTarget::new(which, who, ctx)?;
14    debug!("get_priority prio_target: {:?}", prio_target);
15
16    let processes = get_processes(prio_target)?;
17    let highest_prio = {
18        let mut nice = Nice::MAX.value().get();
19        for process in processes.iter() {
20            let proc_nice = process.nice().load(Ordering::Relaxed).value().get();
21            // Returns the highest priority enjoyed by the processes
22            if proc_nice < nice {
23                nice = proc_nice;
24            }
25        }
26
27        // The system call returns nice values translated to the range 40 to 1,
28        // since a negative return value would be interpreted as an error.
29        20 - nice
30    };
31
32    Ok(SyscallReturn::Return(highest_prio as _))
33}

35pub(super) fn get_processes(prio_target: PriorityTarget) -> Result<Vec<Arc<Process>>> {
36    Ok(match prio_target {
37        PriorityTarget::Process(pid) => {
38            let process = process_table::get_process(pid).ok_or(Error::new(Errno::ESRCH))?;
39            vec![process]
40        }
41        PriorityTarget::ProcessGroup(pgid) => {
42            let process_group =
43                process_table::get_process_group(&pgid).ok_or(Error::new(Errno::ESRCH))?;
44            let processes: Vec<Arc<Process>> = process_group.lock().iter().cloned().collect();
45            if processes.is_empty() {
46                return_errno!(Errno::ESRCH);
47            }
48            processes
49        }
50        PriorityTarget::User(uid) => {
51            // Get the processes that are running under the specified user
52            let processes: Vec<Arc<Process>> = process_table::process_table_mut()
53                .iter()
54                .filter(|process| {
55                    let main_thread = process.main_thread();
56                    let posix_thread = main_thread.as_posix_thread().unwrap();
57                    uid == posix_thread.credentials().ruid()
58                })
59                .cloned()
60                .collect();
61            if processes.is_empty() {
62                return_errno!(Errno::ESRCH);
63            }
64            processes
65        }
66    })
67}

69#[derive(Debug)]
70pub(super) enum PriorityTarget {
71    Process(Pid),
72    ProcessGroup(Pgid),
73    User(Uid),
74}
75
76impl PriorityTarget {
77    pub(super) fn new(which: i32, who: u32, ctx: &Context) -> Result<Self> {
78        let which = Which::try_from(which)
79            .map_err(|_| Error::with_message(Errno::EINVAL, "invalid which value"))?;
80        Ok(match which {
81            Which::PRIO_PROCESS => {
82                let pid = if who == 0 {
83                    ctx.process.pid()
84                } else {
85                    who as Pid
86                };
87                Self::Process(pid)
88            }
89            Which::PRIO_PGRP => {
90                let pgid = if who == 0 {
91                    ctx.process.pgid()
92                } else {
93                    who as Pgid
94                };
95                Self::ProcessGroup(pgid)
96            }
97            Which::PRIO_USER => {
98                let uid = if who == 0 {
99                    ctx.posix_thread.credentials().ruid()
100                } else {
101                    Uid::new(who)
102                };
103                Self::User(uid)
104            }
105        })
106    }
107}

109#[expect(non_camel_case_types)]
110#[derive(Clone, Debug, TryFromInt)]
111#[repr(i32)]
112pub(super) enum Which {
113    PRIO_PROCESS = 0,
114    PRIO_PGRP = 1,
115    PRIO_USER = 2,
