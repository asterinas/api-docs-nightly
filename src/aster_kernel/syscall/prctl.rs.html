13pub fn sys_prctl(
14    option: i32,
15    arg2: u64,
16    arg3: u64,
17    arg4: u64,
18    arg5: u64,
19    ctx: &Context,
20) -> Result<SyscallReturn> {
21    let prctl_cmd = PrctlCmd::from_args(option, arg2, arg3, arg4, arg5)?;
22    debug!("prctl cmd = {:x?}", prctl_cmd);
23    match prctl_cmd {
24        PrctlCmd::PR_SET_PDEATHSIG(signum) => {
25            ctx.process.set_parent_death_signal(signum);
26        }
27        PrctlCmd::PR_GET_PDEATHSIG(write_to_addr) => {
28            let write_val = {
29                match ctx.process.parent_death_signal() {
30                    None => 0i32,
31                    Some(signum) => signum.as_u8() as i32,
32                }
33            };
34
35            ctx.user_space().write_val(write_to_addr, &write_val)?;
36        }
37        PrctlCmd::PR_GET_DUMPABLE => {
38            // TODO: when coredump is supported, return the actual value
39            return Ok(SyscallReturn::Return(Dumpable::Disable as _));
40        }
41        PrctlCmd::PR_SET_DUMPABLE(dumpable) => {
42            if dumpable != Dumpable::Disable && dumpable != Dumpable::User {
43                return_errno!(Errno::EINVAL)
44            }
45
46            // TODO: implement coredump
47        }
48        PrctlCmd::PR_GET_KEEPCAPS => {
49            let keep_cap = {
50                let credentials = ctx.posix_thread.credentials();
51                if credentials.keep_capabilities() {
52                    1
53                } else {
54                    0
55                }
56            };
57
58            return Ok(SyscallReturn::Return(keep_cap as _));
59        }
60        PrctlCmd::PR_SET_KEEPCAPS(keep_cap) => {
61            if keep_cap > 1 {
62                return_errno!(Errno::EINVAL)
63            }
64            let credentials = ctx.posix_thread.credentials_mut();
65            credentials.set_keep_capabilities(keep_cap != 0)?;
66        }
67        PrctlCmd::PR_GET_NAME(write_to_addr) => {
68            let thread_name = ctx.posix_thread.thread_name().lock();
69            ctx.user_space()
70                .write_bytes(write_to_addr, thread_name.name().to_bytes_with_nul())?;
71        }
72        PrctlCmd::PR_SET_NAME(read_addr) => {
73            let new_thread_name = ctx
74                .user_space()
75                .read_cstring(read_addr, MAX_THREAD_NAME_LEN)?;
76            let mut thread_name = ctx.posix_thread.thread_name().lock();
77            thread_name.set_name(&new_thread_name);
78        }
79        PrctlCmd::PR_SET_CHILD_SUBREAPER(is_set) => {
80            let process = ctx.process.as_ref();
81            if is_set {
82                process.set_child_subreaper();
83            } else {
84                process.unset_child_subreaper();
85            }
86        }
87        PrctlCmd::PR_GET_CHILD_SUBREAPER(write_addr) => {
88            let process = ctx.process.as_ref();
89            ctx.user_space()
90                .write_val(write_addr, &(process.is_child_subreaper() as u32))?;
91        }
92        PrctlCmd::PR_GET_SECUREBITS => {
93            let credentials = ctx.posix_thread.credentials();
94            let securebits = credentials.securebits();
95            return Ok(SyscallReturn::Return(securebits.bits() as _));
96        }
97        PrctlCmd::PR_SET_SECUREBITS(securebits) => {
98            let credentials = ctx.posix_thread.credentials_mut();
99            credentials.set_securebits(securebits)?;
100        }
101        PrctlCmd::PR_GET_TIMERSLACK => {
102            let slack_ns = ctx.posix_thread.timer_slack_ns();
103            return Ok(SyscallReturn::Return(slack_ns as _));
104        }
105        PrctlCmd::PR_SET_TIMERSLACK(slack_ns) => {
106            // Negative values are invalid
107            if (slack_ns as i64) < 0 {
108                return_errno!(Errno::EINVAL);
109            }
110
111            // In Linux, a value of 0 means "use default slack"
112            if slack_ns == 0 {
113                ctx.posix_thread.reset_timer_slack_to_default();
114            } else {
115                ctx.posix_thread.set_timer_slack_ns(slack_ns);
116            }
117        }
118    }
119    Ok(SyscallReturn::Return(0))
120}

122const PR_SET_PDEATHSIG: i32 = 1;
123const PR_GET_PDEATHSIG: i32 = 2;
124const PR_GET_DUMPABLE: i32 = 3;
125const PR_SET_DUMPABLE: i32 = 4;
126const PR_GET_KEEPCAPS: i32 = 7;
127const PR_SET_KEEPCAPS: i32 = 8;
128const PR_SET_NAME: i32 = 15;
129const PR_GET_NAME: i32 = 16;
130const PR_GET_SECUREBITS: i32 = 27;
131const PR_SET_SECUREBITS: i32 = 28;
132const PR_SET_TIMERSLACK: i32 = 29;
133const PR_GET_TIMERSLACK: i32 = 30;
134const PR_SET_CHILD_SUBREAPER: i32 = 36;
135const PR_GET_CHILD_SUBREAPER: i32 = 37;
136
137#[expect(non_camel_case_types)]
138#[derive(Debug, Clone, Copy)]
139pub enum PrctlCmd {
140    PR_SET_PDEATHSIG(SigNum),
141    PR_GET_PDEATHSIG(Vaddr),
142    PR_SET_NAME(Vaddr),
143    PR_GET_NAME(Vaddr),
144    PR_GET_KEEPCAPS,
145    PR_SET_KEEPCAPS(u32),
146    PR_SET_TIMERSLACK(u64),
147    PR_GET_TIMERSLACK,
148    PR_SET_DUMPABLE(Dumpable),
149    PR_GET_DUMPABLE,
150    PR_SET_CHILD_SUBREAPER(bool),
151    PR_GET_CHILD_SUBREAPER(Vaddr),
152    PR_GET_SECUREBITS,
153    PR_SET_SECUREBITS(SecureBits),
154}
155
156#[repr(u64)]
157#[derive(Debug, Clone, Copy, PartialEq, Eq, TryFromInt)]
158pub enum Dumpable {
159    Disable = 0, /* No setuid dumping */
160    User = 1,    /* Dump as user of process */
161    Root = 2,    /* Dump as root */
162}
163
164impl PrctlCmd {
165    fn from_args(option: i32, arg2: u64, _arg3: u64, _arg4: u64, _arg5: u64) -> Result<PrctlCmd> {
166        match option {
167            PR_SET_PDEATHSIG => {
168                let signum = SigNum::try_from(arg2 as u8)?;
169                Ok(PrctlCmd::PR_SET_PDEATHSIG(signum))
170            }
171            PR_GET_PDEATHSIG => Ok(PrctlCmd::PR_GET_PDEATHSIG(arg2 as _)),
172            PR_GET_DUMPABLE => Ok(PrctlCmd::PR_GET_DUMPABLE),
173            PR_SET_DUMPABLE => Ok(PrctlCmd::PR_SET_DUMPABLE(Dumpable::try_from(arg2)?)),
174            PR_SET_NAME => Ok(PrctlCmd::PR_SET_NAME(arg2 as _)),
175            PR_GET_NAME => Ok(PrctlCmd::PR_GET_NAME(arg2 as _)),
176            PR_GET_TIMERSLACK => Ok(PrctlCmd::PR_GET_TIMERSLACK),
177            PR_SET_TIMERSLACK => Ok(PrctlCmd::PR_SET_TIMERSLACK(arg2)),
178            PR_GET_KEEPCAPS => Ok(PrctlCmd::PR_GET_KEEPCAPS),
179            PR_SET_KEEPCAPS => Ok(PrctlCmd::PR_SET_KEEPCAPS(arg2 as _)),
180            PR_SET_CHILD_SUBREAPER => Ok(PrctlCmd::PR_SET_CHILD_SUBREAPER(arg2 > 0)),
181            PR_GET_CHILD_SUBREAPER => Ok(PrctlCmd::PR_GET_CHILD_SUBREAPER(arg2 as _)),
182            PR_GET_SECUREBITS => Ok(PrctlCmd::PR_GET_SECUREBITS),
183            PR_SET_SECUREBITS => Ok(PrctlCmd::PR_SET_SECUREBITS(SecureBits::try_from(
184                arg2 as u16,
185            )?)),
186            _ => {
187                debug!("prctl cmd number: {}", option);
188                return_errno_with_message!(Errno::EINVAL, "unsupported prctl command");
189            }
190        }
191    }
