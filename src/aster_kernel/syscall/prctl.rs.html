15pub fn sys_prctl(
16    option: i32,
17    arg2: u64,
18    arg3: u64,
19    arg4: u64,
20    arg5: u64,
21    ctx: &Context,
22) -> Result<SyscallReturn> {
23    let prctl_cmd = PrctlCmd::from_args(option, arg2, arg3, arg4, arg5)?;
24    debug!("prctl_cmd = {:x?}", prctl_cmd);
25
26    match prctl_cmd {
27        PrctlCmd::PR_SET_PDEATHSIG(signum) => {
28            ctx.process.set_parent_death_signal(signum);
29        }
30        PrctlCmd::PR_GET_PDEATHSIG(write_to_addr) => {
31            let write_val = match ctx.process.parent_death_signal() {
32                None => 0i32,
33                Some(signum) => signum.as_u8() as i32,
34            };
35            ctx.user_space().write_val(write_to_addr, &write_val)?;
36        }
37        PrctlCmd::PR_GET_DUMPABLE => {
38            // TODO: When coredump is supported, return the actual value.
39            return Ok(SyscallReturn::Return(Dumpable::Disable as _));
40        }
41        PrctlCmd::PR_SET_DUMPABLE(dumpable) => {
42            if dumpable != Dumpable::Disable && dumpable != Dumpable::User {
43                return_errno_with_message!(Errno::EINVAL, "invalid dumpable attribute");
44            }
45            // TODO: Implement coredump.
46        }
47        PrctlCmd::PR_GET_KEEPCAPS => {
48            let keep_cap = {
49                let credentials = ctx.posix_thread.credentials();
50                if credentials.keep_capabilities() {
51                    1
52                } else {
53                    0
54                }
55            };
56            return Ok(SyscallReturn::Return(keep_cap as _));
57        }
58        PrctlCmd::PR_SET_KEEPCAPS(keep_cap) => {
59            if keep_cap > 1 {
60                return_errno_with_message!(Errno::EINVAL, "invalid keep-capabilities flag");
61            }
62            let credentials = ctx.credentials_mut();
63            credentials.set_keep_capabilities(keep_cap != 0)?;
64        }
65        PrctlCmd::PR_SET_NAME(read_addr) => {
66            let new_thread_name = ctx
67                .user_space()
68                .read_cstring(read_addr, MAX_THREAD_NAME_LEN)?;
69            let mut thread_name = ctx.posix_thread.thread_name().lock();
70            thread_name.set_name(&new_thread_name);
71        }
72        PrctlCmd::PR_GET_NAME(write_to_addr) => {
73            let thread_name = ctx.posix_thread.thread_name().lock();
74            ctx.user_space()
75                .write_bytes(write_to_addr, thread_name.name().to_bytes_with_nul())?;
76        }
77        PrctlCmd::PR_CAPBSET_READ(capability) => {
78            let credentials = ctx.posix_thread.credentials();
79            let is_in_bounding_set = credentials.bounding_capset().contains(capability);
80            return Ok(SyscallReturn::Return(is_in_bounding_set as _));
81        }
82        PrctlCmd::PR_CAPBSET_DROP(capability) => {
83            let credentials = ctx.credentials_mut();
84            credentials.drop_bounding_capability(capability)?;
85        }
86        PrctlCmd::PR_GET_SECUREBITS => {
87            let credentials = ctx.posix_thread.credentials();
88            let securebits = credentials.securebits();
89            return Ok(SyscallReturn::Return(securebits.bits() as _));
90        }
91        PrctlCmd::PR_SET_SECUREBITS(securebits) => {
92            let credentials = ctx.credentials_mut();
93            credentials.set_securebits(securebits)?;
94        }
95        PrctlCmd::PR_SET_TIMERSLACK(slack_ns) => {
96            // Negative values are invalid.
97            if (slack_ns as i64) < 0 {
98                return_errno_with_message!(Errno::EINVAL, "invalid timer slack");
99            }
100            // In Linux, a value of 0 means "use default slack".
101            if slack_ns == 0 {
102                ctx.posix_thread.reset_timer_slack_to_default();
103            } else {
104                ctx.posix_thread.set_timer_slack_ns(slack_ns);
105            }
106        }
107        PrctlCmd::PR_GET_TIMERSLACK => {
108            let slack_ns = ctx.posix_thread.timer_slack_ns();
109            return Ok(SyscallReturn::Return(slack_ns as _));
110        }
111        PrctlCmd::PR_SET_CHILD_SUBREAPER(is_set) => {
112            let process = ctx.process.as_ref();
113            if is_set {
114                process.set_child_subreaper();
115            } else {
116                process.unset_child_subreaper();
117            }
118        }
119        PrctlCmd::PR_GET_CHILD_SUBREAPER(write_addr) => {
120            let process = ctx.process.as_ref();
121            ctx.user_space()
122                .write_val(write_addr, &(process.is_child_subreaper() as u32))?;
123        }
124        PrctlCmd::PR_SET_NO_NEW_PRIVS => {
125            let credentials = ctx.credentials_mut();
126            credentials.set_no_new_privs();
127        }
128        PrctlCmd::PR_GET_NO_NEW_PRIVS => {
129            let credentials = ctx.posix_thread.credentials();
130            return Ok(SyscallReturn::Return(credentials.no_new_privs() as _));
131        }
132        PrctlCmd::PR_CAP_AMBIENT(cmd) => match cmd {
133            CapAmbientCmd::IsSet(capability) => {
134                let credentials = ctx.posix_thread.credentials();
135                let is_set = credentials.ambient_capset().contains(capability);
136                return Ok(SyscallReturn::Return(if is_set { 1 } else { 0 }));
137            }
138            CapAmbientCmd::Raise(capability) => {
139                let credentials = ctx.credentials_mut();
140                credentials.raise_ambient_capability(capability)?;
141            }
142            CapAmbientCmd::Lower(capability) => {
143                let credentials = ctx.credentials_mut();
144                credentials.lower_ambient_capability(capability);
145            }
146            CapAmbientCmd::ClearAll => {
147                let credentials = ctx.credentials_mut();
148                credentials.clear_ambient_capset();
149            }
150        },
151    }
152
153    Ok(SyscallReturn::Return(0))
154}

156const PR_SET_PDEATHSIG: i32 = 1;
157const PR_GET_PDEATHSIG: i32 = 2;
158const PR_GET_DUMPABLE: i32 = 3;
159const PR_SET_DUMPABLE: i32 = 4;
160const PR_GET_KEEPCAPS: i32 = 7;
161const PR_SET_KEEPCAPS: i32 = 8;
162const PR_SET_NAME: i32 = 15;
163const PR_GET_NAME: i32 = 16;
164const PR_CAPBSET_READ: i32 = 23;
165const PR_CAPBSET_DROP: i32 = 24;
166const PR_GET_SECUREBITS: i32 = 27;
167const PR_SET_SECUREBITS: i32 = 28;
168const PR_SET_TIMERSLACK: i32 = 29;
169const PR_GET_TIMERSLACK: i32 = 30;
170const PR_SET_CHILD_SUBREAPER: i32 = 36;
171const PR_GET_CHILD_SUBREAPER: i32 = 37;
172const PR_SET_NO_NEW_PRIVS: i32 = 38;
173const PR_GET_NO_NEW_PRIVS: i32 = 39;
174const PR_CAP_AMBIENT: i32 = 47;

176#[expect(non_camel_case_types)]
177#[derive(Clone, Copy, Debug)]
178enum PrctlCmd {
179    PR_SET_PDEATHSIG(SigNum),
180    PR_GET_PDEATHSIG(Vaddr),
181    PR_GET_DUMPABLE,
182    PR_SET_DUMPABLE(Dumpable),
183    PR_GET_KEEPCAPS,
184    PR_SET_KEEPCAPS(u32),
185    PR_SET_NAME(Vaddr),
186    PR_GET_NAME(Vaddr),
187    PR_CAPBSET_READ(CapSet),
188    PR_CAPBSET_DROP(CapSet),
189    PR_GET_SECUREBITS,
190    PR_SET_SECUREBITS(SecureBits),
191    PR_SET_TIMERSLACK(u64),
192    PR_GET_TIMERSLACK,
193    PR_SET_CHILD_SUBREAPER(bool),
194    PR_GET_CHILD_SUBREAPER(Vaddr),
195    PR_SET_NO_NEW_PRIVS,
196    PR_GET_NO_NEW_PRIVS,
197    PR_CAP_AMBIENT(CapAmbientCmd),
198}
199
200#[repr(u64)]
201#[derive(Clone, Copy, Debug, Eq, PartialEq, TryFromInt)]
202enum Dumpable {
203    Disable = 0, /* No setuid dumping */
204    User = 1,    /* Dump as user of process */
205    Root = 2,    /* Dump as root */
206}
207
208#[derive(Clone, Copy, Debug)]
209enum CapAmbientCmd {
210    IsSet(CapSet),
211    Raise(CapSet),
212    Lower(CapSet),
213    ClearAll,
214}
215
216impl PrctlCmd {
217    fn from_args(option: i32, arg2: u64, arg3: u64, arg4: u64, arg5: u64) -> Result<PrctlCmd> {
218        match option {
219            PR_SET_PDEATHSIG => {
220                let signum = SigNum::try_from(arg2 as u8)?;
221                Ok(PrctlCmd::PR_SET_PDEATHSIG(signum))
222            }
223            PR_GET_PDEATHSIG => Ok(PrctlCmd::PR_GET_PDEATHSIG(arg2 as _)),
224            PR_GET_DUMPABLE => Ok(PrctlCmd::PR_GET_DUMPABLE),
225            PR_SET_DUMPABLE => Ok(PrctlCmd::PR_SET_DUMPABLE(Dumpable::try_from(arg2)?)),
226            PR_GET_KEEPCAPS => Ok(PrctlCmd::PR_GET_KEEPCAPS),
227            PR_SET_KEEPCAPS => Ok(PrctlCmd::PR_SET_KEEPCAPS(arg2 as _)),
228            PR_SET_NAME => Ok(PrctlCmd::PR_SET_NAME(arg2 as _)),
229            PR_GET_NAME => Ok(PrctlCmd::PR_GET_NAME(arg2 as _)),
230            PR_CAPBSET_READ => Ok(PrctlCmd::PR_CAPBSET_READ(parse_capability(arg2)?)),
231            PR_CAPBSET_DROP => Ok(PrctlCmd::PR_CAPBSET_DROP(parse_capability(arg2)?)),
232            PR_GET_SECUREBITS => Ok(PrctlCmd::PR_GET_SECUREBITS),
233            PR_SET_SECUREBITS => {
234                let securebits = u16::try_from(arg2).map_err(|_| {
235                    Error::with_message(Errno::EPERM, "the bits are not valid secure bits")
236                })?;
237                Ok(PrctlCmd::PR_SET_SECUREBITS(SecureBits::try_from(
238                    securebits,
239                )?))
240            }
241            PR_SET_TIMERSLACK => Ok(PrctlCmd::PR_SET_TIMERSLACK(arg2)),
242            PR_GET_TIMERSLACK => Ok(PrctlCmd::PR_GET_TIMERSLACK),
243            PR_SET_CHILD_SUBREAPER => Ok(PrctlCmd::PR_SET_CHILD_SUBREAPER(arg2 > 0)),
244            PR_GET_CHILD_SUBREAPER => Ok(PrctlCmd::PR_GET_CHILD_SUBREAPER(arg2 as _)),
245            PR_SET_NO_NEW_PRIVS => {
246                if arg2 != 1 || arg3 != 0 || arg4 != 0 || arg5 != 0 {
247                    return_errno_with_message!(
248                        Errno::EINVAL,
249                        "invalid PR_SET_NO_NEW_PRIVS arguments"
250                    );
251                }
252                Ok(PrctlCmd::PR_SET_NO_NEW_PRIVS)
253            }
254            PR_GET_NO_NEW_PRIVS => {
255                if arg2 != 0 || arg3 != 0 || arg4 != 0 || arg5 != 0 {
256                    return_errno_with_message!(
257                        Errno::EINVAL,
258                        "invalid PR_GET_NO_NEW_PRIVS arguments"
259                    );
260                }
261                Ok(PrctlCmd::PR_GET_NO_NEW_PRIVS)
262            }
263            PR_CAP_AMBIENT => Ok(PrctlCmd::PR_CAP_AMBIENT(CapAmbientCmd::from_args(
264                arg2, arg3, arg4, arg5,
265            )?)),
266            _ => {
267                debug!("prctl cmd number: {}", option);
268                return_errno_with_message!(Errno::EINVAL, "unsupported prctl command");
269            }
270        }
271    }
272}

274impl CapAmbientCmd {
275    // Reference: <https://elixir.bootlin.com/linux/v6.16.5/source/include/uapi/linux/prctl.h#L196-L199>
276    const PR_CAP_AMBIENT_IS_SET: u64 = 1;
277    const PR_CAP_AMBIENT_RAISE: u64 = 2;
278    const PR_CAP_AMBIENT_LOWER: u64 = 3;
279    const PR_CAP_AMBIENT_CLEAR_ALL: u64 = 4;
280
281    fn from_args(operation: u64, capability: u64, arg4: u64, arg5: u64) -> Result<Self> {
282        if arg4 != 0 || arg5 != 0 {
283            return_errno_with_message!(Errno::EINVAL, "unused PR_CAP_AMBIENT arguments are not 0");
284        }
285
286        match operation {
287            Self::PR_CAP_AMBIENT_IS_SET => Ok(Self::IsSet(parse_capability(capability)?)),
288            Self::PR_CAP_AMBIENT_RAISE => Ok(Self::Raise(parse_capability(capability)?)),
289            Self::PR_CAP_AMBIENT_LOWER => Ok(Self::Lower(parse_capability(capability)?)),
290            Self::PR_CAP_AMBIENT_CLEAR_ALL => {
291                if capability != 0 {
292                    return_errno_with_message!(
293                        Errno::EINVAL,
294                        "PR_CAP_AMBIENT_CLEAR_ALL requires all arguments to be 0"
295                    );
296                }
297                Ok(Self::ClearAll)
298            }
299            _ => return_errno_with_message!(Errno::EINVAL, "invalid PR_CAP_AMBIENT operation"),
300        }
301    }
302}
303
304fn parse_capability(capability: u64) -> Result<CapSet> {
305    CapSet::from_capability_number(capability)
306        .ok_or_else(|| Error::with_message(Errno::EINVAL, "invalid capability number"))
