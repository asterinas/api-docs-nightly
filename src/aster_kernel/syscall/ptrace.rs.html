13pub fn sys_ptrace(
14    request: u32,
15    tid: Tid,
16    addr: u64,
17    data: u64,
18    ctx: &Context,
19) -> Result<SyscallReturn> {
20    let request = PtraceRequest::try_from(request)
21        .map_err(|_| Error::with_message(Errno::EIO, "invalid ptrace request"))?;
22    debug!(
23        "ptrace: request = {:?}, tid = {}, addr = 0x{:x}, data = 0x{:x}",
24        request, tid, addr, data
25    );
26
27    match request {
28        PtraceRequest::PTRACE_TRACEME => {
29            let current_thread = current_thread!();
30            let parent_guard = ctx.process.parent().lock();
31            let parent_main_thread = parent_guard.process().upgrade().unwrap().main_thread();
32
33            do_ptrace_attach(&parent_main_thread, current_thread)?;
34        }
35        PtraceRequest::PTRACE_CONT => {
36            let sig_num = parse_ptrace_injected_signal(data)?;
37
38            let tracee = ctx.posix_thread.get_tracee(tid)?;
39            tracee
40                .as_posix_thread()
41                .unwrap()
42                .ptrace_continue(PtraceContRequest::Continue(sig_num), ctx)?;
43        }
44    }
45
46    Ok(SyscallReturn::Return(0))
47}
48
49fn do_ptrace_attach(tracer_thread: &Arc<Thread>, tracee_thread: Arc<Thread>) -> Result<()> {
50    let tracer = tracer_thread.as_posix_thread().unwrap();
51    let tracee = tracee_thread.as_posix_thread().unwrap();
52    if !Arc::ptr_eq(&tracer.process().main_thread(), tracer_thread) {
53        return_errno_with_message!(
54            Errno::EOPNOTSUPP,
55            "using a non-main thread as a tracer is not supported currently"
56        );
57    }
58
59    if Weak::ptr_eq(tracer.weak_process(), tracee.weak_process()) {
60        return_errno_with_message!(
61            Errno::EPERM,
62            "tracer and tracee must be in different processes"
63        );
64    }
65
66    tracee.check_alien_access_from(tracer, AlienAccessMode::ATTACH_WITH_REAL_CREDS)?;
67
68    tracer.attach_to(tracer_thread, tracee_thread)
69}

71fn parse_ptrace_injected_signal(data: u64) -> Result<Option<SigNum>> {
72    if data == 0 {
73        return Ok(None);
74    }
75
76    let sig_num = u8::try_from(data)
77        .map_err(|_| Error::with_message(Errno::EINVAL, "invalid signal number"))?;
78    let sig_num = SigNum::try_from(sig_num)?;
79
80    Ok(Some(sig_num))
81}

94    // /// Return the word in the thread's data space at address ADDR.
95    // PTRACE_PEEKDATA = 2,
96    // /// Return the word in the thread's user area at offset ADDR.
97    // PTRACE_PEEKUSER = 3,
98    // /// Write the word DATA into the thread's text space at address ADDR.
99    // PTRACE_POKETEXT = 4,
100    // /// Write the word DATA into the thread's data space at address ADDR.
101    // PTRACE_POKEDATA = 5,
102    // /// Write the word DATA into the thread's user area at offset ADDR.
103    // PTRACE_POKEUSER = 6,
104    // /// Kill the thread.
105    // PTRACE_KILL = 8,
106    // /// Single step the thread.
107    // PTRACE_SINGLESTEP = 9,
108    // /// Get all general purpose registers used by a thread.
109    // PTRACE_GETREGS = 12,
110    // /// Set all general purpose registers used by a thread.
111    // PTRACE_SETREGS = 13,
112    // /// Get all floating point registers used by a thread.
113    // PTRACE_GETFPREGS = 14,
114    // /// Set all floating point registers used by a thread.
115    // PTRACE_SETFPREGS = 15,
116    // /// Attach to a thread that is already running.
117    // PTRACE_ATTACH = 16,
118    // /// Detach from a thread attached to.
119    // PTRACE_DETACH = 17,
120    // /// Get all extended floating point registers used by a thread.
121    // PTRACE_GETFPXREGS = 18,
122    // /// Set all extended floating point registers used by a thread.
123    // PTRACE_SETFPXREGS = 19,
124    // /// Continue and stop at the next entry to or return from syscall.
125    // PTRACE_SYSCALL = 24,
126    // /// Continue and stop at the next syscall, it will not be executed.
127    // PTRACE_SYSEMU = 31,
128    // /// Single step the thread, the next syscall will not be executed.
129    // PTRACE_SYSEMU_SINGLESTEP = 32,
130    // /// Set ptrace filter options.
131    // PTRACE_SETOPTIONS = 0x4200,
132    // /// Get the siginfo of the last ptrace-stop.
133    // PTRACE_GETSIGINFO = 0x4202,
134    // /// Get register content.
135    // PTRACE_GETREGSET = 0x4204,
136    // /// Set register content.
137    // PTRACE_SETREGSET = 0x4205,
