8pub fn sys_madvise(addr: Vaddr, len: usize, behavior: i32, ctx: &Context) -> Result<SyscallReturn> {
9    let behavior = MadviseBehavior::try_from(behavior)?;
10    debug!(
11        "addr = 0x{:x}, len = 0x{:x}, behavior = {:?}",
12        addr, len, behavior
13    );
14
15    if !addr.is_multiple_of(PAGE_SIZE) {
16        return_errno_with_message!(Errno::EINVAL, "the mapping address is not aligned");
17    }
18    if len == 0 {
19        return Ok(SyscallReturn::Return(0));
20    }
21    if VMAR_CAP_ADDR.checked_sub(addr).is_none_or(|gap| gap < len) {
22        // FIXME: Linux returns `EINVAL` if `(addr + len).align_up(PAGE_SIZE)` overflows. Here, we
23        // perform a stricter validation.
24        return_errno_with_message!(Errno::EINVAL, "the mapping range is not in userspace");
25    }
26    let addr_range = addr..(addr + len).align_up(PAGE_SIZE);
27
28    let user_space = ctx.user_space();
29    let vmar = user_space.vmar();
30
31    match behavior {
32        MadviseBehavior::MADV_DONTNEED => {
33            vmar.discard_pages(addr_range)?;
34        }
35        _ if DUMMY_MADVISE.contains(&behavior) => {
36            let query_guard = vmar.query(addr_range);
37            if !query_guard.is_fully_mapped() {
38                return_errno_with_message!(
39                    Errno::ENOMEM,
40                    "the range contains pages that are not mapped"
41                );
42            }
43            // For `DUMMY_MADVISE`, doing nothing is correct, though it may not be efficient.
44        }
45        _ => return_errno_with_message!(Errno::EINVAL, "the madvise behavior is not supported yet"),
46    }
47
48    Ok(SyscallReturn::Return(0))
49}

51// Reference: <https://elixir.bootlin.com/linux/v4.8/source/include/uapi/asm-generic/mman-common.h#L37>
52#[repr(i32)]
53#[derive(Debug, Clone, Copy, PartialEq, Eq, TryFromInt)]
54#[expect(non_camel_case_types)]
55enum MadviseBehavior {
56    MADV_NORMAL = 0,     /* no further special treatment */
57    MADV_RANDOM = 1,     /* expect random page references */
58    MADV_SEQUENTIAL = 2, /* expect sequential page references */
59    MADV_WILLNEED = 3,   /* will need these pages */
60    MADV_DONTNEED = 4,   /* don't need these pages */
61
62    /* common parameters: try to keep these consistent across architectures */
63    MADV_FREE = 8,           /* free pages only if memory pressure */
64    MADV_REMOVE = 9,         /* remove these pages & resources */
65    MADV_DONTFORK = 10,      /* don't inherit across fork */
66    MADV_DOFORK = 11,        /* do inherit across fork */
67    MADV_HWPOISON = 100,     /* poison a page for testing */
68    MADV_SOFT_OFFLINE = 101, /* soft offline page for testing */
69
70    MADV_MERGEABLE = 12,   /* KSM may merge identical pages */
71    MADV_UNMERGEABLE = 13, /* KSM may not merge identical pages */
72
73    MADV_HUGEPAGE = 14,   /* Worth backing with hugepages */
74    MADV_NOHUGEPAGE = 15, /* Not worth backing with hugepages */
75
76    MADV_DONTDUMP = 16, /* Explicitly exclude from the core dump,
77                        overrides the coredump filter bits */
78    MADV_DODUMP = 17, /* Clear the MADV_DONTDUMP flag */
79
80    MADV_WIPEONFORK = 18, /* Zero memory on fork, child only */
81    MADV_KEEPONFORK = 19, /* Undo MADV_WIPEONFORK */
82
83    MADV_COLD = 20,    /* deactivate these pages */
84    MADV_PAGEOUT = 21, /* reclaim these pages */
85
86    MADV_POPULATE_READ = 22,  /* populate (prefault) page tables readable */
87    MADV_POPULATE_WRITE = 23, /* populate (prefault) page tables writable */
88
89    MADV_DONTNEED_LOCKED = 24, /* like DONTNEED, but drop locked pages too */
90}

92/// Madvise that a dummy implementation is also correct.
93///
94/// This list can only contain madvise behaviors that do not alter the semantics of the user
95/// program. In other words, they are intended solely for performance optimization and can safely
96/// be ignored by the kernel.
97///
98/// **Please think twice before adding a new behavior to this list. Not all madvise behaviors can
99/// be no-ops.**
100const DUMMY_MADVISE: &[MadviseBehavior] = &[
101    MadviseBehavior::MADV_NORMAL,
102    MadviseBehavior::MADV_RANDOM,
103    MadviseBehavior::MADV_SEQUENTIAL,
104    MadviseBehavior::MADV_WILLNEED,
105    MadviseBehavior::MADV_FREE,
106    MadviseBehavior::MADV_MERGEABLE,
107    MadviseBehavior::MADV_UNMERGEABLE,
108    MadviseBehavior::MADV_HUGEPAGE,
109    MadviseBehavior::MADV_NOHUGEPAGE,
110];
