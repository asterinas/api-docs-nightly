11/// This struct corresponds to `SystemTime` in Rust std.
12#[derive(Debug, PartialEq, Eq, PartialOrd, Ord)]
13pub struct SystemTime(PrimitiveDateTime);
14
15pub static START_TIME: Once<SystemTime> = Once::new();
16pub(super) static START_TIME_AS_DURATION: Once<Duration> = Once::new();
17
18pub(super) fn init() {
19    let start_time = convert_system_time(read_start_time()).unwrap();
20    START_TIME_AS_DURATION
21        .call_once(|| start_time.duration_since(&SystemTime::UNIX_EPOCH).unwrap());
22    START_TIME.call_once(|| start_time);
23}
24
25impl SystemTime {
26    /// The unix epoch, which represents 1970-01-01 00:00:00
27    pub const UNIX_EPOCH: SystemTime = SystemTime::unix_epoch();
28
29    const fn unix_epoch() -> Self {
30        // 1970-01-01 00:00:00
31        let Ok(date) = Date::from_ordinal_date(1970, 1) else {
32            unreachable!()
33        };
34        let Ok(time) = Time::from_hms_nano(0, 0, 0, 0) else {
35            unreachable!()
36        };
37
38        SystemTime(PrimitiveDateTime::new(date, time))
39    }
40
41    /// Returns the current system time
42    pub fn now() -> Self {
43        // The get real time result should always be valid
44        START_TIME
45            .get()
46            .unwrap()
47            .checked_add(read_monotonic_time())
48            .unwrap()
49    }
50
51    /// Add a duration to self. If the result does not exceed inner bounds return Some(t), else return None.
52    pub fn checked_add(&self, duration: Duration) -> Option<Self> {
53        let duration = convert_to_time_duration(duration);
54        self.0.checked_add(duration).map(SystemTime)
55    }
56
57    /// Subtract a duration from self. If the result does not exceed inner bounds return Some(t), else return None.
58    #[expect(dead_code)]
59    pub fn checked_sub(&self, duration: Duration) -> Option<Self> {
60        let duration = convert_to_time_duration(duration);
61        self.0.checked_sub(duration).map(SystemTime)
62    }
63
64    /// Returns the duration since an earlier time. Return error if `earlier` is later than self.
65    pub fn duration_since(&self, earlier: &SystemTime) -> Result<Duration> {
66        if self.0 < earlier.0 {
67            return_errno_with_message!(
68                Errno::EINVAL,
69                "duration_since can only accept an earlier time"
70            );
71        }
72        let duration = self.0 - earlier.0;
73        Ok(convert_to_core_duration(duration))
74    }
75
76    /// Return the difference between current time and the time when self was created.
77    /// Return Error if current time is earlier than creating time.
78    /// The error can happen if self was created by checked_add.
79    #[expect(dead_code)]
80    pub fn elapsed(&self) -> Result<Duration> {
81        let now = SystemTime::now();
82        now.duration_since(self)
83    }
84}

86/// convert ostd::time::Time to System time
87fn convert_system_time(system_time: aster_time::SystemTime) -> Result<SystemTime> {
88    let month = match Month::try_from(system_time.month) {
89        Ok(month) => month,
90        Err(_) => return_errno_with_message!(Errno::EINVAL, "unknown month in system time"),
91    };
92    let date = match Date::from_calendar_date(system_time.year as _, month, system_time.day) {
93        Ok(date) => date,
94        Err(_) => return_errno_with_message!(Errno::EINVAL, "Invalid system date"),
95    };
96    let time_ = match Time::from_hms_nano(
97        system_time.hour,
98        system_time.minute,
99        system_time.second,
100        system_time.nanos.try_into().unwrap(),
101    ) {
102        Ok(time_) => time_,
103        Err(_) => return_errno_with_message!(Errno::EINVAL, "Invalid system time"),
104    };
105    Ok(SystemTime(PrimitiveDateTime::new(date, time_)))
106}

108/// FIXME: need to further check precision loss
109/// convert core::time::Duration to time::Duration
110const fn convert_to_time_duration(duration: Duration) -> time::Duration {
111    let seconds = duration.as_secs() as i64;
112    let nanoseconds = duration.subsec_nanos() as i32;
113    time::Duration::new(seconds, nanoseconds)
114}

116/// FIXME: need to further check precision loss
117/// convert time::Duration to core::time::Duration
118const fn convert_to_core_duration(duration: time::Duration) -> Duration {
119    let seconds = duration.whole_seconds() as u64;
120    let nanoseconds = duration.subsec_nanoseconds() as u32;
121    Duration::new(seconds, nanoseconds)
