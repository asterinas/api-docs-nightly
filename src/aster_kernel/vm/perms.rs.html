9bitflags! {
10    /// The memory access permissions of memory mappings.
11    // NOTE: `check` hardcodes `MAY_READ >> 3 == READ`, and so for r/w/x bits.
12    pub struct VmPerms: u32 {
13        /// Readable.
14        const READ    = 1 << 0;
15        /// Writable.
16        const WRITE   = 1 << 1;
17        /// Executable.
18        const EXEC   = 1 << 2;
19        /// May be protected to readable.
20        const MAY_READ = 1 << 3;
21        /// May be protected to writable.
22        const MAY_WRITE = 1 << 4;
23        /// May be protected to executable.
24        const MAY_EXEC = 1 << 5;
25        /// All permissions (READ | WRITE | EXEC).
26        const ALL_PERMS = Self::READ.bits | Self::WRITE.bits | Self::EXEC.bits;
27        /// All `MAY_*` permissions (MAY_READ | MAY_WRITE | MAY_EXEC).
28        const ALL_MAY_PERMS =  Self::MAY_READ.bits | Self::MAY_WRITE.bits | Self::MAY_EXEC.bits;
29    }
30}
31
32impl VmPerms {
33    /// Checks whether all requested permissions (`READ`, `WRITE`, `EXEC`) are
34    /// allowed by their corresponding `MAY_*` capabilities.
35    pub fn check(&self) -> Result<()> {
36        let requested = *self & Self::ALL_PERMS;
37        // NOTE: `MAY_READ >> 3 == READ`, and so for r/w/x bits.
38        let allowed = VmPerms::from_bits_truncate((*self & Self::ALL_MAY_PERMS).bits >> 3);
39        if !allowed.contains(requested) {
40            return_errno_with_message!(Errno::EACCES, "permission denied");
41        }
42
43        Ok(())
44    }
45
46    /// Parses `bits` as requested permissions from user programs and returns errors
47    /// if there are unknown permissions.
48    pub fn from_user_bits(bits: u32) -> Result<Self> {
49        if let Some(vm_perms) = VmPerms::from_bits(bits)
50            && Self::ALL_PERMS.contains(vm_perms)
51        {
52            Ok(vm_perms)
53        } else {
54            return_errno_with_message!(Errno::EINVAL, "invalid permissions");
55        }
56    }
57
58    /// Parses `bits` as requested permissions from user programs and ignores any
59    /// unknown permissions.
60    pub fn from_user_bits_truncate(bits: u32) -> Self {
61        VmPerms::from_bits_truncate(bits) & Self::ALL_PERMS
62    }
63}
64
65impl From<Rights> for VmPerms {
66    fn from(rights: Rights) -> VmPerms {
67        let mut vm_perm = VmPerms::empty();
68        if rights.contains(Rights::READ) {
69            vm_perm |= VmPerms::READ;
70        }
71        if rights.contains(Rights::WRITE) {
72            vm_perm |= VmPerms::WRITE;
73        }
74        if rights.contains(Rights::EXEC) {
75            vm_perm |= VmPerms::EXEC;
76        }
77        vm_perm
78    }
79}
80
81impl From<VmPerms> for Rights {
82    fn from(vm_perms: VmPerms) -> Rights {
83        let mut rights = Rights::empty();
84        if vm_perms.contains(VmPerms::READ) {
85            rights |= Rights::READ;
86        }
87        if vm_perms.contains(VmPerms::WRITE) {
88            rights |= Rights::WRITE;
89        }
90        if vm_perms.contains(VmPerms::EXEC) {
91            rights |= Rights::EXEC;
92        }
93        rights
94    }
95}

97impl From<PageFlags> for VmPerms {
98    fn from(flags: PageFlags) -> Self {
99        let mut perms = VmPerms::empty();
100        if flags.contains(PageFlags::R) {
101            perms |= VmPerms::READ;
102        }
103        if flags.contains(PageFlags::W) {
104            perms |= VmPerms::WRITE;
105        }
106        if flags.contains(PageFlags::X) {
107            perms |= VmPerms::EXEC;
108        }
109        perms
110    }
111}

113impl From<VmPerms> for PageFlags {
114    fn from(val: VmPerms) -> Self {
115        let mut flags = PageFlags::empty();
116        if val.contains(VmPerms::READ) {
117            flags |= PageFlags::R;
118        }
119        if val.contains(VmPerms::WRITE) {
120            flags |= PageFlags::W;
121        }
122        if val.contains(VmPerms::EXEC) {
123            flags |= PageFlags::X;
124        }
125        flags
126    }
