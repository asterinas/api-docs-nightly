12impl Vmar {
13    /// Reads memory from the process user space.
14    ///
15    /// This method reads until one of the conditions is met:
16    /// 1. The writer has no available space.
17    /// 2. Reading from the process user space or writing to the writer encounters some error.
18    ///
19    /// On success, the number of bytes read is returned;
20    /// On error, both the error and the number of bytes read so far are returned.
21    ///
22    /// The `VmSpace` of the process is not required to be activated on the current CPU.
23    pub fn read_remote(
24        &self,
25        vaddr: Vaddr,
26        writer: &mut VmWriter,
27    ) -> core::result::Result<usize, (Error, usize)> {
28        let len = writer.avail();
29        let read = |frame: UFrame, skip_offset: usize| {
30            let mut reader = frame.reader();
31            reader.skip(skip_offset);
32            reader.read_fallible(writer)
33        };
34
35        self.access_remote(vaddr, len, PageFlags::R, read)
36    }
37
38    /// Writes memory to the process user space.
39    ///
40    /// This method writes until one of the conditions is met:
41    /// 1. The reader has no remaining data.
42    /// 2. Reading from the reader or writing to the process user space encounters some error.
43    ///
44    /// On success, the number of bytes written is returned;
45    /// On error, both the error and the number of bytes written so far are returned.
46    ///
47    /// The `VmSpace` of the process is not required to be activated on the current CPU.
48    pub fn write_remote(
49        &self,
50        vaddr: Vaddr,
51        reader: &mut VmReader,
52    ) -> core::result::Result<usize, (Error, usize)> {
53        let len = reader.remain();
54        let write = |frame: UFrame, skip_offset: usize| {
55            let mut writer = frame.writer();
56            writer.skip(skip_offset);
57            writer.write_fallible(reader)
58        };
59
60        self.access_remote(vaddr, len, PageFlags::W, write)
61    }
62
63    /// Writes zeros to the process user space.
64    ///
65    /// This method writes at most `len` bytes of zeros to the process user space.
66    /// On success, the number of bytes written is returned; on error, both the
67    /// error and the number of bytes written so far are returned.
68    ///
69    /// The `VmSpace` of the process is not required to be activated on the current CPU.
70    pub fn fill_zeros_remote(
71        &self,
72        vaddr: Vaddr,
73        len: usize,
74    ) -> core::result::Result<usize, (Error, usize)> {
75        let mut remain = len;
76        let write = |frame: UFrame, skip_offset: usize| {
77            let mut writer = frame.writer();
78            writer.skip(skip_offset);
79            let res = writer.fill_zeros(remain);
80            remain -= res;
81            Ok(res)
82        };
83
84        self.access_remote(vaddr, len, PageFlags::W, write)
85    }
86
87    /// Accesses memory at `vaddr..vaddr+len` within the process user space using `op`.
88    ///
89    /// The `VmSpace` of the process is not required to be activated on the current CPU.
90    /// If any page in the range is not mapped or does not have the required page
91    /// flags, a page fault will be handled to try to make the page accessible.
92    fn access_remote<F>(
93        &self,
94        vaddr: Vaddr,
95        len: usize,
96        required_page_flags: PageFlags,
97        mut op: F,
98    ) -> core::result::Result<usize, (Error, usize)>
99    where
100        F: FnMut(UFrame, usize) -> core::result::Result<usize, (ostd::Error, usize)>,
101    {
102        if len == 0 {
103            return Ok(0);
104        }
105
106        if !is_userspace_vaddr_range(vaddr, len) {
107            return Err((
108                Error::with_message(Errno::EINVAL, "the address range is not in userspace"),
109                0,
110            ));
111        }
112        let range = vaddr.align_down(PAGE_SIZE)..(vaddr + len).align_up(PAGE_SIZE);
113
114        let mut current_va = range.start;
115        let mut bytes = 0;
116
117        while current_va < range.end {
118            let frame = self
119                .query_page_with_required_flags(current_va, required_page_flags)
120                .map_err(|err| (err, bytes))?;
121
122            let skip_offset = if current_va == range.start {
123                vaddr - range.start
124            } else {
125                0
126            };
127            match op(frame, skip_offset) {
128                Ok(n) => bytes += n,
129                Err((err, n)) => return Err((err.into(), bytes + n)),
130            }
131
132            current_va += PAGE_SIZE;
133        }
134
135        Ok(bytes)
136    }
137
138    fn query_page_with_required_flags(
139        &self,
140        vaddr: Vaddr,
141        required_page_flags: PageFlags,
142    ) -> Result<UFrame> {
143        let mut item = self.query_page(vaddr)?;
144
145        let vm_item = loop {
146            match item {
147                Some(vm_item) if vm_item.prop().flags.contains(required_page_flags) => {
148                    break vm_item;
149                }
150                Some(_) | None => (),
151            }
152
153            let page_fault_info = PageFaultInfo {
154                address: vaddr,
155                required_perms: required_page_flags.into(),
156            };
157            self.handle_page_fault(&page_fault_info)?;
158
159            item = self.query_page(vaddr)?;
160
161            // Note that we are not holding `self.inner.lock()` here. Therefore, in race conditions
162            // (e.g., if the mapping is removed concurrently), we will need to try again. The same
163            // is true for real page faults; they may occur more than once at the same address.
164        };
165
166        match vm_item {
167            VmQueriedItem::MappedRam { frame, .. } => Ok(frame),
168            VmQueriedItem::MappedIoMem { .. } => {
169                return_errno_with_message!(
170                    Errno::EOPNOTSUPP,
171                    "accessing remote MMIO memory is not supported currently"
172                );
173            }
174        }
175    }
176
177    fn query_page(&self, vaddr: Vaddr) -> Result<Option<VmQueriedItem>> {
178        debug_assert!(is_userspace_vaddr(vaddr) && vaddr.is_multiple_of(PAGE_SIZE));
179
180        let preempt_guard = disable_preempt();
181        let vmspace = self.vm_space();
182        let mut cursor = vmspace.cursor(&preempt_guard, &(vaddr..vaddr + PAGE_SIZE))?;
183        let (_, item) = cursor.query()?;
184
185        Ok(item)
186    }
