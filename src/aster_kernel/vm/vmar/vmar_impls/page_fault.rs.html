3use super::{Interval, RssDelta, Vmar};
4use crate::{prelude::*, thread::exception::PageFaultInfo};
5
6impl Vmar {
7    pub fn handle_page_fault(&self, page_fault_info: &PageFaultInfo) -> Result<()> {
8        let inner = self.inner.read();
9
10        let address = page_fault_info.address;
11        if let Some(vm_mapping) = inner.vm_mappings.find_one(&address) {
12            debug_assert!(vm_mapping.range().contains(&address));
13
14            let mut rss_delta = RssDelta::new(self);
15            return vm_mapping.handle_page_fault(&self.vm_space, page_fault_info, &mut rss_delta);
16        }
17
18        return_errno_with_message!(
19            Errno::EACCES,
20            "no VM mappings contain the page fault address"
21        );
22    }
23}
