3//! This module enables access to a VMAR
4//! from an [_alien thread_] (crate::process::posix_thread::alien_access),
5//! which is a non-member of the owner process of the VMAR.
6
7use align_ext::AlignExt;
8use ostd::{
9    mm::{PAGE_SIZE, PageFlags, UFrame, io::util::HasVmReaderWriter, vm_space::VmQueriedItem},
10    task::disable_preempt,
11};
12
13use super::{Vmar, is_userspace_vaddr};
14use crate::{
15    prelude::*,
16    vm::vmar::{PageFaultInfo, is_userspace_vaddr_range},
17};
18
19impl Vmar {
20    /// Reads memory in the context of an alien thread.
21    ///
22    /// This method reads until one of the conditions is met:
23    /// 1. The writer has no available space.
24    /// 2. Reading from the process user space or writing to the writer encounters some error.
25    ///
26    /// On success, the number of bytes read is returned;
27    /// On error, both the error and the number of bytes read so far are returned.
28    ///
29    /// The `VmSpace` of the process is not required to be activated on the current CPU.
30    pub fn read_alien(
31        &self,
32        vaddr: Vaddr,
33        writer: &mut VmWriter,
34    ) -> core::result::Result<usize, (Error, usize)> {
35        let len = writer.avail();
36        let read = |frame: UFrame, skip_offset: usize| {
37            let mut reader = frame.reader();
38            reader.skip(skip_offset);
39            reader.read_fallible(writer)
40        };
41
42        self.access_alien(vaddr, len, PageFlags::R, read)
43    }
44
45    /// Writes memory in the context of an alien thread.
46    ///
47    /// This method writes until one of the conditions is met:
48    /// 1. The reader has no remaining data.
49    /// 2. Reading from the reader or writing to the process user space encounters some error.
50    ///
51    /// On success, the number of bytes written is returned;
52    /// On error, both the error and the number of bytes written so far are returned.
53    ///
54    /// The `VmSpace` of the process is not required to be activated on the current CPU.
55    pub fn write_alien(
56        &self,
57        vaddr: Vaddr,
58        reader: &mut VmReader,
59    ) -> core::result::Result<usize, (Error, usize)> {
60        let len = reader.remain();
61        let write = |frame: UFrame, skip_offset: usize| {
62            let mut writer = frame.writer();
63            writer.skip(skip_offset);
64            writer.write_fallible(reader)
65        };
66
67        self.access_alien(vaddr, len, PageFlags::W, write)
68    }
69
70    /// Writes zeros in the context of an alien thread.
71    ///
72    /// This method writes at most `len` bytes of zeros to the process user space.
73    /// On success, the number of bytes written is returned; on error, both the
74    /// error and the number of bytes written so far are returned.
75    ///
76    /// The `VmSpace` of the process is not required to be activated on the current CPU.
77    pub fn fill_zeros_alien(
78        &self,
79        vaddr: Vaddr,
80        len: usize,
81    ) -> core::result::Result<usize, (Error, usize)> {
82        let mut remain = len;
83        let write = |frame: UFrame, skip_offset: usize| {
84            let mut writer = frame.writer();
85            writer.skip(skip_offset);
86            let res = writer.fill_zeros(remain);
87            remain -= res;
88            Ok(res)
89        };
90
91        self.access_alien(vaddr, len, PageFlags::W, write)
92    }
93
94    /// Accesses memory at `vaddr..vaddr+len` in the context of an alien thread using `op`.
95    ///
96    /// The `VmSpace` of the process is not required to be activated on the current CPU.
97    /// If any page in the range is not mapped or does not have the required page
98    /// flags, a page fault will be handled to try to make the page accessible.
99    fn access_alien<F>(
100        &self,
101        vaddr: Vaddr,
102        len: usize,
103        required_page_flags: PageFlags,
104        mut op: F,
105    ) -> core::result::Result<usize, (Error, usize)>
106    where
107        F: FnMut(UFrame, usize) -> core::result::Result<usize, (ostd::Error, usize)>,
108    {
109        if len == 0 {
110            return Ok(0);
111        }
112
113        if !is_userspace_vaddr_range(vaddr, len) {
114            return Err((
115                Error::with_message(Errno::EINVAL, "the address range is not in userspace"),
116                0,
117            ));
118        }
119        let range = vaddr.align_down(PAGE_SIZE)..(vaddr + len).align_up(PAGE_SIZE);
120
121        let mut current_va = range.start;
122        let mut bytes = 0;
123
124        while current_va < range.end {
125            let frame = self
126                .query_page_with_required_flags(current_va, required_page_flags)
127                .map_err(|err| (err, bytes))?;
128
129            let skip_offset = if current_va == range.start {
130                vaddr - range.start
131            } else {
132                0
133            };
134            match op(frame, skip_offset) {
135                Ok(n) => bytes += n,
136                Err((err, n)) => return Err((err.into(), bytes + n)),
137            }
138
139            current_va += PAGE_SIZE;
140        }
141
142        Ok(bytes)
143    }
144
145    fn query_page_with_required_flags(
146        &self,
147        vaddr: Vaddr,
148        required_page_flags: PageFlags,
149    ) -> Result<UFrame> {
150        debug_assert!(is_userspace_vaddr(vaddr) && vaddr.is_multiple_of(PAGE_SIZE));
151
152        let vmspace = self.vm_space();
153
154        loop {
155            let preempt_guard = disable_preempt();
156            let mut cursor = vmspace.cursor(&preempt_guard, &(vaddr..vaddr + PAGE_SIZE))?;
157
158            match cursor.query()?.1 {
159                Some(vm_item) if vm_item.prop().flags.contains(required_page_flags) => {
160                    match vm_item {
161                        VmQueriedItem::MappedRam { frame, .. } => return Ok((*frame).clone()),
162                        VmQueriedItem::MappedIoMem { .. } => {
163                            return_errno_with_message!(
164                                Errno::EOPNOTSUPP,
165                                "accessing alien MMIO memory is not supported currently"
166                            );
167                        }
168                    }
169                }
170                Some(_) | None => (),
171            }
172
173            drop(cursor);
174            drop(preempt_guard);
175
176            let page_fault_info = PageFaultInfo::new(vaddr, required_page_flags.into()).force();
177            self.handle_page_fault(&page_fault_info)?;
178
179            // Note that we are not holding `self.inner.lock()` here. Therefore, in race conditions
180            // (e.g., if the mapping is removed concurrently), we will need to try again. The same
181            // is true for real page faults; they may occur more than once at the same address.
182        }
183    }
184}
