3use core::ops::Range;
4
5use ostd::sync::RwMutexReadGuard;
6
7use super::{VmMapping, Vmar, VmarInner};
8
9impl Vmar {
10    /// Finds all the mapped regions that intersect with the specified range.
11    pub fn query(&self, range: Range<usize>) -> VmarQueryGuard<'_> {
12        VmarQueryGuard {
13            vmar: self.inner.read(),
14            range,
15        }
16    }
17}
18
19/// A guard that allows querying a [`Vmar`] for its mappings.
20pub struct VmarQueryGuard<'a> {
21    vmar: RwMutexReadGuard<'a, VmarInner>,
22    range: Range<usize>,
23}
24
25impl VmarQueryGuard<'_> {
26    /// Returns an iterator over the [`VmMapping`]s that intersect with the
27    /// provided range when calling [`Vmar::query`].
28    pub fn iter(&self) -> impl Iterator<Item = &VmMapping> {
29        self.vmar.query(&self.range)
30    }
31
32    /// Returns whether the range is fully mapped.
33    ///
34    /// In other words, this method will return `false` if and only if the
35    /// range contains pages that are not mapped.
36    pub fn is_fully_mapped(&self) -> bool {
37        let mut last_mapping_end = self.range.start;
38
39        for mapping in self.iter() {
40            if last_mapping_end < mapping.map_to_addr() {
41                return false;
42            }
43            last_mapping_end = mapping.map_end();
44        }
45
46        if last_mapping_end < self.range.end {
47            return false;
48        }
49
50        true
51    }
52}
