3use ostd::{mm::vm_space::VmQueriedItem, task::disable_preempt};

5use super::{RssDelta, Vmar, util::is_intersected};
6use crate::{prelude::*, vm::vmar::is_userspace_vaddr_range};

8impl Vmar {
9    /// Resizes the original mapping.
10    ///
11    /// The range of the mapping goes from `map_addr..map_addr + old_size` to
12    /// `map_addr..map_addr + new_size`.
13    ///
14    /// If the new mapping size is smaller than the original mapping size, the
15    /// extra part will be unmapped. If the new mapping is larger than the old
16    /// mapping and the extra part overlaps with existing mapping, resizing
17    /// will fail and return `Err`.
18    ///
19    /// - When `check_single_mapping` is `true`, this method will check whether
20    ///   the range of the original mapping is covered by a single [`VmMapping`].
21    ///   If not, this method will return an `Err`.
22    /// - When `check_single_mapping` is `false`, The range of the original
23    ///   mapping does not have to solely map to a whole [`VmMapping`], but it
24    ///   must ensure that all existing ranges have a mapping. Otherwise, this
25    ///   method will return an `Err`.
26    ///
27    /// [`VmMapping`]: crate::vm::vmar::vm_mapping::VmMapping
28    pub fn resize_mapping(
29        &self,
30        map_addr: Vaddr,
31        old_size: usize,
32        new_size: usize,
33        check_single_mapping: bool,
34    ) -> Result<()> {
35        let mut inner = self.inner.write();
36        let mut rss_delta = RssDelta::new(self);
37
38        if check_single_mapping {
39            inner.check_lies_in_single_mapping(map_addr, old_size)?;
40        } else if inner.vm_mappings.find_one(&map_addr).is_none() {
41            return_errno_with_message!(Errno::EFAULT, "there is no mapping at the old address")
42        }
43        // FIXME: We should check whether all existing ranges in
44        // `map_addr..map_addr + old_size` have a mapping. If not,
45        // we should return an `Err`.
46
47        inner.resize_mapping(&self.vm_space, map_addr, old_size, new_size, &mut rss_delta)
48    }
49
50    /// Remaps the original mapping to a new address and/or size.
51    ///
52    /// If the new mapping size is smaller than the original mapping size, the
53    /// extra part will be unmapped.
54    ///
55    /// - If `new_addr` is `Some(new_addr)`, this method attempts to move the
56    ///   mapping from `old_addr..old_addr + old_size` to `new_addr..new_addr +
57    ///   new_size`. If any existing mappings lie within the target range,
58    ///   they will be unmapped before the move.
59    /// - If `new_addr` is `None`, a new range of size `new_size` will be
60    ///   allocated, and the original mapping will be moved there.
61    ///
62    /// # Panics
63    ///
64    /// This method panics if `new_addr` is `None` and `new_size <= old_size`.
65    /// Use `resize_mapping` instead in this case.
66    pub fn remap(
67        &self,
68        old_addr: Vaddr,
69        old_size: usize,
70        new_addr: Option<Vaddr>,
71        new_size: usize,
72    ) -> Result<Vaddr> {
73        debug_assert_eq!(old_addr % PAGE_SIZE, 0);
74        debug_assert_eq!(old_size % PAGE_SIZE, 0);
75        debug_assert_eq!(new_size % PAGE_SIZE, 0);
76
77        let mut inner = self.inner.write();
78        let mut rss_delta = RssDelta::new(self);
79
80        let Some(old_mapping) = inner.vm_mappings.find_one(&old_addr) else {
81            return_errno_with_message!(
82                Errno::EFAULT,
83                "remap: there is no mapping at the old address"
84            )
85        };
86        if new_size > old_size {
87            if !old_mapping.can_expand() {
88                return_errno_with_message!(
89                    Errno::EFAULT,
90                    "remap: device mappings cannot be expanded"
91                );
92            }
93            inner.check_extra_size_fits_rlimit(new_size - old_size)?;
94        }
95
96        // Shrink the old mapping first.
97        if old_addr.checked_add(old_size).is_none() {
98            return_errno_with_message!(Errno::EINVAL, "remap: the address range overflows");
99        }
100        let (old_size, old_range) = if new_size < old_size {
101            inner.alloc_free_region_exact_truncate(
102                &self.vm_space,
103                old_addr + new_size,
104                old_size - new_size,
105                &mut rss_delta,
106            )?;
107            (new_size, old_addr..old_addr + new_size)
108        } else {
109            (old_size, old_addr..old_addr + old_size)
110        };
111
112        // Allocate a new free region that does not overlap with the old range.
113        let new_range = if let Some(new_addr) = new_addr {
114            if !new_addr.is_multiple_of(PAGE_SIZE) || !is_userspace_vaddr_range(new_addr, new_size)
115            {
116                return_errno_with_message!(
117                    Errno::EINVAL,
118                    "remap: the new range is not aligned or not in userspace"
119                );
120            }
121            if is_intersected(&old_range, &(new_addr..new_addr + new_size)) {
122                return_errno_with_message!(
123                    Errno::EINVAL,
124                    "remap: the new range overlaps with the old one"
125                );
126            }
127            inner.alloc_free_region_exact_truncate(
128                &self.vm_space,
129                new_addr,
130                new_size,
131                &mut rss_delta,
132            )?
133        } else {
134            debug_assert!(new_size > old_size);
135
136            // Fast path: expand the old mapping in place to the new size
137            if is_userspace_vaddr_range(old_addr, new_size)
138                && inner
139                    .alloc_free_region_exact(old_range.end, new_size - old_size)
140                    .is_ok()
141            {
142                let old_mapping_addr = inner.check_lies_in_single_mapping(old_addr, old_size)?;
143                let old_mapping = inner.remove(&old_mapping_addr).unwrap();
144                let new_mapping = old_mapping.enlarge(new_size - old_size);
145                inner.insert_try_merge(new_mapping);
146                return Ok(old_range.start);
147            }
148
149            inner.alloc_free_region(new_size, PAGE_SIZE)?
150        };
151
152        // Create a new `VmMapping`.
153        let old_mapping = {
154            let old_mapping_addr = inner.check_lies_in_single_mapping(old_addr, old_size)?;
155            let vm_mapping = inner.remove(&old_mapping_addr).unwrap();
156            let (left, old_mapping, right) = vm_mapping.split_range(&old_range);
157            if let Some(left) = left {
158                inner.insert_without_try_merge(left);
159            }
160            if let Some(right) = right {
161                inner.insert_without_try_merge(right);
162            }
163            old_mapping
164        };
165        // Note that we have ensured that `new_size >= old_size` at the beginning.
166        let new_mapping = old_mapping.clone_for_remap_at(new_range.start);
167        inner.insert_try_merge(new_mapping.enlarge(new_size - old_size));
168
169        let preempt_guard = disable_preempt();
170        let total_range = old_range.start.min(new_range.start)..old_range.end.max(new_range.end);
171        let vmspace = self.vm_space();
172        let mut cursor = vmspace.cursor_mut(&preempt_guard, &total_range).unwrap();
173
174        // Move the mapping.
175        let mut current_offset = 0;
176        while current_offset < old_size {
177            cursor.jump(old_range.start + current_offset).unwrap();
178            let Some(mapped_va) = cursor.find_next(old_size - current_offset) else {
179                break;
180            };
181            let (va, Some(item)) = cursor.query().unwrap() else {
182                panic!("Found mapped page but query failed");
183            };
184            debug_assert_eq!(mapped_va, va.start);
185            cursor.unmap(PAGE_SIZE);
186
187            let offset = mapped_va - old_range.start;
188            cursor.jump(new_range.start + offset).unwrap();
189
190            match item {
191                VmQueriedItem::MappedRam { frame, prop } => {
192                    cursor.map(frame, prop);
193                }
194                VmQueriedItem::MappedIoMem { paddr, prop } => {
195                    // For MMIO pages, find the corresponding `IoMem` and map it
196                    // at the new location
197                    let (iomem, offset) = cursor.find_iomem_by_paddr(paddr).unwrap();
198                    cursor.map_iomem(iomem, prop, PAGE_SIZE, offset);
199                }
200            }
201
202            current_offset = offset + PAGE_SIZE;
203        }
204
205        cursor.flusher().dispatch_tlb_flush();
206        cursor.flusher().sync_tlb_flush();
207
208        Ok(new_range.start)
209    }
