3//! Lower-level virtual memory object machinery used by the page cache.
4//!
5//! This submodule defines [`Vmo`] and [`VmoOptions`], the generic memory-object
6//! layer underneath [`crate::vm::page_cache::PageCache`]. A [`Vmo`] owns a sparse
7//! array of pages, commits them on demand, and optionally cooperates with a
8//! [`crate::vm::page_cache::PageCacheBackend`] to populate or write back pages with
9//! a backend.
10//!
11//! # When to Use This Module
12//!
13//! - Filesystem buffered I/O, invalidation, and resize should usually stay at
14//!   [`crate::vm::page_cache::PageCache`].
15//! - Memory-mapping, page-fault, and other lower-level VM paths that need
16//!   direct page-oriented access should work with [`Vmo`].

43/// Page-indexed memory object used by the page cache and mapping code.
44///
45/// A `Vmo` represents a contiguous byte range backed by a sparse array of
46/// [`CachePage`]s. Pages are materialized on demand: anonymous VMOs allocate
47/// zero-filled RAM pages, while VMOs with a backend consult a
48/// [`PageCacheBackend`] to populate or write back individual pages.
49///
50/// `Vmo` is the storage engine underneath [`crate::vm::page_cache::PageCache`]. It
51/// understands page indices, per-page state transitions, and optional backend
52/// I/O, but it deliberately stays below filesystem policy. In particular, it
53/// does not define logical EOF semantics, extent mapping policy, or whole-file
54/// synchronization.
55///
56/// # Backing Modes
57///
58/// - **Anonymous VMO.** RAM-backed and zero-filled on first access. This is
59///   used for anonymous memory and in-memory page caches.
60/// - **Backend VMO.** Connected to a [`PageCacheBackend`]. Pages are loaded
61///   lazily and can later be written back.
62///
63/// # What `Vmo` Owns
64///
65/// - the page-aligned byte span returned by [`Vmo::size`];
66/// - the sparse page array and page-commit path;
67/// - per-page initialization, dirtying, and writeback coordination; and
68/// - writable-mapping tracking for callers that opt into it.
69///
70/// # What Stays Outside `Vmo`
71///
72/// Filesystems and higher layers still own:
73///
74/// - logical file size and inode metadata;
75/// - extent or block mapping decisions embodied by the backend; and
76/// - the higher-level locking that excludes conflicting buffered I/O,
77///   invalidation, truncate, or page-fault operations.
78///
79/// # Size Synchronization
80///
81/// A file that uses [`crate::vm::page_cache::PageCache`] has two related sizes:
82///
83/// - the logical file size stored in filesystem metadata; and
84/// - the page-aligned accessible range returned by [`Vmo::size`].
85///
86/// The logical file size defines EOF for filesystem operations. [`Vmo::size`]
87/// defines how far lower-level VMO operations such as reads, page commits, and
88/// page faults may access, and it may be larger because it is rounded up to
89/// whole pages.
90///
91/// These two sizes must be updated while holding the same higher-level
92/// inode- or file-level lock that serializes resize against buffered I/O, page
93/// faults, and invalidation. The filesystem-facing resize API,
94/// [`crate::vm::page_cache::PageCache::resize`], documents the required ordering
95/// between logical file size updates and [`Vmo::size`] changes.
96///
97/// # Backend Page Lifecycle
98///
99/// For pages with a backend, the state machine is:
100///
101/// - `Uninit -> UpToDate`: the page is populated from the backend, or a hole is
102///   materialized as zeros.
103/// - `Uninit -> Dirty`: a full-page overwrite commits and keeps the page
104///   locked until the writer has copied its new contents.
105/// - `UpToDate -> Dirty`: buffered writes modify the page under the page lock.
106/// - `Dirty -> UpToDate`: the VMO writeback path hands the page to the backend.
107///
108/// The auxiliary `is_writing_back` bit is set under the page lock, then cleared
109/// later by the BIO completion callback after the writeback state has been
110/// handed off. Anonymous VMOs stay `UpToDate` in steady state once a page is
111/// committed.
112pub struct Vmo {
113    /// The backend that provides disk I/O operations, if any.
114    //
115    // TODO: Using `Weak` here is to avoid circular references in exfat file systems.
116    // We should avoid the circular reference by design, and then we can change this to `Arc`.
117    pub(super) backend: Option<Weak<dyn PageCacheBackend>>,
118    /// Flags
119    pub(super) flags: VmoFlags,
120    /// The virtual pages where the VMO resides.
121    pub(super) pages: XArray<CachePage>,
122    /// The size of the VMO.
123    ///
124    /// This is the page-aligned VMO size, which may differ from the
125    /// filesystem's logical file size. Updates must happen after locking the
126    /// [`XArray`] in `pages`, so a size read after taking that lock observes
127    /// the latest resize performed under the required higher-level file-size
128    /// synchronization.
129    pub(super) size: AtomicUsize,
130    /// The status of writable mappings of the VMO.
131    //
132    // TODO: This field is used only by VMOs belonging to memfd (i.e., `MemfdInode`). But VMOs do
133    // not have the knowledge to determine if they belong to memfd. We may want to enhance
134    // `VmoOptions` to make VMOs aware of whether its writable mappings should be tracked.
135    pub(super) writable_mapping_status: WritableMappingStatus,
136}

154        /// Set this flags if a VMO is backed by physically contiguous memory

166/// The error type used for commit operations of [`Vmo`].
167#[derive(Debug)]
168pub enum VmoCommitError {
169    /// A general error occurred during the commit operation.
170    Err(Error),
171    /// The commit operation requires backend I/O for the page at `index`.
172    NeedIo { index: usize },
173    /// The page exists but is not yet initialized.
174    ///
175    /// The caller should wait for initialization of `page` at `index` to
176    /// complete.
177    WaitUntilInit { index: usize, page: CachePage },
178}
179
180#[derive(Debug, Clone, Copy, PartialEq, Eq)]
181enum CommitMode {
182    Read,
183    Overwrite,
184}
185
186impl CommitMode {
187    fn skips_backend_read(self) -> bool {
188        matches!(self, Self::Overwrite)
189    }
190}
191
192impl VmoCommitError {
193    /// Returns the page index whose commit is pending on I/O or initialization.
194    pub fn pending_index(&self) -> Result<usize> {
195        match self {
196            Self::NeedIo { index } | Self::WaitUntilInit { index, .. } => Ok(*index),
197            Self::Err(e) => Err(*e),
198        }
199    }
200}
201
202impl From<Error> for VmoCommitError {
203    fn from(e: Error) -> Self {
204        VmoCommitError::Err(e)
205    }
206}

208impl From<ostd::Error> for VmoCommitError {
209    fn from(e: ostd::Error) -> Self {
210        Error::from(e).into()
211    }
212}

214impl Vmo {
215    /// Commits the page at `page_idx`, blocking on backend I/O if required.
216    ///
217    /// For anonymous VMOs the page is zero-filled on first access.
218    /// For VMOs with a backend this may perform synchronous I/O.
219    pub fn commit_on(&self, page_idx: usize) -> Result<CachePage> {
220        self.commit_on_internal(page_idx, CommitMode::Read)
221            .map(CommittedPage::into_page)
222    }
223
224    fn commit_on_internal(
225        &self,
226        page_idx: usize,
227        commit_mode: CommitMode,
228    ) -> Result<CommittedPage> {
229        let mut locked_pages = self.pages.lock();
230        if page_idx >= self.size().div_ceil(PAGE_SIZE) {
231            return_errno_with_message!(Errno::EINVAL, "the offset is outside the VMO");
232        }
233
234        if let Some(backed_vmo) = self.as_backed_vmo() {
235            backed_vmo.commit_on(locked_pages, page_idx, commit_mode)
236        } else {
237            let mut cursor = locked_pages.cursor_mut(page_idx as u64);
238            if let Some(page) = cursor.load() {
239                return Ok(page.clone().into());
240            }
241
242            let new_page = CachePage::alloc_zero()?;
243            cursor.store(new_page.clone());
244
245            Ok(new_page.into())
246        }
247    }
248
249    /// Tries to commit the page covering byte `offset` without blocking on I/O.
250    ///
251    /// If the commit operation needs to perform I/O, it will return a
252    /// [`VmoCommitError::NeedIo`].
253    pub fn try_commit_page(&self, offset: usize) -> Result<CachePage, VmoCommitError> {
254        let page_idx = offset / PAGE_SIZE;
255        if offset >= self.size() {
256            return Err(VmoCommitError::Err(Error::with_message(
257                Errno::EINVAL,
258                "the offset is outside the VMO",
259            )));
260        }
261
262        let guard = disable_preempt();
263        let mut cursor = self.pages.cursor(&guard, page_idx as u64);
264        self.try_commit_with_cursor(&mut cursor, CommitMode::Read)
265            .map(|(_, page)| page.into_page())
266    }
267
268    fn try_commit_with_cursor(
269        &self,
270        cursor: &mut Cursor<'_, CachePage>,
271        commit_mode: CommitMode,
272    ) -> Result<(usize, CommittedPage), VmoCommitError> {
273        if let Some(backed_vmo) = self.as_backed_vmo() {
274            if let Some((index, page)) = backed_vmo.try_commit_with_cursor(cursor, commit_mode)? {
275                return Ok((index, page));
276            }
277        } else if let Some(page) = cursor.load() {
278            let index = cursor.index() as usize;
279            return Ok((index, page.clone().into()));
280        }
281
282        // Need to commit. Only Anonymous VMOs can reach here, because VMOs
283        // with a backend will return `Err` if the page is not loaded.
284        let index = cursor.index() as usize;
285        Ok((index, self.commit_on_internal(index, commit_mode)?))
286    }
287
288    /// Traverses the indices within a specified range of a VMO sequentially.
289    ///
290    /// For each index position, you have the option to commit the page as well as
291    /// perform other operations.
292    ///
293    /// Once a commit operation needs to perform I/O, it will return a
294    /// [`VmoCommitError::NeedIo`].
295    pub fn try_operate_on_range<F>(
296        &self,
297        range: &Range<usize>,
298        mut operate: F,
299    ) -> Result<(), VmoCommitError>
300    where
301        F: FnMut(
302            &mut dyn FnMut() -> Result<(usize, CachePage), VmoCommitError>,
303        ) -> Result<(), VmoCommitError>,
304    {
305        self.try_operate_on_range_internal(
306            range,
307            |commit_fn| {
308                let mut commit_page_fn =
309                    || commit_fn().map(|(index, page)| (index, page.into_page()));
310                operate(&mut commit_page_fn)
311            },
312            CommitMode::Read,
313        )
314    }
315
316    fn try_operate_on_range_internal<F>(
317        &self,
318        range: &Range<usize>,
319        mut operate: F,
320        commit_mode: CommitMode,
321    ) -> Result<(), VmoCommitError>
322    where
323        F: FnMut(
324            &mut dyn FnMut() -> Result<(usize, CommittedPage), VmoCommitError>,
325        ) -> Result<(), VmoCommitError>,
326    {
327        if range.end > self.size() {
328            return Err(VmoCommitError::Err(Error::with_message(
329                Errno::EINVAL,
330                "the range to operate exceeds the VMO size",
331            )));
332        }
333
334        let page_idx_range = get_page_idx_range(range);
335        let guard = disable_preempt();
336        let mut cursor = self.pages.cursor(&guard, page_idx_range.start as u64);
337        for _ in page_idx_range {
338            let mut commit_fn = || self.try_commit_with_cursor(&mut cursor, commit_mode);
339            operate(&mut commit_fn)?;
340            cursor.next();
341        }
342        Ok(())
343    }

345    /// Returns the current page-aligned size of the VMO in bytes.
346    ///
347    /// Higher layers may keep a smaller logical EOF alongside this capacity.
348    pub fn size(&self) -> usize {
349        self.size.load(Ordering::Acquire)
350    }

352    /// Returns the status of writable mappings of the VMO.
353    pub fn writable_mapping_status(&self) -> &WritableMappingStatus {
354        // Currently, only VMOs used by `MemfdInode` (anonymous) track writable mapping status.
355        // VMOs with a backend do not use this field.
356        debug_assert!(!self.has_backend());
357        &self.writable_mapping_status
358    }

360    /// Decommits anonymous pages in the specified byte range.
361    pub(super) fn decommit_anon_pages(
362        &self,
363        mut locked_pages: LockedXArray<CachePage>,
364        range: Range<usize>,
365    ) -> Result<()> {
366        debug_assert!(!self.has_backend());
367
368        let page_idx_range = get_page_idx_range(&range);
369        let mut cursor = locked_pages.cursor_mut(page_idx_range.start as u64);
370
371        loop {
372            cursor.remove();
373            let page_idx = cursor.next_present();
374            if page_idx.is_none_or(|idx| idx >= page_idx_range.end as u64) {
375                break;
376            }
377        }
378
379        Ok(())
380    }

382    /// Converts this VMO to a backend VMO wrapper if it has a backend.
383    ///
384    /// Returns `None` if this is an anonymous VMO.
385    pub(super) fn as_backed_vmo(&self) -> Option<BackedVmo<'_>> {
386        self.backend.as_ref().and_then(|weak_backend| {
387            weak_backend
388                .upgrade()
389                .map(|backend| BackedVmo { vmo: self, backend })
390        })
391    }

393    /// Returns whether this VMO has a backend.
394    fn has_backend(&self) -> bool {
395        self.backend.is_some()
396    }

399// Implement the read/write methods for `Vmo`.
400impl Vmo {
401    /// Maximum number of pages collected per batch from the `XArray`.
402    const PAGE_BATCH_CAPACITY: usize = 32;
403
404    /// Reads data from the VMO at `offset` into `writer`.
405    pub fn read(&self, offset: usize, writer: &mut VmWriter) -> Result<()> {
406        let read_len = writer.avail().min(self.size().saturating_sub(offset));
407        if read_len == 0 {
408            return Ok(());
409        }
410        let read_end = offset + read_len;
411
412        let range = offset..read_end;
413        let page_idx_range = get_page_idx_range(&range);
414        let mut current_idx = page_idx_range.start;
415        let mut page_offset = offset % PAGE_SIZE;
416        let mut page_batch =
417            Vec::with_capacity(min(page_idx_range.len(), Self::PAGE_BATCH_CAPACITY));
418
419        while current_idx < page_idx_range.end {
420            self.collect_pages(
421                current_idx,
422                page_idx_range.end,
423                CommitMode::Read,
424                &mut page_batch,
425            )?;
426
427            for (_, page) in &page_batch {
428                page.reader().skip(page_offset).read_fallible(writer)?;
429                page_offset = 0;
430            }
431
432            // `current_idx < page_idx_range.end` guarantees at least one page is successfully collected here.
433            current_idx = page_batch.last().unwrap().0 + 1;
434        }
435
436        Ok(())
437    }
438
439    /// Writes data from `reader` into the VMO at `offset`.
440    pub fn write(&self, offset: usize, reader: &mut VmReader) -> Result<()> {
441        let write_len = reader.remain().min(self.size().saturating_sub(offset));
442        if write_len == 0 {
443            return Ok(());
444        }
445        let write_end = offset + write_len;
446
447        let write_range = offset..write_end;
448        let mut page_offset = offset % PAGE_SIZE;
449        let mut page_batch = Vec::with_capacity(min(
450            write_range.len().div_ceil(PAGE_SIZE),
451            Self::PAGE_BATCH_CAPACITY,
452        ));
453
454        if !self.has_backend() {
455            return self.write_anonymous_pages(
456                &write_range,
457                &mut page_offset,
458                reader,
459                &mut page_batch,
460            );
461        }
462
463        // VMOs with a backend require dirty tracking and may skip backend reads
464        // for page-aligned ranges that will be entirely overwritten.
465        if write_range.len() < PAGE_SIZE {
466            return self.write_pages_with_backend(
467                &write_range,
468                &mut page_offset,
469                reader,
470                CommitMode::Read,
471                &mut page_batch,
472            );
473        }
474
475        // Split into head (unaligned), middle (aligned), and tail (unaligned).
476        let up_align_start = write_range.start.align_up(PAGE_SIZE);
477        let down_align_end = write_range.end.align_down(PAGE_SIZE);
478
479        if write_range.start != up_align_start {
480            let head = write_range.start..up_align_start;
481            self.write_pages_with_backend(
482                &head,
483                &mut page_offset,
484                reader,
485                CommitMode::Read,
486                &mut page_batch,
487            )?;
488        }
489        if up_align_start != down_align_end {
490            let mid = up_align_start..down_align_end;
491            self.write_pages_with_backend(
492                &mid,
493                &mut page_offset,
494                reader,
495                CommitMode::Overwrite,
496                &mut page_batch,
497            )?;
498        }
499        if down_align_end != write_range.end {
500            let tail = down_align_end..write_range.end;
501            self.write_pages_with_backend(
502                &tail,
503                &mut page_offset,
504                reader,
505                CommitMode::Read,
506                &mut page_batch,
507            )?;
508        }
509
510        Ok(())
511    }

513    /// Fills the specified range with zeros.
514    pub(super) fn fill_zeros(&self, range: Range<usize>) -> Result<()> {
515        if range.is_empty() {
516            return Ok(());
517        }
518        if range.end > self.size() {
519            return_errno_with_message!(Errno::EINVAL, "the range to fill exceeds the VMO size");
520        }
521
522        static ZERO_PAGE: [u8; PAGE_SIZE] = [0; PAGE_SIZE];
523
524        let mut current_offset = range.start;
525        while current_offset < range.end {
526            let page_remaining = PAGE_SIZE - current_offset % PAGE_SIZE;
527            let write_len = (range.end - current_offset).min(page_remaining);
528            let mut reader = VmReader::from(&ZERO_PAGE[..write_len]).to_fallible();
529
530            self.write(current_offset, &mut reader)?;
531            current_offset += write_len;
532        }
533
534        Ok(())
535    }

537    /// Writes data to pages without dirty tracking.
538    ///
539    /// Used for anonymous VMO writes where pages are always in the `UpToDate`
540    /// state.
541    fn write_anonymous_pages(
542        &self,
543        range: &Range<usize>,
544        page_offset: &mut usize,
545        reader: &mut VmReader,
546        page_batch: &mut Vec<(usize, CommittedPage)>,
547    ) -> Result<()> {
548        let page_idx_range = get_page_idx_range(range);
549        let mut current_idx = page_idx_range.start;
550
551        while current_idx < page_idx_range.end {
552            self.collect_pages(
553                current_idx,
554                page_idx_range.end,
555                CommitMode::Read,
556                page_batch,
557            )?;
558
559            for (_, page) in page_batch.iter() {
560                page.writer().skip(*page_offset).write_fallible(reader)?;
561                *page_offset = 0;
562            }
563
564            // `current_idx < page_idx_range.end` guarantees at least one page is successfully collected here.
565            current_idx = page_batch.last().unwrap().0 + 1;
566        }
567
568        Ok(())
569    }

571    /// Writes data to pages with backend dirty tracking.
572    ///
573    /// Each page is locked before writing to ensure correct state transitions
574    /// and marked dirty for later writeback.
575    fn write_pages_with_backend(
576        &self,
577        range: &Range<usize>,
578        page_offset: &mut usize,
579        reader: &mut VmReader,
580        commit_mode: CommitMode,
581        page_batch: &mut Vec<(usize, CommittedPage)>,
582    ) -> Result<()> {
583        let page_idx_range = get_page_idx_range(range);
584        let mut current_idx = page_idx_range.start;
585
586        while current_idx < page_idx_range.end {
587            self.collect_pages(current_idx, page_idx_range.end, commit_mode, page_batch)?;
588
589            // `current_idx < page_idx_range.end` guarantees at least one page is successfully collected here.
590            let next_idx = page_batch.last().unwrap().0 + 1;
591
592            for (_, page) in page_batch.drain(..) {
593                let locked_page = page.into_locked();
594                let written_size = match locked_page
595                    .writer()
596                    .skip(*page_offset)
597                    .write_fallible(reader)
598                {
599                    Ok(written_size) => written_size,
600                    Err((err, written_size)) => {
601                        if written_size > 0 {
602                            locked_page.set_dirty();
603                        }
604                        return Err(Error::from(err));
605                    }
606                };
607
608                if written_size > 0 {
609                    locked_page.set_dirty();
610                }
611
612                *page_offset = 0;
613            }
614
615            current_idx = next_idx;
616        }
617
618        Ok(())
619    }

621    /// Collects a batch of committed pages from `start_idx` to `end_idx`.
622    ///
623    /// Returns up to a small batch of consecutive committed pages. The
624    /// caller receives only successfully committed pages and never needs to
625    /// deal with [`VmoCommitError`] directly.
626    ///
627    /// This helper keeps the returned pages alive, but it does not pin them in
628    /// the `XArray`. Callers must still serialize against `evict_range()` /
629    /// `invalidate_range()` with a higher-level invalidation lock before they
630    /// take per-page locks, otherwise buffered writes may dirty a page that has
631    /// already been detached from the page cache.
632    fn collect_pages(
633        &self,
634        start_idx: usize,
635        end_idx: usize,
636        commit_mode: CommitMode,
637        pages: &mut Vec<(usize, CommittedPage)>,
638    ) -> Result<()> {
639        let end_idx = end_idx.min(start_idx + pages.capacity());
640        pages.clear();
641        let range = (start_idx * PAGE_SIZE)..(end_idx * PAGE_SIZE);
642        let mut current_range = range.clone();
643
644        'retry: loop {
645            let mut operate =
646                |commit_fn: &mut dyn FnMut() -> Result<(usize, CommittedPage), VmoCommitError>| {
647                    let (idx, page) = commit_fn()?;
648                    pages.push((idx, page));
649                    Ok(())
650                };
651            match self.try_operate_on_range_internal(&current_range, &mut operate, commit_mode) {
652                Ok(()) => break 'retry,
653                Err(err) => {
654                    let (idx, page) = self.handle_commit_error(err, commit_mode)?;
655                    pages.push((idx, page));
656                    current_range.start = (idx + 1) * PAGE_SIZE;
657                    if current_range.start >= range.end {
658                        break 'retry;
659                    }
660                }
661            }
662        }
663
664        Ok(())
665    }
666
667    /// Handles a commit error by performing the necessary I/O or initialization.
668    ///
669    /// Returns the page index and page that were recovered.
670    fn handle_commit_error(
671        &self,
672        err: VmoCommitError,
673        commit_mode: CommitMode,
674    ) -> Result<(usize, CommittedPage)> {
675        match err {
676            VmoCommitError::Err(e) => Err(e),
677            VmoCommitError::NeedIo { index } => {
678                Ok((index, self.commit_on_internal(index, commit_mode)?))
679            }
680            VmoCommitError::WaitUntilInit { index, page } => {
681                page.ensure_init(|locked_page| {
682                    self.as_backed_vmo()
683                        .unwrap()
684                        .backend
685                        .read_page(index, locked_page)
686                })?;
687                Ok((index, page.into()))
688            }
689        }
690    }
691}

693/// A wrapper around a VMO with a backend that provides specialized operations.
694///
695/// This structure is created by calling [`Vmo::as_backed_vmo()`] and provides
696/// access to backend-specific functionality like reading from storage and
697/// managing dirty pages.
698pub struct BackedVmo<'a> {
699    vmo: &'a Vmo,
700    backend: Arc<dyn PageCacheBackend>,
701}
702
703impl<'a> BackedVmo<'a> {
704    /// Writes back dirty pages in the specified byte range to the backend storage.
705    pub(super) fn flush_dirty_pages(&self, range: &Range<usize>) -> Result<()> {
706        let page_idx_range = get_page_idx_range(range);
707        let npages = self.backend.npages();
708        if page_idx_range.start >= npages {
709            return Ok(());
710        }
711        let page_idx_range = page_idx_range.start..page_idx_range.end.min(npages);
712
713        let locked_pages = self.vmo.pages.lock();
714        if range.start > self.size() {
715            return Ok(());
716        }
717
718        let dirty_pages =
719            self.collect_pages_if(locked_pages, page_idx_range, |_, page| page.is_dirty());
720
721        let mut io_batch = IoBatch::with_capacity(dirty_pages.len());
722        for (idx, page) in dirty_pages {
723            let locked_page = page.lock();
724            self.backend
725                .write_page_async(idx, locked_page, &mut io_batch)?;
726        }
727
728        io_batch.wait_all().map_err(Into::into)
729    }
730
731    /// Removes up-to-date (clean) pages in the specified byte range from the page cache.
732    ///
733    /// Only pages in the `UpToDate` state are removed. Dirty and uninitialized
734    /// pages are left in place.
735    //
736    // TODO: Returns `Err` if any up-to-date page has been mapped.
737    // TODO: Integrate reverse mappings or an explicit invalidation lock so this
738    // path can coordinate with page faults.
739    pub(super) fn evict_up_to_date_pages(&self, range: &Range<usize>) -> Result<()> {
740        if range.start > self.size() {
741            return Ok(());
742        }
743
744        let page_idx_range = get_page_idx_range(range);
745        let npages = self.backend.npages();
746        let locked_pages = self.vmo.pages.lock();
747        let pages_to_evict = self.collect_pages_if(locked_pages, page_idx_range, |idx, page| {
748            idx >= npages || page.is_up_to_date()
749        });
750        self.wait_for_writeback_and_remove_pages(pages_to_evict);
751
752        Ok(())
753    }
754
755    /// Decommits pages with a backend in the specified byte range.
756    ///
757    /// Dirty pages in the range are discarded without being written back. If a
758    /// page is already under writeback, decommit waits for that writeback to
759    /// finish before removing the page from the `XArray`.
760    pub(super) fn decommit_pages(
761        &self,
762        locked_pages: LockedXArray<CachePage>,
763        range: &Range<usize>,
764    ) -> Result<()> {
765        if range.start > self.size() {
766            return Ok(());
767        }
768
769        let page_idx_range = get_page_idx_range(range);
770        let pages_to_decommit = self.collect_pages_if(locked_pages, page_idx_range, |_, _| true);
771        self.wait_for_writeback_and_remove_pages(pages_to_decommit);
772
773        Ok(())
774    }
775
776    /// Commits a page at the given index for a VMO with a backend.
777    fn commit_on(
778        &self,
779        mut locked_pages: LockedXArray<'_, CachePage>,
780        page_idx: usize,
781        commit_mode: CommitMode,
782    ) -> Result<CommittedPage> {
783        let mut cursor = locked_pages.cursor_mut(page_idx as u64);
784        if let Some(page) = cursor.load() {
785            let page = page.clone();
786            drop(locked_pages);
787            if page_idx < self.backend.npages() && !commit_mode.skips_backend_read() {
788                page.ensure_init(|locked_page| self.backend.read_page(page_idx, locked_page))?;
789                return Ok(page.into());
790            }
791
792            return Ok(page.into());
793        };
794
795        // Page is within the file bounds - need to allocate a cache page.
796        let locked_page = CachePage::alloc_uninit()?
797            .try_lock()
798            .expect("newly allocated cache page must be unlocked");
799
800        cursor.store(locked_page.clone());
801
802        drop(locked_pages);
803
804        if page_idx < self.backend.npages() {
805            // Page is within the file bounds
806            if commit_mode.skips_backend_read() {
807                // Page will be completely overwritten, no need to read
808                Ok(locked_page.into())
809            } else {
810                let new_page = locked_page.clone();
811                // Read the page from backend storage
812                self.backend.read_page(page_idx, locked_page)?;
813                Ok(new_page.into())
814            }
815        } else {
816            // Page is beyond file bounds - treat as hole (zero-filled)
817            locked_page.fill_zeros(0, PAGE_SIZE).unwrap();
818            locked_page.set_up_to_date();
819
820            Ok(locked_page.unlock().into())
821        }
822    }
823
824    /// Attempts to commit a page using a cursor, without blocking on I/O.
825    fn try_commit_with_cursor(
826        &self,
827        cursor: &mut Cursor<'_, CachePage>,
828        commit_mode: CommitMode,
829    ) -> Result<Option<(usize, CommittedPage)>, VmoCommitError> {
830        let page_idx = cursor.index() as usize;
831
832        let Some(page) = cursor.load() else {
833            return Err(VmoCommitError::NeedIo { index: page_idx });
834        };
835
836        // If page is within file bounds, check if it's initialized
837        if !commit_mode.skips_backend_read() && page_idx < self.backend.npages() && page.is_uninit()
838        {
839            return Err(VmoCommitError::WaitUntilInit {
840                index: page_idx,
841                page: page.clone(),
842            });
843        }
844
845        Ok(Some((page_idx, page.clone().into())))
846    }
847
848    /// Collects pages in the specified page-index range that satisfy `should_collect`.
849    ///
850    /// The pages are only read (not removed) from the XArray.
851    fn collect_pages_if<F>(
852        &self,
853        locked_pages: LockedXArray<CachePage>,
854        page_idx_range: Range<usize>,
855        mut should_collect: F,
856    ) -> Vec<(usize, CachePage)>
857    where
858        F: FnMut(usize, &CachePage) -> bool,
859    {
860        if page_idx_range.is_empty() {
861            return Vec::new();
862        }
863
864        let mut collected_pages = Vec::new();
865
866        let mut cursor = locked_pages.cursor(page_idx_range.start as u64);
867        if let Some(page) = cursor.load()
868            && should_collect(page_idx_range.start, &page)
869        {
870            collected_pages.push((page_idx_range.start, page.clone()));
871        }
872
873        while let Some(page_idx) = cursor.next_present() {
874            let page_idx = page_idx as usize;
875            if page_idx >= page_idx_range.end {
876                break;
877            }
878
879            let page = cursor.load().unwrap();
880            if should_collect(page_idx, &page) {
881                collected_pages.push((page_idx, page.clone()));
882            }
883        }
884
885        collected_pages
886    }
887
888    /// Waits for writeback on collected pages, then removes them from the `XArray`.
889    ///
890    /// The pages are collected before this function is called so the `XArray`
891    /// lock can be released while waiting for in-flight writeback to finish.
892    /// Waiting before removal also prevents a later commit from installing a
893    /// new page for the same index and starting duplicate BIOs while the old
894    /// page is still under writeback.
895    fn wait_for_writeback_and_remove_pages(&self, pages_to_remove: Vec<(usize, CachePage)>) {
896        for (_, page) in &pages_to_remove {
897            let locked_page = page.lock_guard();
898            locked_page.wait_until_finish_writing_back();
899        }
900
901        let mut locked_pages = self.vmo.pages.lock();
902
903        for (page_idx, page) in pages_to_remove {
904            let mut cursor = locked_pages.cursor_mut(page_idx as u64);
905
906            // The caller will hold the higher-level lock that excludes concurrent
907            // removal, read/write and page fault handling, so the collected index
908            // still identifies the page that need to be deleted.
909            debug_assert!(
910                cursor
911                    .load()
912                    .is_some_and(|current_page| current_page.paddr() == page.paddr())
913            );
914            cursor.remove();
915        }
916    }
917}
918
919impl Deref for BackedVmo<'_> {
920    type Target = Vmo;
921
922    fn deref(&self) -> &Self::Target {
923        self.vmo
924    }
925}

927/// Gets the page index range that contains the offset range of VMO.
928pub fn get_page_idx_range(vmo_offset_range: &Range<usize>) -> Range<usize> {
929    let start = vmo_offset_range.start.align_down(PAGE_SIZE);
930    let end = vmo_offset_range.end.align_up(PAGE_SIZE);
931    (start / PAGE_SIZE)..(end / PAGE_SIZE)
932}

934/// The status of writable mappings of a VMO, i.e., shared mappings that may
935/// include the `PROT_WRITE` permission.
936///
937/// Internally, it uses an `AtomicIsize` counter with the following rules:
938///
939/// - **Positive values**: number of active writable mappings.
940/// - **Zero**: no writable mappings, and writable mappings are still allowed.
941/// - **Negative values**: writable mappings are denied.
942#[derive(Debug, Default)]
943pub struct WritableMappingStatus(AtomicIsize);
944
945impl WritableMappingStatus {
946    /// Builds a new writable mapping.
947    ///
948    /// Fails with `EPERM` if writable mappings have been denied.
949    pub fn map(&self) -> Result<()> {
950        // Increase unless negative
951        self.0
952            .fetch_update(Ordering::Relaxed, Ordering::Relaxed, |v| {
953                (v >= 0).then(|| v + 1)
954            })
955            .map_err(|_| Error::with_message(Errno::EPERM, "writable mappings have been denied"))?;
956        Ok(())
957    }
958
959    /// Denies any future writable mapping.
960    ///
961    /// Fails with `EBUSY` if there are still active writable mappings.
962    pub fn deny(&self) -> Result<()> {
963        // Decrease unless positive
964        self.0
965            .fetch_update(Ordering::Relaxed, Ordering::Relaxed, |v| {
966                (v <= 0).then_some(-1)
967            })
968            .map_err(|_| {
969                Error::with_message(Errno::EBUSY, "there are still active writable mappings")
970            })?;
971        Ok(())
972    }
973
974    /// Increments the writable mapping counter.
975    ///
976    /// Typically used when splitting an existing mapping, or forking a process.
977    pub fn increment(&self) {
978        self.0.fetch_add(1, Ordering::Relaxed);
979    }
980
981    /// Decrements the writable mapping counter.
982    ///
983    /// Typically used when unmapping a region, exiting a process, or merging mappings.
984    pub fn decrement(&self) {
985        self.0.fetch_sub(1, Ordering::Relaxed);
986    }
987}
