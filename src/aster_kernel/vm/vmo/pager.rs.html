3use ostd::mm::UFrame;
4
5use crate::prelude::*;
6
7/// Pagers provide frame to a VMO.
8///
9/// A `Pager` object can be attached to a VMO. Whenever the
10/// VMO needs more frames (i.e., on commits), it will turn to the pager,
11/// which should then provide frames whose data have been initialized properly.
12/// Any time a frame is updated through the VMO, the VMO will
13/// notify the attached pager that the frame has been updated.
14/// Finally, when a frame is no longer needed (i.e., on decommits),
15/// the frame pager will also be notified.
16pub trait Pager: Send + Sync {
17    /// Ask the pager to provide a frame at a specified index.
18    ///
19    /// After a page of a VMO is committed, the VMO shall not call this method
20    /// again until the page is decommitted. But a robust implementation of
21    /// `Pager` should not rely on this behavior for its correctness;
22    /// instead, it should returns the _same_ frame.
23    ///
24    /// If a VMO page has been previously committed and decommited,
25    /// and is to be committed again, then the pager is free to return
26    /// whatever frame that may or may not be the same as the last time.
27    ///
28    /// It is up to the pager to decide the range of valid indices.
29    fn commit_page(&self, idx: usize) -> Result<UFrame>;
30
31    /// Notify the pager that the frame at a specified index has been updated.
32    ///
33    /// Being aware of the updates allow the pager (e.g., an inode) to
34    /// know which pages are dirty and only write back the _dirty_ pages back
35    /// to disk.
36    ///
37    /// The VMO will not call this method for an uncommitted page.
38    /// But a robust implementation of `Pager` should not make
39    /// such an assumption for its correctness; instead, it should simply ignore the
40    /// call or return an error.
41    fn update_page(&self, idx: usize) -> Result<()>;
42
43    /// Notify the pager that the frame at the specified index has been decommitted.
44    ///
45    /// Knowing that a frame is no longer needed, the pager (e.g., an inode)
46    /// can free the frame after writing back its data to the disk.
47    ///
48    /// The VMO will not call this method for an uncommitted page.
49    /// But a robust implementation of `Pager` should not make
50    /// such an assumption for its correctness; instead, it should simply ignore the
51    /// call or return an error.
52    fn decommit_page(&self, idx: usize) -> Result<()>;
53
54    /// Ask the pager to provide a frame at a specified index.
55    /// Notify the pager that the frame will be fully overwritten soon, so pager can
56    /// choose not to initialize it.
57    fn commit_overwrite(&self, idx: usize) -> Result<UFrame>;
58}
