132impl From<&RawPageFaultInfo> for PageFaultInfo {
133    fn from(raw_info: &RawPageFaultInfo) -> Self {
134        let required_perms = if raw_info
135            .error_code
136            .contains(PageFaultErrorCode::INSTRUCTION)
137        {
138            VmPerms::EXEC
139        } else if raw_info.error_code.contains(PageFaultErrorCode::WRITE) {
140            // On x86_64, writable pages must also be readable.
141            // Reference: Section 5.11.3 from <https://www.intel.com/content/dam/www/public/us/en/documents/manuals/64-ia-32-architectures-software-developer-vol-3a-part-1-manual.pdf>.
142            VmPerms::READ | VmPerms::WRITE
143        } else {
144            VmPerms::READ
145        };
146
147        PageFaultInfo {
148            address: raw_info.addr,
149            required_perms,
150        }
151    }
152}

201/// A collection of CPU cores.
202///
203/// Note that hyperthreading can result in multiple logical processors within one core.
204static CPU_CORES: SpinLock<BTreeSet<u32>> = SpinLock::new(BTreeSet::new());

206/// CPU information to be shown in `/proc/cpuinfo`.
207///
208/// Different CPUs may have different information, such as the core ID. Therefore, [`Self::new`]
209/// should be called on every CPU.
210//
211// Implementation notes: For each field in this structure that is conditionally shown in Linux, it
212// should be wrapped in an `Option`. Please conduct the Linux implementation when adding a new
213// field, see <https://elixir.bootlin.com/linux/v6.16/source/arch/x86/kernel/cpu/proc.c#L63>.
214pub struct CpuInformation {
215    processor: u32,
216    vendor_id: String,
217    cpu_family: u8,
218    model: u8,
219    model_name: String,
220    stepping: Option<u8>,
221    cpu_khz: Option<u32>,
222    cache_size: Option<u32>,
223    core_id: u32,
224    apicid: u32,
225    cpuid_level: u32,
226    flags: Vec<&'static str>,
227    tlb_size: Option<u32>,
228    clflush_size: u32,
229    cache_alignment: u32,
230    address_sizes: (u32, u32),
231}

233impl CpuInformation {
234    /// Constructs the information for the current CPU.
235    pub fn new(guard: &DisabledPreemptGuard) -> Self {
236        let mut result = Self {
237            processor: guard.current_cpu().into(),
238            vendor_id: "unknown".to_owned(),
239            cpu_family: 0,
240            model: 0,
241            model_name: "unknown".to_owned(),
242            stepping: None,
243            // FIXME: The CPU frequency may not be equal to the TSC frequency. It may not even be a
244            // constant (i.e., the real CPU frequency can be adjusted due to the workload).
245            cpu_khz: Some((tsc_freq() / 1000) as u32),
246            cache_size: None,
247            core_id: 0,
248            apicid: 0,
249            cpuid_level: 0,
250            flags: Vec::new(),
251            tlb_size: None,
252            clflush_size: 64,
253            cache_alignment: 64,
254            address_sizes: (36, 48),
255        };
256
257        let vendor = result.fill_vendor_info();
258        result.fill_version_info(vendor);
259        result.fill_brand_info();
260        result.fill_cache_info(vendor);
261        result.fill_topology_info();
262        result.fill_feature_info();
263        result.fill_address_info();
264
265        result
266    }

268    fn fill_vendor_info(&mut self) -> CpuVendor {
269        let Some(CpuidResult { eax, ebx, ecx, edx }) = cpuid(CpuidLeaf::Base as u32, 0) else {
270            return CpuVendor::Unknown;
271        };
272
273        self.cpuid_level = eax;
274
275        let vendor = [ebx.to_le_bytes(), edx.to_le_bytes(), ecx.to_le_bytes()];
276        let Ok(vendor_str) = str::from_utf8(vendor.as_flattened()) else {
277            return CpuVendor::Unknown;
278        };
279        self.vendor_id = vendor_str.to_owned();
280
281        CpuVendor::parse(vendor_str)
282    }

284    fn fill_version_info(&mut self, vendor: CpuVendor) {
285        let Some(CpuidResult { eax, ebx, .. }) = cpuid(CpuidLeaf::Feature as u32, 0) else {
286            return;
287        };
288
289        let stepping = eax & 0xf;
290        let base_model = (eax >> 4) & 0xf;
291        let base_family = (eax >> 8) & 0xf;
292        let ext_model = (eax >> 16) & 0xf;
293        let ext_family = (eax >> 20) & 0xff;
294
295        // "The Extended Family ID needs to be examined only when the Family ID is 0FH."
296        let family = if base_family == 0xf {
297            base_family + ext_family
298        } else {
299            base_family
300        };
301
302        let has_ext_model = match vendor {
303            // The Intel manual says: "The Extended Model ID needs to be examined only when the
304            // Family ID is 06H or 0FH."
305            CpuVendor::Intel => base_family == 0x6 || base_family == 0xf,
306            // The AMD manual says: "If BaseFamily[3:0] is less than Fh, then ExtFamily is reserved
307            // and Family is equal to BaseFamily[3:0]."
308            CpuVendor::Amd | CpuVendor::Unknown => base_family == 0xf,
309        };
310        let model = if has_ext_model {
311            base_model | (ext_model << 4)
312        } else {
313            base_model
314        };
315
316        self.cpu_family = family as u8;
317        self.model = model as u8;
318        self.stepping = Some(stepping as u8);
319
320        // "Bits 15-08: CLFLUSH line size (Value âˆ— 8 = cache line size in bytes; used also by
321        // CLFLUSHOPT)."
322        self.clflush_size = ((ebx >> 8) & 0xff) * 8;
323        self.cache_alignment = self.clflush_size;
324        // Bits 31-24: Initial APIC ID.
325        self.apicid = ebx >> 24;
326        self.core_id = self.apicid;
327    }

329    fn fill_brand_info(&mut self) {
330        let Some(CpuidResult {
331            eax: eax1,
332            ebx: ebx1,
333            ecx: ecx1,
334            edx: edx1,
335        }) = cpuid(CpuidLeaf::ExtBrand1 as u32, 0)
336        else {
337            return;
338        };
339        let Some(CpuidResult {
340            eax: eax2,
341            ebx: ebx2,
342            ecx: ecx2,
343            edx: edx2,
344        }) = cpuid(CpuidLeaf::ExtBrand2 as u32, 0)
345        else {
346            return;
347        };
348        let Some(CpuidResult {
349            eax: eax3,
350            ebx: ebx3,
351            ecx: ecx3,
352            edx: edx3,
353        }) = cpuid(CpuidLeaf::ExtBrand3 as u32, 0)
354        else {
355            return;
356        };
357
358        #[rustfmt::skip]
359        let brand = [
360            eax1.to_le_bytes(), ebx1.to_le_bytes(), ecx1.to_le_bytes(), edx1.to_le_bytes(),
361            eax2.to_le_bytes(), ebx2.to_le_bytes(), ecx2.to_le_bytes(), edx2.to_le_bytes(),
362            eax3.to_le_bytes(), ebx3.to_le_bytes(), ecx3.to_le_bytes(), edx3.to_le_bytes(),
363        ];
364        let Ok(brand_cstr) = CStr::from_bytes_until_nul(brand.as_flattened()) else {
365            return;
366        };
367        let Ok(brand_str) = brand_cstr.to_str() else {
368            return;
369        };
370
371        self.model_name = brand_str.to_owned();
372    }

374    fn fill_cache_info(&mut self, vendor: CpuVendor) {
375        match vendor {
376            CpuVendor::Intel => self.fill_cache_info_intel(),
377            CpuVendor::Amd | CpuVendor::Unknown => self.fill_cache_info_amd(),
378        }

381    fn fill_cache_info_intel(&mut self) {
382        let mut l1size = 0;
383        let mut l2size = 0;
384        let mut l3size = 0;
385
386        for subleaf in 0.. {
387            let Some(CpuidResult { eax, ebx, ecx, .. }) = cpuid(CpuidLeaf::Cache as u32, subleaf)
388            else {
389                break;
390            };
391
392            // Bits 04-00, Cache Type Field: Null - No more caches.
393            if eax & 0x1f == 0 {
394                break;
395            }
396
397            let system_coherency_line_size = ebx & 0xfff;
398            let physical_line_partitions = (ebx >> 12) & 0x3ff;
399            let ways_of_associativity = (ebx >> 22) & 0x3ff;
400            let number_of_sets = ecx;
401
402            let size = (system_coherency_line_size + 1)
403                * (physical_line_partitions + 1)
404                * (ways_of_associativity + 1)
405                * (number_of_sets + 1);
406
407            // Bits 07-05, Cache Level (starts at 1).
408            match (eax >> 5) & 0x7 {
409                1 => l1size += size,
410                2 => l2size += size,
411                3 => l3size += size,
412                _ => (),
413            }
414        }
415
416        if l3size != 0 {
417            self.cache_size = Some(l3size / 1024);
418        } else if l2size != 0 {
419            self.cache_size = Some(l2size / 1024);
420        } else if l1size != 0 {
421            self.cache_size = Some(l1size / 1024);
422        }
423    }

425    fn fill_cache_info_amd(&mut self) {
426        let l1_cache_size =
427            if let Some(CpuidResult { ecx, edx, .. }) = cpuid(CpuidLeaf::ExtCacheL1 as u32, 0) {
428                // Bits 31-24, L1DcSize: L1 data cache size in KB.
429                let l1d_cache_size = ecx >> 24;
430                // Bits 31-24, L1IcSize: L1 instruction cache size KB.
431                let l1i_cache_size = edx >> 24;
432                l1d_cache_size + l1i_cache_size
433            } else {
434                0
435            };
436
437        let (l2_tlb_size, l2_cache_size) =
438            if let Some(CpuidResult { ebx, ecx, .. }) = cpuid(CpuidLeaf::ExtCacheL2L3 as u32, 0) {
439                // Bits 11-0, L2ITlb4KSize: L2 instruction TLB number of entries for 4-KB pages.
440                let l2i_tlb_size = ebx & 0xfff;
441                // Bits 27-16, L2DTlb4KSize: L2 data TLB number of entries for 4-KB pages.
442                let l2d_tlb_size = (ebx >> 16) & 0xfff;
443                // Bits 31-16, L2Size: L2 cache size in KB.
444                let l2_cache_size = ecx >> 16;
445                (l2i_tlb_size + l2d_tlb_size, l2_cache_size)
446            } else {
447                (0, 0)
448            };
449
450        // The Linux implementation here is quite odd, we just follow its logic:
451        //  - If the L3 cache is present, it will be ignored on AMD CPUs [1] but it will be reported as
452        //    the cache size on Intel CPUs [2].
453        //  - The TLB size is never counted on Intel CPUs.
454        //  - The L1 TLB is never counted on 64-bit AMD CPUs. But the L2 TLB will be counted.
455        //
456        // [1]: https://elixir.bootlin.com/linux/v6.16/source/arch/x86/kernel/cpu/common.c#L811
457        // [2]: https://elixir.bootlin.com/linux/v6.16/source/arch/x86/kernel/cpu/cacheinfo.c#L359
458        if l2_tlb_size != 0 {
459            self.tlb_size = Some(l2_tlb_size);
460        }
461        if l2_cache_size != 0 {
462            self.cache_size = Some(l2_cache_size);
463        } else if l1_cache_size != 0 {
464            self.cache_size = Some(l1_cache_size);
465        }
466    }

468    fn fill_topology_info(&mut self) {
469        let Some(CpuidResult { eax, edx, .. }) = cpuid(CpuidLeaf::Topology as u32, 0) else {
470            return;
471        };
472
473        // "Bits 31-00: x2APIC ID of the current logical processor."
474        self.apicid = edx;
475        // "Bits 04-00: The number of bits that the x2APIC ID must be shifted to the right to
476        // address instances of the next higher-scoped domain."
477        self.core_id = self.apicid >> (eax & 0x1f);
478
479        CPU_CORES.lock().insert(self.core_id);
480    }

788    fn fill_address_info(&mut self) {
789        let Some(CpuidResult { eax, .. }) = cpuid(CpuidLeaf::ExtAddrSizes as u32, 0) else {
790            return;
791        };
792
793        // "Bits 07-00: #Physical Address Bits*."
794        self.address_sizes.0 = eax & 0xff;
795        // "Bits 15-08: #Linear Address Bits."
796        self.address_sizes.1 = (eax >> 8) & 0xff;
797    }

800impl fmt::Display for CpuInformation {
801    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
802        write!(
803            f,
804            "processor\t: {}\n\
805             vendor_id\t: {}\n\
806             cpu family\t: {}\n\
807             model\t\t: {}\n\
808             model name\t: {}\n",
809            self.processor, self.vendor_id, self.cpu_family, self.model, self.model_name,
810        )?;
811
812        if let Some(stepping) = self.stepping {
813            writeln!(f, "stepping\t: {}", stepping)?;
814        } else {
815            writeln!(f, "stepping\t: unknown")?;
816        }
817
818        // TODO: Add the `microcode` field.
819
820        if let Some(cpu_khz) = self.cpu_khz {
821            writeln!(f, "cpu MHz\t\t: {}.{:03}", cpu_khz / 1000, cpu_khz % 1000)?;
822        } else {
823            writeln!(f, "cpu MHz\t\t: Unknown")?;
824        }
825
826        if let Some(cache_size) = self.cache_size {
827            writeln!(f, "cache size\t: {} KB", cache_size)?;
828        }
829
830        // Note that we don't support NUMA now, so we assume that all CPUs are on the same package
831        // (i.e., their physical IDs are all zeros).
832        let siblings = num_cpus();
833        let cpu_cores = CPU_CORES.lock().len();
834        write!(
835            f,
836            "physical id\t: 0\n\
837             siblings\t: {}\n\
838             core id\t\t: {}\n\
839             cpu cores\t: {}\n\
840             apicid\t\t: {}\n\
841             initial apicid\t: {}\n",
842            siblings, self.core_id, cpu_cores, self.apicid, self.apicid
843        )?;
844
845        write!(
846            f,
847            "fpu\t\t: yes\n\
848             fpu_exception\t: yes\n\
849             cpuid level\t: {}\n\
850             wp\t\t: yes\n",
851            self.cpuid_level,
852        )?;
853
854        write!(f, "flags\t\t:")?;
855        for flag in self.flags.iter() {
856            write!(f, " {}", flag)?;
857        }
858
859        // TODO: Add the `vmx flags` field.
860
861        // TODO: Fill the `bugs` field.
862        write!(
863            f,
864            "\n\
865             bugs\t\t:\n"
866        )?;
867
868        // TODO: Add the `bogomips` field.
869
870        if let Some(tlb_size) = self.tlb_size {
871            writeln!(f, "TLB size\t: {} 4K pages", tlb_size)?;
872        }
873
874        write!(
875            f,
876            "clflush size\t: {}\n\
877             cache_alignment\t: {}\n\
878             address sizes\t: {} bits physical, {} bits virtual\n",
879            self.clflush_size, self.cache_alignment, self.address_sizes.0, self.address_sizes.1,
880        )?;
881
882        // TODO: Fill the `power management` field.
883        writeln!(f, "power management:")?;
884
885        Ok(())
886    }
