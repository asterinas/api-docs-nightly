156/// error used in this crate
157#[derive(Debug, Clone, Copy)]
158pub struct Error {
159    errno: Errno,
160    #[expect(dead_code)]
161    msg: Option<&'static str>,
162}
163
164impl Error {
165    pub const fn new(errno: Errno) -> Self {
166        Error { errno, msg: None }
167    }
168
169    pub const fn with_message(errno: Errno, msg: &'static str) -> Self {
170        Error {
171            errno,
172            msg: Some(msg),
173        }
174    }
175
176    pub const fn error(&self) -> Errno {
177        self.errno
178    }
179}
180
181impl From<Errno> for Error {
182    fn from(errno: Errno) -> Self {
183        Error::new(errno)
184    }
185}
186
187impl AsRef<Error> for Error {
188    fn as_ref(&self) -> &Error {
189        self
190    }
191}

193impl From<ostd::Error> for Error {
194    fn from(ostd_error: ostd::Error) -> Self {
195        match ostd_error {
196            ostd::Error::AccessDenied => Error::new(Errno::EFAULT),
197            ostd::Error::NoMemory => Error::new(Errno::ENOMEM),
198            ostd::Error::InvalidArgs => Error::new(Errno::EINVAL),
199            ostd::Error::IoError => Error::new(Errno::EIO),
200            ostd::Error::NotEnoughResources => Error::new(Errno::EBUSY),
201            ostd::Error::PageFault => Error::new(Errno::EFAULT),
202            ostd::Error::Overflow => Error::new(Errno::EOVERFLOW),
203        }
204    }
205}

207impl From<(ostd::Error, usize)> for Error {
208    // Used in fallible memory read/write API
209    fn from(ostd_error: (ostd::Error, usize)) -> Self {
210        Error::from(ostd_error.0)
211    }
212}

221impl From<aster_block::bio::BioEnqueueError> for Error {
222    fn from(error: aster_block::bio::BioEnqueueError) -> Self {
223        match error {
224            aster_block::bio::BioEnqueueError::IsFull => {
225                Error::with_message(Errno::EBUSY, "The request queue is full")
226            }
227            aster_block::bio::BioEnqueueError::Refused => {
228                Error::with_message(Errno::EBUSY, "Refuse to enqueue the bio")
229            }
230            aster_block::bio::BioEnqueueError::TooBig => {
231                Error::with_message(Errno::EINVAL, "Bio is too big")
232            }
233        }
234    }
235}

237impl From<aster_block::bio::BioStatus> for Error {
238    fn from(err_status: aster_block::bio::BioStatus) -> Self {
239        match err_status {
240            aster_block::bio::BioStatus::NotSupported => {
241                Error::with_message(Errno::EIO, "I/O operation is not supported")
242            }
243            aster_block::bio::BioStatus::NoSpace => {
244                Error::with_message(Errno::ENOSPC, "Insufficient space on device")
245            }
246            aster_block::bio::BioStatus::IoError => {
247                Error::with_message(Errno::EIO, "I/O operation fails")
248            }
249            status => panic!("Can not convert the status: {:?} to an error", status),
250        }
251    }
252}

304            }

306                Error::with_message(Errno::EIO, "CPIO buffer I/O error")
307            }
308        }

312impl From<Error> for ostd::Error {
313    fn from(error: Error) -> Self {
314        match error.errno {
315            Errno::EACCES => ostd::Error::AccessDenied,
316            Errno::EIO => ostd::Error::IoError,
317            Errno::ENOMEM => ostd::Error::NoMemory,
318            Errno::EFAULT => ostd::Error::PageFault,
319            Errno::EINVAL => ostd::Error::InvalidArgs,
320            Errno::EBUSY => ostd::Error::NotEnoughResources,
321            _ => ostd::Error::InvalidArgs,
322        }
323    }
324}

338impl From<aster_systree::Error> for Error {
339    fn from(err: aster_systree::Error) -> Self {
340        use aster_systree::Error::*;
341        match err {
342            NotFound => Error::new(Errno::ENOENT),
343            InvalidOperation => Error::new(Errno::EINVAL),
344            ResourceUnavailable => Error::new(Errno::EBUSY),
345            AttributeError => Error::new(Errno::EIO),
346            PermissionDenied => Error::new(Errno::EACCES),
347            InternalError(msg) => Error::with_message(Errno::EIO, msg),
348            AlreadyExists => Error::new(Errno::EEXIST),
349            Overflow => Error::new(Errno::EOVERFLOW),
350            PageFault => Error::new(Errno::EFAULT),
351            IsDead => Error::new(Errno::ENODEV),
352        }
353    }
354}
