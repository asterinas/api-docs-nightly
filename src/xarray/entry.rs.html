10use crate::node::XNode;

12/// A type serving as the basic unit of storage for `XArray`s, used in the head of the `XArray` and
13/// the slots of `XNode`s.
14///
15/// There are the following types of `XEntry`:
16/// - Internal entries: These are invisible to users. Currently these entries represent pointers to
17///   `XNode`s (`Arc<XNode<P>>`).
18/// - Item entries: These represent user-given items of type `P`.
19///
20/// An `XEntry` owns the item or node that it represents. Once an `XEntry` generated from an item
21/// or an `XNode`, the ownership of the item or the `XNode` will be transferred to the `XEntry`.
22pub(super) type XEntry<P> = Either<Arc<XNode<P>>, P>;
23
24/// A type that represents the reference to `XEntry`.
25pub(super) type XEntryRef<'a, P> = Either<ArcRef<'a, XNode<P>>, <P as NonNullPtr>::Ref<'a>>;
26
27/// A type that represents the interior entries in `XArray`.
28pub(super) type NodeEntry<P> = Arc<XNode<P>>;
29
30/// A type that represents the reference to `NodeEntry`.
31pub(super) type NodeEntryRef<'a, P> = ArcRef<'a, XNode<P>>;
