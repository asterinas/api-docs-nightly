21/// The height of an `XNode` within an `XArray`.
22///
23/// In an `XArray`, the head has the highest height, while the `XNode`s that
24/// directly store items are at the lowest height, with a height value of 1.
25/// Each level up from the bottom height increases the height number by 1.
26/// The height of an `XArray` is the height of its head.
27#[derive(Debug, PartialEq, Eq, PartialOrd, Ord, Copy, Clone)]
28pub(super) struct Height {
29    height: u8,
30}
31
32impl Deref for Height {
33    type Target = u8;
34
35    fn deref(&self) -> &Self::Target {
36        &self.height
37    }
38}
39
40impl DerefMut for Height {
41    fn deref_mut(&mut self) -> &mut Self::Target {
42        &mut self.height
43    }
44}
45
46impl PartialEq<u8> for Height {
47    fn eq(&self, other: &u8) -> bool {
48        self.height == *other
49    }
50}
51
52impl PartialOrd<u8> for Height {
53    fn partial_cmp(&self, other: &u8) -> Option<Ordering> {
54        self.height.partial_cmp(other)
55    }
56}
57
58impl Height {
59    /// Creates a `Height` directly from a height value.
60    pub(super) fn new(height: u8) -> Self {
61        Self { height }
62    }
63
64    /// Creates a minimal `Height` that allows the `index`-th item to be stored.
65    pub(super) fn from_index(index: u64) -> Self {
66        let mut height = Height::new(1);
67        while index > height.max_index() {
68            *height += 1;
69        }
70        height
71    }
72
73    /// Goes up, which increases the height value by one.
74    pub(super) fn go_root(&self) -> Self {
75        Self::new(self.height + 1)
76    }
77
78    /// Goes down, which decreases the height value by one.
79    pub(super) fn go_leaf(&self) -> Self {
80        Self::new(self.height - 1)
81    }
82
83    fn height_shift(&self) -> u8 {
84        (self.height - 1) * BITS_PER_LAYER as u8
85    }
86
87    /// Calculates the corresponding offset for the target index at
88    /// the current height.
89    pub(super) fn height_offset(&self, index: u64) -> u8 {
90        ((index >> self.height_shift()) & SLOT_MASK as u64) as u8
91    }
92
93    /// Calculates the maximum index that can be represented in an `XArray`
94    /// with the current height.
95    pub(super) fn max_index(&self) -> u64 {
96        ((SLOT_SIZE as u64) << self.height_shift()) - 1
97    }
98
99    /// Calculates the index step representing one offset at the current height.
100    pub(super) fn index_step(&self) -> u64 {
101        1 << self.height_shift()
102    }
103}

105/// The `XNode` is the intermediate node in the tree-like structure of the `XArray`.
106///
107/// It contains `SLOT_SIZE` number of `XEntry`s, meaning it can accommodate up to
108/// `SLOT_SIZE` child nodes. The `height` and `offset_in_parent` attributes of an
109/// `XNode` are determined at initialization and remain unchanged thereafter.
110pub(super) struct XNode<P>
111where
112    P: NonNullPtr + Send + Sync,
113{
114    /// The pointer that refers to the parent node.
115    ///
116    /// If the current node is the head node, its parent pointer will be `None`.
117    parent: RcuOption<NodeEntry<P>>,
118    /// The height of the subtree rooted at the current node.
119    ///
120    /// The height of a leaf node, which stores the user-given items, is 1.
121    height: Height,
122    /// This node is its parent's `offset_in_parent`-th child.
123    ///
124    /// This field will be zero if this node is the root, as the node will be
125    /// the 0-th child of its parent once the height of `XArray` is increased.
126    offset_in_parent: u8,
127    /// The slots in which `XEntry`s are stored.
128    ///
129    /// The entries point to user-given items for leaf nodes and other `XNode`s for
130    /// interior nodes.
131    slots: [RcuOption<XEntry<P>>; SLOT_SIZE],
132    /// The marks representing whether each slot is marked or not.
133    ///
134    /// Users can set mark or unset mark on user-given items, and a leaf
135    /// node or an interior node is marked if and only if there is at least
136    /// one marked item within the node.
137    marks: [Mark; NUM_MARKS],
138}
139
140impl<P: NonNullPtr + Send + Sync> XNode<P> {
141    pub(super) fn new_root(height: Height) -> Self {
142        Self::new(height, 0)
143    }
144
145    pub(super) fn new(height: Height, offset: u8) -> Self {
146        Self {
147            parent: RcuOption::new_none(),
148            height,
149            offset_in_parent: offset,
150            slots: [const { RcuOption::new_none() }; SLOT_SIZE],
151            marks: [const { Mark::new_empty() }; NUM_MARKS],
152        }
153    }
154
155    /// Gets the slot offset at the current `XNode` for the target index `target_index`.
156    pub(super) fn entry_offset(&self, target_index: u64) -> u8 {
157        self.height.height_offset(target_index)
158    }
159
160    pub(super) fn height(&self) -> Height {
161        self.height
162    }
163
164    pub(super) fn parent<'a>(&'a self, guard: &'a dyn InAtomicMode) -> Option<NodeEntryRef<'a, P>> {
165        let parent = self.parent.read_with(guard)?;
166        Some(parent)
167    }
168
169    pub(super) fn offset_in_parent(&self) -> u8 {
170        self.offset_in_parent
171    }
172
173    pub(super) fn entry_with<'a>(
174        &'a self,
175        guard: &'a dyn InAtomicMode,
176        offset: u8,
177    ) -> Option<XEntryRef<'a, P>> {
178        self.slots[offset as usize].read_with(guard)
179    }
180
181    pub(super) fn is_marked(&self, offset: u8, mark: usize) -> bool {
182        self.marks[mark].is_marked(offset)
183    }
184
185    pub(super) fn is_mark_clear(&self, mark: usize) -> bool {
186        self.marks[mark].is_clear()
187    }
188
189    pub(super) fn is_leaf(&self) -> bool {
190        self.height == 1
191    }
192
193    pub(super) fn next_marked(&self, offset: u8, mark: usize) -> Option<u8> {
194        self.marks[mark].next_marked(offset)
195    }
196}

198impl<P: NonNullPtr + Send + Sync> XNode<P> {
199    /// Sets the parent pointer of this node to the given `parent`.
200    fn set_parent(&self, _guard: XLockGuard, parent: NodeEntry<P>) {
201        self.parent.update(Some(parent));
202    }
203
204    /// Clears the parent pointers of this node and all its descendant nodes.
205    ///
206    /// This method should be invoked when the node is being removed from the tree.
207    pub(super) fn clear_parent(&self, guard: XLockGuard) {
208        self.parent.update(None);
209        for child in self.slots.iter() {
210            if let Some(node) = child.read_with(guard.0).and_then(|entry| entry.left()) {
211                node.clear_parent(guard);
212            }
213        }
214    }
215
216    /// Sets the slot at the given `offset` to the given `entry`.
217    ///
218    /// If `entry` represents an item, the old marks at the same offset will be cleared.
219    /// Otherwise, if `entry` represents a node, the marks at the same offset will be
220    /// updated according to whether the new node contains marked items.
221    ///
222    /// This method will also propagate the updated marks to the ancestors.
223    pub(super) fn set_entry(
224        self: &Arc<Self>,
225        guard: XLockGuard,
226        offset: u8,
227        entry: Option<XEntry<P>>,
228    ) {
229        let old_entry = self.slots[offset as usize].read_with(guard.0);
230        if let Some(node) = old_entry.and_then(|entry| entry.left()) {
231            node.clear_parent(guard);
232        }
233
234        let is_new_node = match &entry {
235            Some(Either::Left(new_node)) => {
236                new_node.set_parent(guard, self.clone());
237                true
238            }
239            _ => false,
240        };
241        let is_new_item = matches!(&entry, Some(Either::Right(_)));
242
243        self.slots[offset as usize].update(entry);
244
245        if is_new_node {
246            self.update_mark(guard, offset);
247        } else {
248            for i in 0..NUM_MARKS {
249                if i == PRESENT_MARK && is_new_item {
250                    self.set_mark(guard, offset, i);
251                } else {
252                    self.unset_mark(guard, offset, i);
253                }
254            }
255        }
256    }
257
258    /// Sets the input `mark` at the given `offset`.
259    ///
260    /// This method will also update the marks on the ancestors of this node
261    /// if necessary to ensure that the marks on the ancestors are up to date.
262    pub(super) fn set_mark(&self, guard: XLockGuard, offset: u8, mark: usize) {
263        let changed = self.marks[mark].update(guard, offset, true);
264        if changed {
265            self.propagate_mark(guard, mark);
266        }
267    }
268
269    /// Unsets the input `mark` at the given `offset`.
270    ///
271    /// This method will also update the marks on the ancestors of this node
272    /// if necessary to ensure that the marks on the ancestors are up to date.
273    pub(super) fn unset_mark(&self, guard: XLockGuard, offset: u8, mark: usize) {
274        let changed = self.marks[mark].update(guard, offset, false);
275        if changed {
276            self.propagate_mark(guard, mark);
277        }
278    }
279
280    /// Updates the mark at the given `offset`.
281    ///
282    /// This method does nothing if the slot at the given `offset` does not represent
283    /// a node. It assumes the marks of the child node are up to date, and ensures
284    /// the mark at the given `offset` is also up to date.
285    ///
286    /// This method will also update the marks on the ancestors of this node
287    /// if necessary to ensure that the marks on the ancestors are up to date.
288    fn update_mark(&self, guard: XLockGuard, offset: u8) {
289        let entry = self.slots[offset as usize].read_with(guard.0);
290        let Some(node) = entry.and_then(|entry| entry.left()) else {
291            return;
292        };
293
294        for i in 0..NUM_MARKS {
295            let changed = self.marks[i].update(guard, offset, !node.is_mark_clear(i));
296            if changed {
297                self.propagate_mark(guard, i);
298            }
299        }
300    }
301
302    /// Propagates the mark updates on this node to the ancestors.
303    ///
304    /// This method must be called after the marks are updated to ensure that the marks on the
305    /// ancestors are up to date.
306    fn propagate_mark(&self, guard: XLockGuard, mark: usize) {
307        let Some(parent) = self.parent(guard.0) else {
308            return;
309        };
310
311        let changed =
312            parent.marks[mark].update(guard, self.offset_in_parent, !self.is_mark_clear(mark));
313        if changed {
314            parent.propagate_mark(guard, mark);
315        }
316    }
