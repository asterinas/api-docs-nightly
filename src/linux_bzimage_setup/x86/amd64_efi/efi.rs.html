15/// SAFETY: The name does not collide with other symbols.
16#[unsafe(no_mangle)]
17unsafe extern "sysv64" fn main_efi_common64(
18    handle: Handle,
19    system_table: *const SystemTable,
20    boot_params_ptr: *mut BootParams,
21) -> ! {
22    // SAFETY: We get `handle` and `system_table` from the UEFI firmware, so by contract the
23    // pointers are valid and correct.
24    unsafe {
25        boot::set_image_handle(handle);
26        uefi::table::set_system_table(system_table);
27    }
28
29    uefi::helpers::init().unwrap();
30
31    let boot_params = if boot_params_ptr.is_null() {
32        allocate_boot_params()
33    } else {
34        // SAFETY: We get boot parameters from the boot loader, so by contract the pointer is valid and
35        // the underlying memory is initialized. We are an exclusive owner of the memory region, so we
36        // can create a mutable reference of the plain-old-data type.
37        unsafe { &mut *boot_params_ptr }
38    };
39
40    efi_phase_boot(boot_params);
41
42    // SAFETY: All previously opened boot service protocols have been closed. At this time, we have
43    // no references to the code and data of the boot services.
44    unsafe { efi_phase_runtime(boot_params) };
45}

47fn allocate_boot_params() -> &'static mut BootParams {
48    let boot_params = {
49        let bytes = alloc_pages(AllocateType::AnyPages, size_of::<BootParams>());
50        bytes.write_filled(0);
51        // SAFETY: Zero initialization gives a valid representation for `BootParams`.
52        unsafe { &mut *bytes.as_mut_ptr().cast::<BootParams>() }
53    };
54
55    boot_params.hdr.header = linux_boot_params::LINUX_BOOT_HEADER_MAGIC;
56
57    boot_params
58}

60fn efi_phase_boot(boot_params: &mut BootParams) {
61    uefi::println!(
62        "[EFI stub] Loaded with offset {:#x}",
63        crate::x86::image_load_offset(),
64    );
65
66    // Load the command line if it is not loaded.
67    if boot_params.hdr.cmd_line_ptr == 0
68        && boot_params.ext_cmd_line_ptr == 0
69        && let Some(cmdline) = load_cmdline()
70    {
71        boot_params.hdr.cmd_line_ptr = cmdline.as_ptr().addr().try_into().unwrap();
72        boot_params.ext_cmd_line_ptr = 0;
73        boot_params.hdr.cmdline_size = (cmdline.count_bytes() + 1).try_into().unwrap();
74    }
75
76    // Load the init ramdisk if it is not loaded.
77    if boot_params.hdr.ramdisk_image == 0
78        && boot_params.ext_ramdisk_image == 0
79        && let Some(initrd) = load_initrd()
80    {
81        boot_params.hdr.ramdisk_image = initrd.as_ptr().addr().try_into().unwrap();
82        boot_params.ext_ramdisk_image = 0;
83        boot_params.hdr.ramdisk_size = initrd.len().try_into().unwrap();
84        boot_params.ext_ramdisk_size = 0;
85    }
86
87    // Fill the boot params with the RSDP address if it is not provided.
88    if boot_params.acpi_rsdp_addr == 0 {
89        boot_params.acpi_rsdp_addr =
90            find_rsdp_addr().expect("ACPI RSDP address is not available") as usize as u64;
91    }
92
93    // Fill the boot params with the screen info if it is not provided.
94    if boot_params.screen_info.lfb_base == 0 && boot_params.screen_info.ext_lfb_base == 0 {
95        fill_screen_info(&mut boot_params.screen_info);
96    }
97
98    // Decode the payload and load it as an ELF file.
99    uefi::println!("[EFI stub] Decoding the kernel payload");
100    let kernel = decode_payload(crate::x86::payload());
101    uefi::println!("[EFI stub] Loading the payload as an ELF file");
102    crate::loader::load_elf(&kernel);
103}

105fn load_cmdline() -> Option<&'static CStr> {
106    uefi::println!("[EFI stub] Loading the cmdline");
107
108    let loaded_image = open_protocol_exclusive::<uefi::proto::loaded_image::LoadedImage>(
109        uefi::boot::image_handle(),
110    )
111    .unwrap();
112
113    let Some(load_options) = loaded_image.load_options_as_bytes() else {
114        uefi::println!("[EFI stub] Warning: No cmdline is available!");
115        return None;
116    };
117
118    if !load_options.len().is_multiple_of(2)
119        || load_options.iter().skip(1).step_by(2).any(|c| *c != 0)
120    {
121        uefi::println!("[EFI stub] Warning: The cmdline contains non-ASCII characters!");
122        return None;
123    }
124
125    // The load options are a `Char16` sequence. We should convert it to a `Char8` sequence.
126    let cmdline_bytes = alloc_pages(
127        AllocateType::MaxAddress(u32::MAX as u64),
128        load_options.len() / 2 + 1,
129    );
130    for i in 0..load_options.len() / 2 {
131        cmdline_bytes[i].write(load_options[i * 2]);
132    }
133    cmdline_bytes[load_options.len() / 2].write(0);
134
135    // SAFETY: We've initialized all the bytes above.
136    let cmdline_str =
137        CStr::from_bytes_until_nul(unsafe { cmdline_bytes.assume_init_ref() }).unwrap();
138
139    uefi::println!("[EFI stub] Loaded the cmdline: {:?}", cmdline_str);
140
141    Some(cmdline_str)
142}

144// Linux loads the initrd either using a special protocol `LINUX_EFI_INITRD_MEDIA_GUID` or using
145// the file path specified on the command line (e.g., `initrd=/initrd.img`). We now only support
146// the former approach, as it is more "modern" and easier to implement. Note that this approach
147// requires the boot loader (e.g., GRUB, systemd-boot) to implement the protocol, while the latter
148// approach does not.
149fn load_initrd() -> Option<&'static [u8]> {
150    uefi::println!("[EFI stub] Loading the initrd");
151
152    // Note that we should switch to `uefi::proto::media::load_file::LoadFile2` once it provides a
153    // more ergonomic API. Its current API requires `alloc` and cannot load files on pages (i.e.,
154    // ensure that the initrd is aligned to the page size).
155    #[repr(transparent)]
156    // SAFETY: The protocol GUID matches the protocol itself.
157    #[uefi::proto::unsafe_protocol(uefi_raw::protocol::media::LoadFile2Protocol::GUID)]
158    struct LoadFile2(uefi_raw::protocol::media::LoadFile2Protocol);
159
160    let mut device_path_buf = [MaybeUninit::uninit(); 20 /* vendor */ + 4 /* end */];
161    let mut device_path = {
162        use uefi::proto::device_path::build;
163        build::DevicePathBuilder::with_buf(&mut device_path_buf)
164            .push(&build::media::Vendor {
165                // LINUX_EFI_INITRD_MEDIA_GUID
166                vendor_guid: uefi::guid!("5568e427-68fc-4f3d-ac74-ca555231cc68"),
167                vendor_defined_data: &[],
168            })
169            .unwrap()
170            .finalize()
171            .unwrap()
172    };
173
174    let Ok(handle) = uefi::boot::locate_device_path::<LoadFile2>(&mut device_path) else {
175        uefi::println!("[EFI stub] Warning: Failed to locate the initrd device!");
176        return None;
177    };
178
179    let Ok(mut load_file2) = uefi::boot::open_protocol_exclusive::<LoadFile2>(handle) else {
180        uefi::println!("[EFI stub] Warning: Failed to open the initrd protocol!");
181        return None;
182    };
183
184    let mut size = 0;
185    // SAFETY: The arguments are correctly specified according to the UEFI specification.
186    let status = unsafe {
187        (load_file2.0.load_file)(
188            &mut load_file2.0,
189            device_path.as_ffi_ptr().cast(),
190            false, /* boot_policy */
191            &mut size,
192            core::ptr::null_mut(),
193        )
194    };
195    if status != uefi::Status::BUFFER_TOO_SMALL {
196        uefi::println!("[EFI stub] Warning: Failed to get the initrd size!");
197        return None;
198    }
199
200    let initrd = alloc_pages(AllocateType::MaxAddress(u32::MAX as u64), size);
201    // SAFETY: The arguments are correctly specified according to the UEFI specification.
202    let status = unsafe {
203        (load_file2.0.load_file)(
204            &mut load_file2.0,
205            device_path.as_ffi_ptr().cast(),
206            false, /* boot_policy */
207            &mut size,
208            initrd.as_mut_ptr().cast(),
209        )
210    };
211    if status.is_error() {
212        uefi::println!("[EFI stub] Warning: Failed to load the initrd!");
213        return None;
214    }
215    assert_eq!(
216        size,
217        initrd.len(),
218        "the initrd size has changed between two EFI calls"
219    );
220
221    uefi::println!(
222        "[EFI stub] Loaded the initrd: addr={:#x}, size={:#x}",
223        initrd.as_ptr().addr(),
224        initrd.len()
225    );
226
227    // SAFETY: We've initialized all the bytes in `load_file`.
228    Some(unsafe { initrd.assume_init_ref() })
229}

231fn find_rsdp_addr() -> Option<*const ()> {
232    use uefi::table::cfg::{ACPI_GUID, ACPI2_GUID};
233
234    // Prefer ACPI2 over ACPI.
235    for acpi_guid in [ACPI2_GUID, ACPI_GUID] {
236        if let Some(rsdp_addr) = uefi::system::with_config_table(|table| {
237            table
238                .iter()
239                .find(|entry| entry.guid == acpi_guid)
240                .map(|entry| entry.address.cast::<()>())
241        }) {
242            uefi::println!("[EFI stub] Found the ACPI RSDP at {:p}", rsdp_addr);
243            return Some(rsdp_addr);
244        }
245    }
246
247    uefi::println!("[EFI stub] Warning: Failed to find the ACPI RSDP address!");
248
249    None
250}

252fn fill_screen_info(screen_info: &mut linux_boot_params::ScreenInfo) {
253    use uefi::{
254        boot::{OpenProtocolAttributes, OpenProtocolParams, open_protocol},
255        proto::console::gop::{GraphicsOutput, PixelFormat},
256    };
257
258    let Ok(handle) = uefi::boot::get_handle_for_protocol::<GraphicsOutput>() else {
259        uefi::println!("[EFI stub] Warning: Failed to locate the graphics handle!");
260        return;
261    };
262
263    // We don't use `open_protocol_exclusive` here for `GraphicsOutput` because it may disconnect
264    // the console.
265    //
266    // UEFI Specification, 7.3.9 EFI_BOOT_SERVICES.OpenProtocol():
267    //   EXCLUSIVE [..] If any drivers have the protocol interface opened with an attribute of
268    //   BY_DRIVER, then an attempt will be made to remove them by calling the driver's Stop()
269    //   function.
270    //
271    // SAFETY: No one will change the graphics mode at this point. It is safe to query it through
272    // shared access.
273    let Ok(mut protocol) = (unsafe {
274        open_protocol::<GraphicsOutput>(
275            OpenProtocolParams {
276                handle,
277                agent: uefi::boot::image_handle(),
278                controller: None,
279            },
280            OpenProtocolAttributes::GetProtocol,
281        )
282    }) else {
283        uefi::println!("[EFI stub] Warning: Failed to open the graphics protocol!");
284        return;
285    };
286
287    if !matches!(
288        protocol.current_mode_info().pixel_format(),
289        PixelFormat::Rgb | PixelFormat::Bgr
290    ) {
291        uefi::println!(
292            "[EFI stub] Warning: Ignored the framebuffer as the pixel format is not supported!"
293        );
294        return;
295    }
296
297    let addr = protocol.frame_buffer().as_mut_ptr().addr();
298    let (width, height) = protocol.current_mode_info().resolution();
299
300    // TODO: We are only filling in fields that will be accessed later in the kernel. We should
301    // fill in other important information such as the pixel format.
302    screen_info.lfb_base = addr as u32;
303    screen_info.ext_lfb_base = (addr >> 32) as u32;
304    screen_info.lfb_width = width.try_into().unwrap();
305    screen_info.lfb_height = height.try_into().unwrap();
306    screen_info.lfb_depth = 32; // We've checked the pixel format above.
307
308    uefi::println!(
309        "[EFI stub] Found the framebuffer at {:#x} with {}x{} pixels",
310        addr,
311        width,
312        height
313    );
314}

316unsafe fn efi_phase_runtime(boot_params: &mut BootParams) -> ! {
317    uefi::println!("[EFI stub] Exiting EFI boot services");
318    // SAFETY: The safety is upheld by the caller.
319    let memory_map = unsafe { exit_boot_services(uefi::table::boot::MemoryType::LOADER_DATA) };
320
321    crate::println!(
322        "[EFI stub] Processing {} memory map entries",
323        memory_map.entries().len()
324    );
325    #[cfg(feature = "debug_print")]
326    {
327        memory_map.entries().for_each(|entry| {
328            crate::println!(
329                "    [{:#x}] {:#x} (size={:#x}) {{flags={:#x}}}",
330                entry.ty.0,
331                entry.phys_start,
332                entry.page_count,
333                entry.att.bits()
334            );
335        })
336    }
337
338    // Write the memory map to the E820 table in `boot_params`.
339    let e820_table = &mut boot_params.e820_table;
340    let mut num_entries = 0usize;
341    for entry in memory_map.entries() {
342        let typ = if let Some(e820_type) = parse_memory_type(entry.ty) {
343            e820_type
344        } else {
345            // The memory region is unaccepted (i.e., `MemoryType::UNACCEPTED`).
346            crate::println!("[EFI stub] Accepting pending pages");
347            for page_idx in 0..entry.page_count {
348                // SAFETY: The page to accept represents a page that has not been accepted
349                // (according to the memory map returned by the UEFI firmware).
350                unsafe {
351                    tdx_guest::tdcall::accept_page(0, entry.phys_start + page_idx * PAGE_SIZE)
352                        .unwrap();
353                }
354            }
355            linux_boot_params::E820Type::Ram
356        };
357
358        if num_entries != 0 {
359            let last_entry = &mut e820_table[num_entries - 1];
360            let last_typ = last_entry.typ;
361            if last_typ == typ && last_entry.addr + last_entry.size == entry.phys_start {
362                last_entry.size += entry.page_count * PAGE_SIZE;
363                continue;
364            }
365        }
366
367        if num_entries >= e820_table.len() {
368            crate::println!("[EFI stub] Warning: The number of E820 entries exceeded 128!");
369            break;
370        }
371
372        e820_table[num_entries] = linux_boot_params::BootE820Entry {
373            addr: entry.phys_start,
374            size: entry.page_count * PAGE_SIZE,
375            typ,
376        };
377        num_entries += 1;
378    }
379    boot_params.e820_entries = num_entries as u8;
380
381    crate::println!(
382        "[EFI stub] Entering the Asterinas entry point at {:p}",
383        super::ASTER_ENTRY_POINT,
384    );
385    // SAFETY:
386    // 1. The entry point address is correct and matches the kernel ELF file.
387    // 2. The boot parameter pointer is valid and points to the correct boot parameters.
388    unsafe { super::call_aster_entrypoint(super::ASTER_ENTRY_POINT, boot_params) }
389}

391fn parse_memory_type(
392    mem_type: uefi::table::boot::MemoryType,
393) -> Option<linux_boot_params::E820Type> {
394    use linux_boot_params::E820Type;
395    use uefi::table::boot::MemoryType;
396
397    match mem_type {
398        // UEFI Specification, 7.2 Memory Allocation Services:
399        //   Following the ExitBootServices() call, the image implicitly owns all unused memory in
400        //   the map. This includes memory types EfiLoaderCode, EfiLoaderData, EfiBootServicesCode,
401        //   EfiBootServicesData, and EfiConventionalMemory
402        //
403        // Note that this includes the loaded kernel! The kernel itself should take care of this.
404        //
405        // TODO: Linux takes memory attributes into account. See
406        // <https://github.com/torvalds/linux/blob/b7f94fcf55469ad3ef8a74c35b488dbfa314d1bb/arch/x86/platform/efi/efi.c#L133-L139>.
407        MemoryType::LOADER_CODE
408        | MemoryType::LOADER_DATA
409        | MemoryType::BOOT_SERVICES_CODE
410        | MemoryType::BOOT_SERVICES_DATA
411        | MemoryType::CONVENTIONAL => Some(E820Type::Ram),
412
413        // Some memory types have special meanings.
414        MemoryType::PERSISTENT_MEMORY => Some(E820Type::Pmem),
415        MemoryType::ACPI_RECLAIM => Some(E820Type::Acpi),
416        MemoryType::ACPI_NON_VOLATILE => Some(E820Type::Nvs),
417        MemoryType::UNUSABLE => Some(E820Type::Unusable),
418        MemoryType::UNACCEPTED => None,
419
420        // Other memory types are treated as reserved.
421        MemoryType::RESERVED
422        | MemoryType::RUNTIME_SERVICES_CODE
423        | MemoryType::RUNTIME_SERVICES_DATA
424        | MemoryType::MMIO
425        | MemoryType::MMIO_PORT_SPACE => Some(E820Type::Reserved),
426        _ => Some(E820Type::Reserved),
427    }
