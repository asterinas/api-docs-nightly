3use core::mem::MaybeUninit;
4
5use uefi::boot::AllocateType;
6
7pub fn alloc_at(addr: usize, size: usize) -> &'static mut [MaybeUninit<u8>] {
8    assert_ne!(addr, 0, "the address to allocate is zero");
9    assert!(
10        addr.checked_add(size).is_some(),
11        "the range to allocate overflows"
12    );
13
14    let allocated = alloc_pages(AllocateType::Address(addr as u64), size);
15    assert_eq!(
16        allocated.as_ptr().addr(),
17        addr,
18        "the allocated address is not the request address"
19    );
20
21    allocated
22}
23
24pub(super) fn alloc_pages(ty: AllocateType, size: usize) -> &'static mut [MaybeUninit<u8>] {
25    assert!(
26        size <= isize::MAX as usize,
27        "the size to allocate exceeds `isize::MAX`"
28    );
29
30    let allocated = uefi::boot::allocate_pages(
31        ty,
32        uefi::boot::MemoryType::LOADER_DATA,
33        size.div_ceil(super::efi::PAGE_SIZE as usize),
34    )
35    .expect("the UEFI allocation fails");
36
37    // SAFETY:
38    // 1. The address is not zero and the size is reasonable (there are less the `isize::MAX` bytes
39    //    and the range won't overflow the address space), as asserted above or guaranteed by the
40    //    implementation of `allocate_pages`.
41    // 2. The memory region is allocated via the UEFI firmware, so it is valid for reading and
42    //    writing. We will not deallocate it, so it live for `'static`.
43    // 3. The type alignment is 1 and the type can contain uninitialized data.
44    unsafe { core::slice::from_raw_parts_mut(allocated.as_ptr().cast(), size) }
45}
